function: autoreroll REROLLS:n {
result: [highest 1 of REROLLSd{0,0,0,0,0,1}]
}

function: lar DICE:n REROLLS:n {
TOTAL: 0
loop N over {1..DICE} {
    INC: [autoreroll REROLLS]
    TOTAL: TOTAL + INC
  }
result: TOTAL
}

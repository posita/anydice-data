function: roll ROLL:n {
 if ROLL <= COMP { result: 0 }
 if ROLL  + MOD >= DIFF { result: 2 }
 if ROLL + MOD < DIFF { result: 1 }
 result: 0
}

MOD: 0
\ -2 to 2 \

COMP: 2
\ 2 to 5 \

loop DIFF over {8..14} {
 output [roll 2d6] named "[DIFF] Difficulty"
}

loop DIFF over {8..14} {
 output [roll 1d6 + 1d8] named "[DIFF] Difficulty (Concept)"
}

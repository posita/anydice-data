ARMOR_RATING: 6
DAMAGE: 0

function: hit ATK:n greater than AR:n {
  if ATK > AR {
      DAMAGE: DAMAGE + (ATK - AR)
      ARMOR_RATING: AR - 1
  }
  result: DAMAGE
}

loop N over {1..20}{
   ATTACK: d8
   output [hit ATTACK greater than ARMOR_RATING]
}
output ARMOR_RATING named "AR: [ARMOR_RATING]"
output DAMAGE named "Total Damage: [DAMAGE]"

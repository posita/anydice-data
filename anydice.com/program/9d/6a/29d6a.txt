\Alternate 2d50 calculator. Each 'die' rolls 2d50 dice.\
\Rolling too many dice will cause an error, so adjust DICEMIN and DICEMAX to keep your dice reasonable!\

DICEMIM: 1 \Starting number of dice rolled\
DICEMAX: 5 \Ending number of dice rolled\
DIEBONUS: 32 \Bonus per die\
CURRENTPROGRESS: 87 \Your current project progress\
PROGRESSTARGET: 200 \The total progress the project needs\
OMAKEBONUS: 15 \Omake bonus points\
NOBONUSDICE: 0 \Number of extra dice rolled without a bonus\

function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 0
 if SUCCESSES > 0 { result: 1 }
 result: 0
}

loop DICE over {DICEMIM..DICEMAX} {
 output [evaluate (DICE*2)d50+(NOBONUSDICE*2)d50+DIEBONUS*DICE+CURRENTPROGRESS-PROGRESSTARGET+OMAKEBONUS] named "[DICE]"
}

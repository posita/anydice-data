function: distance A:n B:n {
  if A >= B { result: A-B }
  result: B - A
}

function: closest to T:n in SEQ:s {
  WINNER: 1000
  loop N over {1..#SEQ} {
    if [distance (N@SEQ*10) T] < [distance (WINNER*10) T] { WINNER: N@SEQ }
  }
  result: WINNER
}

output 2@3d10
output [closest to 55 in 3d10]
output [closest to 1d{54,55} in 3d10]

function: attack ROLL:n    plus BONUS:n    vs AC:n    for DMG:d    crit CRIT_DMG:d    on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

output [attack 1d20   plus 0   vs 0   for 1d10   crit 2d10   on 20] named "crit d10 normal"
output [attack 1d20   plus 0   vs 0   for 1d10   crit 3d10   on 20] named "crit d10 +1d10"

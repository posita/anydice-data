PROF: 4
DEX: 4

function: attack ROLL:n vs DC:n {
   if ROLL = 1 { result: 0 }
   if ROLL + PROF + DEX >= DC { result: 1}
   result: 0
}

function: damage A:n B:n {
   N:0
   if A = 1 {
      N: N + 3d4
   }
   if B = 1 {
      N: N + [highest of DEX+1 and 3d4]
   }
   result: N
}

loop DC over {10..18} {
   output 2d[damage [attack 2@2d20 vs DC] [attack 2@2d20 vs DC]] named "[DC]"
}

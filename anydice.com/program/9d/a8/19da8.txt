function: mightattack A:n with S:n bonus B:n type X:n{
H:A + S
D:0

if H >14 {
 D:D+1+B}
if H >19 {
 D:D+1}
if H >24 {
 D:D+1}
 result: DdX
}

output [ mightattack d20 with 3 bonus 1 type 4] named "Blunt"

output [ mightattack d20 with 3 bonus 1 type 6] named "Pointy"

output [ mightattack d20 with 3 bonus 1 type 8] named "Cutting"

function: reroll R:n under N:n {
   if R < N { result: 1d6 } else {result: R}
}
output 2d6 named "2d6, not exploded no GWM"
output [reroll 1d6 under 3] + [reroll 1d6 under 3] named "2d6, not exploded, with GWM"

loop DEPTH over {1..2} {
 set "explode depth" to DEPTH
 output [explode [reroll 1d6 under 3]] + [explode [reroll 1d6 under 3]] named "2d6, exploded at most [DEPTH] times with GWM"
}

set "maximum function depth" to 6

function: explode DIE:d plus MOD:n {
 MAX: [maximum of DIE]
 result: [explode DIE + MOD helper]
}

function: explode N:n helper {
 if N >= MAX { result: MAX + [explode DIE + MOD helper] }
 result: N
}

loop N over {-6..6} { 
  output [explode d12 plus N] - d12 named "d12+[N]"
}

function: forbiddenwhite R:n push N:n {
if N = 1 { result: 0}
else{
if N < 5 { if R > 0 { result: 0 + [forbiddenwhite R-1 push d5] }}
result: N = 5
}}

function: forbiddenred R:n push N:n {
if N < 5 { if R > 0 { result: 0 + [forbiddenred R-1 push d5] }}
result: N = 5
}

FORBIDDENA: [forbiddenwhite 1 push d5]
FORBIDDENB: [forbiddenred 1 push d5]


loop N over {1..10} {
output [lowest of 10 and NdFORBIDDENA] named "[N]d, hvit d5 med push"
output [lowest of 10 and NdFORBIDDENB] named "[N]d, rød d5 med push"
}

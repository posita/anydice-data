function: roll N:n with Limit L:n {
result: [lowest of [count {5, 6} in Nd6] and L]
}

function: extended roll N:n with Limit L:n{
S: 0
loop X over{0..N}{
S:  S + [roll (N-X) with Limit L] 
}
result: S
}

N: 12
L: 7
T: 30
output [roll N with Limit L] named "Single Test"
output [extended roll N with Limit L] named "Extended Test"

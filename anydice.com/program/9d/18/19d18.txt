BONUSONE: 5
BONUSTWO: 5
BONUSTHREE: 6

TARGETEASY: 10
TARGETMEDIUM: 15
TARGETHARD: 20
TARGETCRAZY: 25

ROLLONE: 1d20+BONUSONE
ROLLTWO: 1d20+BONUSTWO
ROLLTHREE: 1d20+BONUSTHREE

output (ROLLONE>=TARGETEASY) + (ROLLTWO>= TARGETEASY) + (ROLLTHREE >= TARGETEASY) named "Easy"
output (ROLLONE>=TARGETMEDIUM) + (ROLLTWO>= TARGETMEDIUM) + (ROLLTHREE >= TARGETMEDIUM) named "Medium"
output (ROLLONE>=TARGETHARD) + (ROLLTWO>= TARGETHARD) + (ROLLTHREE >= TARGETHARD) named "Hard"
output (ROLLONE>=TARGETCRAZY) + (ROLLTWO>= TARGETCRAZY) + (ROLLTHREE >= TARGETCRAZY) named "Crazy"

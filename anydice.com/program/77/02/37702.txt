function: attack A:n skill B:n ac C:n damage D:n crit chance E:n crit die F:n crit mult G:n {
 if A >= E{ 
  result: G*(D+F)
 }
 if A=1{
  result: -1
 }
 if A+B>=C{
  result: D 
 }
 result: 0
}

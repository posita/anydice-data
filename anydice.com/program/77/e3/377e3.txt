CHANTFAIL: d3
CHANT: d6
CPCHANT: {1,0,2,3,4,5}

function: chant roll a FIRSTROLL:n {
if FIRSTROLL = 1 {
RP: 0
}
if FIRSTROLL > 1 {RP: FIRSTROLL}
result: RP
}
output [chant roll a 1d6] named "First chant roll"

function: chant roll b FIRSTROLL:n SECONDROLL:n {
if FIRSTROLL = 1 {RP: 0}
if FIRSTROLL > 1 {RP: FIRSTROLL}
if SECONDROLL = 1 {RP: FIRSTROLL - CHANTFAIL}
if SECONDROLL = 2 {RP: FIRSTROLL + 0}
if SECONDROLL > 2 {RP: FIRSTROLL + SECONDROLL - 1}
result: RP
}
output [chant roll b 1d6 1d6] named "CP Chant"

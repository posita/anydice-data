SKILL: d6
MODIFIER: 0
TARGET: 4

function: if CONDITION:n then A else B {
  if CONDITION { result: A } else { result: B }
}

ROLL: [highest of [explode SKILL] and [explode d6]]
CRIT_FAIL: ROLL = 1
RESULT: ROLL + MODIFIER
SUCCESS: RESULT >= TARGET & !CRIT_FAIL
RAISES: [
 if CRIT_FAIL
 then -2
 else [
  if SUCCESS
  then RESULT / TARGET
  else -1  
 ]
]

output SUCCESS named "Success"
output RAISES named "Raises"

function: check new roll D:n R:n N:n {
  if D = R {result: 1d10}
  else {result: R}
}

function: fancy reroll D:n over N:n {
  if D < N {result: [check new roll D 1d10 N]}

  else {result: D}
}

TOHIT: 8
output 1d10 >= TOHIT
output [highest 1 of 2d10] >= TOHIT
output [fancy reroll 1d10 over TOHIT] >= TOHIT

function: normal ROLL:s target TARGET:n {
  result: ROLL >= TARGET
}

function: challenged ATTACKER:n with DEFENDER:n {
  if ATTACKER >= DEFENDER { result: 1 }
  result: -1
}

loop DICE over {3} {
  loop TARGET over {5,3} {
    output [normal DICEd6 target TARGET] named "[DICE]d6 w/ success >= [TARGET]"
  }
}

loop DICE over {3} {
  loop TARGET over {5} {
    A: [normal DICEd6 target TARGET]
    B: [normal DICEd6 target TARGET]
    output [challenged A with B] named "Challenged [DICE]d6 results"
  }
}

function: success from POOL {
  SUCCESS: 0
  loop TARGET over POOL {
    SUCCESS: SUCCESS + [normal 1d6 target TARGET]
  }
  result: SUCCESS
}

NPC: {5,5,5}
PLAYER: {5,3,3}
output [success from NPC] - [success from PLAYER] named "Success with NPC 0 upgrades vs Player 2 upgrades"

\**** Damage calculator normal Vs advantage ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{25}

ATTACKMOD: 5
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 2
CRITROLL: 20
ATTACKTIMES: 4
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS + 1d6
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS + 2d6
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "+1 second"

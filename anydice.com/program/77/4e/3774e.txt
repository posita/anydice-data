function: spellduelbase A:s vs B:s {
   if(A > B) {result: 2} \ ADVANTAGE FOR CASTER AGAINST COUNTER \
   if(A <= B) {result: -1} \ ADVANTAGE FOR COUNTERS \
}

function: spellduelextra A:s vs B:s {
   if(A >= B+5) {result: 2}  \ ADVANTAGE FOR CASTER AGAINST COUNTER \
   if(A > B)    {result: 1}  \ NO ADVANTAGE FOR CASTER \
   if(A = B)    {result: 0}  \ DON'T KNOW. MAYBE COPY DC20 WILD MAGIC? \
   if(A+5 <= B) {result: -2} \ COUNTERSPELL \
   if(A < B)    {result: -1} \ ADVANTAGE FOR COUNTERS \
}

CASTER: 9
COUNTER: 1
DICE: 1d8

output [spellduelbase CASTER + DICE vs COUNTER + DICE]
output [spellduelextra CASTER + DICE vs COUNTER + DICE]

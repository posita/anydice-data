 DPERCENTAGE: A * 10 + B
 if DPERCENTAGE > TN { result: -1 * DPERCENTAGE }
 if A = B { result: DPERCENTAGE + 100 }
 result: DPERCENTAGE
}

 result: [compute 1d10 and 1d10 to TN]
}

 if VALUE >= 100 {result: !0}
 result: 0
}

function: VALUE is fail {
 if VALUE < 0 {result: !0}
 result: 0
}

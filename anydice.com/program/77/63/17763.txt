function: rerollnew DICE:d {
  result: [rerollnewhelper DICE DICE]
}

function: rerollnewhelper DICE:d ROLL:n {
 if ROLL <= [maximum of DICE]/2 {
   result: DICE
 }
 result: ROLL
}

function: rerolleither DICE:d {
 result: [highest of DICE and DICE]
}

DICE: 2d6+3
output DICE named "Normal roll"
output [rerollnew DICE] named "Reroll, take new (rerolling if result < 1/2 max)"
output [rerolleither 2d6+3] named "Reroll, take either (always roll twice and take best)"

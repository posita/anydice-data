function: bleem A:n and B:n and D:n {
  AD: [absolute A - (D/2)]
  BD: [absolute B - (D/2)]
  if AD >= BD {
    result: A
  } else if BD > AD {
    result: B
  }
  result: d{}
}

output [bleem d4 and d4 and 4]
output [bleem d6 and d6 and 6]
output [bleem d8 and d8 and 8]
output [bleem d10 and d10 and 10]
output [bleem d12 and d12 and 12]
output [bleem d20 and d20 and 20]
output [bleem d100 and d100 and 100]

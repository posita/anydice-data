function: roll A:n stat B target C{
 if A+B>=C{
  result: A+B-C
 }
 else {
  result: 0
 }
}

output 0 named "single roll group"
output [roll 1d20 stat 0 target 10] named "No talent, on rank"
output [roll 1d20 stat 2 target 10] named "moderate talent, on rank"
output [roll 1d20 stat 2 target 15] named "moderate talent, -1 rank"
output [roll 1d20 stat 4 target 20] named "perfect talent, -2 rank"

output 0 named "full roll group"
output [roll 1d20 stat 0 target 10] + [roll 1d20 stat 0 target 10] named "No attribute or approach"

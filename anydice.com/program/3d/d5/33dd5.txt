function: roll A:n and S:n vs T:n {
  GAP: 3
  SUM: A + S
  TN: T
  TNFC: T - 7
  TNF: T - 4
  TNS:  T + 4
  TNSC: T + 7
  if SUM <= TNFC  {result: -3}
  if SUM <= TNF {result: -2}
  if SUM < TN {result: -1}
  if SUM >= TNSC {result: 3}
  if SUM >= TNS {result: 2}
  if SUM >= TN {result: 1}
}

output [roll d4 and d4 vs 5]

output [roll d4 and d6 vs 7]

function: damage ROLL:n vs AC:n {
 if ROLL = 20 {result: CRIT}
 if ROLL + P + DMOD >= AC {result: DAMAGE}
 result: 0
}

function: condc DAMAGE:n{
 if DAMAGE < 10 {result:10}
 if DAMAGE >= 20 {result: DAMAGE/2}
}

function: save ROLL:n vs DC:n{
 if ROLL + CMOD + CSP >= DC {result:0}
 result: 1
}

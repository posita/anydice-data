function: attack A:n B:n roll D:n plus BONUS:n vs AC:n {
  if D = 20 {
    result: A+A+B
  } else if D = 1 {
    result: 0
  } else if D + BONUS >= AC {
    result: A+B
  } else {
    result: 0
  }
}

function: save DC:n vs ROLL:n plus BONUS:n damage DICE:d {
  if ROLL >= DC {
    result: DICE
  } else {
    result: DICE/2
  }
}

function: saveb DC:n vs ROLL:n plus BONUS:n damage DICE:d {
  if ROLL >= DC {
    result: DICE
  } else {
    result: 0
  }
}

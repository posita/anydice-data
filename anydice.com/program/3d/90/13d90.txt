S: d{0:1, 1:4, 2:1}

function: ROLL:s with N:n retries {
 FAILS: ROLL = 0
 REROLLS: ROLL = 1
 HITS: ROLL = 2

 if FAILS > 1 {
  result:0
 }

 if REROLLS > 0 & N > 0 {
  result: HITS + [REROLLS dS with N - 1 retries]
 }

 result: HITS
}

output [5dS with 0 retries]
output [5dS with 1 retries]
output [5dS with 2 retries]

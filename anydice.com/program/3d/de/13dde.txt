\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 10-20\
ENEMYAC: d{10..20}

ATTACKMOD: 5
DMGBONUS: 1d6 + 10
PROFICIENCY: 3
ACCBONUS: -5
CRITROLL: 20
ATTACKTIMES: (5/2)
DAMAGE: 2d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd [attackroll d20 vs ENEMYAC] named "GWM/Hex"

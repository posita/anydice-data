\ A variant of exploding dice, where you replace the maximum result of one die with two new dice. \

DIE: 1d6

function: ob NUMBER:n {
  TOTALSUM: 0
  loop LOOPING over {1..NUMBER} {
    X: DIE
    X: [checkmax X]
    TOTALSUM: TOTALSUM + X
  }
  result: TOTALSUM
}

function: checkmax N:n {
  if N = 6 {
    result: [ob 2]
  }
  result: N
}

function: kontroll A:n B:n {
 if A >= B { result: ((A - B) / 5) + 1 }
 result: 0
}

set "maximum function depth" to 4    \ Change this to a lower number if you get a timeout. \

output [kontroll [ob 6] 12] + [kontroll [ob 5] 10] + [kontroll [ob 5] 10]

function: roll NATURAL:n plus MOD vs AC:n {
    if NATURAL <= 1 { result: 0 }
    if NATURAL = 20 { result: 2 }
    result: NATURAL+MOD >= AC
}

function: hit HIT:n for DAMAGE:d plus DMOD:n {
    if HIT = 0 { result: 0 }
    result: (HIT d DAMAGE) + DMOD
}

function: smite HIT:n for DAMAGE:d plus DMOD {
    if HIT = 0 { result: 0 }
    if HIT > 1 { DMOD: DMOD + 7 }
    result: (HIT d DAMAGE) + DMOD
}

function: attack AMOD:n vs AC:n for DAMAGE:d plus DMOD:n {
    result: [hit [roll d20 plus AMOD vs AC] for DAMAGE plus DMOD]
}

function: smiting AMOD: n vs AC:n for DAMAGE:d plus DMOD:n {
    result: [smite [roll d20 plus AMOD vs AC] for DAMAGE plus DMOD]
}

function: attack A:n B:n roll D:n plus BONUS:n vs ac AC:n {
  if D = 20 {
    result: A+A+B
  } else if D = 1 {
    result: 0
  } else if D + BONUS >= AC {
    result: A+B
  } else {
    result: 0
  }
}

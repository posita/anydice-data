\ first style  \
\--------------\
function: roll DICE:d and use WEAPON:n by SKILL:n {
  result: DICE + WEAPON + SKILL
}

function: hit TARGET:n by ACTION:d {
  result: 1dACTION > TARGET
}


\ second style \
\--------------\
function: hit TARGET:n by WEAPON:n and SKILL:n {
  result: 2d6 > TARGET - WEAPON - SKILL
}

\ Example      \
\--------------\

SWORD:3
BODY:1

GOBLIN:10

\ first style \

HACK:[roll 2d6 and use SWORD by BODY]
FIRST:1d[hit GOBLIN by HACK]

SECOND:[hit GOBLIN by SWORD and BODY]

output 1dFIRST
output 1dSECOND

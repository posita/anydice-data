function: owod N:n tn TN:n {
 if N = 1 { result: 0 }
 if N < TN { result: 0 }
 if N < 13 { result: 1 }
 result: 1 + [owod d10 tn TN]
}

TAKU: [owod d12 tn 10]

 output [highest of [lowest of NdOWOD and 10] and -4] named "[N]d"
}

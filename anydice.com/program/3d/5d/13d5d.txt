function: skill N:n {
 if N >= 11 {result: 1 + [skill d12]}
 if N < 6 {result: 0}
 result: 1
}

function: ring N:n {
 if N = 6 {result: 1 + [ring d6]}
 if N < 4 {result: 0}
 result: 1
}

function: highest N:n of A:s B:s {
    result: {1..N}@[sort {A, B}]
}

SKILL: [skill d12]
RING: [ring d6]

set "maximum function depth" to 5
output 1dSKILL named "Skill"
output 1dRING named "Ring"

\loop R over {1..1}{
 loop S over {0..5}{
  P:R+S
  output [highest R of RdRING SdSKILL] named "[P]k[R]"

 }
}\

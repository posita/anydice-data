function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d10] }
 result: N >= 8
}

NTEN: [nwod 10 again d10]

NNINE: [nwod 9 again d10]

NEIGHT: [nwod 8 again d10]

loop N over {3..4} {
 output [lowest of 10 and NdNTEN] named "[N]-10A"
\ output [lowest of 10 and NdNNINE] named "[N]-9A" \
 output [lowest of 10 and NdNEIGHT] named "[N]-8A"
}

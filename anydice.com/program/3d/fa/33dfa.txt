function: cut R:n at T:n {
  if R <= T { result: R }
  result: 0
}

function: decide R:s {
  if 1@R = 0 { result: 0 }
  result: 1@R + (R > 0) - 1
}

D: [ cut 1d10 at 4 ]
loop M over { 1..10 } {
   output [ decide MdD ] named "[M]d10"
}  

E: [ cut 1d6 at 3 ]
loop M over { 1..10 } {
   output [ decide MdE ] named "[M]d6"
}

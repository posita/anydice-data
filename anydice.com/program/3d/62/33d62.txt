function: roll B:n plus A:n against TN:n {
H: [highest 1 of 3d6] + A
M: [middle 1 of 3d6] + A
L: [lowest 1 of 3d6] + A
if B = 1 {result: H-TN}
if B = 2 {result: M-TN}
if B = 3 {result: L-TN}
}


loop A over {1..4} {
  loop T over {1..9} {
    output [roll 1 plus A against T] named "H+[A]vs[T]"
    output [roll 2 plus A against T] named "M+[A]vs[T]"
    output [roll 3 plus A against T] named "L+[A]vs[T]"
  }
}


function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output [highest 3 of 5d[reroll {1} on d6]] named "Highest 3 of 5d6, rerolling 1s"

STATS: 6d[highest 3 of 5d[reroll {1} on d6]]

loop N over {1 .. 6} {
    output N@STATS named "Stat [N]"
}

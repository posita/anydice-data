
function: roll RHITS:n RDISTANCE:n RCALLEDSHOT:n {
 SCATTER: [scatter RDISTANCE RCALLEDSHOT]
 loop S over {1..#SCATTER} {
  if S@SCATTER < 0 {
   SCATTER: [set element S in SCATTER to (S@SCATTER + RHITS)]
   if S@SCATTER > 0 {
    SCATTER: [set element S in SCATTER to 0]
   }
  }
  else {
   SCATTER: [set element S in SCATTER to (S@SCATTER - RHITS)]
   if S@SCATTER < 0 {
    SCATTER: [set element S in SCATTER to 0]
   }
  }
 }
 result:SCATTER
}
function: set element I:n in SEQ:s to N:n {
  NEW: {}
  loop J over {1 .. #SEQ} {
    if I = J { NEW: {NEW, N} }
    else { NEW: {NEW, J@SEQ} }
  }
  result: NEW
}
function: scatter DISTANCE:n CALLEDSHOT:n  {
  NEW: {}
  loop J over {1..6} {
    loop K over {1..6} {
      NEW: {NEW, ((J + K - 7) * (2 ^ DISTANCE) * (2 ^ (1 - CALLEDSHOT)))}
    }
  }
  result: NEW
}

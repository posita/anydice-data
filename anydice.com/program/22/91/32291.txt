\ hardcaml \
\ SCROLL DOWN TO SEE DISPLAY SETTINGS \
\ -1 indicates CRITICAL SUCCESS, 1 indicates CRITICAL FAILURE \
\ BUILD SETTINGS \
   MY_DEX:     18
   BARB_DEX:   20
   PWT:        1              \Pass Without Trace; set 1 or 0 for ON or OFF\
   WOLF:       1              \Aspect of the Wolf; set 1 or 0 for ON or OFF\
   GUIDANCE:   1              \set 1 or 0 for ON or OFF\
   SS_RING:    1              \Shapeshifter's Ring; set 1 or 0 for ON or OFF\
   LEVEL:      12
   PROF:       2              \set 0 for none, 1 for proficiency, 2 for expertise\
   ITEM:       0              \any other items that are increasing your Stealth modifier\

\ DISPLAY SETTINGS \
   SHOW_BASIC_DICE:     1     \shows basic dice\
   SHOW_BUILD_CURVE:    1     \shows build results based on above\
   SHOW_NORMAL:         1     \shows no adv. or disadv. for build stealth checks -- useful as of 
                               Hotfix #8 because Stealth Advantage is bugged, but usually won't 
                               affect your Greater Invis DC\
   SHOW_DISADV:         0     \shows disadvantage for builds -- why are you in disadvantage???\
   SHOW_GREATER_INVIS:  0     \shows Greater Invis curves assuming adv. - ENABLE TRANSPOSED VIEW. Disables above.
                               automatically shows up to highest rate of failure.\





DICE: d20     \if for whatever reason your dice isn't a d20\

function: lucky R:n {
   if R = 1 { result: DICE } else { result: R }
}

function: reliable R:n {
   if R < 10 { result: 10 } else { result: R }
}

LUCKY: [lucky DICE]
RELIABLE: [reliable DICE]
RELIABLY_LUCKY: [reliable LUCKY]

function: luckyadv RX:n and RY:n {
   if RX = 1 {
      result: [highest of DICE and RY]
   } else if RY = 1 {
      result: [highest of RX and DICE]
   } else {
      result: [highest of RX and RY]
   }
}

function: luckydis RX:n and RY:n {
   if RX = 1 {
      result: [lowest of DICE and RY]
   } else if RY = 1 {
      result: [lowest of RX and DICE]
   } else {
      result: [lowest of RX and RY]
   }
}

function: reliablyluckyadv RX:n and RY:n {
   if RX = 1 {
      result: [highest of RELIABLE and [reliable RY]]
   } else if RY = 1 {
      result: [highest of [reliable RX] and RELIABLE]
   } else {
      result: [highest of [reliable RX] and [reliable RY]]
   }
}

function: reliablyluckydis RX:n and RY:n {
   if RX = 1 {
      result: [lowest of RELIABLE and [reliable RY]]
   } else if RY = 1 {
      result: [lowest of [reliable RX] and RELIABLE]
   } else {
      result: [lowest of [reliable RX] and [reliable RY]]
   }
}

LUCKY_ADV: [luckyadv DICE and DICE]
LUCKY_DIS: [luckydis DICE and DICE]

RELIABLE_ADV: [highest of RELIABLE and RELIABLE]
RELIABLE_DIS: [lowest of RELIABLE and RELIABLE]

RELIABLY_LUCKY_ADV: [reliablyluckyadv DICE and DICE]
RELIABLY_LUCKY_DIS: [reliablyluckydis DICE and DICE]

if SHOW_BASIC_DICE & !SHOW_GREATER_INVIS {
   output DICE named "Normal [DICE]"
   output [highest 1 of 2d20] named "Normal Advantage"
   output [lowest 1 of 2d20] named "Normal Disadvantage"

   output LUCKY named "Lucky [DICE]"
   output LUCKY_ADV named "Lucky w/ Advantage"
   output LUCKY_DIS named "Lucky w/ Disadvantage"

   output RELIABLE named "Reliable Talent"
   output RELIABLE_ADV named "Reliable Talent w/ Advantage"
   output RELIABLE_DIS named "Reliable Talent w/ Disadvantage"

   output RELIABLY_LUCKY named "Reliably Lucky"
   output RELIABLY_LUCKY_ADV named "Reliably Lucky w/ Advantage"
   output RELIABLY_LUCKY_DIS named "Reliably Lucky w/ Disadvantage"
}

function: build INPUT:n {
   if INPUT = 1 {
      result: 1
   } else if INPUT = 20 {
      result: -1
   }
   else {
      result: INPUT + PWT*10 + WOLF*((BARB_DEX-10)/2) + GUIDANCEd4 + SS_RINGd4 + PROF*((LEVEL+3)/4 + 1) + ((MY_DEX-10)/2) + ITEM
   }
}

if SHOW_BUILD_CURVE & !SHOW_GREATER_INVIS {
   output [build [highest 1 of 2d20]] named "Adv. Stealth"
   output [build LUCKY_ADV] named "Adv. Stealth w/ Lucky"
   output [build RELIABLE_ADV] named "Adv. Stealth w/ Reliable Talent"
   output [build RELIABLY_LUCKY_ADV] named "Adv. Stealth w/ Lucky & Reliable Talent"
   if SHOW_NORMAL {
      output [build DICE] named "Stealth"
      output [build LUCKY] named "Stealth w/ Lucky"
      output [build RELIABLE] named "Stealth w/ Reliable"
      output [build RELIABLY_LUCKY] named "Stealth w/ Lucky & Reliable Talent"
   }
   if SHOW_DISADV {
      output [build [lowest 1 of 2d20]] named "Disadv. Stealth"
      output [build LUCKY_DIS] named "Disadv. Stealth w/ Lucky"
      output [build RELIABLE_DIS] named "Disadv. Stealth w/ Reliable"
      output [build RELIABLY_LUCKY_DIS] named "Disadv. Stealth w/ Lucky & Reliable Talent"
   }
}

function: success INPUT:n over DC:n {
   if INPUT = -1 { result: 1 }
   else { result: INPUT >= DC }
}

if SHOW_GREATER_INVIS {
   loop N over {1..([maximum of [build [highest 1 of 2d20]]]-13)} {
      output [success [build [highest 1 of 2d20]] over (N+14)] named "Standard GINVIS @ Check #[N]"
   }
   loop N over {1..([maximum of [build LUCKY_ADV]]-13)} {
      output [success [build LUCKY_ADV] over (N+14)] named "Lucky GINVIS @ Check #[N]"
   }
   loop N over {1..([maximum of [build RELIABLE_ADV]]-13)} {
      output [success [build RELIABLE_ADV] over (N+14)] named "Reliable GINVIS @ Check #[N]"
   }
   loop N over {1..([maximum of [build RELIABLY_LUCKY_ADV]]-13)} {
      output [success [build RELIABLY_LUCKY_ADV] over (N+14)] named "Reliably Lucky GINVIS @ Check #[N]"
   }
}

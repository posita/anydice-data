\ --- CONSTANTS & SETTINGS --- \
PMISS: 1
PHIT: 2
PCRIT: 3
MISS: 400
HIT: 500
CRIT: 600
MISSCOMP: 401
HITCOMP: 501
CRITCOMP: 601
MISSOP: 402
HITOP: 502
CRITOP: 602
OPA: 5
OPB: 6

DAMAGEA: 1d6
MOD: 5
DEFENSE: 14

\ --- FUNCTIONS --- \

function: attack ROLL:n plus MOD:n vs DEF:n {
  if ROLL = 20 { result: PCRIT }
  if ROLL + MOD >= DEF { result: PHIT }
  result: PMISS
}

function: opportunity ROLL:n VALA:n VALB:n {
  if ROLL = 1 { result: 2 }
  if ROLL = 2 { result: 4 }
  if ROLL = 5 { result: VALA }
  if ROLL = 6 { result: VALB }
  result: 0
}

function: attackstakes ROLL:n plus MOD:n plus OPP:n vs DEF:n {
  if ROLL = 20 { result: 300 + OPP }
  if ROLL + MOD + OPP >= DEF { result: 200 + OPP }
  result: 100 + OPP
}

function: stakes BASE:n {
  if BASE >= 300 { 
    if BASE >= 305 { result: CRITOP }
    if BASE >= 302 { result: CRITCOMP }
    result: CRIT
  }
  if BASE >= 200 {
    if BASE >= 205 { result: HITOP }
    if BASE >= 202 { result: HITCOMP }
    result: HIT
  }
  if BASE >= 105 { result: MISSOP }
  if BASE >= 102 { result: MISSCOMP }
  result: MISS
}

function: damagestakes STATUS:n for DIE:d plus MOD:n {
  if STATUS >= 600 { result: [maximum of DIE] + MOD } 
  if STATUS = HITOP { result: [maximum of DIE] + MOD }
  if STATUS >= 500 { result: DIE + MOD } 
  if STATUS = MISSOP { result: [maximum of DIE] }
  result: 0
}

function: grazestakes STATUS:n for DIE:d plus MOD:n {
  if STATUS >= 600 { result: [maximum of DIE] + MOD }
  if STATUS = HITOP { result: [maximum of DIE] + MOD }
  if STATUS >= 500 { result: DIE + MOD }
  if STATUS = MISSOP { result: [maximum of DIE] }
  result: DIE 
}

function: damagebase STATUS:n for DIE:d plus MOD:n {
  if STATUS = PCRIT { result: [maximum of DIE] + MOD }
  if STATUS = PHIT { result: DIE + MOD }
  result: 0
}

function: grazebase STATUS:n for DIE:d plus MOD:n {
  if STATUS = PCRIT { result: [maximum of DIE] + MOD }
  if STATUS = PHIT { result: DIE + MOD }
  result: DIE
}

function: gaindice RESULT:n {
  if RESULT = PCRIT { result: [highest 1 of 3d20] }
  if RESULT = PHIT { result: [highest 1 of 2d20] }
  result: 1d20
}

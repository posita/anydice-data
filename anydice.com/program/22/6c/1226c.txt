MOD: 4

function: gwf DIE:d {
  result: [reroll DIE as DIE if under 3]
}
function: reroll ROLL:n as DIE:d if under LIMIT:n {
  if ROLL < LIMIT { result: DIE }
  else { result: ROLL }
}

loop SIDES over {6} {
  output 4d[gwf dSIDES]+(MOD*2) named "4d[SIDES] with GWF"
}

output 2d10+1d4+(MOD*3)
output 3d8+(MOD*3)

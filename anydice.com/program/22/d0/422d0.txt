\ --- CONFIG --- \
LEVEL: 11

\ --- WEAPON SETUPS --- \
DAMAGEA: 3d10
MODA: 12
DAMAGEB: 1d6
MODB: 12

\ --- DYNAMIC SCALING --- \
function: get defense {
  if LEVEL < 6 { result: 14 }
  if LEVEL < 11 { result: 15 }
  if LEVEL < 16 { result: 17 }
  result: 17
}

function: get deflect {
  if LEVEL < 6 { result: 1 }
  if LEVEL < 11 { result: 1 }
  if LEVEL < 16 { result: 2 }
  result: 4
}

\ --- OPPORTUNITY LOGIC --- \
\ We pass ROLL:n so AnyDice calculates the probability for us \
function: get opportunity ROLL:n reroll REROLL:n {
  if ROLL >= 5 {
    if ROLL = 5 { result: OPA }
    result: OPB
  }
  if REROLL = 1 {
    \ This triggers a second 1d6 roll only if the first failed and reroll is active \
    result: [get opportunity 1d6 reroll 0]
  }
  if ROLL = 1 { result: 2 }
  if ROLL = 2 { result: 4 }
  result: 0
}

DEFENSE: [get defense]
DEFLECT: [get deflect]

\ --- CONSTANTS --- \
PMISS: 1
PHIT: 2
PCRIT: 3
MISS: 400
HIT: 500
CRIT: 600
MISSCOMP: 401
HITCOMP: 501
CRITCOMP: 601
MISSOP: 402
HITOP: 502
CRITOP: 602
OPA: 5
OPB: 6

\ --- ATTACK FUNCTIONS --- \
function: attack ROLL:n plus MOD:n vs DF:n {
  if ROLL = 20 { result: PCRIT }
  if ROLL + MOD >= DF { result: PHIT }
  result: PMISS
}

function: attackstakes ROLL:n plus MOD:n plus OPP:n vs DF:n {
  if ROLL = 20 { result: 300 + OPP }
  if ROLL + MOD + OPP >= DF { result: 200 + OPP }
  result: 100 + OPP
}

function: stakes BASE:n {
  if BASE >= 300 { 
    if BASE >= 305 { result: CRITOP }
    if BASE >= 302 { result: CRITCOMP }
    result: CRIT
  }
  if BASE >= 200 {
    if BASE >= 205 { result: HITOP }
    if BASE >= 202 { result: HITCOMP }
    result: HIT
  }
  if BASE >= 105 { result: MISSOP }
  if BASE >= 102 { result: MISSCOMP }
  result: MISS
}

\ --- DAMAGE FUNCTIONS --- \
function: damagestakes STATUS:n for DIE:d plus MOD:n USEDEF:n {
  SUB: DEFLECT * USEDEF
  if STATUS >= 600 { result: [highest of 0 and ([maximum of DIE] + MOD - SUB)] } 
  if STATUS = HITOP { result: [highest of 0 and ([maximum of DIE] + MOD - SUB)] }
  if STATUS >= 500 { result: [highest of 0 and (DIE + MOD - SUB)] } 
  if STATUS = MISSOP { result: [highest of 0 and ([maximum of DIE] - SUB)] }
  result: 0
}

function: grazestakes STATUS:n for DIE:d plus MOD:n USEDEF:n {
  SUB: DEFLECT * USEDEF
  if STATUS >= 600 { result: [highest of 0 and ([maximum of DIE] + MOD - SUB)] }
  if STATUS = HITOP { result: [highest of 0 and ([maximum of DIE] + MOD - SUB)] }
  if STATUS >= 500 { result: [highest of 0 and (DIE + MOD - SUB)] }
  if STATUS = MISSOP { result: [highest of 0 and ([maximum of DIE] - SUB)] }
  result: [highest of 0 and (DIE - SUB)] 
}

function: damagebase STATUS:n for DIE:d plus MOD:n USEDEF:n {
  SUB: DEFLECT * USEDEF
  if STATUS = PCRIT { result: [highest of 0 and ([maximum of DIE] + MOD - SUB)] }
  if STATUS = PHIT { result: [highest of 0 and (DIE + MOD - SUB)] }
  result: 0
}

function: gaindice RESULT:n {
  if RESULT = PCRIT { result: [highest 1 of 3d20] }
  if RESULT = PHIT { result: [highest 1 of 2d20] }
  result: 1d20
}

\ --- CALCULATIONS --- \
OPPS: [get opportunity 1d6 reroll 0]
OPPR: [get opportunity 1d6 reroll 1]

ADV: [highest 1 of 2d20]
STD: 1d20
GDIEA: [gaindice [attack STD plus MODA vs DEFENSE]]
GDIEB: [gaindice [attack STD plus MODB vs DEFENSE]]

\ --- 5-LETTER BLOCKS: SET A --- \
ADBSY: [damagebase [attack STD plus MODA vs DEFENSE] for DAMAGEA plus MODA 1]

\ Standard vs Reroll \
ADSSY: [damagestakes [stakes [attackstakes STD plus MODA plus OPPS vs DEFENSE]] for DAMAGEA plus MODA 1]
ADSRY: [damagestakes [stakes [attackstakes STD plus MODA plus OPPR vs DEFENSE]] for DAMAGEA plus MODA 1]

\ --- 5-LETTER BLOCKS: SET B --- \
BDBSY: [damagebase [attack STD plus MODB vs DEFENSE] for DAMAGEB plus MODB 1]

\ Standard vs Reroll \
BDSSY: [damagestakes [stakes [attackstakes STD plus MODB plus OPPS vs DEFENSE]] for DAMAGEB plus MODB 1]
BDSRY: [damagestakes [stakes [attackstakes STD plus MODB plus OPPR vs DEFENSE]] for DAMAGEB plus MODB 1]

\ --- FINAL OUTPUTS --- \
output ADSSY named "Atk A: Standard Stakes"
output ADSRY named "Atk A: Reroll Stakes"
output BDSSY named "Atk B: Standard Stakes"
output BDSRY named "Atk B: Reroll Stakes"

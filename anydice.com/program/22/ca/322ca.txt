function: reroll REROLL:s on COUNT:n of DICE:d {
  RET: 0
  loop I over {1..COUNT} {
    RET: RET + [reroll REROLL on DICE to DICE]
  }
  result: RET
}

function: reroll REROLL:s on DICE:d {
  result: [reroll REROLL on DICE to DICE]
}

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
  if [REROLL contains ROLL] {
    result: REPLACEMENT
  }
  result: ROLL
}

function: crit TOHIT:n with COUNT:n of DICE:d on CRIT_RANGE:n{
	result: [crit TOHIT with COUNT of DICE on CRIT_RANGE add DICE]
}

function: crit TOHIT:n with COUNT:n of DICE:d on CRIT_RANGE:n add CRIT_DAMAGE:n{
	RET: 0
	if TOHIT >= CRIT_RANGE {
		COUNT: COUNT+1
		RET: RET + CRIT_DAMAGE
	}
	loop I over {1..COUNT} {
		RET: RET + DICE
	}
	
	result: RET
}

\Not full parity. crit damage assumed in one, extra assumed in other\
function: check NUM:n with COUNT:n of DICE:d on CRIT_RANGE:n plus BONUS:n {
	result: [check d20 against NUM with COUNT of DICE crit on CRIT_RANGE add 0 plus BONUS extra 1]
}

function: check NUM:n with COUNT:n of DICE:d on CRIT_RANGE:n add CRIT_DAMAGE:d plus BONUS:n {
	result: [check d20 against NUM with COUNT of DICE crit on CRIT_RANGE add CRIT_DAMAGE plus BONUS extra COUNT]
}

function: check TOHIT:n against NUM:n with COUNT:n of DICE:d crit on CRIT_RANGE:n add CRIT_DAMAGE:n plus BONUS:n extra EXTRA:n {
	RET: 0
	ROLLED_EXTRA: 0
	
	loop I over {1..NUM} {
		if TOHIT >= CRIT_RANGE & ROLLED_EXTRA < EXTRA{
			ROLLED_EXTRA: ROLLED_EXTRA + 1
			RET: RET + CRIT_DAMAGE
		}
		RET: RET + [crit TOHIT with COUNT of DICE on CRIT_RANGE add CRIT_DAMAGE] + BONUS
	}
	
	result: RET
}

\WIP\
\second dmg source with separate attacks amount (e.g. maneuvers)\

ATTACKS: 1
ADVANTAGE: 0
TO_HIT_FLAT: 16
DEFENSE: 18
WEAPON_DMG_DIE: 3d12
EXTRA_DMG_DIE: 0
FLAT_DMG: 5 + 5 + 10
MIN_CRIT_PROC: 20

function: if HIT:n hit if CRIT:n crit {
  if HIT + TO_HIT_FLAT >= DEFENSE {
    if HIT >= MIN_CRIT_PROC {result: 2}
    result: 1
  }
  else {
    if HIT >= MIN_CRIT_PROC {result: 2}
    result: 0
  }
}

\final dmg considering miss, normal hit and crit\
function: HIT_TYPE:n unified dmg CRUNCHY {
  \crit dmg\
  if HIT_TYPE = 2 {
    if CRUNCHY {DAMAGE: [maximum of WEAPON_DMG_DIE] + WEAPON_DMG_DIE + EXTRA_DMG_DIE + EXTRA_DMG_DIE + FLAT_DMG}
    else {DAMAGE: WEAPON_DMG_DIE + WEAPON_DMG_DIE + EXTRA_DMG_DIE + EXTRA_DMG_DIE + FLAT_DMG}
    result: DAMAGE
  }
  \normal dmg\
  if HIT_TYPE = 1 {
    DAMAGE: WEAPON_DMG_DIE + EXTRA_DMG_DIE + FLAT_DMG
    result: DAMAGE
  }
  \miss\
  result: 0
}

HIT_RESULT: 0
HIT_ROLL: d20
if ADVANTAGE = 1 {
  HIT_ROLL: [highest 1 of 2d20]
}

\hit chance\
HIT_RESULT: HIT_ROLL + TO_HIT_FLAT >= DEFENSE
CRIT: HIT_ROLL >= MIN_CRIT_PROC
HIT_TYPE: [if HIT_ROLL hit if CRIT crit]
output HIT_TYPE named "crit chance: 0 miss, 1 hit, 2 crit"

\normal dmg\
DAMAGE: WEAPON_DMG_DIE + EXTRA_DMG_DIE + FLAT_DMG
output DAMAGE named "normal hit DMG"

\crit dmg\
DAMAGE: WEAPON_DMG_DIE + WEAPON_DMG_DIE + EXTRA_DMG_DIE + EXTRA_DMG_DIE + FLAT_DMG
output DAMAGE named "crit hit DMG"

\crunchy crit dmg\
DAMAGE: [maximum of WEAPON_DMG_DIE] + WEAPON_DMG_DIE + EXTRA_DMG_DIE + EXTRA_DMG_DIE + FLAT_DMG
output DAMAGE named "CRUNCHY crit DMG"

\unified dmg\
UNIFIED_DMG: [HIT_TYPE unified dmg 0]
UNIFIED_DMG_CRUNCHY: [HIT_TYPE unified dmg 1]
loop N over {2..ATTACKS} {
  UNIFIED_DMG: UNIFIED_DMG + [HIT_TYPE unified dmg 0]
  UNIFIED_DMG_CRUNCHY: UNIFIED_DMG_CRUNCHY + [HIT_TYPE unified dmg 1]
}
output UNIFIED_DMG named "unified DMG per [ATTACKS] attack/s"
output UNIFIED_DMG_CRUNCHY named "unified DMG CRUNCHY per [ATTACKS] attack/s"

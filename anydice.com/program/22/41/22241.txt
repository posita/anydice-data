MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs BS:n {
 if ROLL = 100 { result: MISS }
 if ROLL <= BS/5 { result: CRITICALHIT }
 if ROLL <= BS { result: HIT }
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE/A
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE/A
 }
 result: MISSDAMAGE/A
}

function: attack with MOD:n bonus die {
 if MOD >= 0 {
  result: [damage for [attack [lowest 1 of (MOD+1)d100] vs KNIGHTWEAPONS]]
 }
 if MOD < 0 {
  result: [damage for [attack [highest 1 of ([absolute MOD]+1)d100] vs KNIGHTWEAPONS]]
 }
}

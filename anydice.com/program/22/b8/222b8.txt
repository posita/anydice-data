\ Hero Quest Dice Sim \
\ save is 5 for attacks against players and 6 for those against monsters \

ATTACKS:4
SAVE:5
DEFENCE:5

function: over H:n on NN:n {
  result: [count {H..6} in NN d6]
}

function: under H:n on NN:n {
  result: [count {H..6} in NN d6]
}

output [over 4 on ATTACKS] named "Hits"
output [under SAVE on DEFENCE] named "Blocks"
output [highest of 0 and [over 4 on ATTACKS]-[under SAVE on DEFENCE]] named "Damage"

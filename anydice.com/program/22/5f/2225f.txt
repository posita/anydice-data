\ Brin's Ham Roller \

\ Attacker's stats \
ATTACK:2
HIT:4
WOUND:4
DAMAGE:1
AP:0
\ Defender's stats \
SAVE:4
FNP:3
\ --------------- \

function: hitting NN:n H:n{
  result: [count {H..6} in NN d6]
}
\ works out the count of rolls between the Hit value and six \

function: wounding HH:n W:n {
result: [count {W..6} in HH d6]
}
\ works out the count of rolls between the Wound value and six \

function: saving HH:n S:n{
result: [count {1..S-1} in HH d6]
}
\ works out the count of rolls between 1 and one less than the Save value, a 1 is always a fail, so starting at 1 keeps this true even when save if save is 1+ \

output [hitting ATTACK HIT] named "Successful Hit rolls"
output [hitting ATTACK 6] named "Hit rolls of 6"
output [wounding [hitting ATTACK HIT] WOUND] named "Successful Wound rolls"
output [wounding [hitting ATTACK HIT] 6] named "Wound rolls of 6"
output [saving [wounding [hitting ATTACK HIT] WOUND] SAVE+AP] named "Failed Saves"
output [saving [wounding [hitting ATTACK HIT] WOUND] SAVE+AP]*DAMAGE named "Damage"
output [saving [saving [wounding [hitting ATTACK HIT] WOUND] SAVE+AP]*DAMAGE FNP] named "Damage after feel no pain"

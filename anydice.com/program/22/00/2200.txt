function: compute A:n B:n C:n {
  SEQ: [sort {A, B, C}]
  AA: 3@SEQ
  BB: 2@SEQ
  CC: 1@SEQ
  if AA = 1 {
    if CC = 20 { result: BB }
    else { result: AA }
  } else { result: CC }
}

function: roll quest style {
  result: [compute 1d20 1d20 1d20]
}

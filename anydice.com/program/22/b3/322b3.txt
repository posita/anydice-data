function: regular DIE:d {
  result: [regular DIE on DIE] }
function: regular X: n on DIE:d {
  D: {DIE}
  if X = #D@D { result: +1}
  else if X = 1@D { result: -1}
  else {result: 0}
}

function: spread DIE:d {
  result: [spread DIE on DIE] }
function: spread X:n on DIE:d {
  D: {DIE}
  if X = #D@D { result: 1 + [spread DIE] }
  else if X = 1@D {result: 0}
  else {result: 1}
}

function: explosive DIE:d {
  result: [explosive DIE on DIE] }
function: explosive X:n on DIE:d {
  D: {DIE}
  if X = #D@D { result: [explosive DIE] + 1 }
  else if X = 1 @D {result:[explosive DIE] - 1 }
  else {result: 0}
}

loop N over {0..10} {
output Nd[spread d6] named "spread [N]d6"
\output Nd[regular d6] named "regular [N]d6"\
\output Nd[explosive d6] named "explosive [N]d6"\
}

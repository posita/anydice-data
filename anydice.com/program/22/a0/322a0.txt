\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: roll ROLL:n plus MOD:n vs DC:n {
 if ROLL + MOD < DC { result: DAMAGE }
 result: DAMAGE/2
}




\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}
SAVEMOD: d{5..10}



DMGBONUS: 5
DMGDIE: 1d{1d4, 1d6, 1d8, 1d10, 1d12}
DC: 17
DAMAGE: 2d10 + DMGBONUS + DMGDIE

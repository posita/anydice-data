TARGETAC : 10
TARGETREFLEX : 6

SPLATTKB : 7
SPLDC : 17
ATTKB : 3
MCROSSATTKB : 6

DAMAGEBONUS : d6

MOP : -5
MOPP : -10

function: and A:n B:n {
    if A=1{
	    if B=1{
		    result :1
		}
    }
    result :0
}


function: attack X:n bonus B:n ac A:n {
    R : 0
	
    if X+B >= 10+A {
        R: 4
    }
    else if X+B >= A {
        R: 3
    }
    else if X+B <= A-10 {
        R: 1
    }
    else if X+B < A{
        R: 2
    }
	
	if [and R!=4 X=20]{
        R: R+1
    }
	if [and R!=1 X=1]{
        R: R-1
    }
	
    result : R
}

function: wattack X:n bonus B:n ac A:n damage D:n {
    X : [attack X bonus B ac A]
    if X = 4 {
        result: 2*D
    }
    if X = 3 {
        result: D
    }
    result: 0
}


function: save X:n bonus B:n dc A:n damage D:n {
    X : [attack X bonus B ac A]
    if X = 1 {
        result: 2*D
    }
    if X = 2 {
        result: D
    }
    if X = 3{
        result: [highest of D/2 and 1]
    }
    result: 0    
}

output [save 1d20 bonus 7 dc 17 damage 3d4] named "needle darts"
output [wattack 1d20 bonus 7 ac 17 damage 3d4] named "needle darts (ac)"

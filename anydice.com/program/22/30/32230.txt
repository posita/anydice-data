
function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

output [reroll {1,2} on 1d6 to d6]
output [reroll {1,2} on 1d6 to d6] + [reroll {1,2} on 1d6 to d6]
output [reroll {1,2} on d{1, 3, 4, 5, 6, 8} to d{1, 3, 4, 5, 6, 8}] + [reroll {1,2} on d{1,2,2,3,3,4} to d{1,2,2,3,3,4}]

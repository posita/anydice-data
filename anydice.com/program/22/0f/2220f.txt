function: twohanded attack ROLL:n with MOD:n and ATTR:n against AC:n {
    if ROLL = 20 {
        result: 4d6 + ATTR
    }
    if ROLL + MOD >= AC {
        result: 2d6 + ATTR
    }
    result: 0
}

function: twohanded ATT:n with MOD:n and ATTR:n against AC:n {
    if ATT > 0 {
        result: [twohanded attack 1d20 with MOD and ATTR against AC] + [twohanded (ATT - 1) with MOD and ATTR against AC]
    }
}

function: twohandedstyle damage ROLL:n {
    if ROLL > 2 {
        result: ROLL
    }
    result: 1d6
}

function: twohandedstyle attack ROLL:n with MOD:n and ATTR:n against AC:n {
    if ROLL = 20 {
        result: 4 d [twohandedstyle damage 1d6] + ATTR
    }
    if ROLL + MOD >= AC {
        result: 2 d [twohandedstyle damage 1d6] + ATTR
    }
    result: 0
}

    if ATT > 0 {
        result: [twohandedstyle attack 1d20 with MOD and ATTR against AC] + [twohanded (ATT - 1) with MOD and ATTR against AC]
    }
}

        result: 1d6 + ATTR

        result: [dualwield attack 1d20 with MOD and ATTR against AC] + [dualwield (ATT - 1) with MOD and ATTR against AC]

function: maxwidth VALUES:s in SEQUENCE:s {
 FINAL: 0
 loop P over {1..#VALUES} {
  COUNT: P@VALUES = SEQUENCE
  if COUNT >= 2 {
   FINAL: [highest of FINAL and COUNT]
  }
 }
 result: FINAL
}

output [maxwidth {1..10} in 2d10] named "2 dice"

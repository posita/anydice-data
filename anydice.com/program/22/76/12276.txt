
function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output [highest 3 of 5d[reroll {1} on d6]]


output [lowest 1 of 6d[highest 3 of 5d[reroll {1} on d6]]]

output [highest 1 of 6d[highest 3 of 5d[reroll {1} on d6]]]

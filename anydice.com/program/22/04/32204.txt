function: roll A:s vs B:s {
  R: 0
  loop N over {1.. [lowest of #A and #B]} {
    if ( N@A > N@B ) {R: R + 1} 
    else {R: R - 1 }
  }
  \ R: R + (#A - #B) \
  result: R
}

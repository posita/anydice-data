function: spread DIE:d {
  SEQ: {DIE}
  result: [roll spread DIE from 1@SEQ to #SEQ@SEQ]
}

function: roll spread N:n from MIN:n to MAX:n {
 if N = MAX { result: 1 + [roll spread DIE from MIN to MAX] }
 else if N=MIN {result: 0}
 else {result: 1}
}

function: chaos DIE:d {
  result: [roll chaos DIE on DIE]
}

function: roll chaos N:n on DIE:d {
  SEQ: {DIE}
  if N=#SEQ@SEQ { result: [spread DIE] }
  else if N=1@SEQ {result: -[spread DIE] }
  else {result: 0}
}

function: natural spread DIE:d {
  SEQ: {DIE}
  result: [roll natural spread DIE from 1@SEQ to #SEQ@SEQ]
}

function: roll natural spread N:n from MIN:n to MAX:n {
 if N = MAX { result: 1 + [roll natural spread DIE from MIN to MAX] }
 else if N=MIN {result: -1 + [roll natural spread DIE from MIN to MAX]
}
 else {result: 0}
}

N: 4

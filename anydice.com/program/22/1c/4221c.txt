function: resolve action ACTION:s danger DANGER:s {
  loop N over {6, 5, 4, 3, 2, 1} {
    A_COUNT: ACTION = N
    D_COUNT: DANGER = N
    
    if A_COUNT > D_COUNT {
      result: N
    }
  }
  \ If all action dice are cancelled, it results in a botch (1) \
  result: 1 
}

function: filter N:n {
  if [{6} contains N] {result: 2}
  if [{5,4} contains N] {result: 1}
  else {result: 0}
}

  loop D over {0..4}{  
    output [filter [resolve action Ad6 danger Dd6]] named "[A] vs [D]"
  }
}

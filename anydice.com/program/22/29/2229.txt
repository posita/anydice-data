X: 1d12
Y: 1d12
Z: 1d12

function: grok A:n B:n C:n {
if A = B = C {result:A+B+C}
result: 0
}

function: grab A:n B:n C:n {
if 1@[sort {A, B, C}] = 2@[sort{A,B,C}] {result:A+B}
result:0
}

function: hero A:n B:n C:n {
if [grok A B C] > 0 {result: A+B+C}
result: {[grab A B C]}
}

function: myth A:n B:n C:n {
if [hero A B C] > 0 {result: [hero A B C]}
result: 1@[sort{A,B,C}]
}

output [myth X Y Z]

COMP:0

function: interpret ROLL:s COMP:n {
  S:[count {4..6,6} in ROLL]
B:[count {3,5} in ROLL]
C:[count {2} in ROLL]
D:[count {1} in ROLL]
if COMP > B+C+C {result: S+B+C+ [lowest of (COMP-B-C-C)/3 and D]}
if COMP > B {result: S+B+ [lowest of (COMP-B)/2 and C]}
result:S + [lowest of COMP and B]
}

loop DICE over {1..8} {
 output [interpret DICE d6 COMP] named "[DICE]d compétence [COMP]"
}
loop DICE over {1..8} {
 output [interpret DICE d6 COMP+1] named "[DICE]d compétence [COMP]+1"
}

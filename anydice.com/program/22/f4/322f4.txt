loop CHARACTERISTIC over {1..4} {
loop SPECIALS over {0..3} {
if SPECIALS < 2 {
  output 1d12 < CHARACTERISTIC+SPECIALS+1 named "Characteristic: [CHARACTERISTIC] Skill: 1 Specials: [SPECIALS]"
}
if SPECIALS < 3 {
  output [lowest 1 of 2d12] < CHARACTERISTIC+SPECIALS+1 named "Characteristic: [CHARACTERISTIC] Skill: 2 Specials: [SPECIALS]"
}
if SPECIALS < 4 {
  output [lowest 1 of 3d12] < CHARACTERISTIC+SPECIALS+1 named "Characteristic: [CHARACTERISTIC] Skill: 3 Specials: [SPECIALS]"
}
if SPECIALS < 5 {
  output [lowest 1 of 4d12] < CHARACTERISTIC+SPECIALS+1 named "Characteristic: [CHARACTERISTIC] Skill: 4 Specials: [SPECIALS]"
}
}
}

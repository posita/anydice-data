function: X:n atk WEA:d MALUS:n DEX:n {
    if MALUS>0 {
        if(X-MALUS+3+DEX)>18 {result: (WEA+WEA)+10+DEX}
        if(X-MALUS+3+DEX)<14 {result: 0}
        result: WEA+10+DEX
    }
    if (X+3+DEX)>18 {result: (WEA+WEA)+DEX}
    if (X+3+DEX)<14 {result: 0}
    result: WEA+DEX
}
ATK: [d20 atk 1d8 0 4]
output ATK+ATK named "1:dex"
ATK: [d20 atk 1d10 0 3]
output ATK+ATK named "1:CB"
ATK: [d20 atk 1d8 5 3]
output ATK+ATK named "1:SS"
ATK: [d20 atk 1d8 0 5]
output ATK+ATK named "1:dex, 2:dex"
ATK: [d20 atk 1d10 0 4]
output ATK+ATK named "1:dex, 2:CB"
ATK: [d20 atk 1d8 5 4]
output ATK+ATK named "1:dex, 2:SS"

function: interpret ROLL:s COMP:n RISK:n {
  S:[count {4..6,6} in ROLL]
B:[count {3,5} in ROLL]
C:[count {2} in ROLL]
D:[count {1} in ROLL]
if RISK>0{
if COMP > B+C+C {result: S+B+C+[lowest of (COMP-B-C-C)/3 and D]}
if COMP > B {result: S+B+ [lowest of (COMP-B)/2 and C]}
result:S + [lowest of COMP and B]}
if COMP-D > B {result: S+B+ [lowest of (COMP-B-D)/2 and (C+D)]}
if COMP > D {result: S+ [lowest of (COMP-D) and B]}
result:S
}

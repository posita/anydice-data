function: reroll ROLL:s target TARGET:n {
   R: 1d6
   if ROLL < TARGET { result: R} else {result: ROLL}
}

function: successcount ROLL:s target TARGET:n {
  SUCCESSES: 0

  loop I over ROLL {
    if I >= 5  { SUCCESSES: SUCCESSES + 1 }
    if I = TARGET { 
        SUCCESSES: SUCCESSES + 1 
        reroll I target TARGET
   }
  }
  result: SUCCESSES
}

DICE: 3d6
TARGET: 4

output [successcount DICE target TARGET]

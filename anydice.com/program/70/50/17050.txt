function: ROLL:n with SUCCESSROLL:n over SUCCESS:n {
  if SUCCESS < SUCCESSROLL {result: 0}
  else { result: ROLL }
}
function: ROLL:n with SUCCESS:n { result: [ROLL with 1d100 over SUCCESS] }

function: sneak TURNS { result: (TURNS * SNEAK_DICE)d6 }

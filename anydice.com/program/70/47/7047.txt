function: dpr weapon WEA:d sneak SD:d str STR:n actions ACT:n prof PR:n
{RES: 0
loop I over {1..ACT} 
{RES: RES+[dpa WEA SD STR PR]}
result: RES}


function: dmg WEA:d SD:d STR:n CRIT:n {
if CRIT > 0 {result: ((1d8+SD)*2+1d8+STR)} 
result: (1d8+SD+STR) 
}
CRIT: [dmg 1d8 7d6 5 1] 
HIT: [dmg 1d8 7d6 5 0] 

function: dpa X:n {
if X>=(19-10) {result:CRIT}
result: HIT
} 

output [dpa d20] + [dpa d20] + [dpa d20]

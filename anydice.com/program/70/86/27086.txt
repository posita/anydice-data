function: roll HIT_ROLL:n MIN_CRIT:n AC:n PROF_STR:n NORMAL_DAMAGE:s CRIT_DAMAGE:s {
  if HIT_ROLL >= MIN_CRIT {
    result: CRIT_DAMAGE
  }
  HIT: HIT_ROLL + PROF_STR
  if HIT >= AC {
    result: NORMAL_DAMAGE
  }
  result: 0
}

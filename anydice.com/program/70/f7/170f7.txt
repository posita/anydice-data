\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 13-20\
ENEMYAC: d{13..20}

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 5
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 1d10 + ATTACKMOD
CRITDAMAGE: 2d10 + ATTACKMOD

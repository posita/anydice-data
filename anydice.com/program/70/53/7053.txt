DEF: 1

function: sequence compare ATT:s and DEF:s {
  loop N over {6,5,4,3,2,1} {
   if (ATT = N) > (DEF = N) { result: 1 }
   if (DEF = N) > (ATT = N) { result: 0 }
  }
  result: 0
}


loop N over {1..6}{
  output (2d6 + N) > (2d6 + DEF) named "2d6+[N] > 2d6+[DEF]"
}

loop N over {1..6}{
  output Nd(1d6 >= 5) > DEFd(1d6 >= 5) named "[N]d > [DEF]d"
}

loop N over {1..6}{
  output [sequence compare Nd6 and DEFd6] named "Comp [N]d v [DEF]d"
}


function: bucketize ROLL:n with highest miss HIGHESTMISS:n and lowest complete success LOWESTCOMPLETE:n {
    if ROLL <= HIGHESTMISS {
        result: 0
    }
    else {
        if ROLL >= HIGHESTMISS + 1 & ROLL <= LOWESTCOMPLETE -1 {
            result: 1
        }
        else {
            if ROLL >= LOWESTCOMPLETE {
                result: 2
            }
            else {
                result: 99
            }
        }
    }
}

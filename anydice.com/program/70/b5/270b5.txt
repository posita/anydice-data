function: H dice { 
 result: [count {5,6} in Hd6]
}

function: A vs B {
 result: [A dice] - [B dice]
}

function: AL lead A cavalry AC assault DL lead D cavalry DC {
 RA: A
 if D + DC > 0 { RA: (A + 1) / 2 }
 result: [(RA+AL) dice] >= D + DC &
         [(RA+AL) dice] > 0 &
         [(D+DL+1) dice] < A + AC
}

function: L lead A pirate B {
 result: (A = [B dice] + 1 & [(1+L) dice] > 0) |
         (A > [B dice] + 1 & [(2+L) dice] > 0)
}

function: A reform B break ties T modifier M {
 if T = 1 { result: [highest 1 of Ad6] + M >= [highest 1 of Bd6] }
 else { result: [highest 1 of Ad6] + M > [highest 1 of Bd6] }
}

function: A debate B committed C {
 C: 2 - C
 result: ([(A+3) dice] - [(B+C) dice]) > B
}

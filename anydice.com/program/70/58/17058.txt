function: roll D:n needing TARGET:n for DAMAGE:d {
    if D = 1 {result: 0}
    else if D = 20 {result: DAMAGE*2} 
    else if D > TARGET {result: DAMAGE}
    else {result: 6}
}

function: doubleroll X:n x D:n needing TARGET:n for DAMAGE:d {
    if X=0 {result: [roll D needing TARGET for DAMAGE]}
    else {result: [roll D needing TARGET for DAMAGE]+ [roll D needing TARGET for DAMAGE]} 
}

output [roll 1d20 needing 8 for 6d8+2] named "lightning arrow"
output [roll 1d20 needing 10 for 6d6+10] named "single arrow"
output [doubleroll d{0,1} x 1d20 needing 10 for 6d4+10] named "double shot"
output [doubleroll d{0,1} x 1d20 needing 8 for 6d6+2] named "double lightning arrow"
output 6d6+2 named "magic missile"

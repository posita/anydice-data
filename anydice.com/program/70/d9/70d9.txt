function: countl VALUES:s in SEQUENCE:s limit by L{
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 if COUNT > L {result: L}
 result: COUNT
}

loop LIMIT over {4..6}{
    output [countl {5,6} in 15d6 limit by LIMIT] > [countl {5,6} in 8d6 limit by 4] named "15d6[[LIMIT]] vs 8d6[4]"
}

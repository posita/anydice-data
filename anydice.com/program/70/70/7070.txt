function: well of DICE:d luck LUCK:n
{
    if (LUCK <= 6)
    {
        result: [lowest 1 of DICE]
    }
    result: [highest 1 of DICE]
}

output [well of 3d12 luck d12-5] named "-5 Luck 3d12 mod on flip"
output [well of 3d12 luck d12-2] named "-2 Luck 3d12 mod on flip"
output [well of 3d12 luck d12] named "-0 Luck 3d12 mod on flip"
output [well of 3d12 luck d12+2] named "+2 Luck 3d12 mod on flip"
output [well of 3d12 luck d12+5] named "+5 Luck 3d12 mod on flip"

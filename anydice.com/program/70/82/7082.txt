function: pick preference PREF:s from SEQ:s {
  loop N over {1..#PREF} { \ Go through the ranked list of preferences one by one and pick the first hit \
    if SEQ = N@PREF { result: N@PREF }
  }
  result: 1@PREF \ Default to the highest die \
}

output [pick preference {6,5,4,2,3,1} from 2d6]

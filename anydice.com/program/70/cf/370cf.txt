A: 20
TH: 5

S: 3
T: 3

ARM: 5
PEN: 1

REG: 0
WARD: 0

function: hits ATTACKS:n {
  DICE: ATTACKS d6
  result: [count {TH..6} in DICE]
}

function: wounds HITS:n {
  DICE: HITS d6
  DIFF: T - S
  if(DIFF > 3) { result: 0 }
  if(DIFF = 3 | DIFF = 2) { result: [count {6} in DICE]}
  if(DIFF = 1) { result: [count {5..6} in DICE]}
  if(DIFF = 0) { result: [count {4..6} in DICE]}
  if(DIFF = -1) { result: [count {3..6} in DICE]}
  if(DIFF <= -2) { result: [count {2..6} in DICE]}
}

function: armor WOUNDS:n {
  DICE: WOUNDS d6
  if(ARM>0) {  result: [count {1..ARM+PEN-1} in DICE] } else { result: WOUNDS }
}

function: regen WOUNDS:n {
  DICE: WOUNDS d6
  if(REG>0) {  result: [count {1..REG-1} in DICE] } else { result: WOUNDS }
}

function: ward WOUNDS:n {
  DICE: WOUNDS d6
  if(WARD>0) {  result: [count {1..WARD-1} in DICE] } else { result: WOUNDS }
}

output [ ward [ regen [ armor [ wounds [hits A] ] ] ] ]

\ 6 or more +3 \
\ 4 or 5 +2 \
\ 3 or 2 +1 \
\ -3 or -2 -1 \
\ -4 or -5 -1 \
\ -6 or more -3 \

function: skillcheck A:n {
  if A >= 6 {
   result: 3
  }
  if A >= 4 & A <= 5 {
   result: 2
  }
  if A >= 2 & A <= 3 {
   result: 1
  }
  if A <= -6 {
   result: -3
  }
  if A <= -4 & A >= -5 {
   result: -2
  }
  if A <= -2 & A >= -3 {
   result: -1
  }
}

loop N over {1..10} {
  output [skillcheck N-d10]+[skillcheck N-d10]+[skillcheck N-d10]+[skillcheck N-d10]
}

\**** Damage calculator ****\

function: damage CONF:n {
 if CONF = 0 {result: ATTDAMA+ATTACKMOD+DMGBONUS} 
 if CONF = 1 {result: ATTDAMA+ATTACKMOD+DMGBONUS+ATTDAMA}
 if CONF = 2 {result: ATTDAMA+ATTACKMOD+DMGBONUS+[maximum of ATTDAMA]}
}

\function: advantage ADV:n ROLL:n{
 if ADV=1{result: [highest 1 of 2dROLL]}
 if ADV=-1 {result: [lowest 1 of 2dROLL]}
 if ADV!=1|ADV!=1{result: 1dROLL}
}\

function: attackroll ROLL:n vs AC:n{
 \VAL: [advantage ADV]\
 if ROLL>= CRITROLL {result:1d20+ROLL*1000}
 if ROLL+ PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: ROLL+ PROFICIENCY + ATTACKMOD + ACCBONUS }
 result: 0
}

\looped against enemy AC from 15-20\
\ENEMYAC: d{15..20}\

ATTACKMOD: 3
DMGBONUS: 0
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20
F:1
S:1
E:1
TH:1
TO:1

ATTDAMA: Fd4+Sd6+Ed8+THd10+TOd12

AD:0

loop ENEMYAC over {7..30}{
 output [attackroll d20 vs ENEMYAC] named "attack roll vs [ENEMYAC]"
}


loop CRITIND over {0..2}{
 output [damage CRITIND] named "Damage [CRITIND]"
}

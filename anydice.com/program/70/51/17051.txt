function: ROLL:n with SUCCESSROLL:n over SUCCESS:n {
  if SUCCESS < SUCCESSROLL {result: 0}
  else { result: ROLL }
}
function: ROLL:n with SUCCESS:n { result: [ROLL with 1d100 over SUCCESS] }

DEX_MOD: 4
SPELL_MOD: 3

SNEAK_DICE: 2
GREENHIT_DICE: 1
GREENEXTRA_DICE: 1
BOOMHIT_DICE: 1
BOOMEXTRA_DICE: 2

function: boom TIMES TARGETS { X: (TIMES * BOOMHIT_DICE)d8 loop N over {1..TARGETS} { X: X + [BOOMEXTRA_DICEd8 with 1d100 over 30] } result: X }
function: green TIMES { result: (TIMES * GREENHIT_DICE)d8 + (TIMES * GREENEXTRA_DICE)d8 + (TIMES * SPELL_MOD) }
function: weapon TIMES TURNS { result: (TIMES * WEAPON_DICE)d8 + (TURNS * DEX_MOD) }
function: sneak TURNS { result: (TURNS * SNEAK_DICE)d6 }

function: onestacking SEQ:s {
ONES: [count 1 in SEQ]
SIXES: [count 6 in SEQ]
EXTRAONES: [count 1 in SIXES * d6]
if SIXES > 0 { result: 1@SEQ + ONES + EXTRAONES }
if #{SEQ} > 1 { result: 1@SEQ + ONES }
result: 1@SEQ
}

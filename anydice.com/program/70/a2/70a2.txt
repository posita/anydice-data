function: owodzu N:n tn TN:n {
 if N < TN { result: 0 }
 result: 1
}

function: owodsp N:n tn TN:n {
 if N = 1 { result: -1 }
 if N < TN { result: 0 }
 if N < 10 { result: 1 }
 result: 1 + [owodzu d10 tn TN]
}

OWOD: [owodsp d10 tn 6]

loop N over {1..15} {
 output [highest of NdOWOD and -1] named "[N] Würfel"
}

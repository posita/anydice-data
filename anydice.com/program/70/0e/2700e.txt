 NUM_REROLL: [lowest of [count BAD in INITIAL] and MAX_REROLL]
 result: INITIAL - NUM_REROLL * BAD + NUM_REROLL d DIE
}

function: N:n pool DIE:d reroll up to MAX_REROLL:n {

}

function: max I:n J:n {
  if I >= J { 
    result: I
  }
  result: J
}

function: attack ROLL:n vs ADJDX:n doing DAMDICE:n against ARMOR:n {
  if ROLL = 3 {
    result: [max (DAMDICE * 3 - ARMOR) 0]
  }
  if ROLL = 4 {
    result: [max (DAMDICE *2 - ARMOR) 0]
  }
  if ROLL >= 16 {
    result: 0
  }
  if ROLL <= ADJDX {
    result: [max(DAMDICE - ARMOR) 0]
  }
  result: 0
}

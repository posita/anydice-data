function: sn DICE:n DC:n{
 SUCC: 0
 loop N over {1..DICE} {
  ROLL: [explode 1d10]
  SUCC: SUCC + [helper ROLL DC]
 }
 result: SUCC
}

function: helper ROLL:n DC:n {
 if ROLL > DC {
  result: 1 + ((ROLL-DC)/4)
 }
 result: 0
}
output [sn 1 6] named "Test"
output [sn 2 6] named "Test"
output [sn 3 6] named "Test"

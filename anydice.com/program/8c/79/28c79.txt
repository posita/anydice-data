D: 1d20
MAX: [maximum of D]

function: atleasthalf DICE:n {
if DICE > MAX/2 {
 result: DICE
}
result: DICE + MAX/2
}


output D named "basic roll"
output 1@2d D named "best of 2 (advantage)"
output [atleasthalf D] named "+half if < half"
output [lowest of ((MAX/2)+D) and MAX] named "+half limited"
output (MAX/2)+D named "+half unlimited"

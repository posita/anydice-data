function: ROLL:n vs AC:n {

 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: [threat ROLL]
 }
 result: 0
}
function: threat ROLL:n {
 if ROLL >= THREAT {
  result: CRITICAL d DAMAGE + STRENGTH
 }
 result: DAMAGE
}
function: ROLL:n crit {
 if ROLL = 20 {
  result: 1
 }
 result: 0
}
function: ROLL:n hit {
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: 1
 }
 result: 0
}
function: ROLL:n HIT:n save {
 if ROLL >= (HIT / 2) & ROLL >= 10 {
  result: 0
 }
 result: 1
}

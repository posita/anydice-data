function: hit AC:n on ROLL:n with MOD:n range CRIT:n {
  if (ROLL = 20) | (ROLL >= CRIT & ROLL + MOD >= AC) {
    result: 2
  }
  if (ROLL + MOD >= AC) {
    result: 1
  }
  result: 0
}

function: damage WEAPON:d hit AC:n on ROLL:n with MOD:n range CRIT:n {
  HIT: [hit AC on ROLL with MOD range CRIT]
  if HIT = 2 {
    result: WEAPON + WEAPON
  }
  if HIT = 1 {
    result: WEAPON
  }
  result: 0
}

output [damage 1d10+2 hit 15 on 1d20 with 5 range 20] named "normal crit"
output [damage 1d10+2 hit 15 on 1d20 with 5 range 19] named "improved crit"

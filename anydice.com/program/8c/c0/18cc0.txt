ROLL_DIFFICULTY_X: [highest 2 of 4d6]
ROLL_DIFFICULTY_Y: [lowest 2 of 5d6]

loop DC over {1..15} {
  loop ADEPTSCR over {-3..3} {
    if ADEPTSCR >= 0 {
      output (( [highest 2 of (ADEPTSCR + 2)d6]+3 >= DC ) > 0 ) named "AS: [ADEPTSCR] vs DC [DC]"
    }
    else {
      output (( [lowest 2 of (-ADEPTSCR + 2)d6]+3 >= DC ) > 0 ) named "AS: [ADEPTSCR] vs DC [DC]"
    }
  }
}

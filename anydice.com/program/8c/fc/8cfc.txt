function: N:n success S failure F critical C {
if N = C { result: 10 }
if N >= S { result: 1 }
if N <= F { result: -1 }
result: 0
}

function: pool ROLL:s {
if (ROLL = 10) >= 2 { result: 99 }
result: (ROLL >= 1) - (ROLL = -1)
}

SD: [d100 success 99 failure 2 critical 100]
loop N over {2..10} {
output [pool NdSD]
}

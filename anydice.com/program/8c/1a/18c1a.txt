\**** Damage calculator normal Vs advantage ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: attackrolls ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE + [attackroll [highest 1 of 2d20] vs ENEMYAC]} 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

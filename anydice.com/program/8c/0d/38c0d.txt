\
 Blades in the Dark
 0: Things go bad.
 1: Success, but with undesired consequences.
 2: Clean success.
 3: Critical success!
\

function: blades ROLL:s {
 if 1@ROLL = 12 {
  result: 2 + ((2@ROLL = 12) | (2@ROLL = 11))
 }
 result: 1@ROLL >= 7
}

output [blades 2@2d12] named "0d"
loop P over {1..7} {
 output [blades Pd12] named "[P]d"
}

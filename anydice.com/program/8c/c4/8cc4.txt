\ DO NOT TOUCH \
function: for ROLL:s count crit condition equal to A:n then sum success condition gte B:n and fail condition lte C:n  {
  if {1,2}@ROLL = 24 { 
    result: 99 }
  else { result: (ROLL >= B) - (ROLL <= C)}
}

function: for ROLL:s count crit condition equal to A:n then sum success discounting first eleven with success condition gte eleven and fail condition lte one {
  if {1,2}@ROLL = 24 { 
    result: 99 }
  else { S: (ROLL = 12) 
         P: (ROLL = 10) 
         L: (ROLL = 1)
         if P > 0 { P: P - 1}
         result: S + P - L}
}
\ DO NOT TOUCH \



\ Can only run one at a time -- please delete others prior to running \
loop N over {2..6} {
  X: N
  output [for Xd12 count crit condition equal to 24 then sum success condition gte 12 and fail condition lte 1] named "Distribution[X]d12 @ success = 12)"
  output [for Xd12 count crit condition equal to 24 then sum success condition gte 11 and fail condition lte 1] named "Distribution[X]d12 @ success >= 11)"
  output [for Xd12 count crit condition equal to 24 then sum success discounting first eleven with success condition gte eleven and fail condition lte one] named "Distribution[X]d12 @ success >= 11 discount first 11)"
}

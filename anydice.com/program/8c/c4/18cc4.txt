\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
if ROLL >= CRITROLL { result: CRITDAMAGE }
if ROLL = 1 { result: 0 }
if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
result: 0
}

\looped against enemy AC from 13-18\
ENEMYAC: d{13..18}

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d4 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d4 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Dagger +2 proficiency +2 Attack modifier"

SAVEBONUS: 5
DAMAGE: 0
FAIL:1
SAVE:2

function: save ROLL:n vs dc DC:n {
	if ROLL + SAVEBONUS < DC {result: FAIL}
	if ROLL + SAVEBONUS >= DC {result: SAVE}
}

function: damage for SAVERESULT:n {
	if SAVERESULT = FAIL {
		result: 1
	 }
	if SAVERESULT = SAVE {
		result: 0
	}
}

output [damage for [save [lowest 1 of 2d20] vs dc 13]]

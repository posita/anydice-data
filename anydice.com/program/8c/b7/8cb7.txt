function: sign A:n {
  if A > 0 { result: 1 }
  if A < 0 { result: -1}
  result: 0
}

\ player 1 rolls Ad6 and player 2 rolls Bd6. Both pick the single highest die \
\ 1 indicates that player 1 won, -1 that player 2 won, and 0 indicates a tie \
\ In FU speak, Ad6 is "Positive" and Bd6 is "Negative" so 1 = YES and -1 = NO \

loop A over {1..10}{
  loop B over {1..A}{
    output [sign ([highest 1 of Ad6] - [highest 1 of Bd6])] named "[A]d6 vs. [B]d6"
  }
}

function: providence F:n S:n T:n {
 if F = 1 {
  if S = 1 {
   if T = 1 {result: 3}
   result: 2
  }
  result: 1
 }
 if F = -1 {
  if S = -1 {
   if T = -1 {result: -3}
   result: -2
  }
  result: -1
 }
 result: 0
}

output [providence 1d{-1,0:4,1} 1d{-1,0:4,1} 1d{-1,0:4,1}] named "providence"
output 1d6

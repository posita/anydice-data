function: roll hit ROLLER:n at ATTACK:n vs ARMOR:n at SECONDROLLER:n{
 if (ROLLER = 1)*(SECONDROLLER>ARMOR){
   result:2}
 if (ROLLER <= ATTACK)*(SECONDROLLER>ARMOR){
  result:1}
 result:0}
function: roll hit{
 result: [roll hit ROLLER at ATTACK vs ARMOR at SECONDROLLER]}
function: roll damage for HIT:n{
 if HIT = 2{
  result: CRITDAMAGE}
 if HIT = 1{
  result: DAMAGE}
 result: MISSDAMAGE}
function: roll damage{
 result: [roll damage for [roll hit]]}

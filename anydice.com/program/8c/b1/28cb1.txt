\This script checks the likelihood of passing a check within a certain number of attempts\

\Set TARGET as AC or DC\
 TARGET:15

\Set BP as the Bonus or Penalty to the roll\
 BP:-4

\Set LOOPS to the number of attempts max\
 LOOPS:10

X: 0
Y: 0
Z: 0
function: pass {
 if 1d20+BP>TARGET{
  X: X+1 }
 else {
  Y: 1 }
 Z: Z+1 }

function: passFail {
 if Y = 0 & Z < LOOPS {
  [pass] [passFail] }
   result: X}

output passFail

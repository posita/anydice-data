set "maximum function depth" to 4
function: roll N:n magic{
  X: 0
  loop I over {1..N}
  {
    X: X + [select 1d20 magic]
  }
  result: X
}
function: select N:n magic {
  if N >= 1 & N <= 3 {
   result: 0
  }
  if N >= 4 & N <= 11 {
   result: 1   
  }
  if N >= 12 & N <= 18 {
   result: 2
  }
  if N = 19 {
   result: 1 + [roll 2 magic]
  }
  if N = 20 {
   result: [roll 3 magic]
  }
}

loop I over {1..10} {
  output [roll I magic] named "[I] magic"
}

ROLL_DIFFICULTY_X: [highest 2 of 4d6]
ROLL_DIFFICULTY_Y: [lowest 2 of 5d6]

loop ROLL_DIFFICULTY over {-3..3} {
  loop ADEPTSCR over {-3..3} {
    if ADEPTSCR >= 0 and {
      output (( [highest 2 of (ADEPTSCR + 2)d6]+1 >= ROLL_DIFFICULTY_X ) > 0 ) named "Adept Score: [ADEPTSCR]"
    }
    else {
      output (( [lowest 2 of (-ADEPTSCR + 2)d6]+1 >= ROLL_DIFFICULTY_X ) > 0 ) named "Adept Score: [ADEPTSCR]"
    }
  }
}

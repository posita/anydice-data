function: vampire on BENIJAKE_HP:n{
    if BENIJAKE_HP <= 0{
        result: -100
    }
    if BENIJAKE_HP >= 48{
        result: 53
    }
    if BENIJAKE_HP <= 10{
            result: BENIJAKE_HP + 30
    }
    else {
        result: BENIJAKE_HP + 5
    }
}

    if ROLL <= MISS_BORDER {
        result: 0
    }
    result: ROLL
}

    result: [missshot 1d40 on 12]
}

}

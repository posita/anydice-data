function: decay N:n S:n {
 E: 0
 loop P over{1..N}{
  E: E + [exploder 1dS S]
 }
 result: E
}

function: exploder N:n S:n {
 if S != 4 {
  if N = S{
   result: N+[exploder 1d(S-2) S-2]
  }
 }else{
  result: N
 }
}

output [decay 3 4] named "modified"
output [explode 1d4] + [explode 1d4] + [explode 1d4] named "vanilla"

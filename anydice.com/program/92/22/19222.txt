
AC: 16

MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}
function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {result: CRITICALDAMAGE }
 if ATTACKRESULT = HIT {result: HITDAMAGE}
 result: 0
}
function: attack {
 result: [damage for [attack ROLL vs AC]]
}

ROLL: 1d20

CRITICALDAMAGE: 4d6 + 4
HITDAMAGE: 2d6 + 4
ATTACK: 4 + 2
output [attack] named "+STR"
CRITICALDAMAGE: 4d6 + 3 + 10
HITDAMAGE: 2d6 + 3 + 10
ATTACK: 3 + 2 - 5
output [attack] named "GWM"

ROLL: [highest 1 of 2d20]

CRITICALDAMAGE: 4d6 + 4
HITDAMAGE: 2d6 + 4
ATTACK: 4 + 2
output [attack] named "+STR Reckless"
CRITICALDAMAGE: 4d6 + 3 + 10
HITDAMAGE: 2d6 + 3 + 10
ATTACK: 3 + 2 - 5

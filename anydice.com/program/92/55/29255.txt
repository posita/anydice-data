SAVEBONUS: 0
DAMAGE: 3d4+1
FAIL:1
SAVE:2
BASE:14

function: save ROLL:n vs dc DC:n {
	if ROLL + SAVEBONUS < DC {result: FAIL}
	if ROLL + SAVEBONUS >= DC {result: SAVE}
}

function: damage for SAVERESULT:n {
	if SAVERESULT = FAIL {
		result: DAMAGE
	 }
	if SAVERESULT = SAVE {
		result: DAMAGE/2
	}
}
function: sum CHECKNO:n rolls of DAMAGE:d {
	TOTAL: 0
	loop P over {1..CHECKNO} {
		TOTAL: TOTAL + DAMAGE
		}
	result: TOTAL
}

output [sum 10 rolls of [damage for [save [highest 1 of 1d20] vs dc 17]]] + 14

function: limitedreroll ROLL:s
{
  RR: 0
  NEW_ROLL: {}
  loop I over ROLL
  {
    if I < SUCESS_VALUE & RR < MAX_REROLLS
    {
      NEW_ROLL: NEW_ROLL + dFACES_PER_DICE
      MAX_REROLLS: MAX_REROLLS+ 1
    }
    else
    {
      NEW_ROLL: NEW_ROLL + I
    }
  }
  result: NEW_ROLL
}

function: count successes ROLL:s
{
  result: ROLL >= SUCESS_VALUE
}

MAX_REROLLS: 1
DICE_COUNT: 3
FACES_PER_DICE: 6
SUCESS_VALUE: 4

output [limitedreroll dFACES_PER_DICE]

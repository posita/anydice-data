ADICE:3         \ Player 1's number of dice in pool \
BDICE:2         \ Player 2's number of dice in pool \
DICESIDES:6     \ Number of sides on each dice \

function: check BIGPOOL:n vs SMALLPOOL:n {
   SEQ: {}
   loop N over {1..#(BIGPOOL)} {
      if {N@BIGPOOL} > {N@SMALLPOOL} {SEQ: {SEQ, 1}}
      if {N@BIGPOOL} < {N@SMALLPOOL} {SEQ: {SEQ, 2}}
      if {N@BIGPOOL} = {N@SMALLPOOL} {SEQ: {SEQ, 0}}
   
   }
   result: SEQ
}

output [check (1@[sort {ADICE, BDICE}])dDICESIDES vs (2@[sort {ADICE, BDICE}])dDICESIDES] named "result of check, should be SEQ"
output (1@[sort {ADICE, BDICE}])dDICESIDES
output (2@[sort {ADICE, BDICE}])dDICESIDES

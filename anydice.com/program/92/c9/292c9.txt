\Function Definitions\
function: does D:n hit AC:n {
  result: D >= AC
}

function: MANY:n attacks with MOD:n doing DAMAGE:d hits AC:n killing HP:n {
  result: ([does 1d20+MOD hit AC]*(DAMAGE)*MANY) >= HP
}

function: modifier M:n saves against DC:n {
  result: [does 1d20+M hit DC]
}

function: damage distro with MOD:n modifier DMG:n damage on ACRNG {
  VAL: 0
  loop AC over ACRNG {
    VAL: VAL + [does 1d20+MOD hit AC]*DMG
  }
  result: VAL/(#ACRNG)
}


\Player Stat Generation\
LVL: 4
PROFICIENCIES: {2,2,2,2,3}
PROF_BON: LVL@PROFICIENCIES
SKILL_MOD: d{3..5}
ATK_MOD: PROF_BON+SKILL_MOD
DAMAGE: 2d6

\Outputs\
\output [damage distro with ATK_MOD modifier DAMAGE+SKILL_MOD damage on {14..14}]\
output ATK_MOD

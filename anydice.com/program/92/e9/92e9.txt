ADICE:2         \ Player 1's number of dice in pool \
BDICE:3         \ Player 2's number of dice in pool \
DICESIDES:6     \ Number of sides on each dice \

DICESETASIDE: 0
AVICTORIES: {}
BVICTORIES: {}

function: check BIGPOOL:d vs SMALLPOOL:d {
   loop N over {1..#(BIGPOOL)} {
      if {N@BIGPOOL} > {N@SMALLPOOL} {SEQ: SEQ + 1}
      if {N@BIGPOOL} < {N@SMALLPOOL} {BVICTORIES: BVICTORIES + 1}
      if {N@BIGPOOL} = {N@SMALLPOOL} {DICESETASIDE: DICESETASIDE + 1}
   
   }
   SEQ : {AVICTORIES, BVICTORIES }
   result: SEQ
}

output [check (1@[sort {ADICE, BDICE}])dDICESIDES vs (2@[sort {ADICE, BDICE}])dDICESIDES] named "result of check, should be SEQ"
output (1@[sort {ADICE, BDICE}])dDICESIDES
output (2@[sort {ADICE, BDICE}])dDICESIDES

output AVICTORIES named "avictories"
output BVICTORIES named "bvictories"
output DICESETASIDE named "dice set aside"

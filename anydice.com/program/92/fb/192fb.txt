\ dice to roll, including bonuses: \
DICE: (14d6+10)+3+1d12
\ list of DCs to compare against: \
DC: {45,65,80}

function: threshold ROLL:n LEVELS:s {
  loop N over [sort LEVELS] {
    if ROLL >= N {result: N}
  }
  result: 0
}

output [threshold DICE DC] named "DC brackets"
output DICE named "dice roll"

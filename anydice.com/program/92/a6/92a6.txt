YOURMELEE: 20
YOUREVASION: 20
MONSTERMELEE: 20
MONSTEREVASION: 20
DAMAGEDICE: 1
DAMAGESIDES: 8
WEAPONWEIGHT: 1
CRITMOD: 4
PROTECTION: 3d4

function: ATTACK:n vs EVASION:n {
HIT: ATTACK - EVASION
if HIT <= 0 { result: 0 }

CRITS: HIT / (CRITMOD + WEAPONWEIGHT)

result: [highest of ((DAMAGEDICE + CRITS) d DAMAGESIDES) - PROTECTION and 0]
}

MONSTERMELEEROLL: (MONSTERMELEE + 1d20) - (YOUREVASION + 1d20)

function: MONSTERMELEEROLL :n {
if MONSTERMELEEROLL <= -10 - WEAPONWEIGHT {result: 1}
if MONSTERMELEEROLL > -10 - WEAPONWEIGHT {result: 0}
}


output [d20 + YOURMELEE vs d20 + MONSTEREVASION] + [MONSTERMELEEROLL]*[d20 + YOURMELEE vs d20 + MONSTEREVASION]

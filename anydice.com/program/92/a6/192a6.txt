LV: 43     \ character level \
INT: 0     \ -1 for 18/210 Int, -2 for 18/200 Int \

METEOR: 4*51
RIFT: 40+LVd7
REND: 11dLV
CHAOS: 13dLV
MANA: 300 + 2*LV

FCHAOS: (49-LV + INT) *3
if FCHAOS < 0 { FCHAOS: 0 }       \ Fail rate Chaos Strike: 0% at level 49 ? \
FMANA: 14 + ((50-LV) - INT) *3    \ Fail rate Mana Storm:  14% at level 50 ? \

output LV     named "Level"
output -INT*3 named "Intelligence Malus%"
output FCHAOS named "Chaos Strike Fail%"
output FMANA  named "Mana Storm Fail%"

CHAOS: CHAOS * (100 - FCHAOS) / 100
MANA:  MANA  * (100 - FMANA ) / 100

output REND   named "Rend Soul (non-evil)"
output METEOR named "Meteor Swarm"
output RIFT   named "Rift"
output CHAOS  named "Chaos Strike"
output MANA   named "Mana Storm"

output REND   / 15 named "Rend Soul / Mana"
output METEOR / 14 named "Meteor Swarm / Mana"
output RIFT   / 20 named "Rift / Mana"
output CHAOS  / 15 named "Chaos Strike / Mana"
output MANA   / 16 named "Mana Storm / Mana"

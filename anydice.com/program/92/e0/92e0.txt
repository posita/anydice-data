ADICE:1         \ Player 1's number of dice in pool \
BDICE:5         \ Player 2's number of dice in pool \
DICESIDES:6     \ Number of sides on each dice \

DICESETASIDE: 0

loop N over {1..1@[sort {ADICE, BDICE}]} {
   if N@[sort {ADICEdDICESIDES}] != N@[sort {BDICEdDICESIDES}] {
      if N@[sort {ADICEdDICESIDES}] > N@[sort {BDICEdDICESIDES}] {WINNER: 1} else {WINNER: 0}
      


   }
   else {
      DICESETASIDE: DICESETASIDE + 1
   }
}

output 1@[sort ADICEdDICESIDES] != 1@[sort BDICEdDICESIDES] named "winnervar is [WINNER]"
output DICESETASIDE

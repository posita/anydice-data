\ taking highest 2 from a pool of 4 different die types \

function: highest N:n of A:s and B:s and C:s and D:s {
 result: {1..N} @ [sort {A, B, C, D}]
}

loop X over {4,6}
{

 loop Y over {10,12}
 {

  loop Z over {10,12}
  {

   loop N over {10,12}
   {

    if X <= Y & X <= Z
    {
     output [highest 2 of dX and dY and dZ and dN] >= 8 named "[X]-[Y]-[Z]-[N]"
    }
   }
  }
 }
}

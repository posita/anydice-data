function: roll A:n t T:n v B:n r R:n {
 E: 3
 if A > T {E: E-1}
 if B <= R {E: E-1}
 result: E
}

function: t T:n adv A:n vs R:n {
 M: 0
 if A > 0 {
  if A > 3 {M: -d12 - d(4*(A-3))}
  else {M: -d(4*A)}
 }
 if A < 0 {
  A: A*-1
  if A > 3 {M: d12 + d(4*(A-3))}
  else {M: d(4*A)}
 }
 result: [roll 2d20 + M t T v d20 r R]
}


output [t 15*2 adv 5 vs 17]

MISS: 0
HIT: 1
CRITHIT: 2
ABILITYMOD: 5
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
ATTACKTIMES: 1
DAMAGE: 1d6 + ABILITYMOD + DMGBONUS
CRITDAMAGE: 1d6 + 6 + ABILITYMOD + DMGBONUS
ATTACK: ABILITYMOD + PROFICIENCY + ACCBONUS

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20{ result: CRITHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

loop AC over {15..30} {
 output [attack d20 vs AC] named "n [AC]"
 output [attack [highest 1 of 3d20] vs AC] named "a [AC]"
 \output [attack [lowest 1 of 2d20] vs AC] named "d [AC]"\
}

\-- Pistol ToHit+Damage calculator --\
\ Rolls percentages on shots fired, coupling the likelyhood of
dealing damage with the chance to hit at a certain range \


function: damage die D sides S bonus B { result: DdS + B }

function: pistol sog G sob B pow P agil A range R
{ 
result: [damage die 2 sides (4+P) bonus G + B] * (10 + A + 4 - (R / 3) > 3d6)
}

output [pistol sog 0 sob 0 pow 0 agil 0 range 2] named "Normal pistol range 2"
\
output [pistol sog 0 sob 0 pow 0 agil 2 range 2] named "EE 1 range 2"


output [pistol sog 0 sob 0 pow 0 agil 0 range 5] named "Normal pistol range 5"

output [pistol sog 0 sob 0 pow 0 agil 2 range 5] named "EE 1 range 5"

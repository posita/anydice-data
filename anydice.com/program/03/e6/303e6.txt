function: explode N:n on MAGICNUMBER:n {
 if N = MAGICNUMBER {
  result: N + [explode d4 on MAGICNUMBER]
 }
 result: N
}
function: explodewithone N:n on MAGICNUMBER:n {
 if N = MAGICNUMBER {
  result: N + [explodewithone d4 on MAGICNUMBER]
 }
 if N = 1 {
  result: 1 + [explodewithone d4 on MAGICNUMBER]
 }
 result: N
}

output [explode 1d8 on 8] +[explode 1d8 on 8] named "explode 2x 1d8 on 8"
output [explodewithone 1d8 on 8] +[explodewithone 1d8 on 8] named "explode 2x 1d8 on 1 and 8"

\/Variables/\
NUM_A: 1
NUM_B: 1

BONUS_AA: 10
BONUS_BA: 10
PROGRESS_A: 0
PROGRESS_B: 0
CLASH_NUM: 1

\/Pre-Processing & (Eventual) Helper Functions/\
BONUS_A: BONUS_AA \Future group totalling\
BONUS_B: BONUS_BA 

\/Clash Loop/\

loop CNUM over {1..CLASH_NUM} {
 loop DICE over {1..2} {
  RESULT: (1d100 + BONUS_AA) - (1d100 + BONUS_BA)

  PROGRESS_A: PROGRESS_A + [highest of RESULT and 0]
  PROGRESS_B: PROGRESS_B + [highest of -RESULT and 0]
 }
}


\/Results/\
output PROGRESS_A/50
output PROGRESS_B/50


\Full Output; Truncated Full Output\
\output PROGRESS_A
output PROGRESS_B\
output [lowest of PROGRESS_A and (75 + BONUS_A)*CLASH_NUM]
output [lowest of PROGRESS_B and (75 + BONUS_B)*CLASH_NUM]

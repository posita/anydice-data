function: check out N:n {
 W: 1d6
 D: Nd6
 H: [highest 1 of D]
 L: [lowest 1 of D]
 if W = 1 {
  result: [D + 1 - H]
 }
 if W = 6 {
  result: [D + 6 - L]
 }
 result: D
}

function: check min N:n {
 W: 1d6
 D: Nd6
 H: [highest 1 of D]
 L: [lowest 1 of D]
 if W < H {
  result: D + W - H
 }
 result: D
}

function: check max N:n {
 W: 1d6
 D: Nd6
 H: [highest 1 of D]
 L: [lowest 1 of D]
 if W > L {
  result: D + W - L
 }
 result: D
}

output [check min 3]
output [check out 3]
output [check max 3]

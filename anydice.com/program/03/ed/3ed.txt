set "maximum function depth" to 5
DICE: 6
EDGESTAT: 3
EDGEDICE: 1

function: evaluate ROLL:s {
FIVES: [count {5} in ROLL]
SIXES: [count {6} in ROLL]
result: FIVES+SIXES

}

function: evaluateedge ROLL:s {
FIVES: [count {5} in ROLL]
SIXES: [count {6} in ROLL]
EDGE: [evaluate SIXES d6]
result: FIVES+SIXES+EDGE
}



 output [evaluate DICE d6] named "[DICE]d"  

 EDGEDICE: EDGESTAT + DICE 
 output [evaluateedge EDGEDICE d6] named "[DICE]d +[EDGESTAT] Edge"

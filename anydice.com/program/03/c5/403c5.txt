\**spell save???**\
\*variables*\
MISS: 0
HIT: 1
ABILITYMOD: 2
PROFICIENCY: 5
SPELLDMG: 5d8
MISSDMG: (5d8)/2

function: save ROLL:n vs SAVE:n {
 if ROLL + ABILITYMOD + PROFICIENCY >= SAVE { result: MISS }
 result: HIT
}

loop DC over {17}{
 output [save d20 vs DC] named "[DC]"
}

function: damage for SAVERESULT:n {
 if SAVERESULT = HIT {
  result: SPELLDMG
 }
  result: MISSDMG
}

\loop DC over {16}{\
 \output [damage for [save d20 vs DC]] named "[DC]"\
\}\

function: explode D:d {
 result: [open D]
}

function: open R:n {
 if R > 1 {result: 1 + [explode 1dTEN]}
 result: R
}

set "maximum function depth" to 4

TEN: d{0:7, 1:2, 2:1}
EIGHT: d{0:7, 1:1}

DIFFICULTY: 3
SKILL: 5

output [highest of (SKILLd[explode 1dTEN] - DIFFICULTYd[explode 1dTEN]) and 0] named "roll"
output [highest of (SKILLd[explode 1dTEN] - DIFFICULTYd[explode 1dTEN] + 1dEIGHT) and 0] named "adv"
output [highest of (SKILLd[explode 1dTEN] - DIFFICULTYd[explode 1dTEN] - 1dEIGHT) and 0] named "disadv"

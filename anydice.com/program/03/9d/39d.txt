DIFFICULTY:5
ATTACK:1
DEFENSE:1

function: attack ROLL:s {
 SUCCESSES: ROLL >= DIFFICULTY
 if SUCCESSES > 0 { 
  loop DICE over {DEFENSE} {
   DEFENSEROLL: [defense DICE d6]
  }
  if SUCCESSES > 0 { result: SUCCESSES - DEFENSEROLL }
 }
  result: 0
}

function: defense ROLL:s {
 SUCCESSES: ROLL >= DIFFICULTY
 if SUCCESSES > 0 { result: SUCCESSES }
 result: 0
}

loop DICE over {ATTACK} {
 output [attack DICE d6] named "[DICE]d"
}

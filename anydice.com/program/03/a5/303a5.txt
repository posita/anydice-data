MAX_BAG: 24 \ Needs to be a least the max of all possible dice added, or results will be wonky \
\ Simple N+ with various dice combinations \
loop N over {4..24} {
  \output [count {N..MAX_BAG} in d6+d6] named "[N]+ 2d6"
  output [count {N..MAX_BAG} in d6+d8] named "[N]+ 1d6 1d8"
  output [count {N..MAX_BAG} in d6+d6+d6] named "[N]+ 3d6"
  output [count {N..MAX_BAG} in d6+d6+d8] named "[N]+ 2d6 1d8"
  output [count {N..MAX_BAG} in d6+d6+d6+d6] named "[N]+ 4d6"
  output [count {N..MAX_BAG} in d8+d8] named "[N]+ 2d8"
\
  output [count {N..MAX_BAG} in d6+d8+d8] named "[N]+ 1d6 2d8"
}

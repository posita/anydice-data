
function: if ROLL:n plus BONUS fails SAVE deal DAMAGE plus MOD else X times EFF
{
  if ROLL + BONUS < SAVE { 
    if EFF = 1 { result : (DAMAGE + MOD)*2 } 
    else if EFF = -1 { result : (DAMAGE + MOD)/2 }
    else if EFF = -2 { result : 0 }
    else { result : DAMAGE + MOD }
  }
  else if X = 0 { result: 0 }
  else if X = 1 {
     R : (DAMAGE + MOD)
     if EFF = 1 { result : (R)*2 } 
     else if EFF = -1 { result : (R)/2 }
     else if EFF = -2 { result : 0 }
     else { result : R }
  }
  else { result: 0 }
}

function: if ROLL:n plus MOD hits AC deal X plus MODD times EFF 
{
  TOTAL:0
  
    if (ROLL >= AC - MOD)
    {
      R: TOTAL + X + MODD
      if EFF = 1 { result : (R)*2 } 
      else if EFF = -1 { result : (R)/2 }
      else if EFF = -2 { result : 0 }
      else { result : R }
    }
  
  result: TOTAL
}

\syntax: output <number of attacks>d[if d20 plus <to-hit bonus> hits <enemy AC> deal <damage dice> plus <damage modifier(0 for no modifier, 1 for vulnerability, -1 for resistance, -2 for immunity>] named "<Name of set>"\
\examples\

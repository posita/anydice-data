

function: evaluate ROLL:s with N:n special {
  SUCCESSES: 0
  T: 5 - N
  loop I over {1..T}{
    if I@ROLL >= 4 {
      SUCCESSES: SUCCESSES + 1
    }
  }

  X: T + 1
  loop I over {X..5}{
    if I@ROLL >= 3 {
      SUCCESSES: SUCCESSES + 1
    }
  }
  result: SUCCESSES
}

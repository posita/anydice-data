function: rolling DICE:n of DIE:n success NUMBER:n {
 V: 0
 if DICE > 0 {
  if DIE >= NUMBER {
   V: 1
  }
  result: V + [rolling (DICE - 1) of d6 success NUMBER]
 }
 result: 0
}

function: skill DICE:n success NEED:n required VALS:n {
 V: [rolling DICE of d6 success NEED]
 VALS: VALS - V
 if V <= 0 { 
  result: 0 
 }
 if VALS > 0 {
  result: [skill DICE success NEED required VALS]
 }
 result: 1
}

output [rolling 4 of d6 success 4]
output [skill 4 success 3 required 4]

function: self SELF:n and SELFTWO:n approach APPROACH:n {
 POOL: [sort { SELF, APPROACH }]
 OUT: [count {4..12} in POOL]
 OUT: OUT + [count {8..12} in POOL]
 OUT: OUT + [count {12..12} in POOL]
 if SELF >= 6 { OUT: OUT + (SELFTWO >= 4) }
 if (1@POOL = 1) { result: -1 }
 result: OUT
}

loop N over {0,4,6,8,10,12}{
 output [self 1d6 and 1d6 approach 1dN] named "d6/d[N]"
}

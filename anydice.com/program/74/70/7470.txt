\this is for MOV: Skill Challenges in which opposing hits are subtracted from own hits.\

function: ROLL:s blocked by POWER:s {
R: 0
PP: 1
loop P over {1..#ROLL} {
if P@ROLL < 4 { result: R } \set the <# for the to-hit threshold\
if P@ROLL > PP@POWER { R: R + 1 }
if P@ROLL = PP@POWER { R: R + 1 } \enabling this line cancels higher colors from blocking equal hits from lower colors; use for adjacent CWDS colors\
else { PP: PP + 1 }
}
result: R
}

function: A:s p PA:s vs B:s p PB:s {
result: [A blocked by PB] + (PA > 3) - [B blocked by PA] - (PB > 3)
}
\output [0 p 2d6 vs d6 p 0]\

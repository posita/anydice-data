\function: excess A {
if A > 0 {result: (A-1)}
else {result: 0}
}\

DIFFICULTY_POOL: 3d{0..9}
DIFFICULTY: [highest 1 of DIFFICULTY_POOL]
NINES: [count 9 in DIFFICULTY_POOL]
EXCESS_NINES: (NINES*9)/10

SKILL_POOL_I: 1d{0..9}
SKILL_I: [lowest 1 of SKILL_POOL_I]
ZEROES_I: [count 0 in SKILL_POOL_I]
EXCESS_ZEROES_I: (ZEROES_I*9/10)

SKILL_POOL_II: 2d{0..9}
SKILL_II: [lowest 1 of SKILL_POOL_II]
ZEROES_II: [count 0 in SKILL_POOL_II]
EXCESS_ZEROES_II: (ZEROES_II*9/10)

SKILL_POOL_III: 3d{0..9}
SKILL_III: [lowest 1 of SKILL_POOL_III]
ZEROES_III: [count 0 in SKILL_POOL_III]
EXCESS_ZEROES_III: (ZEROES_III*9/10)

SKILL_POOL_IV: 4d{0..9}
SKILL_IV: [lowest 1 of SKILL_POOL_IV]
ZEROES_IV: [count 0 in SKILL_POOL_IV]
EXCESS_ZEROES_IV: (ZEROES_IV*9/10)

SKILL_POOL_V: 5d{0..9}
SKILL_V: [lowest 1 of SKILL_POOL_V]
ZEROES_V: [count 0 in SKILL_POOL_V]
EXCESS_ZEROES_V: (ZEROES_V*9/10)

output DIFFICULTY + SKILL_I + EXCESS_NINES - EXCESS_ZEROES_I named "Skill 1"
output DIFFICULTY + SKILL_II + EXCESS_NINES - EXCESS_ZEROES_II named "Skill 2"
output DIFFICULTY + SKILL_III + EXCESS_NINES - EXCESS_ZEROES_III named "Skill 3"
output DIFFICULTY + SKILL_IV + EXCESS_NINES - EXCESS_ZEROES_IV named "Skill 4"
output DIFFICULTY + SKILL_V + EXCESS_NINES - EXCESS_ZEROES_V named "Skill 5"

function: roll against SKILL:n {
  ROLL: d12
  if ROLL = 12 { result: -1 }
  if ROLL = 1 {
    if SKILL > 0 { result: 2 }
    else { result: 1 }
  }
  if ROLL <= 4 + SKILL { result: 1 }
  result: 0
}

function: test ATTRIBUTE:n SKILL:n {
  TOTAL: 0
  loop I over {1..ATTRIBUTE} {
    TOTAL: TOTAL + [roll against SKILL]
  }
  result: TOTAL
}

output [test 3 2] named "Test with 3 ATTRIBUTE and 2 SKILL"

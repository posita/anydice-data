function: roll D:n damage DD:d {
 if D<=6{result:(-d4)}
 if D>=25{result:DD+DD}
 if D>=13{result:DD}
 result: 0
}
HD: d20+5
SSMH: d6+3
SSOH: d6
RMH: d8+3
ROH: d8
DMH: d4+3
DOH: d4

SSS: [roll HD damage SSMH]
output SSS named "single shortsword"

SSSS: [roll HD damage (SSMH+d6)]
output SSSS named "single shortsword + sneak"

DSS: [roll HD damage SSMH] + [roll HD damage SSOH]
output DSS named "dual shortsword"

SSD: [roll HD damage SSMH] + [roll HD damage DOH]
output SSD named "shortsword + dagger"

SDP: [roll HD damage DMH]
output SDP named "single dagger"
SDS: [roll HD damage DMH+d6]
output SDS named "single dagger + sneak"
DDP: [roll HD damage DMH] + [roll HD damage DOH]
output DDP named "dual dagger"

SR: [roll d20+3 damage RMH]
output SR named "single rapier"
SRS: [roll d20+3 damage (RMH+d6)]
output SRS named "single rapier + sneak"
DR: [roll d20+3 damage RMH] + [roll d20+3 damage ROH]
output DR named "dual rapier"

output SRP named "single rapier (pro)"
SRSP: [roll HD damage (RMH+d6)]

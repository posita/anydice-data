function: breakdown WILD:n REST:n FAILREST:n{
if WILD = 1 {result: FAILREST}
if WILD = 6 {result: REST + 6 + [explode 1d6]}
result: REST + WILD
}

function: sw SKILLDICE:d pips PIPS:n{
SKILL: #SKILLDICE
TAME: SKILL-1
LAME: TAME-1
TAMEROLLED: TAMEd6
if LAME < 1 { LAMETOTAL: 0 + PIPS }
else { LAMETOTAL: [lowest LAME of TAMEROLLED] + PIPS }
TAMETOTAL: TAMEROLLED + PIPS
result: [breakdown 1d6 TAMETOTAL LAMETOTAL]
}

STATVAL: 5d6

output [sw STATVAL pips 0] named "With Wild Die"
output STATVAL named "Standard"

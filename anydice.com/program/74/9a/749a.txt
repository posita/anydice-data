function: A:s and B:s and C:s stunt {
  if A = B { result: C }
  else if A = C { result: C }
  else if B = C { result: C }
  else { result: 0 }
}

function: A:n and B:n and C:n hitandstunt D:n {
  if A+B+C < D { result : -1 }  
  result : [A and B and C stunt]
}

loop N over { 1..18 } {
  output [ d6 and d6 and d6 hitandstunt N ] 
}

function: explode N:n {
 if N = 1 {
  result: N - d10
 }
 if N = 10 {
  result: N + d10
 }
 result: N
}

ATTACK: [explode 1d10]+10
DODGE: [explode 1d10]+7
DV: 13

output (ATTACK > DODGE) named "Hit Chance(dodging)"
output (ATTACK > DV) named "Hit Chance(not dodging)"

output (ATTACK > DODGE) & (ATTACK <= DV) named "Hit by Attack that Would've Missed"
output (ATTACK <= DODGE) & (ATTACK > DV) named "Dodged Attack that Would've Hit"

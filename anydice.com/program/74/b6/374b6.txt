TN: 6
DICE: 4
DIEMOD: 0
MOD: 0

SAVAGE: 0
loop N over {1..DICE}{
 SAVAGE: SAVAGE + (([explode 1d6]) >= TN)
}
output SAVAGE named "Base"

SAVAGE: 0
loop N over {1..DICE}{
 SAVAGE: SAVAGE + (([explode 1d6]) >= (TN + MOD))
}
output SAVAGE named "Vs Mod"

SAVAGE: 0
loop N over {1..(DICE + DIEMOD)}{
 SAVAGE: SAVAGE + (([explode 1d6]) >= (TN + MOD))
}
output SAVAGE named "Extra Dice v Mod"

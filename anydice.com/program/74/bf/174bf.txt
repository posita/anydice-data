function: save ROLL:n vs SPELLSAVEDC:n {
 if ROLL + SAVEBONUS >= SPELLSAVEDC{ result: MISS}
 result: HIT
}


function: savedamage for SAVERESULT:n {
 if SAVERESULT= HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}


function: saveattack {
 result: [savedamage for [save 1d20 vs SPELLSAVEDC]]
}
function: saveattackhighten {
 result: [savedamage for [save [lowest 1 of 2d20]vs SPELLSAVEDC]]
}



function: attack ROLL:n vs DEFENCEAC:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENCEAC { result: HIT }
 result: MISS
}



function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENCEAC]]
}

function: attackadvantage {
 result: [damage for [attack [highest 1 of 2d20]vs DEFENCEAC]]
}

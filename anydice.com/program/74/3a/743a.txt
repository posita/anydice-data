\ This stuff sets up the Skew rules; you should leave it alone.  Scroll to the bottom to decide what dice will be rolled. \

function: U:s {
  result: {U}
}
function: U:s and V:s {
  result: {U,V}
}
function: U:s and V:s and W:s {
  result: {U,V,W}
}
function: U:s and V:s and W:s and X:s {
  result: {U,V,W,X}
}
function: U:s and V:s and W:s and X:s and Y:s {
  result: {U,V,W,X,Y}
}
function: skew S:s {
  R: 0
  loop I over {1..1@S} {
    Y: I * [count I in S]
    if Y > R { R: Y }
  }
  result: R
}
function: degrees N:n {
  if N < 0 {
    result: -[degrees -N]
  } else {
    result: ([absolute N] + 4) / 5
  }
}
function: A:d versus B:d {
  result: [degrees [skew A] - [skew B]]
}

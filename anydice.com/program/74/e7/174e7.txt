function: table A:n{
	result: (A=4) +(A=5) +(A=6)*2
}

	loop EXTRA over {1,2}{

			output [table	[highest 1 of DICEd6]+[highest 1 of EXTRAd6]*MOD	] named "[DICE] standard dice and highest of [EXTRA] extra at [MOD]"
			output [table	[highest 1 of DICEd6]+[lowest 1 of EXTRAd6]*MOD	] named "[DICE] standard dice and lowest of [EXTRA] extra at [MOD]"
		}
	}
}

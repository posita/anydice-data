\ set value to given position in sequence \
function: setat SEQ:s I:n V:n {
  NEW: {}
  loop J over {1..#SEQ} {
    VAL: J@SEQ
    if J = I {
      VAL: V
    }
    NEW: {NEW, VAL}
  }
  result: NEW
}

\ find matching [ or ] \
function: match PPTR:n PRGM:s BAL:n MAT:n DIR:n {
  if DIR = 1 {
    PPTRS: {PPTR+1..#PRGM}
  } else {
    PPTRS: [reverse {1..PPTR-1}]
  }

  COUNT: 0
  loop PTR over PPTRS {
    INST: PTR@PRGM
    if INST = MAT {
      if COUNT = 0 {
        result: PTR
      } else {
        COUNT: COUNT - 1
      }
    } else if INST = BAL {
      COUNT: COUNT + 1
    }
  }

  result: -1
}

\ main program executor \
function: exec PPTR:n PRGM:s DPTR:n DATA:s IPTR:n IN:s OUT:s {
  \ do 100 iters per call, to curb recursion \
  loop ITER over {1..100} {
    if PPTR > #PRGM {
      result: OUT
      \ result: DATA \ \ for debugging \
    }

    INST: PPTR@PRGM

    if INST = 1 {
      \ > : increase data ptr \
      DPTR: DPTR + 1
      PPTR: PPTR + 1
    } else if INST = 2 {
      \ < : decrease data ptr \
      DPTR: DPTR - 1
      PPTR: PPTR + 1
    } else if INST = 3 {
      \ + : increase value at data ptr \
      DATA: [setat DATA DPTR (DPTR@DATA + 1)]
      PPTR: PPTR + 1
    } else if INST = 4 {
      \ - : decrease value at data ptr \
      DATA: [setat DATA DPTR (DPTR@DATA - 1)]
      PPTR: PPTR + 1
    } else if INST = 5 {
      \ . : output value \
      OUT: {OUT, DPTR@DATA}
      PPTR: PPTR + 1
    } else if INST = 6 {
      \ , : input value \
      if IPTR > #IN {
        IV: -1
      } else {
        IV: IPTR@IN
        IPTR: IPTR + 1
      }
      DATA: [setat DATA DPTR IV]
      PPTR: PPTR + 1
    } else if INST = 7 {
      \ [ : if value is zero, jump to matching ] \
      if DPTR@DATA = 0 {
        PPTR: [match PPTR PRGM 7 8 1] + 1
      } else {
        PPTR: PPTR + 1
      }
    } else if INST = 8 {
      \ ] : if value is nonzero, jump to matching [ \
      if DPTR@DATA != 0 {
        PPTR: [match PPTR PRGM 8 7 0] + 1
      } else {
        PPTR: PPTR + 1
      }
    } else {
      \ unknown inst \
      PPTR: PPTR + 1
    }
  }

  result: [exec PPTR PRGM DPTR DATA IPTR IN OUT]
}

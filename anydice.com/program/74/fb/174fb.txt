function: explode N:n with M:n {
 if N = M { result: M + [explode dM with M] }
 result: N
}

function: minsplode N:n with M:n {
 if N = M { result: M - 1 + [minsplode dM with M] }
 result: N
}

function: maxplode N:n {
 if N >= 2 { result: 1 + [maxplode dN] }
 result: N
}

set "maximum function depth" to 10
\ loop X over {2..4} { \
\  output [explode dX with X] named "explode d[X]" \
\  output [minsplode dX with X] named "minsplode d[X]" \
\  output [maxplode dX] named "maxplode d[X]" \
\ } \

output [minsplode d2 with 2]
output [minsplode d3 with 3]
output [minsplode d4 with 4]
output [minsplode d5 with 5]
output [maxplode d2]
output [maxplode d3]
output [maxplode d4]
output [maxplode d5]

ACC: 1d4+16
AWA: 20
EA: 50
AC: 0
DC: -1
if DC = -2 {
output ACC+(AWA-10) >= d10+(EA-10) named "Original Hit Rate: [AWA] Awareness vs. [EA] Alacrity"
}
if DC = -1 {
output ACC+(AWA-10) >= d20+(EA-10) named "Original Hit Rate: [AWA] Awareness vs. [EA] Alacrity"
}
if DC = 0 {
output ACC+(AWA-10) >= d10+EA named "Original Hit Rate: [AWA] Awareness vs. [EA] Alacrity"
}
if DC = 1 {
output ACC+(AWA-10) >= d20+EA named "Original Hit Rate: [AWA] Awareness vs. [EA] Alacrity"
}
output [highest of ACC+(AWA-10)-(d10+EA) and 0] named "Original Hit Difference: [AWA] Awareness vs. [EA] Alacrity"
EDR: EA/5
AR: (AWA-10)/5
ELO: EA - EDR*5
AWALO: (AWA-10) - AR*5
PACC: ACC+(AR*3)d(4-DC*2)+d(2+2*AWALO)-2*AR+(5*(AC+1))d2
PED: (1+EDR)d(10+2*DC)+d(ELO*2+DC*2)
output PACC >= PED named "Planned Hit Rate: [AWA] Awareness vs. [EA] Alacrity"
output [highest of PACC-PED and 0] named "Planned Hit Difference: [AWA] Awareness vs. [EA] Alacrity"

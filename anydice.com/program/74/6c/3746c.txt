TB: 8
AP: 6
WOUNDS: 15
PEN: 4
DMG: 9
if AP-PEN < 0 {AP: 0}
else {AP: AP-PEN}

SOAK: TB+AP
\SOAK: SOAK-DMG\

FIRST_RESULT: 0
SEC_RESULT: 0
THIRD_RESULT: 0

if TB+AP+WOUNDS < 10 {output 1 named "Instant kill w/ 1 shot!"} else
if TB+AP+WOUNDS < 20 {output 1 named "Instant kill w/ 2 shots!"} else
if TB+AP+WOUNDS < 30 {output 1 named "Instant kill w/ 3 shots!"}

if TB+AP < 10 {output 1 named "Damage is automatic!"}

FIRST_ROLL: [highest 1 of 2d10]
FIRST_ATTACK: [explode FIRST_ROLL]
FIRST_RESULT: FIRST_ATTACK+DMG-SOAK

SEC_ROLL: [highest 1 of 2d10]
SEC_ATTACK: [explode SEC_ROLL]
SEC_RESULT: SEC_ATTACK+DMG-SOAK

THIRD_ROLL: [highest 1 of 2d10]
THIRD_ATTACK: [explode THIRD_ROLL]
THIRD_RESULT: THIRD_ATTACK+DMG-SOAK

output FIRST_RESULT + SEC_RESULT + THIRD_RESULT

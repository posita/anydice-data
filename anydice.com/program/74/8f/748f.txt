function: rollp D:d {result:[rolla d20+5 damage D]}
function: rolla D:n damage DD:d {
 if D<=6{result:(-d4)}
 if D>=25{result:DD+DD}
 if D>=13{result:DD}
 result: 0
}
function: rollnp D:d {result:[rollanp d20+3 damage D]}
function: rollanp D:n damage DD:d {
 if D<=4{result:(-d4)}
 if D>=23{result:DD+DD}
 if D>=13{result:DD}
 result: 0
}

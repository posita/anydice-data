function: stepup APPROACH:n {
 SELF: [explode d6]
 POOL: [sort { APPROACH }]
 OUT: APPROACH >= 4
 OUT: OUT + ( APPROACH >= 8 )
 OUT: OUT + ( APPROACH >= 12 )
 OUT: OUT + SELF
 result: OUT
}

function: explode N:n {
 if N >= 6 { result: 1 + [explode d6] }
 if N >= 4 { result: 1 }
 result: 0
}

set "maximum function depth" to 4

loop N over {0,4,6,8,10,12}{
 output [stepup 1dN] named "d6/d[N]"
}

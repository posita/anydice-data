set "maximum function depth" to 3

function: doubleexplode A:n and B:n on SIZE:n {
if A = SIZE { DA: [highest of [doubleexplode dSIZE and dSIZE on SIZE] and SIZE] }
else { DA: A }

if B = SIZE { DB: [highest of [doubleexplode dSIZE and dSIZE on SIZE] and SIZE] }
else { DB: B }

result: DA+DB
}

loop D over {4,6,8,10,12}
{
output [doubleexplode dD and 0 on D] named "d[D] double explode"
}

AC: 15
BASE_DMG: 1d8
STAT_DMG_BONUS: +3
ATTACK_BONUS: +7

function: ROLL:n hit {
    if ROLL = 20 {
        result: BASE_DMG * 2 + STAT_DMG_BONUS
    } else {
        if ROLL + ATTACK_BONUS > AC {
            result: BASE_DMG + STAT_DMG_BONUS
        } else {
            result: 0
        }
    }
}

function: ROLL:n sharpshooter {
    if ROLL = 20 {
        result: BASE_DMG * 2 + STAT_DMG_BONUS + 10
    } else {
        if ROLL + ATTACK_BONUS - 5 > AC {
            result: BASE_DMG + STAT_DMG_BONUS + 10
        } else {
            result: 0
        }
    }
}

output [1d20 hit] named "Regular Attack"
output [1d20 sharpshooter] named "Sharpshooter"
output [[highest 1 of 2d20] hit] named "Advantage Attack"
output [[highest 1 of 2d20] sharpshooter] named "Sharpshooter Advantage"
output [[highest 1 of 3d20] hit] named "Elven Attack"
output [[highest 1 of 3d20] sharpshooter] named "Elven Sharpshooter"

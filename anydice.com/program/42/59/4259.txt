function: wwcount SQ:s
{
  S: [count 1 in SQ] + [count 2 in SQ] * 2
  F: [count -1 in SQ]
  R: S - F
  if R < 0 {R: 0}
  if S = 0
  {
    if F > 0 { result: -1 }
  }
  result: R
}

function: wwdice NUM:n diff DIF:n
{
  if NUM = 10
  {
    result: 2
  }else if NUM >= DIF
  {
    result: 1
  }else if NUM = 1
  {
    result: -1
  }
  result: 0
}

{
  D: DICEd[wwdice d10 diff DIF]

}

POS: 0
NEG: 0

function: ROLL:s count groups {
  MAX: 0
  loop N over {1..6} {
    MAX: [highest of MAX and N * (ROLL = N)]
  }
  result: MAX
}

function: ova roll at modifier MOD:n {
  if MOD < 0 {
    COUNT: -MOD
    result: [lowest 1 of COUNTd6]
  } else {
    COUNT: MOD + 2
    result: [COUNTd6 count groups]
  }
}

output [ova roll at modifier POS] - [ova roll at modifier NEG] named "[POS] (positive result) vs. [NEG] (negative result)"

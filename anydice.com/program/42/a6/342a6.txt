SN: 1
R: 2 \ 1=untrained, 2=trained \
L: 5
U: 1d12 <= SN
T: 1d10 <= SN
A: 1d8 <= SN
E: 1d6 <= SN
M: 1d4 <= SN
if L = 1 {
 output U
}
if L = 2 {
 Y: U + T
 output Y >= R
}
if L = 3 {
 Y: U + T + A
 output Y >= R
}
if L = 4 {
 Y: U + T + A + E
 output Y >= R
}
if L = 5 {
 Y: U + T + A + E + M
 output Y >= R
}

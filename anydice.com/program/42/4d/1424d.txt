CUN_SCALE: 2
FER_PER_BONUS: 1
DAMAGE_DIE_CONVERSION: 4

function: attack A_ROLL:d and damage D_ROLL:d with cunning C_NUM:n and ferocity F_NUM:n against TN:n{
result: [damage roll D_ROLL plus (F_NUM/FER_PER_BONUS) with [attack roll (A_ROLL+(C_NUM*CUN_SCALE)) against TN]]
}

function: attack roll ROLL:n against TN:n {
if ROLL >= TN{
result: 0
}
else{
result: ((TN-ROLL) + (DAMAGE_DIE_CONVERSION-1))/DAMAGE_DIE_CONVERSION
}
}

function: damage roll ROLL:d plus BONUS:n with NUM:n{
if(NUM >= #ROLL){
result: 0
}
else{
result: ({(NUM+1)..#ROLL}@(ROLL))+((#ROLL-NUM)*BONUS)
}
}

HIGHTN: 18
MIDDLETN: 14
LOWTN: 10

ATTACK_ROLL:2d10
DAMAGE_ROLL:2d6
CUNNING:3
FEROCITY:3

output [attack ATTACK_ROLL  and damage DAMAGE_ROLL with cunning CUNNING and ferocity FEROCITY against HIGHTN] named "[CUNNING]/[FEROCITY] Against [HIGHTN]"

output [attack ATTACK_ROLL and damage DAMAGE_ROLL with cunning CUNNING and ferocity FEROCITY against MIDDLETN] named "[CUNNING]/[FEROCITY] Against [MIDDLETN]"

output [attack ATTACK_ROLL and damage DAMAGE_ROLL with cunning CUNNING and ferocity FEROCITY against LOWTN] named "[CUNNING]/[FEROCITY] Against [LOWTN]"

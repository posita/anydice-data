CRITVAL: 10
DICE: 5

function: dxcrit NUM:n
{
  if NUM >= CRITVAL
  {
    result: 0
  }else
  {
    result: NUM
  }
}

function: dxroll DICE:d
{
  P: [count 0 in DICE]
  D: [dxcrit Pd10]
  R: 1@DICE
  Q: 1@D

  R: [highest of R and Q] + P * 10
  result: R
}

output [dxroll DICEd[dxcrit d10]] named "5d10, Critical Value 10"

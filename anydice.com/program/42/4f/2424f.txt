function: attack ROLL:n vs DEFENSE:n and ATTACK:n{
  if ROLL = 0 {result:MISS}
  if ROLL = 20 {result:CRITHIT}
  if ROLL + ATTACK >= DEFENSE {result:HIT}
  result:MISS
}

function: damage RESULT:n HITDAMAGE:n CRITDAMAGE:n{
  if RESULT = MISS {result : 0}
  if RESULT = HIT {result : HITDAMAGE}
  if RESULT = CRITHIT {result : CRITDAMAGE}
}


MISS : -1
HIT : -2
CRITHIT : -3

DEFENSE : 10

ATTACK: 8
HITDAMAGE: 1d8 + 2d6 + 4
CRITDAMAGE: 2d8 + 4d6 + 4


LOOP : {15}

loop DEFENSE over LOOP {
  output [damage [attack d20 vs DEFENSE and ATTACK] HITDAMAGE CRITDAMAGE] + [damage [attack d20 vs DEFENSE and ATTACK] HITDAMAGE CRITDAMAGE] + [damage [attack d20 vs DEFENSE and ATTACK] HITDAMAGE CRITDAMAGE] + [damage [attack d20 vs DEFENSE and ATTACK] HITDAMAGE CRITDAMAGE] + [damage [attack d20 vs DEFENSE and ATTACK] HITDAMAGE CRITDAMAGE] named "Damage for 5 attacks Flaming Rapier"
  output [damage [attack [highest 1 of 2d20] vs DEFENSE and ATTACK] HITDAMAGE CRITDAMAGE] + [damage [attack [highest 1 of 2d20] vs DEFENSE and ATTACK] HITDAMAGE CRITDAMAGE] + [damage [attack [highest 1 of 2d20] vs DEFENSE and ATTACK] HITDAMAGE CRITDAMAGE] + [damage [attack [highest 1 of 2d20] vs DEFENSE and ATTACK] HITDAMAGE CRITDAMAGE] + [damage [attack [highest 1 of 2d20] vs DEFENSE and ATTACK] HITDAMAGE CRITDAMAGE] named "Damage for 5 attacks with advantage Flaming Rapier"
}

MOD: 0

function: ROLL:s count groups {
  MAX: 0
  loop N over {1..6} {
    MAX: [highest of MAX and N * (ROLL = N)]
  }
  result: MAX
}

if MOD < 0 {
  COUNT: -MOD
  output [lowest 1 of COUNTd6] named "roll [COUNT] take lowest @ [MOD]"
} else {
  COUNT: MOD + 2
  output [COUNTd6 count groups] named "roll [COUNT] count groups @ [MOD]"
}

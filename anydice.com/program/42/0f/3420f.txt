MISS: 0
HIT: 1
CRIT: 2
LUCKY: 3

ATK: 9

DEF: 13

function: attack ROLL:n vs AC:n {
 if ROLL = 1 { result: MISS }
 if ROLL + ATK >= AC { result: ROLL }
 result: MISS
}

function: dodge ROLL:n vs DC:n {
  if DC = MISS { result: MISS }
  if ROLL + DEF >= DC + ATK { result: MISS }
  result: DC
}

function: is crit ROLL:n {
  if ROLL = MISS { result: MISS }
  if ROLL = {19..20} { result: CRIT}
  result: HIT
}

function: is lucky ROLL:n CRITHITMISS:n {
  if CRITHITMISS != CRIT { result: CRITHITMISS}
  if ROLL = {5..6} { result: LUCKY }
  result: CRIT
}

output [attack 2@2d20 vs 20] named "to hit AC 20"
output [dodge 1d20 vs [attack 2@2d20 vs 20]] named "after dodge"
output [is crit [dodge 1d20 vs [attack 2@2d20 vs 20]]] named "after crit"
output [is lucky 1d6 [is crit [dodge 1d20 vs [attack 2@2d20 vs 20]]]] named "after d6"

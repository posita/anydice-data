\\\*** Base Damage Calc Info ***\\\
ENEMY_AC: 10 \looped against enemy AC from 14-18, DMG suggested AC from CR4-16\
LEVEL: 5
STAT_MOD: 4
TURNS: 1
PROF: (LEVEL+7)/4
ACTIONSURGE: 0
SUPRISE: 2d6
CURSE: 0
RAYS: 3

\\\*** Scorching Ray ***\\\
ATTACK_DICE_A: 2d6
BONUS_DAM_MOD_A: 0
BONUS_DICE_A: SUPRISE
HIT_BONUS_A: 0
MAGIC_A: 1
SINGLE_HIT_A: PROF
N_ATTACKS_A: TURNS*RAYS + RAYS*ACTIONSURGE

\\\*** Genie EB ***\\\
ATTACK_DICE_B: 1d10
BONUS_DAM_MOD_B: STAT_MOD
BONUS_DICE_B: SUPRISE
HIT_BONUS_B: 0
MAGIC_B: 1
SINGLE_HIT_B: PROF
N_ATTACKS_B: TURNS*2 + 2*ACTIONSURGE

\\\*** Hex EB ***\\\
ATTACK_DICE_C: 1d10
BONUS_DAM_MOD_C: STAT_MOD + CURSE
BONUS_DICE_C: SUPRISE
HIT_BONUS_C: 0
MAGIC_C: 1
SINGLE_HIT_C: 0
N_ATTACKS_C: TURNS*2 + 2*ACTIONSURGE

\\\*** General Info ***\\\
CRIT_RANGE: 20
CRIT_BONUS: 0 \Any additional dice used specifically for a critical\
NORMAL_HIT: d20
HIT_ADV: [highest 1 of 2d20]


\\\*** Functions ***\\\
function: attack ROLL:n vs ENEMY_AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL >= CRIT_RANGE { result: CRITICALDAMAGE }
 if ROLL + ATTACK >= ENEMY_AC { result: HITDAMAGE }
 result: 0
}

function: single ROLL:n vs ENEMY_AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL + ATTACK >= ENEMY_AC { result: SINGLEDAMAGE }
 result: 0
}

ATTACK: STAT_MOD + PROF + MAGIC_A + HIT_BONUS_A
HITDAMAGE: ATTACK_DICE_A + MAGIC_A + BONUS_DAM_MOD_A + BONUS_DICE_A
CRITICALDAMAGE: 2dATTACK_DICE_A + STAT_MOD + MAGIC_A + BONUS_DAM_MOD_A + 2dBONUS_DICE_A + CRIT_BONUS
SINGLEDAMAGE: SINGLE_HIT_A
output N_ATTACKS_Ad[attack NORMAL_HIT vs ENEMY_AC] + [single NORMAL_HIT vs ENEMY_AC] named "S.Ray"
output N_ATTACKS_Ad[attack HIT_ADV vs ENEMY_AC] + [single NORMAL_HIT vs ENEMY_AC] named "Adv S.Ray"

ATTACK: STAT_MOD + PROF + MAGIC_B + HIT_BONUS_B
HITDAMAGE: ATTACK_DICE_B + MAGIC_B + BONUS_DAM_MOD_B + BONUS_DICE_B
CRITICALDAMAGE: 2dATTACK_DICE_B + STAT_MOD + MAGIC_B + BONUS_DAM_MOD_B + 2dBONUS_DICE_B + CRIT_BONUS
SINGLEDAMAGE: SINGLE_HIT_B
output N_ATTACKS_Bd[attack NORMAL_HIT vs ENEMY_AC] + [single NORMAL_HIT vs ENEMY_AC] named "Genie EB"
output N_ATTACKS_Bd[attack HIT_ADV vs ENEMY_AC] + [single NORMAL_HIT vs ENEMY_AC] named "Adv Genie EB"

ATTACK: STAT_MOD + PROF + MAGIC_C + HIT_BONUS_C
HITDAMAGE: ATTACK_DICE_C + MAGIC_C + BONUS_DAM_MOD_C + BONUS_DICE_C
CRITICALDAMAGE: 2dATTACK_DICE_C + STAT_MOD + MAGIC_C + BONUS_DAM_MOD_C + 2dBONUS_DICE_C + CRIT_BONUS
SINGLEDAMAGE: SINGLE_HIT_C
output N_ATTACKS_Cd[attack NORMAL_HIT vs ENEMY_AC] + [single NORMAL_HIT vs ENEMY_AC] named "Hex EB"
output N_ATTACKS_Cd[attack HIT_ADV vs ENEMY_AC] + [single NORMAL_HIT vs ENEMY_AC] named "Adv Hex EB"

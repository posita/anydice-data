function: attack ROLL:n vs DEFENSE:n and ATTACK:n{
  result:CRITHIT
  
}

function: damage RESULT:n HITDAMAGE:n CRITDAMAGE:n ARCANEWEAPON:n{
  if RESULT = MISS {result : 0}
  if RESULT = HIT {result : HITDAMAGE + ARCANEWEAPON}
  if RESULT = CRITHIT {result : CRITDAMAGE + (2 * ARCANEWEAPON)}
}


MISS : -1
HIT : -2
CRITHIT : -3

DEFENSE : 10

DAGGERATTACK: 7
DAGGERHITDAMAGE: 1d4 + 4
DAGGERCRITDAMAGE: 2d4 + 4
CROSSBOWATTACK: 7
CROSSBOWHITDAMAGE: 1d8 + 4
CROSSBOWCRITDAMAGE: 2d8 + 4

LOOP : {15}

loop DEFENSE over LOOP {
  output [damage [attack d20 vs DEFENSE and DAGGERATTACK] DAGGERHITDAMAGE DAGGERCRITDAMAGE 0] + [damage [attack d20 vs DEFENSE and DAGGERATTACK] DAGGERHITDAMAGE DAGGERCRITDAMAGE 0] named "Damage against [DEFENSE] for Dagger"
  output [damage [attack d20 vs DEFENSE and CROSSBOWATTACK] CROSSBOWHITDAMAGE CROSSBOWCRITDAMAGE 0] + [damage [attack d20 vs DEFENSE and CROSSBOWATTACK] CROSSBOWHITDAMAGE CROSSBOWCRITDAMAGE 0] named "Damage against [DEFENSE] for Crossbow"
  output [damage [attack d20 vs DEFENSE and DAGGERATTACK] DAGGERHITDAMAGE DAGGERCRITDAMAGE 1d6] + [damage [attack d20 vs DEFENSE and DAGGERATTACK] DAGGERHITDAMAGE DAGGERCRITDAMAGE 1d6] named "Damage against [DEFENSE] for Dagger with Arcane Weapon"
  output [damage [attack d20 vs DEFENSE and CROSSBOWATTACK] CROSSBOWHITDAMAGE CROSSBOWCRITDAMAGE 1d6] + [damage [attack d20 vs DEFENSE and CROSSBOWATTACK] CROSSBOWHITDAMAGE CROSSBOWCRITDAMAGE 1d6] named "Damage against [DEFENSE] for Crossbow with Arcane Weapon"
}

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15\
ENEMYAC: d{19}

ATTACKMOD: 2
DMGBONUS: 5
PROFICIENCY: 6
ACCBONUS: 5
CRITROLL: 18
ATTACKTIMES: 4
WEAPONDICE: 1d12
DAMAGE: WEAPONDICE + ATTACKMOD + DMGBONUS
CRITDAMAGE: WEAPONDICE + WEAPONDICE + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "+2 Greataxe +6 Proficiency +5 Attack Modifier"

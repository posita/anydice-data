function: rolls ROLL:s with REROLL:s{
COUNTONES: [count 1 in ROLL]
COUNTSIXES: [count 6 in ROLL]
REROLLCOUNTONE : [count 1 in REROLL]
RESERVE: COUNTONES

 if COUNTONES > 0 {
D : ROLL - 1 + REROLL


if REROLLCOUNTONE > 0 { 
RESERVE : RESERVE + 1
if COUNTSIXES > 0 { 
  D: D - 5 
  RESERVE : RESERVE + 1
}
 }
}

else {D : ROLL}

result: D
}

output 3d6 named "3d6"
output [rolls 3d6 with 1d6] named "reroll only rules"
output [highest 3 of 4d6] named "4d6 drop highest"

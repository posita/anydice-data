function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  result: CRITICALHIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage RESULT:n {
  if RESULT = MISS {result : 0}
  if RESULT = HIT {result : HITDAMAGE}
  if RESULT = CRITICALHIT {result: CRITDAMAGE}
}

MISS: -1
HIT: -2
CRITICALHIT: -3

ATTACK: 8
DEFENSE: 15
HITDAMAGE: 1d4 + 1
CRITDAMAGE: 2d4 + 1

loop ATTACK over {0..5} {
 output [damage [attack d20 vs DEFENSE]] named "[ATTACK]"
}

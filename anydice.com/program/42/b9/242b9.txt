function: calculatedamage DAMAGE:n plus NETHITS:n {
 if NETHITS < 1 { result: 0 }
 result: DAMAGE + NETHITS
}

function: postdamage NETDAMAGE:n minus REDUCTION:n {
 if (NETDAMAGE - REDUCTION) < 1 { result: 0 }
 result: NETDAMAGE - REDUCTION
}

AT: 10d (d6>4)
DE: 7d (d6>4)

DA: 10
AR: 3d6

NH: AT-DE

EXPDAM: [calculatedamage DA plus NH]
output EXPDAM

DAMTAK: [postdamage EXPDAM minus AR]
output DAMTAK

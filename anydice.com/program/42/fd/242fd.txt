\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ABILITYMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

ABILITYMOD: 4
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 1
CRITDAMAGE: 1
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs 15] named "+6 advantage vs dc 15"

function: rerolltwo NUMBER:n{
 if NUMBER = 2 { 
 NUMBER: 1d20 }
 result: NUMBER
}


function: ROLL:n with ATTACKBONUS:n for DAMAGE vs DEFENSE:n {
 if ROLL = 1 { result: 1 }
 if ROLL = 20 {
  if ROLL + ATTACKBONUS >= DEFENSE { result: [maximum of DAMAGE] }
  result: DAMAGE
 }
 result: (ROLL + ATTACKBONUS >= DEFENSE) * DAMAGE
}

loop AC over {15..20} {
 output [[rerolltwo d20] with +4 for d8 + 3 vs AC] named "TWF [AC]"
}

loop AC over {15..20} {
 output [d20 with +4 for d10 + 3 vs AC] named "2h [AC]"
}

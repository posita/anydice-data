AC: 16
DEX: 4
PROF: 3
DAGGER: 1d4
SHORTS: 1d6
RAPIER: 1d8
PISTOL: 1d10
SA: 3d6

function: first A:n for AD:d plus AB:n second B:n for BD:d plus BB:n{
  if A = 20 & B = 20 {result: 2dAD + 2dBD + 2dSA + DEX}
  if A = 20 & B + BB + DEX + PROF >= AC & B != 1 {result: 2dAD + 2dSA + 1dBD + DEX}
  if A = 20 {result: 2dAD + 2dSA + DEX}
  if A + AB + DEX + PROF >= AC & A != 1 & B = 20 {result: 1dAD + SA + 2dBD + DEX}
  if A + AB + DEX + PROF >= AC & A != 1 & B + BB + DEX + PROF >= AC & B != 1 {result: 1dAD + SA + 1dBD + DEX}
  if A + AB + DEX + PROF >= AC & A != 1 {result: 1dAD + SA + DEX}
  if B = 20 {result: 2dBD + 2dSA}
  if B + BB + DEX + PROF >= AC & B != 1 {result: 1dBD + SA}
  result: 0
}

\output [first d20 for RAPIER plus 1 second d20 for DAGGER plus 0 ]

output [first d20 for RAPIER plus 1 second d20 for PISTOL plus 1 ]\

output [first d20 for DAGGER plus 0 second d20 for DAGGER plus 0 ]
output [first d20 for RAPIER plus 1 second 1 for 0 plus 0 ]

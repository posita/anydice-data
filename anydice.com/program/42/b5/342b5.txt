function: swap X:n {
 if X=1006 {X: 1006} else {X: X/1001}
 result: X
}
function: reset X:n {
 if X=1006 {X: 1006} else {X: X*1001}
 result: X
}

function: compare X:n Y:n {
 if X>Y {result: X} else {result: Y}
}

function: apply edge to A:n B:n C:n {
 BX: [swap B]
 MLOWEST: 3 @ [sort {A,BX,C}]
 if BX=MLOWEST {BXX: MEDGE} else {BXX: BX}
 BFIX: [compare BX BXX]
 if A=MLOWEST {if BX>MLOWEST {AX: 1d6} else {AX: A}} else {AX: A} \Compare A and C values as well\
 if C=MLOWEST {if A>MLOWEST {CX: 1d6} else {CX: C}} else {CX: C}
 AFIX: [compare A AX]
 CFIX: [compare C CX]
 result: AFIX+[reset BFIX]+CFIX
}

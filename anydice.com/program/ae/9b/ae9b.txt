function: better of A:n and B:n target T:n {
  if [absolute A - T] < [absolute B - T] { result: A }
  if [absolute A - T] > [absolute B - T] { result: B }
  result: 1d{A, B}
}
function: better of A:n plus minus B:n target T {
  result: [better of A+B and A-B target T]
}

loop T over {1..10} {
  output [absolute T - [better of 1d10 plus minus 1d6 target T]] named "1d10 +/- 1d6, distance from [T]"
}

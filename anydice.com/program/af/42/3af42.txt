function: POS:n at better of STATS_A:s and STATS_B:s {
  SUM_A: 0 + STATS_A
  SUM_B: 0 + STATS_B
  if SUM_A > SUM_B { result: POS @ STATS_A }
  else { result: POS @ STATS_B }
}

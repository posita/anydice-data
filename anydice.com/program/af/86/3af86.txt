IGNORE: d{}

function: POS:n at ROLL:s when sum is SUM:n helper {
  if 0 + ROLL = SUM { result: POS @ ROLL }
  result: IGNORE
}

function: POS at ROLL when sum is SUM:n {
  result: [POS at ROLL when sum is SUM helper]
}

N: 3
ABILITIES: N d [highest 3 of 4d6]

BETTER_SUM: [highest 1 of 2 d ABILITIES]

loop P over {1..N} {
  output [P at ABILITIES when sum is BETTER_SUM] named "Ability [P]/[N]"
}

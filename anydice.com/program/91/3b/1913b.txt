function: normal ROLL:s TARGET:n {
 result: ROLL >= TARGET
}

function: exploding ROLL:s TARGET:n {
 if ROLL = 6 { result: 1 + [exploding d6 TARGET] }
 result: ROLL >= TARGET
}

set "maximum function depth" to 5

loop DICE over {3} {
 loop TARGET over {5,3} {
   output [normal DICE d6 TARGET] named "Normal [DICE]d6 w/ success >= [TARGET]"
   output DICEd[exploding d6 TARGET] named "Exploding [DICE]d6 w/ success >= [TARGET]"
  }
}

CRITRANGE: 20   \ Lowest rolled number where the character triggers a critical hit. \
CRITONLYDAM: 0  \ Damage only added on a critical hit. \

function: proficiency {
    if LEVEL >=17 {
        result: 6
    }
    if LEVEL >= 13 {
        result: 5
    }
    if LEVEL >= 9 {
        result: 4
    }
    if LEVEL >= 5 {
        result: 3
    }
    else {
        result: 2
    }
}

function: roll ROLL:n at plus ATTACKBONUS:n for CRITABLEDAM:d plus HITDAM:d {
    if ROLL >= CRITRANGE {
        result: CRITONLYDAM + CRITABLEDAM + CRITABLEDAM + HITDAM
    }
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= AC - ATTACKBONUS {
        result: CRITABLEDAM + HITDAM
    }
    else {
        result: 0
    }
}

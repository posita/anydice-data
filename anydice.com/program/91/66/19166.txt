function: attack D:n vs AC:n {
if D = 20 {result: CRIT}
if D >= AC  {result: DAMAGE}
if D < AC {result: 0}
}

function: explode DIE:d {
 result: [explode DIE helper]
}

function: explode N:n helper {
 if N = {1,2} { result: DIE }
 result: N
}

DAMAGE: 1d[explode 1d12]+4
CRIT: 2d[explode 1d12]+4

loop AC over {10..20} {

output [attack d20 vs AC]
}

DAMAGE: 1d8+4
CRIT: 2d8+4


loop AC over {10..20} {

output 2d[attack d20 vs AC]
}

MISS: 0
HIT: 1
CRITICALHIT: 2

FAIL: 0
SAVE: 1
FSAVE: 2


\Your Stats\
ATTACK: 0
DC: 0


\Range of Enemy ACs and Profiencies\
DEFENSE: d{13..25}
PROF: d{0..16}


\Attack Roll Functions (DON'T TOUCH)\
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}






\Spell Save Functions (DON'T TOUCH)\
function: spell ROLL:n and PROF:n {
 if ROLL + PROF >= DC{ 
  if PASSNOD = 1 { result: FSAVE}  
   result: SAVE 
 }
 result: FAIL
}

function: spell damage for SPELLRESULT:n {
 if SPELLRESULT = FAIL{
  result: SPELLDAMAGE
 }
 if SPELLRESULT = SAVE{
  result: (SPELLDAMAGE / 2)
 }
 if SPELLRESULT = FSAVE{
 result: MISSDAMAGE
 }
}

function: spellattack {
 result: [spell damage for [spell d20 and PROF]]
}

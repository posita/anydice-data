function: attack ROLL:n ATTACK:n vs DEFENSE:n dice DAMAGEDICE:d CRITDAMAGEDICE:d{
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITDAMAGEDICE }
 if ROLL + ATTACK >= DEFENSE+10 { result: CRITDAMAGEDICE }
 if ROLL + ATTACK >= DEFENSE { result: DAMAGEDICE}
 result: MISS
}

FINISHER: 2d6+1 + (3d6)*2
CRITFINISHER: (2d6+1+(3d6)*2)*2

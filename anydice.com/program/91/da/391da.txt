function: ffxivdmg checkdie CDIE:n checkmod CMOD:n def DEF:n base BASE:n direct DIRECT:n crit CRIT:n {
  if (CDIE = 20) {result: BASE+DIRECT+CRIT }
  if (CDIE+CMOD >= DEF) {result: BASE+DIRECT }
  result: BASE
}

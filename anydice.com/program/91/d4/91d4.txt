function: damage from AV:s attack with DAM:n damage against DEF:n defense {
 DAMAGE: 0
 ATTACK: 0
 
 loop P over {1..#AV} {
  if P@AV > 2 {
   ATTACK: ATTACK + 1
  }
  if P@AV > 5 {
   DAMAGE: DAMAGE + 1
  }
 }
 
 if ATTACK >= DEF {
  result: 2
 }

 if (ATTACK + DAMAGE) >= DEF {
  result: 1
 }

 result: 0

}

DM:1

loop A over {3..7} {
 loop D over {2..5} {
   output [damage from Ad6 attack with DM damage against D defense] named "Attack +[A], damage [DM] against defense [D]"
 }
}

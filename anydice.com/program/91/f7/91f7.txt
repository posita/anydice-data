function: calculate with target TN:s skill SKILL:s and expertise EXPERTISE:n {
  VALUE: 0
  loop N over {1..#SKILL}{
    ROLL: N@SKILL
    if ROLL > 0 {
      if ROLL <= TN { VALUE: VALUE + 1 }
      if ROLL <= EXPERTISE { VALUE: VALUE + 1 }
      if ROLL = 20 { VALUE: VALUE - 1 }
    }
  }
  result: VALUE
}

output [calculate with target 3 skill 3d12 and expertise 0] + [calculate with target 3 skill 3d10 and expertise 0]

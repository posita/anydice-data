function: missshot ROLL:n on MISS_BORDER:n{
  if ROLL <= MISS_BORDER {
    result: 0
  }
  result: ROLL
}

function: magical forest ROLL:n on REP_BORDER:n to REPLACE:n{
  if ROLL <= 1 {
    result: REPLACE + 1d45
  }
  if ROLL <= REP_BORDER {
    result: REPLACE
  }
  result: ROLL
}


function: master spark MAX:n{
  result: [missshot 1d(MAX) on 22]
}

function: magical spark MAX:n{
  result: [missshot [magical forest 1d(MAX) on 11 to MAX] on 22]
}


function: witch power MAX:n{
  result: 1d(MAX + 10)
}

function: magical power MAX:n{
  result: [magical forest 1d(MAX + 10) on 11 to (MAX + 10)]
}

loop N over {0..6}{
 output [magical spark 45 - (N * 7)] >= 57 named "ろっく×[N]"
}

output [magical power 45] + [magical spark 45] >= 57 named "先発から2パン"

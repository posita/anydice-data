YOURMELEE: 20
YOUREVASION: 20
MONSTERMELEE: 20
MONSTEREVASION: 20
DAMAGEDICE: 1
DAMAGESIDES: 8
WEAPONWEIGHT: 1


YOURMELEEROLL: [highest of (YOURMELEE + 1d20) - (MONSTEREVASION + 1d20) and 0]
MONSTERMELEEROLL: (MONSTERMELEE + 1d20) - (YOUREVASION + 1d20)


function: YOURMELEEROLL :n {
if YOURMELEEROLL = 0 {result: 1000}
if YOURMELEEROLL > 0 {result: 3d4} 
}


function: MONSTERMELEEROLL :n {
if MONSTERMELEEROLL <= -10 - WEAPONWEIGHT {result: 2}
if MONSTERMELEEROLL > -10 - WEAPONWEIGHT {result: 1}
}


CRITS: YOURMELEEROLL / (4 + WEAPONWEIGHT)


output [MONSTERMELEEROLL]d[highest of ((CRITS+DAMAGEDICE)dDAMAGESIDES - [YOURMELEEROLL]) and 0]

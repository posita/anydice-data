SAVE: 5
REROLLMAX: 0
PALAWOUNDBONUS: 0
HITBONUS: 0

function: ROLL:n hit TARGET:n {
 if ROLL >= TARGET {
  result: 1
 }
 result:0
}

function: hit TARGET:n {
 result: [d6 hit TARGET]
}

function: ROLL:n rehit TARGET:n REHIT:n {
 if ROLL >= TARGET {
  result: 1
 }
 if ROLL <= REHIT {
  result: [hit TARGET]
 }
 result:0
}

function: rehit TARGET:n REHIT:n {
 result: [d6 rehit TARGET REHIT]
}

function: ROLL:n nosave REND:n {
 if ROLL >= (SAVE+REND) {
  result: 0
 }
 if ROLL <= REROLLMAX+REND & REROLLMAX > 0 {
  result: d6 < (SAVE+REND)
 }
 result: 1
}

function: nosave REND:n {
 result: [d6 nosave REND]
}

function: dowound TOHIT:n TOWOUND:n REND:n REHIT:n REWOUND:n {
 result: [rehit TOHIT REHIT]*[rehit TOWOUND REWOUND]*[nosave REND]
}

function: attacks ATTACKS:n hit TOHIT:n wound TOWOUND:n rend REND:n dmg DAMAGE:n {
 result: [attacksr ATTACKS hit TOHIT wound TOWOUND rend REND dmg DAMAGE reroll 0 0]
}

function: attacksr ATTACKS:n hit TOHIT:n wound TOWOUND:n rend REND:n dmg DAMAGE:n reroll REHIT:n REWOUND:n {
 TOTAL: 0
 loop N over {1..ATTACKS} {
   TOTAL: TOTAL + DAMAGE * [dowound TOHIT TOWOUND REND REHIT REWOUND]
 }
 result: TOTAL
}

function: attackdthree ATTACKS:n hit TOHIT:n wound TOWOUND:n rend REND:n {
 result: [attackrdthree ATTACKS hit TOHIT wound TOWOUND rend REND reroll 0 0]
}

function: attackrdthree ATTACKS:n hit TOHIT:n wound TOWOUND:n rend REND:n reroll REHIT:n REWOUND:n {
 TOTAL: 0
 loop N over {1..ATTACKS} {
   TOTAL: TOTAL + d3 * [dowound TOHIT TOWOUND REND REHIT REWOUND]
 }
 result: TOTAL
}

function: daemonbladeattack HITROLL:n HITREROLL:n wnd WOUNDROLL:n WOUNDREROLL:n reroll REHIT:n REWOUND:n {
 HIT: 0
 DAEMONHIT: 0
 if HITROLL >= 4 {
  HIT: 1
  if HITROLL >= 6 {
   DAEMONHIT: 1
  }
 } else {
  if HITROLL <= REHIT {
   if HITREROLL >= 4 {
    HIT: 1
    if HITREROLL >= 6 {
     DAEMONHIT: 1
    }
   }
  }
 }
 
 WOUND: 0
 WOUNDRESULT: WOUNDROLL
 if WOUNDROLL >= 3 {
  WOUND: 1
 } else {
  if WOUNDROLL <= REWOUND {
   WOUNDRESULT: WOUNDREROLL
   if WOUNDREROLL >= 3 {
    WOUND: 1
   }
  }
 }

 TOTAL: 0
 if WOUNDRESULT >= 2 {
  TOTAL: DAEMONHIT
 }

 TOTAL: TOTAL + HIT * WOUND * [hit (8-SAVE)]

 result: TOTAL
}

function: skullreaperattacks ATTACKS:n reroll REHIT:n REWOUND:n {
 TOTAL: 0
 loop N over {1..ATTACKS} {
  TOTAL: TOTAL + [daemonbladeattack (d6) (d6) wnd (d6) (d6) reroll REHIT REWOUND]
 }
 result: TOTAL
}

function: retriattack HITROLL:n {
 if HITROLL >= 6 {
   result: 2
 }
 result: [HITROLL hit 3]*[hit (3-PALAWOUNDBONUS)]*[hit (8-SAVE-1)]*2

}

function: retriattacks ATTACKS:n {
 TOTAL: 0
 loop N over {1..ATTACKS} {
  TOTAL: TOTAL + [retriattack (d6 + HITBONUS)]
 }
 result: TOTAL
}

 if HITROLL = 6 {
   result: [HITROLL hit 3]*[hit 3]*[hit (8-SAVE-5)]*d3
 }

function: insensateattack WOUNDROLL:n charge CHARGE:n carnage CARNAGE:n {
 DMG: d6

 CARNAGEBONUS: 3
 if CARNAGE = 2 {
  CARNAGEBONUS: d3
 }
 if CARNAGE = 3 {
  CARNAGEBONUS: 1
 }

 TOTAL: [WOUNDROLL hit 2]*[rehit 4 CHARGE]*[hit (8-SAVE-2)]*DMG
 if WOUNDROLL = 6 {
   TOTAL: TOTAL + CARNAGEBONUS
 }
 result: TOTAL
}

function: insensateattacks ATTACKS:n charge CHARGE:n carnage CARNAGE:n {
 TOTAL: 0
 loop N over {1..ATTACKS} {
  TOTAL: TOTAL + [insensateattack d6 charge CHARGE carnage CARNAGE]
 }
 result: TOTAL
}

function: carnageattack CARNAGE:n reroll REHIT:n REWOUND:n {
 TOTAL: [rehit CARNAGE REWOUND]
 DMG: 1 + TOTAL * 7
 result: [rehit 4 REHIT] * DMG
}

function: carnageattacks ATTACKS:n carnage CARNAGE:n reroll REHIT:n REWOUND:n {
 TOTAL: 0
 loop N over {1..ATTACKS} {
  TOTAL: TOTAL + [carnageattack CARNAGE reroll REHIT REWOUND]
 }
 result: TOTAL
}

function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

function: DICE:s reroll N:n less than BOUND:s as REROLL:d{
 TOSUB: 0
 NUMSUB: 0
 loop I over {0..N-1} {
  if (#DICE-I)@DICE < BOUND {
   TOSUB: TOSUB + (#DICE-I)@DICE
   NUMSUB: NUMSUB + 1}
 }
 result: DICE - TOSUB + NUMSUBdREROLL
}

\Attribute Dice\
AA: 1d12
AB: 1d10
AC: 3d20
\Skill TN\
TN: 3
\Expertise Rank\
EXP: 1

function: roll A:s B:s C:s {
SUM: 0
if A = 12 {SUM: SUM - 1}
if B = 10 {SUM: SUM - 1}
if C = 20 {SUM: SUM - 1}
if A <= TN {SUM: SUM + 1 + (A <= EXP)}
if B <= TN {SUM: SUM + 1 + (B <= EXP)}
if C <= TN {SUM: SUM + 1 + (C <= EXP)}
result: SUM
}
output [roll AA AB AC]

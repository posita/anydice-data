\Skill adds +2 to total roll \
loop S over {0..7} {

    \Talent gives number of dice rolled, everyone gets 1D6 free of charge\
    loop T over {1..4} {

        \Attributes are mods provided by boons, modify the roll total\
        loop A over {-1,0,1, 2, 3} {

            B: S*2 \Skills adding the +2 modified by Attributes\
            D: T + A + 1 \The free D6 to ensure its a sort of bellcurve at the minimum\

                        \Check qualifications - you can't learn a Skill Level that is higher than your Att + Tal score \
                        if(S <= A + T) {
                 \Output result\
                 output [highest 2 of Dd6] + B named "Skill Level: [S], Talent: [T], Attribute: [A], Bonus [B] => Rolling [D]d6 + [B]:"
                        }

        }    
    }
}

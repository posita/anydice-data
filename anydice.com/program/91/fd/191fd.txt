function: gate X:n under T:n {
  if X > T {
    result: 0
  } else {
    result: X
  }
}

function: roll S:n vs T:n {
  R: {}
  loop N over {1..S} {
    R: R + [gate d{0..9} under T]
  }
  result: R
}

loop S over {1..5} {
  loop T over {3..7} {
    output [roll S vs T] named "Skill [S] vs trait [T]"  
  }
}

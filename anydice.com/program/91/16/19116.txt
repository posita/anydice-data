function: attack ATTACKMOD:n vs ac AC:n {
 ROLL:d20
 result: ROLL + ATTACKMOD >= AC
} 

function: ROLL:n damage DAMAGEDIE:n plus MOD:n {
 if ROLL < 20 {result:DAMAGEDIE + MOD}
 if ROLL = 20 {result:DAMAGEDIE + DAMAGEDIE + MOD}
}

function: ROLL:n plus ATTACKMOD:n vs ac AC:n for DAMAGEDIE:n plus MOD:n damage{
 if (20 + ATTACKMOD) < AC {result:0}
 else{
  if ROLL = 20 {result:DAMAGEDIE + DAMAGEDIE + MOD}
  if (ROLL + ATTACKMOD) >= AC {result:DAMAGEDIE + MOD}
  if (ROLL + ATTACKMOD) < AC {result:0}
 }
}

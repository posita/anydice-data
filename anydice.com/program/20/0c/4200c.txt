function: attack ROLL:s on HIT:n energized TOKEN:n {

    NUM_HITS: [count {HIT..10} in ROLL]
    NUM_CRITS: [count {9, 10} in ROLL]
    DAMAGE: NUM_HITS + NUM_CRITS
    
    \ has energized token \
    if TOKEN = 1 {
        if NUM_CRITS <= 2 {
            DAMAGE: DAMAGE + 2
        }
        else if NUM_CRITS = 3 {
            DAMAGE: DAMAGE + 1
        }
    }

    result: DAMAGE
}

output [attack 5d10 on 4 energized 1] named "[White Devil] Gundam Hyper Bazooka w/ Energized token"

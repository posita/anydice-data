\chose what you want to display (1) or hide (0)\
DES: {4, 6, 8, 10}
NBR: {1..2}
SEU: {7, 6, 5}

function: push D:n {
 result: [highest 1 of [explode 2dD]]
}

function: JET:s over SEUIL:n{
  loop I over JET{
  if I < S {result: 0} 
    else {
      if I < (S+5) {result: 1} 
        else {result: 2}
}}}

function: N:n dc D:n over S:n with P:n {
  if P = 0 
    {result: [[explode NdD] over S]}
    else {
      X:{}
      loop I over {1..N} {
        X: [[push D] over S] | X
      }
      result: X}
}

loop N over NBR {
  loop D over DES {
    loop S over SEU {
      R: [N dc D over S with 0]
      output R named "[N]:[D]:[S]"
}}}

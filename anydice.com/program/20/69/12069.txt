\**** Damage calculator ****\

function: divine {
}

function: bonusroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: BCRIT } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD >= AC { result: BDMG }
 result: 0
}

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD >= AC { result: DAMAGE }
 result: 0
}

function: multi X:n  {
 TOTAL: 0
 BA: 0
  loop N over {1..X} {
    DAMAGE: [attackroll ATKROLL vs ENEMYAC]
    TOTAL: TOTAL + DAMAGE
  }

 if BONUS = 1 {BA:[bonusroll ATKROLL vs ENEMYAC]}
 TOTAL: TOTAL + BA
 if L >= 3 {TOTAL: [divine]}
 BONUS: 0
 result: TOTAL
}

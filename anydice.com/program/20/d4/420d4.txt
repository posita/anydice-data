
function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output [reroll {1} on d20] + 5 named "luck"
output 1d20 + 5 named "flat"
output [reroll {1} on [highest 1 of 2d20]] + 5 named "luck & advantage"
output [highest 1 of 2d20] + 5 named "advantage"

function: explodingsuccess ROLL:n success SUCCESS:s explode EXPLODE:s {
 if ROLL = EXPLODE {
   result: (ROLL = SUCCESS) + [explodingsuccess d6 success SUCCESS explode EXPLODE]
 }
 else {
   result: ROLL = SUCCESS
 }
}

output [explodingsuccess d6 success {4..6} explode 6] named "d6 >= 4, explode 6"
output [explodingsuccess d6 success {4..6} explode 1] named "d6 >= 4, explode 1"
output [explodingsuccess d6 success {4..6} explode {1,6}] named "d6 >= 4, explode 1 and 6"
output [explodingsuccess d6 success {4..6} explode {1,3,5}] named "d6 >= 4, explode odd rolls"

function: attack AC GWM {
  R: [highest 1 of 2d20] + 7 - 5 * GWM
  result: [recifcrit R AC GWM]
}
function: recifcrit R:n AC GWM {
  if R = 20 {
    result: 2 * d12 + [attack AC GWM] + 2 + 10 * GWM
  } else {
    if R >= AC {
      result: d12 + 2 + 10 * GWM
    } else {
      result: 0
    }
  }
}
function: attackifop P:n AC GRM {
  if P = 1 {
    result: 0
  } else {
    result: [attack AC GRM]
  }
}
output [attack 16 1] + [attackifop d2 16 1] + [attack 16 1] + [attack 16 1]

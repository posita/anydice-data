function: drakonym X:s {
  A: 0
  loop N over {1..#X}{
    if N@X = 5 { A: A + 1 }
    if N@X = 6 { A: A + 2 }
  }
  result: A
}

function: drakonym X:s with Y:n boon {
  A: 0
  B: Y
  loop N over {1..#X}{
    if N@X = 4 & B > 0 {
      A: A + 1
      B: B - 1
    }
    if N@X = 5 { A: A + 1 }
    if N@X = 6 { A: A + 2 }
  }
  result: A
}

function: drakonym X:s with Y:n bane {
  A: 0
  B: Y
  loop N over {1..#X}{
    if N@X = 5 & B > 0 {
      A: A - 1
      B: B - 1
    }
    if N@X = 5 { A: A + 1 }
    if N@X = 6 { A: A + 2 }
  }
  result: A
}


loop I over {8..8} {
  output [drakonym Id6] named "[I] dice"
  loop J over {1..4} {
    output [drakonym Id6 with J boon] named "[I] dice, [J] boon"
  }
  loop K over {1..4} {
    output [drakonym Id6 with K bane] named "[I] dice, [K] bane"
  }
}

TARGETNUMBER: 10

function: roll against target number TN:n with a value of VALUE:n {
  result: [check 1@DICE 2@DICE TN VALUE]
}

function: check FIRST:n SECOND:n TN:n VALUE:n {
  if SECOND+VALUE >= TN { result: FIRST }
  if FIRST+VALUE >= TN { result: SECOND }
  else {result: 0}

}

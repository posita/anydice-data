
function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 7
 if SUCCESSES > 0 { result: SUCCESSES + (ROLL = 10) }
 if [count 1 in ROLL] { result: -1 }
 result: 0
}

loop DICE over {1..10} {
 output [evaluate DICE d10] named "[DICE]d"
}

function: blades ROLL:s {
 if 1@ROLL = 6 {
  result: 2 + (2@ROLL = 6)
 }
 result: 1@ROLL >= 4
}

loop DICE over {1..10} {
 output DICE d [blades Pd6] named "[DICE]d"
}

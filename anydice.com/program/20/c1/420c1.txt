function: roll X:n for target T:n{
 if ((X<T) & (X!=20)) | ((X=20) & (X<T-10)) | (X=1 & X<T+10){
  result: 0
 }
 else if ((T<=X & X<(T+10))&(X!=20)) | ((X=20) & (X<T)){
  result: 1
 }
 else {
  result: 2
 }
}

function: triple a X for T:n {
 result: [roll X for target T]+[roll X for target T+5]+[roll X for target T+10]
}

function: first a X for T:n{
 result: [roll X for target T-1]+[roll X for target T+5]
}
function: second a X for T{
 result: [roll X for target T]+[roll X for target T+5-1]
}

output [second a 1d20 for TT] named "boost the second"

}

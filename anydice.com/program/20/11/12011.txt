function: explode N:n of DICE:d {
  LEN: #(DICE)
  TEMP: 0
  EXPLODES: 0
  loop I over {1..LEN} {
    MAX: [maximum of I@DICE]
    if (EXPLODES < N) {
      TEMP: TEMP + [explode 1dMAX helper MAX]
      EXPLODES: EXPLODES + 1
    }
    else {
      TEMP: TEMP + 1dMAX
    }
  }

  result: TEMP
}

function: explode ROLL:n helper MAX:n {
  if ROLL = MAX {
     result: ROLL + [explode 1dMAX]
  }
  else {
      result: ROLL
  }
}

set "explode depth" to 8

output [explode 0 of 2d6] named "explode 0 of 2d6"
output 2d6 named "straight 2d6"

output [explode 2 of 2d6] named "explode 2 of 2d6"
output 2d[explode d6] named "straight explode"

TYPE: d6
NUM: 3

function: iseven N:n {
  result: (2*(N/2))=N
}

function: xplhigh ROLL:n {
  if ((2*(ROLL/2))=ROLL) {
    result: [highest of ROLL and [xplhigh TYPE]]
  } else {
    result: ROLL
  }
}

set "maximum function depth" to 5

TMP: 1@NUMd[xplhigh TYPE]
output TMP named "Highest roll"
output [iseven TMP] named "Highest roll is even"

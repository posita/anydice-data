function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE }
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result:0
}

function: pierceroll DICE:s below UNDERDICE:n as RERE:n {
 NEW_DICE: {RERE}
 if 1@DICE <= UNDERDICE {
  result: RERE
 } else {
  result: DICE
 }
}

function: piercerollcrit DICE:s below UNDERDICE:n as RERE:n {
 NEW_DICE: {RERE}
 if 3@DICE <= UNDERDICE {
  PAUSEDICE: {1,2}@DICE + {RERE}
  result: PAUSEDICE
 } else {
  result: DICE
 }
}

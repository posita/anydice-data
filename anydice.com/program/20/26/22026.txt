ATK:  1d20  \ attack roll: use 1@2d20 for advantage or 2@2d20 for disadvantage \
ATK_DIS: 2@2d20
HIT:    +8  \ bonus to attack roll \
SS_HIT: -5  \ Sharpshooter bonus to attack roll \
\AC:      8   target AC: attack hits if ATK + HIT >= AC \
CRIT:   20  \ minimum attack roll to crit \
DMG:   2d10 \ damage rolled on successful hit \
SS_DMG: 2d10+10 \ Sharpshooter damage bonus \

function: attack ROLL:n plus BONUS:n {
  \ return the number of times damage is applied on an attack roll of ROLL \
  if ROLL <= 1 { result: 0 }     \ natural 1 is always a miss \
  if ROLL >= CRIT { result: 2 }  \ critical hit deals damage damage twice \
  result: ROLL + HIT + BONUS >= AC
}

function: multiattack N:n times {
  result: [roll ATK plus HIT N times for DMG]
}


function: automultiattack N:n times {
  result: [roll ATK_DIS plus HIT N*2 times for DMG]
}


function: ssmultiattack N:n times {
  result: [roll ATK plus SS_HIT N times for SS_DMG]
}


function: ssautomultiattack N:n times {
  result: [roll ATK_DIS plus SS_HIT N*2 times for SS_DMG]
}

function: roll TH:d plus B:n N:n times for D:d{
  if N <= 0 {
    result: 0
  }
  DAMAGE: 0
  loop I over {1..N} 
  {
    DAMAGE: DAMAGE + [attack TH plus B] d D
  }
  result: DAMAGE
}


loop AC over {8..20} {
output [multiattack 2 times] named "Multiattack vs AC:[AC]"
output [automultiattack 2 times] named "Auto-Multiattack vs AC:[AC]"
output [ssmultiattack 2 times] named "Sharpshooter Multiattack vs AC:[AC]"
output [ssautomultiattack 2 times] named "Auto Sharpshooter Multiattack vs AC:[AC]"
}

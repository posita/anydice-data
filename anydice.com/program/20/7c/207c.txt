function: DIE:n vs AC:n {
  result: DIE != 1 & (DIE >= AC | DIE = 20)
}

function: DIE:n crits TH:n vs AC:n conf CONFDIE:n {
  result: [DIE vs AC] & DIE >= TH & (CONFDIE >= AC | CONFDIE = 20)
}

loop AC over {1..21}{
  output [d20 crits 19 vs AC conf d20] named "full confirmation vs [AC] AC"
  output [d20 vs AC] named "vs [AC] AC"
}

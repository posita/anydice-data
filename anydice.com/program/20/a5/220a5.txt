function: what W:n how H:n{
  if W < 0{
    result: [lowest 2 of (2-W)d6] + H
  }
  else{
    result: [highest 2 of (2+W)d6] + H
  }
}

loop AW over {-1..3}{
  loop AH over {-1..3}{
    loop BW over {-1..3}{
      loop BH over {-1..3}{
        output [what AW how AH] >= [what BW how BH] named "([AW],[AH]) vs ([BW],[BH])"
}}}}

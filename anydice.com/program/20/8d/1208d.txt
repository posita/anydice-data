function: A:n B:n C:n conditional {

if B<=A & A<=C {X:A}
if C<=A & A<=B {X:A}

if A<=B & B<=C {X:B}
if C<=B & B<=A {X:B}

if B<=C & C<=A {X:C}
if A<=C & C<=B {X:C}

if X - MOD = 1 {result: 0}

if X - MOD = 20 {result: 3}

if X >= AC {result: 2}

else { result: 1 }
}

AC: 15
MOD: 2
output [d20 d20 d20 conditional]

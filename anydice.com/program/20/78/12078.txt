function: attack A:n plus B:n vs D:n{
 if A = 1 {result:0}
 if A = 20 {result:2}
 if A+B>=D {result:1}
 result:0
}

function: damage for A:n with DN:d damage dice and DC:d crit dice plus DB:n{
 if A = 1 {result:DN+DB}
 if A = 2 {result:DN+DC+DB}
 result:0
}

loop DC over {10..25}{
 output 3d[damage for [attack d20 plus 10 vs DC] with 1d6 damage dice and 1d6 crit dice plus 4] named "Base Damage vs DC[DC]"
 output 4d[damage for [attack d20 plus 10 vs DC] with 1d6 damage dice and 1d6 crit dice plus 4] 
       \+[damage for [attack 21-(1@2d20) plus 10 vs DC] with 1d6 damage dice and 1d6 crit dice plus 4]\ named "TWF Disadvantage Damage vs DC[DC]"
 output 3d[damage for [attack d20 plus 10 vs DC] with 2d6 damage dice and 2d6 crit dice plus 4] named "TWF Add Dice Damage vs DC[DC]"
}

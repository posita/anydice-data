CRITICALMISS: 1
MISSDAMAGE: 0
HIT: 2
CRITICALHIT: 20
MODIFIER:     4
GREATSWORD: 2d6
GREATAXE:   d12
CRITICAL: 2

function: average for ATTACKRESULT:n {
if ATTACKRESULT = CRITICALHIT { result: HITDAMAGE + HITDAMAGE + MODIFIER }
if ATTACKRESULT = CRITICALMISS { result: MISSDAMAGE }
result: HITDAMAGE + MODIFIER
}

function: nocrit for ROLL:n {
 if ROLL = 0 { result: MISSDAMAGE }
result: HITDAMAGE + MODIFIER
}

HITDAMAGE: GREATSWORD
output [average for d20] named "Great Sword with Crit"

HITDAMAGE: GREATAXE
output [average for d20] named "Great Axe with Crit"

HITDAMAGE: GREATAXE
output [nocrit for d20] named "Greataxe no Crit"

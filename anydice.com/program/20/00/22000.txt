
function: get die SIZE:n {
 FAILS: 3              \ numbers from 1 to this number result in a failure \
 CRITS: {6,20}<=SIZE   \ these numbers result in a full success \
 result: d{0:FAILS,    \ 0’s represent failure \
 1:SIZE-FAILS-CRITS,   \ 1’s represent mixed success \
 2:CRITS}              \ 2’s represent full success \
}

DICE: {6,6,8,10,12,20} \ dice sizes gained for each additional dot \
POOL: [get die 4]      \ start the pool with a d4 \
loop N over {0..6} {
 POOL: [highest of POOL and [get die N@DICE]]
 output POOL named "[N] dots"
}

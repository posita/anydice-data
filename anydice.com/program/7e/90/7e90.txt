SKILL: 10
MODIFIER: 0
DAMAGE: 1
PASS: 1
FAIL: 0
CRITPASS: 2
CRITFAIL: -1
ZERO: 0

function: skillcheck ROLL:n vs SKILL:n{
if ROLL <= SKILL{
if ROLL <= SKILL-10{
if ROLL <= 6{result: CRITPASS}
}
if ROLL <= 4{result: CRITPASS}
result: PASS
}
if ROLL >= SKILL + 10{
result: CRITFAIL
}
if SKILL < 16{
if ROLL > 16{result: CRITFAIL}
}
result: FAIL
}

function: damage for SKILLCHECK:n{
if SKILLCHECK = PASS{
result: DAMAGE
}
result: ZERO
}

output [skillcheck 3d6-MODIFIER vs SKILL]
output [damage for [skillcheck 3d6-MODIFIER vs SKILL]]

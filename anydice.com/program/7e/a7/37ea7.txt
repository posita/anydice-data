set "explode depth" to 3

NA: 1
N: 2
NB: 3
YB: 5
Y: 7
YA: 15

function: roll D:n
{if D < N {result: -2}
else if N <= D & D < NB {result: -1}
else if NB <= D & D < YB {result: 0}
else if YB <= D & D < Y {result: 1}
else if Y <= D & D < YA {result: 2}
else if YA <= D {result: 3}
}

output [roll [explode d4]] named "d4"
output [highest of [roll [explode d4]] and [roll [explode d4]]] named "d4 adv"
output [lowest of [roll [explode d4]] and [roll [explode d4]]] named "d4 dis"

function: success A:n vs TN:n with PENALTY:n {
  if A + PENALTY >= TN {
    result: 1
  } else {
    result: 0
  }
}

function: count successes in DICE:s vs TN:n with PENALTY:n {
  COUNT: 0
  loop I over DICE {
    COUNT: COUNT + [success I vs TN with PENALTY]
  }
  result: COUNT
}

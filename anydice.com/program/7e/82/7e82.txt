\ D6 System rolls with wild die \

function: wild N:n K:n D:d M:n {
  if N = 1 {
    if K < 1 { result: M }
    result: [lowest K of D] + M
  }
  if N = 6 { result: 6 + D + M + [explode d6] }
  result: N + D + M
}
function: roll N:n M:n {
 result: [wild 1d6 (N-2) (N-1)d6 M]
}

output [roll 1 0] named "1D"
output [roll 2 0] named "2D"
output [roll 3 0] named "3D"
output [roll 4 0] named "4D"

\ Attack Sequence \

\ Hit Values \
A: 1 	\ # Attacks \
H: 3 	\ Hit Roll Target \
HR: 0   \ Hit Re-Roll \
W: 3 	\ Wound Roll Target \
WR: 0   \ Wound Re-Roll \
R: 0 	\ Rend (Negative Number) \
D: 1 	\ Damage \
F: 7	\ Feel No Pain Roll Target \
MWHT: 6 \ Mortal Wound on Hit Target \
MWHD: 2 \ Mortal Wound on Hit Damage \


\ Die Values \
HD: 1d6 \ Hit Die \
HRD: 1d6\ Hit RR Die\
WD: 1d6 \ Wound Die \
WRD: 1d6\ Wound RR Die \
SD: 1d6 \ Save Die\
FD: 1d6 \ Feel No Pain Die \


function: hit TO_HIT:s HREROLL:s {
	if TO_HIT <= HR {result: HREROLL >= H} 
        result: TO_HIT >= H
}

function: wound TO_WOUND:s WREROLL:s {
	if TO_WOUND <= WR {result: WREROLL >= W} 
        result: TO_WOUND >= W
}

function: save TO_SAVE:s {
	result: TO_SAVE < S-R
}

function: gate HR:s WR:s SR:s D:s {
	result: HR * WR * SR * D
}

function: fnp DICE:s{
	if DICE < F {result: 1}
	result: 0
}

function: mwonhit TO_HIT:s HREROLL:s {
	if TO_HIT <= HR {result: (HREROLL >= MWHT) * MWHD} 
        result: (TO_HIT >= MWHT) * MWHD
}

loop S over {2..7}{

output Ad([gate [hit HD HRD] [wound WD WRD] [save SD] D]d[fnp  FD] + d[mwonhit HD HRD]) named "[A] Attacks, [H]+ To Hit w/ Re-Roll at <=[HR], [W]+ To Wound w/ Re-Roll at <=[WR], [S]+ Save w/ Rend [R], [D] Damage, [F]+ FNP"

}

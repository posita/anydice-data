AC: 12
PROFBONUS: 4
MAGICAL: 0
ABMOD: 5 + MAGICAL
DICENUM: 2
DICESIZE: 6
BARB: 0
VANTAGE: 0
ATTACK: ABMOD + PROFBONUS
MISS: 0
HIT: 1
CRITICALHIT: 2
HITDAMAGE: DICENUMdDICESIZE + ABMOD
MISSDAMAGE: 0
CRITDAMAGE: (2*DICENUM + BARB)dDICESIZE + ABMOD



function: attack TWENTYRESULT:n vs DEFENSE:n { 
if TWENTYRESULT = 1{result: MISS}
if TWENTYRESULT = 20{
result: CRITICALHIT
}
if TWENTYRESULT + ATTACK >= DEFENSE{result: HIT}
result: MISS
}

function: damage for ATTACKRESULT:n{
if ATTACKRESULT = HIT{
result: HITDAMAGE
}
if ATTACKRESULT = CRITICALHIT{
result: CRITDAMAGE
}
result: MISSDAMAGE
}

output [damage for [attack d20 vs AC]]

function: evaluate ROLL:s {
 SUCCESSES: ROLL >=4
 if SUCCESSES > 0 { result: SUCCESSES + (ROLL = 6)}
 result: 0
}

loop DICE over {1..10} {
 output [evaluate DICE d6] named "[DICE]d"
}
output 5d{0,0,1,1,1,2} named "Super Spark"
output 5d{0,0,0,1,1,2} named "Spark"
output 5d{0,0,0,1,1,1} named "Normal"

AC: 12
PROFBONUS: 4
MAGICAL: 0
ABMOD: 5 + MAGICAL
DICENUM: 2
DICESIZE: 6
BARB: 0
DICE: 2
VANTAGE: 1
ATTACK: ABMOD + PROFBONUS
MISS: 0
HIT: 1
CRITICALHIT: 2
HITDAMAGE: DICENUMdDICESIZE + ABMOD
MISSDAMAGE: 0
CRITDAMAGE: (2*DICENUM + BARB)dDICESIZE + ABMOD

function: attack ROLL:n vs DEFENSE:n { 
if VANTAGE = 1{
result: VANTAGE@{DICEd20}
}else if VANTAGE = 2{
result: VANTAGE@{DICEd20}
}

if ROLL = 1{result: MISS}
if ROLL = 20{
if ROLL + ATTACK >= DEFENSE{result: CRITICALHIT}
result: HIT
}
if ROLL + ATTACK >= DEFENSE{result: HIT}
result: MISS
}

function: damage for ATTACKRESULT:n{
if ATTACKRESULT = HIT{
result: HITDAMAGE
}
if ATTACKRESULT = CRITICALHIT{
result: CRITDAMAGE
}
result: MISSDAMAGE
}

output [damage for [attack DICEd20 vs AC]]

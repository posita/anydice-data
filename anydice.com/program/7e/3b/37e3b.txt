function: dist A:n {
    NEG: A-1
    POS: [absolute A-20]
    if NEG>POS {result: NEG}
    else {result: POS}
}

function: emphasis A:n B:n {
    DA: [dist A]
    DB: [dist B]
    
    if DA > DB { result: A }
    result: B
}

function: disregard A:n B:n {
    DA: [dist A]
    DB: [dist B]
    
    if DA < DB { result: A }
    result: B
}

output 1d20 named "Straight"
output 1@2d20 named "Advantage"
output 2@2d20 named "Disadvantage"
output [emphasis 1d20 1d20] named "Emphasis"
output [disregard 1d20 1d20] named "Disregard"

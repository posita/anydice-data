HIT: 1
CRIT: 2
RANGE: 19
MISS: 0

ATTACK: {31, 31, 26, 26, 21, 21, 16, 31}

function: roll A:n {
 result: A
}

function: attack R:n plus A:n vs B {
  if (R >= RANGE) {
    result: CRIT
  }
  if R+A >= B {
    result: HIT
  }
  result: MISS
}

loop A over ATTACK {
  output [attack d20 plus A vs 33]
}

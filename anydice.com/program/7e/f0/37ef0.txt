function: roll DA:n and DB:n {
 X: DA - DB
 if DA = 1 {result: -2}
 else if X < 0 {result: -1}
 else if X >= 0 & X < S {result: 0}
 else if X >= S & X < S*2 {result: 1}
 else if X >= S*2 {result: 2}
}

function: attack ATK:n vs DEF:n {
 X: ATK - DEF
 if X > S*NUM-1 {result: NUM}
 if X <= 0 {result: 0}
 if X > 0 & X < S {result: 1}
 if X >= S & X < S*2 {result: 2}
 if X >= S*2 & X < S*3 {result: 3}
 if X >= S*3 & X < S*4 {result: 4}
 if X >= S*4 {result: 5}
 
}

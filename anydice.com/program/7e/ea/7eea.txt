function: reroll R:n under N:n this D {
   if R < N { result: D } else {result: R}
}

function: hit H:n mod M crit C dmg D bonus B ac A {
   if H >= C {result: D+D+B}
   else if H+M > A {result: D+B} 
   else {result: 0}
}

GWA: [reroll 1d12 under 3 this d12]
GWS: [reroll 1d6 under 3 this d6]
DA: [highest of 1d20 and 1d20]

AXEA: [hit DA mod 9 crit 19 dmg GWA bonus 5 ac 16]
SWORD: [hit d20 mod 9 crit 19 dmg GWS+GWS bonus 5 ac 16]

output AXEA+AXEA named "Greataxe with Advantage"
output SWORD+SWORD named "Greatsword"

function: explode N:n on THRESHOLD:n{
 HITS: 0
 if N >= TN { HITS: 1 }
 if N >= THRESHOLD { HITS: HITS + [explode d6 on 7] }
 result: HITS
}

TN: 4

output 10d[explode d6 on 7] named "hits with [TN]"
output 10d[explode d6 on 6] named "hits A+ with [TN]"
output 10d[explode d6 on 5] named "hits A++ with [TN]"

function: ekr ROLL:s {
  WINS: 0
  \ check for at least one 4 \
  if ([ROLL contains 4]) {
    WINS: WINS + 1
  }
  \ check for at least one 6, and at least 2 results (including that 6) that are higher than 1 \
  if ([ROLL contains 6] & [count {2..6} in ROLL] >= 2) {
    WINS: WINS + 1
  }
  \ check that we rolled no more than a single 1 \
  if ([count {2..6} in ROLL] >= 3) {
    WINS: WINS + 1
  }
  result: WINS
}

output [ekr 3d6]

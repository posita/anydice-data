
CRIT_THRESHOLD: 20
TARGET_AC: 16
CRITICAL_DICE: 2
BASE_DICE: 1
ROLL: 1d20

function: R:n firstattack{
    if R >= CRIT_THRESHOLD {result: CRITICAL_DICE d DMG_A + DMG_MOD_A}
    if (R + HIT_MOD_A) >= TARGET_AC {result: BASE_DICE d DMG_A + DMG_MOD_A}
    if (R + HIT_MOD_A) < TARGET_AC {result: 0}
}


function: R:n secondattack{
    if R >= CRIT_THRESHOLD {result: CRITICAL_DICE d DMG_B + DMG_MOD_B}
    if (R + HIT_MOD_B) >= TARGET_AC {result: BASE_DICE d DMG_B + DMG_MOD_B}
    if (R + HIT_MOD_B) < TARGET_AC {result: 0}
}

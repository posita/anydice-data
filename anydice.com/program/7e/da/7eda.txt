TN:14
FOCUS:0

function: evaluate ROLL:s {
\ SUCCESSES: ROLL <= TN \
\ FSUCCESS: ROLL <= FOCUS \
\ if SUCCESSES + FSUCCESS > 0 { result: 1 } \
\ if [count 20 in ROLL] { result: -1 } \
 result: (ROLL <= TN) + (ROLL <= FOCUS)
}

loop DICE over {1..5} {
 loop FOCUS over {1..4} {
  loop TN over {8..15} {
 output [evaluate DICE d20] named "[DICE]d20: TN [TN], Focus [FOCUS]"
  }
 }
}

function: explode N:n {
 if N = 10 { result: N + [explode d10]}
 if N = 1 { result: N - [explode d10]}
 result: N
}
set "maximum function depth" to 2

function: dor amX dmY {
 result: 1d[explode d10] + X > 1d[explode d10]+Y | 1d[explode d10]+X>1d[explode d10]+Y
}
function: dand amX dmY {
 result: 1d[explode d10] + X > 1d[explode d10]+Y & 1d[explode d10]+X>1d[explode d10]+Y
}
output [dor am14 dm14]
output [dand am14 dm14]
output 1d[explode d10] + 14 > 1d[explode d10] + 14 - 4

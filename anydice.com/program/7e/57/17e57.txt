function: SEQUENCE:s contains NUMBER:n{
 result: (SEQUENCE = NUMBER) > 0
}

function: count VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: COUNT
}

output 4d[6d6 contains 1]
output 8d[6d6 contains 1]

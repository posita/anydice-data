function: roll D:n plus ATTR:n  {
    if D + ATTR >= 20{ result: 2 } 
    else if D + ATTR >= 14 { result: 1}
    else { result:0}
}

DICE: {4,6,8,10,12}

loop X over DICE {
    output [roll [highest 1 of 2d20] plus dX] named "Complete ADV"
}

loop X over DICE {
    output [roll d20 plus [highest 1 of 2dX]] named "Partial ADV"
}

loop X over DICE {
    output [roll d20 plus dX] named "Straight"
}

loop X over DICE {
    output [roll d20 plus [lowest 1 of 2dX]] named "Partial DADV"
}

loop X over DICE {
    output [roll [lowest 1 of 2d20] plus dX] named "Complete DADV"
}

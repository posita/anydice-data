MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL + ATTACK >= (DEFENSE + 10) { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 result: HITDAMAGE * ATTACKRESULT
}

ATTACK: 18
HITDAMAGE: 2d10 + 8
loop AC over {21..25} {
 output [damage for [attack d20 vs AC]] named "[AC]"
}

ATTACK: 15
HITDAMAGE: 2d8 + 4
loop MAP over {0, -5, -10} {
 output [damage for [attack (d20 + MAP) vs 16]] named "[MAP]"
}

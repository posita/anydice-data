\**** Damage calculator ****\
function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}
\**** Attack roll with archery and sneak attack damage ****\
function: sneakattackroll ROLL:n vs AC:n {

 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + ARCHERY >= AC { result: SNEAKDAMAGE }
 result: 0
}

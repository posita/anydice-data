THRESHOLDNOSUCCESS: 4
THRESHOLDMINORSUCCESS: 9
DIESIZES: {4,6,8,10,12}
SKILLSIZES: {0..5}

function: compute success of N:n {
 if (N <= THRESHOLDNOSUCCESS) {
   result: 0
 } else if (N <= THRESHOLDMINORSUCCESS) {
   result: 1
 } else {
   result: 2
 }
}

loop FIRSTDIE over DIESIZES {
  loop SECONDDIE over DIESIZES {
    loop DARKDIE over DIESIZES {
      loop SKILL over SKILLSIZES {
         output [compute success of 1dFIRSTDIE+1dSECONDDIE-1dDARKDIE+SKILL] named "1d[FIRSTDIE]+1d[SECONDDIE]-1d[DARKDIE]+[SKILL]"
      }
    }
  }
}

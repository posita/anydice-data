BASE:{11,5,6}
LIGHT:{4,-1,0} MEDIUM:{2,2,0} HEAVY:{1,4,0}


DIE:{0,0,0,0,0,0,1,1,1,2}

function: WA:s vs WB:s {
          DEF: 3@WB + 3@BASE
          ATK: (1@WA + 1@BASE)dDIE
          RES: ATK - DEF
          result: [damroll RES]
          }

function: damroll N:n {
          if N >= 0 {result: (2@ WA + 2@BASE + N)dDIE}
          result: 0
          }

output [LIGHT vs LIGHT] named "light vs light"
output [LIGHT vs MEDIUM] named "light vs medium"
output [LIGHT vs HEAVY] named "light vs heavy"
output [MEDIUM vs LIGHT] named "medium vs light"
output [MEDIUM vs MEDIUM] named "medium vs medium"

loop AC over {13..23}{
loop GWM over {0,5}{
loop ST over {4,5}{
PRF:4
RAGE:2
ATMOD: ST+PRF-GWM
DMGMOD:ST+RAGE+(GWM*2)
CRTMLT:3+(5-ST)
function: switch S:n {
 if S=1 {result:-10}
 if S=20 {result: CRTMLTd12+DMGMOD*CRTMLT}
 if S+ATMOD < AC {result: 0}
 if S < 19 { result: d12+DMGMOD}
 result: CRTMLTd12+(DMGMOD*CRTMLT)
}

output [switch d20] named "[AC]_attack_GWM[GWM]"
output [highest 1 of 2d[switch d20]] named "[AC]_advantage_GWM[GWM]"
}
}
}

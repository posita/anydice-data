function: crit CRITS:s {
D:0
if CRITS >= 6 {D:D+1}
result: CRITS + 1d4*D
}

function: do FIRST:s and SEC:s and THIRD:s for john {
C:0
if FIRST >= 6 {C:C+1}
if SEC >= 6 {C:C+1}
if THIRD >= 6 {C:C+1}
result: FIRST + SEC + THIRD + [crit 1d(C*4)]
}

output [do 1d6 and 1d6 and 1d6 for john] named "3d6"
output [do 1d8 and 1d8 and 0 for john] named "2d8"

ATT: 20
SKILL: 3
MOD: 0
TOUGH: 3
WMOD: 0

function: ATTACK:n attacks {
 HIT: 0 
 loop N over {1.. ATTACK}{
   HIT: HIT + [hitroll d6]
 }
 result: HIT
}

function: hitroll ROLL:s{
 X: 0
 if (ROLL = 6){
  X: X + SUSTAIN
  if (LETHAL = 1){
   X: X + 10000
   result: X
  }
 }
 if (ROLL + MOD >= SKILL){
  X: X+1
 }
 result: X
}

function: HITS:n wounds {
 LETHALHITS: 0
 if (HITS >= 10000){
  LETHALHITS: HITS / 10000
  HITS: HITS - (LETHALHITS * 10000)
 }
 WOUND: 0 
 loop N over {1.. HITS}{
   WOUND: WOUND + [woundroll d6]
 }
 result: WOUND + LETHALHITS
}

function: woundroll ROLL:s{
 X: 0
 if (ROLL + WMOD >= TOUGH){
  X: X+1
 }
 result: X
}

SUSTAIN: 0
LETHAL: 0

output [[ATT attacks]wounds] named "Regular Attacks"

SUSTAIN: 0
LETHAL: 1

output [[ATT attacks]wounds] named "Lethal Hits"

SUSTAIN: 1
LETHAL: 0

output [[ATT attacks]wounds] named "Sustained Hits"

SUSTAIN: 1
LETHAL: 1

output [[ATT attacks]wounds] named "Lethal Sustained Hits"

\ Functions \

function: skill of ROLL:d {
  result: [highest 1 of ROLL]
} 

function: base LEVEL:n {
  if LEVEL >= 4 {result: 2}
  else {result:1}
}

function: crit LEVEL:n {
  if LEVEL >= 4 {result: 3}
  else {result:2}
}

function: damage of ROLL:n with LEVEL:n level and HARMONY:n harmony {
  CRITS:  ([count 2 in ROLL] * [crit LEVEL]) + ([count 2 in ROLL] * HARMONY)
  BASE: [count 1 in ROLL] * [base LEVEL]
  result: CRITS+BASE
} 

function: wither of ROLL:n with LEVEL:n level and RIVAL:n rivals {
  BASE: ([count 1 in ROLL] * [base LEVEL]) + ([count 1 in ROLL] * RIVAL * 2)
  result: BASE
}

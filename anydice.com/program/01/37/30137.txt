function: weaponone ROLL:n vs DEFENSE:n {
 if ROLL >= DEFENSE { result: (WEAPONONEDAM+DAMAGEMOD) * WEAPONONEMULTIPLIER }
 result: (WEAPONONEDAM+DAMAGEMOD)
}

function: weapontwo ROLL:n vs DEFENSE:n {
 if ROLL >= DEFENSE { result: (WEAPONTWODAM+DAMAGEMOD) * WEAPONTWOMULTIPLIER }
 result: (WEAPONTWODAM+DAMAGEMOD)
}

DAMAGEMOD:2

WEAPONONEDAM:d8
WEAPONONECRIT:d8+1
WEAPONONEMULTIPLIER:3

WEAPONTWODAM:3d4
WEAPONTWOCRIT:1d10
WEAPONTWOMULTIPLIER:3

loop TTN over {6..8} {
output [weaponone WEAPONONECRIT vs TTN] named "First weapon Target Number [TTN]"
output [weapontwo WEAPONTWOCRIT vs TTN] named "Second Weapon Target Number [TTN]"
}

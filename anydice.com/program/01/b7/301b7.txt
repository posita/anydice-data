AC: 13
H: 7
POISON: ((1d20 +2 + 4 - AC)>0)*1d12
THUNDER: ((1d20 +2 + 4 - AC)>0)*3d8
DAGGER: ((1d20 + 2 - AC)>0)*(4+1d4)+((1d20 + 2 - AC)>0)*(1d4)
DEPTH: 7

function: turns HP:n TURNS:n ATTACK COST:n{
  if (HP<0) {
    result: TURNS
  }
  if (TURNS>DEPTH) {
    result: TURNS+HP*COST
  }

  result: [turns HP-ATTACK TURNS+COST ATTACK COST]


  
}

output [turns H 0 POISON 2]
output [turns H 0 THUNDER 3]
output [turns H 0 DAGGER 2]

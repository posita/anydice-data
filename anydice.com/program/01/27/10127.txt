 result: d4 + 1

function: half on save rolling ROLL:n against AC:n ac doing DAMAGE:d damage
{
  DIVISION: 1
  if ROLL >= AC { 
    DIVISION: 2
  }
  result: DAMAGE / DIVISION
}

loop LVL over {2..4}{ output (LVL-1)d(d4 + 1) named "Magic Missle [LVL]" }
output 2d10 named "Fire Bolt"

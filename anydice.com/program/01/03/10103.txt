\**** Damage calculator ****\

\**** Great Weapon Fighting Plain Calculator ****\
function: sixgwfight ROLLX:n {
 if ROLLX <= 2 { result:d6 }
 result: ROLLX
}

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 4
DMGBONUS: 1d4
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "5th BH Dueling & Rite + Extra Att"
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "5th BH Dueling & Rite + Extra Att + ADV"

LVL: 1
ATK: 1+LVL*1/2
DEF: 1+LVL*1/2
MGK: 1+LVL*1/2
HP: 20+LVL*(2+DEF)
MP: 10+(MGK)+(LVL)

output ATK named "ATK"
output DEF named "DEF"
output MGK named "MGK"
output HP named "HP"
output MP named "MP"

TOHIT: (2d10 + ATK)
TODODGE: (1d10 + DEF) 
DAMAGE: [highest of ATK*2d6-(DEF*3)and 1]
CRITDAMAGE: ATK*2*2d6-(DEF*2)

output TOHIT > TODODGE named "To-hit"
output TOHIT - TODODGE > 10 named "To-crit"
output DAMAGE named "Damage"
output [highest of CRITDAMAGE and 1] named "Crit Damage"
output (HP/DAMAGE) named "Time to kill"
output (HP/CRITDAMAGE) named "Time to kill critical"

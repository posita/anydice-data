MAIL: {0,0,4,4,4,4}
PLATE: {3,5,5,5,5,5}
ARMOR: MAIL

function: thrustraw ROLL:s {
  HIGH: {1@ROLL}
  LOW: {#ROLL@ROLL}
  result: [damage (HIGH - [armor cover LOW])]
}
function: damage N:n {
  if N < 0 {
    result: 0
  } else {
    result: N
  }
}
function: normal {
  result: [damage (1d6 - [armor cover 1d6])]
}
function: armor cover AD:n {
  result: AD@ARMOR
}
function: precise {
  result: [damage (1d6 - [armor cover [lowest 1 of 2d6]])]
}
function: thrust {
  ROLL: 2d6
  DAMAGE: [thrustraw ROLL]
  result: DAMAGE
}
output [armor cover 1d6] named "Armor"
output [normal] named "Normal"
output [precise] named "Precise"
output [thrust] named "Driving Thrust"

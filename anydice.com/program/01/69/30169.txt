function: A:n boon B:n bane {
X:8
Y:8
Z:8
if A>0{
  loop N over {1..A} {
      if X<12 {X:X+2}
      else if Y<12 {Y:Y+2}
      else if Z<12 {Z:Z+2}
    }
  }

if B>0{
  loop N over {1..B} {
      if Z>4 {Z:Z-2}
      else if Y>4 {Y:Y-2}
      else if X>4 {X:X-2}
    }
  }

result: 1dX+1dY+1dZ
}

output [0 boon 0 bane]
output [1 boon 1 bane]

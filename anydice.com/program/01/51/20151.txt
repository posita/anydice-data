\**** Damage calculator for GWM normal Vs advantage ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 14-18\
ENEMYAC: d{14..18}

ATTACKMOD: 5
DMGBONUS: 10
PROFICIENCY: 5
ACCBONUS: -5
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 4d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 8d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "GWM Normal"

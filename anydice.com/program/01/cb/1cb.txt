function: match M:n against S:s{
 \ calculate how many evens were rolled on the dS pool \
 SE: (S = 2) + (S = 4) + (S = 6)
 
 \ calculate how many evens were rolled on the single dM \
 ME: M = {2, 4, 6}
 
 \ finish if the dM didn't roll even \
 if ME = 0 { result: SE + 1 }
 
 \ dM rolled even, add how many times dS match the dM \
 result: SE + 2 + (S = M)
}

output [match d6 against 2d6] named "2dS + dM (+3 shift)"
output [match d6 against 3d6] named "3dS + dM (+4 shift)"
output [match d6 against 4d6] named "4dS + dM (+5 shift)"
output [match d6 against 5d6] named "5dS + dM (+6 shift)"

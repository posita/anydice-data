function: success LEVEL:n {
  if LEVEL >= 4 {S: 2}
  else {S:1}
  result: S
}

function: critical LEVEL:n HARMONY:n {
if LEVEL >= 4 {C: 3 + (HARMONY * 2)}
else {C:2 + (HARMONY * 2)}
result: C
}

  result: {[critical LEVEL HARMONY],0,0,0,0,0}
}

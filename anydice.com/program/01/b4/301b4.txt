function: if CONDITION:n then A else B {
  if CONDITION { result: A } else { result: B }
}

function: danger roll T:n of size S:n {
  TOTAL: 0
  ONES: 0
  loop N over {1..T} {
    OUTCOME: 1dS
    ONES: [if OUTCOME = 1 then ONES + 1 else ONES]
    TOTAL: TOTAL + OUTCOME
  }

  result: [if ONES > 0 then 1 else TOTAL]
}

output [danger roll 4 of size 6]

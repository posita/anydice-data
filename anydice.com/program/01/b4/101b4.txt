function: rerollone A:n B:n under N:n {
    \Only reroll if less than N\
    if A + B < 19-7+1 {
        \Find the lowest dice\
             if A < B  { result: B + 1d8 }
        else           { result: A + 1d8 }
    }
    else { result: A + B }
}

function: rerollall R:n under N:n {
   \Only reroll if less than N\
   if R < 12 { result: 2d8 }
   else { result: R }
}

function: A:n greater B:n {
result: A>B
}
function: easy SUR sur INV inv at level N {
result: 20 * N * [2d20 + SUR + INV greater 2d10 + 2] - 15
}
function: medium SUR sur INV inv at level N {
result: 50 * N * [2d20 + SUR + INV greater 4d10 + 4] - 30
}
function: hard SUR sur INV inv at level N {
result: 20 * N * [2d20 + SUR + INV greater 6d10 + 6] - 45
}
output [easy 10 sur 5 inv at level 7]
output [medium 10 sur 5 inv at level 7]
output [hard 10 sur 5 inv at level 7]

D:{0:6,1,1,2,2}

function: roll R:s one fewer success {
  S:{1,2}@R-1
  if S<0 { result:0 } else { result:S }
}

loop N over {5,4,3} {
  output {N,N-1}@NdD named "lowest 2 of [N]d10"
  \output [roll NdD one fewer success] named "lowest 2 of [N]d10 one fewer success"\
}

output 2@2dD+1dD named "2d10 reroll highest"
output 2dD named "2d10"
output 1@2dD+1dD named "2d10 reroll lowest"
\output [roll 2dD one fewer success] named "2d10 one fewer success"\\die roll has one fewer successes\
\output 1@2dD named "highest 1 of 2d10"\\one fewer die can generate successes\

loop N over {3..5} {
  output {1,2}@NdD named "highest 2 of [N]d10"

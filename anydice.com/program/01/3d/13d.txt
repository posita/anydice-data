function: dig NUM{
 TEMP: 0
 loop I over {0..(NUM -1)}{
  TEMP: TEMP + ((1d10-1) * (10^I))
 }
 result: TEMP
}

function: sum NUM{
 result: NUMd10
}

function: comp SUM:n{
 if SUM > 0 {
  result: 1
 } else if SUM < 0 {
  result: -1
 } else {
  result: 0
 }
}

loop I over {1..3}{
 output [comp [dig I] - [dig 2]] named "Dig [I] vs 2"
}
loop I over {1..3}{
 output [comp [sum I] - [sum 2]] named "Sum [I] vs 2"
}

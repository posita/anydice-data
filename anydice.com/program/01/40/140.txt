\ Only counts rolls of up to five degrees of success; hopefully enough \
function: ROLL:s count TARGET:n degree DSEP:n {
  SONE: (ROLL >= TARGET)
  STWO: (ROLL >= TARGET+DSEP)
  STHREE: (ROLL >= TARGET+2*DSEP)
  SFOUR: (ROLL >= TARGET+3*DSEP)
  SFIVE: (ROLL >= TARGET+4*DSEP)
  result: SONE + STWO + STHREE + SFOUR + SFIVE
}

\ TS is threshold - skill \
TS: 7
\ DS is the increment between degrees of success \
DS: 3
\ MAXPOOL is the maximum dice pool size \
MAXPOOL: 5

loop N over {1..MAXPOOL} {
  output [Nd10 count TS degree DS] named "Dice pool [N], threshold - skill [TS]"
}

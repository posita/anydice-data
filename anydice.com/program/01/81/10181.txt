function: rerollone A:n B:n under N:n {
    \Only reroll if less than N\
    if A + B < N {
        \Find the lowest dice\
             if A < B { result: B +  1d6 }
        else                  { result: A + 1d6 }
    }
    else { result: A + B }
}


output [rerollone 1d6 1d6 under 7] named "Reroll lowest of 2D6 if under 7"

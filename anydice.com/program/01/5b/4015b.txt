LVL: 3
ATK: 1+LVL*1/2
DEF: 1+LVL*1/2
MGK: 1+LVL*1/2
HP: 20+LVL*(2+DEF)
MP: 10+LVL/4*(MGK)

output ATK named "ATK"
output DEF named "DEF"
output MGK named "MGK"
output HP named "HP"
output MP named "MP"

TOHIT: (2d10 + ATK)
TODODGE: (1d10 + DEF)
DAMAGE: ATK*2d6-(DEF*3)
OGDAMAGE: ATK*4-DEF*2
CRITDAMAGE: ATK*4d6-(DEF*3)

output TOHIT > TODODGE
output TOHIT - TODODGE > 10
output DAMAGE
output OGDAMAGE
output CRITDAMAGE
output (HP/DAMAGE)+1
output (HP/CRITDAMAGE)+1

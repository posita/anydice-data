MISS: 0
HIT: 1
CRITICALHIT: 2
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  result: CRITICALHIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}
function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: HITDAMAGE*2
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}


function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{12..22}

ATTACK: 2 + 4 + 1
HITDAMAGE: d10 + 4
MISSDAMAGE: 0
output [attack] named "norm"

ATTACK: 2 + 4 + 1 -5 
HITDAMAGE: d10 + 4 +10
MISSDAMAGE: 0
output [attack] named "feat"

function: action roll ADICE:n AMOD:n CROLLS:n {
  if ADICE + AMOD < {1,2}@CROLLS {result: 0}
  result: [success check 1d6 X 2d10]
}

function: success check ADICE:n AMOD:n CROLLS:n {
  if ADICE + AMOD > 1@CROLLS {result: 2}
  result: 1
}

loop X over {1..3}{
 output [action roll 1d6 X 2d10] named "1d6+[X]"
}

DICEDAMAGE: 0
FLATDAMAGE: 1
HITBONUS: 14
AC: 22
ATTACKS: 6
OPROHB: 0
CRIT: 20
CRITB: 0

function: attack B:n HIT:n DD:n FD:n {
   if HIT >= CRIT {
      \on crit\
      result: DD + FD + DD + CRITB
   }
   if HIT + B >= 9 {
      \on hit\
      result: DD + FD
   } else {
      \on miss\
      result: 0
   }
}

function: attackac B:n AC:n HIT:n DD:n FD:n {
   result: [attack B-AC+9 HIT DD FD]
}

function: oproh OHB:n ATK:n {
   if ATK = 0 {
      result: 0
   }
   result: ATK + OHB * (1 + (1d20>=CRIT))
}

function: atk {
   result: [attack HITBONUS 1d20 DICEDAMAGE FLATDAMAGE]
}

function: atkh H:n {
   result: [attack HITBONUS+H 1d20 DICEDAMAGE FLATDAMAGE]
}

function: atkd D:d {
   result: [attack HITBONUS 1d20 DICEDAMAGE+D FLATDAMAGE]
}

function: adv {
   result: [highest of 1d20 and 1d20]
}

function: eadv {
   result: [highest of 1d20 and [highest of 1d20 and 1d20]]
}

function: aatk {
   result: [attack HITBONUS [adv] DICEDAMAGE FLATDAMAGE]
}


function: eatk {
   result: [attack HITBONUS [eadv] DICEDAMAGE FLATDAMAGE]
}

function: atkac {
   result: [attackac HITBONUS AC 1d20 DICEDAMAGE FLATDAMAGE]
}

function: aatkac {
   result: [attackac HITBONUS AC [adv] DICEDAMAGE FLATDAMAGE]
}

function: eatkac {
   result: [attackac HITBONUS AC [eadv] DICEDAMAGE FLATDAMAGE]
}

function: resolveseq S:d

function: dimindam HITS:n DDS:s {
   CF: 1\(1 + (1d20>=CRIT))\
   loop N over {1..#DDS}{
      if HITS = N {result:N@DDS * CF}
   }
   result: 0
}

function: temp T:n {
  CF: (1 + (1d20>=CRIT))
  if T = 6 {result: 25d8 * CF}
  if T = 5 {result: 22d8 * CF}
  if T = 4 {result: 18d8 * CF}
  if T = 3 {result: 14d8 * CF}
  if T = 2 {result: 10d8 * CF}
  if T = 1 {result: 5d8 * CF}
  if T = 0 {result: 0}
}

output [dimindam ATTACKSd[aatkac] {1,d6}]

\function: rr one D:n DD:d {
  if D = 1  {
    result: DD
  } 
  result: D
}
function: two one D:n {
  if D = 1  {
    result: 2
  } 
  result: D
}\

\function: rrlcorrection A:n B:n AT:d BT:d {
  if A < B {
    result: AT-A
  }
  result: BT-B
}

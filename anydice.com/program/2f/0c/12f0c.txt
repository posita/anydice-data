function: save ROLL:s {
if (ROLL = 6) > 0 {result: 1}
result: 0
}

function: saveenc ROLL:s {
if (ROLL = 6) > 1 {result: 1}
result: 0
}

function: check ROLL:s {
if (ROLL = 6) > (ROLL = 1) {result: 1}
result: 0
}

function: flog ROLL:s vs T:n {
if (1)@ROLL = 6 {result: 100}
if (1)@[reverse ROLL] = 1 & {1..10}@ROLL < T {result: 0}
if {1..10}@ROLL < T {result: 1}
result: 2
}

\output [flog 1d6 vs 10] named "Flog Dismal"
output [flog 2d6 vs 10] named "Flog Low"
output [flog 3d6 vs 10] named "Flog Typical"
output [flog 4d6 vs 10] named "Flog High"
output [flog 5d6 vs 10] named "Flog Peak"\



output [check 1d6]
output [check 2d6]
output [check 3d6]
output [check 4d6]
output [check 5d6]

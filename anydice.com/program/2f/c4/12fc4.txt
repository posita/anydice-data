function: shortsword ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: SHORTSWORDCRIT } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: SHORTSWORD }
 result: 0
}

function: shortswordtwf ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: SHORTSWORDCRIT-ATTACKMOD } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: SHORTSWORD-ATTACKMOD }
 result: 0
}

function: shortswordtwftwo ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: SHORTSWORDCRIT-ATTACKMOD } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS -2 >= AC { result: SHORTSWORD-ATTACKMOD }
 result: 0
}

function: greataxe ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: GREATAXECRIT } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: GREATAXE }
 result: 0
}

function: shortsworddamagedie ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: SHORTSWORDCRIT+2d6 } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: SHORTSWORD+1d6 }
 result: 0
}

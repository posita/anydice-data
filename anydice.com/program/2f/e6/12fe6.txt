TARGET_AC: {18}

SLOT: 1
CHA: 5
PROF: 3
DIE: 4
REROLL: 1

DIESQ: DIE * DIE
REROLLSQ: REROLL * REROLL
MODONE: REROLL + 1
MODTWO: DIESQ - REROLLSQ

DART: {{1..REROLL}:REROLL, {MODONE..DIE}:(MODTWO / (DIE - REROLL))}

output (1dDART + 1 + CHA) * (2 + SLOT)


ELDRITCH_BLAST_ATTACK: {CHA + PROF}

ELDRITCH_BLAST_ROLL: 1d10
ELDRITCH_BLAST_ROLL_CRIT: 2d10

ELDRITCH_BLAST_DAMAGE: ELDRITCH_BLAST_ROLL + CHA
ELDRITCH_BLAST_DAMAGE_CRIT: ELDRITCH_BLAST_ROLL_CRIT + CHA


MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{TARGET_AC}

ATTACK: ELDRITCH_BLAST_ATTACK
HITDAMAGE: ELDRITCH_BLAST_DAMAGE * 3
CRITICALDAMAGE: ELDRITCH_BLAST_DAMAGE_CRIT * 3
MISSDAMAGE: 0
output [attack] named "Eldritch Blast"

function: attack {
 result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]
}

ATTACK: ELDRITCH_BLAST_ATTACK
HITDAMAGE: ELDRITCH_BLAST_DAMAGE * 3
CRITICALDAMAGE: ELDRITCH_BLAST_DAMAGE_CRIT * 3
MISSDAMAGE: 0
output [attack] named "Adv Eldritch Blast"

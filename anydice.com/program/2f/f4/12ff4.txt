U: 1
D: 0

P: 3 + U + D
Z: Pd6
T: 0
R: 0

loop L over {0..U+D} {

  if T = 0 & U > 0 {
    set "position order" to "highest first"
    U: U - 1
    X: 3 + U + D
    R: {1..X}@Z
  }

  if T = 1 & D > 0 {
    set "position order" to "lowest first"
    D: D - 1
    X: 3 + U + D
    R: {1..X}@Z
  }

  if T = 0 {
    T: 1
  } else {
    T: 0
  }
}

output R

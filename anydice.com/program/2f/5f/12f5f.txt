TARGET_AC: {18}

DEX: 5 
PROF: 3
WEAPON_DIE_SIZE: 6
WEAPON_DIE_NUMBER: 1
MOD: 1
CRIT: 2

WEAPON_ATTACK: {DEX + PROF + MOD}
SHADOWBLADE_ATTACK: {DEX + PROF}


WEAPON_ROLL: {WEAPON_DIE_NUMBER}d{WEAPON_DIE_SIZE}
WEAPON_ROLL_CRIT: {CRIT * WEAPON_DIE_NUMBER}d{WEAPON_DIE_SIZE}
WEAPON_ROLL: 2d8
SHADOWBLADE_ROLL_CRIT: {CRIT * 2}d8


WEAPON_DAMAGE: WEAPON_ROLL + DEX + MOD
WEAPON_DAMAGE_CRIT: WEAPON_ROLL_CRIT + DEX + MOD
SHADOWBLADE_DAMAGE: SHADOWBLADE_ROLL + DEX
SHADOWBLADE_DAMAGE_CRIT: SHADOWBLADE_ROLL_CRIT + DEX


MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{TARGET_AC}

ATTACK: WEAPON_ATTACK
HITDAMAGE: WEAPON_DAMAGE
CRITICALDAMAGE: WEAPON_DAMAGE_CRIT
MISSDAMAGE: 0
output [attack] named "Weapon"

ATTACK: SHADOWBLADE_ATTACK
HITDAMAGE: SHADOWBLADE_DAMAGE
CRITICALDAMAGE: SHADOWBLADE_DAMAGE_CRIT
MISSDAMAGE: 0
output [attack] named "Shadowblade"

function: attack {
 result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]
}

ATTACK: WEAPON_ATTACK
HITDAMAGE: WEAPON_DAMAGE
CRITICALDAMAGE: WEAPON_DAMAGE_CRIT
output [attack] named "Adv Weapon"

ATTACK: SHADOWBLADE_ATTACK
HITDAMAGE: SHADOWBLADE_DAMAGE
CRITICALDAMAGE: SHADOWBLADE_DAMAGE_CRIT
output [attack] named "Adv Shadowblade"

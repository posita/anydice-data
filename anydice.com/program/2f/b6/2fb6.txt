function: pickcrit NUMBER:n of DICE:d {
 \loop CRIT over {1..#DICE} {
   if [{CRIT@DICE} contains 1] {result: CRIT@DICE} }\
 REV: #DICE - NUMBER + 1
 if NUMBER < 1 { REV: #DICE }
 if NUMBER > #DICE { REV: 1  }
 DICEFACE: #{1@DICE}
 result: [pickhelper DICE]
 
}

function: pick NUMBER:n of DICE:d {
 \loop CRIT over {1..#DICE} {
   if [{CRIT@DICE} contains 1] {result: CRIT@DICE} }\
 REV: #DICE - NUMBER + 1
 if NUMBER < 1 { REV: #DICE }
 if NUMBER > #DICE { REV: 1  }

 result: REV@DICE
 
}

function: pickhelper N:s {
  if N = 1 {
   if N = DICEFACE { result: REV@DICE }
  }
  if N = 1 {result: 1}
  if N = DICEFACE {result: DICEFACE}
  result: REV@DICE
}

function: compare ONE:d TWO:d {
  result: [comparehelp ONE TWO]
}

  if FIRST > SECOND {result: 1}
  if FIRST = SECOND {result: 2}

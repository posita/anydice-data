function: ace D:n {
 result: [tace dD on D]
}

function: tace X:n on D:n {
 if X = D {
  result: X + [tace dD on D]
 } else {
  result: X
 }
}

function: wild D:d {
 result: [highest of D and [ace 6]]
}

function: treroll D:d R:n T:n {
 if R < T {
  result: D
 } else {
  result: R
 }
}

function: reroll D:d with T:n {
 result: [treroll D D T]
}

output [wild [ace 8]] named "wild 8"
output [wild [ace 10]] named "wild 10"

function: myex N:n C:n T:n {
 Y: Nd10
 loop I over {1..N} {
  result: [convert I@Y C 10+T I]
 }
 result: 1@Y+T
}

function: dcross A:n cross B:n plus C:n {
 Y: Ad10
 loop N over {1..A} {
  result: [convert N@Y B 10+C N]
 }
 result: 1@Y+C
}

function: convert N:n C:n T:n I:n {
 if N >= C {result: [myex I C 10+T]}
 else {result: T+N}
}

function: boons NAME:n and SKILL:n and ASSET:n {
 POOL: {NAME, SKILL, ASSET}
 BOONS: [count {4..20} in POOL]
 BOONS: BOONS + [count {8..20} in POOL]
 BOONS: BOONS + [count {12..20} in POOL]
 OUT: 0
 if BOONS > 0 { OUT: BOONS }
 result: OUT
}

loop M over {0,6}{
 loop N over {4,8,12}{
  output [boons 1d6 and 1dN and 1dM] named "BOON 1d6/S[N]/A[M]"
 }
}

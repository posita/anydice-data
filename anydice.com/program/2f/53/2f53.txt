function: sqrthelper N:n guess R:n {
 if [absolute ((N/R)-R)] <= 1 
  {
   if (N/R) > R
   {
    result: R
   }
   result: (N/R)
  }
 result: [sqrthelper N guess [absolute (((N/R)+(R))/2)]]
}

function: sqrt of N:n {
 result: [sqrthelper N guess 2] 
}

function: sdie S:n success R:n reroll D:n T:n {
 if D >= R { result: 1 + [sdie S success R reroll dT T]}
 result: D >= S
}

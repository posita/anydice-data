ROLL:6
KEEP:5

DIE: d10
EXPLODEFROM: [maximum of DIE]
EXPLODER: [explode DIE]

function: explode highest N:n of ROLL:s {
  result: {1..N}@ROLL + [lowest of N and ROLL >= EXPLODEFROM] d EXPLODER
}

loop R over {1..10}{
 loop K over {1..R}{
   output [explode highest K of RdDIE] named "[R]k[K]"
  }
}

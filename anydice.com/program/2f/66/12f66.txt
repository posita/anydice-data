function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: 2dDMGDICE + ATTACKMOD + DMGBONUS } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DMGDICE + ATTACKMOD + DMGBONUS }
 result: 0
}

\helper functions for Great Weapon Fighting and rogue offhand attack\
function: reroll ROLL:n as DIE:d if less than MIN:n {
  if ROLL < MIN { result: DIE }
  else { result: ROLL }
}
function: gwf DIE:d {
  result: [reroll DIE as DIE if less than 3]  \GWF allows reroll on 1 or 2\
}
function: retry ATTACK:d on miss {
  result: [reroll ATTACK as ATTACK if less than 1]  \assume less than 1 damage => miss => retry\
}

\averaged against enemy AC from 15-20\
ENEMYAC: d{15..20}


ATTACKMOD: 5
DMGBONUS: 1
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
DMGDICE: 1d4 + 1d8 + 1d6 + 4d6
output [retry [attackroll [highest 1 of 2d20] vs ENEMYAC] on miss] named "Rogue"

ATTACKMOD: 5
DMGBONUS: 11
PROFICIENCY: 3
ACCBONUS: -5

function: ob T:n in A:d with reroll {
   result: [count T in A] + [ob T in (#A-[count T in A])d6 with reroll]
}

function: ob at least T:n in A:d with reroll {
  result: [count {T..6} in A] + [ob at least T in (#A-[count {T..6} in A])d6 with reroll]
}
output [ob 6 in 3d6 with reroll]
output [ob at least 5 in 3d6 with reroll]

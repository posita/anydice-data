AC:  16
DEX: 5
WIS: 4

\ Pre-combat Buffs \
OBLIVION: d4
BLESS: d4

\ Permanent buffs \
ENHANCEMENT: 3
ARCHERY: 2

\ Penalties \
PENALTY_SHARPSHOOTER: -5

\ Ranger \
HUNTERS_MARK : d6

\ Other \
BANE: 3d6
ACID: 2d6
DIVINE_FURY: 1d6
SHARPSHOOTER: 10
DREAD_AMBUSHER: d8
RAGE: 2

function: attack AC:n ac roll ROLL:n plus BONUS:n for DAMAGE:d and EXTRA_DAMAGE:d {
  TO_HIT: ROLL + BONUS

  if (ROLL = 1 | TO_HIT < AC) {
    result: 0
  } else if (ROLL = 20) {
    result: DAMAGE + DAMAGE + EXTRA_DAMAGE
  } else {
    result: DAMAGE + EXTRA_DAMAGE
  }
}

output
  [attack AC ac
    roll d20 plus DEX + ARCHERY + ENHANCEMENT
    for d8
    and DEX + ENHANCEMENT
  ] +
  [attack AC ac
    roll d20 plus DEX + ARCHERY + ENHANCEMENT
    for d8
    and DEX + ENHANCEMENT
  ]

function: seq X:n and Y:n {
  DICE: Xd6
  Z: Y - 1
  result: [count Y in DICE] + Z * [DICE contains Y]
}

function: model X:n {
  result: [highest of [seq X and 1] and [highest of [seq X and 2] and [highest of [seq X and 3] and [highest of [seq X and 4] and [highest of [seq X and 5] and [seq X and 6]]]]]]
}

function: seq ROLL:s and FACE:n {
  OFFSET: FACE - 1
  result: [count FACE in ROLL] + OFFSET * [ROLL contains FACE]
}

function: model DICE:d {
  result: [highest of [seq DICE and 1] and [
    highest of [seq DICE and 2] and [
      highest of [seq DICE and 3] and [
        highest of [seq DICE and 4] and [
          highest of [seq DICE and 5] and [seq DICE and 6]]]]]]
}

output [model 1d6]

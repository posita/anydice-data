D: 1
DARK: Dd6

function: dark or light L:s D:s{
  if(1@L > 1@D){
    result: 1
  }
  if(1@L < 1@D){
    result: -1
  }
  if(1@L=1@D){
    result:0
  }
}

function: power level L:s D:s{
  result: 1@{L,D}
}

loop L over {1..3}{
  LIGHT: Ld6
  loop D over {1..3}{
    DARK: Dd6

    output [dark or light LIGHT DARK] named "Dark or Light on [L] Light dice and [D] Dark dice"
    output [power level LIGHT DARK] named "Power on [L] Light dice and [D] Dark dice"
  }
}

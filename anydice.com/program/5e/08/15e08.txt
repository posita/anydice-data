function: craft DIE:n double THRESHOLD:n {
  if DIE = 10 {result: 2 + [craft 1d10 double THRESHOLD]}
  if DIE >= THRESHOLD {result: 2}
  if DIE >= 7 {result: 1}

  if DIE = 6 {result: [craft 1d10 double THRESHOLD]}
  result: 0
}

function: forge SUCCESSES:n difficulty DIFF:n {
  if SUCCESSES < DIFF {result: 0}
  result: SUCCESSES - DIFF + 1
}

set "maximum function depth" to 2
output [craft d10 double 10]
output [craft d10 double 9]
output [craft d10 double 8]
output [craft d10 double 7]
output [forge 10d[craft d10 double 9] difficulty 10]
output [forge 20d[craft d10 double 9] difficulty 10]
output [forge 20d[craft d10 double 8] difficulty 10]
output [forge 20d[craft d10 double 7] difficulty 10]

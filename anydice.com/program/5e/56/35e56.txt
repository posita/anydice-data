function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

X: [d4 reroll {1} as d4]
output (1dX + 1) * 3 named "Magic Missile (Reroll on 1)"

X: [d4 reroll {1,2} as d4]
output (1dX + 1) * 3 named "Magic Missile (Reroll on 1 or 2)"

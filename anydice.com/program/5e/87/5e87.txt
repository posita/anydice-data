function: reroll D:n if below S:n and R:n above zero {
X: R - 1
if D <= S & R > 0 {result: [reroll 1d10 if below S and X above zero]}
result: D
}

function: roll D:n dice with R:n rerolls and strategy S:n {
X: 0
loop N over {1..D}{
X: X + [reroll 1d10 if below S and R above zero]
}
result: X
}

output [roll 1 dice with 2 rerolls and strategy 5]


function: level LEVEL:n target CL:n STAT:n {
 if STAT = 3 {result: d20+LEVEL-3 >= CL} 
 if STAT >= 4 & STAT <= 5 {result: d20+LEVEL-2 >= CL}
 if STAT >= 6 & STAT <= 8 {result: d20+LEVEL-1 >= CL}
 if STAT >= 13 & STAT <= 15 {result: d20+LEVEL+1 >= CL}
 if STAT >= 16 & STAT <= 17 {result: d20+LEVEL+2 >= CL}
 if STAT >= 18 & STAT <= 19 {result: d20+LEVEL+3 >= CL}
 result: d20+LEVEL >= CL
}

loop STAT over {3..19} {
CLI: 18
output [level 1 target 18 STAT] named "[STAT]"
}

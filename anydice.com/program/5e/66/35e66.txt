L: 5   \Desired level\
H: 10   \Desired Hit Die size\

function: std X:n hd Y:n {
 if X > 1 { result: [highest of [std X-1 hd Y] and XdY] }
 result: 1dY
}

function: adv X:n hd Y:n {
 if X > 1 { result: [highest of [adv X-1 hd Y] and XdY] }
 result: [highest 1 of 2dY]
}

function: max X hd Y {
 if X > 1 { result: [highest of [max X-1 hd Y] and XdY] }
 result: Y
}

output [std L hd H] named "1st level flat HP"
output [adv L hd H] named "1st lvl Adv. HP"
output [max L hd H] named "1st level max HP"
output LdH named "LdH (for reference)"

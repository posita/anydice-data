function: seq ROLL:s and FACE:n {
  OFFSET: FACE - 1
  result: [count FACE in ROLL] + OFFSET * [ROLL contains FACE]
}

function: model NUM:n FACES:n{
  DICE:NUMdFACES
  MAX: [seq DICE and 1]
  loop ITER over {2..FACES}{
    MAX:[highest of [seq DICE and ITER] and MAX]
  }

  result: MAX
}

output [model 1 6]

CHECKDIE: 6
SKILLDIE: 6
CHECKTARGET: 5
SKILLTARGET: 6
ATLEAST: 1

function: check ROLL:s {
  result: (ROLL >= CHECKTARGET) >= ATLEAST
}

function: skill ROLL:s {
  result: (ROLL >= SKILLTARGET) >= ATLEAST
}

function: test CHECKROLL:s {
  if [check CHECKROLL] = 1 {
    result: 1
  }
  result: 0
}

function: test CHECKROLL:s with SKILLROLL:s {
  if 1@[check CHECKROLL] = 1 {
    result: 1
  }
  if [skill SKILLROLL] = 1 {
    result: 1
  }
  result: 0
}

loop N over {1..3} {
  output [test NdCHECKDIE] named "test [N]d[CHECKDIE]"
  loop I over {1..5} {
    output [test NdCHECKDIE with IdSKILLDIE] named "test [N]d[CHECKDIE] with [I]d[SKILLDIE]"
  }
}

function: explode N:n THRESHOLD:n{
 if N >= THRESHOLD { result: 10 + [explode d10] }
 result: N
}

function: maximum of SEC:s {
 result: 1@[reverse {SEC}]
}

function: X:n r C:n{
 if X=1{result:[explode d10 C]}
 if X=2{result:[highest of [explode d10 C] and [explode d10 C]]}
 result: [highest of [X-1 r C] and [explode d10 C]]
}

set "explode depth" to 10
set "maximum function depth" to 10
output [10 r 7]

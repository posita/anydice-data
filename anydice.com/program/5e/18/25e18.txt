function: roll A:n plus C:n vs D:n {
if A = 2 { result: 0 }
if A = 12 { result: 3 }
if A + C >= D+D { result: 3 }
if A + C >= D { result: 2}
if A + C <= D/2 { result: 0 }
result: 1
}

loop Y over {-3..3} {
output [roll 2d6 plus Y vs 2d6-3] named "Stat (+[Y]) vs Easy"
output [roll 2d6 plus Y vs 2d6] named "Stat (+[Y]) vs Medium"
output [roll 2d6 plus Y vs 2d6+3] named "Stat (+[Y]) vs Hard"
output [roll 2d6 plus Y vs 3d6+3] named "Stat (+[Y]) vs Impossible"
}

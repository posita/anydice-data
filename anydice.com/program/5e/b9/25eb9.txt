function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

function: reroll R:n under N:n {
   if R < N { result: 1d6 } else {result: R}
}

output 4d[reroll {1,2} on d6]  named "4d6r2"

output 4d[reroll {1,2} on d6] + [reroll {1,2} on d4]  named "4d6r2 enlarge"

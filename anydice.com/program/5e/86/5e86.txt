function: reroll D:n if below five and R:n above zero {
X: R - 1
if D <= 5 & R > 0 {result: [reroll 1d10 if below five and X above zero]}
result: D
}

function: roll D:n dice with R:n rerolls {
X: 0
loop N over {1..D}{
X: X + [reroll 1d10 if below five and R above zero]
}
result: X
}

output [roll 3 dice with 2 rerolls]

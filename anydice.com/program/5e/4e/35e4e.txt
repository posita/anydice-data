function: attack hit HIT:n on H wound WOUND:n on W:n save SAVE:n on S:n {
result: HIT >= H & WOUND >= W & SAVE < S
}

function: devastating hit HIT:n on H wound WOUND:n on W:n save SAVE:n on S:n {
if WOUND = 6 {result: HIT >= H}
result: HIT >= H & WOUND >= W & SAVE < S
}

function: sustain SUS:n hit HIT:n on H wound WOUND:n on W:n save SAVE:n on S:n {
if HIT = 6 {result: (SUS+1)d(HIT >= H & WOUND >= W & SAVE < S)}
result: HIT >= H & WOUND >= W & SAVE < S
}

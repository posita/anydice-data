MINOR: 5
MAJOR: 10
SEVERE: 15

function: range on DIE:n between LOW:n and HIGH:n {
if DIE >= LOW & DIE <= HIGH { result: 1 }
result: 0
}

loop EV over {9,10,11} {
  HIT: (1d20) > EV
  DAMAGE: (1d14+4) * HIT
  
  if EV = 9 {
    DAMAGE: DAMAGE - 3
  }
  if EV = 10 {
    DAMAGE: DAMAGE - 5
  }
  if EV = 11 {
    DAMAGE: DAMAGE - 7
  }
  HP: (DAMAGE >= SEVERE) * 3 + [range on DAMAGE between MAJOR and SEVERE-1] * 2 + [range on DAMAGE between MINOR and MAJOR-1] * 1

  \output HIT named "Hit? at EV:[EV]"\
  \output DAMAGE named "Damage at EV:[EV]"\
  output HP named "HP loss at EV:[EV]"
}

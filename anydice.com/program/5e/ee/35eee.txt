MINOR: 5
MAJOR: 10
SEVERE: 15

loop EV over {9,10,11} {
  HIT: (1d20) > EV
  DAMAGE: (1d14+4) * HIT
  HP: (DAMAGE >= SEVERE) * 3 + (DAMAGE >= MAJOR & DAMAGE < SEVERE) * 2 + (DAMAGE >= MINOR & DAMAGE < MAJOR) * 1

  output HIT named "Hit? at EV:[EV]"
  output DAMAGE named "Damage at EV:[EV]"
  output HP named "HP loss at EV:[EV]"
}

function: roll ATKDICE:n plus ATKBONUS vs AC:n damage DMGDICE:d damagebonus DMGBONUS:n {
    DMGROLL: DMGDICE + DMGBONUS
    CRITROLL: DMGDICE*2 + DMGBONUS
    MISS: 0

    if ATKDICE = 1 {
        result: MISS
    }
    if ATKDICE >= 20 {
        result: CRITROLL
    }
    if ATKDICE+ATKBONUS >= AC {
        result: DMGROLL
    }
    result: MISS
}

AC: 18
ATKDICE: 1d20
ADVATK: [highest 1 of 2d20]
DISATK: [lowest 1 of 2d20]
ATK: 10
DMG: 1d6
DMGB: 5
SHARPATK: -5
SHARPDMG: 10
HUNTERSMARK: 1d6


output [roll ATKDICE plus ATK vs AC damage DMG+HUNTERSMARK damagebonus DMGB] named "Raw"
output [roll ATKDICE plus ATK+SHARPATK vs AC damage DMG+HUNTERSMARK damagebonus DMGB+SHARPDMG] named "Sharpshooter"

output [roll ADVATK plus ATK vs AC damage DMG+HUNTERSMARK damagebonus DMGB] named "Raw advantage"
output [roll ADVATK plus ATK+SHARPATK vs AC damage DMG+HUNTERSMARK damagebonus DMGB+SHARPDMG] named "Sharpshooter advantage"

output [roll DISATK plus ATK vs AC damage DMG+HUNTERSMARK damagebonus DMGB] named "Raw disadvantage"
output [roll DISATK plus ATK+SHARPATK vs AC damage DMG+HUNTERSMARK damagebonus DMGB+SHARPDMG] named "Sharpshooter disadvantage"

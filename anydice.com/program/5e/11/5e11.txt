function: R:s and O:s with B:n bonuses and P:n penalties {
 result: R + O + [lowest of O = -1 and B] - [lowest of O = 0 and P]
}

RUNES: 14
FOCUS: 0
OMENS: 5
BONUSES: 1
PENALTIES: 0
OD: 10

RUNE_POOL: RUNES d{0,1}
OMEN_POOL: OMENS d{-1,0}
OD_POOL: OD d{-1,0}

output RUNE_POOL + FOCUS + OMEN_POOL + OD_POOL named "Regular"
output [RUNE_POOL and OMEN_POOL with BONUSES bonuses and PENALTIES penalties] + OD_POOL + FOCUS named "With bonuses or penalties"
output ([RUNE_POOL and OMEN_POOL with BONUSES bonuses and PENALTIES penalties] + OD_POOL + FOCUS) <0 named "Failure Rate"
output ([RUNE_POOL and OMEN_POOL with BONUSES bonuses and PENALTIES penalties] + OD_POOL + FOCUS) =0 named "Stalemate Rate"
output ([RUNE_POOL and OMEN_POOL with BONUSES bonuses and PENALTIES penalties] + OD_POOL + FOCUS) >0 named "Success Rate"

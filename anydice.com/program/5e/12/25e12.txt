function: explodesix A:n 
{
if A<4 {result: 0}
if A<6 {result: 1}
result:[explodesix d6]+1
}

function: explodeone A:n
{
if A>3 {result: 0}
if A>1 {result: -1}
result:[explodeone d6]-1
}

function: romaroll A:s
{
if A=1 {result:[explodeone d6]-1}
if A<4 {result:0}  
if A<6 {result:1}
if A=6 {result:[explodesix d6]+1}
}
set "maximum function depth" to 6
output 1d[romaroll d6]

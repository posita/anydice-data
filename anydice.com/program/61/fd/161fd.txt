function: X:n gwf {
if X = 1 {result: 1d10}
if X = 2 {result: 1d10}
result: X
}

function: gwfb DIE:d {
  result: [reroll DIE as DIE if under 3]
}
function: reroll ROLL:n as DIE:d if under LIMIT:n {
  if ROLL < LIMIT { result: DIE }
  else { result: ROLL }
}

output [d10 gwf] + 3 named "Longsword GWF"
output d10 + 3 named "Longsword Hexblade"
output d8 + 5 named "Longsword Dueling"

GWF: [gwfb d6]
output 2dGWF + 2 named "Maul with GWF"

output 2d6 + 2 named "Maul without GWF"

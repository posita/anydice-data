DIE: 1d10          \ Rolling 100 is OK if you got a skill of 100. \

function: roll A:n B:n C:n {
    if [highest of A and B] <= SKILLVALUE {
    if (C) <= SKILLVALUE { result: 2}
  }
    if [highest of A and B] <= SKILLVALUE {
    if (C) > SKILLVALUE { result: 1}
  }
    if (C) <= SKILLVALUE {
    if [highest of A and B] > SKILLVALUE { result: 1}
  }
    if [highest of A and B] > SKILLVALUE {
    if (C) > SKILLVALUE { result: 0}
  }
}

loop SKILLVALUE over {2..7} {
  output [roll DIE DIE DIE] named "doppio d10 A SVANTAGGIO against skill value [SKILLVALUE]"
}

function: hilo D:s to X:n {
  LO: [count {1..X} in D]
  result: #D - 2 * LO
}

function: test T:s against GM:s {
  if T = -5 { result: -3 }
  if T = 5 { result: 3 }
  if T < GM & T > 0 { result: -1 }
  if T > GM & T < 0 { result: 1 }
  if T > GM | T > 0 { result: 2 }
  else { result: -2}
}

output [test [hilo 5d10 to 5] against [hilo 5d10 to 5]]

CB: 4
AC: 16

function: roll A:d plus E:d if hit landed with B:n plus C:n vs D:n crit on G:n confirm with F:n {
    if B = 20 {
        if ((F+C) >= D) & F > 1 { result: 2*A + E }
        result: A + E
    } else  {
        if B+C >= D & B > 1 { result: A + E }
        result: 0
    }
}

function: full attack against A:n with bonuses B:s with damage C:d plus D:d {
    DAMAGE: 0
    loop BONUS over B{
        DAMAGE: DAMAGE + [roll C plus D if hit landed with 1d20 plus BONUS vs A crit on 20 confirm with 1d20]
    }
    result: DAMAGE
}

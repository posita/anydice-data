CB: 4
AC: 16

function: roll A:d if hit landed with B:n plus C:n vs D:n {
    if B = 20 | (B+C >= D & B > 1) { result: A }
    result: 0
}

function: full attack against A:n with bonuses B:s with damage C:d {
    DAMAGE: 0
    loop BONUS over B{
        DAMAGE: DAMAGE + [roll C+5d6 if hit landed with 1d20 plus BONUS vs A]
    }
    result: DAMAGE
}

loop AC over {16, 20, 24, 28} {
  output [full attack against AC with bonuses {8, 8, 3} with damage 5d6+6] named "TWF 8,8,3 (5d6+6) vs [AC]"
  output [full attack against AC with bonuses {10, 5} with damage 5d6+6] named "FA 10,5 (5d6+6) vs [AC]"
  output [full attack against AC-10 with bonuses {20} with damage 9d6] named "Sc.Ray SA vs [AC]"
  output [full attack against AC with bonuses {27} with damage 2d6+5d6+14+4] named "TStrk C.SG PA(7) vs [AC]"
  output [full attack against AC with bonuses {14} with damage 2d6+5d6+4] named "C.S.Grasp vs [AC]"
  output [full attack against AC with bonuses {14, 9} with damage 2d6+7] named "Full attack 10,5 vs [AC]"
  output [full attack against AC-10 with bonuses {12, 12} with damage 4d6] named "Sc.Ray vs [AC]"
}

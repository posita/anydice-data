function: gamble G:s with A:s and B:s and C:s {
 SUCCESS: ((1d20+A)>(2d10+5)) + ((1d20+B)>(2d10+5)) + ((1d20+C)>(2d10+5))
 result: (SUCCESS = 1) * G + (SUCCESS = 2) * (G + 3 * G / 4) + (SUCCESS = 3) * (G + 6 * G / 4)
}

loop G over {1000} {
output [gamble G with 0 and 0 and 0] - G named "Gambling [G]gp"
}

    if ROLL = 1 {
        result: 0
    }
    if ROLL = 20 {
        if ROLL + BONUS >= AC{
            result: DMG * 2 + CRIT_DMG  
        }
        result: DMG
    }
    if ROLL + BONUS >= AC + 10 {
        result: DMG * 2 + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

         result: DAMAGE
    }
    result: SAVE
}

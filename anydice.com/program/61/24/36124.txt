\
 Heart
 0: Critical Failure
 1: Failure
 2: Success at Cost
 3: Success
 4: Critical Success
\

function: heart ROLL:s {
 if 1@ROLL = 10 {
  result: 4 
 }
 if 1@ROLL >= 8 {
  result: 3 
 }
 if 1@ROLL >= 6 {
  result: 2 
 }
 if 1@ROLL >= 2 {
  result: 1 
 }
 if 1@ROLL = 1 {
  result: 0 
 }
}

loop P over {1..4} {
 output [heart Pd10] named "[P]d"
}

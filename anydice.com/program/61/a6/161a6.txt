DIE: 1d10          \ Rolling 100 is OK if you got a skill of 100. \

function: roll A:n B:n {
    if (A) <= SKILLVALUE {
    if (B) <= SKILLVALUE { result: 2}
  }
    if (A) <= SKILLVALUE {
    if (B) > SKILLVALUE { result: 1}
  }
    if (B) <= SKILLVALUE {
    if (A) > SKILLVALUE { result: 1}
  }
    if (A) > SKILLVALUE {
    if (B) > SKILLVALUE { result: 0}
  }
}

loop SKILLVALUE over {2..7} {
  output [roll DIE DIE] named "doppio d10 against skill value [SKILLVALUE]"
}

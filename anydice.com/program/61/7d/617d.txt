    if DIEROLL = 20 | (DIEROLL >= CRIT & DIEROLL+ATKBONUS >= AC) {
        if (CONFIRM = 20 | (CONFIRM+ATKBONUS) >= AC) & CONFIRM > 1 { result: 2*DMG+ PRECISION }
        result: DMG + PRECISION
    } else  {
        if DIEROLL+ATKBONUS >= AC & DIEROLL > 1 { result: DMG + PRECISION }
        result: 0
    }
}

}

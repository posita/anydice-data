
function: helper ROLL:s {
 SCORE: 0
 if ( ROLL >= 4)> 0 { SCORE: 1 + [evaluate 1d10]}
 result: SCORE
}

TM: 1d6

function: thelper G:s{
result: [highest of G and TM]
}

function: evaluate S:s{
 result: [helper S]}

function: advantage T:s{
 result: [evaluate [thelper T]]
}



output [advantage 1d10]

function: check A:n vs B:n {
 if A < B - 3 { result: -2 } \ CF \
 if A < B { result: -1 } \ F \
 if A > B + 3 { result: 2 } \ CS \
 if A > B { result: 1 } \ S \
 result: 0 \ DR \
}

SD: d{-1, 0:2, 1:2, 2}
MIN_A : 10
MAX_A : 15
DL : 5

loop A over {MIN_A..MAX_A} {
  output [check AdSD vs DL] named "[A] vs [DL]"
}

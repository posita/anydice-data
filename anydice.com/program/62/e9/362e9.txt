function: find X:n sets of Y:n on TARGET:n plus in ROLL:s {
UN: 0
DEUX: 0
TROIS: 0
ZERO: 0
 IDX: 1
 END: 0
 loop M over {1..#ROLL} {
    if (M@ROLL >= 2) & ((M+1)@ROLL < 2) & (END=0) {
      IDX: M+1
      END: 1
    }
 }
 LOWPART: {IDX..#ROLL}@ROLL              \sequence with 1s & 0s\
 UN: {1:LOWPART}
 DEUX: {1:((IDX-1)*3 - {1..IDX-1}@ROLL)}
 TROIS: {1:((IDX-1) - DEUX)}
 ZERO: {1:#ROLL -UN-DEUX-TROIS}
 SIZE: {#ZERO, #UN, #DEUX, #TROIS}

 RES: 0
 loop N over {1+TARGET..#SIZE}{
   if(N@SIZE >= Y){RES: RES+1}
   if(RES >= X) {result: RES}
 }
 result: 0
}


function: unique values in S:s {
  U: {}
  loop N over S { if !(N = U) { U: {U, N} } }
  result: U
}
function: find PATTERN:s in ROLL:s {
  COUNTS: {}
  loop N over [unique values in ROLL] {
    COUNTS: {COUNTS, N = ROLL}
  }
  result: [sort PATTERN] = [sort COUNTS]
}

function: PATTERN:s upto SIZE:n{
 SEQ: PATTERN
 START: 1+PATTERN
 loop N over {START..SIZE} {
   SEQ: {SEQ,1}
 }
 result: SEQ
}

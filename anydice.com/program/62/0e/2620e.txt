
function: evaluate ATTR_POOL:s SKILL:n {
 SUCCESSES: (ATTR_POOL <= SKILL) - (ATTR_POOL = 0) + (ATTR_POOL = 9)
 if SKILL > 0{
  SUCCESSES: SUCCESSES + [count SKILL in ATTR_POOL]
 }
 result: SUCCESSES
}

\loop TARGET over {0..9}{\
TARGET: 1
 output [evaluate 9d{0..9} TARGET ] named "target [TARGET]"
\}\

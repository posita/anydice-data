SPELLTN: 10
WILL:3
MAGIC: 3

function: fade ROLL:d {
result: ROLL = 6
}
function: possess ROLL:d {
result: ROLL < 14
}
function: dragon D:d {
result: D = 1
}
function: magic ROLL:d DRAGON:d {
result: (ROLL+DRAGON < SPELLTN) + [dragon DRAGON] + [fade 1d6] + [possess 3d6+WILL]
}

output [magic 2d6+MAGIC 1d6]

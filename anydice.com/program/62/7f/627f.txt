function: sign S:s {
if S>0 { result: 1 }
if S<0 { result: -1 }
result: 0
}

function: A:n beats B:n {
A_DICE: Ad6
B_DICE: Bd6

A_MAX: [highest 1 of A_DICE]
B_MAX: [highest 1 of B_DICE]

WINS: [count B_MAX in A_DICE]-[count A_MAX in B_DICE]

result: [sign WINS]
}

loop A over {1..5} {
 loop B over {1..5} {
  output [A beats B] named "[A] beats [B]"
 }
}

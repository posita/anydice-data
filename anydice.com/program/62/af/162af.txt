function: roll gwf damage with SIDES_ON_DMG_DIE:n sided die {
  \ create a die that simulates rerolling the first 1 or 2.
    This creates a sequence of 1,2,2,3...X
\
  GWF_DIE: {{1..2}:2, {3..SIDES_ON_DMG_DIE}:SIDES_ON_DMG_DIE+2}
  result: dGWF_DIE
}

\output 2d[roll gwf damage with 6 sided die]\

function: dees greatsword {
    result: 3d[roll gwf damage with 6 sided die]
}

function: roll elementaladept damage with SIDES_ON_DMG_DIE:n sided die {
  \ create a die that simulates rolling a 1 as if it were a 2 for an arbitrary sided die \
  EA_DIE: {2, 2, {3..SIDES_ON_DMG_DIE}}
  result: dEA_DIE
}

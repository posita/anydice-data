POW: 10
DEF: 12
ARM: 18


function: norm ATT:n TGT:n MELEE:n {
  if ATT >= TGT {
    if MELEE {
      DAM: 3d6 + POW - ARM
      result: [highest of 0 and DAM]
    }
    DAM: 2d6 + POW - ARM
    result: [highest of 0 and DAM]
  }
  result: 0
}

function: crit ATT:n TGT:n FST:n SEC:n MELEE:n {
  if ATT >= TGT {
    if FST = SEC {
      if MELEE {
        DAM: 4d6 + POW - ARM
        result: [highest of 0 and DAM]
      }
      DAM: 3d6 + POW - ARM
      result: [highest of 0 and DAM]
    }
  }
  result: 0
}

function: barrier MELEE:n {
  ATT: 2d6 + 2
  result: [norm ATT DEF MELEE]
}

function: deadeye MELEE:n {
  ATT: 3d6
  FST: [highest 1 of ATT]
  SEC: [middle 1 of ATT]
  result: [crit ATT DEF FST SEC MELEE]
}


output [barrier 0] named "Temp Barrier"
output [deadeye 0] named "Deadeye"
output [barrier 1] named "Temp Barrier Melee"
output [deadeye 1] named "Deadeye Melee"

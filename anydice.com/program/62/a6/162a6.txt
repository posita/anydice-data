function: burn DICE:s {
 FULL: [count 6 in DICE]
 HALF: [count {4,5} in DICE]
 if FULL >= 2 { result: 3 }
 if FULL >= 1 { result: 2 }
 if HALF >= 1 { result: 1 }
 result: 0
}

function: low DICE:s {
 FULL: [count 6 in DICE]
 HALF: [count {4,5} in DICE]
 if FULL = 2 { result: 2 }
 if HALF + FULL = 2 { result: 1 }
 result: 0
}

output [low 2d6] named "0D"

loop N over {1..6}{
output [burn Nd6] named "[N]D"
}

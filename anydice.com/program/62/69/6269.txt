function: check A:n vs B:n {
 if A < B - 3 { result: -2 } \ CF \
 if A < B { result: -1 } \ F \
 if A > B + 3 { result: 2 } \ CS \
 if A > B { result: 1 } \ S \
 result: 0 \ DR \
}

SD: d{-1, 0:2, 1:2, 2}
MIN_A : 1
MAX_A : 3
MIN_B : 1
MAX_B : 3

loop A over {MIN_A..MAX_A} {
 A_POOL: AdSD
 loop B over {MIN_B..MAX_B} {
  output [check A_POOL vs BdSD] named "[A] vs [B]"
 }
}

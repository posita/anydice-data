\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 2
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 2d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] + ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "67 Rifle Multiattack"

DAMAGE: 2d8 + ATTACKMOD + 1d6 + DMGBONUS
CRITDAMAGE: 4d8 + ATTACKMOD + 2d6 + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] + ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "67 Hunter's Mark Rifle Multiattack"

DAMAGE: 2d8 + 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d8 + 2d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] + ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "67 Zephyr Strike Rifle Multiattack"

DAMAGE: 2d8 + ATTACKMOD + 1d6 + 1d8 + DMGBONUS
CRITDAMAGE: 4d8 + ATTACKMOD + 2d6 + 2d8 + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] + ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "67 Hunter's Mark Zephyr Strike Rifle Multiattack"

ACCBONUS: 0
DAMAGE: 1d4 + 1d4 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d4 + 2d4 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] + ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "67 Dagger Multiattack"

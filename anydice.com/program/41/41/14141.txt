function: attack A:n B:n C:d D:n{
 if A>B {result: 0}
 if A>(B-20) {result: [dam C D]}
 if A>(B-40) {result: [dam C D]+[dam C D]}
 if A>(B-60) {result: [dam C D]+[dam C D]+[dam C D]}
}

function: dam C:n D:n{
 if D>C {result: 0}
 result: C-D
}

\output [attack 1d100 "WS" "Weapon Dam" "Enemy Armor"]\
output [attack 1d100 50 1d10+3 0]

function: two to X:n if Z:n is one {
  if Z = 1 {
   result: X + 2
  } else {
   result: X 
  }
}
function: get int with plus two if highest from STR:n CON:n DEX:n INT:n WIS:n CHA:n {
 GREATER: 1
 TIES: 0
 loop N over {STR, CON, DEX, WIS, CHA} {
   if INT < N {
     GREATER: 0
   }
   if INT = N {
     TIES: TIES + 1
   }
 }
 if GREATER = 1 {
   if TIES = 0 {
    result: INT + 2
   } else {
    result: [two to INT if 1dTIES is one]
   }
 } else {
   result: INT
 }
}

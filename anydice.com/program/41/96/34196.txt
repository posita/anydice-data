function: insp A:n and B:n {
 if A = B { result: 1 }
 result: 0
}

function: neatroll A:d and B:d and C:d plus D:n {
 X: [insp A and B]
 result: A + B + [highest 1 of C] + D + X*100
}

output [neatroll d10 and d10 and 2d6 plus 4]
output [neatroll d10 and d10 and -2d6 plus 4]
output [neatroll d6 and d6 and 0d6 plus 0]

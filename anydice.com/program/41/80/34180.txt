function: adv X:d {
  result: [highest of X and X]
}

function: dmg ROLL:n CRITMOD:n {
  if ROLL >= 20 - CRITMOD {
    result: [adv d6] + [adv d10] + [adv d4] + [adv d6] + [adv d4] + 4 + [adv d6] + [adv d10] + [adv d4] + [adv d6] + [adv d4] + 10
  } else {
    result: [adv d6] + [adv d10] + [adv d4] + [adv d6] + [adv d4] + 4
  }
}

output [dmg d20 1]

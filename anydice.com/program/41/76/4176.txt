function: magic BIG:d OPERATOR:n SMALL:d
{
    if (OPERATOR = 1)
    {
        result: BIG / SMALL
    }
    if (OPERATOR = 2)
    {
        result: BIG - SMALL
    }
    if (OPERATOR = 3)
    {
        result: BIG + SMALL
    }
    if (OPERATOR = 4)
    {
        result: BIG * SMALL
    }
}

output [magic (1d10-1)*10 1d4 1d10-1] named "Calculator"


D: d{0,0, 0,0, 0,0, 1,1}


function: sum ROLL:n {
  result: ROLL
}

function: check ROLL:s T:n {
  SUM: [sum ROLL]
  result: (SUM >= T)
  \result: (T@ROLL >= 1)\
}

function: roll ROLL:s T:n {
  result: [check ROLL T] + [check ROLL (T-1)]
}

C: 2

output [roll {2..3}@3dD C] named "1 + 0"
output [roll {2..4}@4dD C] named "2 + 0"
output [roll {2..5}@5dD C] named "3 + 0"

loop N over {1..16} {
  output [roll NdD C] named "[N]"
}

    \ Rolling Xd6 against a fixed target number, and comparing it to the game master's roll against the same target number. \
DIETYPE: d6       \ Change this to any other die type. \
GAMEMASTER: 4     \ The number of dice the game master rolls. \
TARGETNUMBER: 4   \ The target number to beat. \

loop PLAYER over {1..9} {
  output [count {TARGETNUMBER..6} in PLAYERdDIETYPE] >= [count {4..6} in GAMEMASTERdDIETYPE] named "Player rolled [PLAYER]d6 against the game master's [GAMEMASTER],  with a target number of [TARGETNUMBER]"
}

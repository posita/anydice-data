function: adv X:d {
  result: [highest of X and X]
}

\ this considers we have the (BG3) Savage Attacker feat \
function: weapon damage {
  result: [adv d6] + [adv d10] + [adv d4] + [adv d6] + [adv d4]
}

function: ROLL:n damage roll {
  ROLL_CRIT_MOD:1
  STR_MOD: 5
  DMG_MOD: STR_MOD + 2 + 2
  CRIT_DMG_MOD: 10

  \ weapon damage is defined in the dedicated function \
  DMG: [weapon damage] + DMG_MOD

  if ROLL + ROLL_CRIT_MOD >= 20 {
    DMG: DMG + [weapon damage] + CRIT_DMG_MOD
  }
  result: DMG
}

output [d20 damage roll] named "with SA"

\ this considers we do *NOT* have the (BG3) Savage Attacker feat \
function: weapon damage without savage attacker {
  result: d6 + d10 + d4 + d6 + d4
}

function: ROLL:n damage roll without savage attacker {
  ROLL_CRIT_MOD:1
  STR_MOD: 5
  DMG_MOD: STR_MOD + 2 + 2
  CRIT_DMG_MOD: 10

  \ weapon damage is defined in the dedicated function \
  DMG: [weapon damage without savage attacker] + DMG_MOD

  if ROLL + ROLL_CRIT_MOD >= 20 {
    DMG: DMG + [weapon damage without savage attacker] + CRIT_DMG_MOD
  }
  result: DMG
}

output [d20 damage roll without savage attacker] named "without SA"

function: roll M:n vs DC:n{
  if M >= DC {
    result: 1
  } else { 
   result: 0
  }
}

function: bard D:n {
  if D = 1 {
    result: 0
  } else if D = 20 {
    result: 31
  } else if D <= 9 {
    result: 10+M
  } else {
    result: D+M
  }
}

DC: 23
M: 7
ADV: [highest 1 of 2d20]
DIS: [lowest 1 of 2d20]
NORM: d20
D: DIS

loop N over {1..30}{
  output [roll [bard D] vs N] named "[N]"
}

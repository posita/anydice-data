\ Taking the highest outcome in an exploding dice pool \
\ It will only re-roll exploding dice three times \

loop DICETYPE over {6, 8, 10} {
  output 1 named "~~~~~~~~~~~~~~~~~~ NEW DIE TYPE ~~~~~~~~~~~~~~"
  loop NUMBEROFDICE over {3..6} {
   output 1@NUMBEROFDICEd[explode dDICETYPE] named "Exploding [NUMBEROFDICE]d[DICETYPE]"
  }
}

function: luck A:n {
if A = 0 {result: 0}
if A > 0 {result: d{1..2+2*A}}
if A < 0 {result: d{2*A-2..-1}}
}

function: die A:n difficulty B:n luck C:n {
if A-B+C <= 7 {result: 1}
if A-B+C >= 18 {result: 3}
result: 2
}

loop D over {0, 4, 8, 12, 16} {
loop L over {-4..4} {
output [die d20 difficulty D luck [luck L]] named "Difficulty [D] Luck [L]"
}}

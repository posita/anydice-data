ATB: 7
DMG: 6
AVR: 0

function: X:n gwf{
  if X > 2 {result: X}
  result: 1d10
}

loop N over {10..20}{

  AC: N

  function: X:n hit{
    if (X+ATB) >= AC | X = 20 {result: 1}
    result: 0
  }

  AVR: AVR + (([1d10 gwf]+DMG) * [1d20 hit])

  output ([1d10 gwf]+DMG) * [1d20 hit] named "AC [N]"

}

output AVR / 11 named "average"

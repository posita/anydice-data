function: count_successes ROLLS:s difficulty DIFFICULTY:n {
  SUCCESSES: ROLLS >= DIFFICULTY
  FAILURES: ROLLS = 1
  if SUCCESSES >= 1 {
    result: [highest of 4 and SUCCESSES]
  }
  else {
    if FAILURES > (ROLLS - SUCCESSES) {
      result: -2
    }
    else {
      result: -1
    }
  }
}

output [count_successes 1d6 difficulty 3] named "Test"

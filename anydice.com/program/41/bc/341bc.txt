function: attack for ROLL:n plus BONUS:n vs DEFENSE:n for DAMAGE_COUNT:n die DAMAGE:n plus DB:n {
 if ROLL = 1 { result: 0 }
 if ROLL = 20 { result: (2*DAMAGE_COUNT)dDAMAGE + DB }
 if ROLL + BONUS >= DEFENSE { result: DAMAGE_COUNTdDAMAGE + DB }
 result: 0
}

function: zombie against DEFENSE:n {
 result: [attack for d20 plus 1 vs DEFENSE for 1 die 6 plus 1]
}

BODY_COUNT: 32
SWARM_SIZE: BODY_COUNT / 4
ATTACK_COUNT: BODY_COUNT / 8

DICE_COUNT: BODY_COUNT / 8
DICE_TYPE: 6


function: swarm against DEFENSE:n {
 result: SWARM_SIZEd[attack for d20 plus BONUS vs DEFENSE for DICE_COUNT die DICE_TYPE plus BONUS]
}

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: bonusattackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: BONUSCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: BONUSDAMAGE }
 result: 0
}
\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 9
DMGBONUS: 0
PROFICIENCY: 0
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 3d6
CRITDAMAGE: 6d6
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Accursed Specter"

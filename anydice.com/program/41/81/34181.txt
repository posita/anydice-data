function: adv X:d {
  result: [highest of X and X]
}

function: ROLL:n damage roll with CRITMOD:n critmod {
  STR_MOD: 5
  DMG_MOD: 2 + 2
  DMG: [adv d6] + [adv d10] + [adv d4] + [adv d6] + [adv d4] + STR_MOD + DMG_MOD

  CRIT_DMG_MOD: 10
  CRIT_DMG: [adv d6] + [adv d10] + [adv d4] + [adv d6] + [adv d4] + CRIT_DMG_MOD

  if ROLL >= 20 - CRITMOD {
    result: DMG + CRIT_DMG
  } else {
    result: DMG
  }
}

output [d20 damage roll with 1 critmod]

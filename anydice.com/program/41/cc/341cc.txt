function: crit ROLL:d THRESHOLD {
 HITROLL: [highest 1 of ROLL]
 CRIT: [count 20 in HITROLL] > 0
 result: CRIT
}

function: attack ATKS DMG CRIT {
 TOTAL: 0
 loop I over {1..ATKS}{
  TOTAL: TOTAL+DMG
  TOTAL: TOTAL + (CRIT * [crit 2d20 20])
 }
 result: TOTAL
}

output [attack 1 1d12+4+1d6+1d6+1d8+10 12+6+6+8] named "Bonnie Full power"
output [attack 1 1d12+4+10 12] named "Bonnie sharpshooter only"
output [attack 1 5d6+4+1d8 30+8] named "5d6+4 rogue with die"

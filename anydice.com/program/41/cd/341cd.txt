\ shorthand for advantage \
function: adv X:d {
  result: [highest of X and X]
}

function: roll X:d {
  result: [highest of 0 and X]
}

function: attack ROLL:n with modifier CRIT_MOD:n {
  if ROLL + CRIT_MOD >= 20 {
    result: 1
  } else {
    result: 0
  }
}

\ fluent versions of booleans \
WITH: 1
WITHOUT: 0

\ -------------------------------------------------------------- \

ASTARION_ROLL_CRIT_MOD:1
ASTARION_DEX_MOD: 5
ASTARION_DMG_MOD: ASTARION_DEX_MOD + 2 + 2
ASTARION_CRIT_DMG_MOD: 10
ASTARION_ATTACKS: 6

function: astarion weapon damage SA:n savage attacker {
  if SA {
    result: [adv d6] + [adv d10] + [adv d4] + [adv d6] + [adv d4] + [adv d4]
  } else {
    result: d6 + d10 + d4 + d6 + d4 + d4
  }
}

function: astarion ROLL:n damage roll SA:n savage attacker {
  DMG: [astarion weapon damage SA savage attacker] + ASTARION_DMG_MOD
  if ROLL + ASTARION_ROLL_CRIT_MOD >= 20 {
    DMG: DMG + [astarion weapon damage SA savage attacker] + ASTARION_CRIT_DMG_MOD
  }
  result: ASTARION_ATTACKS * DMG
}

\ -------------------------------------------------------------- \

LAEZEL_ROLL_CRIT_MOD: 0
LAEZEL_STR_MOD: 5
LAEZEL_DMG_MOD: LAEZEL_STR_MOD + 2
LAEZEL_CRIT_DMG_MOD: 0
LAEZEL_ATTACKS: 4

function: laezel weapon damage SA:n savage attacker and level N:n smites {
  if SA {
    DMG: [adv d6] + [adv d6] + [adv d4] + [adv d4]
  } else {
    DMG: d6 + d6 + d4 + d4
  }

  \ apply smites \
  if N > 0 {
    if SA {
      DMG: DMG + [adv d8]
    } else {
      DMG: DMG + d8
    }

    MAX_SMITES: [lowest of N and 4]
    loop N over {1..MAX_SMITES}{
      if SA {
        DMG: DMG + [adv d8]
      } else {
        DMG: DMG + d8
      }
    }
  }

  result: DMG
}

function: laezel ROLL:n damage roll SA:n savage attacker and level N:n smites {
  CRITICAL: [attack ROLL with modifier LAEZEL_ROLL_CRIT_MOD]
  DMG: [laezel weapon damage SA savage attacker and level N smites] + LAEZEL_DMG_MOD
  if CRITICAL {
    DMG: DMG + [laezel weapon damage SA savage attacker and level N smites] + LAEZEL_CRIT_DMG_MOD
  }
  result: DMG * LAEZEL_ATTACKS
}

output [astarion d20 damage roll WITH savage attacker] named "Astarion with SA (6 attacks)"
\output [astarion d20 damage roll WITHOUT savage attacker] named "Astarion without SA"\
loop N over {1..4} {
  \output [laezel d20 damage roll WITH savage attacker and level N smites] named "Laezel with SA (level [N] smites, 4 attacks)"\
  output [laezel d20 damage roll WITH savage attacker and level N smites] named "Laezel without SA (level [N] smites, 4 attacks)"
}

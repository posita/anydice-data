\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC+10 { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\

DMGBONUS: 0

ACCBONUS: -2
ATTACKTIMES: 1
DAMAGE: 2d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2*DAMAGE

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: bonusattackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: BONUSCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: BONUSDAMAGE }
 result: 0
}
\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 2
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 2d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d6 + ATTACKMOD + DMGBONUS
BONUSDAMAGE: 2d6 + DMGBONUS
BONUSCRITDAMAGE: 4d6 + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] + ATTACKTIMESd[attackroll d20 vs ENEMYAC] + ATTACKTIMESd[bonusattackroll d20 vs ENEMYAC] named "67 No Spells, 2d6 gun with bonus attack"

ACCBONUS: 4
DAMAGE: 2d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] + ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "67 No Spells, Sharpshooter, 2d8 Gun no bonus attack"

function: crit DIE:d {
 MAX: [maximum of DIE]
 result: [explode DIE helper]
}

function: explode N:n helper {
 if N = 8 & N = MAX { result: N + 1d4 }
 if N = 10 & N = MAX { result: N + 1d6 } 
 if N = 12 & N = MAX { result: N + 1d8 }
 result: N
}

output [crit 1d8]
output [crit 1d10]
output [crit 1d12]

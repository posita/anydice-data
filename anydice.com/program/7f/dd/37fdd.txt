LOOP: {1..5}

SUCCESSES: {6} \ change to { 6 } for high pressure or {4, 5, 6} for low pressure \
RRALLOWED: {2, 3, 4, 5} \ which dice can be rerolled, 0 for "none", {1} for training or e.g. {1, 2, 3, 4} for expert \
MAXREROLLS: 5 \ set to how many dice can be rerolled... set to the max LOOP value or some high number if "all" is desired \

function: pool DSP:s {
 S: [count SUCCESSES in DSP]
 REROLLS: [lowest of MAXREROLLS and [count RRALLOWED in DSP]]
 S: S + [count SUCCESSES in REROLLSd6]
 result: S
}

loop N over LOOP {
  output [pool Nd6] named "[N]d6"
}

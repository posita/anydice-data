LOOP: 0
TOTALDAMAGE: 0
ATTACK: 21
TARGETAC: 35
CRITRANGE: 4

function: confirm ROLL:n mod ATTACK target AC damage DAMAGE extra EXTRA {
 if ROLL > 1 & ROLL+ATTACK >= AC {
  result: DAMAGE + DAMAGE + EXTRA
 } else {
  result: DAMAGE + EXTRA
 }
}

function: roll ROLL:n mod ATTACK target AC damage DAMAGE extra EXTRA {
 if ROLL >= 21 - CRITRANGE {
  result: [confirm 1d20 mod ATTACK target AC damage DAMAGE extra EXTRA]
 } else if ROLL > 1 & ROLL + ATTACK >= AC {
  result: DAMAGE + EXTRA
 }
 else
 {
  result: 0
 }
}

loop LOOP over {1..4}
{
 if LOOP > 2 {
  DAMAGE: [roll 1d20 mod ATTACK - ((LOOP - 2) * 5) target TARGETAC damage 1d6+14 extra 2d6+5]
 } else {
  DAMAGE: [roll 1d20 mod ATTACK target TARGETAC damage 1d6+14 extra 2d6+5]
 }
 TOTALDAMAGE: TOTALDAMAGE + DAMAGE
 output DAMAGE named "Dueling Sword #[LOOP]"
}

loop LOOP over {1..4}
{
 DAMAGE: [roll 1d20 mod ATTACK target TARGETAC damage 1d6+9 extra 1d6]
 TOTALDAMAGE: TOTALDAMAGE + DAMAGE
 output DAMAGE named "Storm of Swords #[LOOP]"
}

DAMAGE: [roll 1d20 mod ATTACK target TARGETAC damage 1d6+14 extra 2d6+5]
TOTALDAMAGE: TOTALDAMAGE + DAMAGE
output DAMAGE named "Mirror Strike"

output TOTALDAMAGE named "Total Damage"

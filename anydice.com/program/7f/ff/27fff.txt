set "maximum function depth" to 4

function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 4
 if ROLL = 6 { result: SUCCESSES + [evaluate 1d6]}
 if SUCCESSES > 0 { result: SUCCESSES}
 result: 0
}

function: nodice ROLL:s {
 SUCCESSES: 0
 if ROLL = 6 { result: SUCCESSES + [evaluate 1d6]}
 if SUCCESSES > 0 { result: SUCCESSES}
 result: 0
}

output [evaluate 2@2d6] named "0d6 take lowest"
loop DICE over {1..6} {
 output [evaluate DICE d6] named "[DICE]d6"
}

DICEPOOL: {4,6,8,10,12,20}
function: explodenth DIE:d {
 MAX: [maximum of DIE]
 result: [explodenth DIE helper]
}

function: explodenth N:n helper {
if MAX != 20 {
	loop F over {1..#(DICEPOOL)}{
 		if MAX = F@DICEPOOL{ 
			B: (F+1)@DICEPOOL
			
		}
	}
}
else { B : MAX }
 if N = MAX { result: N + [explodenth 1dB] }
 result: N
}
set "maximum function depth" to 15
output [explodenth 1d4] named "explode 1"

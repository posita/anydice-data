\******** DEFINITIONS ********\
ARCHERY_STYLE: 2
SHARPSHOOTER_ATTACK_PENALTY: 5
SHARPSHOOTER_DMG_BONUS: 10

\******** ENEMY ********\
AC: d{10..20}

\******** ATTACK FUNCTIONS ********\
function: sharpshooter at level LEVEL:n with DEX:n dex for DMG:d crit CRIT_DMG:d roll ROLL:n against AC:n{
    ATTACK_MODIFIER: (DEX - 10)/2
    PROFICIENCY: ((LEVEL - 1)/4) + 2
    DMG_WITH_BONUS: DMG + ATTACK_MODIFIER + SHARPSHOOTER_DMG_BONUS
	
    if ROLL = 1 {result:0}
    if ROLL = 20 { result: DMG_WITH_BONUS + CRIT_DMG }
    if ROLL + ATTACK_MODIFIER + ARCHERY_STYLE >= AC { result: DMG_WITH_BONUS }

    result: 0
}

function: beast attack plus ATTACK_MODIFIER:n to hit for DMG:d crit CRIT_DMG:d roll ROLL:n against AC:n{
    if ROLL = 1 {result:0}
    if ROLL = 20 { result: DMG + CRIT_DMG }
    if ROLL + ATTACK_MODIFIER >= AC { result: DMG}

    result: 0
}


output 
 2d[sharpshooter at level 8 with 20 dex for 1d8 crit 1d8 roll 1@3d20 against AC] 
 named "Sharpshooter Samurai w/ Fighting Spirit, Elven Accuracy, 20 Dex"

output 8d[beast attack plus 4 to hit for 1d8+2 crit 1d8 roll d20 against AC] named "8 Constrictor Snakes (No Restraint)"

output 7d[beast attack plus 4 to hit for 1d8+2 crit 1d8 roll d20 against AC] +
    [beast attack plus 5 to hit for 3d6+3 crit 1d8 roll 1@2d20 against AC]  named "8 Constrictor Snakes (Restrained After Seventh)"

output 6d[beast attack plus 4 to hit for 1d8+2 crit 1d8 roll d20 against AC] +
    2d[beast attack plus 5 to hit for 1d8+2 crit 1d8 roll 1@2d20 against AC]  named "8 Constrictor Snakes (Restrained After Sixth)"

output 5d[beast attack plus 4 to hit for 1d8+2 crit 1d8 roll d20 against AC] +
    3d[beast attack plus 5 to hit for 1d8+2 crit 1d8 roll 1@2d20 against AC]  named "8 Constrictor Snakes (Restrained After Fifth)"

output 4d[sharpshooter at level 8 with 20 dex for 1d8 crit 1d8 roll 1@3d20 against AC] 
    named "Action Surging Sharpshooter Samurai w/ Fighting Spirit, Elven Accuracy, 20 Dex"

output 4d[beast attack plus 4 to hit for 1d8+2 crit 1d8 roll d20 against AC] +
    4d[beast attack plus 5 to hit for 1d8+2 crit 1d8 roll 1@2d20 against AC]  named "8 Constrictor Snakes (Restrained After Fourth)"

output 3d[beast attack plus 4 to hit for 1d8+2 crit 1d8 roll d20 against AC] +
    5d[beast attack plus 5 to hit for 1d8+2 crit 1d8 roll 1@2d20 against AC]  named "8 Constrictor Snakes (Restrained After Third)"

output 2d[beast attack plus 4 to hit for 1d8+2 crit 1d8 roll d20 against AC] +
    6d[beast attack plus 5 to hit for 1d8+2 crit 1d8 roll 1@2d20 against AC]  named "8 Constrictor Snakes (Restrained After Second)"

output [beast attack plus 4 to hit for 1d8+2 crit 1d8 roll d20 against AC] + 
     7d[beast attack plus 5 to hit for 1d8+2 crit 1d8 roll 1@2d20 against AC]  named "8 Constrictor Snakes (Restrained After First)"

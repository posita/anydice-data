function: attack N:n plus B:n dmg D:n plus DB:n vs AC:n crit C:n{
 if N>=C {result: D * 2 + DB}
 if N+B >= AC {result: D + DB}
 result: 0
}

loop AC over {8,9,10,12,14,16,18}{
output 2d[attack 1d20 plus 4 dmg 3d10 + 4d4 + 1d6 plus 14 vs AC crit 20] named "sharpshoot vs [AC]"
output 2d[attack 1d20 plus 9 dmg 3d10 + 4d4 + 1d6 plus 4 vs AC crit 20] named "normal vs [AC]"
}

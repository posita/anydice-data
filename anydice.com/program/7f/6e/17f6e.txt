\**** Damage calculator gwm barb ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL >= AC + 10 { result: CRITDAMAGE }
 if ROLL = 1 { result: 0 }
 if ROLL = AC - 10 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 17-17\
ENEMYAC: d{17..17}

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 1d6 * 2 + 1d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of 1d20] vs ENEMYAC] named "1st attack"

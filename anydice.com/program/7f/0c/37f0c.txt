function: attack DICE:n HIT_BONUS:n DC:n DAMAGE_BONUS:n DAMAGE:n CRIT_DAMAGE:n {
 if DICE = 20 | DICE + HIT_BONUS >= DC + 10 { result: CRIT_DAMAGE + DAMAGE_BONUS }
 if DICE + HIT_BONUS >= DC { result: DAMAGE + DAMAGE_BONUS }
 result: 0
}

DICE: d20
HIT_BONUS: 9
DC: 15
DAMAGE_BONUS: 4

output [attack DICE HIT_BONUS DC DAMAGE_BONUS 1d8 2*(1d8+4)+1d8] named "Glaive"
output [attack DICE HIT_BONUS DC DAMAGE_BONUS 1d10 2*(1d10+4)] named "Falchion"
output [attack DICE HIT_BONUS DC DAMAGE_BONUS 1d12 2*(1d12+4)] named "Greatsword"

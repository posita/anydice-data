function: swizzle S:s {
 if 2@S < 5 {result:1@S * 10 + 3@S}
result: 3@S * 10 + 1@S
}
function: swozzle S:s {
 if 2@S >= 5 {result:1@S * 10 + 3@S}
result: 3@S * 10 + 1@S
}
function: swag S:s N:n {
 if N >= 5 {result:1@S * 10 + 2@S}
result: 2@S * 10 + 1@S
}
output [swag 2d{0..9} 1d6]
output [swizzle 3d{0..9}]
output [swozzle 3d{0..9}]
output 1d10 * 1d10

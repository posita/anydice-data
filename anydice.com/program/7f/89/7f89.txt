\ v MODIFY THESE VALUES IF NEEDED v \
PLAYER_ONE: 0
PLAYER_TWO: 0
\ ^ MODIFY THESE VALUES IF NEEDED ^ \



function: compare A and B{
 if A = 10 + PLAYER_ONE & A > B + 1 {result: 1} \ 1 = Triumph \
 if                       A > B     {result: 2} \ 2 = Win \
 if                       A = B     {result: 3} \ 3 = Tie \
 if A = 1  + PLAYER_ONE & A < B - 1 {result: 5} \ 5 = Failure. We check this before Loss to make sure it isn't counted there. \
 if                       A < B     {result: 4} \ 4 = Loss \
}

TOTAL:   0
TRIUMPH: 0
WIN:     0
TIE:     0
LOSS:    0
FAILURE: 0

loop I over {1..10}{
 loop J over {1..10}{
  TOTAL: TOTAL + 1
  if [compare I+PLAYER_ONE and J+PLAYER_TWO] = 1 {TRIUMPH: TRIUMPH+1}
  if [compare I+PLAYER_ONE and J+PLAYER_TWO] = 2 {WIN: WIN+1}
  if [compare I+PLAYER_ONE and J+PLAYER_TWO] = 3 {TIE: TIE+1}
  if [compare I+PLAYER_ONE and J+PLAYER_TWO] = 4 {LOSS: LOSS+1}
  if [compare I+PLAYER_ONE and J+PLAYER_TWO] = 5 {FAILURE: FAILURE+1}
 }
}

output TOTAL   named "Total"
output TRIUMPH named "Triumph"
output WIN     named "Win"
output TIE     named "Tie"
output LOSS    named "Loss"
output FAILURE named "Failure"

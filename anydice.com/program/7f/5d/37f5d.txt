function: onering FEAT:s SUCCESS:s{
ONES: [count 1 in SUCCESS]
TWOS: [count 2 in SUCCESS]
THREES: [count 3 in SUCCESS]
S: SUCCESS - ONES - (TWOS*2) - (THREES * 3)
 if FEAT = 12 {TOTAL: 1000}
 else {
 if FEAT = 1 {TOTAL: S}
 else {TOTAL: FEAT -1 + S}
 }
 DIFF: 20 - TOTAL
 if DIFF < 2 {result: 0}
 else 
{
 if DIFF > 8 {result: 10}
}
 result : DIFF
}
loop N over {0..6}{
 output [onering [lowest 1 of 1d12] Nd6] named "[N] ranks in skill"
}

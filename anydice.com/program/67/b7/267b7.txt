function: noir DICE:d {
  NUMBEROF: #DICE
  HIGHESTDIE: 1@DICE
  LOWESTDIE: NUMBEROF@DICE
  DIETYPE: [maximum of HIGHESTDIE]

  result: [check HIGHESTDIE LOWESTDIE NUMBEROF 1dDIETYPE]
}

function: check HIGHEST:n LOWEST:n NUMBEROF:n DIETYPE:d {
  if LOWEST = 1 {
    if NUMBEROF = 1 {
      result: LOWEST
    } else {
      result: HIGHEST + LOWEST
    }
  }

  if LOWEST = 10 {
    if NUMBEROF = 1 {
      result: LOWEST + 1d[explode DIETYPE]
    } else {
      result: LOWEST + HIGHEST + 1d[explode DIETYPE] + 1d[explode DIETYPE]
    }
  }

  if HIGHEST = 10 {
    if NUMBEROF = 1 {
      result: HIGHEST + 1d[explode DIETYPE]
    } else {
      result: LOWEST + HIGHEST + 1d[explode DIETYPE]
    }
  }

  result: HIGHEST + LOWEST
}

output [noir 2d10]

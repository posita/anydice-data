function: best match in ROLL:s {
 BEST: 0
 MATCH: 0
 CURRENT: 0
 
 loop V over ROLL {
  if V = CURRENT {
   MATCH: MATCH + V
  }
  else {
   if MATCH > BEST {
    BEST: MATCH
   }
   MATCH: V
   CURRENT: V
  }
 }
 if MATCH > BEST {
  result: MATCH
 }
 result: BEST
}

function: ctech ROLL:s {
 result: [best match in ROLL]
}

output [ctech 3d8+2d12]

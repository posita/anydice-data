SOGLIA: 4
function: calcolo su A:s B:s C:s D:s E:s {
 SORTED: [sort {A, B, C, D, E}]
 R: 1@SORTED
 
 NEFFETTO: 0
 loop N over {2..6}{
  if N@SORTED >= 4 {NEFFETTO : NEFFETTO + 1}
 }
 NEFFETTO : NEFFETTO / 2
 if NEFFETTO > 0 {
  loop N over {2..1+NEFFETTO}{
   R : R + N@SORTED
  }
 }
 
 result: R
}

output [calcolo su 2d12 0 2d4 0 0]

output [calcolo su 2d12 0 2d10 0 0]
output [calcolo su 2d12 0 2d12 0 0]

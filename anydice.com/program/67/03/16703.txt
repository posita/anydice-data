set "maximum function depth" to 5
function: limit N:n {
 if N >= 15 {result: 15}
 if N <= 7 {result: 7}
 result: N
}
function: rolllimit N:n {
 if N <= 15 & N >= 7 {result: N}
 result: [rolllimit 3d{2..6}]
}
ABILITIES: 6 d [highest 3 of 4d6]
loop P over {1..6} {
 output P @ ABILITIES named "Ability [P]"
}

output [limit [highest 3 of 4d6]] named "4d6, drop 1, limit"
output [limit 3d6 + 2] named "3d6 + 2, limit"

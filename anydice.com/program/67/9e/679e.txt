function: burst N:n {
 if N = 12 { result: N + d6 }
 result: N
}

loop S over {1..17} {
loop C over {1..11} {
output H > S named "overall success [S]/[C]"
\output H > S & L > C named "success [S]/[C]"
output H > S & L <= C named "consequences [S]/[C]"
output H <= S named "failure [S]/[C]"\
}}

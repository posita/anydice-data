function: count V:s in MS:s and SS:s {
 COUNT: 0
 loop P over {1..#V} {
  COUNT: COUNT + (P@V = MS)
  COUNT: COUNT + (P@V = SS)
 }
 result: COUNT
}

MOD: 4
output [count {8,9,10,11,12} in d10 + MOD] named "1d10"

loop DICE over {1..5} {
  output [count {8,9,10,11,12} in DICE d10 and d10 + MOD] named "[DICE]d10"
}

function: roll ROLL:n threshold T {
  if ROLL >= T {result: 1}
  result: 0
}

function: t DICE:n threshold T {
  RES:0
  loop N over {1..DICE} {
    RES:RES+[roll 1d{20,10,9,87,6,5,4,3,2,1,0} threshold T]
  }
  result:RES
}

loop Q over {1..4} {
output [t Q threshold 5] named "[Q] t 5"
output [t Q threshold 4] named "[Q] t 4"
}

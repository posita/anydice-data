function: owod N:n tn TN:n {
 if N = 1 { result: -1 }
 if N < TN { result: 0 }
 result: 10 
}

function:  norm N:n {
 if N < 0 { result: -1 }
 if N < 10 { result: 0 }
 if N = (N / 10) * 10 { result: N/10 }
 else { result: [highest of (N/10 + 1 - ((N/10 + 1) * 10 - N)) and 0] }
}

OWOD: [owod d10 tn 6]

loop N over {1..15} {
 output [norm NdOWOD] named "[N]d"
}

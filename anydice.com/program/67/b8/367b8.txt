function: marvel ONE:n TWO:n THREE:n plus BONUS:n vs TARGET:n {
  if TWO = 1 {
    if ONE + 6 + THREE + BONUS >= TARGET {
      result: 2
    }
    result: -1
  }
  if ONE + TWO + THREE + BONUS >= TARGET {
    result: 1
  }
  result: 0
}

loop B over {0..8} {
  loop T over {10..18} {
    output [marvel 1d6 1d6 1d6 plus B vs T] named "+[B] vs [T]"
  }
}

FAIL: 0
PARTIAL: 1
SUCCESS: 2

function: successtype ROLL:n {
 if ROLL < 7 { result: FAIL }
 if ROLL < 10 { result: PARTIAL }
 result: SUCCESS
}

loop DIE_FIRST over {4,6,8,10,12}{
  loop DIE_SECOND over {4,6,8,10,12}{
    if DIE_FIRST <= DIE_SECOND {
      output [successtype 1dDIE_FIRST + 1dDIE_SECOND] named "d[DIE_FIRST]+d[DIE_SECOND]"
    }
  }
}

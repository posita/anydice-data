
function: simpleroll B:n with TAW:n {
  if (B <= TAW) {
    result: TAW - B
  }
  result: -1
}
function: kroll A:n second B:n with TAW:n {
  if A = 1 {
    if [simpleroll B with TAW] >= 0 {
      result: 100
    }
  }
  if A = 20 {
    if [simpleroll B with TAW] < 0 {
      result: -100
    }
  }
  result: [simpleroll A with TAW]
}
output [kroll d20 second d20 with 20]

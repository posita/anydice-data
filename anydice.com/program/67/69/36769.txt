function: astorm with ACCROLL:n mr MISSRATE:n cr CRITRATE:n {
  if ACCROLL >= (20 - CRITRATE) {result: 3d8+2}
  if ACCROLL >= MISSRATE { result: 1d8+2 }
  result: 0  
}
X: 0
HAWKEYE: 0
loop I over {1..4} {
  X: X + [astorm with (1d20+(HAWKEYE*2)) mr 4 cr (2+(HAWKEYE*2))]
}

output X + 16 - (HAWKEYE*5)

function: owod N:n tn TN:n {
 if N = 1 { result: -1 }
 if N < TN { result: 0 }
 result: 100 
}

function:  norm N:n {
 if N < 0 { result: -1 }
 if N < 100 { result: 0 }
 if N = (N / 100) * 100 { result: N/100 }
 else { result: [highest of (N/100 + 1 - ((N/100 + 1) * 100 - N)) and 0] }
}

OWOD: [owod d10 tn 6]

loop N over {1..15} {
 output [norm NdOWOD] named "[N]d"
}

MISS: 0
HIT: 1
CRIT: 2

function: attack ROLL:n vs AC:n {
 if ROLL >= 20 { result: CRIT }
 if ROLL = 1 { result: MISS }
 if ROLL + TOHIT >= AC {result: HIT}
 result: MISS
}

function: damage for ATKRSLT:n {
 if ATKRSLT = CRIT { result: CRITDMG }
 if ATKRSLT = HIT { result: HITDMG }
 if ATKRSLT = MISS { result: MISSDMG }
}

function: swing {
 result: [damage for [attack ([highest 1 of 1d20]) vs AC]]
}

function: save ROLL:n {
 if ROLL + SAVEMOD >= 8+TOHIT { result: SAVEEFFECT*TARGETS }
 result: EFFECT*TARGETS
}

TOHIT: 3\INT18\ + 2\PROF\

AC: d{10..16}

HITDMG: 3d4+3
CRITDMG: HITDMG
MISSDMG: HITDMG
output [swing] named "Magic Missile"

HITDMG: 3d8
CRITDMG: 6d8
MISSDMG: 0
output [swing] named "Chromatic Orb"

loop TARGETS over {1..3} {
SAVEMOD: {-1..3}

EFFECT: 2d6
SAVEEFFECT: EFFECT/2
HITDMG: 1d10
CRITDMG: 2d10
MISSDMG: 0
output [swing] + [save d20] named "Ice Knife"

EFFECT: 3d6
SAVEEFFECT: EFFECT/2
output [save d20] named "Burning Hands"
}

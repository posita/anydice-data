function: fuhc P:s pos N:s neg {
  \ for each positive die in order from highest to lowest... \
  loop PD over {1..#P} {
    \ check to see if the dice was rolled by the negative dice \
    if [N contains PD@P] = 0 {
      \ if it wasn't, this is the result of the roll. Else, move on to the next die \
      result: PD@P
    }
  }
  \ if all dice have been cancelled out... \
  result: 1
}

output [fuhc 1d6 pos 1d6 neg]

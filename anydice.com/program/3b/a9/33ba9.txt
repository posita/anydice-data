RED: {0,d{0,1},1,d{1,2},d{1,2},2}
BLACK: {d{0,1},d{1,2},d{1,2},2,d{2,3},d{2,3,3,4}}
function: rolls A:n plus B:n power P:d bonus Q:s tohit H:n{
  RESULT: 1d0
  ROLL: d10 + B >= H
  loop X over {1..A}{
    if X=1 {
      RESULT: RESULT + ROLLd(P + Q)
    }
    else {
      RESULT: RESULT + ROLLdP
    }
  }
  
  result: RESULT

}

output [rolls 2 plus 0 power 1dBLACK bonus 1dRED tohit 8]
output [rolls 1 plus 0 power 2dBLACK bonus 1dRED tohit 8]

\ Difficulty/Armor Class \
DC: 10
\ Modifier \
MOD: 0

function: X:n conditional
{
 if X >= DC {result: 1}
 else {result: 2}
}

function: X:n conditionalcrit
{
 if X - MOD = 1 {result: 3}
 if X - MOD = 20 {result: 0}
 if X >= DC {result: 1}
 else {result: 2}
}

output [1d20 + MOD conditional] named "regular"
output [[highest 1 of 2d20] + MOD conditional] named "advantage"
output [[lowest 1 of 2d20] + MOD conditional] named "disadvantage"
output [1d20 + MOD conditionalcrit] named "regular - crit"
output [[highest 1 of 2d20] + MOD conditionalcrit] named "advantage - crit"
output [[lowest 1 of 2d20] + MOD conditionalcrit] named "disadvantage - crit"

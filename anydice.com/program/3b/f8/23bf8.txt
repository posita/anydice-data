function: action ROLL:s vs DIFFICULTY:n add RISK:s {
  if ROLL >= DIFFICULTY {result: 1 + (ROLL >= DIFFICULTY + RISK)}
  else {result: 0}
}

loop MOD over {0, 2, 4, 6, 8}{
 loop T over {4, 6, 8, 10}{
  loop DIFFICULTY over {3, 6, 9, 12}{
    output [action (1d12 + MOD) vs DIFFICULTY add 1dT] named "+[MOD] vs [DIFFICULTY] + d[T]"
}}}

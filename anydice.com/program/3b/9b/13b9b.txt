function: apoc A:n {
   if A >= 10 { result: 10 }
   if A >= 7 { result: 7 }
   result: 0
}

function: napoc A:n {
   if A >= 15 { result: 15 }
   if A > 10 { result: 11 }
   result: 0
}

function: xapoc A:n plus B:n {
   X: A + B
   if X >= 15 { result: 15 }
   if X > 10 { result: 11 }
   result: 0
}

output [apoc 2d6] named "2d6"
output [napoc 3d6] named "3d6"
output [xapoc 1d6 plus 2d4] named "1d6 + 2d4"
output [xapoc 2d6 plus 1d4] named "2d6 + 1d4"

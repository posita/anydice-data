\**** EK vs. Full Caster Save Calculations ****\


function: castersave ROLL:n {
if ROLL + SAVEMOD >= CASTERSAVEDC { result: 1 }
if ROLL + SAVEMOD < CASTERSAVEDC { result: 0 }
result: 0
}


function: eksave ROLL:n {
if ROLL + SAVEMOD >= EKSAVEDC { result: 1 }
if ROLL + SAVEMOD < EKSAVEDC { result: 0 }
result: 0
}

SAVEMOD: 6

CASTERSAVEDC: 18
EKSAVEDC: 14

output 1d[castersave 1d20] named "Full Caster"
output 1d[eksave [lowest 1 of 2d20]] named "Eldritch Knight"

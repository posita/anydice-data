function: health on R:n under N:n die X {
    if R < N {result: [health on X under N die X]}
    else {result: R}
}

function: healthall X:n times on R under N:n with endurance Y:n {
    TOTAL: 0
    N: N + Y
    loop P over {1..X} {
        ATTACK: [health on R under N die R] + Y
        TOTAL: TOTAL + ATTACK 
    }
    result: TOTAL
}

set "maximum function depth" to 3

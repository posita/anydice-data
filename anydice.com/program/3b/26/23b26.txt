function: health on R:n under N:n {
    if R < N {result: [health on R under N]}
    else {result: R}
}

function: healthall X:n times on R under N:n {
    TOTAL: 0
    loop P over {1..X} {
        ATTACK: [health on R under N]
        TOTAL: TOTAL + ATTACK 
    }
    result: TOTAL
}

\Magic Missile Damage\
MM_LEVEL: 6
MM_DICE: 2+MM_LEVEL
MM_BONUS: MM_DICE*6
output MM_DICEd4 + MM_BONUS named "Level [MM_LEVEL] Magic Missile [MM_DICE]d4+[MM_BONUS]"

\Fireball Damage\
FIREBALL_LEVEL: 6
if FIREBALL_LEVEL < 3 {
   output 0 named "Fireball is a minimum 3rd level spell"
}
else {
   FIREBALL_DICE: 5+FIREBALL_LEVEL
   TARGETS: 2
   output (FIREBALL_DICEd6 + 5) * TARGETS named "Level [FIREBALL_LEVEL] Fireball [FIREBALL_DICE]d6+5 against [TARGETS] targets"
}

\Overchannel Damage\
OC_LEVEL: 5
OC_USES: 1

if OC_USES <= 0 {
   OC_DICE: 0
}
else if OC_USES > 0 {
   OC_DICE: (2*OC_LEVEL)+((OC_USES-1)*OC_LEVEL)
}

SHIELD_GUARDIAN: 0
if SHIELD_GUARDIAN = 1 {
   output OC_DICEd12/2 named "Overchannel Spell Level [OC_LEVEL] Uses [OC_USES] [OC_DICE]d12 Split With Shield Guardian"
}
else if SHIELD_GUARDIAN = 0 {
   output OC_DICEd12 named "Overchannel Spell Level [OC_LEVEL] Uses [OC_USES] [OC_DICE]d12"
}

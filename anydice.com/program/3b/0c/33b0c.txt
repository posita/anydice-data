CRIT_ENABLED: 1

function: tier X:n {
  if X < 9 {
    result: -1
  } else if X < 14 {
    result: 0
  } else if X < 20 | !CRIT_ENABLED {
    result: 1
  } else {
    result: 2
  }
}

\MOD: 0\

loop ADV over {-1..1} {
 loop MOD over {-7, -3, 0, 3, 7} {
  if ADV >= 0 {
   output [tier [highest 2 of ([absolute ADV]+2)d10]+MOD] named "A [ADV], M [MOD]"
  } else {
   output [tier [lowest 2 of ([absolute ADV]+2)d10]+MOD] named "A [ADV], M [MOD]"
  }
 }
}

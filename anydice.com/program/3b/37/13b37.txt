function: clashduel MOD:n ONE:n TWO:n {
 \Crits\
 if ONE = 20 & TWO-MOD = 20{
  result: 0
 }
 if ONE = 20 & TWO-MOD != 20{
  result: 2
 }
 if ONE != 20 & TWO-MOD = 20{
  result: -2
 }

 \Basics\
 if ONE > TWO{
  result: 1
 }
 if ONE < TWO{
  result: -1
 }
 if ONE = TWO{
  result: 0
 }

}

loop MOD over {0,5,10,15,20}{
 output [clashduel MOD (d20) (d20+MOD)] named "d20+[MOD]"
}

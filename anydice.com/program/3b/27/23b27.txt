function: health on R:n under N:n die X {
    if R < N {result: [health on X under N die X]}
    else {result: R}
}

function: healthall X:n times on R under N:n {
    TOTAL: 0
    loop P over {1..X} {
        ATTACK: [health on R under N die R]
        TOTAL: TOTAL + ATTACK 
    }
    result: TOTAL
}

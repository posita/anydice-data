loop DIE over [6, 10] {
    loop MOD over [-5..5] {
        if MOD > 0 {
            output 2d[DIE] + MOD named "2d[DIE] + [MOD]"
        } else if MOD < 0 {
            output 2d[DIE] + MOD named "2d[DIE] - [absolute MOD]"
        } else {
            output 2d[DIE] + MOD named "2d[DIE]"
        }
    }
}

FOR: 9
function: damage DAMAGE:n crit CRIT:n mult MULT:n{
 result: [roll d20 damage DAMAGE crit CRIT mult MULT]
}

function: roll ROLL:n damage DAMAGE:n crit CRIT:n mult MULT:n{
 if(ROLL >= CRIT){
  result: (DAMAGE+FOR)*MULT
 }
 result: DAMAGE+FOR
}

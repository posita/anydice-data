function: recursive X:n depth Y with Z again and MINIMUM min {
  if X>=Z {
    result: [recursive d10 depth Y+1 with Z again and MINIMUM min]
  }
  if X>=MINIMUM {
    result: Y+1
  }
  result: Y
}

function: storytelling with AGAIN again and MINIMUM min {
  result: [recursive d10 depth 0 with AGAIN again and MINIMUM min]
}

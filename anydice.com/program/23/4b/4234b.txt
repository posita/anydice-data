BASEDIE: d6

set "maximum function depth" to 3

function: target TARGET:n reroll REROLL:n VALUE:n {
  if VALUE = REROLL {
    result: 1 + [target TARGET reroll REROLL BASEDIE]
  }
  if VALUE  > TARGET {
    result: 1
  }
  result: 0
}

DIE: [target 3 reroll 6 BASEDIE]

loop N over {1..10}{
  output NdDIE named "[N]d6"
}

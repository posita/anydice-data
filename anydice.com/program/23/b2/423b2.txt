function: compare R:n T:n {
  if R >= T {
    result: 1
  }
  result: 0
}

function: roll ATTACKS:n TARGET:n {
  ROLL: ATTACKSd6
  DAMAGE: 0
  loop I over {1..ATTACKS} {
    DAMAGE: DAMAGE + [compare I@ROLL TARGET]
  }
  result: DAMAGE
}

loop I over {1..5} {
  output [roll I 3] named "[I] 3+"
}

loop I over {1..5} {
  output [roll I 5] named "[I] 5+"
}

function: sucessos A:s{
 result:(1@A>=4)+(1@A=6)+(2@A=6)
}
function: deducao A:s{
 result:[count {7,8} in A]
}
function: A:s thorns B:n
{
 C:[sucessos A]-B
 if C>=3{result:3}
 if C<=-1{result:-1}
 result:C
}

 TT:[deducao Td8]
 loop X over {1..7}{
  output [Xd6 thorns TT] named "[X]d6 com [T] thorns"
 }

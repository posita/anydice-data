\ Attack \
AZERO: 1d6
AONE: 1@2d6
ATWO: [highest of 1@2d6 and 1d8]
\ Defense \
DZERO: 1d6
DONE: 1d6
DTWO: 1@2d6
\ Damage \
set "maximum function depth" to 15
function: DIE:n at least NUM:n {
 if DIE < NUM {
  result: 0
 }
 result:1
}

function: ATT:n against DEF:n DAM:n  {
 DAMAGE: ATT - DEF
 HURT: 0
 if DAMAGE > 0 { 
  loop N over {1..DAMAGE} {
   HURT: HURT + [1d6 at least DAM]
  }
 }
 result: HURT
}

output [ATWO against DZERO 5] named "2d6+1d8R vs 1d6W, at least 5"
output [ATWO against DTWO 4] named "2d6+1d8R vs 2d6B, at least 4"

function: attack ATTACK:d {
  result: ATTACK
}

function: reroll R:n D:d N:n{ 
  if R < N { result: D } else {result: R}
}

function: multiattackreroll X:n Y:d M:n{
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: [reroll Y 3 Y] + M
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

function: multiattack X:n Y:d M:n{
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: [attack Y] + M
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

MULTIATTACK: 3
ATTACK: 1d8
MODIFIER: 4+2

MULTIATTACKB: 3
ATTACKB: 1d6
MODIFIERB: 4+2

output [multiattackreroll MULTIATTACK ATTACK MODIFIER] + [multiattack MULTIATTACKB ATTACKB MODIFIERB] named "Barb2Monk5"

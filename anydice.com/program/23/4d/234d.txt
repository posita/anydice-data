function: swap random from BONUS:s and PENALTY:s with RB:n and RP:n {
if RB@BONUS < RP@PENALTY {
result: BONUS-PENALTY + 2*RP@PENALTY - 2*RB@BONUS
}
else {
result: BONUS-PENALTY
}
}

function: swap SWAPA:s and SWAPB:s plus BONUS:s minus PENALTY:s {
if (SWAPA > SWAPB) {
result: SWAPA + BONUS - SWAPB - PENALTY
}
else {
result: SWAPB + BONUS - SWAPA - PENALTY
}
}

function: swap totals between BONUS:n and PENALTY:n {
if PENALTY > BONUS {
result: PENALTY-BONUS
}
else {
result: BONUS-PENALTY
}
}

function: swap BONUSA:n with PENALTYA:n and BONUSB:n with PENALTYB:n {
if (BONUSA > PENALTYA) {
TOTAL: BONUSA - PENALTYA
}
else {
TOTAL: PENALTYA - BONUSA
}
if (BONUSB > PENALTYB) {
result: TOTAL + BONUSB - PENALTYB
}
else {
result: TOTAL + PENALTYB - BONUSB
}
}

function: highest minus lowest of ROLLS:s {
result: {1,2}@ROLLS - {3,4}@ROLLS
}

function: highestsum A:s a B:s {
 result: {1,2,3,4} @ [sort {A, B}]
}

function: kosc I:n in SEQ:s to N:n {
  NEW: {}
  loop J over {1 .. #SEQ} {
    if J@SEQ = I { NEW: {NEW, N} }
    else { NEW: {NEW, J@SEQ} }
  }
  result: NEW
}

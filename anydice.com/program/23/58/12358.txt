AC: 22
DEX: 3
THEODC: 16

function: tohit A:n plus M with D plus DB {
 if A = 20 {
  result: D + D + DB
 }else if A = 1 {
  result: 0
 }else if A+M >= AC {
  result: D + DB
 }else {
  result: 0
 }
}

function: eldritch blast {
 result: 2d[tohit [highest 1 of 3d20] plus 8 with d10 plus 5]
}

function: spiritual weapon {
 result: [tohit [highest 1 of 3d20] plus 8 with 2d8 plus 5]
}

function: varis {
 result: [eldritch blast] + [spiritual weapon]
}

function: yorrn attacks {
 result: [tohit [highest 1 of 2d20] plus 8 with 3d8 plus 7] +
         [tohit [highest 1 of 2d20] plus 8 with 3d8 plus 7] +
         [tohit [highest 1 of 2d20] plus 8 with 1d8 plus 2] + 1d6
}

function: save S:n with D{
 if S + DEX >= THEODC{
  result: 0
 }else {
  result: D
 }
}

function: theo L{
 result: [theo catapult L]  + [tohit [highest 1 of 2d20] plus 5 with 2d6 plus 3]
}

function: theo catapult L {
 result: [save [lowest 1 of 2d20] with (L+2)d8]
}

output [theo 4] + [varis] + [yorrn attacks] named "All Out"

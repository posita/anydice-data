function: adv SEQUENCE:s {
 VALUES: {4,5,6}
 COUNT: 0
 ONES: 1 = SEQUENCE
 COUNT: COUNT + [count VALUES in ONESd6]
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: COUNT
}

function: disadv SEQUENCE:s {
 VALUES: {4,5,6}
 COUNT: 0
 SIXES: 6 = SEQUENCE
 COUNT: COUNT + [count VALUES in SIXESd6] - SIXES
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: COUNT
}

output [disadv 6d6] named "disADV"
output [count {4,5,6} in 6d6] named "Normal"
output [adv 6d6] named "ADV"

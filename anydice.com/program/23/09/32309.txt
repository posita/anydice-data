function: draw DRAW:n lands left LL:n cards left CL:n card drawn CD:n muligans MUL:n lands in hand LH:n {
HAND:LH

DRAW:DRAW-1
CL:CL-1
if (CD<=LL) {
LL:LL-1
HAND:HAND+1
}

if (DRAW>0) {HAND: [draw DRAW lands left LL cards left CL card drawn 1dCL muligans MUL lands in hand HAND]}


\if (MUL>0) {HAND: [draw DRAW lands left LL cards left CL card drawn 0 muligans MUL-1 lands in hand HAND]}\
result: HAND
}

DRAW: 1
LANDSINDECK: 1
CARDSINDECK: 1
MULLIGANS: 0

output [draw DRAW lands left LANDSINDECK cards left CARDSINDECK card drawn 1dCARDSINDECK muligans MULLIGANS lands in hand 0]

function: random BITS:n bit number {
T:0
loop N over {1..BITS}{
T:T*10
T:T+(1d2>1)
}
result: T
}
function: bit to integer BITS:n {
result: #BITS
}
output [random 8 bit number]
output [bit to integer [random 8 bit number]]

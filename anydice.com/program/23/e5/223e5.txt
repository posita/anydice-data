function: crop A:n{
 if A=0 {result: 100}
 if A>100 {result: 101}
result: A}

loop N over {0..3}{
 output [crop 10*[highest 1 of (N+1)d{0..9}]+d{0..9}] named "Normal with [N] penalty"
 output [crop 2*(10*[highest 1 of (N+1)d{0..9}]+d{0..9})] named "hard with [N] penalty"
 output [crop 5*(10*[highest 1 of (N+1)d{0..9}]+d{0..9})] named "extreme with [N] penalty"
}

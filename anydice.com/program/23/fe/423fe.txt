\ --- DYNAMIC SCALING --- \
function: get defense {
  if LEVEL < 6 { result: 14 }
  if LEVEL < 11 { result: 15 }
  if LEVEL < 16 { result: 17 }
  result: 17
}
function: get deflect {
  if LEVEL < 6 { result: 1 }
  if LEVEL < 11 { result: 1 }
  if LEVEL < 16 { result: 2 }
  result: 4
}
DEFENSE: [get defense]
DEFLECT: [get deflect]

\ --- CONSTANTS & LOGIC --- \
PMISS: 1 PHIT: 2 PCRIT: 3
MISS: 400 HIT: 500 CRIT: 600
MISSCOMP: 401 HITCOMP: 501 CRITCOMP: 601
MISSOP: 402 HITOP: 502 CRITOP: 602

function: floor ROLL:n MIN:n { 
  result: [highest of ROLL and MIN] 
}

function: reroll VAL:n {
  if VAL < 5 { result: 1d6 }
  result: VAL
}

function: attack ROLL:n plus MOD:n vs DF:n {
  if ROLL = 20 { result: PCRIT }
  if ROLL + MOD >= DF { result: PHIT }
  result: PMISS
}

function: attackstakes ROLL:n plus MOD:n plus OPP:n vs DF:n {
  if ROLL = 20 { result: 300 + OPP }
  if ROLL + MOD + OPP >= DF { result: 200 + OPP }
  result: 100 + OPP
}

function: stakes BASE:n {
  if BASE >= 300 { if BASE >= 305 { result: CRITOP } if BASE >= 302 { result: CRITCOMP } result: CRIT }
  if BASE >= 200 { if BASE >= 205 { result: HITOP } if BASE >= 202 { result: HITCOMP } result: HIT }
  if BASE >= 105 { result: MISSOP } if BASE >= 102 { result: MISSCOMP } result: MISS
}

function: gaindice RESULT:n {
  if RESULT = PCRIT { result: [highest 1 of 3d20] }
  if RESULT = PHIT { result: [highest 1 of 2d20] }
  result: 1d20
}

\ --- DAMAGE FUNCTIONS --- \

function: damagestakes STATUS:n for DIE:d plus MOD:n plus EXTRA:n USEDEF:n MIN:n {
  SUB: DEFLECT * USEDEF
  F: [floor DIE MIN]
  if STATUS >= 600 { result: [highest of 0 and ([maximum of DIE] + MOD + EXTRA - SUB)] } 
  if STATUS = HITOP { result: [highest of 0 and ([maximum of DIE] + MOD + EXTRA - SUB)] }
  if STATUS >= 500 { result: [highest of 0 and (F + MOD + EXTRA - SUB)] } 
  if STATUS = MISSOP { result: [highest of 0 and ([maximum of DIE] - SUB)] }
  result: 0
}

function: grazestakes STATUS:n for DIE:d plus MOD:n plus EXTRA:n USEDEF:n MIN:n {
  SUB: DEFLECT * USEDEF
  F: [floor DIE MIN]
  if STATUS >= 600 { result: [highest of 0 and ([maximum of DIE] + MOD + EXTRA - SUB)] }
  if STATUS = HITOP { result: [highest of 0 and ([maximum of DIE] + MOD + EXTRA - SUB)] }
  if STATUS >= 500 { result: [highest of 0 and (F + MOD + EXTRA - SUB)] }
  if STATUS = MISSOP { result: [highest of 0 and ([maximum of DIE] - SUB)] }
  result: [highest of 0 and (F - SUB)] 
}

function: damagebase STATUS:n for DIE:d plus MOD:n plus EXTRA:n USEDEF:n MIN:n {
  SUB: DEFLECT * USEDEF
  F: [floor DIE MIN]
  if STATUS = PCRIT { result: [highest of 0 and ([maximum of DIE] + MOD + EXTRA - SUB)] }
  if STATUS = PHIT { result: [highest of 0 and (F + MOD + EXTRA - SUB)] }
  result: 0
}

function: grazebase STATUS:n for DIE:d plus MOD:n plus EXTRA:n USEDEF:n MIN:n {
  SUB: DEFLECT * USEDEF
  F: [floor DIE MIN]
  if STATUS = PCRIT { result: [highest of 0 and ([maximum of DIE] + MOD + EXTRA - SUB)] }
  if STATUS = PHIT { result: [highest of 0 and (F + MOD + EXTRA - SUB)] }
  result: [highest of 0 and (F - SUB)]
}

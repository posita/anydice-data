\**** Bobby Bones Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: d20 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC + 5 { result: DUELISTDAMAGE }
 result: 0
}

\looped against enemy AC from 12-14\
ENEMYAC: d{12..14}

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 19
ATTACKTIMES: 1
DAMAGE: 2d6 + DMGBONUS
DUELISTDAMAGE: 3d6 + DMGBONUS
CRITDAMAGE: 4d6 + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Revolver"

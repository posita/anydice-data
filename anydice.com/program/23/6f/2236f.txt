set "position order" to "lowest first"

function: reroll lowest A:n of B:d if below C:n {
  DIESIZE: [maximum of 1@B]
  if C = 0 { C: (DIESIZE / 2) + 1 }  
  DIE: 1dDIESIZE
  LOWEST: 1@B
  LOCKED: {2..#(B)}@B
  LOWEST: [reroll DIE of LOWEST if below C]
  result: LOWEST + LOCKED
}

function: reroll A:d of B:n if below C:n {
  if B < C { result: A }
  result: B
}

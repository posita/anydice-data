\**** Damage calculator ****\

function: multiattack ATTACKS:n DICE:n {
 TOTAL: 0
 loop N over {1..ATTACKS}{
  ADVANTAGE: [highest 1 of DICEd20]
  TOTAL: TOTAL + [attackroll ADVANTAGE vs ENEMYAC]
 }
 result: TOTAL
}

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 12-20\
ENEMYAC: d{12..20}


ATTACKMOD: 5
DMGBONUS: 5
PROFICIENCY: 4
ACCBONUS: -5
CRITROLL: 20
DAMAGE: 2d6 + ATTACKMOD + DMGBONUS + 2d8 + 4d8 + 3d6 + 3d6
CRITDAMAGE: 4d6 + ATTACKMOD + DMGBONUS + 4d8 + 8d8 + 6d6 + 6d6
output [multiattack 1 1] named "18 Cha Maul +1 Aasimar Hexblade 5/Whisper Bard 6 (Total 11)"

ATTACKMOD: 6
DMGBONUS: 15
PROFICIENCY: 4
ACCBONUS: -5
CRITROLL: 20
DAMAGE: 2d6 + ATTACKMOD + DMGBONUS + 1
CRITDAMAGE: 4d6 + ATTACKMOD + DMGBONUS + 3
output [multiattack 3 1] named "20 Str Greatsword +1 GWM Aasimar Fighter 11"

function: check N:n skill {
  if N > 16 {
    result: 3d6 <= 16
  }
  else
  {
    result: 3d6 <= N
  }  
}

function: skill ATTSKILL: n attack DEFSKILL: n dodge {
  result: [check ATTSKILL skill] & (! [check DEFSKILL skill])
}

function: DMG: n dices damages DR: n dr with MODIFIER: n mod {
  DEALT: DMG - DR
  if DEALT < 0 {
    result: 0
  }
  else {
    result: DEALT * MODIFIER
  }
}

function: COMBAT: n results damaging DR:n dr by DMG:n dices with MODIFIER: n mod {
  if ! COMBAT {
    result: 0
  }
  else
  {
    result: [DMG dices damages DR dr with MODIFIER mod]
  }
}


output [[skill 17 attack 9 dodge] results damaging 7 dr by 8d6 dices with 2 mod]

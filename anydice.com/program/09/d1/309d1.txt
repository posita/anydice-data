function: ROLL:n rerollones as REROLL:d {
  if ROLL = 1 { 
                   result: REROLL
                  }
  result: ROLL
}

function: countrerolls ROLL:n {
  if ROLL = 1 { 
                   result: 1 + [countrerolls d6]
                  }
  result: 0
}

function: produceflame ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: PRODUCEFLAMECRIT } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: PRODUCEFLAME }
 result: 0
}

function: quarterstaff ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: QUARTERSTAFFCRIT } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + 2 + ACCBONUS >= AC { result: QUARTERSTAFF }
 result: 0
}

function: poisonspray ROLL:n vs DC:n {
 if ROLL < ENEMYSAVE { result: 0 }

}

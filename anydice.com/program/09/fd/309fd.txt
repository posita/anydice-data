  if A = 1 {
    result: d6
  } else {
    result: A

}

function: rolloff A:s {
  if A >= 4 {
    result: 1
  } else {
    result: 0
  }
}

function: equal fight A:s vs B:s {
  if A > B { result: 1 }
  if A < B { result: 0 }

}

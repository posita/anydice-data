function: explode DIE:d SPECIALTY:n BONUS:n STRENGTH:n {
 MAX: [maximum of DIE] - SPECIALTY
 STRLOW: [lowest of STRENGTH and 0]
 result: [explode DIE helper (-BONUS) STRENGTH]
}
function: explode N:n helper PENALTY:n STRENGTH:n {
  if PENALTY >= (N+STRENGTH) {
   result: 0
  }
  if N >= MAX {
   result: N + [explode DIE helper PENALTY+1 STRENGTH-1] - PENALTY + [highest of STRENGTH and STRLOW]
  }
  result: N - PENALTY + [highest of STRENGTH and STRLOW]
}

set "explode depth" to 10

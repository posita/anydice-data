function: limitedex DICENUM with SIDES for NUMEX{
	STORE: 0
	loop COUNT over {1..DICENUM}{
	 STORE : STORE + ( NUMEXdSIDES = NUMEX * SIDES) 
	 }
        result: STORE
}
function: rollfn A:n B:n EX:n{
        EXC: [lowest of B and EX]
	UNEX: [highest of A-EX and 0]
	UNEXC:[highest of B-EX and 0]
        result: [highest UNEXC of UNEXd9] + [highest EXC of EXd[banger]]
}

function: banger{
      result: 10+ d[explode d10]
}

function: A legend B{
	EX: [limitedex A with 10 for 1]
	EXC: [lowest of B and EX]
	UNEX: [highest of A-EX and 0]
	UNEXC:[highest of B-EX and 0]
	result: [rollfn A B EX]
}

function: produceflame ROLL:n {
  if ROLL = 20 { result: 2d8 }
  if ROLL + 5 >= 13 { result: 1d8 }
  result: 0
}

function: averonsflame ROLL:n {
  if ROLL = 20 { result: 2d4 }
  if ROLL + 5 >= 13 { result: 1d4 }
  result: 0
}

function: averonsflametwice {
  result: [averonsflame d20] + [averonsflame d20]
}

function: flameseed ROLL:n {
  if ROLL = 20 { result: 2d6 + 2 }
  if ROLL + 5 >= 13 { result: 1d6 + 2 }
  result: 0
}

function: averonsflameandflameseed {
  result: [averonsflame d20] + [flameseed d20]
}

function: produceflameandflameseed {
  result: [produceflame d20] + [flameseed d20]
}

function: averonsflametwiceandflameseed {
  result: [averonsflametwice] + [flameseed d20]
}

output [averonsflame d20] named "Attacking with Averon's Flame once (Action)"
output [produceflame d20] named "Attacking with Produce Flame once (Action)"
output [averonsflametwice] named "Attacking with Averon's Flame twice (multi-attack in a single Action)"
output [flameseed d20] named "Attacking with Flame Seed once (Bonus Action)"
output [averonsflameandflameseed] named "Attacking with Averon's Flame once (Action), and with Flame Seed (Bonus Action)"
output [produceflameandflameseed] named "Attacking with Produce Flame once (Action), and with Flame Seed (Bonus Action)"
output [averonsflametwiceandflameseed] named "Attacking with Averon's Flame twice (multi-attack in a single Action), and with Flame Seed (Bonus Action)"

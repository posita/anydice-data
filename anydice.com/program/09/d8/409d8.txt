function: dice N: n {
    result: Nd{0, 0, 1, 1, 1, 2}
}

loop I over {2..8} {
    output [dice I] > 3
}

loop J over {1..3} {
    loop I over {2..(8 - J)} {
        if I < J {
            X: 0
        } else {
            X: I - J
        }
        output [highest X of [dice I]] + [dice J] > 3
    }
}

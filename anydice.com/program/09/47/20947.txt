\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: DAMAGE + DAMAGE + ATTACKMOD + DMGBONUS } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE + ATTACKMOD + DMGBONUS }
 result: 0
}

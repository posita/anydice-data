TURNS: 10

POWER: 10
SHARPNESS: 5

HEALTH: 50
CRUMBLE_MAX: 2

CRUMBLE: 0
CRUMBLE_INCR: POWER / SHARPNESS
DAMAGE: 0
DID_CRUMBLE: 0
loop N over {1:TURNS}{
  DAMAGE: DAMAGE + POWER + 2d6
  CRUMBLE: CRUMBLE + CRUMBLE_INCR + 2d6
  CRUMBLE_CH: (CRUMBLE*100)/(CRUMBLE_MAX*100)
  DID_CRUMBLE: DID_CRUMBLE | CRUMBLE_CH > 1d100
}

output !DID_CRUMBLE & DAMAGE > HEALTH

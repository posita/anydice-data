TURNS: 5

POWER: 10
SHARPNESS: 5

HEALTH: 50
CRUMBLE_MAX: 30

CRUMBLE: 0
CRUMBLE_INCR: POWER / SHARPNESS
DAMAGE: 0
loop N over {1..TURNS}{
  DAMAGE: DAMAGE + POWER + 2d6
  CRUMBLE: CRUMBLE + CRUMBLE_INCR + 2d6
  CRUMBLE_CH: (CRUMBLE*100)/(CRUMBLE_MAX*100)
  output DAMAGE > HEALTH & CRUMBLE_CH = 0 named "Win on Turn [N]"
}

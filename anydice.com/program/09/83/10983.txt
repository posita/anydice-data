CUSTOM_AC: 13
CUSTOM_BONUS: 2

function: reroll ROLL:n with BONUS:n if it misses AC:n {
  if ROLL + BONUS > AC {
    result: ROLL + BONUS
  } else {
    result: 1d20 + BONUS
  }
}

output [reroll 1d20 with CUSTOM_BONUS if it misses CUSTOM_AC] named "proposed"
output 1@2d20 + CUSTOM_BONUS named "advantage"
output 1d20 + CUSTOM_BONUS named "normal"
output CUSTOM_AC named "AC reference point"


function : attack map PLUS:n vs AC:n
{
 G : {1,AC-PLUS,20}
 G : 2@[sort G]
 result : 1d{0:(G-1),1:(20-G),2}
}

function : attack MAP:s damage DAMAGE:d DPLUS:n
{
 D : 0
 if MAP >= 1
 {
  D : DPLUS
  D : D + MAPdDAMAGE
 }
 
 result : D
}

function : make attack HP MHP vs DAMAP times TIMES
{

 loop N over {1..TIMES}
 {
  HP : HP - DAMAP
 }
 result : HP
}

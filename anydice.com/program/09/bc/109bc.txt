MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLLONE:n and ROLLTWO:n vs DEFENSE:n {
 if ROLLONE + ROLLTWO = 2 { result: MISS }
 if ROLLONE > DEFENSE & ROLLTWO > DEFENSE { result: CRITICALHIT }
 if ROLLONE <= DEFENSE & ROLLTWO <= DEFENSE { result: MISS }
 result: HIT
}

output [attack [explode d6] and [explode d6] vs [highest 1 of 2d6]-2] named "Easy"
output [attack [explode d6] and [explode d6] vs [highest 1 of 2d6]] named "Medium"
output [attack [explode d6] and [explode d6] vs [highest 1 of 2d6]+2] named "Hard"

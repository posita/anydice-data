function: rolla A and B {
  result: [highest of [lowest of A and d6] - [lowest of B and d6] and 0]
}

function: rollb A and B{
  result: [highest of [lowest of d6 and A - B] and 0]
}

function: rollc A and B {
  result: [highest of d6 + A - d6 - B and 0]
}

output [rolla 2 and 3]
output [rollb 2 and 3]
output [rollc 2 and 3]

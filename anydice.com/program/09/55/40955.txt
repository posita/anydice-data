ATR: 1 \ Coloque o valor de atributo \
PER: 0 \ Coloque o valor de perícia \
SAC: 0 \ Coloque quantos pontos de sacrifício serão usados \

\ Coloque o valor 2 nos pontos de sacrifício caso queria a rolagem dificultada; \
\ Coloque o valor 0 nos pontos de sacrifício caso queria a rolagem base; \
\ Coloque o valor -2 nos pontos de sacrifício caso queria a rolagem facilitada; \







DADOS: ATR+PER+SAC

if SAC = 2 {
output DADOSd(d8 >= 6)
}

if SAC = 0 {
output DADOSd(d10 >= 6)
}

if SAC = -2 {
output DADOSd(d12 >= 6)
}

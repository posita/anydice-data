function: tens A:n {
 X: (A / 10)
 if X = 10 { X: 0 }
 result: X
}

function: digits A:n {
 X: A - [tens A]*10
 if X = 100 { X: 0 }
 result: X
}

function: flip to lower B:n C:n{
 X: [tens B]
 Y: [digits B]
 if X < Y {
  Z: (Y*10 + X)
 }
 if Y < X {
  Z: (X*10 + Y)
 }
 if X = Y {
  Z: (X*10 + Y)
 }
 if Z < C {
  result: 1
 }
 result: 0
}

output [flip to lower d100 67]

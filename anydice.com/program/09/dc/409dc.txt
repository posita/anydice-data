function: if CONDITION:n then A:s else B:s {
  if CONDITION { result: A } else { result: B }
}

function: rerollok HIGHEST:n MID:n LOWEST:n {
   if HIGHEST>=10 {
      if MID + LOWEST < 10 {
         result: 1
      }
      result: 0
   }
   result: 0
}


function: secondroll OLDHIGHEST:n {
  ROLL: 2d12
\  HIGHEST: {1}@ROLL
  LOWEST: {2}@ROLL
  result: [if  LOWEST < OLDHIGHEST then HIGHEST + OLDHIGHEST else ROLL]
\
  result: 2
}

function: roll {
   ROLL: 3d6
   BASE: {1,2}@ROLL
   HIGHEST: {1}@ROLL
   MID: {2}@ROLL
   LOWEST: {3}@ROLL
   result: [if  [ rerollok HIGHEST MID LOWEST ] then {1,2}@3d6 else BASE]
}

ROLL: [roll]

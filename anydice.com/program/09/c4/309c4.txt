function: attack ATTACK:d {
  result: ATTACK
}

function: roll gwf damage with SIDES_ON_DMG_DIE:n sided die {
    \ create a die that simulates rerolling the first 1 or 2 as a sequence \
    GWB_DIE: {{1..2}:2, {3..SIDES_ON_DMG_DIE}:SIDES_ON_DMG_DIE+2}
    result: dGWB_DIE
}


function: multiattack X:n Y:d {
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: [attack Y]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

ATTACK_A: 1d8 + 5
ATTACK_B: 2d6 + 5

output [multiattack 4 ATTACK_A] named "4 Monk attacks w/ boosted die (d8 instead of d6)"
output [multiattack 4 ATTACK_B] named "4 Monk attacks w/ extra die (2d6)"

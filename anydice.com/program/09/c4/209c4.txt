function: tens A:n {
 X: (A / 10)
 if X = 10 { X: 0 }
 result: X
}

function: digits A:n {
 X: A - [tens A]*10
 if X = 100 { X: 0 }
 result: X
}

function: flip to lower B:n C:n D:n{
 X: [tens B]
 Y: [digits B]
 if X < C {
  if X < D {
   if Y < C {
    if Y < D {
     result: 1
    }
   }
  }
 }
 result: 0
}

output [flip to lower d100 9 9]

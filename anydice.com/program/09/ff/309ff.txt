function: attack ATTACK:d {
  result: ATTACK
}

function: reroll R:n under N:n {
   if R < N { result: d8 } else {result: R}
}

function: multiattack X:n Y:d {
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: [attack Y]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

ATTACK_A: [reroll 1d8 under 3] + 5
ATTACK_B: 2d6 + 5

output [multiattack 4 ATTACK_B] named "Current Damage: 4 Monk attacks w/ extra die(2d6)"
output [multiattack 4 ATTACK_A] named "Proposed Damage: 4 Monk attacks w/ boosted die (d8 instead of d6, reroll 1's and 2's)"

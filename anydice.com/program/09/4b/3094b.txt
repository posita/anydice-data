function: get highest {
 A: {d10-1}
 R: {}
 loop N over A{
  loop M over A{
   if N >= M {
    R: {R, N*10+M}
   }
   else {
    R: {R, M*10+N}
   }
  }
 }
 result: R
}

function: get lowest {
 A: {d10-1}
 R: {}
 loop N over A{
  loop M over A{
   if N < M {
    R: {R, N*10+M}
   }
   else {
    R: {R, M*10+N}
   }
  }
 }
 result: R
}

\ output [get highest] named "take highest" \
\ output [get lowest] named "take lowest" \

function: get highest reduced {
 A: {d10-1}
 R: {}
 loop N over A{
  loop M over A{
   if N = M {
    R: {R, N*11}
   }
   if N > M {
    R: {R, (N-1)*10+(N-1)}
   }
   if N < M {
    R: {R, (M-1)*10+(M-1)}
   }
  }
 }
 result: R
}
function: get lowest reduced {
 A: {d10-1}
 R: {}
 loop N over A{
  loop M over A{
   if N = M {
    R: {R, N*11}
   }
   if N < M {
    R: {R, (N)*10+(N)}
   }
   if N > M {
    R: {R, (M)*10+(M)}
   }
  }
 }
 result: R
}

\ output [get highest reduced] named "take highest" \
\ output [get lowest reduced] named "take lowest" \

function: get highest mushed {
 A: {d10-1}
 R: {}
 loop N over A{
  loop M over A{
   if N = M {
    if N = 0 {
     R: {R, -1}
    }
    else if N = 9 {
     R: {R, 10}
    }
    else {
     R: {R, N}
    }
   }
   if N > M {
    R: {R, (N-1)}
   }
   if N < M {
    R: {R, (M-1)}
   }
  }
 }
 result: R
}
function: get lowest mushed {
 A: {d10-1}
 R: {}
 loop N over A{
  loop M over A{
   if N = M {
    if N = 0 {
     R: {R, -1}
    }
    else if N = 9 {
     R: {R, 10}
    }
    else {
     R: {R, N}
    }
   }
   if N < M {
    R: {R, (N)}
   }
   if N > M {
    R: {R, (M)}
   }
  }
 }
 result: R
}

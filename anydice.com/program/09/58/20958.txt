function: opposed ATTACK_SKILL:n vs DEFEND_SKILL:n {
  ATTACK_ROLL: d100
  ATTACK_SUCCESS: ATTACK_ROLL <= ATTACK_SKILL
  ATTACK_SUCCESS_LEVEL: (ATTACK_SKILL / 10) - (ATTACK_ROLL / 10)
  DEFEND_ROLL: d100
  DEFEND_SUCCESS: DEFEND_ROLL <= DEFEND_SKILL
  DEFEND_SUCCESS_LEVEL: (DEFEND_SKILL / 10) - (DEFEND_ROLL / 10)
  result: ATTACK_SUCCESS_LEVEL - DEFEND_SUCCESS_LEVEL
}

output [opposed 35 vs 75] named "Attacker wins by"

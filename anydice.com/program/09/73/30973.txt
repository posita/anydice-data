TEN: {{-9..0},{2..9}:10,{11..20}}

ATK: 14
DV: 25
DMG:3d6
PRECISIONATTACK: 0

SPOTWEAKNESS: 3
SP: 4

function: attack N:n{
if ATK+N+PRECISIONATTACK>DV{result:DMG}
result:0}

function: damage N:n{
if N+SPOTWEAKNESS>SP{result:SPOTWEAKNESS+N-SP}
result:0}

output [attack dTEN]
output [damage [attack dTEN]]

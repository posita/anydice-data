function: roll A:n under B:n low L:s high H:s {
  if [L contains A] {result: 1}
  if [H contains A] {result: 0}
  result: A <= B
}

function: roll A:n bonus B over T:n low L:s high H:s {
  if [L contains A] {result: 0}
  if [H contains A] {result: 1}
  result: A + B >= T
}

loop B over {1..20}{
 output [roll 3d6 under B low {3,4} high {17,18}] named "3d6 ≤ [B]"
}

loop B over {1..20}{
 output [roll 3d6 bonus B over 21 low {3,4} high {17,18}] named "3d6 + [B] ≥ 21"
}

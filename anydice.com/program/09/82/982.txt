EXPWILDDIE:[explode d6]
TRAITDIE:d12

function: fail on TD:d with mod MOD:n {
 EXPTRAITDIE: [explode TD]
 TMOD: EXPTRAITDIE + MOD
 WMOD: EXPWILDDIE + MOD
 result: (EXPTRAITDIE != 1 & (TMOD < 4 & WMOD < 4 )) 
}

function: critfail on TD:d with mod MOD:n {
 EXPTRAITDIE: [explode TD]
 result: EXPTRAITDIE = 1
}

function: success on TD:d with mod MOD:n {
 EXPTRAITDIE: [explode TD]
 TMOD: EXPTRAITDIE + MOD
 WMOD: EXPWILDDIE + MOD
 TDSUCCESSONLY: (TMOD >= 4 & TMOD < 8) & WMOD < 4
 WDSUCCESSONLY: (WMOD >= 4 & WMOD < 8) & (EXPTRAITDIE != 1 & TMOD < 4)
 BOTHSUCCESS: (WMOD >= 4 & WMOD < 8) & (TMOD >= 4 & TMOD < 8)
 result: (TDSUCCESSONLY | WDSUCCESSONLY) | BOTHSUCCESS
}

function: raisedsuccess on TD:d with mod MOD:n {
 EXPTRAITDIE: [explode TD]
 TMOD: EXPTRAITDIE + MOD
 WMOD: EXPWILDDIE + MOD
 TDSUCCESSONLY: TMOD >= 8 & WMOD < 8
 WDSUCCESSONLY: WMOD >= 8 & (EXPTRAITDIE != 1 & TMOD < 8)
 BOTHSUCCESS: WMOD >= 8 & TMOD >= 8
 result: (TDSUCCESSONLY | WDSUCCESSONLY) | BOTHSUCCESS
}

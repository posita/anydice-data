\2d6+Range of 0-5+2 vs TN 10\
TN: 10

loop N over {0..5} {
 output (2d6 + N + 2) >= TN named "[N]+2"
}

loop N over {0..5} {
 output ((3d6 + N ) >= TN + 1d6)*2 named "[N]+2"
}

loop N over {0..5} {
 output (({1,2}@3d6 + N ) >= TN + 1d6)*3 named "[N]+2"
}

\ result 0 = less than 10
  result 1 = greater or equal to 10
\

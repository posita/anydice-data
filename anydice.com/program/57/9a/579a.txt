DIE: d6
LOWESTAMOUNT: 3
HIGHESTAMOUNT: 7
TARGETNUMBER: 9

function: roll AMOUNT:n {
  SUCCESSES: 0

  loop N over {1..AMOUNT} {
    SUCCESSES: SUCCESSES + [check N@AMOUNTd[explode DIE] against target number]
  }

  result: SUCCESSES
}

function: check SINGLEDIE:n against target number {
  result: SINGLEDIE >= TARGETNUMBER
}

loop AMOUNT over {LOWESTAMOUNT..HIGHESTAMOUNT} {
  output [roll AMOUNT] named "Rolling [AMOUNT][DIE] against [TARGETNUMBER]"
}

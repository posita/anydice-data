MOD:5
PROF:3
MISC:0
WEPDIEF:1d8+2d6
WEPDIES:1d8
WEPDIET:2d8
CRIT:2

ATTACK:MOD+PROF+MISC
DAMAGEF:MOD+MISC+(WEPDIEF+WEPDIEF)
DAMAGES:MOD+MISC+(WEPDIES+WEPDIES)
DAMAGET:MOD+MISC+(WEPDIET+WEPDIET)
ATTACKROLL: [highest 1 of 2d20]

SDAMAGEF:MOD+MISC+(WEPDIEF)
SDAMAGES:MOD+MISC+(WEPDIES)

loop AC over {10..20} {
  output (ATTACKROLL+ATTACK >=AC)* (DAMAGEF)+(ATTACKROLL+ATTACK >=AC)* (DAMAGES)+(ATTACKROLL+ATTACK >=AC)* (DAMAGET) named "first turn vs AC:[AC]"
  output (1d20+ATTACK >=AC)* (SDAMAGEF)+(1d20+ATTACK >=AC)* (SDAMAGES) named "rest of the turns vs AC:[AC]"
}

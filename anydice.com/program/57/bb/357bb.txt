ADVANTAGE: [highest 1 of 2d20]
PROFICIENCY: 3
DEXTERITY: 5
EXPERTISE: 3
GUIDANCE: 1d4
BLESS: 1d4
BARDIC_INSPIRATION: 1d10

function: reliable talent ROLL:n with BONUS:n {
    TRUE_ROLL: ROLL
    if ROLL < 10 {TRUE_ROLL: 10}
    result: TRUE_ROLL + DEXTERITY + PROFICIENCY + EXPERTISE + BONUS
}

function: unreliable talent ROLL:n with BONUS:n {
    if ROLL = 1 {result: 1}
    result: ROLL + DEXTERITY + PROFICIENCY + EXPERTISE + BONUS
}

RT: [reliable talent ADVANTAGE with GUIDANCE + BARDIC_INSPIRATION]
URT: [unreliable talent ADVANTAGE with GUIDANCE]

output RT
output URT

\ Input: DICE - number of dice to roll. 0 (or lower) is correctly      \
\               handled as "roll 2, take worse."                       \

\ Output:                                                              \
\ -1 Bad outcome                                                       \
\  0 Partial success                                                   \
\  1 Full success                                                      \
\  2 Critical success - NOT YET IMPLEMENTED TODO                       \

\ Example: show probabilities when 4 dice are rolled:                  \
\   output [blades roll 4] named "4 dice"                              \


function: bladeshelper THERESULT:n {
  if (THERESULT >= 6) {
    result: 1
  }
  if (THERESULT >= 4) {
    result: 0
  }
  result: -1
}

function: blades roll DICE:n {
  if (DICE <= 0) {
    result: [bladeshelper [lowest 1 of 2d6]]
  } else {
    result: [bladeshelper [highest 1 of DICEd6]]
  }
}


output [blades roll 0] named "0 dice"
output [blades roll 1] named "1 die"
output [blades roll 2] named "2 dice"
output [blades roll 3] named "3 dice"
output [blades roll 4] named "4 dice"
output [blades roll 5] named "5 dice"

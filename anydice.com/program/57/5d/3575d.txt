BONUS: 17

function: pfe healing compare ROLL:n with dc DC:n extra healing EXTRA:n risky surgery RS:n{
RS_DAMAGE: 0
if RS = 1 {ROLL: ROLL +2}
if RS = 1 {RS_DAMAGE: -1d8}
CRIT_FAIL: DC-10
CRIT_SUCCESS: DC+10
if ROLL + BONUS <=CRIT_FAIL                                  {HEAL: -1d8}
if ROLL + BONUS >CRIT_FAIL      & ROLL + BONUS <DC           {HEAL: 0}
if ROLL + BONUS >=DC            & ROLL + BONUS <CRIT_SUCCESS {HEAL: 2d8 + EXTRA}
if ROLL + BONUS >=CRIT_SUCCESS                               {HEAL: 4d8 + EXTRA}
result: HEAL + RS_DAMAGE
}

output [pfe healing compare 1d20 with dc 15 extra healing 0 risky surgery 0] named "DC 15"
output [pfe healing compare 1d20 with dc 20 extra healing 15 risky surgery 0] named "DC 20"
output [pfe healing compare 1d20 with dc 30 extra healing 40 risky surgery 0] named "DC 30"

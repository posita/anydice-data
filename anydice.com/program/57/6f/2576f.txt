
loop MOD over {0..10} {
T: {-1,0,1} 
TT: {0,1,1} 

output (10-MOD)dT + MODdTT named "+[MOD]"
}

loop MOD over {1..10} {
TT: {0,1,1} 
TTT: {1,1,2} 
MODD: MOD+10

output (10-MOD)dTT + MODdTTT named "+[MODD]"
}


loop MOD over {1..10} {
TTT: {1,1,2} 
MODD: MOD+20

output (10)dTTT + MOD*2/3 named "+[MODD]"
}

function: get largest die size with A:n or less sides {
  if A <= 4 {result: 4}
  if A <= 6 {result: 6}
  if A <= 8 {result: 8}
  if A <= 10 {result: 10}
  if A <= 12 {result: 12}
  if A <= 20 {result: 20}
}

POWER_DICE_SIZE: [get largest die size with POWER_TWELVE or less sides]
output POWER_DICE_SIZE named "POWER_DICE_SIZE"


function: highest of A:s if highest is not greater than B else zero {
  SORTED: [sort A]
  HIGHEST: 1 @ SORTED
  if HIGHEST > B {result: 0} else {result: HIGHEST} 
}

function: lowest of A:s if highest is not greater than B else zero {
  SORTED: [sort A]
  HIGHEST: 1 @ SORTED
  LOWEST: #SORTED @ SORTED
  if HIGHEST > B {result: 0} else {result: LOWEST} 
}

function: lowest of A:s where anything below or equal to B is zero {
  SORTED: [sort A]
  LOWEST: #SORTED @ SORTED
  if LOWEST <= B {result: 0} else {result: LOWEST} 
}

function: highest of A:s where anything below or equal to B is zero {
  SORTED: [sort A]
  loop N over SORTED {
    if N <= B {result: 0} else {result: N}
  }
  result: 0
}

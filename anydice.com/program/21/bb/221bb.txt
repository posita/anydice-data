
SHOW_ADV: 1   \ 1=true, 0=false \
SHOW_DADV: 1   \ 1=true, 0=false \

loop DIFFNUM over {1,2} {
 loop DIFFDIE over {4,8,12,20} {
  output 1d20 + DIFFNUMdDIFFDIE named "difficulty [DIFFNUM]d[DIFFDIE]"
  if SHOW_ADV {
   output 1@2d20 + DIFFNUMdDIFFDIE named "difficulty [DIFFNUM]d[DIFFDIE] adv"
  }
  if SHOW_DADV {
   output 2@2d20 + DIFFNUMdDIFFDIE named "difficulty [DIFFNUM]d[DIFFDIE] adv"
  }
 }
}

\ for advantage, replace “1d20” with “1@2d20” \
\ for disadvantage, replace “1d20” with “2@2d20” \

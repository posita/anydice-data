function: arrange A:n and B:n {
  if (A < B) {
    result: B * 20 + A - 1
  } else {
    result: A * 20 + B - 1
  }
}

function: mid AB:n and C:n {
  DH: AB / 20
  DL: AB - DH * 20 + 1

  if (DH < C) {
    result: DH
  } else if (DL < C) {
    result: C
  } else {
    result: DL
  }
}

function: min AB:n and C:n {
  DH: AB / 20
  DL: AB - DH * 20 + 1

  if (DL < C) {
    result: DL
  } else {
    result: C
  }
}

function: max AB:n and C:n {
  DH: AB / 20

  if (DH < C) {
    result: C
  } else {
    result: DH
  }
}

function: medmin AB:n and C:n {
  DH: AB / 20
  DL: AB - DH * 20 + 1

  if (DH < C) {
    result: DL + DH
  } else {
    result: DL + C
  }
}

function: medmax AB:n and C:n {
  DH: AB / 20
  DL: AB - DH * 20 + 1

  if (DL < C) {
    result: DH + C
  } else {
    result: DH + DL
  }
}

function: maxmin AB:n and C:n {
  DH: AB / 20
  DL: AB - DH * 20 + 1

  if (DH < C) {
    result: DL + C
  } else if (DL < C) {
    result: DH + DL
  } else {
    result: DH + C
  }
}

function: doubles AB:n and C:n {
  DH: AB / 20
  DL: AB - DH * 20 + 1

  if (DH = DL) {
    result: 1
  } else if (DH = C) {
    result: 1
  } else {
    result: DL = C
  }
}

function: ricochet AB:n and C:n {
  DH: AB / 20
  DL: AB - DH * 20 + 1

  if ((DH = C) + (DL = C) + (DH = DL)) {
    R: 1
  } else {
    R: 0
  }

  if (DH < C) {
    result: C + DL * R
  } else if (DL < C) {
    result: DH + DL * R
  } else {
    result: DH + C * R
  }
}

loop II over {1..5} {
  I: 1d(2 + 2 * II)
  loop JJ over {II..5} {
    J: 1d(2 + 2 * JJ)
    IJ: [arrange I and J]
    loop KK over {JJ..5} {
      K: 1d(2 + 2 * KK)
      output [doubles IJ and K] named "doubles of [I], [J], and [K]"
    }
  }
}

loop P over {0,+3,-3} {
    Q: 13+P
    if Q = 13 {  output Q named "~~~ NORMAL VALUES ~~~" }
    if Q = 10 {  output Q named "~~~ IF CHARACTER HAVE ATTACK TOKEN ~~~" }
    if Q = 16 {  output Q named "~~~ IF OPPONENT HAVE DEFENCE TOKEN ~~~" }
    output 2d10+1d6>= Q named "Basic roll: 2d10+1d6"
    output 1d10+2d6>= Q named "1d10+2d6"
    output 3d6>= Q named "3d6"
}

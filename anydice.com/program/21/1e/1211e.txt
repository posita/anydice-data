
function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll {1,2} on DICE to DICE]
}

output [reroll {1,2} on d4] named "reroll 1s and 2s on a d4"

output [reroll {1,2} on d6] named "reroll 1s and 2s on a d6"

output [reroll {1,2} on d8] named "reroll 1s and 2s on a d8"

output [reroll {1,2} on d10] named "reroll 1s and 2s on a d10"

output [reroll {1,2} on d12] named "reroll 1s and 2s on a d12"

output 2d[reroll {1,2} on d6] named "reroll 1s and 2s on 2d6"

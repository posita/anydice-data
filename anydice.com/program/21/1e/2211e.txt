function: clamp N:n between MIN:n and MAX:n {
  if N <= MIN { result: MIN }
  if N >= MAX { result: MAX }
  result: N
}
function: minimum of D:d {
 result: 1@{D}
}


function: explode D:d both {
  MIN: [minimum of D]
  MAX: [maximum of D]
  result: [explode D using D with MIN and MAX]
}
function: explode N:n using D:d with MIN:n and MAX:n {
  if N = MIN {
    result: N - [explode D] 
  }
  if N = MAX {
    result: N + [explode D] 
  }
  result: N
}

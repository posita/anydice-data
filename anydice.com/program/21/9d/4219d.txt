function: explode A:n and B:n {
 if A = B { result: A + B + [explode d10 and d10] }
 
 result: A + B
}

function: adv A:n and B:n and C:n {
 if A = B { result: A + B + [explode d10 and d10] }
 if A = C { result: A + C + [explode d10 and d10] }
 if B = C { result: B + C + [explode d10 and d10] }

 FIRST: 0
 SECOND: 0
 
 loop N over {1..10} { 
  X: 11 - N

  if X = A | X = B | X = C {
   if FIRST = 0 { FIRST: X }
   else if SECOND = 0 { SECOND: X }
  }

  if FIRST > 0 & SECOND > 0 {
   result: FIRST + SECOND
  }
 }
}

function: dis A:n and B:n and C:n {
 if A = B & B = C { result: A + B + [explode d10 and d10] }

 FIRST: 0
 SECOND: 0
 THIRD: 0
 
 loop N over {1..10} { 
  X: 11 - N

  if X = A {
   if FIRST = 0 { FIRST: X }
   else if SECOND = 0 { SECOND: X }
   else if THIRD = 0 { THIRD: X }
  }
  if X = B {
   if FIRST = 0 { FIRST: X }
   else if SECOND = 0 { SECOND: X }
   else if THIRD = 0 { THIRD: X }
  }
  if X = C {
   if FIRST = 0 { FIRST: X }
   else if SECOND = 0 { SECOND: X }
   else if THIRD = 0 { THIRD: X }
  }

  if SECOND > 0 & THIRD > 0 {
   if SECOND = THIRD {
    result: FIRST + THIRD
   }

   result: SECOND + THIRD
  }
 }
}

MISS: 0
HIT: 1
CRITICALHIT: 2
NORMAL: d20
ADVANTAGE: [highest 1 of 2d20]
DISADVANTAGE: [lowest 1 of 2d20]

function: attack ROLL:n BONUS:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + BONUS>= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + BONUS>= DEFENSE { result: HIT }
 result: MISS
}

function: damage ROLL:n CRITROLL:n BONUS:n for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITROLL + BONUS
 }
 if ATTACKRESULT = HIT {
  result: ROLL + BONUS
 }
 result: 0
}

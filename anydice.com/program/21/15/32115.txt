function: get largest die size with A:n or less sides {
  if A >= 20 {result: 20}
  if A >= 12 {result: 12}
  if A >= 10 {result: 10}
  if A >= 8 {result: 8}
  if A >= 6 {result: 6}
  if A >= 4 {result: 4}
  result: 0
}
\

\
EXP_DIE_SIZE: [get largest die size with ATTRIBUTE+LEVEL or less sides]
output EXP_DIE_SIZE named "EXP_DIE_SIZE"
MAG_DIE_SIZE: [get largest die size with 3*MAGIC or less sides]
output MAG_DIE_SIZE named "MAG_DIE_SIZE"

function: highest of A:s if highest is not greater than B else zero {
  SORTED: [sort A]
  HIGHEST: 1 @ SORTED
  if HIGHEST > B {result: 0} else {result: HIGHEST} 
}

function: lowest of A:s if highest is not greater than B else zero {
  SORTED: [sort A]
  HIGHEST: 1 @ SORTED
  LOWEST: #SORTED @ SORTED
  if HIGHEST > B {result: 0} else {result: LOWEST} 
}

function: lowest of A:s where anything below or equal to B is zero {
  SORTED: [sort A]
  LOWEST: #SORTED @ SORTED
  if LOWEST <= B {result: 0} else {result: LOWEST} 
}

function: highest of A:s where anything below or equal to B is zero {
  SORTED: [sort A]
  loop N over SORTED {
    if N <= B {result: 0} else {result: N}
  }
  result: 0
}

function: lowest N:n of dice A:s and B:s {
  \This doesn't work without a function, you need to tell AnyDice to interpret A and B as sequences rather than dice.\
  result: {1..N}@ [reverse [sort {A, B}]]
}


function: roll power pool with EXPERIENCE_DIE_SIZE:n and MAGIC_DIE_SIZE:n as full class {
  if EXPERIENCE_DIE_SIZE = 0 & MAGIC_DIE_SIZE = 0 {
    result: [lowest 1 of dice d4 and d4]
  } else {
    result: [highest 1 of dice dEXPERIENCE_DIE_SIZE and dMAGIC_DIE_SIZE]
  }
}

function: roll power pool with EXPERIENCE_DIE_SIZE:n and MAGIC_DIE_SIZE:n as half class {
  POWER_DIE_SIZE: [lowest of EXPERIENCE_DIE_SIZE and MAGIC_DIE_SIZE]
  if POWER_DIE_SIZE = 0 {
    result: [lowest 1 of dice d4 and d4]
  } else {
    result: dPOWER_DIE_SIZE
  }
}

GRIT:3
ACCURACY:-1 \negative if difficulty\
DMG:2d6+3   


EVASION:13   
ARMOUR:1   
INVISIBLE:0


function: attack ROLL:n vs EVASION:n for DMG:d crit CRIT_DMG:d {

    if ROLL + GRIT >= 20 {
        result: CRIT_DMG
    }
    if ROLL + GRIT>= EVASION {
        result: DMG
    }
    result: 0

}

function: invis CLOAK:n {

if CLOAK = 1 {
        result: d2
    }
    
    result: 5
}

function: miss COIN:n{

if COIN = 1 {
        result: 0
    }
    if COIN = 2 {
        result: 1
    }
    result: 1
}

output[attack d20 + GRIT + [highest 1 of ACCURACYd6] vs EVASION for (DMG-ARMOUR)*[miss [invis INVISIBLE]] crit ([highest 1 of 1d DMG]-ARMOUR)*[miss [invis INVISIBLE]] ]
named "DMG/chance"

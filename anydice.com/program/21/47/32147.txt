N: 5
ROLL: 2d10

function: success ROLL:n {
  if [highest N of ROLL] >= N { result: 1 }
  if [highest N of ROLL] = N { result: 1 }
  if [highest N of ROLL] = [lowest N of ROLL] { result: 1 }
  result: 0
}

function: critical_success ROLL:n {
  if [highest N of ROLL] = N { result: 1 }
  if [highest N of ROLL] = [lowest N of ROLL] { result: 1 }
  result: 0
}

function: critical_failure ROLL:n {
  if [highest N of ROLL] = N { result: 1 }
  if [highest N of ROLL] = [lowest N of ROLL] { result: 1 }
  result: 0
}

output [success ROLL]
output [critical_success ROLL]
output [critical_failure ROLL]

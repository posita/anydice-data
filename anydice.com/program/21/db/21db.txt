function: ndice N {
  DIE: d{0,0,0,0,0,0,1,1,1,2}
  DICE: DIE
  loop X over {2..N} {
    DICE: DICE + DIE
  }
  result: DICE
}

loop A over {1..15} {
  loop B over {1..15} {
    output [ndice A] > [ndice B] named "[A] dice vs [B] dice"
  }
}

loop A over {1..15} {
  loop B over {1..15} {
    output [ndice A] >= B named "[A] dice vs difficulty [B]"
  }
}

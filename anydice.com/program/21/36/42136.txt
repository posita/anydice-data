function: tokens DIE:d target TARGET:n {
  ROLL: [highest of [explode DIE] and [explode d6]]
  result: [highest of 0 and (ROLL - TARGET + 4) / 4]
}

PLAYER_A: [tokens d4 target 5]
PLAYER_B: [tokens d6 target 6]
PLAYER_C: [tokens d12 target 8]
TOTAL: PLAYER_A + PLAYER_B + PLAYER_C

ROUNDS: 10
output ROUNDS d TOTAL named "tokens gained over [ROUNDS] rounds"

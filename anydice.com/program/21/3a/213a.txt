function: ntwf A:n vs B:n {
  TEMP: 0
  if (A+2)>B {
    TEMP: MAIN
    if (A-3)>B {
      TEMP: TEMP + OFF
    }
  }
  result: TEMP
}
function: nofeat A:n vs B:n {
  TEMP: 0
  if (A-1)>B {
    TEMP: MAIN
    if (A-6)>B {
      TEMP: TEMP + OFF
    }
  }
  result: TEMP
}

loop AC over {4..14}{
OTWF: OTWF + ((d20-2 > AC) * MAIN + (d20-2 > AC) * OFF)
MTWF: MTWF + (d20-2 > AC)* (MAIN + OFF)
NTWF: NTWF + [ntwf d20 vs AC]
NOFEAT: NOFEAT + [nofeat d20 vs AC]
THF: THF + (d20 > AC) * DOUBLE
SOBO: SOBO + (d20 > AC-2) * MAIN
}
\output NTWF
output THF\
\output OTWF
output MTWF\
output NOFEAT
output SOBO

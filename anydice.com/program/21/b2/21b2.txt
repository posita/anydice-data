function: attackroll with DICE:n target TARGET:n {
  SUCCESSES: [count {1..TARGET} in DICEd10]
  result: SUCCESSES
}

function: damage for ATTACK:n with DICE:n {
  HIT: ATTACK > 0
  CRIT: 0 
  if DICE > 2 {
    CRIT: ATTACK = DICE
  }
  result: HIT 
}

function: attack with DICE:n target TARGET:n {
  result: [damage for [attackroll with DICE target TARGET] with DICE]
}


output [attack with 1 target 7] named "1:7"
output [attack with 2 target 6] named "2:6"

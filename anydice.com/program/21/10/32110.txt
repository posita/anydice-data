loop N over {1, 2, 3, 4, 5, 7, 9, 10} {
  DIFICULDADE: N + 10
  NIVEL_PERSONAGEM: N-1
  PROFICIENCIA: [lowest of 5 and NIVEL_PERSONAGEM]
  PERICIA: [lowest of 5 and PROFICIENCIA+1]
  DOMINIO: [lowest of 5 and PROFICIENCIA+1]

  X: (3+DOMINIO)d(d20 >= DIFICULDADE)
  Y: (3+DOMINIO)d([count {DIFICULDADE-PERICIA..DIFICULDADE-1} in d20])
  output [lowest of 3 and X + [lowest of 1 and Y]] named "Desafio [N]"
}

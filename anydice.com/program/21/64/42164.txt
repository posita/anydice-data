STATIC: 8

loop SKILL over {0..10}
{
    if SKILL < 6 {
        output (STATIC-SKILL)d10+10*SKILL named "Rank [SKILL]"
    }
    loop ADVANTAGE over {1..3}
    {
        if SKILL < 3 {
            output {0..ADVANTAGE}@(ADVANTAGE*2)d10+(STATIC-SKILL-ADVANTAGE)d10+10*SKILL named "Rank [SKILL] Advantage [ADVANTAGE]"
        }
    }
    if SKILL > 5 {
        N: SKILL - 5
        output {1,2,3}@(STATIC-(5-N))d10+50 named "Rank [SKILL]"
    }
}

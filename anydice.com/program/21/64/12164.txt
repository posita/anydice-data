

function: attack ROLL:n plus BONUS:n vs AC:n hit HIT:d miss MISS:d crit CRIT:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: MISS
    }
    if ROLL >= CRIT_RANGE {
        result: CRIT
    }
    if ROLL + BONUS >= AC {
        result: HIT
    }
    result: MISS
}


function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

function: precise SUP:n attack ROLL:n vs AC:n for DMG:d {
    if ROLL + SUP >= AC {
        result: DMG
    }
    result: 0
}

function: battlemaster SUP_DICE:n superiority SUP:d precise PRECISE_RANGE:s attacks ATTACKS:n roll ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:s use USE:d pass PASS:d {
    if ROLL = 1 {
        result: PASS
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG + 2dSUP + USE
    }
    if [PRECISE_RANGE contains (ROLL + BONUS)] {
        result: [precise SUP attack ROLL + BONUS vs AC for DMG] + USE
    }
    if ROLL + BONUS >= AC {
        if SUP_DICE >= ATTACKS {
            result: DMG + SUP + USE
        }
        result: DMG + PASS
    }

    result: PASS
}

function: battlemaster SUP_DICE:n superiority SUP:d precise PRECISE_RANGE:s attacks ATTACKS:n roll ROLL:d plus BONUS:d vs AC:d for DMG:d crit CRIT:d on CRIT_RANGE:d {
    if ATTACKS = 0 {
        result: 0
    }
    if SUP_DICE = 0 {
        result: ATTACKSd[attack ROLL plus BONUS vs AC for DMG crit CRIT on CRIT_RANGE]
    }
    USE:  [battlemaster SUP_DICE - 1 superiority SUP precise PRECISE_RANGE attacks ATTACKS-1 roll ROLL plus BONUS vs AC for DMG crit CRIT on CRIT_RANGE]
    PASS: [battlemaster SUP_DICE     superiority SUP precise PRECISE_RANGE attacks ATTACKS-1 roll ROLL plus BONUS vs AC for DMG crit CRIT on CRIT_RANGE]
    result : [battlemaster SUP_DICE superiority SUP precise PRECISE_RANGE attacks ATTACKS roll ROLL plus BONUS vs AC for DMG crit CRIT on CRIT_RANGE use USE pass PASS]
}

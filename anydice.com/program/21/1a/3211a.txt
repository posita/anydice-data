function: savage D:d {
  result: 1@2dD
}

function: gwf die N:n {
  result: d{1..N:2, 3..N:N}
}

function: gwf S:s {
  SUM: 0
  loop N over S {
    SUM: SUM + [gwf die N]
  }
  result: SUM
}

GS: 2d6
GA: 1d12

output GA
output [gwf 12]
output [savage GA]
output [savage [gwf 12]]

output GS
output [gwf {6,6}]
output [savage GS]
output [savage [gwf {6,6}]]

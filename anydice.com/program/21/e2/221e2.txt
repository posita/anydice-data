TO_HIT_BONUS: 2
DAMAGE_DICE: 1d6
DAMAGE_BONUS: 2
NUMBER_OF_CREATURES: 3
MIN_AC: 15
MAX_AC: 21

function: attack D:n plus BONUS:n vs AC:n {
    if D = 20 { result: 1 } 
    else if D = 1 { result: 0}
    else if D + BONUS >= AC { result: 1 } 
    else { result:0}
}

loop AC over {MIN_AC..MAX_AC} {
    output [attack d20 plus TO_HIT_BONUS vs AC]  named "[AC]"
}

output [highest 1 of (DAMAGE_DICE + DAMAGE_BONUS) * NUMBER_OF_CREATURES] named "Max Damage"

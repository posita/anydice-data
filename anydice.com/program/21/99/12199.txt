\**** Damage calculator duelling Vs archery ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15}

ATTACKMOD: 3
DMGBONUS: 10
PROFICIENCY: 3
ACCBONUS: -3
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 1d10 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 1d10 + 10 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "heavy xbow ss"

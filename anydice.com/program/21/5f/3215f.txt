function: reroll D:n less than R:n up to N:n times {
  if N = 0 { result: D }
  \ if D = 1 { result: 1 } \
  if D >= R { result: D }
  result: [reroll 1d12 less than R up to N-1 times]
}

output [highest of 1d12 and 1d[reroll d12 less than 0 up to 1 times]]

output [highest of 1d12 and 1d[reroll d12 less than 6 up to 1 times]]
output [highest of 1d12 and 1d[reroll d12 less than 9 up to 1 times]]
output [highest of 1d12 and 1d[reroll d12 less than 12 up to 1 times]]

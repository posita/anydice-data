function: crit ROLL:d THRESHOLD {
 HITROLL: [highest 1 of ROLL]
 CRIT: [count 20 in HITROLL] > 0
 result: CRIT
}

function: attack ATKS DMG CRIT {
 TOTAL: 0
 loop I over {1..ATKS}{
  TOTAL: TOTAL+DMG
  TOTAL: TOTAL + (CRIT * [crit 1d20 20])
 }
 result: TOTAL
}

output [attack 2 1d4+3+1 4] named "2 1d4 claws"
output [attack 2 1d6+3+1 6] named "2 1d6 claws"
output [attack 3 1d4+3+1 4] named "3 1d4 claws"
output [attack 2 1d12+3+1 12] named "2 1d12 maul"

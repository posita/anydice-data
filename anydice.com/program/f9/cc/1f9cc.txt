function: near dupes in A:s B:s C:s {
  DICE: [sort {A, B, C}]
  DUPES: 0
  loop X over {1..#DICE} {
    PREV_MATCH: X > 1 & (X-1)@DICE <= X@DICE + 1
    NEXT_MATCH: X < #DICE & (X+1)@DICE >= X@DICE - 1
    if PREV_MATCH | NEXT_MATCH { DUPES: DUPES + 1 }
  }
  result: DUPES
}

output [near dupes in 4d12 {} {}]

\**** 2d6 Damage calculator with attack numbers ****\

function: damage ROLL:n vs DEF:n {
 if ROLL = 12 { result: 12 + ATTACKBONUS + DAMAGEBONUS }
 if ROLL + ATTACKBONUS > DEF { result: ROLL + ATTACKBONUS + DAMAGEBONUS - DEF } 
 else { result: 0 }
}

DEFENSE: 5
ATTACKBONUS: 3
DAMAGEBONUS: 4
ATTACKTIMES: 2

output ATTACKTIMESd[damage 2d6 vs DEFENSE] named "Damage"

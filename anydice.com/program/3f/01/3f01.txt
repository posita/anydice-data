\ function to find if a roll beats a specified DC with 1 auto-fail 20 auto-passing \
function: skill N:n DC:n
{
  if N=1 { result: 0 }
  if N=20 { result: 1 }
  result: N>=DC
}

\ this is the targets bonus to save \
SAVE: 8
\ player DC is buffed by this amount \
BUFF: 0

\ round 1: evil eye \
PEN: 0
DC: 16
output [skill d20+SAVE-PEN DC+BUFF] named "r1: evil eye"

\ round 2: ray of sickening \
PEN: 2
DC: 15
output [skill d20+SAVE-2 DC+BUFF] named "r2: ray o sick"

\ round 3: casting misfortune \
PEN: 4
DC: 16
output [skill d20+SAVE-PEN DC+BUFF] named "r3: misfortune"

\ if affected by misfortune they'll be rolling twice \
PEN: 4
DC: 16
output [skill [lowest of d20 and d20]+SAVE-PEN DC+BUFF] named "save VS something with Misfortune"

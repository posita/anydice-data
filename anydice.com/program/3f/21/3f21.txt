function: roll DIE:d with N:n TAR:n {
 if N < 0 { result: [deroll DIE with -N TAR] }
 if N > 0 { result: [reroll DIE with N TAR] }
 result: DIE >= TAR
}

function: deroll D:d with N:n TAR:n {
 if N > 0 { result: ((D >= TAR ) & [deroll D with (N-1) TAR])}
 result: (D >= TAR)
}

function: reroll R:d with N:n TAR:n {
 if N > 0 { result: ((R >= TAR ) | [reroll R with (N-1) TAR])}
 result: (R >= TAR)
}

\**** 2d6 Damage calculator with improved crits ****\

function: damagecrit ROLLA:n and ROLLB:n vs DEF:n {
 if ROLLA = 6 { if ROLLB = 6 { result: 12 + ATTACKBONUS + DAMAGEBONUS } }
 if ROLLA = 5 { if ROLLB = 5 { result: 10 + ATTACKBONUS + DAMAGEBONUS } }
 if ROLLA = 4 { if ROLLB = 4 { result: 8 + ATTACKBONUS + DAMAGEBONUS } }
 if ROLLA = 3 { if ROLLB = 3 { result: 6 + ATTACKBONUS + DAMAGEBONUS } }
 if ROLLA = 2 { if ROLLB = 2 { result: 4 + ATTACKBONUS + DAMAGEBONUS } }
 if ROLLA + ROLLB + ATTACKBONUS > DEF { result: ROLLA + ROLLB + ATTACKBONUS + DAMAGEBONUS - DEF } 
 else { result: 0 }
}

function: damage ROLL:n vs DEF:n {
 if ROLL = 12 { result: 12 + ATTACKBONUS + DAMAGEBONUS }
 if ROLL + ATTACKBONUS > DEF { result: ROLL + ATTACKBONUS + DAMAGEBONUS - DEF } 
 else { result: 0 }
}

DEFENSE: 7
ATTACKBONUS: 3
DAMAGEBONUS: 10
ATTACKTIMES: 1

output ATTACKTIMESd[damagecrit 1d6 and 1d6 vs DEFENSE] named "Damage"
output ATTACKTIMESd[damage 2d6 vs DEFENSE] named "Damage"

function: resolve A:n {
	if A < 0 { result: -1}
	if A = 0 { result: 0}
	if A > 0 & A < 3 { result: 1}
	if A >= 3 { result: 2}
}
function: succeed A:n {
	if A > 0 { result: 3}
	result: -2
}
F: 4d{-1,-1,0,0,1,1}
loop M over {-6..6} {
  output [resolve F - F + M] named "4df[M]"
  output [succeed F - F + M] named "4df[M]"
}

\function: roll DIE:d TAR N:n {
 if N < 0 { [deroll DIE -N] }
 if N > 0 { [reroll DIE N] }
 result: DIE
}\

function: deroll D:d N:n TAR:n {
 if N > 1 { result: ((D >= TAR ) | [deroll D (N-1) TAR])}
 result: (D >= TAR)
}

function: reroll R:d N:n TAR:n {
 if N > 1 { result: ((R >= TAR ) & [reroll R (N-1) TAR])}
 result: (R >= TAR)
}

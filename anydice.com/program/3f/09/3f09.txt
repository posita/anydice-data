function: reroll ATT:n DIE:d TAR:n TIM:n times{
 result: [rehelper DIE TIM ATT]
}

function: rehelper N:n T:n R:n {
 if T <= 0 { result: 1 }
 if N >= TAR & T > 0 { result: [rehelper DIE (T-1) R] }
 if N < TAR & R > 0 { result: [rehelper DIE T (R-1)] }
 if N < TAR & T > 0 { result: 0  }
}

loop N over {0..3} {
 loop D over {1..3} {
  output [reroll N 1@6d6 6 D times] named "[D]d [N]r"
 }
}
output 1@6d6 = 6
output [lowest of 1@6d6 and 1@6d6] = 6
output [lowest of [lowest of 1@6d6 and 1@6d6] and 1@6d6] = 6

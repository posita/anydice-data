AC: 12
PROFICIENCY: 3

function: attack DMG:d crit CRIT_DMG:d penalty PENALTY:n rider RIDER:n stat STAT:n roll ROLL:n{
    DMG_WITH_BONUS: DMG + STAT + RIDER
	
    if ROLL = 1 {
        result: 0
    }
    if ROLL = 20 {
        result: DMG_WITH_BONUS + CRIT_DMG
    }
    if ROLL + STAT + PROFICIENCY - PENALTY >= AC {
        result: DMG_WITH_BONUS
    }
    result: 0
}

named "+2 Attack ASI"

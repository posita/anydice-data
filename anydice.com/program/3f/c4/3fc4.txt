function: fate { 
    result: d{-1,0,1}
}


function: reroll ROLL:n REROLLS:n {
if ((ROLL=-1)&(REROLLS>0))
{
result: [reroll [fate] REROLLS-1]
}
else
{
result:ROLL
}
}

function: rerollturbo ROLL:n REROLLS:n {
if ((ROLL<1)&(REROLLS>0))
{
result: [rerollturbo [fate] REROLLS-1]
}
else
{
result:ROLL
}
}

output 4d[fate]
output 4d[reroll [fate] 1]
output 4d[rerollturbo [fate] 1]

AC: 20
CHA: 5
PROF: 5
MISF_CAST: 1
if MISF_CAST = 1 {
 MISFORTUNE: 1d{{2..6}:5, {1}}
}
else {
 MISFORTUNE: 0
}
EB: 1d{{2..10}:9, {1}} + 5 + MISFORTUNE

function: first A:n second B:n third C:n {
  
  if A + CHA + PROF >= AC & B + CHA + PROF >= AC & C + CHA + PROF >= AC {result: EB + EB + EB}
  if A + CHA + PROF >= AC & B + CHA + PROF >= AC {result: EB + EB}
  if A + CHA + PROF >= AC & C + CHA + PROF >= AC {result: EB + EB}
  if B + CHA + PROF >= AC & C + CHA + PROF >= AC {result: EB + EB}
  if A + CHA + PROF >= AC {result: EB}
  if B + CHA + PROF >= AC {result: EB}
  if C + CHA + PROF >= AC {result: EB}
  result: 0
}

output [first d20 second d20 third d20]

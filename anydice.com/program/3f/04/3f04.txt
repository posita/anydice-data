function: chaos TOTAL:n {
 HALF: TOTAL / 2
 CHECK: (TOTAL + 1) / 2
 if HALF = CHECK {
  result: TOTAL * -1
 }
 result: TOTAL * 2
}

function: chaosboost SIZE:n {
 result: [chaos [explode dSIZE]]
}

set "explode depth" to 3

output [explode d4] named "normal boost"
output [chaosboost 4] named "chaos boost"

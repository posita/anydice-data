B: 1
D: Bd{-3,-2,-1,1,2,3}
M: 0

N: A+B
W: (1+(2*A-B))/2
X: C+D+M
output X named "Raw Values"
Y: (X/2+W)
Z: [highest of Y and -(N+1)]
output [lowest of Z and (N+1)] named "Adjusted Scores"

function: fate { 
    result: d{-1,0,1}
}


function: reroll ROLL:n REROLLS:n {
if ((ROLL=-1)&(REROLLS>0))
{
result: [reroll [fate] REROLLS-1]
}
else
{
result:ROLL
}
}

function: rerollturbo ROLL:n REROLLS:n {
if ((ROLL<1)&(REROLLS>0))
{
result: [rerollturbo [fate] REROLLS-1]
}
else
{
result:ROLL
}
}

function: underexplode ROLL:n {
if (ROLL<1)
{
result: ROLL + [fate]
}
else
{
result:ROLL
}
}

function: overexplode ROLL:n {
if (ROLL>-1)
{
result: ROLL + [fate]
}
else
{
result:ROLL
}
}

function: minusexplode ROLL:n {
if (ROLL=-1)
{
result: ROLL + [fate]
}
else
{
result:ROLL
}
}

function: zeroexplode ROLL:n {
if (ROLL=0)
{
result: ROLL + [fate]
}
else
{
result:ROLL
}
}

function: plusexplode ROLL:n {
if (ROLL=1)
{
result: ROLL + [fate]
}
else
{
result:ROLL
}
}

output 4d[fate]
output 4d[reroll [fate] 1]
output 4d[rerollturbo [fate] 1]
output 4d[minusexplode[fate]]
output 4d[zeroexplode[fate]]
output 4d[plusexplode[fate]]
output 4d[underexplode[fate]]
output 4d[overexplode[fate]]

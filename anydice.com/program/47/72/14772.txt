\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
if ROLL >= CRITROLL { result: CRITDAMAGE }
if ROLL = 1 { result: 0 }
if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 2
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 2d10 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d10 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Pistol +3 Proficiency +4 Dexterity Modifier and Archery Style"

\ A variant of exploding dice, where you replace the maximum result of one die with two new dice. \

NUMBEROFDICE: 5    \ THESE ARE THE ONLY VARIABLES... \
DIE: 6           \ ...THAT YOU NEED TO CHANGE. \

function: ob NUMBER:n {
  TOTALSUM: 0
  loop LOOPING over {1..NUMBER} {
    X: 1dDIE
    X: [checkmax X]
    TOTALSUM: TOTALSUM + X
  }
  result: TOTALSUM
}

function: checkmax N:n {
  if N = DIE {
    result: [ob 2]
  }
  result: N
}

set "maximum function depth" to 14    \ Change this to a lower number if you get a timeout. \

output [ob NUMBEROFDICE] named "OB[NUMBEROFDICE]T[DIE] "

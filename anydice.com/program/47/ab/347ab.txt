function: poolsix DSP:s {
 result: [highest of 0 and [count 6 in DSP] - ([count 1 in DSP]/2)]
}

function: poolfive DSP:s {
 result: [highest of 0 and [count 6 in DSP] - ([count 1 in DSP]/2) + [count 5 in DSP]]
}

loop N over {1..6}{
  output [poolsix Nd6] named "[N]d6=6"
}
loop N over {1..6}{
  output [poolfive Nd6] named "[N]d6=5+"
}

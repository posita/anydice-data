\ Comparing a 1d10 with frequent advantage to a 1d12 without \

NORMAL_AR: 1d20
ADVANTAGE_AR: [ highest 1 of 2d20 ]
FIFTY_FIFTY: d{0, 1}
MISS: 0
NORMAL_HIT: 1
CRITICAL_HIT: 2
CRITICAL_THRESHOLD: 20
STR_MOD: 5
PROFICIENCY: 3
UNSEEN_MENACE: 1d10
GREATSWORD: 2d6
RAGE_BONUS: 2

function: hit with BONUS:n vs AC:n rolling ROLL:n {
   if ROLL >= CRITICAL_THRESHOLD { result: CRITICAL_HIT }

   if ROLL = 1 { result: MISS }

   if ROLL + BONUS - AC >= 0 { result: NORMAL_HIT }

   result: MISS
}

function: attack with BONUS:n vs AC:n rolling ROLL:n with DICE:d and MOD:n {
   HIT: [ hit with BONUS vs AC rolling ROLL ]

   if HIT = MISS { result: 0 }
   
   if HIT = CRITICAL_HIT { result: DICE + DICE + MOD }
   
   result: DICE + MOD
}

function: determine roll with PROB:n {
  if PROB = 1 {
    result: ADVANTAGE_AR
  }
  result: NORMAL_AR
}

DAMAGE_BONUS: STR_MOD + RAGE_BONUS
HIT_BONUS: STR_MOD + PROFICIENCY
TARGET_AC: 22

CRITICAL_THRESHOLD: 21
output [ hit with HIT_BONUS vs TARGET_AC rolling NORMAL_AR ] named "Odds of hit (No advantage)"

output [ hit with HIT_BONUS vs TARGET_AC rolling ADVANTAGE_AR ] named "Odds of hit  (Advantage)"

CRITICAL_THRESHOLD: 20
WEAPON: UNSEEN_MENACE
output [ attack with HIT_BONUS vs TARGET_AC rolling ADVANTAGE_AR with WEAPON and DAMAGE_BONUS ] named "Unseen Menace"

MAYBE_ADV: [ determine roll with FIFTY_FIFTY ]
output MAYBE_ADV named "Adjusted roll"
output [ attack with HIT_BONUS vs TARGET_AC rolling MAYBE_ADV with WEAPON and DAMAGE_BONUS ] named "Unseen Menace adjusted"

WEAPON: GREATSWORD
output [ attack with HIT_BONUS vs TARGET_AC rolling NORMAL_AR with WEAPON and DAMAGE_BONUS ] named "Greatsword"

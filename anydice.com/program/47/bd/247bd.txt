STR: 21
INT: 28
STRROLL: [highest 1 of 2d20]+5 >= STR

function: oneround strh STRHELP:n strb STRBONUS:n inth INTHELP:n intb INTBONUS:n  {
  PERCENT: {}
  STRBASE: 1d20+5
  if STRHELP {
    STRBASE: [highest 1 of 2d20]+5
  }
  if STRBONUS {
    STRBASE: STRBASE+1d4
  }
  STRROLL: STRBASE >= STR
  result: [strroll STRROLL INTHELP INTBONUS]

}

function: strroll STRROLL:s INTHELP:n INTBONUS:n {
  INTBASE: [lowest 1 of 2d20]+9

  if STRROLL = 1 & INTHELP {
    INTBASE: [highest 1 of 2d20]+9
  }
  if (STRROLL = 1 & !INTHELP) | (STRROLL = 0 & INTHELP) {
    INTBASE: 1d20+9
  } 
  if INTBONUS {
    INTBASE: INTBASE+1d4
  }
  INTROLL: INTBASE >= INT
  result: INTROLL
}

function: allrounds N:n strh STRHELP:n strb STRBONUS:n inth INTHELP:n intb INTBONUS:n {
  TOTAL: 0
  loop Y over {1..N}{
    CHANCE: [oneround strh STRHELP strb STRBONUS inth INTHELP intb INTBONUS]
    TOTAL: TOTAL | CHANCE
  }
  result: TOTAL
}

function: range on DIE:n between LOW:n and HIGH:n {
  if DIE >= LOW & DIE <= HIGH { result: 1 }
  result: 0
}

output [range on 1d6 between 6 and 6]
output [range on 1d6 between 4 and 5]
output [range on 1d6 between 1 and 3]

output [range on 1d6+1d4 between 6 and 10]
output [range on 1d6+1d4 between 4 and 10]
output [range on 1d6+1d4 between 1 and 3]

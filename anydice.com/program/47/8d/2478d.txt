\ ------------------------------------------ \
\ ENDURING TRAIT                             \
\ TRAIT BONUS Roll 2d4 for STAMINA roll      \
\ SET BONUS   Roll 4d2 for STAMINA roll      \
\ ------------------------------------------ \
\ TIME TRAIT                                 \
\ TRAIT BONUS Get +1 extra STAMINA each turn \
\ SET BONUS   ADV ON STAMINA ROLLS           \
\ ------------------------------------------ \

function: advantage each of A B {
  COUNT: 0
  loop N over {1..A} {
    COUNT: COUNT + [highest 1 of 2d(B)]
  }
  result: COUNT
}

output 1d8 named "base stamina roll"
output 2d4 named "enduring trait bonus"
output 4d2 named "enduring set bonus"
output 1d8+1 named "time trait bonus"
output [advantage each of 1 8]+1 named "time set bonus"
output [advantage each of 2 4]+1 named "time set bonus + enduring trait bonus"
output [advantage each of 4 2]+1 named "time set bonus + enduring set bonus"

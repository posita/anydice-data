function: range on DIE:n between LOW:n and HIGH:n {
  if DIE >= LOW & DIE <= HIGH { result: 1 }
  result: 0
}

output [range on 1d6 between 5 and 6]
output [range on 1d6 between 3 and 4]
output [range on 1d6 between 1 and 2]

output [range on 1d6+1d4 between 5 and 10]

function: explode N:n {
 if N = 6 { result: 1 + [explode d6] }
 result: 0
}

function: base N:n T:n {
 if N < T { result: 0 }
 if N < 6 { result: 1 }
 result: 1 + [explode d6]
}

function: truncate A:n {
 N: 3
 if A > N { result: N }
 result: A
}

set "maximum function depth" to 4
TARGET: 4
loop N over {1..3} {
 output [truncate Nd[base d6 TARGET]] named "[N]d6"
 output [truncate Nd[base d6+1 TARGET]] named "[N]d6+1"
}

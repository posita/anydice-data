\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + MAGICBONUS + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: savingthrow ROLL:n vs DC:n {
 if ROLL + PROFICIENCY + ATTACKMOD + MAGICBONUS + ACCBONUS >= DC { result: SAVEDAMAGE }
 result: DAMAGE
}

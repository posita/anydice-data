function: successes in DICE:s {
  TOTALCOUNT: 0
  loop I over {1..5} {
    COUNT: 0
    loop J over {1..#DICE} {
      if (J@DICE = I) {COUNT: COUNT + 1}
    }
    if (COUNT > TOTALCOUNT) {TOTALCOUNT: COUNT}
  }

  loop I over {1..#DICE} {
    if (I@DICE = 6) {TOTALCOUNT: TOTALCOUNT + 1}
  }

  result: TOTALCOUNT
}

loop X over {3..8} {
  output [successes in Xd6] >= 3 named "[X]d6"
}

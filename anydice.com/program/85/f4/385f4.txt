output "d4, not exploded" named "d4, not exploded"

loop DEPTH over {1..1} {
    set "explode depth" to DEPTH
    
    function rollD4WithAdvantageAndExplode(depth) {
        roll1 = rollD4()
        roll2 = rollD4()
        highest = max(roll1, roll2)
        
        if highest == 4 and depth > 0 {
            additional_roll = rollD4WithAdvantageAndExplode(depth - 1)
            return 4 + additional_roll
        } else {
            return highest
        }
    }
    
    exploded_result = rollD4WithAdvantageAndExplode(DEPTH)
    
    output exploded_result named "d4, exploded at most [DEPTH] times"
}

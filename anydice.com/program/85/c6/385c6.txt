function: roll {
  result: 1d10
}
function: interlock N:n STAT:n SKILL:n {
  if N = 1 { result: 0 }
  if N = 10 { result: STAT + SKILL + 10 + [explode [roll]] }
  result: STAT + SKILL + N
}

STAT: 5
SKILL: 5
ROLL: [roll]
output [highest of [lowest of [interlock ROLL STAT SKILL] - (10-1) and 1] and 0] named "Easy (10)"
output [highest of [lowest of [interlock ROLL STAT SKILL] - (15-1) and 1] and 0] named "Average (15)"
output [highest of [lowest of [interlock ROLL STAT SKILL] - (20-1) and 1] and 0] named "Difficult (20)"
output [highest of [lowest of [interlock ROLL STAT SKILL] - (25-1) and 1] and 0] named "Very Difficult (25)"

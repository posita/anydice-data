ACMIN: 10
ACMAX: 20
TOHITBONUS: 10
DMGONHIT: 15


function: chancetohitall TOHIT {
   range: {ACMIN .. ACMAX}
   R: {}
   loop N over {1 .. #range} {
      N@range = N@range*[chancetohitspecific TOHIT to {ACMIN .. ACMAX}]
   }
   result: R
}

function: chancetohitspecific TOHIT to AC {
   HITCOUNT: 0
   loop M over {1 .. 20} {
      if M + TOHIT >= AC {
         HITCOUNT: HITCOUNT + 1
      }
   }
   result: HITCOUNT*100/20
}

output [chancetohitspecific TOHITBONUS to 15]
output [chancetohitall TOHITBONUS]

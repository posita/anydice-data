function: multis in DICE:s {
  D: 1
  loop X over {1..#DICE} {
    loop Y over {(X+1)..#DICE} {
      if (Y@DICE = X@DICE) { D: 2}
    }
  }
  loop X over {1..#DICE} {
    loop Y over {(X+1)..#DICE} {
      loop Z over {(Y+1)..#DICE} {
        if (Z@DICE = Y@DICE) {
          if (Y@DICE = X@DICE) {D: 3}
        }
      }
    }
  }
  result: D
}

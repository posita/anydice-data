function: chancetohitspecific TOHIT to AC {
   HITCOUNT: 0
   loop M over {1 .. 20} {
      if M + TOHIT >= AC {
         HITCOUNT: HITCOUNT + 1
      }
   }
   result: HITCOUNT*100/20
}

output [chancetohit TOHITBONUS to 20]

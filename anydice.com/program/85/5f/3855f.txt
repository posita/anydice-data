ADV: 6
THT: 2

SUC: 1d8
FAIL: 1d8
ADVANTAGE: ADVd{0:4, 1:2}
THREAT: THTd{0:4, 1:2}

function: success S:s and A:s {
 if A > 0 & S < 6 {
      result: 6
 }
 else {
  result: S
 }
}
function: failure F:s and A:s {
 if A < 0 {
   result: F - #(A)
 }
 else {
  result: F
 }
}
function: advtht ADV:s and THT:s {
 result: ADV - THT
}

output 2d8 named "baseline"
output [success SUC and [advtht ADVANTAGE and THREAT]] + [failure FAIL and [advtht ADVANTAGE and THREAT]] named "success/failure"
output [advtht ADVANTAGE and THREAT] named "adv/tht"

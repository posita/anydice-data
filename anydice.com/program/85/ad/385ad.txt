\
  ASSUMPTIONS:
  Using no resources, only features that are always available
  Nat 20 is crit, nat 1 is miss
  Enemies make saves at +3
  Wound only counts once
  Two enemies are in range for AoEs
  Enemy has no resistances
\
set "explode depth" to 3

\ Static Variables \
MISS: 0
HIT: 1
MIGHTY: 2
POWER: 3
CRIT: 4

\ Mutable Variables \
STR: 3
PROF: 2

SAVE_DC: 8+STR+PROF
BONUS: STR+PROF


\ Determine what kind of hit happens \
function: attack ROLL:n BON:n vs AC:n {
  if ROLL = 20 { result: CRIT }
  if ROLL + BON >= AC + 10 { result: POWER }
  if ROLL + BON >= AC + 5 { result: MIGHTY }
  if ROLL = 1 | ROLL + BON < AC { result: MISS }
  result: HIT
}

\ Standard attack, nothing special \
function: damage DICE:d DAM:n for ATTACK:n {
  if ATTACK = CRIT { result: [maximum of DICE]+DICE+DAM }
  if ATTACK = MISS { result: 0 }
  result: DICE+DAM
}

\ Wound does an extra damage die on mighty blow or better (including crit, for simplicity) \
function: wound damage D_NUM:n D_SIZE:n DAM:n for ATTACK:n {
  if ATTACK = CRIT { result: [maximum of D_NUMdD_SIZE]+(D_NUM+1)dD_SIZE+DAM }
  if ATTACK >= MIGHTY { result: (D_NUM+1)dD_SIZE+DAM }
  if ATTACK = MISS { result: 0 }
  result: D_NUMdD_SIZE+DAM
}

\ Keen adds 1 damage die on powerful blow or crit \
function: keen damage D_NUM:n D_SIZE:n DAM:n for ATTACK:n {
  if ATTACK = CRIT { result: [maximum of D_NUMdD_SIZE]+(D_NUM+1)dD_SIZE+DAM }
  if ATTACK = POWER { result: (D_NUM+1)dD_SIZE+DAM }
  if ATTACK = MISS { result: 0 }
  result: D_NUMdD_SIZE+DAM
}

\ Puncture explodes damage dice \
function: puncture damage D_NUM:n D_SIZE:n DAM:n for ATTACK:n {
  if ATTACK = CRIT { result: [maximum of D_NUMdD_SIZE]+D_NUMd[explode dD_SIZE]+DAM } \ Crit damage isn't a roll \
  if ATTACK = MISS { result: 0 }
  result: D_NUMd[explode dD_SIZE]+DAM
}

\ Keen but also explodes \
function: keen puncture damage D_NUM:n D_SIZE:n DAM:n for ATTACK:n {
  if ATTACK = CRIT { result: [maximum of D_NUMdD_SIZE]+(D_NUM+1)d[explode dD_SIZE]+DAM }
  if ATTACK = POWER { result: (D_NUM+1)d[explode dD_SIZE]+DAM }
  if ATTACK = MISS { result: 0 }
  result: D_NUMd[explode dD_SIZE]+DAM
}

\ Balanced rerolls ones once \
function: reroll ones N:n for D:d{
  if N = 1 { result: D } else { result: N }
}

function: balanced damage DICE:d DAM:n for ATTACK:n {
  if ATTACK = CRIT { result: [maximum of DICE]+[reroll ones DICE for DICE]+DAM }
  if ATTACK = MISS { result: 0 }
  result: [reroll ones DICE for DICE]+DAM
}

\ Balanced but also explodes \
function: balanced puncture damage D_NUM:n D_SIZE:n DAM:n for ATTACK:n {
  if ATTACK = CRIT { result: [maximum of D_NUMdD_SIZE]+D_NUMd[reroll ones [explode dD_SIZE] for [explode dD_SIZE]]+DAM }
  if ATTACK = MISS { result: 0 }
  result: D_NUMd[reroll ones [explode dD_SIZE] for [explode dD_SIZE]]+DAM
}

\ Flat attacks \
\ Template: output [damage 1d6 STR for [attack d20 BONUS vs AC]] named "" \
loop AC over {10..20} {
  output 0 named "AC [AC]"
  \ Light \
  output [keen puncture damage 1 4 STR for [attack d20 BONUS vs AC]] named "Sickle"
  output [puncture damage 1 4 STR for [attack d20 BONUS vs AC]] named "Light Pick"
  output [puncture damage 1 4 (STR+(2*PROF)) for [attack d20 (BONUS-PROF) vs AC]] named "Light Pick (Punch)"
  output [damage 1d6 STR for [attack d20 BONUS vs AC]] named "Scimitar/Light Hammer"
  output [damage 1d6 (STR+(2*PROF)) for [attack d20 (BONUS-PROF) vs AC]] named "Light Hammer (Punch)"
  output [balanced puncture damage 1 4 STR for [attack d20 BONUS vs AC]] named "Hand Spear"

  \ 1H \

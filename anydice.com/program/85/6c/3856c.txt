ADV: 5
THT: 1

ROLL: 2d8
ADVANTAGE: ADVd{0:4, 1:2}
THREAT: THTd{0:4, 1:2}

function: roll R:s and A:s {
 LOW: 1@R
 HIGH: 2@R
 if A > 0 & LOW < 6 {
      result: 6 + HIGH
 }
 if A < 0 & LOW < 6 {
      result: 1 + HIGH
 }
 else {
  result: R
 }
}
function: advtht ADV:s and THT:s {
 result: ADV - THT
}

output 2d8 named "baseline"
output [roll ROLL and [advtht ADVANTAGE and THREAT]] named "roll"
output [advtht ADVANTAGE and THREAT] named "adv/tht"

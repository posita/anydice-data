\ helper function: given two dice, return a new die distributed according \
\ to the normalized pointwise product of their probability mass functions \
function: pointwise product of A:n and B:n {
  if A = B { result: A } else { result: d{} }
}

\ given a die, return a new die with all probability mass concentrated \
\ at the most likely outcome(s) of the original die \
function: most likely result of DIE:d {
  loop I over {1..20} {
    DIE: [pointwise product of DIE and DIE]  \ square the probabilities \
  }
  result: DIE
}

loop N over {1..4} {
  output Nd6 named "[N]d6"
  output [most likely result of Nd6] named "most likely result of [N]d6"
  output Nd6 = [most likely result of Nd6] named "maximum probability of [N]d6"
}

function: pointwise product of A:n and B:n {
  if A = B { result: A } else { result: d{} }
}
function: if COND:n then A:d else B:d {
  if COND { result: A } else { result: B }
}
function: highest probability of A:d and B:d {
  X: [pointwise product of A and 1-B]
  loop I over {1..10} {
    X: [pointwise product of X and X]
  }
  result: [if X then A else B]
}

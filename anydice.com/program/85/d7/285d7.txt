function: compare A:n to B:n
{
S:0
M:0
F:0

loop I over {1..#A}
{
loop V over {1..#B}
{
if I@A = V@B {S: S + 1}
if I@A = (V@B)-1 {M: M+1}
if I@A = (V@B)+1 {M: M+1}
else {F: F+1}
}
}

if S >= 1 {result: 2}
if M >= 1 {result: 1}
else {result: 0}

}

loop L over {1..4}
{
loop Z over {1..4}
{
output [compare Ld6 to Zd6] named "[L]d vs. [Z]d"
}
}

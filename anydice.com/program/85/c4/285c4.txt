function: attack ROLL:n plus BONUS:n vs TARGET_AC:n hit HIT_DIE:d miss MISS_DIE:d crit CRIT_DIE:d on CRIT_RANGE:n {
    if ROLL = 1	{ result: MISS_DIE }
    if ROLL >= CRIT_RANGE { result: CRIT_DIE }
    if ROLL + BONUS >= TARGET_AC { result: HIT_DIE }
	result: MISS_DIE
}
function: attack ROLL:n plus BONUS:n vs AC:n for DAMAGE_DIE:d crit CRIT_DAMAGE_DIE:d on CRIT_RANGE:n {
    result: [attack ROLL plus BONUS vs AC hit DAMAGE_DIE miss 0 crit DAMAGE_DIE + CRIT_DAMAGE_DIE on CRIT_RANGE]
}

function: save ROLL:n plus BONUS:n vs SAVE_DC:n success SUCCESS_DIE:d failure FAILURE_DIE:d {
	if ROLL + BONUS >= SAVE_DC { result: SUCCESS_DIE }
	result: FAILURE_DIE
}

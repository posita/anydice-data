\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + MAGICBONUS + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
PROFICIENCY: 4
MAGICBONUS: 1
DMGBONUS: 0
EXTRADICESIZE: 6
ACCBONUS: 0
CRITROLL: 20
SNEAK: 5
DAMAGE: 1d6 + ATTACKMOD + MAGICBONUS + DMGBONUS + 1dEXTRADICESIZE + SNEAKd6
CRITDAMAGE: 2d6 + ATTACKMOD + MAGICBONUS + DMGBONUS + 2dEXTRADICESIZE + {2*SNEAK}d6
output [attackroll d20 vs ENEMYAC] named "+1 Weapon, 1d6 extra, SA"

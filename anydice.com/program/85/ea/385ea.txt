function: explode N:n {
 if N = 6 { result: 1 + [explode d6] }
 result: 0
}

function: base N:n T:n B:n {
 if N >= 6 { result: 1 + [explode d6] }
 if N + B >= T { result: 1 }
 result: 0
}

function: truncate A:n {
 N: 3
 if A > N { result: N }
 result: A
}

set "maximum function depth" to 4
TARGET: 3
loop N over {1..2} {
 loop BONUS over {0..1} {
  output [truncate Nd[base d6 TARGET BONUS]] named "[N]d6+[BONUS]"
 }
}

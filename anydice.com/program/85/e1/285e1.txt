function: teamwork of A:s and B:s
{

MATCH: 0

loop I over {1..#A}
{

loop L over {1..#B}
{
if I@A = L@B {MATCH:2}
if MATCH != 2
{
if I@A-1 = L@B {MATCH: 1}
if I@A+1 = L@B {MATCH: 1}
}
}

}

result: MATCH

}

loop L over {1..4}
{
loop Z over {L..3}
{
output [teamwork of Ld6 and Zd6] named "[L] dice helping/helped by [Z] dice"
}
}

FUDGE: {-1, 0, +1}

function: evall ROLL:s {
  X: [count -1 in ROLL]
  Y: [count 1 in ROLL]
  Z: [count 0 in ROLL]
 
  if X > Y & X > Z {result: -X}
  if Y > X & Y > Z {result: Y}
  else {result: 0}
}
function: evali ROLL:s {
  X: [count -1 in ROLL]
  Y: [count 1 in ROLL]
 
  if X > Y {result: -X}
  if Y > X {result: Y}
  else {result: 0}
}

output [evall 4dFUDGE]
output [evali 4dFUDGE]

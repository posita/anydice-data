TURNS: 30
TURN_DIE: d{1,2,3,4,5,6}

SIXES_ROLLED: 0

CHANCE: 1

loop P over {0..TURNS}
 {
   \if P > 3 { TURN_DIE: d{1,2,3,4,6,6} }
   if P > 6 { TURN_DIE: d{1,2,3,6,6,6} }
   if P > 9 { TURN_DIE: d{1,2,6,6,6,6} }\

   SIXES_ROLLED: SIXES_ROLLED + [TURN_DIE contains 6]

   TURNCOUNT: P + 2
   \output SIXES_ROLLED named "Crash slots on turn [TURNCOUNT]"\

   output  SIXES_ROLLED named "Crash slots more than five on turn [TURNCOUNT]"

   \output [count {0..CRASH_SLOTS} in TURN_DIE]\
 }

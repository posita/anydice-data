\RESULTS\
MISS: 0
HIT: 1
CRITICALHIT: 2

\FUNCTIONS\

function: attack ROLL:n using WEAPON:s vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= CRIT_MIN {
  result: CRITICALHIT
 }
 if ROLL + 1@WEAPON + 2@WEAPON + 3@WEAPON + ATK_BONUS >= DEFENSE { 
  result: HIT
 }
 result: MISS
}

function: damage for ATTACK_RESULT:n using WEAPON:s {
 if ATTACK_RESULT = CRITICALHIT {
  result: 2dDAMAGE_DICE + 1@WEAPON + 2@WEAPON + DMG_BONUS
 }
 if ATTACK_RESULT = HIT {
  result: 1dDAMAGE_DICE + 1@WEAPON + 2@WEAPON + DMG_BONUS
 }
 result: 0
}

function: average damage for ROLL:n using WEAPON:s vs DEFENSE:n {
 result: [damage for [attack ROLL using WEAPON vs DEFENSE] using WEAPON]
}

function: finesse {
 if DEXTERITY > STRENGTH {
  result: DEXTERITY
 }
 result: STRENGTH
}

function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

function: reroll A:n under B:n as C:n {
  if (A < B) { result: C }
  result: A
}

function: slidingbonus X:n {
  if (X <= 11) { result: X+5 }
  if (X <= 16) { result: X+(5-(X-11)) }
  result: X
}

output 3d6
output [highest 3 of 4d6]
output 3d[reroll d6 under 3 as d6]
output [slidingbonus 3d[reroll d6 under 2 as d6]]

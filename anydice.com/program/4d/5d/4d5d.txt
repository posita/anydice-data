

function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d10] }
 result: N >= 8
}

NWOD: [nwod 10 again d10]








function: nwodraw P:n {
 if P < 8 { result: 0 }
 result: (( P - 8 ) / 3 ) + 1
}

NWODRAW: [nwodraw d10]

loop N over {1..8} {
 output [lowest of 10 and NdNWODRAW] named "[N]draw"
}

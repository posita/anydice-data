\ Assumptions \
\ 20 Primary stat \
\ +4 Proficiency \
\ No feats or fighting styles (to keep things simple) \
\ Using Longsword and Greatsword only \
\ Average AC of enemies is 12 to 22 (probably a little low for level 9+ party) \


\ Generic Magic Weapons \
function: ROLL:n with ATTACKBONUS:n for DMGDIE plus DMGBONUS:n vs AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL = 20 { result: DMGDIE + DMGDIE + DMGBONUS}
 result: (ROLL + ATTACKBONUS >= AC) * (DMGDIE + DMGBONUS) }
output [d20 with +10 for 1d8 plus 6 vs d{12..22}] named "Magic Longsword +1"
output [d20 with +10 for 2d6 plus 6 vs d{12..22}] named "Magic Greatsword +1"
output [d20 with +11 for 1d8 plus 7 vs d{12..22}] named "Magic Longsword +2"
output [d20 with +11 for 2d6 plus 7 vs d{12..22}] named "Magic Greatsword +2"


\ Sword of Sharpness at various magic levels with the +14 bonus damage only on a nat 20 \
function: ROLL:n with ATTACKBONUS:n for DMGDIE plus DMGBONUS:n vs AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL = 20 { result: 2d6 + DMGBONUS + 14 }
 result: (ROLL + ATTACKBONUS >= AC) * (DMGDIE + DMGBONUS) }
\ Longsword \
output [d20 with +9 for 1d8 plus 5 vs d{12..22}] named "Longsword of Sharpness +0" 
output [d20 with +10 for 1d8 plus 6 vs d{12..22}] named "Longsword of Sharpness +1" 
output [d20 with +11 for 1d8 plus 7 vs d{12..22}] named "Longsword of Sharpness +2" 
\ Greatsword \
output [d20 with +9 for 2d6 plus 5 vs d{12..22}] named "Greatsword of Sharpness +0" 
output [d20 with +10 for 2d6 plus 6 vs d{12..22}] named "Greatsword of Sharpness +1" 
output [d20 with +11 for 2d6 plus 7 vs d{12..22}] named "Greatsword of Sharpness +2" 

\ Sword of Sharpness with the +14 bonus damage on any d20 + mods at 20 or higher \
function: ROLL:n with ATTACKBONUS:n for DMGDIE plus DMGBONUS:n vs AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL = 20 { result: 2d6 + DMGBONUS + 14 }
 if ((ROLL + ATTACKBONUS >= 20) & (ROLL + ATTACKBONUS >= AC)) { result: DMGDIE + DMGBONUS + 14 }
 result: (ROLL + ATTACKBONUS >= AC) * (DMGDIE + DMGBONUS) }
output [d20 with +9 for 1d8 plus 5 vs d{12..22}] named "Longsword of Sharpness (high dmg)"
output [d20 with +9 for 2d6 plus 5 vs d{12..22}] named "Greatsword of Sharpness (high dmg)"

 if ROLL = 1 { result: 0 }
 if ROLL = 20 { result: DMGDIE + DMGDIE + DMGBONUS + 2d6}

output [d20 with +11 for 2d6 plus 7 vs d{12..22}] named "Hazirawn (Normal)"

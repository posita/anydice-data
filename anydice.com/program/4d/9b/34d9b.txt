function: tn N:n for D:d {
 result: [highest of [explode D] and [explode d6]]>=N
}

loop STAGE over {0,1,2,3,4}{
 loop TN over {4,6,8,9,10}{
  DICEUP: 4+STAGE*2
  output [explode d DICEUP] >= TN named "d[DICEUP][[TN]]"
  output [highest of [explode d DICEUP] and [explode d6]] >= TN  named "wildcard d[DICEUP][[TN]]"
  output [highest of [explode d DICEUP] and [explode d6]] >= (TN+4)  named "wildcard raise  d[DICEUP][[TN]]"
 }
}

function: lose server on machine TURN:n rolling ROLL:s
{
 SLOTS: 0
 loop P over {2..TURN}
 {
   if [ROLL contains 6] {SLOTS: SLOTS + 1}
 }

 result: [count {0..SLOTS} in 1d6]
}

TURNS: 100
TURN_DIE: d{1,2,3,4,5,6}

CHANCE: 1

loop P over {1..TURNS}
 {
   CHANCE: CHANCE - [lose server on machine P rolling TURN_DIE]
   output CHANCE < 1 named "Chance Lose Server on Turn [P]"  \ NOTE: THIS CHANCE INCREASES EACH TIME LOSE CARD \
 }

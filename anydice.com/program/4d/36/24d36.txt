function: mydice {
  result: {0,0,0,1,1,1}
}

function: dangerdice {
  result: {0, -1}
}

function: adjustroll R:s {
  if R < -1 {
    result: -1
  }
  if R > 2 {
    result: 3
  }
  result: R
}

function: roll N M {
  result: Nd[mydice] + Md[dangerdice]
}

function: realroll N M {
  result: [adjustroll [roll N M]]
}


loop N over {1..6} {
  loop M over {1..6} {
    output [realroll N M] named "roll [N] difficulty [M]"
  }
}

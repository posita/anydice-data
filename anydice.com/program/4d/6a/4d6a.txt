function: sum SEQUENCE:s{
  TOTAL: 0
  loop P over {1..#SEQUENCE}{
    TOTAL: TOTAL + P@SEQUENCE
  }
  result: TOTAL
}

function: foo ROLL:s WANT:n REPLACE:n {
  SUM: [sum ROLL]
  if (SUM >= WANT) {
    result: SUM
  } else {
    result: SUM - 1@[reverse [sort ROLL]] + REPLACE
  }
}

output [foo 4d6 15 1d6]

}

}

    if (R + HIT_MOD) < TARGET_AC {result: 0}
}

function: R:n spellattack{
    if (R + TARGET_SPELL_SAVE_BONUS) < SPELL_DC {result: SPELL_DICE d SPELL_DMG_DIE}
    if (R + TARGET_SPELL_SAVE_BONUS) >= SPELL_DC {result: (SPELL_DICE d SPELL_DMG_DIE)/2}
}

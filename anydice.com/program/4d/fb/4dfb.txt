{

    if CONFIRM + HITADD >= AC
    {
      CRITICAL: 0
      loop N over {1..MULTIPLIER}
      {
        CRITICAL: CRITICAL + DAMAGE
      }
      result: CRITICAL
    }
    else
    {
      result: DAMAGE
    }
  }

  {
    result: 0
  }
  else if HIT = 1
  {
    result: 0
  }
  else
  {
    result: DAMAGE
  }
}

\to-hit calculation\
function: hit F:n H:n I:n J:n{
 if H>F { result:0 }
 else { if J<=I { result:0 }
 else { result:1 }
 }
}

WPNROLL: 3d6
DEFROLL: 3d6

\Damage vs DR calculation\
\A= DAM, B= DR, C= Pen, D= Wounding E= Follow up\
function: damage A:n B:n C:n D:n E:n{
 if C=0 { B: B*2 C: 1}
 if C=1000 {B: 0}
 if D=0 { X: (A - B / C) / 2}
 else {X: (A - B / C) * D}
 if X < 1 {
  if E < B {result: 0}
  else {result: E*[hit WPN WPNROLL DEF DEFROLL]}
 }
 else {result: (X+E)*[hit WPN WPNROLL DEF DEFROLL]}
}

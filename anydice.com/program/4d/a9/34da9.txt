\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + ATTACKMOD >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
\ ENEMYAC: d{15..20} \

ENEMYAC: 17

ATTACKMOD: 6
DMGBONUS: 4
CRITROLL: 20
DAMAGE: 1d8 + DMGBONUS
CRITDAMAGE: 2d8 + DMGBONUS
output [attackroll 1d20 vs ENEMYAC] named "TRANDOSHAN Blaster Rifle"
output [attackroll [highest 1 of 2d20] vs ENEMYAC] named "TRANDOSHAN Blaster Rifle (Advantage)"

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
PROFICIENCY: 5
HEX: 1d6
SMITE: 1d8 + 5d8
DMGBONUS: 5
ACCBONUS: 5
CRITROLL: 19
ATTACKTIMES: 2
DAMAGE: 1d4 + HEX + ATTACKMOD + DMGBONUS + HEX + ATTACKMOD + DMGBONUS + SMITE
CRITDAMAGE: 2 * (1d4 + HEX + SMITE) + 2 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Hexblade Celestial Whip"

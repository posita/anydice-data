function: N:n on D:s successes {
  if {1..#D}@D >= N { result: 1 } result: 0
}

function: check N:n{
  SUM: 0
  H: N/2
  loop DICE over {2..H} {
    D:DICE*2
    SUM: SUM + [4 on 1dD successes]
  }
  result: SUM
}

output [check 4] named "d4"
output [check 6] named "d6"

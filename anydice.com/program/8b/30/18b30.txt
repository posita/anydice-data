STR:4
PROF:3
MISC:1
WEPDIE:1d12
ADVANTAGE:0

ATTACK:STR+PROF+MISC
DAMAGE:STR+MISC+WEPDIE
GWMVALUE: {-5,10}

function: advantage of ADV {
if ADV > 0 {D: [highest 1 of 2d20]}
if ADV < 0 {D: [lowest 1 of 2d20]}
if ADV = 0 {D: 1d20}
result: D
}

DIE: [advantage of ADVANTAGE]  

loop AC over {10..20} {
  output (DIE+ATTACK >=AC)* (DAMAGE) named "without gwm vs [AC]"
  output (DIE+ATTACK+1@GWMVALUE >=AC)* (DAMAGE+2@GWMVALUE) named "with gwm vs [AC]" 
}

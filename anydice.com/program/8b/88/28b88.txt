function: focus DICE:n dice of difficulty DIFF:n and focus FOCUS:n {
    FAIL_COUNT: DIFF - 1
    CRIT_COUNT: FOCUS
    SUCCESS_COUNT: 20 - (FAIL_COUNT + CRIT_COUNT)
    
    result: DICEd{0:FAIL_COUNT, 1:SUCCESS_COUNT, 2:CRIT_COUNT}
}

loop N over {3..8} {
    loop D over {10} {
        loop F over {2} {
            output [focus N dice of difficulty D and focus F] named "[N] Dice with Difficulty [D] and [F] Focus"
        }
    }
}

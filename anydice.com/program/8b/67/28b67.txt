            function: roll ROLL:n at plus ATTACKBONUS:n vs ac AC:n for CRITABLEDAM:d plus HITDAM:d {
                if ROLL >= CRITRANGE {
                    result:  CRITABLEDAM + CRITABLEDAM + HITDAM
                }
                if ROLL = 1 {
                    result: 0
                }
                if ROLL >= AC - ATTACKBONUS {
                    result: CRITABLEDAM + HITDAM
                }
                else {
                    result: 0
                }
            }

            function: roll ROLL:n at plus ATTACKBONUS:n vs ac AC:n for CRITABLEDAM:d plus HITDAM:d with booming secondary at LEVEL: n{
                SECONDARY: [booming secondary LEVEL dice 1d20]

                if ROLL >= CRITRANGE {
                    result:  CRITABLEDAM + CRITABLEDAM + HITDAM + SECONDARY
                }
                if ROLL = 1 {
                    result: 0
                }
                if ROLL >= AC - ATTACKBONUS {
                    result: CRITABLEDAM + HITDAM + SECONDARY
                }
                else {
                    result: 0
                }
            }

            \from anydice tutorial https://www.reddit.com/r/3d6/comments/gf111s/anydice_tutorial_part_3_state_the_great_weapon/\
            function: attack ROLL:n plus BONUS:n vs AC:n hit HIT:d miss MISS:d crit CRIT:d on CRIT_RANGE:n {
                if ROLL = 1 {
                    result: MISS    
                }
                if ROLL >= CRIT_RANGE {
                    result: CRIT
                }
                if ROLL + BONUS >= AC {
                    result: HIT
                }
                result: MISS
            }

            \Flurry function is recursive\
            function: flurry TIMES:n times at level LEVEL:n using AC_SCALING:s {
                \recursion base case\
                if TIMES = 0 {
                    result: 0
                }
                ATT_BONUS: LEVEL@GS_MOD_PROGRESSION+LEVEL@PROF_BONUSES+2-5
                AC: LEVEL@AC_SCALING
                MOD: LEVEL@GS_MOD_PROGRESSION+10
                CRIT: 1d6
                ON_HIT: [flurry TIMES-1 times at level LEVEL using AC_SCALING]+CRIT+MOD
                ON_CRIT: [flurry TIMES-1 times at level LEVEL using AC_SCALING]+CRIT+MOD+CRIT
                D: [roll 1@2d20 at plus ATT_BONUS vs ac AC for CRIT plus MOD] \baseline damage of 1 shot\

                \On a miss, we immediately use the flurry reroll, so damage on miss for this shot is D, and other shots have no flurry effect\
                if TIMES = 3 {
                    ON_MISS: D+D+D 
                }
                if TIMES = 2 {
                    ON_MISS: D+D
                }
                if TIMES = 1 {
                    ON_MISS: D
                }        
                result: [attack 1@2d20 plus ATT_BONUS vs AC hit ON_HIT miss ON_MISS crit ON_CRIT on CRITRANGE]

            }

            \TM baseline warlock with Agonizing Blast and Hex\
            function: warlock LEVEL:n using AC_SCALING:s {
                NUM_BLAST: LEVEL@EB_SCALING
                ATT_BONUS: LEVEL@WARLOCK_MOD_PROGRESSION+LEVEL@PROF_BONUSES
                CRIT: 1d10+1d6 \hex\
                if LEVEL >= 2 {
                    NO_CRIT: LEVEL@WARLOCK_MOD_PROGRESSION \Agonizing\
                } else {
                    NO_CRIT: 0
                }
                AC: LEVEL@AC_SCALING
                TOTAL: 0
                W: [roll 1d20 at plus ATT_BONUS vs ac AC for CRIT plus NO_CRIT]
                loop I over {1..NUM_BLAST}{
                    TOTAL: TOTAL + W
                }
                result: TOTAL    
            }

            function: fighter action surge LEVEL:n using AC_SCALING:s {
                if LEVEL >= 2 {
                    NUM_SHOT: LEVEL@EA_SCALING +LEVEL@EA_SCALING+ 1 \Crossbow expert\
                } else {

                CRIT: 1d6
                if LEVEL >= 4 {

                    NO_CRIT: LEVEL@FIGHTER_MOD_PROGRESSION+10 \Sharpshooter\

                loop I over {1..NUM_SHOT}{

                }
                result: TOTAL

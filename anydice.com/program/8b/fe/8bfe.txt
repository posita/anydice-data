AC: 10
DIE: 1d6
CRIT: 2
ATTACKBONUS: 0
DAMAGE: 0


MARGINALBONUS: ATTACKBONUS+1
MARGINALDAMAGE: DAMAGE+1

Z: (AC-1)
ATTACKROLL: 1d{{0}:Z-ATTACKBONUS,{1}:20-(Z-ATTACKBONUS)}
MARGINALATTACKROLL: 1d{{0}:Z-MARGINALBONUS,{1}:20-(Z-MARGINALBONUS)}
CRITICALATTACKROLL: 1d{{0}:Z-ATTACKBONUS,{1}:20-(Z-ATTACKBONUS)-1,CRIT}

ATTACK: ATTACKROLL*(DIE+DAMAGE)
EXPLODINGATTACK: ATTACKROLL*([explode DIE]+DAMAGE)

output ATTACK named "Base: +[ATTACKBONUS] Attack and [DIE]+[DAMAGE] Damage against AC [AC]"
\output ATTACKROLL*(DIE+MARGINALDAMAGE) - ATTACK named "+1 Damage"\
output MARGINALATTACKROLL*(DIE+DAMAGE) - ATTACK named "+1 Attack"
output EXPLODINGATTACK - ATTACK named "Exploding Damage"
output CRITICALATTACKROLL*(DIE+DAMAGE) - ATTACK named "Crit [CRIT]"

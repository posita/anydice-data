function: rolls DICE:n threshold DC:n{
 SUCCEEDS: 0
 loop N over {1..DICE} {
  ROLL: [explode 1d10]
  SUCCEEDS: SUCCEEDS + [decide ROLL DC]
 }
 result: SUCCEEDS
}
function: decide ROLL:n DC:n {
 if ROLL >= DC {
  result: 1 + (ROLL-DC)/4
 } 
 result: 0
} 
function: judge RESULTS:n {
 if RESULTS >= 4 {
  result: MORE
 } 
 if RESULTS = 3 {
  result: ONE
 } 
 result: FAIL
}
FAIL: 0
ONE: 1
MORE: 2
function: brolls DICE:n threshold DC:n{
 SUCCEEDS: 0
 loop N over {1..DICE} {
  ROLL: [explode 1d10]
  SUCCEEDS: SUCCEEDS + [decide ROLL DC]
 }
 result: [judge SUCCEEDS]
}

output [rolls 1 threshold 5] named "0 Stress, distributed"
output [brolls 1 threshold 5] named "0 Stress, collected"

function: skill SKILL:n with ROLL:s plus A:n and B:n {
if SKILL = 0 { result: ROLL }
if SKILL > 0 { R: [sort {1@ROLL, 2@ROLL, 3@ROLL, A}] }
if SKILL > 1 { R: [sort {1@R, 2@R, 3@R, B}] }
result: R
}

output [skill 0 with 4d6 plus d6 and d6] named "0"
output [skill 1 with 4d6 plus d6 and d6] named "1"
output [skill 2 with 4d6 plus d6 and d6] named "2"
output [skill 2 with 5d6 plus d6 and d6] named "3"

\ D6 System roll probabilities \
set "explode depth" to 5
function: roll DICE:n {
 if DICE < 1 {result: 0}
 result: [rolldetail [explode d6] against DICE-1]
}
function: rolldetail WILD:n against POOL {
 DIE: POOL d6
 if WILD = 1 {
  if POOL <= 1 { result: 0 }
  DIE: [lowest POOL-1 of DIE]
  WILD: 0
 }
 result: DIE + WILD
}
loop AD over {1..16}{
 output [roll AD]
}

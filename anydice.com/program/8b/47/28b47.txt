function: attack with HIT:s plus HITMOD to hit AC ac for DAMAGE:s plus DAMAGEMOD damage {
  if HIT = 20 {
    result: DAMAGE + DAMAGE + DAMAGEMOD
  }
  else if HIT + HITMOD >= AC {
    result: DAMAGE + DAMAGEMOD
  }
  else {
    result: 0
  }
}

DRAGONWING: [attack with d20 plus 6 to hit 20 ac for 2d6 plus 15 damage]
output DRAGONWING named "Dragon wing hand crossbow"

PLUSTWO: [attack with d20 plus 8 to hit 20 ac for d6 plus 17 damage]
output PLUSTWO named "+2 hand crossbow"

DRAGONWINGWITHPRECISION: [attack with d20+d10 plus 6 to hit 20 ac for 2d6 plus 15 damage]
output DRAGONWINGWITHPRECISION named "Dragon wing hand crossbow with Precision maneuver"

PLUSTWOWITHPRECISION: [attack with d20+d10 plus 8 to hit 20 ac for d6 plus 17 damage]
output PLUSTWOWITHPRECISION named "+2 hand crossbow with Precision maneuver"

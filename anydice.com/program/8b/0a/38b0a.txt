function: iscrit ROLL {
  result: ROLL = 20
}

function: ROLL tohit BONUS vs DEF damage DMG bonus DMGBONUS {
  if [iscrit ROLL] {
    result: 2*DMG + DMGBONUS
  } else {
    result: 0
    \if [fix ROLL+BONUS > DEF] {
      result: DMG+DMGBONUS
    } else {
      result: 0
    }\
  }
}
output [1d20 tohit +5 vs 15 damage 1d8 bonus +3]

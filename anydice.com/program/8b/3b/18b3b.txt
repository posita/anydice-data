loop R over {-1, 0, 1, 2, 3} {
  
  \loop A over {9, 10, 11, 12, 14, 15, 16, 17} {\
  loop A over {10, 13, 15, 17} {
    K: R+3
    N: K+1
    if K < 3 {
      K: 3
      N: K
    }
    output [highest K of Nd6] <= A named "[A] >= [N]d6 worst [K]"
  }
  
  \loop A over {-1..5} {\
  loop A over {-1..3} {
    TN: (R+1)*2+6
    output 2d6+A >= TN named "2d6 [A] >= [TN]"
  }
  
}

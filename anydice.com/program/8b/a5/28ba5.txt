\ '''This script covers rogue variants
    Here is a key:
    DP - Double Phantom, as built by ramen_goblin, is a straight Shadar Kai rogue, with Elven accuracy at 4, ritual caster at 8, ASI (DEX) at 10, then defensive feats after
    AT - Arcane Trickster with Wizard 1 dip at level 2. Starts Custom Lineage (Moderately armored), with mobile at 5, sentinel at 9 and then an ASI (DEX) at 11. Gimmick is using Mirror Image + Sentinel to provoke off turn sneak attacks; superceded by other builds, and sentinel could probably be better taken earlier.
    MR - "martial rogue" is an Arcane Trickster with a 4 level dip in Battlemaster Fighter at levels 2-5. Starts Custom Lineage (Sentinel), with Squat Nimbleness at 5 for the extra move and DEX improvement. ASI (DEX) at 8 rounds off the main build. Here, off turn sneak attacks are provoked via battlemaster maneuvers or sentinel or action surge, every round, and familiar is used for advantage on regular turns.
    BA - "Battlemaster Assassin" is an Assassin / Battlemaster hybrid, starting CL(sentinel) again, magic initiate (wizard) at 5 for famiiar and bladetrips, and squat nimbleness at 8, ASI (DEX) at 12. Assume that 50% of combats start with surprise, and use action surge to double crit sneak attack on these rounds. For other 7 rounds to next short rest, 4 get off turn sneak attacks via battlemaster maneuvers, while the last 3 get sneak attacks from sentinel.
    '''\

function: roll ROLL:n at plus ATTACKBONUS:n vs ac AC:n for CRITABLEDAM:d plus HITDAM:d {
    if ROLL >= CRITRANGE {
        result:  CRITABLEDAM + CRITABLEDAM + HITDAM
    }
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= AC - ATTACKBONUS {
        result: CRITABLEDAM + HITDAM
    }
    else {
        result: 0
    }
}

function: roll ROLL:n at plus ATTACKBONUS:n vs ac AC:n for CRITABLEDAM:d plus HITDAM:d with booming secondary at LEVEL: n{
    SECONDARY: [booming secondary LEVEL dice 1d20]

    if ROLL >= CRITRANGE {
        result:  CRITABLEDAM + CRITABLEDAM + HITDAM + SECONDARY
    }
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= AC - ATTACKBONUS {
        result: CRITABLEDAM + HITDAM + SECONDARY
    }
    else {
        result: 0
    }
}

function: wail LEVEL:n {
    DICE: (LEVEL+1)/2
    WAIL_DICE: (DICE+1)/2
    result: WAIL_DICEd6

}

\BB primary\
function: booming primary LEVEL: n {
    DICE:LEVEL@BB_SCALING
    result: DICEd8
}

\BB secondary\
function: booming secondary LEVEL:n dice PROB:n {
    DICE:LEVEL@BB_SCALING+1
    if PROB <= BOOMING_CHANCE {
        result: DICEd8
    }
    else {
        result: 0
    }
}

    MOD: LEVEL@BA_MOD_PROGRESSION
    ROG_LVL: LEVEL@BA_ROG_LEVELS
    SNEAK: [sneak ROG_LVL]
    BOOM_PRIMARY: [booming primary LEVEL]

    MAIN_ATT: [roll 1@2d20 at plus ATT_BONUS vs ac AC for CRIT+BOOM_PRIMARY plus MOD+2 with booming secondary at LEVEL]
    SENTINEL_ATT: [roll 1d20 at plus ATT_BONUS vs ac AC for CRIT plus MOD+2]
    ATT_DAM_SUPDIE: [roll 1d20 at plus ATT_BONUS vs ac AC for CRIT+1d8 plus MOD+2]
    SENTINEL: [proc sentinel die 1d20 chance RAW_SENTINEL proc SENTINEL_ATT fail ATT_DAM_SUPDIE]
    NON_NOVA: MAIN_ATT + SENTINEL
    
    \#Nova calculation - assumes surprise, so advantage, and autocrit, making 2 booming blade swings. \
    \# attack damage and dice are moved to "mod" section because it auto crits and we don't want to crit the already crit damage\
    NOVA_ATT: [roll 1@2d20 at plus ATT_BONUS vs ac AC for 0 plus CRIT+BOOM_PRIMARY+CRIT+BOOM_PRIMARY+MOD+2 with booming secondary at LEVEL]
    NOVA: NOVA_ATT+NOVA_ATT
    
    TOTAL: NOVA+NON_NOVA+NON_NOVA+NON_NOVA+NON_NOVA+NON_NOVA+NON_NOVA+NON_NOVA
    
    result: TOTAL/8 \#averaged over 8 rounds\
    
    
}

\#TM baseline warlock with Agonizing Blast and Hex\
function: warlock LEVEL:n using AC_SCALING:s {
    NUM_BLAST: LEVEL@EB_SCALING
    ATT_BONUS: LEVEL@WARLOCK_MOD_PROGRESSION+LEVEL@PROF_BONUSES
    CRIT: 1d10+1d6 \hex\
    if LEVEL >= 2 {
        NO_CRIT: LEVEL@WARLOCK_MOD_PROGRESSION \Agonizing\
    } else {
        NO_CRIT: 0
    }
    AC: LEVEL@AC_SCALING
    TOTAL: 0
    W: [roll 1d20 at plus ATT_BONUS vs ac AC for CRIT plus NO_CRIT]
    loop I over {1..NUM_BLAST}{
        TOTAL: TOTAL + W
    }
    result: TOTAL
}

\#Subclass-less fighter with archery (lvl1), sharpshooter (lvl4), crossbow expert (lvl1)\
function: fighter LEVEL:n using AC_SCALING:s {
    NUM_SHOT: LEVEL@EA_SCALING + 1 \Crossbow expert\
    CRIT: 1d6
    if LEVEL >= 4 {
        ATT_BONUS: LEVEL@FIGHTER_MOD_PROGRESSION+LEVEL@PROF_BONUSES + 2 - 5 \Archery+Sharpshooter\
        NO_CRIT: LEVEL@FIGHTER_MOD_PROGRESSION+10 \Sharpshooter\
    } else {
        ATT_BONUS: LEVEL@FIGHTER_MOD_PROGRESSION+LEVEL@PROF_BONUSES + 2 \Archery\
        NO_CRIT: LEVEL@FIGHTER_MOD_PROGRESSION
    }
    AC: LEVEL@AC_SCALING
    TOTAL: 0
    loop I over {1..NUM_SHOT}{
        TOTAL: TOTAL + [roll 1d20 at plus ATT_BONUS vs ac AC for CRIT plus NO_CRIT]
    }
    result: TOTAL
}


\#Subclass-less fighter with archery (lvl1), sharpshooter (lvl4), crossbow expert (lvl1); using action surge\
function: fighter action surge LEVEL:n using AC_SCALING:s {
    if LEVEL >= 2 {
        NUM_SHOT: LEVEL@EA_SCALING +LEVEL@EA_SCALING+ 1 \Crossbow expert\
    } else {
        NUM_SHOT: LEVEL@EA_SCALING + 1 \Crossbow expert\
    }

    CRIT: 1d6
    if LEVEL >= 4 {
        ATT_BONUS: LEVEL@FIGHTER_MOD_PROGRESSION+LEVEL@PROF_BONUSES + 2 - 5 \Archery+Sharpshooter\
        NO_CRIT: LEVEL@FIGHTER_MOD_PROGRESSION+10 \Sharpshooter\
    } else {
        ATT_BONUS: LEVEL@FIGHTER_MOD_PROGRESSION+LEVEL@PROF_BONUSES + 2 \Archery\
        NO_CRIT: LEVEL@FIGHTER_MOD_PROGRESSION
    }
    AC: LEVEL@AC_SCALING
    TOTAL: 0
    loop I over {1..NUM_SHOT}{
        TOTAL: TOTAL + [roll 1d20 at plus ATT_BONUS vs ac AC for CRIT plus NO_CRIT]
    }
    result: TOTAL
}

    if LEVEL <= 3 {
        result: [ba twothree LEVEL using AC_SCALING]
    }
    if LEVEL <= 4 {
        result: [ba four LEVEL using AC_SCALING]
    }
    if LEVEL <= 6 {
        result: [ba fivesix LEVEL using AC_SCALING]
    }
    if LEVEL <= 20 {
        result: [ba sevenplus LEVEL using AC_SCALING]
    }
}

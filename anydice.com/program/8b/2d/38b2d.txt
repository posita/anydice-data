INT_MOD: 5
PROF_BONUS: 6
set "maximum function depth" to 15

function: is prone by AC and STR {
  result: 1d20 + INT_MOD + PROF_BONUS >= AC & 1d20+STR < 8 + INT_MOD + PROF_BONUS
}

function: apply adv by IS_PRONE:n {
  if IS_PRONE {
    result: [highest 1 of 2d20]
  } else {
    result: 1d20
  }
}

function: crit DMG with ADD by ATK:n and BONUS and AC {
  if ATK = 20 {
    result: DMG + DMG + ADD
  } else if ATK + BONUS >= AC{
    result: DMG + ADD
  } else {
    result: 0
  }
}

function: champcrit DMG with ADD by ATK:n and BONUS and AC {
  if ATK >= 18 {
    result: DMG + DMG + ADD
  } else if ATK + BONUS >= AC{
    result: DMG + ADD
  } else {
    result: 0
  }
}

function: sharpcrit DMG with ADD by ATK:n and BONUS and AC {
  if ATK = 20 {
    result: DMG + DMG + ADD + 1d6 + 10
  } else if ATK + BONUS - 5>= AC{
    result: DMG + ADD + 10
  } else {
    result: 0
  }
}

function: add PER_TURN per turn to COUNT DAMAGE_ROLL with CONST:s by ADV_COUNT and BONUS and AC {
  TOTAL: 0
  USED: 0
  loop N over {1..COUNT}{
    if N <= ADV_COUNT {
      ATK: [highest 1 of 2d20]
    } else {
      ATK: 1d20
    }
    if USED {
      TOTAL: TOTAL + [crit (DAMAGE_ROLL) with N@CONST by ATK and BONUS and AC]
    } else {
      NEW: TOTAL + [crit (DAMAGE_ROLL + PER_TURN) with N@CONST by ATK and BONUS and AC]
      if {NEW} >= {TOTAL} {
        USED: 1
      }
      TOTAL: NEW
    }
  }
  result: TOTAL
}

function: champadd PER_TURN per turn to COUNT DAMAGE_ROLL with CONST by ADV_COUNT and BONUS and AC {
  TOTAL: 0
  USED: 0
  loop N over {1..COUNT}{
    if N <= ADV_COUNT {
      ATK: [highest 1 of 2d20]
    } else {
      ATK: 1d20
    }
    if USED {
      TOTAL: TOTAL + [champcrit (DAMAGE_ROLL) with CONST by ATK and BONUS and AC]
    } else {
      NEW: TOTAL + [champcrit (DAMAGE_ROLL + PER_TURN) with CONST by ATK and BONUS and AC]
      if {NEW} >= {TOTAL} {
        USED: 1
      }
      TOTAL: NEW
    }
  }
  result: TOTAL
}

function: sharpadd PER_TURN per turn to COUNT DAMAGE_ROLL with CONST:s by ADV_COUNT and BONUS and AC {
  TOTAL: 0
  USED: 0
  loop N over {1..COUNT}{
    if N <= ADV_COUNT {

    } else {

function: smite at SLEVEL {
  result: 1d8 + SLEVELd8
}

function: repeat ROLL times TIMES {
  TOTAL: 0
  loop N over {1..TIMES} {
    TOTAL: TOTAL + ROLL
  }
  result: TOTAL
}

function: repeat ROLL until MOD beats DC {
  TOTAL: 0
  ENDED: 0
  loop N over {1..5} {
    if {1d20 + MOD} >= DC {
      ENDED: 1
    }
    if !ENDED {
      TOTAL: TOTAL + ROLL
    }
  }
  result: TOTAL
}

\ This program calculates the probability of a certain amount of hits with Assault Missiles or Plasma Torpedoes and Guidance Chips without a focus. \

function: ROLL:n dice{
  if ROLL = 400 { result: 4 } \ 4 Hits \
  if ROLL > 300 { result: 4 } \ 3 Hits + Chips \
  if ROLL > 200 { result: 3 } \ 2 Hits + Chips \
  if ROLL > 100 { result: 2 } \ 1 Hit  + Chips \
  result: 1                   \ 0 Hits + Chips \
}

\ This is the actual dice roll. 4 dice are rolled, each of them has the faces: 1, 1, 10, 10, 100, 100, 100, 100 \
\ 1 = Blank, 10 = Eye, 100 = Hit or Crit \

output [4d{1,1,10,10,100,100,100,100} dice]

LOWESTMOD: 1
HIGHESTMOD: 1
\change LOWESTMOD and HIGHESTMOD to use different modifiers \
loop N over {LOWESTMOD..HIGHESTMOD} {
  A: d4
  B: d6
  C: d8
  D: d10
  output d20 + A + N named "[A] + [N]"
  output d20 + B + N named "[B] + [N]"
  output d20 + C + N named "[C] + [N]"
  output d20 + D + N named "[D] + [N]"
}

\ This program calculates the probability of a certain amount of hits with Homing Missiles with a focus. \

function: ROLL:n dice with focus and chips {
  if ROLL >= 30 { result: 4 }                                                      \ 3 or more hits result in 4 hits automatically \
  if ROLL >= 10 { result: ROLL/10 + [[ROLL modulo 10] dice with focus and chips] } \ count all hits, then analyze the rest \
  if ROLL >= 1  { result: 2 }                                                      \ if at least one eye: 2 hits b/c of chips and Prot.Torps \
  result: 1                                                                        \ if no eye: 1 hit b/c of chips \
}

function: A:n modulo B:n {
  result: A-(A/B)*B
}

\ This is the actual dice roll. 4 dice are rolled, each of them has the faces: 0, 0, 1, 1, 10, 10, 10, 10 \
\ 0 = Blank 1 = Eye, 10 = Hit or Crit \

output [4d{0,0,1,1,10,10,10,10} dice with focus and chips]

ATTACKER_TROOPS: 4
DEFENDER_TROOPS: 2

function: min A:n and B:n{
    if A < B { result: A }
    else { result: B }
}
ROLL: 2d{0..3}
ATK_HITS: 1@ROLL + (DEFENDER_TROOPS=0)
DEF_HITS: 2@ROLL 

output [min ATK_HITS and ATTACKER_TROOPS]named "Attacker Hits" 
output [min DEF_HITS and DEFENDER_TROOPS]named "Defender Hits"

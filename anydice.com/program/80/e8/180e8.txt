LEVEL: 7
ROLLTYPE: 0

\ 0 = Regular rolling \
\ 1 = Average \
\ 2 = Roll but reroll 1's \

function: base HD CON
{
  result: D + LEVEL*C
}

function: hp with con C hd D
{
  if ROLLTYPE = 0 {
   result: [base D C] + (LEVEL-1)dD
  }
  
  if ROLLTYPE = 1 {
    result: [base D C] + (LEVEL-1)*(D/2+1)
  }
  if ROLLTYPE = 2 {
    result: [base D C] + (LEVEL-1)d{2..D}
  }
}

output [hp with con 2 hd 12] named "Brynna - HP at level [LEVEL]:"

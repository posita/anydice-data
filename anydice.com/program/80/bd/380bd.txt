function: minimum of DIE:d {
 result: 1@{DIE}
}

function: hit 
dice    A:n
dicemin AM
dicemax AA
atk     B 
alvl    C 
dlvl    D 
dodge   E 
weapon  F:n 
dr      G 
{
 if(A <= AM)
 {
    result : 0
 }

 if(A >= AA)
 {
    result : ((A + B + C - D - E) / 5) + (2 * F) - G
 }

 if( (A + B + (C-D)) < E) {
    result : 0
 }
 result : [highest of 0 and (((A + B + C - D - E) / 10) + F - G)]
}

output [hit dice 2d8 dicemin [minimum of 2d8] dicemax [maximum of 2d8] atk 1 alvl 1 dlvl 1 dodge 9 weapon d6 dr 0] named "Base Lvl 1 vs Lvl 1"

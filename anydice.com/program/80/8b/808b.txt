BLANK: 0
FOCUS: 1
HIT: 10
ATTACK_DIE: {BLANK, BLANK, FOCUS, FOCUS, HIT, HIT, HIT, HIT}
NUM_DICE: 4
ATTACK_ROLL: NUM_DICEdATTACK_DIE

function: A:n modulo B:n {
    result: A-(A/B)*B
}

function: count NEEDLE:n in HAYSTACK:s {
    COUNT: 0
    loop VALUE over HAYSTACK {
        if VALUE = NEEDLE {
            COUNT: COUNT + 1
        }
    }
    result: COUNT
}

function: attack roll {
    result: NUM_DICEdATTACK_DIE
}

function: hits ROLL:n {
    result: ROLL / 10
}

function: focuses ROLL:n {
    result: [ROLL modulo HIT]
}

function: blanks in N:n dice ROLL:n {
    result: N - [hits ROLL] - [focuses ROLL]
}

function: blanks ROLL:n {
    result: [blanks in NUM_DICE dice ROLL]
}

function: focus ROLL:n {
    result: ([hits ROLL] + [focuses ROLL]) * HIT
}

function: target lock N:n dice ROLL:n save SAVED:s {
    SAVED_BLANKS: [count BLANK in SAVED]
    SAVED_FOCUSES: [count FOCUS in SAVED]
    if SAVED_BLANKS > [blanks in N dice ROLL] {
        SAVED_BLANKS: [blanks in N dice ROLL]
    }
    if SAVED_FOCUSES > [focuses ROLL] {
        SAVED_FOCUSES: [focuses ROLL]
    }
    TL_ROLL: [target lock N - SAVED_BLANKS - SAVED_FOCUSES dice ROLL]
    result: TL_ROLL + SAVED_FOCUSES
}
function: target lock N:n dice ROLL:n {
    HITS: [hits ROLL]
    NOT_HITS: N - HITS
    result: HITS * HIT + NOT_HITSdATTACK_DIE
}

function: target lock ROLL:n save SAVED:s {
    result: [target lock NUM_DICE dice ROLL save SAVED]
}

function: target lock ROLL:n {
    result: [target lock NUM_DICE dice ROLL]
}

function: target lock blanks ROLL:n {
    BLANKS: [blanks ROLL]
    TL_ROLL: [target lock BLANKS dice BLANKSdATTACK_DIE]
    result: ROLL + TL_ROLL
}

function: proton torpedoes ROLL:n {
    if [focuses ROLL] > 0 {result: ROLL - FOCUS + HIT}
    result: ROLL
}

function: advanced proton torpedoes ROLL:n {
    if ([blanks ROLL] > 3) {
        ROLL: ROLL + FOCUS * 3
    } else {
        ROLL: ROLL + FOCUS * [blanks ROLL]
    }
    result: ROLL
}

function: concussion missiles ROLL:n {
    if [blanks ROLL] > 0 {result: ROLL + HIT}
    result: ROLL
}

function: guidance chips HITS:n {
    if HITS < NUM_DICE * HIT { result: HITS + HIT }
    result: HITS
}

output [hits [guidance chips [proton torpedoes [target lock [attack roll] save {1}]]]] named "target locked proton torpedoes"

output [hits [guidance chips [focus [concussion missiles [attack roll]]]]] named "focussed concussion missiles"

NUM_DICE: 5

output [hits [guidance chips [focus [advanced proton torpedoes [attack roll]]]]] named "focused advanced proton torpedoes"

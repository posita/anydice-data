AC: 30
GDR: 39
GDSGDR: 44

function: monster BASE attacks with WEAPON {
  result: [highest of 0 and 1dBASE + 1dWEAPON - 1d3 - 1]
}

function: apply ac AC to DAMAGE with GDR gdr and MAX maxdamage {
  REDUCEDBY: 1d(AC+1) - 1
  REDUCEDBY: [highest of REDUCEDBY and [lowest of GDR*MAX/100 and AC/2]]
  result: [highest of 0 and DAMAGE - REDUCEDBY]
}

OGREAHANDA: [monster 20 attacks with 20]
OGREAHANDB: [monster 15 attacks with 20]

output OGREAHANDA + OGREAHANDB named "unreduced"
output [apply ac AC to OGREAHANDA with 0 gdr and 20+20 maxdamage] + [apply ac AC to OGREAHANDB with 0 gdr and 20+15 maxdamage] named "30 AC draconian"
output [apply ac AC to OGREAHANDA with GDR gdr and 20+20 maxdamage] + [apply ac AC to OGREAHANDB with GDR gdr and 20+15 maxdamage] named "30 AC in plate"
output [apply ac AC to OGREAHANDA with GDSGDR gdr and 20+20 maxdamage] + [apply ac AC to OGREAHANDB with GDSGDR gdr and 20+10 maxdamage] named "30 AC in GDS"

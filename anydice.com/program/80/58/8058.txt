\ This program calculates the probability of a certain amount of hits with Homing Missiles without a focus. \

function: ROLL:n dice with rerolls and chips{
  if ROLL = 4 { result: 4 }
  if ROLL = 3 { result: 4 }            \ 3 Hits + Chips \
  if ROLL = 2 { result: 2 + [roll 2d{0,0,0,0,1,1,1,1} chipped 2 times] }  \ 2 Hits + Chips and 2 Rerolls  \
  if ROLL = 1 { result: 1 + [roll 3d{0,0,0,0,1,1,1,1} chipped 3 times] }  \ 1 Hit  + Chips and 3 Rerolls \
  result: [roll 4d{0,0,0,0,1,1,1,1} chipped 4 times]                      \ 0 Hits + Chips and 4 Rerolls \
}

function: roll ROLL:n chipped S:n times {
  if ROLL = S { result: S }
  result: ROLL + 1
}

\ This is the actual dice roll. 4 dice are rolled, each of them has the faces: 0, 0, 0, 0, 1, 1, 1, 1 \
\ 0 = Blank/Eye, 1 = Hit or Crit \

output [4d{0,0,0,0,1,1,1,1} dice with rerolls and chips]

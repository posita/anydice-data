function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

output d12 named "Plain d12"
output [reroll {1,2} on d12 to d12] named "Old GWF d12"
output d{3:2,3..12} named "New GWF d12"

output 2d6 named "Plain 2d6"
output 2d[reroll {1,2} on d6 to d6] named "Old GWF 2d6"
output 2d{3:2,3..6} named "New GWF 2d6"

function: aw A:n {
   if A >= 10 { result: 2 }
   if A >= 7 { result: 1 }
   result: 0
}

function: awbad A:n {
   if A >= 10 { result: 0 }
   if A >= 7 { result: 1 }
   result: 2
}


loop N over {1..3} {
output [awbad 2d6 + N] named "+[N]Harm; high is bad"
}

loop N over {1,0,-1} {
output [aw 2d6 + N] named "-[N]Harm; high is good"
}

function: reroll DICE:n under NUMBER:n {
	GWD: {{1..NUMBER-1}:NUMBER-1, {NUMBER..DICE}:DICE+NUMBER-1}

	result: dGWD
}

function: greatweapon DICE:d {
	TOTAL: {}
	loop N over {1..#DICE} {
		TOTAL: TOTAL + [reroll [maximum of DICE]/#DICE under 3]
	}
	result: TOTAL
}

output [greatweapon 2d6] named "Great Weapon Fighting"

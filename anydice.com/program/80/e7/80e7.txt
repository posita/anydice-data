SKILL: 3
MAX: 5
TYPE: 6

loop MOD over {1..(SKILL-1)} {
 DICE: MAX
 output DICE d (d TYPE < (MOD+1)) = DICE named "[DICE]d:[MOD]"
}
loop REVERSE over {2..MAX} {
 DICE: MAX - REVERSE + 2
 output DICE d (d TYPE < (SKILL+1)) = DICE named "[DICE]d:[SKILL]"
}
loop DICE over {1..MAX} {
 output DICE d (d TYPE < (SKILL+1)) > 0 named "[DICE]d:[SKILL]"
}
loop MOD over {SKILL+1..TYPE-1} {
 DICE: MAX
 output DICE d (d TYPE < (MOD +1)) > 0 named "[DICE]d:[MOD]"
}

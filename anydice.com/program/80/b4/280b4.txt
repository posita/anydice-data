DICE_COUNT_LOWER: 1
DICE_COUNT_UPPER: 5
DIE_SIDES: 6
SUCCESS_THRESHOLD: 4 \inclusive!\
AGAIN: 6
set "maximum function depth" to 5

function: succ D:n {
  if (D >= AGAIN) { result: 1 + [succ dDIE_SIDES] }
  result: (D >= SUCCESS_THRESHOLD)
}

loop I over {DICE_COUNT_LOWER..DICE_COUNT_UPPER} {
  R: 0
  loop J over {1..I} { R: R + [succ dDIE_SIDES] }
  output R named "successes with [I] dice"
}

function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}


output [attack 1d20 plus 5 vs 15 for 2d8 + 5 crit 2d8 on 20] named "with damage modifier"

output [attack 1d20 plus 5 vs 15 for 2d8 crit 4d8 on 19] named "with expanded crit"

function: ROLL:n reroll BAD:s as REROLL:s {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

function : NUMBER:n plus {
  result:(d6>=NUMBER)
}

function : NUMBER:n plus reroll S:s {
  result:([d6 reroll S as d6]>=NUMBER)
}

function : NUMBER:n plus reroll {
  result:([d6 reroll {1..(NUMBER-1)} as d6]>=NUMBER)
}


MODEL: 30
ATTACK: 2
TO_WOUND: [ 4 plus reroll {1}] 
TO_HIT: [ 3 plus reroll ]
REND: 0
DAMAGE: 1

output (MODEL*ATTACK)d(TO_HIT)d(TO_WOUND)*DAMAGE

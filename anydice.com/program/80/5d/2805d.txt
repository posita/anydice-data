function: oppose A to D {
 result: A > D
}

function: oppose A to D with C advantaged attackers {
  if C = 0 {
    result: 0
  }
  if C = 1 {
    result: [oppose A to D]
  }
  if C = 2 {
    result: [oppose ([highest 1 of 2d20]) to D]
  }
  else
  {
     Z: C-1
     result: [highest of [oppose A to D] and [oppose [highest 1 of 2d20] to D with Z advantaged attackers]]
  }
}


function: oppose A to D with C simple attackers {
  if C = 0 {
    result: 0
  }
  if C = 1 {
    result: [oppose A to D]
  }
  else
  {
     Z: C-1
     result: [highest of [oppose A to D] and [oppose A to D with Z simple attackers]]
  }
}

ATTACKERS: 9
BONUS: 1

loop B over {0..BONUS} {
  loop A over {2..ATTACKERS} {
     output ((([oppose (1d20) to (1d20+B) with A advantaged attackers]) - ([oppose (1d20) to (1d20+B) with A simple attackers]))) named "Advantage v Simple, Bonus [B], Attackers [A]" 
  }
}

function: SUM:n vs TARGET:n with SUCCESS:n
{
 if (SUM >= TARGET) { result: 1 + (SUM - TARGET) / SUCCESS}
 else { result: 0 }
}

EDFOUR: [explode d4]
EDSIX: [explode d6]
EDEIGHT: [explode d8]
EDTEN: [explode d10]
EDTWELVE: [explode d12]
EDTWENTY: [explode d20]

function: earthdawn STEP:n
{
 if STEP = 1 { result: EDFOUR - 2 }
 else if STEP = 2 { result: EDFOUR - 1 }
 else if STEP = 3 { result: EDFOUR }
 else if STEP = 4 { result: EDSIX }
 else if STEP = 5 { result: EDEIGHT }
 else if STEP = 6 { result: EDTEN }
 else if STEP = 7 { result: EDTWELVE }
 else if STEP = 8 { result: 2dEDSIX }
 else if STEP = 9 { result: EDEIGHT + EDSIX }
 else if STEP = 10 { result: 2dEDEIGHT }
 else if STEP = 11 { result: EDTEN + EDEIGHT }
 else if STEP = 12 { result: 2dEDTEN }
 else if STEP = 13 { result: EDTWLVE + EDTEN }
 else if STEP = 14 { result: 2dEDTWELVE }
 else if STEP = 15 { result: EDTWELVE + 2dEDSIX }
 else if STEP = 16 { result: EDTWELVE + EDEIGHT + EDSIX }
 else if STEP = 17 { result: EDTWELVE + 2dEDEIGHT }
 else if STEP = 18 { result: EDTWELVE + EDTEN + EDEIGHT }
 else if STEP = 19 { result: EDTWENTY + 2dEDSIX }
 else if STEP = 20 { result: EDTWENTY + EDEIGHT + EDSIX }
 else if STEP = 21 { result: EDTWENTY + 2dEIGHT }
}

function: alternate STEP:n
{
 result: STEP + d12 - d12
}

loop N over {6, 10, 14, 18}
{
 M: N + 2
 output [[earthdawn N] vs N with 5] named "ED STEP [N] vs [N]"
 output [[alternate N] vs N with 5] named "Alternate [N] vs [N]"
 output [[earthdawn N] vs M with 5] named "ED STEP [N] vs [M]"
 output [[alternate N] vs M with 5] named "Alternate [N] vs [M]"
}

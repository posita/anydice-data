\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
DMGBONUS: 2
PROFICIENCY: 6
ACCBONUS: 0
CRITROLL: 19
ATTACKTIMES: 3
DAMAGE: 1d8+5d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + 10d8 +  + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of 3d20] vs ENEMYAC] named "Action attack with rapier twice, Action Surge Haste, Haste Action 1 attack, 4th level divine smite on all 3 attacks"

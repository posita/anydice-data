function: limit X:n to LB:n and UB:n {
  if X < LB { result: LB }
  if X > UB { result: UB }
  result: X
}

function: choose middle of A:n and B:n and C:n {
  if A < B {
    if B < C { result: B }
    else {
      if A < C { result: C }
      else { result: A }
    }
  } else {
    if A < C { result: A }
    else {
      if B < C { result: C }
      else { result: B }
    }
  }
}

loop TN over {3,5,7} {
  loop BONUS over {-1,1,3} {
    output [limit ([choose middle of 1d6 and 1d6 and 1d8]+BONUS-TN) to 0 and 3] named "+[BONUS] against [TN]"
  }
}

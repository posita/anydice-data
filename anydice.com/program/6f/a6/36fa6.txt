HS: 15 \ higher stat/20 \
LS: 8 \ lower stat/20 \
AS: 0 \ affinity stat/5 \

R: 2d20 \ roll primary stat \
BLR: [lowest 1 of R] \ base lowest of roll \
LR: BLR - AS \ lowest of roll, adjusted with affinity \

CS: (BLR = 1) \ critical success \
S: (BLR != 1 & LR <= LS & LS != 1) \ success \
PS: (BLR != 1 & LR > LS & LR <= HS & HS != LS) \ partial success \
F: (BLR != 20 & LR > HS & HS != 20) \ failure \
CF: (BLR = 20) \ critical failure \

output CS named "critical success"
output S named "success"
output PS named "partial success"
output F named "failure"
output CF named "critical failure"

set "maximum function depth" to 10
function: explodecount N:n {
 if N = 6 { result: 1 + [explodecount d6] }
 result: N > 3
 \ N is shorter to write than ROLLEDVALUE \
}

loop N over {3..10} {
  output [count {4,5,6} in Nd6] named "[N]d6"
}

loop N over {3..10} {
  output [count {4,5,6} in (N-1)d6] + [explodecount d6] named "[N]d6 with 1 crit die"
}

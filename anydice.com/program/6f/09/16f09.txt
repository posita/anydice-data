function: conspire dissent ROLL:s {
  result: ROLL >= 5
}

function: conspire suspicion ROLL:s {
  result: ROLL = 4
}

function: conspire actions ROLL:s {
  ACTIONS: 0
  loop DIE over ROLL {
    if DIE <= 3 {
      ACTIONS: ACTIONS + DIE
    }
  }
  result: ACTIONS
}

function: ROLL:s dice plot success with S:n suspicion and M:n military {
  if (ROLL = 4) >= S {
    result: -1
  }
  else {
    if (ROLL >= 5) < M {
      result: 0
    }
  }
  result: 1
}

loop S over {3} {
  loop M over {3} {
    loop D over {7} {
      output [Dd6 dice plot success with S suspicion and M military] named "[D]d[S]s[M]m"
    }
  }
}

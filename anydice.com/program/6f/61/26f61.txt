function: square N:n {
 if N < 1 { result: 0 }
 result: N*N
}

function: negdrop N:n {
 if N < 1 { result: 0 }
 result: N
}

function: compare N:n {
 result: (N-1d12-3)
}

function: dexattack N:n {
 if N > 0 { result: ([negdrop N ])+1d{1,1,2,2,3,4} }
 if N < -2 { result: 0 }
 result: 1d{1,1,2,2,3,4}
}

function: strattack N:n {
 if N > 0 { result: ([square N ])+1d{1,1,2,2,3,4} }
 if N < -2 { result: 0 }
 result: 1d{1,1,2,2,3,4}
}

function: cproll DIE:d {
  MAX: [maximum of DIE]
  MIN: 1
  result: [cproll helper DIE DIE]
}

function: cproll helper DIEN:n DIE:d {
  if DIEN = 10 {result: MAX + DIE} 
  else {
    if DIEN = 1 {result: MIN - DIE} 
    else {
      result: DIEN
    }
  }
}

loop S over {-10..10} {
  output [cproll 1d10] + S > 0 named "DV at [S]"
  output [cproll 1d10] + S > [cproll 1d10] named "Opposed at [S]"
}

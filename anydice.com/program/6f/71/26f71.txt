ATTACKS:4
ACCURACY:3
BASEDAMAGE:3
CRITDAMAGE:4

HITS:0
CRITS:0
DAMAGE:0
loop N over {1..ATTACKS}{
RESULT:1d6
HITS:HITS+[count {ACCURACY..5} in RESULT]
CRITS:CRITS+[count 6 in RESULT]
}
DAMAGE:DAMAGE+(HITS*BASEDAMAGE)
DAMAGE:DAMAGE+(CRITS*CRITDAMAGE)

output HITS named "Regular Hits"
output CRITS named "Critical Hits"
output DAMAGE named "Damage Dealt"

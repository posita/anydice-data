function: edgesnag EDGE:s SNAG:s {
  loop X over {1..#EDGE} {
    if X@EDGE > X@SNAG { result: 2 }
    if X@EDGE < X@SNAG { result: 0 }
if X@EDGE = X@SNAG { result: 1 }
  }
 
}

B: {0,0,0,1,1,2}

function: roll ROLL:s plus BONUS:n vs TARGET:n with optional REROLL:d reroll {
  SUM: ROLL + BONUS + 0  
    \Only reroll if less than TARGET\
  if SUM >= TARGET {
    result: ROLL + BONUS + 0
  } else { 
    REST: {1..#ROLL-1}@ROLL
    result: REST + REROLL + BONUS + 0
  }
}

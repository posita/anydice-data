

function: attack D:n plus BONUS:n vs AC:n  {
    if D = 20 { result: 1 } 
    else if D = 1 { result: 0}
    else if D + BONUS >= AC { result: 1 } 
    else { result:0}
}

AC: 18
FIEND_OR_UNDEAD: 0

DORIC_EB_CURSE: [attack d20 plus DORIC_EB_ATK vs AC]*DORIC_EB_CURSE_DMG

DORIC_TURN_ONE: 3dDORIC_EB

function: smite L:n {
  if L >= 4 { result: 5d8 + (FIEND_OR_UNDEAD & 1d8) }
  else { result: (L+1)d8 + (FIEND_OR_UNDEAD & 1d8)}
}

LIANIRI_ATK: +8
LIANIRI_MAIN_DMG: 1d10+1d8+4
LIANIRI_BA_DMG: 1d4+1d8+4

LIANIRI_MAIN_SMITE_TWO_DMG: LIANIRI_MAIN_DMG + [smite 2]
LIANIRI_MAIN_SMITE_THREE_DMG: LIANIRI_MAIN_DMG + [smite 3]
LIANIRI_BA_SMITE_TWO_DMG: LIANIRI_BA_DMG + [smite 2]
LIANIRI_BA_SMITE_THREE_DMG: LIANIRI_BA_DMG + [smite 3]

LIANIRI_MAIN: [attack d20 plus LIANIRI_ATK vs AC]*LIANIRI_MAIN_DMG
LIANIRI_MAIN_SMITE_TWO: [attack d20 plus LIANIRI_ATK vs AC]*LIANIRI_MAIN_SMITE_TWO_DMG
LIANIRI_MAIN_SMITE_THREE: [attack d20 plus LIANIRI_ATK vs AC]*LIANIRI_MAIN_SMITE_THREE_DMG

LIANIRI_BA: [attack d20 plus LIANIRI_ATK vs AC]*LIANIRI_BA_DMG
LIANIRI_BA_SMITE_TWO: [attack d20 plus LIANIRI_ATK vs AC]*LIANIRI_BA_SMITE_TWO_DMG
LIANIRI_BA_SMITE_THREE: [attack d20 plus LIANIRI_ATK vs AC]*LIANIRI_BA_SMITE_THREE_DMG

\ Burning all smite slots ASAP. Unrealistic upper bound \

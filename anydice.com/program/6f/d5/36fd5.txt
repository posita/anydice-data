DIE: d{0,0,0,0,1}

function: count WINS:n in DICE:s {
  result: [count WINS in DICE]
}

function: reroll DICE:s {
  WINS: [count 1 in DICE]
  LOSSES: [count 0 in DICE]
  if LOSSES = 0 {
    result: WINS
  } else {
    result: WINS + [reroll LOSSES d DIE]
  }
}

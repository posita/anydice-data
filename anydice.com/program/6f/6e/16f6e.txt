function: dado U:s D:s {
if U = D { 
  if U = 5 { result: d6+5 }
  if U = 6 { result: -10 }
  else { result: D*-1 }
} else { result: [absolute U - D]}}

output [dado 1d6 1d6] named "My"
output d{-1..1}+d{-1..1}+d{-1..1} named "3dF"
output d{-1..1}+d{-1..1}+d{-1..1}+d{-1..1} named "4dF"
output d{-1..1}+d{-1..1}+d{-1..1}+d{-1..1}+d{-1..1} named "5dF"

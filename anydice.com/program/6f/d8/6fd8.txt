function: roll plus A against B {
  result: [determine 2d6+A of B]
}

function: determine R:n of B:n {
  if R < B-4 { result: -2 }
  if R < B-1 { result: -1 }
  if R < B+2 { result: 0 }
  if R < B+5 { result: 1 }
  result: 2
}

SG: 10
output [roll plus 0 against SG] named "+0"
output [roll plus 1 against SG] named "+1"
output [roll plus 2 against SG] named "+2"
output [roll plus 3 against SG] named "+3"
output [roll plus 4 against SG] named "+4"
output [roll plus 5 against SG] named "+5"

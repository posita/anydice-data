\*** Probabilities Using a 2 Success Range ***\
ADVANTAGE:1 \* 0 = Normal, 1 = Advantage, -1 = Disadvantage *\ 
EDGE_CHANCE:6

function: evaluate ROLL:s EDGE:s{
 SUCCESSES: ROLL >= 6
 
 if ADVANTAGE = 0 { EDGE_RESULT: 0 }
 else { EDGE_RESULT: EDGE >= EDGE_CHANCE }

 if EDGE_RESULT = 1 
 { 
   if ADVANTAGE = 1 { SUCCESSES: SUCCESSES + 1 }
   else { SUCCESSES: SUCCESSES - 1 }
 }

 if SUCCESSES = 1 { result: 1}
 else if SUCCESSES > 1 { result: 2}
 else { result: 0}
}

loop DICE over {2..8} {
 output [evaluate DICE d6 1d10] named "[DICE]d"
}

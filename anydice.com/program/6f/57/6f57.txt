\*** Normal Dice Mechanic using Wild Die ***\
DICE_POOL:5 \* Maximum number of dice to be rolled *\

function: evaluate ROLL:s {
 SUCCESS: ROLL >= 5
 SUCCESSES: SUCCESS
 if SUCCESSES = 1 { result: 1}
 if SUCCESSES > 1 { result: 2}
 else { result: 0}
 result: 0
}

loop DICE over {2..DICE_POOL} {
 DIE: DICE
 output [evaluate DICE d6] named "[DIE]d"
}

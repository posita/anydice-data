HIT: 0
CRIT: 1

function: attackr ROLL:n {
 if ROLL = 20 {
  result: CRIT
  } else {
  result: HIT
  }
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRIT {
  result: CRITICALDAMAGE
 } else {
  result: HITDAMAGE
  }
}


HITDAMAGE: 2d6
CRITICALDAMAGE: 3d6
output [damage for [attackr d20]] named "2d6"

CRITICALDAMAGE: 2d12

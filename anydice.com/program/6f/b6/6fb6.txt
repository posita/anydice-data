ADVANTAGE:1
FATE_CHANCE:6

function: evaluate ROLL:s FATE:s{
 SUCCESSES: ROLL >= 5
 
 if ADVANTAGE = 0 { FATE_RESULT: 0 }
 else { FATE_RESULT: FATE >= FATE_CHANCE }

 if FATE_RESULT = 1 
 { 
   if ADVANTAGE = 1 { SUCCESSES: SUCCESSES + 1 }
   else { SUCCESSES: SUCCESSES - 1 }
 }

 if SUCCESSES = 1 { result: 1}
 else if SUCCESSES = 2 { result: 2}
 else if SUCCESSES > 2 { result: 3}
 else { result: 0}
}

loop DICE over {2..6} {
 output [evaluate DICE d6 1d10] named "[DICE]d"
}

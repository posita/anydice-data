X: 1d6
Y: 1d12
A: 0
B: 0

function: reroll ROLL:n if lower then A:n {
if ROLL <= A {result: X}
result: ROLL
}
loop N over {1..24}{
A: A+[reroll X if lower then 2]}
A: A+[reroll 1d8 if lower then 2]+60
output A named "giant fighter damage RO"
output 24d6+60+1d8 named "giant fighter damage"

loop N over {1..12}{
B: B+[reroll Y if lower then 2]}
B: B+[reroll 1d8 if lower then 2]+60
output B named "giant fighter damage GA RO"
output 12d12+60+1d8 named "giant fighter damage GA"

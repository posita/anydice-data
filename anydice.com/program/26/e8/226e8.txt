function: roll D:n with ATK:n for DMG:n against DEF:n with AMR:n {
  ROLL: (D-ATK)
  if (ROLL >= (10+DEF)) {
    result: [highest of (DMG - AMR) and 0]
  } else {
    result: 0
  }
}

function: sucesso em A:s{
 result:[count 6 in A]*10
}

function: sucesso e risco em A:s{
 result:[count 6 in A]*10 + [count 1 in A]
}

function: sucesso forcando A:s{
 B:[count 6 in A]
 C:[count 1 in A]
 if (B+C)=(#A){result:(B*10)}
 result:(B*10+C)+[sucesso e risco em (#A-B-C)d6]
}

\ DEZENAS SÃO A QUANTIDADE DE SUCESSOS \
\ UNIDADES SÃO A QUANTIDADE DE SUCESSOS \
loop X over {1..5}{
 XX:Xd6
 output [sucesso forcando XX] named "[X]d6"
}

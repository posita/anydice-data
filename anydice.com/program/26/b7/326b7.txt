
loop N over {1..10} {
 output (N-1)d(d12<=N)+1d(d10=1) named "Skill [N] -1d"
 output Nd(d12<=N)+1d(d10=1) named "Skill [N]"
 output (N+1)d(d12<=N)+1d(d10=1) named "Skill [N] +1d"
  if N>5 {
  output Nd(d12<=N)+1d(d10=1)+2 named "Skill [N] +2S" }
  else {
  output Nd(d12<=N)+1d(d10=1)+1 named "Skill [N] +1S" }
}

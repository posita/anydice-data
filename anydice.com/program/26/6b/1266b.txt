function: twenty A:n {
   if A >= 15 { result: 15 }
   if A >= 10 { result: 10 }
   result: 0
}

function: apoc A:n {
   if A >= 10 { result: 10 }
   if A >= 7 { result: 7 }
   result: 0
}

output [twenty 1d20]
output [apoc 2d6]
output [twenty 1d20+4] named "1d20 plus 4"
output [apoc 2d6+1] named "2d6+1"

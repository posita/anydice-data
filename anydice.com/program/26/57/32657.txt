MISS: 0
HIT: 1
CRITICALHIT: 2
CRITICALRANGE: 19
ATTACK: 9
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= CRITICALRANGE {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

output [attack d20 vs 15] named "chance to hit"
output 2d6+5+1d4 named "normal"
output [highest of 2d6+5 and 2d6+5]+1d4 named "savage attacker"

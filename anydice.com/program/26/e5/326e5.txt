function: explode N:n M:n{
 if N <= M { 
    result: N + d6
 }
 else{
   if N = 1{
     result: N
   }
   else{
     result: N
   }
 }
}

function: pool P:n M:n{
 X: 0
 loop N over {1..P}{
   X: X + [explode d6 M]
 }
 result: X
}

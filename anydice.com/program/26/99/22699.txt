function: sol SEQ:s{
  ONES:  ([count 1 in SEQ] *1) +2
  TWOS:  ([count 2 in SEQ] * 2)+2
  THREES:([count 3 in SEQ] * 3)+2
  FOURS: ([count 4 in SEQ] * 4)+2
  FIVES: ([count 5 in SEQ] * 5)+2
  SIXES: ([count 6 in SEQ] * 6)+2

  result: 1 @ [sort {ONES, TWOS, THREES, FOURS, FIVES, SIXES}]
}

output [sol(1d6)]
output [sol(2d6)]
output [sol(3d6)]
output [sol(4d6)]
output [sol(5d6)]
output [sol(6d6)]
output [sol(7d6)]

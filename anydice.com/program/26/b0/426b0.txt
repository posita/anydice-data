function: ironsworn X:n vs Y:n and Z:n {
  BEAT_BOTH: X >= Y & X >= Z
  BEAT_SINGLE: X >= Y | X >= Z
  BEAT_NONE: !(X >= Y | X >= Z)
  DOUBLE: Y=Z
  if BEAT_BOTH & DOUBLE  {result: 4}
  if BEAT_BOTH           {result: 2}
  if BEAT_SINGLE         {result: 1}
  if BEAT_NONE & DOUBLE  {result: -1}
  result: 0
}
output [ironsworn d6 vs d10 and d10] named "Ironsworn roll"

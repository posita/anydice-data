function: reroll R:n under N:n {
   if R < N { 
      result: 2d6 
   } else {
      result: R
   }
}

function: rerollone F:n S:n under N:n {
   if F+S < N {
      if F<S {
         result: 1d6+S
      } else {
         result: 1d6+F
      }
   } else {
      result: F+S
   }
}

output [reroll 2d6 under 9] named "reroll"
output [rerollone 1d6 1d6 under 9] named "reroll one"
output 2d6 named "straight"

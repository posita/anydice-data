MISS: 0
NORMALHIT: 1

function: ROLL:n plus ATTACK:n vs AC:n th THREAT:n x MUL:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= THREAT & ROLL + ATTACK >= AC {
  result: [threat d20 plus ATTACK vs AC for MUL]
 }
 if ROLL + ATTACK >= AC | ROLL = 20 { result: NORMALHIT }
 result: MISS
}

function: threat ROLL:n plus ATTACK:n vs AC:n for MUL:n {
 if ROLL = 1 { result: NORMALHIT }
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: MUL
 }
 result: NORMALHIT
}

function: HIT:n for DAMAGE:d from HP:n {
  if HIT > MISS {
    RESULT: DAMAGE
    loop N over {2..HIT} {
      RESULT: RESULT + DAMAGE
    }
    result: HP - RESULT
  }
  else {
    result: HP
  }
}

\ anydice, why must I hack you so? \
function: rounds HITPOINTS:n survived past CURROUND:n {
  if HITPOINTS < 0 {
    result: CURROUND
  }
  result: CURROUND + 1
}

function: rounds at PA:n to kill HP:n past AC:n {
  ATK: 10-PA+2  \ charging \
  CHP: HP
  R: 1
  loop N over {1..6} {
    CHP: [[d20 plus ATK vs AC th 19 x 2] for 2d6+7+PA*2 from CHP]
    R: [rounds CHP survived past R]
    ATK: 10-PA  \ not charging anymore \
  }
  result: R
}

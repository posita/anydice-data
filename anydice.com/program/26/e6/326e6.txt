function: explode N:n M:n DM:n D:n{
if D <= DM{
  if N <= M { 
      result: N + [explode d6 M DM D+1]
    }
    else{
      result: N
    }
  }
  else{
      result: 0
  }
}

function: pool P:n M:n DM:n{
 X: 0
 loop N over {1..P}{
   X: X + [explode d6 M DM 0]
 }
 result: X
}

DM: 1

output [pool 1 0 DM] named "Atributo 0 y Técnica 0"
output [pool 1 1 DM] named "Atributo 0 y Técnica 1"
output [pool 1 2 DM] named "Atributo 0 y Técnica 2"
output [pool 1 3 DM] named "Atributo 0 y Técnica 3"

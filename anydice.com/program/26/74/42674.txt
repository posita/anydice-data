
ATTACKBONUS: 5
DAMAGE: 1d8
CRITDAMAGE: 2d8
ATTACKS: 1
ADVANTAGE: 0
DISADVANTAGE: 0


MISS: 0
MISSDAMAGE: 0
HIT: 1
CRIT: 2
VANTAGE: ADVANTAGE - DISADVANTAGE

VANTAGE: (ADVANTAGE - DISADVANTAGE)
if VANTAGE > 0 {
ROLL: [highest 1 of (VANTAGE + 1)d20]
}
if VANTAGE < 0 {
ROLL: [lowest 1 of (-1*(VANTAGE) + 1)d20]
}
if VANTAGE = 0 {
ROLL: 1d20
}


function: attack ROLL:n vs AC:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  result: CRIT
 }
 if ROLL + ATTACKBONUS >= AC { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRIT {
  result: CRITDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: DAMAGE
 }
 result: MISSDAMAGE
}

loop AC over {5..30} {
 output (ATTACKS * [damage for [attack ROLL vs AC]]) named "[AC]"
}

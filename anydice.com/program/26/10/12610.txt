function: rapier ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: RAPIERCRIT } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: RAPIER }
 result: 0
}


function: shortswordplusone ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: SHORTSWORDCRIT+1 } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS +1 >= AC { result: SHORTSWORD+1 }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 3
DMGBONUS: 0
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
RAPIER: 1d8 + ATTACKMOD + DMGBONUS
RAPIERCRIT: 2d8 + ATTACKMOD + DMGBONUS
SHORTSWORD: 1d6 + ATTACKMOD + DMGBONUS
SHORTSWORDCRIT: 2d6 + ATTACKMOD + DMGBONUS

output ATTACKTIMESd[rapier d20 vs ENEMYAC] named "rapier"

output ATTACKTIMESd[shortswordplusone d20 vs ENEMYAC] named "shortsword +1"

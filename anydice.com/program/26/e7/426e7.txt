function: sucessos em A:s{
 result:[count 6 in A]
}

function: sucessos forcando A:s{
 B:[count 6 in A]
 C:[count 1 in A]
 if (B+C)=(#A){result:B}
 result:B+[count 6 in(#A-B-C)d6]
}

function: danos A:s{
 B:[count 6 in A]
 C:[count 1 in A]
 if (B+C)=(#A){result:0}
 result:C+[count 1 in(#A-B-C)d6]
}


loop X over {1..5}{
 XX:Xd6
 \output [sucessos em XX] named "[X]d6"\
 \output [sucessos forcando XX] named "forçando [X]d6"\
 output [danos XX] named "danos [X]d6"
}

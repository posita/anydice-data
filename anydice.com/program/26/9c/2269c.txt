function: sol SEQ:s{
  ONES:  ([count 1 in SEQ] *1) +6
  TWOS:  ([count 2 in SEQ] * 2)+6
  THREES:([count 3 in SEQ] * 3)+6
  FOURS: ([count 4 in SEQ] * 4)+6
  FIVES: ([count 5 in SEQ] * 5)+6
  SIXES: ([count 6 in SEQ] * 6)+6

  result: 1 @ [sort {ONES, TWOS, THREES, FOURS, FIVES, SIXES}]
}

output [sol(1d6)]
output [sol(2d6)]
output [sol(3d6)]
output [sol(4d6)]
output [sol(5d6)]
output [sol(6d6)]
output [sol(7d6)]

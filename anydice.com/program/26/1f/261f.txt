function: outer NUMBER:n of DICE:d
{
  if(NUMBER >= #DICE) { result: DICE }

  ODDBUMP: (NUMBER / 2) != ( (NUMBER + 1) / 2)
  BOTTOM: (NUMBER / 2) + ODDBUMP
  TOP: #DICE - (NUMBER / 2) + 1

  result: {1..BOTTOM, TOP..#DICE}@DICE
}

function: reroll highest NUMBER:n of DICE:d with SIDES:s
{
  LOWERDICE: [lowest (#DICE - NUMBER) of DICE]
  result: LOWERDICE
}

D: 4 \ Number of die \
N: 2 \ Number to pull \
S: 6 \ Number of sides \

output [highest N of DdS] named "Highest [N] of [D]d[S], Keeping"

function: roll D:n ESFUERZO:n EQUIPO:n{
   if D <= ESFUERZO{
      result: [highest of d6 and D]
   }
   result: D
}

function: accion TECNICA:n ESFUERZO:n EQUIPO:n{
  X: 0
  loop N over {1..TECNICA+2}{
     X: X + [roll 1d6 ESFUERZO EQUIPO]
  }
  result: X
}




EQUIPO: 0
loop TECNICA over {0..3}{
   loop ESFUERZO over {0..3}{
      \loop EQUIPO over {0..3}{\
         output [accion TECNICA ESFUERZO EQUIPO] named "Técnica [TECNICA] y Esfuerzo [ESFUERZO] y Equipo [EQUIPO]" 
      \}\
   }
}

function: reroll lowest of ROLL:s as REROLL:d if less than MIN:n {
  LOWEST: (#ROLL)@ROLL       \ the lowest die is sorted last \
  if LOWEST >= MIN {
    result: ROLL + 0
  } else {
    REST: {1..#ROLL-1}@ROLL  \ all but the lowest die \
    result: REST + REROLL
  }
}

function: rerollall R:n under N:n as RR:d {
   \Only reroll if less than N\
   if R < N { result: RR }
   else { result: R }
}

output 2d6 named "2d6"
output [rerollall 2d6 under 7 as 2d6] named "2d6 replace all if < 7"
output [reroll lowest of 2d6 as d6 if less than 4] named "2d6 replace lowest if < 4"

output 3d6 named "3d6"
output [rerollall 3d6 under 10 as 3d6] named "3d6 replace all if < 9"
output [reroll lowest of 3d6 as d6 if less than 4] named "3d6 replace lowest if < 4"

function: success graded ROLL:n PASSTHROUGH:d {
 if ROLL = 6 {
   result: 2
 }
 if ROLL = 5 {
   result: 1
 }
 result: 0
}

function: success exploded ROLL:n PASSTHROUGH:d {
 if ROLL = 6 {
   result: 1 + [success exploded PASSTHROUGH PASSTHROUGH]
 }
 if ROLL = 5 {
   result: 2
 }
 result: 0
}



function: graded {
 result: [success graded 1d6 1d6]
}

function: exploded {
 result: [success exploded 1d6 1d6]
}

loop DEPTH over {0..5} {

 output 1d[graded] > DEPTH named "1 graded skill > [DEPTH]"
output 1d[exploded] > DEPTH named "1 exploded skill > [DEPTH]"
}

loop DEPTH over {0..5} {

 output 2d[graded] > DEPTH named "2 graded skill > [DEPTH]"
output 2d[exploded] > DEPTH named "2 exploded skill > [DEPTH]"
}

function: reroll R:n under N:n {
   if R < N { 
      result: 2d6 
   } else {
      result: R
   }
}

function: rerollone F:n S:n under N:n {
   if F+S < N {
      if F<S {
         result: 1d6+S
      } else {
         result: 1d6+F
      }
   } else {
      result: F+S
   }
}

function: rerollonlyfirst F:n S:n under N:n {
   if F+S < N {
      result: 1d6+S
   } else {
      result: F+S
   }
}

function: rerollonlysecond F:n S:n under N:n {
   if F+S < N {
      result: 1d6+F
   } else {
      result: F+S
   }
}

function: rerolork F:n S:n under N:n {
   if F+S < N {
      if F<4 {
         if S<4 {
            result: 2d6
         } else {
            result: [rerollonlysecond 1d6 S under N]
         }
      } else {
         if S<4 {
            result: [rerollonlyfirst F 1d6 under N]
         } else {
            result: [rerollonlysecond S 1d6 under N]
         }
      }
   } else {
      result: F+S
   }
}

loop N over {1..10} {
 loop HN over {2,1} {
  if HN>N {HN:1}
  else {output (N-HN)d(d12<=N)+1d(d10=1) named "Skill [N] -[HN]d" }
 }
 output Nd(d12<=N)+1d(d10=1) named "Skill [N]"
 loop HP over {1,2} {
  output (N+HP)d(d12<=N)+1d(d10=1) named "Skill [N] +[HP]d" }
 if N>5 {
  output Nd(d12<=N)+1d(d10=1)+2 named "Skill [N] +2S" }
 else {
  output Nd(d12<=N)+1d(d10=1)+1 named "Skill [N] +1S" }
}

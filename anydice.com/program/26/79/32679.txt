\**** Damage calculator Eagle Eye Duelist Attack ****\

function: attackroll ROLLA:n with ROLLB:n vs DEF:n {
 if ROLLA + ROLLB <= 2 { result: 0 }
 if ROLLA = ROLLB { 
  if ROLLA + ROLLB + ATTACKBONUS > DEF { result: (ROLLA + ROLLB + ATTACKBONUS - DEF + BONUSDMG + WPNDAMAGEA)*2}
  if ROLLA + ROLLB + ATTACKBONUS <= DEF { result: (BONUSDMG + WPNDAMAGEA)*2}
}
 if ROLLA + ROLLB + ATTACKBONUS >= DEF { result: ROLLA + ROLLB + ATTACKBONUS - DEF + BONUSDMG + WPNDAMAGEA}
 result: 0
}

\looped against enemy DEF\
ENEMYDEF: d{11}

ATTACKBONUS: 5
BONUSDMG: 0
WPNDAMAGEA: 24
output [attackroll 1d6 with 1d6 vs ENEMYDEF]*2 named "Eagle Eye Duelist Attack"
output 2d6 named "Normal 2d6 Distribution"

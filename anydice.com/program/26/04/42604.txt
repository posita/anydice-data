MISS: 0
MISSDAMAGE: 0
HIT: 1
HITDAMAGE: 1d8 + 4
CRIT: 2
CRITDAMAGE: 2d8 + 8
ATTACK:4
AC: 15
function: attack ROLL:n vs DEFENSE:n {
if ROLL = 1 { result: MISS }

if ROLL = 20 { result: CRIT }

if ROLL + ATTACK >= DEFENSE { result: HIT }
result: MISS
}
loop AC over {10, 20, 30} {
output [damage for [attack d20 vs AC]] named "AC: [AC]"
}
function: damage for ATTACKRESULT:n {
if ATTACKRESULT = CRIT {
result: CRITDAMAGE
}
if ATTACKRESULT = HIT {
result: HITDAMAGE
}
result: MISSDAMAGE
}

function: compare LEVEL to DIFFICULTY {
  FULL_DIFFICULTY: DIFFICULTY + 10
  CRITICAL_DIFFICULTY: DIFFICULTY + 20

  PARTIAL: LEVELd12 >= DIFFICULTY
  FULL: LEVELd12 >= FULL_DIFFICULTY
  CRIT: LEVELd12 >= CRITICAL_DIFFICULTY

  result: PARTIAL + FULL + CRIT
}

loop LEVEL over {1..15} {
 TEST_LEVEL: LEVEL+LEVEL/2
 DIFFICULTY: 5*TEST_LEVEL
 output [compare LEVEL to DIFFICULTY] named "PC[LEVEL] vs T[TEST_LEVEL]"
}

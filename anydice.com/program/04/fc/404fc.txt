function: pitch P:n {
 if P = 1 {result: 2d6} \four-seam fastball\
 if P = 2 {result: 1d6+1d4} \cutter\
 if P = 3 {result: 2d4} \sinker\
 if P = 4 {result: [absolute 1d6-1d4]+1d6} \curveball\
 if P = 5 {result: [absolute 1d6-1d4+1d6]} \slider\
 if P = 6 {result: [absolute 1d6-1d4]+1d4} \changeup\
 if P = 7 {result: [absolute 1d6-1d4+1d4]} \splitter\
  result: 0
} 

function: zone Z:n {
 if Z < 5 | Z > 8 {result: 0} \strike zone from 5 to 8 (inclusive)\
  result: 1
}

function: take P:n {
 if [zone P] = 0 {result: 1} \ball\
 if [zone P] != 0 {result: 2} \strike (called)\
}

function: swing B:n at P:n {
 if P < B-C | P > B+C {result: 2} \strike (swing & miss)\
 if P >= B-C & P <= B+C {
  if P >= B-G & P <= B+F {result: 0} \ball in play\
  if P < B-G | P > B+F {
   if S < 2 {result: 3} \strike (foul with less than 2 strikes)\
   result: 5 \foul with 2 strikes\
  }
 }
}

function: bat path BAT:n {
 if BAT = 1 {result: 2@4d6+2} \regular swing\
 if BAT = 2 {result: 2@3d6+3} \two-strike swing\
}

function: BIP:n defended by DEF:n {
 if BIP > DEF {result: 0}
 if BIP < DEF {result: 1}
 if BIP = DEF {result: 2}
}

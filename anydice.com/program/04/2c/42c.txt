function: wren ROLL:s crit at B:n plus {
 if 1@ROLL = 1 {
   result: 0
 }
 if 1@ROLL >= B {
    result: 1
 }
 if 2@ROLL = 1@ROLL {
    result: 1
 }
 result: 0
}

function: wren twinstrike crit at A:n {
 result: [wren 2d20 crit at A plus] + [wren 2d20 crit at A plus]
}

output [wren 2d20 crit at 18 plus] named "No Crit or Crit (18)"
output [wren twinstrike crit at 18] named "Twin strike crit chances (18)"
output [wren 2d20 crit at 16 plus] named "No Crit or Crit (16)"
output [wren twinstrike crit at 16] named "Twin strike crit chances (16)"

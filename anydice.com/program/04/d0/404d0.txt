AC: 12

function: attack DAMAGE:d HIT:n AC:n REPEAT:n {
  TEMP_PRINT: 0
  loop N over {1..REPEAT} {
    CHANCES: [highest of [lowest of (1d20 - AC + HIT) and 1] and 0] * DAMAGE
    TEMP_PRINT: TEMP_PRINT + CHANCES
  }
  result: TEMP_PRINT
}

output [attack 2d8+6 9 AC 1] named "Force Ballista BA"

output [attack 1d8+3 6 AC 1] named "Battle Smith"

output [attack 2d8 9 AC 1] named "Force Ballista"

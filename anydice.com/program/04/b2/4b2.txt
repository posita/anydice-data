\ ******* Calculates if an attack is a miss, hit, or crit ******* \
function: attack ROLL:n vs DEFENSE:n {
  if ROLL = 1 { result: MISS }
  if ROLL >= CRITRANGE {
    if ROLL + ATTACKBONUS >= DEFENSE { result: CRITICALHIT }
    result: HIT
  }
  if ROLL + ATTACKBONUS >= DEFENSE { result: HIT }
  result: MISS
}

\ ******* Calculates if an offhand attack is a miss, hit, or crit ******* \
function: offhand attack ROLL:n vs DEFENSE:n {
  if ROLL = 1 { result: OFFHANDMISS }
  if ROLL >= CRITRANGEOFFHAND {
    if ROLL + ATTACKBONUSOFFHAND >= DEFENSE { result: OFFHANDCRIT }
    result: OFFHANDHIT
  }
  if ROLL + ATTACKBONUSOFFHAND >= DEFENSE { result: OFFHANDHIT }
  result: OFFHANDMISS
}

\ ******* Calculates the damage for a sequence of attacks ******* \
function: damage for ATTACKS:s {
  D: 0
  loop A over {1..#ATTACKS} {
    if A@ATTACKS = CRITICALHIT { D: D + [maximum of ATTACKDAMAGE] + CRITDAMAGE }
    else if A@ATTACKS = HIT { D: D + ATTACKDAMAGE }
    else if A@ATTACKS = MISS { D: D + MISSDAMAGE }
    else if A@ATTACKS = OFFHANDCRIT { D: D + [maximum of ATTACKDAMAGEOFFHAND] + CRITDAMAGEOFFHAND }
    else if A@ATTACKS = OFFHANDHIT { D: D + ATTACKDAMAGEOFFHAND }
    else if A@ATTACKS = OFFHANDMISS { D: D + MISSDAMAGEOFFHAND }
  }
  result: D + [extra damage for ATTACKS]
}

\ ******* Calculates the damage for a sequence of attacks where a hit causes a secondary attack ******* \
function: damage for ATTACKS:s with onhit secondary SECONDARYATTACK {
  D: 0
  loop A over {1..#ATTACKS} {
    if A@ATTACKS = CRITICALHIT { D: D + [maximum of ATTACKDAMAGE] + CRITDAMAGE + SECONDARYATTACK }
    else if A@ATTACKS = HIT { D: D + ATTACKDAMAGE + SECONDARYATTACK }
    else if A@ATTACKS = MISS { D: D + MISSDAMAGE }
    else if A@ATTACKS = OFFHANDCRIT { D: D + [maximum of ATTACKDAMAGEOFFHAND] + CRITDAMAGEOFFHAND + SECONDARYATTACK }
    else if A@ATTACKS = OFFHANDHIT { D: D + ATTACKDAMAGEOFFHAND + SECONDARYATTACK }
    else if A@ATTACKS = OFFHANDMISS { D: D + MISSDAMAGEOFFHAND }
  }
  result: D + [extra damage for ATTACKS]
}

\ ******* Calculates the damage for a burst or blast attack ******* \
function: area damage for ATTACKS:s {
  HITS: 0
  CRITS: 0
  MISSES: 0
  D: 0
  loop A over {1..#ATTACKS} {
    if A@ATTACKS = CRITICALHIT { CRITS: CRITS + 1 }
    else if A@ATTACKS = HIT { HITS: HITS + 1 }
    else if A@ATTACKS = MISS { MISSES: MISSES + 1 }
  }
  D: D + ATTACKDAMAGE * HITS
  D: D + ([maximum of ATTACKDAMAGE] + CRITDAMAGE) * CRITS
  D: D + (MISSDAMAGE) * MISSES
  result: D + [extra damage for ATTACKS]
}

\ ******* Applies all sorts of extra damage that might be applied after all attacks are rolled (IE quarry) ******* \
function: extra damage for ATTACKS:s {
  D: 0
  if USEQUARRY {
    if (ATTACKS >= CRITICALHIT) >= 1 {
      D: D + [maximum of QUARRYDAMAGE]
    } else {
      if (ATTACKS >= HIT) >= 1 { D: D + QUARRYDAMAGE }
    }
  }
  
  if USEONEHITBONUS {
    if (ATTACKS >= HIT) >= 1 { D: D + ONEHITBONUSDAMAGE }
  }
  if USETWOHITBONUS {
    if (ATTACKS >= HIT) >= 2 { D: D + TWOHITBONUSDAMAGE }
  }
  if USETHREEHITBONUS {
    if (ATTACKS >= HIT) >= 3 { D: D + THREEHITBONUSDAMAGE }
  }
  result: D
}

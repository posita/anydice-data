AC: 15

function: attack DAMAGE:d HIT:n AC:n REPEAT:n {
  TEMP_PRINT: 0
  loop N over {1..REPEAT} {
    CHANCES: [highest of [lowest of (1d20 - AC + HIT) and 1] and 0] * DAMAGE
    TEMP_PRINT: TEMP_PRINT + CHANCES
  }
  result: TEMP_PRINT
}

output [attack 5d12 9 AC 1] named "Action"

output [attack 2d8+6 9 AC 1] named "Bonus Action"

output [attack 1d8+3 9 AC 1] named "Battle Forged"

output [attack 2d8 9 AC 1] named "Battle Forged"

output [attack 5d12+5d8+9 9 AC 1] named "Total Damage"

function: high in A:s B:s C:s {
  DICE: [sort {A, B, C}]
  
  if ( 1@DICE = 2@DICE & 2@DICE = 3@DICE ) { H: 1@DICE + 2@DICE + 3@DICE}
  else if ( 1@DICE = 2@DICE ) { H: 1@DICE + 2@DICE}
  else { H: 1@DICE }

  result: H
}

function: midhigh in A:s B:s C:s {
  DICE: [sort {A, B, C}]
  
  if ( 1@DICE = 2@DICE & 2@DICE = 3@DICE ) { MH: 1@DICE + 2@DICE + 3@DICE}
  else if ( 1@DICE = 2@DICE ) { MH: 1@DICE + 2@DICE}
  else if ( 2@DICE = 3@DICE ) { MH: 1@DICE }
  else { MH: 2@DICE }

  result: MH
}

function: midlow in A:s B:s C:s {
  DICE: [sort {A, B, C}]
  
  if ( 1@DICE = 2@DICE & 2@DICE = 3@DICE ) { ML: 1@DICE + 2@DICE + 3@DICE}
  else if ( 2@DICE = 3@DICE ) { ML: 2@DICE + 3@DICE}
  else if ( 1@DICE = 2@DICE ) { ML: 3@DICE }
  else { ML: 2@DICE }

  result: ML
}

function: low in A:s B:s C:s {
  DICE: [sort {A, B, C}]
  
  if ( 1@DICE = 2@DICE & 2@DICE = 3@DICE ) { L: 1@DICE + 2@DICE + 3@DICE}
  else if ( 2@DICE = 3@DICE ) { L: 2@DICE + 3@DICE}
  else { L: 3@DICE }

  result: L
}

output [high in 1d10 1d10 1d10]
output [midhigh in 1d10 1d10 1d10]

\ "Inflorescence: Wildflower Cards and "5E Hack" Dice \ 

\Set Synchronicity Boundry\
function: sync boundary X{
 result: d{-dX,0,dX}
}

\Synchronicity Swing\
SYNC: [sync boundary 3]

\Draw Cards from Deck\
function: draw X cards{
 result: Xd{2,3,4,5,6,7,8,9,10,d{1,11}}
}

\Roll "Wildflower Hack" Dice\
function: roll X dice{
 result: Xd10
}

loop N over {d4}{
 output [draw N cards]+SYNC named "Draw [N]"
 output [roll N dice]+SYNC named "Roll [N]"
 output (([draw N cards]+SYNC)+([roll N dice]+SYNC))/2 named "Average of [N]"
}

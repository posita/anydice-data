AC: 15

function: attack DAMAGE:d HIT:n AC:n REPEAT:n {
  TEMP_PRINT: 0
  loop N over {1..REPEAT} {
    CHANCES: [highest of [lowest of (1d20 - AC + HIT) and 1] and 0] * DAMAGE
    TEMP_PRINT: TEMP_PRINT + CHANCES
  }
  result: TEMP_PRINT
}

output [attack 1d8+5 8 AC 1] named "Unarmed Strike Bonus Action"

output [attack 1d8+5 8 AC 2] named "Flurry Of Blows Bonus Action"

output [attack 1d8+5 8 AC 2] named "Unarmed Strike Action"

function: blades ROLL:s {
 if 1@ROLL = 6 {
  result: 2 + (2@ROLL = 6)
 }
 result: 1@ROLL >= 4
}

output [blades 2@2d6] named "blades 0d"
loop P over {1..7} {
 output [blades Pd6] named "blades [P]d"
}

loop DICE over {1..10} {
 output DICE d (d6 = 1) > 0 named "[DICE]d6, trying for 1"
}

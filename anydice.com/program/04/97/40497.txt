AC: 15

function: attack DAMAGE:d HIT:n AC:n REPEAT:n {
  TEMP_PRINT: 0
  loop N over {1..REPEAT} {
    CHANCES: [highest of [lowest of (1d20 - AC + HIT) and 1] and 0] * DAMAGE
    TEMP_PRINT: TEMP_PRINT + CHANCES
  }
  result: TEMP_PRINT
}

output [attack 1d6+5 10 AC 1] named "Energy Short bow Action"

output [attack 1d8+5 10 AC 2] named "Longbow Action"

output [attack 1d6+2d6+5 10 AC 2] named "Ensnaring Strike Action STR:13"

output 1d6 named "Hunters Mark"

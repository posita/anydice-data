
function: attack DAMAGE:d HIT:n AC:n REPEAT:n {
  TEMP_PRINT: 0
  loop N over {1..REPEAT} {
    CHANCES: [highest of [lowest of ([lowest 1 of 1d20] - AC + HIT) and 1] and 0] * DAMAGE
    TEMP_PRINT: TEMP_PRINT + CHANCES
  }
  result: TEMP_PRINT
}

output [attack 5d8 5 15 1] named "Action"

output [attack 1d6+1 5 15 5] named "Bonus Action"

output [attack 5d8+5d6+5 9 15 1] named "Total Damage"

A: d20
ARM: 0
MOD: 6
ACN: 1
AC: (A = 20) | (A + MOD >= ARM)
ACC: (A = 19) | (AC)
M: ((A = 20) + 1)
MC: ((A >= 19) + 1)
output AC  * (ACN*(Md10 + 4) + Md4 + 4) named "Bare"
output ACC * (ACN*(MCd10 + 6) + MCd4 + 6) named "Curse"
output AC  * (ACN*(Md10 + 4) + Md4 + 4 + ((1 + ACN) * M)d6) named "Hex"
output ACC * (ACN*(MCd10 + 6) + MCd4 + 6 + ((1 + ACN) * MC)d6) named "Curse + Hex"
output AC  * (ACN*(Md10 + 4) + ACN * Md6) named "Hex + hit"
output ACC * (ACN*(MCd10 + 6)) named "Curse + hit"

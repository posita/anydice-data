T:6
N:4
DICE: {6,8,12}
loop I over {1..(#DICE-1)} {
  X: I@DICE
  Y: (I+1)@DICE
  output Nd{0:(T-1), 1:X-(T-1)} named "[N]d[X]"
  loop B over {1..(N-1)} {
    A: N-B
    output Ad{0:(T-1), 1:X-(T-1)} + Bd{0:(T-1), 1:Y-(T-1)} named "[A]d[X] and [B]d[Y]" 
  }
}
output Nd{0:(T-1), 1:Y-(T-1)} named "[N]d[Y]"

function: ROLL:n vs AC:n {
 if ROLL >= THREAT & ROLL + ATTACK >= AC {
  result: [threat d20]
 }
 if ROLL + ATTACK >= AC | ROLL = 20 { result: DAMAGE }
 result: 0
}

function: threat ROLL:n {
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: CRITICAL d DAMAGE
 }
 result: DAMAGE
}

AC: d{14..20}
STRENGTH: 3
BAB: 3

DAMAGE: 1d12 + STRENGTH * 3/2
THREAT: 20
CRITICAL: 3
ATTACK: BAB + STRENGTH + 1
output [d20 vs AC] named "Masterwork Greataxe"

DAMAGE: 1d12 + BAB * 2 + STRENGTH * 3/2
THREAT: 20
CRITICAL: 3
ATTACK: STRENGTH + 1
output [d20 vs AC] named "Masterwork Greataxe (Max Power)"

STRENGTH: 5

DAMAGE: 1d12 + STRENGTH * 3/2
THREAT: 20
CRITICAL: 3
ATTACK: BAB + STRENGTH + 1
output [d20 vs AC] named "Masterwork Greataxe (Bull's Strength)"

DAMAGE: 1d12 + BAB * 2 + STRENGTH * 3/2
THREAT: 20
CRITICAL: 3
ATTACK: STRENGTH + 1

function: glaive ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: GLAIVECRIT } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: GLAIVE }
 result: 0
}

function: handaxe ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: HANDAXECRIT } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: HANDAXE }
 result: 0
}

function: handaxetwf ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: HANDAXECRIT-ATTACKMOD } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: HANDAXE-ATTACKMOD }
 result: 0
}

ENEMYAC: d{12}
ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 2

GLAIVE: 1d10 + ATTACKMOD + DMGBONUS
GLAIVECRIT: 2d10 + ATTACKMOD + DMGBONUS

HANDAXE: 1d6 + ATTACKMOD + DMGBONUS
HANDAXECRIT: 2d6 + ATTACKMOD + DMGBONUS

output ATTACKTIMESd[glaive d20 vs ENEMYAC] named "glaive"
output ATTACKTIMESd[handaxe d20 vs ENEMYAC] named "handaxe"
output ATTACKTIMESd[handaxe d20 vs ENEMYAC] + d[handaxetwf d20 vs ENEMYAC] named "handaxe + handaxetwf"

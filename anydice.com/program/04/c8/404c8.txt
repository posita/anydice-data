function: FDICE:s vs EDICE:s with fight FFV:n vs EFV:n{
  FROLL: {1..1}@FDICE
  EROLL: {1..1}@EDICE
  WIN: 1
  if FROLL > EROLL {WIN: 1}
  if FROLL = EROLL {
    if FFV > EFV {WIN: 1} 
    else {if FFV < EFV {WIN: 0}
       else {WIN: d{0, 1}}}}
  if FROLL < EROLL {WIN: 0}

  result: WIN
}


function: THDICE:s plus FDICE:s vs EDICE:s with fight FFV:n vs EFV:n{
  FROLL: {1..1}@FDICE
  THROLL: {1..1}@THDICE
  if THROLL > FROLL {FROLL: THROLL}
  EROLL: {1..1}@EDICE
  WIN: 1
  if FROLL > EROLL {WIN: 1}
  if FROLL = EROLL {
    if FFV > EFV {WIN: 1} 
    else {if FFV < EFV {WIN: 0}
       else {WIN: d{0, 1}}}}
  if FROLL < EROLL {WIN: 0}

  result: WIN
}

  result: WOUNDS >= NWOUND
}

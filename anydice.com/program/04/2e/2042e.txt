\Overchannel Math\

function: overchannel ROLL:n vs DC:n {
 if ROLL + MOD <= DC { result: DMG }
 if ROLL + MOD > DC { result: DMG/2 }
}

function: overcharge ROLL:n vs DC:n {
 if ROLL + MOD <= DC { result: DMG_O }
 if ROLL + MOD > DC { result: DMG_O/2 }
}

function: overchargemax ROLL:n vs DC:n {
 if ROLL + MOD <= DC { result: DMGM }
 if ROLL + MOD > DC { result: DMGM/2 }
}


X:10
Y:6
I:5
PLAYERDC: 18
MOD: 3
DMG: [maximum of XdY]+5
DMG_O: (X+2)dY + 2*(I+4)+5
DMGM: (X+4)dY + 2*(I+6)+5
output [overcharge d20 vs PLAYERDC] named "Overcharge Fireball @14"

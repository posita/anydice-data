function: ROLL:n vs AC:n {
 if ROLL >= THREAT & ROLL + ATTACK >= AC {
  result: [threat d20]
 }
 if ROLL + ATTACK >= AC | ROLL = 20 { result: DAMAGE }
 result: 0
}

function: threat ROLL:n {
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: CRITICAL d DAMAGE
 }
 result: DAMAGE
}

AC: d{14..20}
STRENGTH: 3
BAB: 1

ATTACK: BAB + STRENGTH
DAMAGE: 2d6 + STRENGTH * 3/2
THREAT: 19
CRITICAL: 2
output [d20 vs AC] named "Greatsword (Basic Attack)"

ATTACK: STRENGTH
DAMAGE: 2d6 + 2*BAB + STRENGTH * 3/2
THREAT: 19
CRITICAL: 2
output [d20 vs AC] named "Greatsword (Power Attack)"

function: cofdseries N:n ROTE:n AGAIN:n{
 TEMP: 0
 loop I over {1..N} {
 TEMP: TEMP+[cofd 1d10 ROTE AGAIN]
 }
 result: TEMP
}

function: cofd R:n ROTE:n AGAIN:n {
   if R >= AGAIN { result: [cofd 1d10 ROTE AGAIN]+1 }
   else if R >= 8 {result: 1}
   else {
    if ROTE = 1 {result: [cofd 1d10 0 AGAIN]}
    else { result: 0 }
   }
}

output [cofdseries 7 0 10]

function: keep and sum A:n of B:d {
 C: 0
 if A < 1 { result: 0 }
 if A > #(B) { A: #(B) }
 loop N over {1..A} {
   C: C + N@B
 }
 result: C
}

S: {4, 6, 8, 10}
loop Y over S {
  loop X over {1..10} {
    loop Z over {1..5} {
      if (Z <= X) { output [keep and sum Z of XdY] named "[X]d[Y] Keep [Z]" }
    }
  }
}

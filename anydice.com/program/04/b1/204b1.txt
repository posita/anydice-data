function: contains VALUES:s in SEQUENCE:s {
 FIVES: 0
 SIXES: 0
 loop P over {1..#VALUES} {
 if (5 = SEQUENCE) {
  if FIVES = 0 {
   FIVES: 1
  }
 }
  SIXES: SIXES + (P@VALUES = SEQUENCE)
 }
 result: FIVES + SIXES
}

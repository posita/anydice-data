function: death D:n P:n F:n {
  if P>=3 {
    result: 1
  } else
  if F>=3 {
    result: 0
  } else
  if D=20 {
    result: 2
  } else
  if D>=10 {
    result: [death d20 P+1 F]
  } else
  if D>=2 {
    result: [death d20 P F+1]
  } else
  if D=1 {
    result: [death d20 P F+2]
  }
}
output [death d20 2 2]

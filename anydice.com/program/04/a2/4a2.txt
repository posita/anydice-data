function: hit chance ATTACKROLL:n with ATTACKBONUS vs DC with crit on CRIT:n {
  if ATTACKROLL = 0 { result: 0 }
  if ATTACKROLL >= CRIT {
    if ATTACKROLL + ATTACKBONUS >= DC { result: 0}
    result: 1
  }
  if ATTACKROLL + ATTACKBONUS >= DC { result: 1 }
  result: 0
}

function: crit chance ATTACKROLL:n with ATTACKBONUS vs DC with crit on CRIT:n {
  if ATTACKROLL >= CRIT {
    if ATTACKROLL + ATTACKBONUS >= DC { result: 1 }
    result: 0
  }
  result: 0
}

function: ATTACKROLL:n with ATTACKBONUS:n vs DC:n for DAMAGE damage plus BONUS on crit of CRIT:n {
  HITDAMAGE: [hit chance ATTACKROLL with ATTACKBONUS vs DC with crit on CRIT] * DAMAGE
  CRITDAMAGE: [crit chance ATTACKROLL with ATTACKBONUS vs DC with crit on CRIT] * ([maximum of DAMAGE]+BONUS)
  result: HITDAMAGE+CRITDAMAGE
}

function: NUM attacks of ATTACKROLL:n with ATTACKBONUS:n vs DC:n for DAMAGE damage plus CRITDAMAGE on crit of CRIT:n plus STRIKERDAMAGE {
  HITS: 0
  CRITS: 0
  TOTALDAMAGE: 0
  loop N over {1..NUM} {
    HITRESULT: [hit chance ATTACKROLL with ATTACKBONUS vs DC with crit on CRIT]
    CRITRESULT: [crit chance ATTACKROLL with ATTACKBONUS vs DC with crit on CRIT]
    if HITRESULT > 0 {HITS: HITS + 1}
    if CRITRESULT > 0 {CRITS: CRITS + 1}
    TOTALDAMAGE: TOTALDAMAGE + (HITRESULT * DAMAGE) + (CRITRESULT * ([maximum of DAMAGE] + CRITDAMAGE))
  }
  if CRITS >= 1 { TOTALDAMAGE: TOTALDAMAGE + [maximum of STRIKERDAMAGE] }
  else { if HITS >= 1 { TOTALDAMAGE: TOTALDAMAGE + STRIKERDAMAGE } }
  result: TOTALDAMAGE
}

function: BONUS bonus damage on NUM hits in ATTACKRESULT:n {
  if ATTACKRESULT > 0 { result: ATTACKRESULT + BONUS }
  result: ATTACKRESULT
}

  result: WEAPONDAMAGE

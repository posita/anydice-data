}

  if ATTACK >= DC {
    result: [DAMAGE damage with crit on CRIT and BONUS bonus]
  } else {
    result: 0
  }
}

  loop N over {1..NUM}{
    TOTAL: TOTAL + [ATTACK vs DC with DAMAGE damage with crit on CRIT and BONUS bonus]
  }

}

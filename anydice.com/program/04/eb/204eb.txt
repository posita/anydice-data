AC: 0

DMGB: 1d8+11
CTA: 17
CTB: 20
ATC: {15,10,4}
ATD: {19,13,9}
DMGC: 1d8+14
DMGD: 1d8+17
CTC: 17
CTD: 20
CB: 0
CM: 2

function: A:n B:n hit C:n {
result: (A=20 | ((A+B)>=C)) & A!=1}

function: A:n B:n atk D:n C:n {
result: [A B hit AC]*(D)*((A>=C & [1d20 B+CB hit AC])*(CM-1)+1)
}

function: A iterative B C:n{
I:0
X:0
loop I over A{
X:X+[1d20 I atk B C]
}
result: X
}

output[ATC iterative DMGC CTC]+[ATD iterative DMGD CTD]

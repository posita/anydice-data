function: wren ROLL:s crit at B:n plus {
 if 1@ROLL = 1 {
   result: 0
 }
 if 1@ROLL >= B {
    result: 1
 }
 if 2@ROLL = 1@ROLL {
    result: 1
 }
 result: 0
}

function: wren attack N:n times crit at A:n {
 D: 0
 loop C over {1..N}{
  D: D + [wren 2d20 crit at A plus]
 }
 result: D
}

\Potent Cantrip Replacement Math\

function: potent ROLL:n vs DC:n {
 if ROLL + MOD <= DC { result: DMG }
 if ROLL + MOD > DC { result: HALFDMG }
}

function: fundamental ROLL:n vs DC:n {
 if ROLL + MOD <= DC { result: DMG_A}
 if ROLL + MOD > DC { result: 0}
}

function: fundamentalb ROLL:n vs DC:n {
 if ROLL + MOD <= DC { result: DMG_B}
 if ROLL + MOD > DC { result: 0}
}

function: std ROLL:n vs DC:n {
 if ROLL + MOD <= DC { result: DMG}
 if ROLL + MOD > DC { result: 0}
}


\X is number of dice 2-4(6th lvl - 17th lvl)\
\Y is die size (6,8,10,12) for wizard damage cantrips\
\PLAYERDC = Spell save DC\
\ MOD = target save modifier\

X:2
Y:8
PLAYERDC: 16
MOD: 2
DMG: XdY
DMG_A: [highest of XdY and [highest of XdY and XdY]]
DMG_B: [highest X of (2*X)dY]
HALFDMG: XdY/2

output [potent d20 vs PLAYERDC] named "With Potent"
output [fundamental d20 vs PLAYERDC] named "With Potent v1.1"
output [fundamentalb d20 vs PLAYERDC] named "With Potent v1.2"
output [std d20 vs PLAYERDC] named "Standard"


function: blades ROLL:s DC {
 result: (1@ROLL >= 4) + (1@ROLL >= 6)
}
function: knave ROLL:s DC {
 result: 2 * (1d20+DC >= 15)
}
function: ogre ROLL:s DC {
 result: (1@ROLL <= DC) + (2@ROLL <= DC)
}
output [blades 2@2d6 3] named "0d"
loop P over {1..6} {
 output [blades Pd6 3] named "[P]d"
}
loop P over {0..10} {
 output [knave 1d20 P] named "d20+[P]"
}
loop P over {1..10} {
 output [ogre 2d12 P] named "under [P]"
}

\ Starting stat value... you can change this! \
STAT: 7

function: increaseroll STAT:n ROLL:n {
    TN: (21 - STAT)*5
    RESULT: STAT
    if ROLL <= TN {
         INC: 1d3-1
         RESULT: STAT + INC
    }
    result: RESULT
}

function: year STAT:n {
    ROLL: d100
    NEWSTAT: [increaseroll STAT ROLL]
    result: NEWSTAT
}

function: growth STAT:n {
    STAT: [year STAT]
    STAT: [year STAT]
    STAT: [year STAT]
    STAT: [year STAT]
    STAT: [year STAT]
    STAT: [year STAT]
    STAT: [year STAT]
    STAT: [year STAT]
    STAT: [year STAT]
    STAT: [year STAT]
    result: STAT
}

output [growth STAT]

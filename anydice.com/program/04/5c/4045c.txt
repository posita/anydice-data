MISS: 0
HIT: 2d6+4
CRITICALHIT: 4d6+4

function: attack ROLL:n vs DEFENSE:n{
 if ROLL=1{result: MISS}
 if ROLL=20{
  if ROLL+ATTACK>=DEFENSE{result: CRITICALHIT}
  result:HIT
 }
 if ROLL+ATTACK>=DEFENSE{result: HIT}
 result: MISS
}

ATTACK: 4+2
output [attack d20 vs 12] named "Total"
output d20+ATTACK named "ToHit"
output HIT named "Damage"

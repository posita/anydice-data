function: ATTACKROLL:n with ATTACKBONUS:n vs DC:n for DAMAGE damage plus BONUS on crit of CRIT:n {
  if ATTACKROLL = 0 { result: 0 }
  if ATTACKROLL >= CRIT {
    if ATTACKROLL + ATTACKBONUS >= DC { result: [maximum of DAMAGE]+BONUS }
    result: DAMAGE
  }
  if ATTACKROLL + ATTACKBONUS >= DC { result: DAMAGE }
  result: 0
}

function: BONUS bonus damage on NUM hits in ATTACKRESULT:n {
  if ATTACKRESULT > 0 { result: ATTACKRESULT + BONUS }
  result: ATTACKRESULT
}

function: NUM:n attacks of ATTACK:d with ATTACKBONUS:n vs DC:n for DAMAGE:d damage plus BONUS:d on crit of CRIT:n {
  TOTAL: 0
  loop N over {1..NUM}{
    TOTAL: TOTAL + [d20 with ATTACK vs DC for DAMAGE damage plus BONUS on crit of CRIT]
  }
  result: TOTAL
}

function: reroll DICE:n if BADRESULT as OTHERDICE:n{
   if DICE = BADRESULT {result: OTHERDICE}
   result: DICE
}

function: roll DICE:n twice reroll BADRESULT {
   result: [highest of [reroll DICE if 1 as DICE] and [reroll DICE if 1 as DICE]]
}

function: count VALUES:s in SEQUENCE:s with master M:n or expert E:n{
 COUNT: 0
 loop P over {1..#VALUES} {
  if (P@VALUES = SEQUENCE) >= COUNT {
    COUNT: (P@VALUES = SEQUENCE)
    if (P@VALUES = E) {
      COUNT: COUNT +1
    }
  }
 }
 result: COUNT+M
}

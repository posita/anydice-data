function: dicepool DA:s DB:s target TGT:n {
  SUCC: 0
  if DA < TGT { SUCC: SUCC + 1 }
  if DB < TGT { SUCC: SUCC + 1 }
  result: SUCC
}

loop TGT over {3..6} {
  loop DA over {6, 8, 12} {
    loop DB over {6, 8, 12} {
      if DA <= DB {
        if !(DA = 6 & DB = 12) {
          output [dicepool 1dDA 1dDB target TGT] named "[DA]/[DB]~[TGT]"
        }
      }
    }
  }
}

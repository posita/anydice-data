A: 3
D: 6
loop N over {(A+1)..(A+5)}{
  output [highest A of NdD] named "[N] standard d[D]"
  loop M over {0..2}{
    X: A-M
    output [highest (A-M) of (N+M)dD] named "([A]-[M])pulled of [N] d[D]"
  }
  loop O over {1..(N-A)} {
    Y: A+O
    output [highest (A) of (N-O)dD] named "([A]+[O])pushed of [N] d[D]"
  }
}

output 3d6

function: damage A:n {
  if A=2 { result: CRIT }
  result: A * DMG
}

function: roll H:n plus B:n against A:n {
  if H=1 { result: 0 }
  if H=20 { result: 2}
  if H+ATKB+B>=A { result:1 }
  result: 0
}

MOD: 3
PB: 2
MAXDMG: 16
DMGDIE: d8+d6+2
ROLL: [highest 1 of 1d20]
ROLLB: 0
RANGE: {12,15,18}

loop AC over RANGE {
  ATKB: MOD+PB
  DMG: DMGDIE+MOD
  CRIT: DMGDIE+MOD+MAXDMG
  
  NORMAL: [damage [roll ROLL plus ROLLB against AC]]
  output NORMAL named "Normal vs [AC]"
  
} loop AC over RANGE {
  ATKB: MOD+PB-5
  DMG: DMGDIE+MOD+10
  CRIT: DMGDIE+MOD+MAXDMG+10
  
  GREAT: [damage [roll ROLL plus ROLLB against AC]]
  output GREAT named "Great vs [AC]"
  
} loop AC over RANGE {
  ATKB: MOD+PB+1
  DMG: DMGDIE+MOD+1
  CRIT: DMGDIE+MOD+MAXDMG+1
  
  PLUS: [damage [roll ROLL plus ROLLB against AC]]
  output PLUS named "Ability+ vs [AC]"
}

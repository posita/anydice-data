function: pbta N:n success S:n partial P:n {
  if N >= S {
    result: 2
  } else if N >= P {
    result: 1
  } else {
    result: 0
  }
}

loop M over {-1..3} {
  if M > 0 {
    output [pbta 2d6-M success 10 partial 7] named "PbtA 2d6+[M]"
  } else if M = 0 {
    output [pbta 2d6-M success 10 partial 7] named "PbtA 2d6"
  } else {
    output [pbta 2d6-M success 10 partial 7] named "PbtA 2d6-[abs M]"
  }
}

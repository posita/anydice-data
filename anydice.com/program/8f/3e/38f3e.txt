function: helper DIE:n BY:n REPLACE:s WITH:n {
  loop X over REPLACE {
    if (DIE = X) {
      result: WITH
    }
  }
  result: DIE
}

function: bonus DIE:d BY:n {
  MAX: [maximum of DIE]
  result: [helper DIE BY {(MAX-BY)..(MAX-1)} MAX]
}

function: penalty DIE:d BY:n {
  result: [helper DIE BY {2..BY+1} 1]
}

output d6

output [bonus d6 1]
output [bonus d6 2]
output [bonus d8 1]
output [penalty d6 1]
output [penalty d6 2]
output [penalty d8 1]

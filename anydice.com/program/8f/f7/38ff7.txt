ATTACK_MOD: 10+2
HITDAMAGE_DICE: 1d10+3d8+1d4+1d6
HITDAMAGE_FIX: 10+2
NUMBER_OF_ATTACK: 4
CRITICAL_MINIMUM_ROLL: 20
MISSDAMAGE: 0
AC_TARGET: 18

MISS: 0
HIT: 1
CRITICALHIT: 2
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= CRITICAL_MINIMUM_ROLL { result: CRITICALHIT }
 if ROLL + ATTACK_MOD >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: HITDAMAGE_DICE*2+HITDAMAGE_FIX
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE_DICE+HITDAMAGE_FIX
 }
 result: MISSDAMAGE
}

output [damage for [attack d20 vs AC_TARGET]]*NUMBER_OF_ATTACK

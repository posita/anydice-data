function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

function: gwf DICE:d {
    result: [reroll {1,2} on DICE]
}

function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d or crit CRIT_DMG:d on CRIT_RANGE:n or VORPAL
{
    if ROLL = 1           {result: 0 }
    if ROLL = 20          {result: VORPAL}
    if ROLL >= CRIT_RANGE {result: CRIT_DMG}
    if ROLL + BONUS >= AC {result: DMG}
    result: 0
}

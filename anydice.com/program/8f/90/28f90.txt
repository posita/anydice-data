function: pbta N:n {
  if N >= 10 {
    result: 2
  } else if N >= 7 {
    result: 1
  } else {
    result: 0
  }
}

output d{} named "0 is failure, 1 is partial success, 2 is success"

loop M over {-2..3} {
  A: [absolute M]
  if M > 0 {
    output [pbta 2d6+M] named "PbtA 2d6+[A]"
  } else if M = 0 {
    output [pbta 2d6+M] named "PbtA 2d6"
  } else {
    output [pbta 2d6+M] named "PbtA 2d6-[A]"
  }
}

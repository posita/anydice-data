function: to hit zil AC:n {
  result: (d20+9)>=AC
}

function: to hit tham AC:n {
  result: (d20+8)>=AC
}

function: X:n crit {
if X = 20 {result: 1d8}
result: 0
}

function: X:n miss {
  if X = 1 {result: 0}
  result: 1
}

loop AC over {10..25} {
  output ((2d8+6)*[to hit zil AC]+[d20 crit])*[d20 miss] named "Zilarra at AC [AC]"

  output ((2d8+5)*[to hit tham AC]+[d20 crit])*[d20 miss] + ((1d8+5)*[to hit tham AC]+[d20 crit])*[d20 miss] named "Thamir at AC [AC]"
}

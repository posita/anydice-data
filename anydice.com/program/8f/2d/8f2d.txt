function: bonushit N:n TN:n{
 COUNT: 0
 ROLLD: 1d100
 loop P over {1..N} {
  if [ROLLD+5] > TN {result: COUNT}
  ROLLD: ROLLD + 5
  COUNT: COUNT + 1
 }
 result: COUNT
}




function: gun N:n ar AR:n ap AP:n dm W:n{
 COUNT: 0
 loop P over {1..N} {
  COUNT: COUNT + [checkhit 2d6]
 }
 result: COUNT*W
}
function: miss N:n ar AR:n ap AP:n dm W:n{
 COUNT: 0
 loop P over {1..N} {
  COUNT: COUNT + [checkmiss 3d6]
 }
 result: COUNT*W
}
function: mortar N:n ar AR:n{
 AP: 2
 W: 1
 COUNT: 0
 loop P over {1..N} {
  COUNT: COUNT + [checkmor 2d6 1d6]
 }
 result: COUNT*W
}
function: checkmiss HH:n {
 if HH >= 15 {result: [checksv 1]}
 if HH + AR > DR {result: [checksv 0]}
 result: 0
}
function: checkhit HH:n {
 if HH >= 11 {result: [checksv 1]}
 if HH + AR > DR+DRM {result: [checksv 0]}
 result: 0
}
function: checkmor HH:n SC:n{
 if HH >= 11 {result: [checksv 1]}
 if HH + AR > DR+DRM {result: [checksv 0]}
 if SC = 1 {result: [checksv -2]}
 result: 0
}
function: checksv VV:n {
 result: (d6 < SV+AP+VV)
}

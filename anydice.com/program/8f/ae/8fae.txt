A: 3
D: 6
loop N over {(A+1)..(A+5)}{
  output [highest A of NdD] named "[N] standard d[D]"
  loop M over {0..2}{
    X: A-M
    output [highest (A-M) of (N+M)dD] named "([A]-[M])pulled of [N] d[D]"
  }
  loop O over {1..(N-A)/2} {
    Y: A+O
    output [highest (A+O) of [lowest (N-A) of NdD]] named "([A]+[O])pushed of [N] d[D]"
  }
}

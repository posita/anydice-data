function: aw A:n {
   if A >= 10 { result: 2 }
   if A >= 7 { result: 1 }
   result: 0
}

function: hl S:s T:n {
  H: 1@S
  L: 2@S
  if H >= T & L >= T { result: 2 }
  if H >= T & L < T { result: 1 }
  result: 0
}


loop N over {-1..3} {
 output [aw 2d6+N] named "AW + [N]"
}


loop N over [reverse {5..17}] {
 output [hl 2d20 N] named "hl [N]"
}

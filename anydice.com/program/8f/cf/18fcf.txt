function: wild DIE:d {
 WILD: [maximum of DIE]
 result: [wild DIE helper]
}

function: wild N:n helper {
 if N = 1 { result: -1 + [wild DIE helper] }
 if N < WILD { result: 0 }
 result: 1 + [wild DIE helper]
}

function: luck D:d {
 LUCK: [maximum of D]
 result: [luck D helper]
}

function: luck L:n helper {
 if L = 1 { result: -1 }
 if L = LUCK { result: 1 }
 result: 0
}

function: roll LEVEL:n {
 L: LEVEL
 loop X over {1..LEVEL} {
  L: L + [luck d6]
 }
 result: L + [wild d6]
}

loop N over {0..5}
{
 output [roll N] named "Level [N]"
}


function: clamp NUMBER:n between MIN:n and MAX:n {
  if NUMBER < MIN { result: MIN }
  if NUMBER > MAX { result: MAX }
  result: NUMBER
}



output 4d{-1,0,1} named "4df" 
output d5-d5 named "d5-d5"
loop N over {5..10} {
  output [ clamp Nd{-1,0,1} between -4 and 4 ] named "[N]df clamped to -4,4"
}

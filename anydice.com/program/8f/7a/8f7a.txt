ATT: 7
POW: 12
DEF: 11
ARM: 16
HITDICE: 2d6
DMGDICE: 3d6

function: tohit ROLL:s {result:((ROLL + ATT) >= DEF)}
function: damage ROLL:s {result: ((ROLL + POW) >= ARM)*((ROLL + POW) - ARM)}

function: warmachine HROLL:s DROLL:s {
if ((HROLL + ATT) >= DEF) & ((DROLL + POW) >= ARM){
result: ((DROLL + POW) - ARM)}
else {
result: 0}
}

output [warmachine HITDICE DMGDICE] named "Warmachine"
output [tohit HITDICE] named "Hit"

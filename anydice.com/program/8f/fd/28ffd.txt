\ Attack Functions \

function: attack ROLL:n vs AC:n {
 if ROLL = 1 { result: FAILURE }
 if ROLL = 20 { result: CRIT }
 if ROLL + ATTACK >= AC { result: SUCCESS }
 result: FAILURE
}

function: damage for ATTACKRESULTS:s {
 D: 0
 loop A over ATTACKRESULTS {
  if A = CRIT {
   D: D + CRIT_DAMAGE
  }
  else if A = SUCCESS {
   D: D + HIT_DAMAGE
  }
  else {
   D: D + MISS_DAMAGE
  }
 }
 result: D
}

function: attack {
 result: [damage for [attack regular vs AC_RANGE]]
}

function: N:n attacks {
 result: [damage for Nd[attack d20 vs AC_RANGE]]
}

function: damage A:n {
  if A=2 { result: CRIT }
  result: A * DMG
}

function: roll H:n against A:n {
  if H=1 { result: 0 }
  if H=20 { result: 2}
  if H+ATKB>=A { result:1 }
  result: 0
}

loop AC over {12,15,18} {
  ATKB: 5
  DMG: d8+3
  CRIT: d8+11
  
  output [damage [roll d20 against AC]] named "Normal vs [AC]"
  
  ATKB: 0
  DMG: d8+13
  CRIT: d8+21
  
  output [damage [roll d20 against AC]] named "Great Weapon vs [AC]"
  
  ATKB: 6
  DMG: d8+4
  CRIT: d8+12
  
  output [damage [roll d20 against AC]] named "Ability+ vs [AC]"
}

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
if ROLL >= CRITROLL { result: CRITDAMAGE }
if ROLL = 1 { result: 0 }
if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
result: 0
}

function: boons BOON:n {
if BOON >= 1 {result: [highest 1 of BOONd6] }
if BOON <= 0 {result: -[highest 1 of -BOONd6] }
result: 0
}

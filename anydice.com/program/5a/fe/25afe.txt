function: ROLL:n with BONUS:n hits AC:n {
  if ROLL = 1 {
    result: MISS
  }
  if ROLL = 20 {
    if ROLL + BONUS > AC {
      result: CRIT
    }

    result: HIT
  }
  if ROLL + BONUS > AC {
    result: HIT
  }
  if ROLL + BONUS = AC {
    result: GLANCE
  }
  result: MISS
}

function: ROLL:n with BONUS:n saves against DC:n {
  if ROLL + BONUS >= DC {
    result: GLANCE
  }

  result: HIT
}

  if RESULT = CRIT {
    result: [maximum of DAMAGE] + DAMAGE + BONUS

  if RESULT = HIT {
    result: DAMAGE + BONUS
  }
  if RESULT = GLANCE {
    result: (DAMAGE + BONUS) / 2
  }

  result: 0
}

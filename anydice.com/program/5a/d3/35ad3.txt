function: powerroll on DICE:n and BONUS:n {
 if DICE + BONUS < 8 {
  result: 1
 }
 if DICE + BONUS > 7 {
  if DICE + BONUS < 10 {
   result: 2
  }
 }
 if DICE + BONUS > 10 {
  result: 3
 }
}

output 1d20 named "Flat d20 roll"
output 2d6 named "Flat 2d6 roll"
output [powerroll on 2d6 and 0] named "Power Roll +0"

function: gimme dmg M:n S:n L:n P:n {
 if P = 1 {A: [highest of (((100+(M*L))*4)/10)+1d101-51 and 0]}
 else     {A: [highest of (((100+(S*L))*4)/10)+1d101-51 and 0]}

 result: A
}

function: cumu dmg M:n S:n L:n {

}

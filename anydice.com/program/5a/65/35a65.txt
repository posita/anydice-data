5d4={0,1,2,2,3}
LR = 2++ to 3++ (1++ is the same as 1+).
HR = 4++ to 6++.
OF = x++ with min target overflowing sum: for 4++, (3+2) ok.
ES = x++ with exact target sum: for 4++, (3+1) ok but (3+2) not ok.

\ Dices based on D4 \
_QUATRE: d{0..3}
_DEUX: d{0..1}
_QUATRE_TO_TROIS: d{0,1,1,2}
Q: _QUATRE

function: count with pairing TARGETS:s in SEQUENCE:s with low range LR:n failures count FC:n and exact sum ES:n {
 REVERSED: [reverse SEQUENCE]
 TARGETS: [sort TARGETS]
 SIZEMAX: #REVERSED
 SIZEMIN: 1
 COUNT: [count TARGETS in SEQUENCE]
 loop N over {1..#SEQUENCE} {
  END: 0
  if N@SEQUENCE < #TARGETS@TARGETS  {
    loop Q over {(SIZEMIN)..(SIZEMAX-N)} {
       if (END=0) & (ES=0) {
         if ((N@SEQUENCE + Q@REVERSED) >= #TARGETS@TARGETS) {
           COUNT: COUNT + 1
           END: 1
           SIZEMIN: Q+1
         }
       }
       if (END=0) & (ES=1) {
         if ((N@SEQUENCE + Q@REVERSED) = #TARGETS@TARGETS) {
           COUNT: COUNT + 1
           END: 1
           SIZEMIN: Q+1
         }
       }
    }
  }
 }
 if (FC=1) {
   COUNT: #SEQUENCE - COUNT
 }
 result: COUNT
}

function: count pairing TARGET:n in SEQUENCE:s {
 SIZEMAX: #SEQUENCE
 SIZEMIN: 1
 COUNT: 0
 loop M over {1..#SEQUENCE} {
  END: 0

  if (M@SEQUENCE >= TARGET) { COUNT: COUNT + 1 }            \COUNT: COUNT + (M@SEQUENCE>=TARGET)\

  if (M@SEQUENCE < TARGET) & (M@SEQUENCE > 0) {
    loop N over {(SIZEMIN)..(SIZEMAX-M)} {
       if (END=0) {
         if ((M@SEQUENCE + (SIZEMAX-N+1)@SEQUENCE) >= TARGET) {
           COUNT: COUNT + 1
           END: 1
           SIZEMIN: N+1
         }
       }
    }
  }
 }
 result: COUNT
}

function: sets DIE:d NUMBER:n TARGET:n {
 result: [setshelper NUMBERdDIE {DIE} TARGET]
}

function: setshelper DICE:s DIESET:s TARGET:n {
 TOTAL: 0
 loop X over {1..#DIESET} {
  if [count X@DIESET in DICE] = TARGET {
   TOTAL: TOTAL + 1
  }
 }
 result: TOTAL
}

 loop B over {2..4} {
loop A over {2..6} {
   if (A >= B) {
     output [sets d6 A B] named "[A]d6 sets of [B]" 
   }
 }
}

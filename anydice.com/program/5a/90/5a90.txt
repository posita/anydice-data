
function: ROLL:s a dificultad DIFICULTAD:n 
{
 result: (ROLL >= DIFICULTAD) - (ROLL = 1)
}

function: ROLL:s a dificultad DIFICULTAD:n con especialidad
{
 result: (ROLL = 10) + (ROLL >= DIFICULTAD) - (ROLL = 1)
}


function: DADOS:n dados a dificultad DIFICULTAD:n 
{
  R: 0
  loop N over {1..DADOS}
  {
   R: R + [1d10 a dificultad DIFICULTAD] 
  }
  result: R
}

function: DADOS:n dados a dificultad DIFICULTAD:n con especialidad
{
  R: 0
  loop N over {1..DADOS}
  {
   R: R + [1d10 a dificultad DIFICULTAD con especialidad] 
  }
  result: R
}

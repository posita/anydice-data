function: compare A:n plus S:n to X:n and Y:n {
 if A + S > X {
  if A + S > Y {
   result : 2
  }
  else {
    result : 1
  }
 }
 else {
  if A + S > Y {
   result : 1
  }
  else {
    result : 0
  }
 }
}

function: compare A:n plus S:n to X:n and Y:n with forced reroll lowest challenge die {
  if X < Y {
    result: [compare A plus S to d10 and Y]
  }
  else {
    result: [compare A plus S to X and d10]
  }
}

output [compare 1d6 plus 1 to d10 and d10] named "+1"
output [compare 1d6 plus 1 to d10 and d10 with forced reroll lowest challenge die] named "+1 with forced reroll lowest challenge die"

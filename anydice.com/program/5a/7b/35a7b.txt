function: overflow pairing SEQUENCE:s on TARGET:n{
 SIZEMAX: #SEQUENCE
 SIZEMIN: 1
 COUNT: 0
 loop M over {1..#SEQUENCE} {
  END: 0

  if (M@SEQUENCE >= TARGET) { COUNT: COUNT + 1 }            \COUNT: COUNT + (M@SEQUENCE>=TARGET)\

  if (M@SEQUENCE < TARGET) & (M@SEQUENCE > 0) {
    loop N over {(SIZEMIN)..(SIZEMAX-M)} {
       if (END=0) {
         if ((M@SEQUENCE + (SIZEMAX-N+1)@SEQUENCE) >= TARGET) {
           COUNT: COUNT + 1
           END: 1
           SIZEMIN: N+1
         }
       }
    }
  }
 }
 result: COUNT
}

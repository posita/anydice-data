function: count pairing TARGET:n in SEQUENCE:s {
 REVERSED: [reverse SEQUENCE]
 SIZEMAX: #SEQUENCE
 SIZEMIN: 1
 COUNT: 0
 loop M over {1..#SEQUENCE} {
  END: 0
  COUNT: COUNT + (M@SEQUENCE>=TARGET)
  \if (M@SEQUENCE >= TARGET) { COUNT: COUNT + 1 }\

  if (M@SEQUENCE < TARGET) & (M@SEQUENCE > 0) {
    loop N over {(SIZEMIN)..(SIZEMAX-M)} {
       if (END=0) {
         if ((M@SEQUENCE + N@REVERSED) >= TARGET) {
           COUNT: COUNT + 1
           END: 1
           SIZEMIN: N+1
         }
       }
    }
  }
 }
 result: COUNT
}


function: bis count pairing TARGET:n in SEQUENCE:s {
 SIZEMAX: #SEQUENCE
 SIZEMIN: 1
 COUNT: 0
 loop M over {1..#SEQUENCE} {
  END: 0

  \COUNT: COUNT + (M@SEQUENCE>=TARGET)\
  if (M@SEQUENCE >= TARGET) { COUNT: COUNT + 1 }

  if (M@SEQUENCE < TARGET) & (M@SEQUENCE > 0) {
    loop N over {(SIZEMIN)..(SIZEMAX-M)} {
       if (END=0) {
         if ((M@SEQUENCE + (SIZEMAX-N+1)@SEQUENCE) >= TARGET) {
           COUNT: COUNT + 1
           END: 1
           SIZEMIN: N+1
         }
       }
    }
  }
 }
 result: COUNT
}


function: tris count pairing TARGET:n in SEQUENCE:s {
 SIZEMAX: #SEQUENCE
 SIZEMIN: 1
 COUNTF: 0
 COUNT: 0
 loop M over {1..#SEQUENCE} {
  END: 0

  if (M@SEQUENCE < TARGET) {
  COUNTF: COUNTF + 1                                    \count Failures\
  if (M@SEQUENCE > 0) {
    loop N over {SIZEMIN..(SIZEMAX-M)} {
       if (END=0) {
         if ((M@SEQUENCE + (SIZEMAX-N+1)@SEQUENCE) >= TARGET) {
           COUNT: COUNT + 1                                              \count Successes by pairing\
           END: 1
           SIZEMIN: N+1
         }
       }
    }
  }}
 }
 result: SIZEMAX-COUNTF+COUNT
}

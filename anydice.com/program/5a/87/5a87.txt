function: sets NUMBER:n DIE:d {
 result: [setshelper NUMBERdDIE {DIE}]
}

function: setshelper DICE:s DIESET:s {
 TOTAL: 0
 loop X over {1..#DIESET} {
  Y: [count X@DIESET in DICE]
  if (Y > 1) {
   TOTAL: TOTAL + (Y - 1)
  }
 }
 result: TOTAL
}

loop X over {2..8} {
 output [sets X 1d6] named "[X] dice"
}

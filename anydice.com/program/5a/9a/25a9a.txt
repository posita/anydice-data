function: critdiff ROLL:s {
 SUCCESSES: [count {4, 5, 6} in ROLL]
 FUMBLES: [count 1 in ROLL]
 result: SUCCESSES-FUMBLES
}

function: explode DIE:d {
  result: [explode DIE helper]
}

function: explode N:n helper {
  if N=6 { result: 1 + [explode DIE helper] }
  else if N>3 { result: 1 }
  else { result: 0 }
}

loop N over {3..8} {
  output [critdiff Nd[explode d6]]<0 named "[N]d6"
}

function: pierce DIE:d {
 MAX: [maximum of DIE]
 result: [pierce DIE helper]
}

function: pierce N:n helper {
 if MAX = 100 {
  if N = MAX {
   result: N - 1 + [pierce 1d20]
  }
 }
 else if MAX = 20 {
  if N = MAX {
   result: N - 1 + [pierce 1d6]
  }
 }
 else if N = MAX {
  result: N - 1 + [pierce DIE helper]
 }
 result: N
}

set "maximum function depth" to 6

output [pierce 1d100] named "1d100p"
output [pierce 1d20] named "1d20p"
output [pierce 1d12] named "1d12p"
output [pierce 1d10] named "1d10p"
output [pierce 1d8] named "1d8p"
output [pierce 1d6] named "1d6p"
output [pierce 1d4] named "1d4p"
output [pierce 1d3] named "1d3p"

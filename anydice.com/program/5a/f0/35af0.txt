MISS: 0
HIT: 1
CRITICAL: 2

function: skill ROLL:n vs TARGET:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + BONUS >= TARGET { result: CRITICAL }
  result: HIT
 }
 if ROLL + BONUS >= TARGET { result: HIT }
 result: MISS
}

BONUS: 3
loop DC over {1..20} {
 output [skill 2d8 vs DC] named "[DC]"
}

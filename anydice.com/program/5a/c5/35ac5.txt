DIFFICULTY: 0
DANGER_PER_HIT: 0

function: roundup D:n {
  if (D < 0){
    result: 0
  }
  result: D 
}

function: rounddown D:n {
  if (D > EVASIONDICE * DANGER_PER_HIT){
    result: EVASIONDICE * DANGER_PER_HIT
  }
  result: D 
}

function: danger gained with armor A:d evasion E:n {
  RESULT: 1d0
  ROLL: d10 + E < DIFFICULTY
  loop X over {1..EVASIONDICE}{
    RESULT: RESULT + (ROLL * DANGER_PER_HIT)
  }
  RESULT: RESULT - A

  RESULT: [roundup RESULT]

  result: RESULT
}

function: danger prevented with armor A:d evasion E:n {
  RESULT: 1d0
  ROLL: d10 + E >= DIFFICULTY
  loop X over {1..EVASIONDICE}{
    RESULT: RESULT + (ROLL * DANGER_PER_HIT)
  }
  RESULT: RESULT + A

  RESULT: [rounddown RESULT]

  result: RESULT
}

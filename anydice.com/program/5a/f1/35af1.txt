FAILURE: 0
SUCCESS: 1

function: skill ROLL:n vs DIFFICULTY:n {
 if HIGHEST 1 of ROLL >= DIFFICULTY { result: SUCCESS }
 result: FAILURE
}

loop DC over {1..10} {
 output [skill 1d4 vs DC] named "1d4"
 output [skill 1d6 vs DC] named "1d6"
 output [skill 1d8 vs DC] named "1d8"
 output [skill 1d10 vs DC] named "1d10"
 output [skill 1d12 vs DC] named "1d12"
}

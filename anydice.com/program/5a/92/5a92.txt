function: overmax A:n above B:n {
if A > B { result: A }
result: B
}

function: bonusdice B:n {
if B=10{result: d6}
result: 0
}

function: bonusdicec B:n {
if B=10{result: 1}
result: 0
}


SUM: 0
DICE: 0
MAX: 0
NBBONUS: 0
NUMBERS: 2

loop N over {1..NUMBERS}{
DICE: 1d10
MAX: [overmax DICE above MAX]
BONUS: [bonusdice MAX]
NBBONUS: NBBONUS+[bonusdicec MAX]
SUM: SUM+BONUS
}
SUM: SUM+MAX

output SUM named "Resultat"
output NBBONUS named "NB de d6 bonus"

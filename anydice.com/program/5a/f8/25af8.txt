function: attackroll BAB with ROLL:n vs AC:n {
 if ROLL >= CRITROLL & ROLL + BAB + STRMOD >= AC { result: [critconfirm BAB with d20]} 
 if ROLL = 1 { result: 0 }
 if ROLL + BAB + STRMOD >= AC | ROLL = 20 { result: DAMAGE }
 result: 0
}

function: multiattackroll {
 FINALDAMAGE: 0
 loop N over ATTACKS{
   FINALDAMAGE: 1d[attackroll N with d20 vs ENEMYAC] + FINALDAMAGE   
  }
  result: FINALDAMAGE
}

function: critconfirm BAB with ROLL:n { 
 if ROLL = 1 { result: DAMAGE }
 if ROLL + BAB + STRMOD >= AC | ROLL = 20 { result: CRITDAMAGE }
 result: DAMAGE
}

function: R:s and O:s with B:n bonuses and P:n penalties {
 result: R + O + [lowest of O = -1 and B] - [lowest of R = 1 and P]
}

RUNES: 3
OMENS: 3
BONUSES: 1
PENALTIES: 0

RUNE_POOL: RUNES d{0,1}
OMEN_POOL: OMENS d{-1,0}

output RUNE_POOL + OMEN_POOL named "regular"
output [RUNE_POOL and OMEN_POOL with BONUSES bonuses and PENALTIES penalties] named "with bonuses and penalties"

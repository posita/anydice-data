function: normaldamage {
 result: 12d8 + 61
}
function: criticaldamage {
 result: [normaldamage]+[normaldamage]+[normaldamage]
}
function: attackroll ADVANTAGECOUNT {
 result: [highest 1 of ADVANTAGECOUNTd20]
}
function: damageswitch ROLL:n {
 if ROLL = 1 { result: 0 }
 if ROLL = 2 { result: [normaldamage] }
 if ROLL = 3 { result: [normaldamage] }
 if ROLL = 4 { result: [normaldamage] }
 if ROLL = 5 { result: [normaldamage] }
 if ROLL = 6 { result: [normaldamage] }
 if ROLL = 7 { result: [normaldamage] }
 if ROLL = 8 { result: [normaldamage] }
 if ROLL = 9 { result: [normaldamage] }
 if ROLL = 10 { result: [normaldamage] }
 if ROLL = 11 { result: [normaldamage] }
 if ROLL = 12 { result: [normaldamage] }
 if ROLL = 13 { result: [normaldamage] }
 if ROLL = 14 { result: [normaldamage] }
 if ROLL = 15 { result: [normaldamage] }
 if ROLL = 16 { result: [normaldamage] }
 if ROLL = 17 { result: [normaldamage] }
 if ROLL = 18 { result: [normaldamage] }
 if ROLL = 19 { result: [criticaldamage] }
 if ROLL = 20 { result: [criticaldamage] }
}
function: getdamage ADVANTAGECOUNT {
 ROLL: [attackroll ADVANTAGECOUNT]
 result: [damageswitch ROLL]
}
function: fullroundattacks ATTACKCOUNT ADVANTAGECOUNT {
 DAMAGETOTAL: 0
 loop N over {1..ATTACKCOUNT}{
  DAMAGETOTAL: DAMAGETOTAL + [getdamage ADVANTAGECOUNT]
 }
 result: DAMAGETOTAL
}
ATTACKCOUNT: 8

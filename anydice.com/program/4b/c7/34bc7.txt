function: count success SUM:n {
  if SUM > 31 { result: 5 }
  if SUM > 15 { result: 4 }
  if SUM > 7 { result: 3 }
  if SUM > 3 { result: 2 }
  if SUM > -1 { result: 1 }
  if SUM < -3 { result: -1 }
  if SUM < -7 { result: -1 }
  result: 0
}

function: matches ATTR:d with SKILL:d vs DIFFI:n  {
  SUM: [explode ATTR] + [explode SKILL] - DIFFI
  result: [count success SUM]
}

function: R:n with damage D:n {
  if R > 0 & D >= 0 {
    result: R + D
  } else {
    if R > 0 { result: R }
    else { result: 0 }
  }
}

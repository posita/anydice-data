\******** DEFINITIONS ********\
ARCHERY_STYLE: 2
SHARPSHOOTER_ATTACK_PENALTY: 5
SHARPSHOOTER_DMG_BONUS: 10

\******** ENEMY ********\
AC: d{10..20}
CON: d{10..20}

\******** ATTACK FUNCTIONS ********\
function: sharpshooter at level LEVEL:n with DEX:n dex for DMG:d crit CRIT_DMG:d roll ROLL:n against AC:n{
    ATTACK_MODIFIER: (DEX - 10)/2
    PROFICIENCY: ((LEVEL - 1)/4) + 2
    DMG_WITH_BONUS: DMG + ATTACK_MODIFIER + SHARPSHOOTER_DMG_BONUS
	
    if ROLL = 20 { result: DMG_WITH_BONUS + CRIT_DMG }
    if ROLL + ATTACK_MODIFIER + ARCHERY_STYLE >= AC { result: DMG_WITH_BONUS }

    result: 0
}

function: beast attack plus ATTACK_MODIFIER:n to hit for DMG:d crit CRIT_DMG:d roll ROLL:n against AC:n{
    DMG_WITH_BONUS: DMG + ATTACK_MODIFIER
	
    if ROLL = 20 { result: DMG_WITH_BONUS + CRIT_DMG }
    if ROLL + ATTACK_MODIFIER >= AC { result: DMG_WITH_BONUS }

    result: 0
}

function: poison attack plus ATTACK_MODIFIER:n to hit for DMG:d crit CRIT_DMG:d roll ROLL:n against AC:n with dc POISON_SAVE_DC:n poison for POISON_DMG:d against CON:n con roll CON_ROLL:n{
    CON_MODIFIER: (CON-10)/2

    if(CON_ROLL + CON_MODIFIER < POISON_SAVE_DC) { DMG: DMG + POISON_DMG }
    result: [beast attack plus ATTACK_MODIFIER to hit for DMG crit CRIT_DMG roll ROLL against AC]
}

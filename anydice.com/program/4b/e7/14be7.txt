CRITICALHIT: 2
function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: [damage for MISS] }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: [damage for CRITICALHIT] }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: [damage for HIT] }
 result: [damage for MISS]
}

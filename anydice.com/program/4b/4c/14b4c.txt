\ one-way Check System for WGRD \


\
function: checkOne CHECK:s CHALLENGE:n {

 loop N over {1..#(CHECK)}{
  if N@CHECK > CHALLENGE{
   result = result + 1
  }
 }
 output RESULT
}

MAXDICE = 5

loop L over {4, 6, 8, 10, 12}{
 loop M over {1..MAXDICE}{
  loop N over {4,6,8,10,12}{
   loop O over {1..MAXDICE}{
    output MdN > dL named "[M]d[N] > d[L]"
   }
  }
 }
}

\ Dice Pool System for WGRD \

function: evaluate ROLL:s vs SAVE:n {
 SUCCESSES: ROLL > SAVE

 if SUCCESSES > 0 { result: SUCCESSES}
 result: 0
}

loop DICE over {4, 6, 8, 10, 12} {
 loop CHALLENGE over {4,6,8,10,12}{
  loop NUMBER over {1..3} {
   output [evaluate NUMBERdDICE vs dCHALLENGE] named "[NUMBER]d[DICE] vs d[CHALLENGE]"
  }
 }
}

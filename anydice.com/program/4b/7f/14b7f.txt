AC: {10..19}

function: attack DMG:d crit CRIT_DMG:d penalty PENALTY:n rider RIDER:n dex DEX:n roll ROLL:n{
    PROFICIENCY: ((LEVEL-1)/4) + 2    
    MODIFIER: (DEX-10)/2
    DMG_WITH_BONUS: DMG + MODIFIER + RIDER
	
    if ROLL = 1 {
        result: 0
    }
    if ROLL = 20 {
        result: DMG_WITH_BONUS + CRIT_DMG
    }
    if ROLL + MODIFIER + PROFICIENCY - PENALTY >= AC {
        result: DMG_WITH_BONUS
    }
    result: 0
}

LEVEL: 1
output 
    [attack 1d8 crit 1d8 penalty 0 rider 0 dex 19 roll 1@3d20] named "Level [LEVEL] Elven Accuracy"
output 
    [attack 1d8 crit 1d8 penalty 5 rider 10 dex 18 roll 1@2d20] named "Level [LEVEL] Sharpshooter"

LEVEL: 4
output 
  [attack 1d8 crit 1d8 penalty 0 rider 0 dex 20 roll 1@3d20] named "Level [LEVEL] Elven Accuracy & Resilient(Dex)"
output 
  [attack 1d6 crit 1d6 penalty 5 rider 10 dex 18 roll 1@2d20] +
  [attack 1d6 crit 1d6 penalty 5 rider 10 dex 18 roll 1@2d20] named "Level [LEVEL] Sharpshooter & Crossbow Expert"

LEVEL: 5
output 
  [attack 1d8 crit 1d8 penalty 0 rider 0 dex 20 roll 1@3d20] +
  [attack 1d8 crit 1d8 penalty 0 rider 0 dex 20 roll 1@3d20] named "Level [LEVEL] Elven Accuracy & Resilient(Dex)"
output 
  [attack 1d6 crit 1d6 penalty 5 rider 10 dex 18 roll 1@2d20] +
  [attack 1d6 crit 1d6 penalty 5 rider 10 dex 18 roll 1@2d20] +
  [attack 1d6 crit 1d6 penalty 5 rider 10 dex 18 roll 1@2d20] named "Level [LEVEL] Sharpshooter & Crossbow Expert"

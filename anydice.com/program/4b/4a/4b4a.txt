\ Root Attack Function \
function: attack ATT:n atbns ABONUS:n crange CRNG:n cmod CMOD:n dmg DMG:d dgbns DBONUS:n ac AC:n {
 if {ATT + ABONUS < AC & ATT != 20} | ATT = 0 { result: 0 }
 DMG_RESULT: DMG+DBONUS
 if ATT >= CRNG | ATT = 20 {
  loop P over {2..CMOD} {
   DMG_RESULT: DMG_RESULT+DMG
  }
 }
 result: DMG_RESULT
}

AC: 23
ATTACK: d20

\ Multiple Attacks Function \
function: attack ATT:d atbns ABONUS:n crange CRNG:n cmod CMOD:n dmg DMG:d dgbns DBONUS:n ac AC:n count COUNT:n {
 DMG_RESULT: 0
 loop P over {1..COUNT} {
  DMG_RESULT: DMG_RESULT+[attack ATT atbns ABONUS crange CRNG cmod CMOD dmg DMG dgbns DBONUS ac AC]
 }
 result: DMG_RESULT
}

\ Baseline level 20 Attack, dual weilding rogue \

function: rogueattwaitcrit ATT:n atbns ABONUS:n crange CRNG:n dmg DMG:d dgbns DBONUS:n cdmg CDMG:d cmod CMOD:n ac AC:n {
 DMGTOTAL: 0
 CRITUSED: 0
 if ATT >= CRNG | ATT = 20 {
  DMGTOTAL: [attack ATT atbns ABONUS crange CRNG cmod CMOD dmg DMG + CDMG dgbns DBONUS ac AC]
  CRITUSED: 1
 } else {
  DMGTOTAL: [attack ATT atbns ABONUS crange CRNG cmod CMOD dmg DMG dgbns DBONUS ac AC]
 }

 if CRITUSED = 0 {
  DMGTOTAL: DMGTOTAL + [attack ATTACK atbns ABONUS crange CRNG cmod CMOD dmg DMG + CDMG dgbns DBONUS ac AC]
 } else {
  DMGTOTAL: DMGTOTAL + [attack ATTACK atbns ABONUS crange CRNG cmod CMOD dmg DMG dgbns DBONUS ac AC]
 }
 result: DMGTOTAL
}

function: rogueattfirsthit ATT:n atbns ABONUS:n crange CRNG:n dmg DMG:d dgbns DBONUS:n cdmg CDMG:d cmod CMOD:n ac AC:n {
 DMGTOTAL: 0
 CRITUSED: 0
 if ATT + ABONUS >= AC & ATT != 0 {
  DMGTOTAL: [attack ATT atbns ABONUS crange CRNG cmod CMOD dmg DMG + CDMG dgbns DBONUS ac AC]
  CRITUSED: 1
 } else {
  DMGTOTAL: 0
 }

 if CRITUSED = 0 {
  DMGTOTAL: DMGTOTAL + [attack ATTACK atbns ABONUS crange CRNG cmod CMOD dmg DMG + CDMG dgbns DBONUS ac AC]
 } else {
  DMGTOTAL: DMGTOTAL + [attack ATTACK atbns ABONUS crange CRNG cmod CMOD dmg DMG dgbns DBONUS ac AC]
 }
 result: DMGTOTAL
}

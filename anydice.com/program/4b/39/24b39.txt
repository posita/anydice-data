LEVEL:16
MOD:2
DIE:d6

LVL:LEVEL-1
DIEMAX:[maximum of DIE]
STR:DIEMAX+MOD

function: hpdie DIE:d {
  result: DIE
}

function: maxhp X:n Y:d {
  TOTAL: 0
  loop N over {1..X} {
    HP: [hpdie Y]
    TOTAL:TOTAL+HP
  }
  result: TOTAL
}

output [maxhp LVL DIE]+STR named "[DIE]"

output 15d6+6+16*2 named "Caleb L16 wizard Con/Mod 2"

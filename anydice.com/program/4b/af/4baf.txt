function: power A:n { if A = 1 { result: A + 1 } result: A }
function: weak A:n { if A = -1 { result: A - 1 } result: A }
output 4d[power d{-1..1}]
output 1d{-1..1} + 3d[power d{-1..1}]
output 2d{-1..1} + 2d[power d{-1..1}]
output 3d{-1..1} + 1d[power d{-1..1}]
output 4d{-1..1}
output 3d{-1..1} + 1d[weak d{-1..1}]
output 2d{-1..1} + 2d[weak d{-1..1}]
output 1d{-1..1} + 3d[weak d{-1..1}]
output 4d[weak d{-1..1}]

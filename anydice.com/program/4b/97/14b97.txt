function: explode DIE:d {
 result: [explode DIE helper]
}

function: explode N:n helper {
 if N = 2{ result: N + [explode DIE helper] -1 }
 result: N
}

function: bound Q:n between LOWER:n and UPPER:n {
 if Q > UPPER{result: UPPER}
 if Q < LOWER{result: LOWER}
 result:Q
}
DEFD:d{-1,0,0,1,1,2}

function: is A:n gte B:n{
 result: A>=B
}

function: battle ATK_SUC:n DEF_SUC:n attack{
 if(ATK_SUC <= 0) {result:0}
 if(ATK_SUC >= DEF_SUC){
  BONUS:[bound ATK_SUC - DEF_SUC between 0 and GLOBAL_CAP]
  result:(BONUS)d6 + [highest of DEF_SUC and 0]
 }
 else
 {
  result:ATK_SUC - [lowest of(DEF_SUC - ATK_SUC) and ATK_SUC]
 }
}
function: battle HIT_SUC:n EVA_SUC:n ATK:n DEF:n hit{
 if(HIT_SUC >= EVA_SUC){
  BONUS:[bound HIT_SUC - EVA_SUC between 0 and GLOBAL_CAP]
  result:[battle (BONUS + ATK)dEXPD DEFdEXPD attack] 
 }
 else{
  result:0
 }
}
function: battle ATK:n HIT:n DEF:n EVA:n {
  result: [battle (HITdEXPD) (EVAdEXPD) ATK DEF hit]
}
function: evenbattle NUM:n {
  result: [battle NUM NUM NUM NUM]
}

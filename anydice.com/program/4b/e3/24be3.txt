\STUFF\
MISS: 0
HIT: 1
CRITICALHIT: 2

\Character\
CRIT_MIN: 20
STRENGTH: 3
PROFICIENCY: 5
MAGIC_WEAPON_BONUS: 1

ATTACK: STRENGTH + PROFICIENCY + MAGIC_WEAPON_BONUS

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  result: CRITICALHIT
 }
 if ROLL + ATTACK >= DEFENSE { 
  result: HIT 
 }
 result: MISS
}



loop AC over {5..30} {
 output [attack d20 vs AC] named "[AC]"
}

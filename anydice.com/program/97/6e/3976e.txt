function: X:n risky {
if X = 1 {result: -1d8}
if X = 20 {result: (4d8+15)}

}

function: X:n safe {

if X >= 14 {result: 4d8}
if X < 4 {result: 0}
if X >= 4 {result: 2d8}

}

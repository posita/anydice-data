STAT:       10
EXTRABONUS: 0

DCAMOUNT:   3

FIRSTDC:    60
SECONDDC:   90
THIRDDC:    110

function: X:n roll {
SUM: X + STAT + 10
if SUM >= THIRDDC  & DCAMOUNT >= 3 {result: 3}
if SUM >= SECONDDC & DCAMOUNT >= 2 {result: 2}
if SUM >= FIRSTDC {result: 1}
result: 0
}

ROLL: [d100 roll]

output ROLL named "DC passed"
output ROLL >= 1 named "Pass chance"

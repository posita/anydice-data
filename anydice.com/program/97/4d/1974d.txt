function: DICE:n dice FOCUS:n f EXPERTISE:n e ATTRIBUTE:n a { result: [test DICEd20 FOCUS (EXPERTISE + ATTRIBUTE)] }
function: DICE:n dice SKILL:n s ATTRIBUTE:n a { result: [test DICEd20 SKILL (SKILL + ATTRIBUTE)] }

function: og DICE:n dice CRIT:n f EXPERTISE:n e ATTRIBUTE:n a {
 PASS: ATTRIBUTE + EXPERTISE - FOCUS
 result: [og DICE CRIT PASS]
}

function: og DICE:n dice CRIT:n s PASS:n a {
 F: 20 - CRIT - PASS
 result: DICEd{2:CRIT, 1:PASS, 0:F}
}
 
function: test ROLLS:s CRIT:n PASS:n {
 V: 1@ROLLS
 loop N over {1..#ROLLS} {
  if N@ROLLS <= CRIT {V: V + 100}
  if N@ROLLS <= PASS {V: V + 100}
 }
 result: V
}

function: X:n beats Y:n {
 if X = Y {result: d{0, 1}}
 result: X > Y
}

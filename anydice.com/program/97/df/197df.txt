function: successes DICE:n threshold DC:n{
 SUCCEEDS: 0
 loop N over {1..DICE} {
  ROLL: [explode 1d10]
  SUCCEEDS: SUCCEEDS+ [helper ROLL DC]
 }
 result: SUCCEEDS
}

function: helper ROLL:n DC:n {
 if ROLL >= DC {
  result: 1 + [exceed (ROLL-DC)]
 } 
 result: 0
}

function: exceed ROLL:n {
 if ROLL >= 4 {
  result: 1 + [exceed (ROLL-4)]
 }
 result: 0
} 

\output [successes 3 threshold 7] named "Charitina"

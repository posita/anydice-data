function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

MISS: 0
HIT: 1d8+3
CRITICALHIT: 11
ATTACK: 5

output [attack d20 vs 13]
output [attack 1@2d20 vs 13]
output [attack 2@2d20 vs 13]

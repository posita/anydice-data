F: d{-1, 0, 1}


function: skillroll TOROLL:n with CUR:n and REROLL:n now ROLL:n {
if (REROLL < 1) {result: CUR + ROLL + (TOROLL)dF}
else if (REROLL > 0 & ROLL < 0) {result: [skillroll TOROLL with CUR and (REROLL-1) now dF] }
else if (TOROLL > 0) { result: [skillroll (TOROLL-1) with (CUR+ROLL) and REROLL now dF] }
else {result: CUR + ROLL}
}

function: skill N:n{
 result: [skillroll 9 with 0 and N now dF]
}

output [skill 1]

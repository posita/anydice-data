function: roll TALENT:n EFFORT:n EASE:n {
  if TALENT = 1 { X: 4 }
  if TALENT = 2 { X: 6 }
  if TALENT = 3 { X: 8 }
  if TALENT = 4 { X: 10 }
  if TALENT = 5 { X: 12 }

  if EASE < 0 {
    Y: EASE*-1 + 1
    result: [lowest 1 of YdX] + 2*EFFORT
  }
  if EASE > 0 {
    Y: EASE + 1
    result: [highest 1 of YdX] + 2*EFFORT
  }
  result: 1dX + 2*EFFORT
}

loop TALENT over {1..5} {
  loop EFFORT over {0..5} {
    loop EASE over {-2..2} {
      output [roll TALENT EFFORT EASE] named "Talent at [TALENT], effort at [EFFORT], and ease at [EASE]"
    }
  }
}

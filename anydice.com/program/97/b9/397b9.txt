set "explode depth" to 3
function: evaluate ROLL:s {
  if [count 6 in ROLL] >= 1 {
    result: ROLL + [explode d6] -1
  } else {
    result: ROLL
  }
}

\ In the example below the "8" is the number of hit points of the target with a shroud (+2hp) and +2 undying (+2hp) and the 11 is the damage from Murdering Winter in cold3 but with +5 CR \
output ([evaluate 2d6]+8+2+2) - ([evaluate 2d6]+11-6)

function: numrerolls with A:n and B:n {
 if A=0 {result: 1}
 else {if A < B { result: A }}
 result: B
}


function: empower REAL:d char CMA:n {
	NUM: #REAL
	SIZE: [sides of REAL]
	C: [numrerolls with CMA and NUM]
	result: [empower REAL char CMA helper]
}

function: sides of DIE:d {
	TOTAL: 1@[reverse {DIE}]
	 result: TOTAL/NUM
}

function: finder DEECE:s P:n {
	result: P@DEECE
}

function: empower DIES:s char CHA:n  helper {
	RES:0
	loop X over {1..C} {
		Y: (NUM - X + 1)@DIES
		if Y<((SIZE + 1)/2) {RES: RES + dSIZE}
		else {RES: RES + Y}
		}
	if (NUM>C){
		REST: NUM - C
		KEPT: {1..REST}@DIES
		RES: RES + KEPT
		}
	result: RES
}

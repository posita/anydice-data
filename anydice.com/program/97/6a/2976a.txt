STR:5
ATH:STR+2*3
ROLL:1@2d20
SPD:40
FEAT:1
RD:10-(5*FEAT)
JD:SPD-RD

function: jump J:n long L:n run RUN:n {
 JDR:ATH+J
 if L>0 {
  if RUN>0 {
   JDT:JDR+20
   if JDT>JD { result: JD }
   result: JDT
  } else {
   JDT:JDR+10
   if JDT>SPD { result: SPD }
   result: JDT
  }
 } else {
  if RUN>0 {
   JDT:JDR+8
   if JDT>JD { result: JD }
   result: JDT
  } else {
   JDT:JDR+4
   if JDT>SPD { result: SPD }
   result: JDT
  }
 }
}

output [jump ROLL long 1 run 1] named "Running Long"
output [jump ROLL long 0 run 1] named "Running High"
output [jump ROLL long 1 run 0] named "Standing Long"
output [jump ROLL long 0 run 0] named "Standing High"

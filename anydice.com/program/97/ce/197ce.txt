function: attack ROLL:n vs AC:n with NUM:n{
 if ROLL = 1 { result: 0 }
 if ROLL = 20 { result: NUM d ( 2 d DAMAGE ) }
 if ROLL + ATTACK >= AC { result: NUM d DAMAGE }
 result: 0
}

function: attackk ROLL:n vs AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL = 20 { result: 2 d DAMAGEE }
 if ROLL + ATTACK >= AC { result: DAMAGEE }
 result: 0
}

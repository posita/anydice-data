MANDIE: 1d8

function: banana for reference { result: 0 }

function: roll DIE:n { result: DIE }

function: manifest power at ORDER:n order with DIE:n die {
  CHECK: DIE
  if CHECK > ORDER { result: 0 }
  if CHECK = ORDER { result: 1 }
  if CHECK < ORDER { result: ORDER }
  result: 10000
}

output [banana for reference] named "reference"
loop O over {2..[lowest of 6 and #{MANDIE}@{MANDIE}]} {
  output [manifest power at O order with MANDIE die] named "strain gained at [O]"
}

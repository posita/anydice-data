function: lightning N:n{
  result: [count {N..6} in 6d6]
}

function: doublelightning N:n{
  result: [count {N..6} in 12d6]
}

output [lightning d6] named "Normal"
output [lightning d{1,1,2,3,4,5}] named "Battalion"
output [doublelightning d6] named "Overcharge"
output [doublelightning d{1,1,2,3,4,5}] named "Battalion Overcharge"

function: poolroll A:s against TN:n {
	COUNT: A > TN
	HIGHR: 0
	loop N over {1..#A}
	{
		NUM: #N@A
   		if NUM > HIGHR { HIGHR: NUM }
	}
	if COUNT >= 1
	{ result: HIGHR + COUNT - 1 }
	else
	{ result: HIGHR }
}

loop N over {3..5}
{
loop T over {2..4}
{
output [poolroll Nd6 against T] named "[N]d6 vs [T]"
}
}

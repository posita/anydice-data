function: exp helper D:n {
  if D = 0 { result: -1000000 }
  if D = 9 { result: 1 + [exp helper DIE] }
  result: 1
}

function: return R:n { result: R }

function: fun BASE:s TN:n DIE:d {
  BR: BASE >= TN
  EXP: 0
  if 1@BASE = BR@BASE {
      loop N over {1..BR} {
        HELPER: 0 + [exp helper DIE]
        if [return HELPER] < 0 { result: -1 }
        EXP: EXP + HELPER
      }
  }
  result: BR + EXP
}

output [fun 5d{0..9} 5 1d{0..9}]

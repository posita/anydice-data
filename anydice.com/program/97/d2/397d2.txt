C: 5
S: 5
N: 5

function: roll DICE:d TIMES:n times {
  if TIMES > 1 {
    result: (DICE + S < 10 + C) + [roll DICE (TIMES-1) times]
  }
  else {
    if TIMES = 1 {
      result: (DICE + S < 10 + C)
    }
    else {
      result: 0
    }
  }
}

output (1d20 + S < 10 + C) * 2d6 * N
output 2d6 * [roll 1d20 N times]

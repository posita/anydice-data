loop SKILL over {0..10} {

STATIC: 10

if SKILL = 0 { output STATICd4 named "Rank [SKILL]" } 

else if SKILL = 1 { output STATICd6 named "Rank [SKILL]" } 

else if SKILL = 2 { output STATICd8 named "Rank [SKILL]" } 

else if SKILL >= 3 { output (STATIC-(SKILL-3))d10+STATIC*(SKILL-3) named "Rank [SKILL]" }

}

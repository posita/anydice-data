function: hit or crit ROLL:n target N:n {
  if ROLL >= N { result: 1 }
  else { result: 0 }
}

function: crit ROLL:n {
  if ROLL = 10 { result: 1 }
  else { result: 0 }
}

function: roll hits DICE:n target N:n {
  result: [count {1} in [sequence DICE of [hit or crit d10 target N]]]
}

function: roll crits DICE:n {
  result: [count {1} in [sequence DICE of [crit d10]]]
}

output [roll hits 4 target 7] named "Hit Count"
output [roll crits 4] named "Critical Count"

function: reroll DICE:s {
 if (1@DICE = 10 & 2@DICE < 6){
   result: 10 + 1d10
 } 
 result: 1@DICE + 2@DICE
}

function: rerolld DICE:s {
 if (2@DICE = 10 & 3@DICE < 6){
   result: 10 + 1d10
 } 
 result: 2@DICE + 3@DICE
}

output [reroll 2d10]
output [reroll 3d10]
output [rerolld 3d10]
output [lowest 2 of 3d10]

function: attack summary tohit ROLL:n plus BONUS:n targetdef TARGETDEF:n targethp TARGETHP:n for DMG:d plus BONUS:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL = 20 {
        result: DMG + DMG + BONUS + BONUS
    }
    if ROLL + BONUS > TARGETDEF {
        result: DMG + BONUS
    }
    result: 0
}

output
[
attack summary
    tohit d20 plus 7 
    targetdef 8 
    targethp 16
    for 1d10 plus 7
]

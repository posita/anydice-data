ATK: 1@2d20 \ attack roll: use 1@2d20 for advantage or 2@2d20 for disadvantage \
HIT:    +13  \ bonus to attack roll \
AC:      20  \ target AC: attack hits if ATK + HIT >= AC \
CRIT:   18  \ minimum attack roll to crit \
DMG:   2d6  \ damage rolled on successful hit \
BONUS: 2d10  \ bonus damage on first hit \
LEVEL: 9 \ level the spell is cast at (+1 from tattoo) \

function: reroll INIROLL:n {
   if INIROLL + HIT < AC { result: 1d20 }
   if INIROLL <= 1 { result: 1d20 }
   else {
   result: INIROLL
   }
}
ENDATK: [reroll ATK]  \ accounts for seeking spell metamagic \

function: attack ROLL:n {
  \ return the number of times damage is applied on an attack roll of ROLL \
  if ROLL <= 1 {result: 0}   \ natural 1 is always a miss \
  if ROLL >= CRIT { result: 2 }  \ critical hit deals damage damage twice \
  result: ROLL + HIT >= AC
}
ATTACK: [attack ENDATK]  \ optimization: precalculate the miss/hit/crit probabilities \

function: multiattack N:n times {
  result: [roll ATTACK to multiattack N times]
}
function: roll FIRSTHIT:n to multiattack N:n times {
  if N <= 0 {
    result: 0
  }
  if FIRSTHIT > 0 {
    result: FIRSTHIT d (DMG + BONUS) + (N-1) d (ATTACK d DMG)
  } else {
    result: [roll ATTACK to multiattack N-1 times]
  }
}
set "maximum function depth" to 999

output [multiattack (LEVEL+1) times] named "Scorching Ray"

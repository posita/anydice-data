 if R+X >= Y+10 {if R = 1 {result: D+S} else { result: 2*D+S+T}}
 else if R+X >= Y {if R = 20 {result: 2*D+S+T} else if R = 1 {result: 0} else {result: D+S}}
 else if R+X>=Y-10 {if R = 20 {result: D+S} else {result: 0}}
 else {result: 0}
}

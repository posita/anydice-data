function: exp helper D:n {
  if D = 10 { result: -1000 }
  if D <= EXP { result: 1 + [exp helper DIE] }
  result: D <= TN
}

function: handful HAND:n DIE:d TN:n EXP:n {
  result: (HAND <= TN) + (HAND <= EXP)d[exp helper DIE]
}

loop N over {1..9} {
output [lowest of [highest of 5d[handful 1d10 1d10 N 1] and -1] and 15]
}

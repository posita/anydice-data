\ D is the size of the dice \
\ Successes are scored on everything over 1 \

D: 6

function: number higher in A:s over B:s 
{
   C: [count {1@B+1..D} in A]
   if C = 0 { C: [count 1@B in A] }
   result: C
}

loop N over {2..6} {
   output [number higher in NdD over 2dD] named "[N]d[D] higher dice v 2d[D]" 
}

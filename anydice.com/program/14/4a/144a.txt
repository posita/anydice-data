\ middle function that works for combinations of three difference dice pools \

function: middle NUMBER:n of A:s and B:s and C:s {
 SEQUENCE: [sort {A, B, C}]
 if NUMBER = #SEQUENCE { result: SEQUENCE }

 if NUMBER = 1 { result: (1 + (#SEQUENCE - 1) / 2) @ SEQUENCE }
 
 FROM: 1 + (#SEQUENCE - NUMBER) / 2
 TO: FROM + NUMBER - 1
 result: {FROM..TO}@SEQUENCE
}

output [middle 1 of 1d4 and 1d6 and 1d8]

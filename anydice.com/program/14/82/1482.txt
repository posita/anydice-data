function: sil DICE:s plus EXTRA:s
{
  SIXES: [count EXTRA in DICE]
  HIGH: 1@DICE
  if HIGH = 6
  {
    result: 5 + SIXES
  }
  else
  {
    result: HIGH
  }
}

output [sil  1d6 plus {6}]
output [sil  2d6 plus {6}]
output [sil  3d6 plus {6}]
output [sil  4d6 plus {6}]
output [sil  5d6 plus {6}]
output [sil  6d6 plus {6}]
output [sil  7d6 plus {6}]
output [sil  8d6 plus {6}]
output [sil  9d6 plus {6}]
output [sil 10d6 plus {6}]

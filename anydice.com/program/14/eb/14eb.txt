function: chance of critical failure with power dice PD:s {
 if #{PD} = 1 & PD <= 2 {result: 1}
 if #{PD} = 2 & {1,2}@PD = 2 {result: 1}
 result: 0
}

 if #{PD} = 1 & PD <= 2 {
  result: 0
 }
 if #{PD} = 2 & {1,2}@PD = 2 {
  result: 0
 }
 result: PD + WL
}

 result: {1,2}@PD = 12
}

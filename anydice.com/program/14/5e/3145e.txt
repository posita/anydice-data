function: reroll REROLL:s on ROLL:n to REPLACEMENT:n {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: redo ROLL:n to at least REPLACEMENT:n {
    if REPLACEMENT >= ROLL {
        result: REPLACEMENT
    }
    result: ROLL
}

output 2d8+4 named "Standard"
output 1d8+8+4 named "Crunchy Crits"
output 2d[reroll {1,2,3} on d8 to 4] + 4 named "Raise Dice to Maximum"
output [redo 2d8 + 4 to at least [maximum of d8] + 4] named "Raise Total to Maximum"

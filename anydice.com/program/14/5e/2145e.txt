\**** Damage calculator ****\

function: mainattackroll ROLL:n vs AC:n {
 if ROLL = 20 { result: MAINCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + DEX >= AC { result: MAINDAMAGE }
 result: 0
}

function: offhandattackroll ROLL:n vs AC:n {
 if ROLL = 20 { result: OFFHANDCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + DEX >= AC { result: OFFHANDDAMAGE }
 result: 0
}

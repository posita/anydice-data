function: middle A:n and B:n and C:n {
  if (A < B) {
    DH: B
    DL: A
  } else {
    DH: A
    DL: B
  }

  if (DH < C) {
    result: DH
  } else if (DL < C) {
    result: C
  } else {
    result: DL
  }
}

loop II over {1..5} {
  I: 1d(2 + 2 * II)
  loop JJ over {II..5} {
    J: 1d(2 + 2 * JJ)
    loop KK over {JJ..5} {
      K: 1d(2 + 2 * KK)
      output [middle I and J and K] named "middle of [I], [J], and [K]"
    }
  }
}

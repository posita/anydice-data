HOPE : d12
FEAR : d12
ADVANTAGE : d6
DISADVANTAGE : -d6

function: compare N:n and M:n {
  if N > M { X: 1 }
  if N < M { X: -1 }
  if N = M { X: 100 }
  result: (N+M)*X
}

function: compare N:n and M:n with A:n {
  if N > M { X: 1 }
  if N < M { X: -1 }
  if N = M { X: 100 }
  result: (N+M+A)*X
}

output [compare HOPE and FEAR] named "Daggerheart Standard Roll"
output [compare HOPE and FEAR with ADVANTAGE] named "Daggerheart Advantage Roll"
output [compare HOPE and FEAR with DISADVANTAGE] named "Daggerheart Disadvantage Roll"

MISS: 0
HIT: 1
CRITICALHIT: 2
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}
function: attack {
 result: [damage for [attack 1d20 vs DEFENSE]]
}

MINAC: 20
MAXAC: MINAC
DEFENSE: d{MINAC..MAXAC}
EXTRA_DMG: 2d8
BLESSED: 1d6
POWER_HIT: -5
POWER_DMG: 10
PROF: 5
STAT: 5
MAGIC: 2
ATTACK: PROF + STAT + 1
MISSDAMAGE: 0

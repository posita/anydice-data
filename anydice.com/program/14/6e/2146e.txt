function: roll A:n and B:n vs D:n with N {
 if A+B=2 { result: 1 }
 if A+B=20 { result: 7 }
 if A+B<5 { result: 2 }
 if A+B+N<D { result: 3 }
 if A+B+N=D | A+B+N=D+1 { result: 4}
 if A+B+N>=D+D { result: 6 }
 result: 5
}


loop N over {-3..3}{
output [roll 1d10 and 1d10 vs 2d10 with N] named "Vs Avg [N]"
}

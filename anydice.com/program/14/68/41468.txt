function: doubles A:n and B:n max M:n {
  if A = B {
    result: M
  } else {
    result: [highest of A and B]
  }
}
function: reroll N:n on R:n size S:n {
  if R <= N {
    result: d{1..S}
  } else {
    result: R
  }
}
function: cascade R:n size S:n {
  if S = 4 { result: d4 }
  if R = S {
    result: R + [cascade d{1..S-2} size S-2]
  } else {
    result: R
  }
}
function: cascadeplus R:n size S:n {
  if S = 4 { result: d4 }
  if R >= S-1 {
    result: R + [cascade d{1..S-2} size S-2]
  } else {
    result: R
  }
}
function: cascadeplusplus R:n size S:n {
  if S = 4 { result: d4 }
  if R >= S-1 {
    result: R + [cascadeplusplus d{1..S-2} size S-2]
  } else {
    result: R
  }
}

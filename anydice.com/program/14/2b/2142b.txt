 if A+B+C=3 { result: 1 }
 if A+B+C=18 { result: 6 }
 if A+B=12 | A+C=12 | B+C=12{ result: 5 }
 if A+B+C<D { result: 2 }
 if A+B+C=D | A+B+C=D+1 { result: 3}
 result: 4
}

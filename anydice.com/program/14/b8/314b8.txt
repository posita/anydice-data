HOPE : d12
FEAR : d12

function: compare N:n and M:n {
  if N > M { X: 1 }
  if N < M { X: -1 }
  if N = M { X: 100 }
  result: (N+M)*X
}

output [compare HOPE and FEAR] named "Daggerheart Standard Roll"
output [compare HOPE and FEAR]+1d6 named "Daggerheart Advantage Roll"
output [compare HOPE and FEAR]-1d6 named "Daggerheart Disadvantage Roll"

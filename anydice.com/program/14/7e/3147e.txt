function: check XROLL:n with X:n against YROLL:n with Y:n using COMPARISON:n {
  XSUCCESS: XROLL <= X
  YSUCCESS: YROLL <= Y

  if (XSUCCESS > YSUCCESS) {
    result: -100
  }
  
  if (XSUCCESS < YSUCCESS) {
    result: 100
  }

  if (XSUCCESS = 1) {
    HIGHEST: 1

    if (COMPARISON = HIGHEST) {
      XWINNER: XROLL > YROLL
      YWINNER: XROLL < YROLL
    } else {
      XWINNER: XROLL < YROLL
      YWINNER: XROLL > YROLL
   }
    
    result: -1 * XWINNER+ YWINNER
  }

  result: 0
}

function: compare COMPARISON:n of XROLL:n X:n and YROLL:n Y:n {
  result: [check XROLL with X against YROLL with Y using COMPARISON]
}

function: reroll DIE:d {
 result: [reroll DIE helper]
}

function: reroll N:n helper {
 if N = {1,2} { result: DIE }
 result: N
}

function: attack ROLL:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL >= 10 { result: CRITICALHIT }
  result: HIT
 }
 if ROLL >= 10 { result: HIT }
 result: MISS
}

MISS: 0
HIT: 2d6
CRITICALHIT: 4d6

output [attack d20]

MISS: 0
HIT: 2d[reroll d6]
CRITICALHIT: 4d[reroll d6]

output [attack d20]

MISS: 0
HIT: 1d12
CRITICALHIT: 2d12

output [attack d20]

MISS: 0
HIT: 1d12
CRITICALHIT: 3d12

output [attack d20]

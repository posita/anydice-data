\ try setting this anywhere from 0 to 10 \
POWERATTACKTARGET: 10

\ example usage: [1d20 plus 5 vs 20 x 3 for 1d8+5] \
function: ROLL:n plus ATTACK:n vs AC:n x MUL:n for DAMAGE:d max BAB:n {
  if ROLL = 1 {
    result: 0
  }
  if ROLL + ATTACK > AC {
    EXTRA: [lowest of ROLL + ATTACK - AC and BAB]
    EXTRADAMAGE: EXTRA * MUL / 2
    result: DAMAGE + EXTRADAMAGE
  }
  if ROLL + ATTACK = AC | ROLL = 20 {
    result: DAMAGE
  }
  result: 0
}

\ iterating refers to number of iterative attacks, e.g. +20/+15/+10/+5 is 4 \
function: ATTACK:n vs AC:n x MUL:n for DAMAGE:d max BAB:n iterating N:n {
  if N = 0 { result: 0 }
  result: [d20 plus ATTACK vs AC x MUL for DAMAGE max BAB] + [ATTACK vs AC x MUL for DAMAGE max BAB iterating N - 1]
}

\ fully expands possibilities, unlike N*2, N*3 \

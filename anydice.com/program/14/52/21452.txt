MISS: 0
HIT: 1
CRITICALHIT: 2
HITDAMAGE: 1d6 + 6
SNEAKDAMAGE: 6d6 + 6
SNEAKCRITDAMAGE: 12d6 + 6d8 + 6
CRITDAMAGE: 2d6 + 6d8 + 6
MISSDAMAGE: 0
ATKMOD:10


function:  ROLL:n vs DEFENSE:n ATTACK:n {
 if ROLL = 1 {result: MISS}
 if ROLL = 20 
 {
   result: CRITICALHIT
 }
 if ROLL + ATTACK>= DEFENSE 
 {
   result: HIT 
 }
 result: MISS
}

\ Damage functions \

function: damage for ATTACKRESULT:n TOTAL:n{
 if TOTAL = 0
 {
   if ATTACKRESULT = CRITICALHIT {
     result: SNEAKCRITDAMAGE
	 }
   if ATTACKRESULT = HIT {
     result: SNEAKDAMAGE
	 }
   result: MISSDAMAGE
 }
 else
 {
	 if ATTACKRESULT = CRITICALHIT {
		 result: CRITDAMAGE
	  }
	  if ATTACKRESULT = HIT {
		 result: HITDAMAGE
	  }
	 result: MISSDAMAGE
 }
}

function: attack {
  TOTAL: 0
  loop N over {1..2} {
    TOTAL: TOTAL + [damage for [d20 vs AC ATKMOD] TOTAL]
  }
  result: TOTAL
}

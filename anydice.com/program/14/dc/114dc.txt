SHOOTING: 1d12
DAMAGE: 2d6

function: wild A {
 result: [highest of [explode 1d6] and [explode A]]
}

function: attack T:n with X:s dealing D:d{
 if X >= T+4 {
  result: D + [explode 1d6]
 } else {
  if X >= T {
   result: D
  } else {
   result: 0
  }
 }
}

output [attack 4 with [wild SHOOTING]+3 dealing [explode DAMAGE]+1] named "shoot"
output [attack 8 with [wild SHOOTING]+3 dealing [explode DAMAGE] + 5] named "headshot"

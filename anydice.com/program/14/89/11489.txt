function: tohit {
  ROLL: [highest 1 of 2 d20]
  result: [highest of ROLL >= CRITON and ROLL + ATKBONUS >= AC]
}

function: tocrit{
  result: [highest 1 of 2 d20] >= CRITON
}

function: hits {
  TOTAL: 0
  loop N over {1..ATKS} {
    TOTAL: TOTAL + [tohit]
  }
  result: TOTAL
}

function: crits {
  TOTAL: 0
  loop N over {1..ATKS} {
    TOTAL: TOTAL + [tocrit]
  }
  result: TOTAL
}

AC: 20
ATKS: 5
ATKBONUS: 10
CRITON: 18

output [hits] named "[ATKS] attacks; +[ATKBONUS] to hit vs AC [AC]: # hits"
output [crits] named "[ATKS] attacks; +[ATKBONUS] to hit: # crits (crit on [CRITON])"

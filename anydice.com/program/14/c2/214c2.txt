function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
      if ROLL = 1           { result: 0 }
      if ROLL >= CRIT_RANGE { result: DMG + CRIT_DMG }
      if ROLL + BONUS >= AC { result: DMG }
      result: 0
}

function: max A:n B:n {
  if A > B { result: A }
  result: B
}

function: advantaged DIE:d {
  result: [max DIE DIE]
}

function: sharp ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
      result: [attack ROLL plus BONUS -5 vs AC for DMG +10 crit CRIT_DMG on CRIT_RANGE]
}

function: reroll A:n B:n T:n DIE:d {
  if A < T { result: [max DIE B] }
  if B < T { result: [max A DIE] }
  result: [max A B]
}

function: elven DIE:d {
  result: [reroll DIE DIE (AC-H) DIE]
}

WD: 1d6

function: tohit {
  result: [highest 1 of 2 d20] >= AC
}

function: tocrit{
  result: [highest 1 of 2 d20] >= CRITON
}

function: hits {
  TOTAL: 0
  loop N over {1..ATKS} {
    TOTAL: TOTAL + [tohit]
  }
  result: TOTAL
}

function: crits {
  TOTAL: 0
  loop N over {1..ATKS} {
    TOTAL: TOTAL + [tocrit]
  }
  result: TOTAL
}

AC: 12
ATKS: 5
CRITON: 18

output [hits] named "[ATKS] attacks, # hits on AC [AC]"
output [crits] named "[ATKS] attacks, # crits"

ISSTRONGHIT: 0
ISWEAKHIT: 0
ISMISS: 0
loop ACTIONDIE over {1..6} {
    loop SUCCESSONE over {1..10} {
        loop SUCCESSTWO over {1..10} {
            if ACTIONDIE > SUCCESSONE {
                if ACTIONDIE > SUCCESSTWO {
                    ISSTRONGHIT: ISSTRONGHIT + 1
                }
                else {
                    ISWEAKHIT: ISWEAKHIT + 1
                }
            }
            else {
                if ACTIONDIE > SUCCESSTWO {
                    ISWEAKHIT: ISWEAKHIT + 1
                }
                else {
                    ISMISS: ISMISS + 1
                }
            }
        }
    }
}
ISALL: (ISMISS + ISWEAKHIT + ISSTRONGHIT)
output ISMISS named "ironsworn miss ([ISMISS] / [ISALL])"
output ISWEAKHIT named "ironsworn weakhit ([ISWEAKHIT] / [ISALL])"
output ISSTRONGHIT named "ironsworn stronghit ([ISSTRONGHIT] / [ISALL])"

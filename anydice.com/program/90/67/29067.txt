PROFICIENCY: 2
DEX_MOD: 4
WIS_MOD: 3

ATTACK_MOD: DEX_MOD + PROFICIENCY

function: adv {
  result: [highest of d20 and d20]
}

function: strike AC {
  result: ((d20 + ATTACK_MOD >= AC) * (d4 + DEX_MOD))
}

function: flankstrike AC {
  result: (([adv]+ ATTACK_MOD >= AC) * (d4 + DEX_MOD))
}

function: attack AC {
  result: [strike AC] + [strike AC]
}

function: flankattack AC {
  result: [flankstrike AC] + [flankstrike AC]
}

function: flurryofblows AC {
  result: [strike AC] + [strike AC] + [strike AC]
}

function: flankflurry AC {
  result: [flankstrike AC] + [flankstrike AC] + [flankstrike AC]
}

output [attack 13] named "Normal Attack"
output [flankattack 13] named "Flank Attack"
output [flurryofblows 13] named "Flurry of Blows"
output [flankflurry 13] named "Flank Flurry of Blows"

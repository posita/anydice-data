function: aw A:n {
   if A >= 10 { result: 2 }
   if A >= 7 { result: 1 }
   result: 0
}

function: k R:s L:n H:n{
  if R <= L { result: 2 }
  if R <= H { result: 1 }
  result: 0
}


loop N over {-1..3} {
 output [aw 2d6+N] named "AW + [N]"
}

output [k 1d100 8 41]
output [k 1d100 17 58]
output [k 1d100 28 72]
output [k 1d100 42 83]
output [k 1d100 58 92]

 output [k 1d100 N] named "hl [N]"

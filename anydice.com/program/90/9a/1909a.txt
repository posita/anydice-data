function: roll A:n B:n C:n {
    result: 2@[sort {A, B, C}]
}

function: rolla A:n B:n C:n {
    result: A *100 + B*10 + C
}

function: rollb A:n B:n C:n {
    result: A + B + C
}

function: rollc A:n B:n C:n {
    AUX: [sort {A, B, C}]
    result: 1@AUX *100 + 2@AUX *10 + 3@AUX
}

PD: {4,6,8,10,12,20}

loop DA over {1..6}{
  loop DB over {DA..6}{
    loop DC over {DB..6}{
      A: DA@PD
      B: DB@PD
      C: DC@PD
      output [roll 1dA 1dB 1dC] named "d[A] d[B] d[C]"
    }
  }
}

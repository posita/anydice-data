function: attack D:n {
if D = {11..19} {result: DAMAGE}
if D = {1..10} {result: 0}
if D = 20 {result: CRIT}
}

function: explode DIE:d {
 result: [explode DIE helper]
}

function: explode N:n helper {
 if N = {1,2} { result: DIE }
 result: N
}

DAMAGE: 1d[explode d12]+4
CRIT: 3d[explode d12]+4

output [attack d20]

\The types of rolls in Dexstrint\

function: count HIGH:s and LOW:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#(HIGH + LOW)} {
  COUNT: COUNT + (P@HIGH = SEQUENCE)
  COUNT: COUNT - (P@LOW  = SEQUENCE)
 }
 result: COUNT
}

output 3d6 named "Conflict Resolution"
output [middle 1 of 3d6] named "Initiative"
output d6/2 - d6/2 named "Quality"
output [count {4, 5, 6} in 3d6] named "Default Pool"
output [count {4, 5, 6} and {1} in 3d6] named "Default Pool"

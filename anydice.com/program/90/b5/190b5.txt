function: ROLL and DMG vs DODGE and ARMOR {
    if ROLL < DODGE {
        result: -1
    }
    if ROLL > ARMOR {
        result: DMG + 1
    }
    if ARMOR - ROLL < DMG {
        result: DMG - (ARMOR - ROLL)
    }
    else {
        result: 0
    }
}

function: blades ROLL:s {
 if 1@ROLL = 6 {
  result: 2 + (2@ROLL = 6)
 }
 result: 1@ROLL >= 4
}

output [blades 2@2d6] named "bitd 0d"
loop P over {1..7} {
 output [blades Pd6] named "bitd [P]d"
}

function: pbta ROLL:s {
 if ROLL >= 10 {
  result: 2
 }
 if ROLL >= 7 & ROLL <= 9 {
   result: 1
 }
 if ROLL <= 6 {
   result: 0
 }
}

loop P over {0..3} {
 output [pbta 2d6+P] named "pbta +[P]"
}

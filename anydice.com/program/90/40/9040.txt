function: risus FIRST:d vs SECOND:d {
  result: [compare FIRST and SECOND]
}

function: compare SUMA:n and SUMB:n {
  \ highest sum wins \
  if SUMA > SUMB { result: 1 }
  if SUMA < SUMB { result: -1 }
  result: 0
}

loop I over {1..6}{
  loop J over {I..6}{
    output [risus Id6 vs Jd6] named "[I]d6 vs [J]d6"
  }
}

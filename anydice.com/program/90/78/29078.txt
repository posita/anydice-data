\ Character information \
LEVEL:20
CON:20
HIT_DIE:d12
HAS_TOUGH:1
CLASS_HP_MOD:0      \ Number of levels that provide a +1 to hp like via sorcerers \
RACE_HP_MOD:LEVEL   \ Number of levels that provide a +1 to hp like via mountain dwarfs \


\ Some short calculations \
CON_MOD:(CON-10)/2
TOUGH:2*LEVEL


\ Tells the output if tough should be added \
function: add tough{
if HAS_TOUGH=1{result: TOUGH}
else {result: 0}
}


\ Perform the actual HP calculation \
output LEVELd{HIT_DIE} + (LEVEL * CON_MOD) + [add tough] + CLASS_HP_MOD + RACE_HP_MOD

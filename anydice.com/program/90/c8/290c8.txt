        result: 6
    }
    if LEVEL >= 13 {

    }
    if LEVEL >= 9 {
        result: 4
    }
    if LEVEL >= 5 {
        result: 3
    }

        result: 2
    }
}

    }

    }
}

function: int {

        result: 4
    }
    if LEVEL >= 11 {
        result: 5
    }
}

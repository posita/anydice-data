function: reroll ROLL:n less than THRESHOLD:n depth DEPTH:n{
  if ROLL < THRESHOLD & DEPTH > 1 {
    DEPTH: DEPTH - 1
    result: [reroll 1d6 less than THRESHOLD depth DEPTH]
  }
  result: ROLL
}

function: calculate DICE:n {
  VALUE: 0
  loop N over {1..DICE}{
    VALUE: VALUE + [reroll 1d6 less than 3 depth 3]
  }
  result: VALUE
}

output [calculate 4]

function: countsplode DIE:d at THRESH lim CAP {
 MAX: [maximum of DIE]
 TICK: 0 
 result: [lowest of [countsplode DIE ticker] and L]
}
function: countsplode N:n ticker{
 TICK: TICK+1
 ROLLRES: N
 if TICK = CAP-1
 {
  if ROLLRES>=THRESH {result: 1}
  else {result: 0}
 }
 else if ROLLRES = MAX
 {
  if ROLLRES>=THRESH {result: 1 + [countsplode N ticker]}
  else  {result: 0 + [countsplode N ticker]}
 }
 else if ROLLRES >= THRESH {result: 1}
 else {result: 0}
}

\Attribute Dice\
AA: d12
AB: d12
AC: d20
AD: d20
AE: d20
\Skill TN\
TN: 4
\Expertise Rank\
EXP: 1

function: roll A:n B:n C:n D:n \E:n\ {
SUM: 0
if A <= TN {SUM: SUM + 1 + (A <= EXP)}
if B <= TN {SUM: SUM + 1 + (B <= EXP)}
if C <= TN {SUM: SUM + 1 + (C <= EXP)}
if D <= TN {SUM: SUM + 1 + (D <= EXP)}
\if E <= TN {SUM: SUM + 1 + (E <= EXP)}\
result: SUM
}
output [roll AA AB AC AD \AE\]

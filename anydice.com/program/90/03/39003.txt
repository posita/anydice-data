\ reroll 1's if BRUTAL is 1. Thanks to Vilarinof of stackexchange! \
function: reroll DICE:n as REROLL:d if BRUTAL:n {
  if BRUTAL=1 {
    if DICE=1 {
      result: REROLL
    }
    result: DICE
  }
  result: DICE
}

\ REROLL whole dice set if HIGHEST is 1, thanks to Vilarinof of stackexchange! \
function: highest of two DICE:d if HIGHEST {
  if HIGHEST=1 {
    result: [highest of DICE and DICE]
  }
  result: DICE
}

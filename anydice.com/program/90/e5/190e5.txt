function: ROLL and DMG vs DODGE and ARMOR {
    if ROLL < DODGE {
        result: 0
    }
    if ROLL > ARMOR {
        result: DMG * 100
    }
    if ARMOR - ROLL < DMG {
        result: DMG - (ARMOR - ROLL)
    }
    else {
        result: 0
    }
}

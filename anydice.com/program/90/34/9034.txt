function: sum unique in A:s {
ROLL: {A}
R: 0
loop N over {1..12} {
if (ROLL = N) = 1 {
R: R + N
}
}
result: R
}

output [sum unique in 2d6] + 1d4
output [sum unique in 2d6] + 1d6
output [sum unique in 2d6] + 1d8
output [sum unique in 2d6] + 1d10
output [sum unique in 2d6] + 1d12
output [sum unique in 2d6] + 1d20

function: calculate with attribute ATT:s skill SKILL:s and expertise EXPERTISE:n {
  VALUE: 0
  loop N over {1..#SKILL}{
    ROLL: N@SKILL
    if ROLL <= ATT { VALUE: VALUE + 1 }
    if ROLL <= EXPERTISE { VALUE: VALUE + 1 }
  }
  result: VALUE
}

output [calculate with attribute 5 skill 2d12 and expertise 2] + [calculate with attribute 5 skill 3d20 and expertise 2]

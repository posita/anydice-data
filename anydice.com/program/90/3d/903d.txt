function: risus FIRST:d vs SECOND:d {
  result: [compare 1@FIRST and 1@SECOND then FIRST and SECOND then #FIRST and #SECOND]
}

function: compare HIA:n and HIB:n then SUMA:n and SUMB:n then NA:n and NB:n {
  \ highest die wins\
  \ if equal, highest sum wins \
  \ if equal, Goliath rule \
  \ if equal, tie \
  if HIA > HIB { result: 1 }
  if HIA < HIB { result: -1 }
  if SUMA > SUMB { result: 1 }
  if SUMA < SUMB { result: -1 }
  if NA < NB { result: 1 }
  if NA > NB { result: -1 }
  result: 0
}

loop I over {1..6}{
  loop J over {I..6}{
    output [risus Id6 vs Jd6] named "[I]d6 vs [J]d6"
  }
}

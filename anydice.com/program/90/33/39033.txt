function: mm L:n MULT_LIGHTNING:n MULT_RADIANT:n MULT_THUNDER:n{
 N: L+3
 A: (Nd4+N)    \base damage, FORCE\
 B: Nd4        \shriek, THUNDER\
 C: 2*N + 2*N  \glow ring for base and shriek, RADIANT\ 
 D: 2*(N-2)    \lightning charge +1 damage for every source except the first two, LIGHTNING\ 
 E: 1d8+1 + 1  \lightning stacks cause d8+1 and another +1 after shriek, LIGHTNING\
 F: 1d4      \shriek d4 from the lightning, THUNDER\
 G: 2 + 2      \glow ring for the lightning d8 and shriek d4 RADIANT\

 \apply multipliers\
 B: MULT_THUNDER*B
 C: MULT_RADIANT*C
 D: MULT_LIGHTNING*D
 E: MULT_LIGHTNING*E
 F: MULT_THUNDER*F
 G: MULT_RADIANT*G

 result: A+B+C+D+E+F+G
}

function: jokerdramatictask RANK {
  TOKENS: 0
  loop N over {1..3} {
    TOKENS: TOKENS + [highest of [explode RANK] and [explode d6]] / 4
  }
  result: TOKENS > 4
}

function: extradramatictask RANK {
  TOKENS: 0
  loop N over {1..3} {
    TOKENS: TOKENS + [explode RANK] / 4
  }
  result: TOKENS > 4
}

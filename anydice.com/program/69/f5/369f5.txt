function: has pair of A:n and B:n in ROLL:s {
  COUNT_A: 0
  COUNT_B: 0
  loop I over ROLL {
    if I = A { COUNT_A: COUNT_A + 1 }
    if I = B { COUNT_B: COUNT_B + 1 }
  }
  result: (COUNT_A >= 1) & (COUNT_B >= 1)
}

function: add probability A to B{
 result: A + B
}

Y:[has pair of 2 and 5 in 5d6]+[has pair of 3 and 4 in 5d6]

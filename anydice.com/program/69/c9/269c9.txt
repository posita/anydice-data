function: roll ROLL:s {
 if 1@ROLL < 1 & (#ROLL)@ROLL = -1 {
  result: -1
 }
 result: [highest of 0 and ROLL]
}

function: roll ROLL:n with willpower {
 result: [highest of 0 and ROLL + 1]
}

function: specialty N:n {
 if N = 10 { result: 2 }
 if N = 1 { result: -1 }
 result: N >= DIFFICULTY
}

function: compspec N:n {
 if N = 10 { result: 2 }
 if N = 1 { result: -1 }
 result: N >= COMPARE
}

DIFFICULTY: 6
COMPARE: 8

SPECIALTY: [specialty d10]
COMPSPEC: [compspec d10]

loop X over {1..14} {output [roll X d SPECIALTY] named "[X]d"}
loop X over {5} {output [roll X d SPECIALTY] named "[X]d Base"}

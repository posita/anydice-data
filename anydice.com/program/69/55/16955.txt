  if SUM >= TARGET{result:SUM}
  SINGLETARGET:4
  if TARGET >= 15 {SINGLETARGET:5}
  if TARGET = 18 {SINGLETARGET:6}
  LOWEST: 1@ROLL
  MIDDLE: 2@ROLL
  HIGHEST: 3@ROLL
  if LOWEST < SINGLETARGET{
    LOWEST: [LOWEST reroll d6 if lower than SINGLETARGET]
    MIDDLE: [MIDDLE reroll d6 if lower than SINGLETARGET]
    HIGHEST: [HIGHEST reroll d6 if lower than SINGLETARGET]
  }
  else{
    LOWEST: d6
  }
  result: LOWEST + MIDDLE + HIGHEST
}

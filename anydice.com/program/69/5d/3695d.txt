function: modified ROLL:n {
  if ROLL <= 45 { result: ROLL + 5 }
  else { result: ROLL }
  \Adds "Add 5 if below X" rule\
  result: ROLL
}

function: round up N:n to nearest multiple of MULTIPLE:n {
  result: ((N + MULTIPLE - 1) / MULTIPLE) * MULTIPLE
}

function: modified success SKILL:n MODIFIER:n vs DC:n {
  ROLL: [modified 1d100]
  ROLLED: [round up (ROLL + MODIFIER + SKILL) to nearest multiple of 5]
  result: ROLLED >= DC
}

SKILL: 5  \ Set the skill level here; change this value to see different outcomes\

loop DC over {10, 15, 20, 25, 30, 35, 40, 45, 50, 55, 60, 65, 70, 75} {
  output [modified success SKILL 0 vs DC] named "Skill [SKILL] vs DC [DC]"
}

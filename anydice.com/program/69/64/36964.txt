\ Dramatic Task for Joker trying to get REACH tokens in ROUNDS rounds with RANK Skill \
function: jokerdramatictask RANK REACH ROUNDS {
  TOKENS: 0
  loop N over {1..ROUNDS} {
    TOKENS: TOKENS + ([highest of [explode RANK] and [explode d6]] / 4)
  }
  result: (TOKENS >= REACH)
}

\ Dramatic Task for Extra trying to get REACH tokens in ROUNDS rounds with RANK Skill \
function: extradramatictask RANK REACH ROUNDS {
  TOKENS: 0
  loop N over {1..ROUNDS} {
    TOKENS: TOKENS + ([explode RANK] / 4)
  }
  result: (TOKENS >= REACH)
}

loop D over {4, 6, 8, 10, 12} {
  output [jokerdramatictask dD 4 3] named "d[D] Joker's Dramatic Task, Challenging (4 tokens in 3 rounds)"
}

loop D over {4, 6, 8, 10, 12} {
  output [extradramatictask dD 4 3] named "d[D] Extra's Dramatic Task, Challenging (4 tokens in 3 rounds)"
}

loop D over {4, 6, 8, 10, 12} {

}

  output [extradramatictask dD 6 4] named "d[D] Extra's Dramatic Task, Difficult (6 tokens in 4 rounds)"
}

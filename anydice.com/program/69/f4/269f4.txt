function: matchtwo POOL:s {
  if 1@POOL = 3@POOL { result: 1 }
  if 1@POOL = 2@POOL { result: 1 }
  if 2@POOL = 3@POOL { result: 1 }
  result: 0
}

function: reroll lowest of ROLL:s as REROLL:d if not match {
  LOWEST: (#ROLL)@ROLL       \ the lowest die is sorted last \
  HIGHEST: 1@ROLL
  if LOWEST = HIGHEST {
    result: 1
  } else {
    result: HIGHEST = REROLL
  }
}

output [matchtwo 3d20] named "3d20"
output [reroll lowest of 2d20 as d20 if not match] named "2d20 reroll if not match"

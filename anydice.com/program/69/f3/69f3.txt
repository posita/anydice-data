ZAC: 16 + 1 \ Zombie AC, chain + 1 from warding bond \
ZSV: d20 + 3 + 1 + d4 \ Zombie Con saving throw for undead fortitude, +3 basic con bonus, +1 from warding bond, +d4 from bless \
ATK: d20
BONUS: 11 \ Attacker's attack bonus, let's say +5 attack bonus \
DMG: 2d6 + 6 \ Attacker's damage, let's say 3d6 \
RESISTEDDMG: DMG / 2 \ Warding bond grants resistance to all \
DMGSV: 5 + (RESISTEDDMG) \ Target value to save against for undead fortitude \

function: ROLL:n plus B:n to hit AC:n {
 if ROLL = 20 {
  result: 100
 }

 if ROLL + BONUS >= AC {
  result: [compare DMGSV and ZSV]
 }

 result: 0
}

function: compare V:n and ZCONSV:n {
 if ZCONSV >= V {
  result: 0
 }

 result: RESISTEDDMG
}

output [ATK plus BONUS to hit ZAC]

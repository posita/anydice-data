function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: bonusattack ROLL:n vs DEFENCE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + BONUSATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + BONUSATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

  result: BONUSCRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: BONUSHITDAMAGE
 }
 result: BONUSMISSDAMAGE
}

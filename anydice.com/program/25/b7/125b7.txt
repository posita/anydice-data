\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: attackrollbon ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGEBON } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGEBON }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
DMGBONUS: 2
PROFICIENCY: 4
ACCBONUS: 2
CRITROLL: 20
ATTACKTIMES: 3
DAMAGE: 1d12 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 3d12 + ATTACKMOD + DMGBONUS
DAMAGEBON: 1d12 + 2d8 + ATTACKMOD + DMGBONUS
CRITDAMAGEBON: 3d12 + 4d8 + ATTACKMOD + DMGBONUS
output (ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC]) named "Derek normal rage"
output (ATTACKTIMESd[attackrollbon [highest 1 of 2d20] vs ENEMYAC]) named "Derek with Holy Weapon"

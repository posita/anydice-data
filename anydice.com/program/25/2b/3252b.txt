\**** Fighting modifiable Calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}


\looped against enemy AC from 15-20\
ENEMYAC: 22



\FS Archery\
ATTACKMOD: 5
DMGBONUS: 0
PROFICIENCY: 6
ACCBONUS: 2
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll 1d20 vs ENEMYAC] named "FS Archery"

\FS Thrown\
ATTACKMOD: 5

PROFICIENCY: 6
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll 1d20 vs ENEMYAC] named "FS Thrown"

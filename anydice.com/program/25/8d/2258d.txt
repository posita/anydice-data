function: coloredhit DICE:n COLOREDMANA:n {
 result: [count {5, 6} in DICE d6] >= COLOREDMANA
}

function: generichit DICE:n GENERICMANA:n {
 result: DICE d6 >= GENERICMANA
}

function: mtgroll DICE:n COLOREDMANA:n GENERICMANA:n {
 result: (([coloredhit DICE COLOREDMANA] > 0) * ([generichit (DICE-COLOREDMANA) GENERICMANA] > 0)) > 0
}

\Define Dice\
ABILITY: d{1,1,2,1,0,0,0,0}
PROFICIENCY: d{1,1,2,2,1,1,1,1,0,0,0,0}

DIFFICULTY: d{-1,-2,-1,0,0,0,0,0}
CHALLENGE: d{-1,-1,-2,-2,-1,-1,-1,0,0,0,0,0}

loop P over {0..5}
{
    loop A over {0..5}
    {
        loop C over {0..5}
        {
            loop D over {0..5}
            {
                 if (A = 0 & P = 0) | (D = 0 &  C = 0)
{ }
else{
if ((A + P) >5) | ((D + C) >5)
{}
else{
output (AdABILITY + PdPROFICIENCY + DdDIFFICULTY + CdCHALLENGE) named "[A],[P],[D],[C]"
}}
            }
         }
     }
}

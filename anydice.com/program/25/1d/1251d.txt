\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
DMGBONUS: 0
PROFICIENCY: 6
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 9
DAMAGE: 4d8 + 3d12 + 1d4 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 8d8 + 6d12 + 2d4 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] + 1d10  named "Battle Master Glyph-Buffed"

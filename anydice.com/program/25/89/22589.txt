function: manahits DICE:n COLORED:n {
 result: [count {5, 6} in DICE d6] >= COLORED
}

function: genvalue DICE:n GENERIC:n {
 result: DICE d6 >= GENERIC
}

function: mtgroll DICE:n COLORED:n GENERIC:n {
 result: (([manahits DICE COLORED] > 0) * ([genvalue (DICE-COLORED) GENERIC] > 0)) > 0
}

output [count {5, 6} in 3d6] named "baseline hits in 3d6"

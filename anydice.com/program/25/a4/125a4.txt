TOHIT: 15
PROF: 2

function: X:n attacks DMG:d {
  TOTAL: 0
  loop N over {1..X} {
    TOTAL: TOTAL + DMG
  }
  result: TOTAL
}

function: ROLL:n damage DAMAGEDIE:n BONUSDIE:n DMGMOD:n ATKMOD:n ABILITY:n {
  if ROLL = 20 {result: DAMAGEDIE + DAMAGEDIE + BONUSDIE + BONUSDIE + DMGMOD + ABILITY}
  else if ROLL + ABILITY + ATKMOD + PROF < TOHIT {result: 0}
  result: DAMAGEDIE + DMGMOD + BONUSDIE + ABILITY
}

function: X:n attackdie {
  result: [highest 1 of Xd20]
}

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
DMGBONUS: 0
PROFICIENCY: 6
ACCBONUS: 0
CRITROLL: 20
DAMAGE: 1d10 + ATTACKMOD + DMGBONUS + 2d8
CRITDAMAGE: 2d10 + ATTACKMOD + DMGBONUS + 4d8
ATTACKS: 4

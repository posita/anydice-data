PARRY: 4
MOD: -2

function: roll A {
 WILD: 1d6
 result: [highest of [explode A] and [explode WILD]]
}

function: fight F vs P {
 result: [roll F]>=P
}

function: shoot S with M {
 result: ([roll S]+M)>=4
}

loop N over {4,6,8,10,12} {
 output [fight 1dN vs PARRY] named "d[N] fight vs 4 parry"
}

loop N over {4,6,8,10,12} {
 output [shoot 1dN with MOD] named "d[N] shoot vs Dodge"
}

loop N over {4,6,8,10,12} {
 output [roll 1dN] named "d[N]"
}

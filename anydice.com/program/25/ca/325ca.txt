function: count ROLL:s successes at S:n {
  SUCCESSES: 0
  loop X over ROLL {
    if X = 6 {
      SUCCESSES: SUCCESSES + 2
    }
    else {
      if X >= S {
        SUCCESSES: SUCCESSES + 1
      }
    }
  }
  result: SUCCESSES
}

loop N over {1..10} {
  output [count Nd6 successes at 4] named "[N]d6"
}

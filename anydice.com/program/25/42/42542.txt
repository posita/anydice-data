function: A:n e B:n{
 C:A+B
 if C>=3{result:3}
 if C<=-1{result:-1}
 result:C
}
function: sucessos A:s{
 result:(1@A>=4)+(1@A=6)+(2@A=6)
}
function: riscos A:s{
 result:[sucessos A]-2*[count {7,8} in A]
}

loop Y over {0..5}{
 loop X over {Y..5}{
  output [[sucessos Xd6] e [riscos Yd8]] named "[X]d6 e [Y]d8"
 }
}

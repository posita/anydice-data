AC: 15
\Target's AC\
ATTACKBONUS: 5
\Your attack bonus\
DAMAGEBONUS: 2
\From STR, magic weapons, whetstones, etc.\
NORMALDAMAGE: 1d8 + DAMAGEBONUS
\Your weapon's damage after all modifiers\
PRONGEDDAMAGE: 1d6 + DAMAGEBONUS
\Damage of a pronged arrow\
PRONGEDTOHIT: ATTACKBONUS + 1
ATTACKROLL: 1d20 + ATTACKBONUS
\Let's see if we hit\
CRITICALCONFIRMATION: 1d20 + ATTACKBONUS

function: hit ac ARMORCLASS:n with attbonus BONUS:n rolling TOTALROLL:n confirming CONFIRMATION:n {
if TOTALROLL = (BONUS + 20) {
\Critical threat scored\
if CONFIRMATION = TOTALROLL {
\Another natural 20\
result: 3
\we have just scored a crit, deal x3 damage\
}
else {
\We did not roll a natural 20 for crith confirmation\
if CONFIRMATION >= ARMORCLASS {
result: 3
\Confirmed, but not with a natural 20\
}
else {
\Critical hit not confirmed\
result: 1
}
}
}
else {
\Not a critical threat\
if TOTALROLL >= ARMORCLASS {
result: 1
}
else {
result: 0
}
}
}


loop AC over {5..27} {
NORMALDAMAGEMULTIPLIER: [hit ac AC with attbonus ATTACKBONUS rolling ATTACKROLL confirming CRITICALCONFIRMATION]
\output DAMAGEMULTIPLIER named "Average damage multiplier, AC [AC]"\

NORMALDAMAGEDEALT: NORMALDAMAGEMULTIPLIER d NORMALDAMAGE

\output NORMALDAMAGEDEALT named "Damage dealt, AC [AC]"\

\Let's check the pronged arrow\


DAMAGEMULTIPLIER: [hit ac AC with attbonus PRONGEDTOHIT rolling ATTACKROLL confirming CRITICALCONFIRMATION]
\output DAMAGEMULTIPLIER named "Average damage multiplier, pronged arrow, AC [AC]"\
DAMAGEDEALT: DAMAGEMULTIPLIER d PRONGEDDAMAGE
\output DAMAGEDEALT named "Damage dealt, pronged arrow, AC [AC]"\
PRONGEDARROWDAMAGE: DAMAGEDEALT

DAMAGEDIFFERENCE: NORMALDAMAGEDEALT - PRONGEDARROWDAMAGE
output DAMAGEDIFFERENCE named "Damage difference, AC [AC]"
\output 0 named "___________"\
}

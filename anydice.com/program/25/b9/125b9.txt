EFFORT: 2d12
loop N over {4,6,8,10,12} {
    loop M over {4,6,8,10,12} {
        EFFORT: 1dM+1dN
        DIFFA: 1d20
        DIFFB: 1d20
        output (EFFORT > DIFFA) & (EFFORT > DIFFB) named "strong success with d[M] + d[N]"
        output (EFFORT > DIFFA) & (EFFORT <= DIFFB) named "weak success with d[M] + d[N]"
        output (EFFORT <= DIFFA) & (EFFORT > DIFFB) named "weak success with d[M] + d[N]"
        output (EFFORT <= DIFFA) & (EFFORT <= DIFFB) named "fail with d[M] + d[N]"
    }
}

MIN: 1
MAX: 6
DICE: 3d6

\output [count MIN in DICE] named "transgenic" \
\output [count MAX in DICE] named "error"\

loop N over {3,4,5} {
  output ([count 1 in Nd6] > 0) & ([count 6 in Nd6] > 0) named "[N] players, transgenic and error"
  output ([count 1 in Nd6] > 0) | ([count 6 in Nd6] > 0) named "[N] players, transgenic or error"
  output ([count 1 in Nd6] > 0) & ([count 6 in Nd6] = 0) named "[N] players, only one"
}

function: warp DICE:s with REROLL:d to AMOUNT:n {
 TARGET: 6
 SCORE: AMOUNT
 OUT: [count {TARGET..6} in DICE]
 loop N over {0..SCORE}{if N > 0 {
  if (SCORE > 0) {
   OUT: OUT + (REROLL >= TARGET)
   SCORE: SCORE - 1
  }
 }}
 result: OUT
}

loop N over {0..5}{
output [warp 5d6 with 1d6 to N] named "[N]R"
}

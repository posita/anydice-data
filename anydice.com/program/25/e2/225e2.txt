function: attackroll highest NUMBER:n of ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 else if ROLL = 1 { result: 0 }
 else if ROLL + PROFICIENCY + ATTACKMOD >= AC { result: DAMAGE }
 result: 0
}

ATTACKMOD: 3
PROFICIENCY: 3
CRITROLL: 20
ATTACKTIMES: 2
WEAPONBONUS: 0
SCIMITAR: 1d6
RAPIER: 1d8
DUELLING: 2

loop ENEMYAC over {12..18}{
DAMAGE: RAPIER + ATTACKMOD + WEAPONBONUS + DUELLING
CRITDAMAGE: RAPIER + RAPIER + ATTACKMOD + WEAPONBONUS + DUELLING
ATTACKTIMES: 2
output ATTACKTIMESd[attackroll highest 1 of 1d20 vs ENEMYAC] named "Rapier Duelling vs AC [ENEMYAC]"
}

loop ENEMYAC over {12..18}{
DAMAGE: SCIMITAR + ATTACKMOD + WEAPONBONUS
CRITDAMAGE: SCIMITAR + SCIMITAR + ATTACKMOD + WEAPONBONUS
ATTACKTIMES: 3
output ATTACKTIMESd[attackroll highest 1 of 1d20 vs ENEMYAC] named "Scimitar 2wf vs AC [ENEMYAC]"
}

E_A : 1d10
E_B : 2d10
E_C : 3d10
P_A : 1d8
P_B : 1d6

function: best of DICE:s {
 MAX: 0
 loop N over DICE{
  if N > MAX { MAX: N }
 }
 result: MAX
}


output [highest 1 of E_A] named "1x Effort"
output [highest 1 of E_B] named "2x Effort"
output [highest 1 of E_C] named "3x Effort"
output [highest of [highest 1 of E_A] and [highest 1 of P_A]] named "1x Effort + 1x Power"
output [highest of [highest 1 of E_B] and [highest 1 of P_A]] named "2x Effort + 1x Power"
output [highest of [highest 1 of E_C] and [highest 1 of P_A]] named "3x Effort + 1x Power"
output [highest of [highest of [highest 1 of E_A] and [highest 1 of P_A]] and [highest 1 of P_B]] named "1x Effort + 2x Power"

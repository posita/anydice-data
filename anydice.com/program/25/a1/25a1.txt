DIE: 1d10-1           \ Rolling 00 is a fumble --> automatic failure. \
SKILLVALUE: 50        \ Change this to anything you like. At 100, 00 is still a failure. \

function: roll ONE:n TWO:n {
  if (ONE*10+TWO) > SKILLVALUE {
    if (TWO*10+ONE) > SKILLVALUE { result: 0 }
    result: TWO*10+ONE
  }

  if (TWO*10+ONE) > SKILLVALUE { result: ONE*10+TWO }

  if ONE > TWO { result: ONE*10+TWO }

  result: TWO*10+ONE  \ If ONE and TWO is below SKILLVALUE, and TWO is bigger than ONE. \
}

output [roll DIE DIE] named "1d100 against skill value [SKILLVALUE]"

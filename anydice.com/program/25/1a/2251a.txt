function: mech ROLL:s BONUS:n {
 \if ROLL = 10 { result: 1 + [mech d10 BONUS]}\
 result: (ROLL >= (10-BONUS))
}
function: multi ROLL:s BONUS:n {
  TOTAL: 0
  loop N over ROLL {
    SUCC: (N >= (10-BONUS))
    if N = 10 {SUCC: SUCC + (d10 >= (10-BONUS))}
    TOTAL: TOTAL + SUCC
  }
  result: TOTAL
}

output [mech 4d10 3]
output [mech 5d10 5]

output [multi 4d10 3]
output [multi 5d10 5]

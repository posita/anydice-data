function: multiply ROLL:s {
  COUNT: ROLL = 6
  \ Sum the dice and subtract the 6s so they count as 0 \
  TOTAL: ROLL - (COUNT * 6)
  
  if COUNT = 0 { result: TOTAL }
  
  \ Exponential: x2, x4, x8... \
  MULT: 1
  loop X over {1..COUNT} {
    MULT: MULT * 2
  }
  
  result: TOTAL * MULT
}

output [multiply 8d{0, 0, 1, 1, 2, 6}] named "6s Double the Rest"

function: successes in ROLL:s with POINTS:n points {
  SUCCESSES: 0
  REMAINING: POINTS
  \ Loop from lowest to highest roll: \
  loop X over [reverse ROLL] {
    if X <= REMAINING {
      SUCCESSES: SUCCESSES + 1
      REMAINING: REMAINING - X
    }
  }
  result: SUCCESSES
}

\ Relabel the sides of the die with the number of points needed to turn them into a success: \
MIN_SUCCESS: 5
D: [highest of 0 and MIN_SUCCESS - d6]

\ N = number of dice, P = number of points: \
N: 6
loop P over {0..5} {
  output [successes in NdD with P points] named "[N]d6 with [P] points"
}

function: roll A {
 WILD: 1d6
 result: [highest of [explode A] and [explode WILD]]
}

function: fight F vs P {
 result: [roll F]>=P
}

function: shoot S vs MOD {
 result: [roll S]>=(4+MOD)
}

loop N over {4,6,8,10,12} {
 output [roll 1dN] named "d[N]"
 output [fight 1dN vs 4] named "d[N] fight vs 4 parry"
 output [shoot 1dN vs -2] named "d[N] shoot vs Dodge"
}

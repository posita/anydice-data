DAC:0
DC:20
DIS:[lowest 1 of 2d20]
BONUS:5
function: impose disadvantage A:n {
 if (A-DC) >= DAC { result: (A-DC) }
 result: A*DAC
}
function: digher A:n {
 if (A-(DC+6)) >= DAC { result: (A-(DC+6)) }
 result: A*DAC
}

output [impose disadvantage DIS + BONUS] named "Disadvantage"
output [digher d20 + BONUS] named "Higher DC"

set "maximum function depth" to 10

function: explode N:n size S:n {
  if (S < 4) {
    result: 0
  }
  if (N>5) {
    result: 1 + [explode d(S-2) size S-2]
  }
  if (N > 3) {
    result: 1
  }
  result: 0
}

loop N over {4, 6, 8, 10, 12}{
  output [explode dN size N] named "3d[N] exploded"
}

output 3d[explode d10 size 10] named "3d10 exploded"

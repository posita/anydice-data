function: shoot WOUND:n SAVE:n {
  BS: 3
  result: (1d6>=BS)&(1d6>=WOUND)&(1d6<SAVE)
}

function: multiattack SHOTS:n shots to wound on WOUND:n save on SAVE:n max damage DMG:n{
  TOTAL: 0
  loop N over {1..SHOTS} {
    DAMAGE: DMG * [shoot WOUND SAVE]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}
output [multiattack 15 shots to wound on 3 save on 5 max damage 1] named "15 shots with autobolt at geq"

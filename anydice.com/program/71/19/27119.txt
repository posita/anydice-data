DEFENSE:4
SAVE:2
HULL:3
ARMOR:1

SAVES:0
CRITSAVES:0
loop N over {1..DEFENSE}{
RESULT:1d6
SAVES:SAVES+[count {SAVE..5} in RESULT]
CRITSAVES:CRITSAVES+[count 6 in RESULT]
}
TOTALSAVES:SAVES+CRITSAVES

output TOTALSAVES named "Total Saves"
output SAVES named "Regular Saves"
output CRITSAVES named "Critical Saves"
output HULLd6+(ARMOR*HULL) named "Effective Health"

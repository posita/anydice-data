function: countl VALUES:s in SEQUENCE:s limit by L{
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 if COUNT > L {result: L}
 result: COUNT
}

function: erfolge SEQUENCE:s {
 result: [count{5,6} in SEQUENCE]
}

function: erfolge SEQUENCE:s with LIMIT{
 result: [countl{5,6} in SEQUENCE limit by LIMIT]
}

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 13-20\
ENEMYAC: d{13..20}

ATTACKMOD: 6
DMGBONUS: 4d6
PROFICIENCY: 4
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d8 + ATTACKMOD + 4d6
CRITDAMAGE: 2d8 + ATTACKMOD + 8d6
MAINHAND: ATTACKTIMESd [attackroll [highest 1 of 3d20] vs ENEMYAC]

ATTACKMOD: 6
DMGBONUS: 0
PROFICIENCY: 4
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d8 + ATTACKMOD
CRITDAMAGE: 2d8 + ATTACKMOD
OFFHAND:  ATTACKTIMESd [attackroll d20 vs ENEMYAC]
 
output MAINHAND + OFFHAND
named "Ray + minus Advantage + Elven accuracy"

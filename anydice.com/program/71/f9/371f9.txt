function: roll ROLL:s  {
 result: (ROLL > 5) + ((ROLL = 10)/2)*2
}
function: desp ROLL:s  {
 result: (ROLL > 5) + ((ROLL > 8)/2)*2
}
function: despb ROLL:s  {
 result: (ROLL < 3)*2 + (ROLL > 5) + ((ROLL = 10)/2)*2
}
function: despc ROLL:s  {
 result: (ROLL = 1) + ((ROLL = 1)/2)*2 + (ROLL > 5) + ((ROLL = 10)/2)*2
}
function: despd ROLL:s  {
 result: (ROLL < 3) + ((ROLL < 3)/2)*2 + (ROLL > 5) + ((ROLL = 10)/2)*2
}

output [desp 10d10]
output [roll 10d10]
output [roll 5d10] + [despd 5d10]

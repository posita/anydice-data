function: roll plus MODIFIER and SKILLED{
 if SKILLED= 1 {
  result: [explode d12] + d8 + MODIFIER
 } else {
  result: [explode d12] + d4 + MODIFIER
 }
}

function: roll plus MODIFIER and SKILLED to hit TARGET {
 result: [roll plus MODIFIER and SKILLED] - TARGET
}

MODIFIER: 6
TARGET: 18
SKILLED: 1 \yes or no\

output [roll plus MODIFIER and SKILLED] named "Distribution"
output [roll plus MODIFIER and SKILLED] > TARGET named "Chance to Hit"
output [roll plus MODIFIER and SKILLED to hit TARGET] named "Missed by"

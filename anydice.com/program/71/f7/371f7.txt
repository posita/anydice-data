SIXPOOL: 1d6
EIGHTPOOL: 1d8

LOWESTSIX: (#SIXPOOL)@SIXPOOL
RESTOFSIX: {1..#SIXPOOL-1}@SIXPOOL

LOWESTEIGHT: (#EIGHTPOOL)@EIGHTPOOL
RESTOFEIGHT: {1..#EIGHTPOOL-1}@EIGHTPOOL

function: choose between FIRST_DICE:s with FIRST_SIDES:n sides and SECOND_DICE:s with SECOND_SIDES:n sides {
  FIRST_S: 
  FIRST_AVG: (FIRST_SIDES + 1) - (FIRST_DICE * 2)
  SECOND_AVG: (SECOND_SIDES + 1) - (SECOND_DICE * 2)

  if FIRST_AVG > SECOND_AVG {
    result: FIRST_SIDES
  } else {
    result: SECOND_SIDES 
  }
}

output [choose between LOWESTSIX with 6 sides and LOWESTEIGHT with 8 sides]

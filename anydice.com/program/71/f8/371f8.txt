function: vamp NCOUNT:n dice with hunger HCOUNT:n {
NORMAL: {2:5, 9:4, 10}
HUNGER: {1:2, 2:3, 9:4, 10}
result: [vamp (NCOUNT-HCOUNT) d NORMAL HCOUNT d HUNGER]
}

function: vamp NORMAL:s HUNGER:s {
ANKH: ([count 10 in NORMAL]) + ([count 10 in HUNGER])
SUCCESS: ([count 9 in NORMAL]) + ([count 9 in HUNGER])
SKULL: ([count 1 in HUNGER])
if (ANKH >= 2) { ANKH: (ANKH + ANKH) }
if (SKULL >= 2) { SKULL: (SKULL + SKULL) }
result: SUCCESS + ANKH + SKULL
}

output [vamp 10 dice with hunger 0]
output [vamp 10 dice with hunger 5]

function: shoot BS:n WOUND:n SAVE:n {
  result: (1d6>=BS)&(1d6>=WOUND)&(1d6<SAVE)
}

function: multiattack SHOTS:n BS:n shots to wound on WOUND:n save on SAVE:n max damage DMG:n{
  TOTAL: 0
  loop N over {1..SHOTS} {
    DAMAGE: DMG * [shoot BS WOUND SAVE]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

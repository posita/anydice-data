function: fifth edition advantage attack ROLL:n {
if ROLL = 1  {result: 0}
if ROLL >= CRIT {result: 2*DMG_DICE + DMG_MOD}
if ROLL+TO_HIT >= AC {result: DMG_DICE + DMG_MOD}
result: 0}

DMG_DICE: 1d10
DMG_MOD: 15
TO_HIT: 6
CRIT: 20

loop AC over {15..20} {
 output [fifth edition advantage attack [highest 1 of 3d20]] named "[AC] AC"
}

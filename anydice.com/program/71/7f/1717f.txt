function: explode DIE:d {
result: [explode DIE helper]
}

function: explode N:n helper {
 if N < 3 { result: DIE  }
 result: N
}
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

ATTACK: 3 + 2
HIT: 2d[explode d6]+3
CRITICALHIT: 4d[explode d6]+3
MISS: 0
loop AC over {10, 15, 20} {
 output [attack d20 vs AC] named "[AC]"
}

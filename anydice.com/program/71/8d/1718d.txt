function: pbta DICE:s mod MOD:n {
 RES: {DICE + MOD}
 if RES >= 10 {
  result: 2
 }
 if RES = 7 | RES = 8 | RES = 9 {
  result: 1
 }
 if RES <= 6 {
  result: 0
 }
}

function: blades ROLL:s {
 if 1@ROLL = 6 {
  result: 2 + (2@ROLL = 6)
 }
 result: 1@ROLL >= 4
}

output [pbta 2d6 mod +3]
output [blades 4d6]

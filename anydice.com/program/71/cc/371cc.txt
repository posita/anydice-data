function: MEDIAN_THREE_D_TWENTY:n with D_TWENTY:n crits {
  X: D_TWENTY
  if X=1 | X=20 {
    result: X
  }
  else {
    Y: MEDIAN_THREE_D_TWENTY
    if Y=1 {
      result: 2
    }
    else {
      if Y=20 {
        result: 19
      }
      else {
        result: Y
      }
    }
  }
}

function: ROLL:n between LOWER and UPPER {
  result: (LOWER < ROLL) & (ROLL < UPPER)
}

ROLL_OLD: d20
ROLL_NEW: [2@(3d20) with d20 crits]
output [ROLL_OLD between 5 and 16]
output [ROLL_NEW between 5 and 16]
\
output (5 < ROLL_OLD) & (ROLL_OLD < 16)
output (5 < ROLL_NEW) & (ROLL_NEW < 16)
\

\D -> Number of Dice\
\GMIN -> Minimum number of Glitches before there is an issue\

loop D over {1..15} {
GMIN: D/2+1

function: ROLL:s glitch
{
 result: ((ROLL >= 5)>0) & ((ROLL = 1)>=GMIN)
}

function: ROLL:s critical glitch
{
 result: ((ROLL >= 5)=0) & ((ROLL = 1)>=GMIN)
}

output [D d6 glitch] named "Probability of glitch with [D] dice"
output [D d6 critical glitch] named "Probability of critical glitch with [D] dice"
}

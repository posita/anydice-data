\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
DMGBONUS: 4
PROFICIENCY: 4
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 3
DAMAGE: 1d10 + 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d10 + 2d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll 1d20 vs ENEMYAC] named "Hexblade EB Damage"

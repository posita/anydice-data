TARGET_AC: 11
DAMAGE: 10
MAP: -5

function: attack DIE:n with MAP:n against AC:n for DAMAGE:n{
  if (DIE + MAP >= (AC+10)) { result: 2*DAMAGE }
  if (DIE + MAP >= AC) {
    if (DIE = 20){ result: 2*DAMAGE }
    result: DAMAGE
  }
  if (DIE = 20){ result: DAMAGE }
  result:0
}

output [attack d20 with 0 against TARGET_AC for DAMAGE] + [attack d20 with MAP against TARGET_AC for DAMAGE] named "two attacks"
output [attack 1@2d20 with 0 against TARGET_AC for DAMAGE] named "true strike"
output [attack 1@2d20 with 0 against TARGET_AC for DAMAGE+1d6] named "true strike w/Bespell"

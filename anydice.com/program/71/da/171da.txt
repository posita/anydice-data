AC : 16

function: S:n hide and hit {
  if (PP - 5) > S {
    R: [highest 1 of (1 + LUCK)d20]
  }
  else {
    R: [highest 1 of (3 + LUCK)d20]
  }
  result : R
}

function: S:n get damage is hidden P:n {
  if(P >= PP) {
    if(S >= 19) {
      R: 10d6 + 2d8 + 5
    } else if (S = 1) {
      R: 0
    } else if (S + 11 >= AC) {
      R: 5d6 + 1d8 + 5
    } else {
      R: 0
    }
  } else {
    if(S >= 19) {
      R: 2d8 + 5
    } else if (S = 1) {
      R: 0
    } else if (S + 11 >= AC) {
      R: 1d8 + 5
    } else {
      R: 0
    }
  }
  result: R
}

output 1d[[R hide and hit] get damage is hidden R]

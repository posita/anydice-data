TARGET: 15
ATTACKMOD: 4 + 3
DAMAGEMOD: 4
DAMAGEDIE: 2d6
ATTACKS: 2






MISS: 0
HIT: 1
CRITICALHIT: 2
 
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITICALHIT }
 if ROLL + ATTACKMOD >= DEFENSE { result: HIT }
 result: MISS
}

function: damage ROLL:n vs DEFENSE:n{
 X: [attack ROLL vs DEFENSE]
 if X = 0 {result: 0}
 if X = 1 {result: DAMAGEDIE + DAMAGEMOD}
 if X = 2 {result: DAMAGEDIE + DAMAGEDIE + DAMAGEMOD}
}

output [attack d20 vs TARGET] named "to Hit"

RESULT: 0

loop N over {1..ATTACKS} {
 RESULT: RESULT + [damage d20 vs TARGET] 
}

output RESULT named "damage with hit chance considered"

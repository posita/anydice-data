I_BREAKS: 0
I_OPENINGS: 0
I_HOPES: 0
RED: {1000, 100, 1, 101, 101, 1002}
BLACK: {1100, 101, 101, 1002, 102, 1202}
WHITE: {1101, 1201, 301, 1102, 302, 1203}
DOES: 1
DOESNT: 0

function: condense FACE:n usetokens T:n{
  BREAKS: I_BREAKS
  HOPES: I_HOPES
  if FACE >= 1000 {
    DOTS: FACE / 1000
    REMAININGDOTS: FACE / 1000
    if HOPES > 0 & T {
      TOKENSUSED: [lowest of HOPES and REMAININGDOTS]
      HOPES: HOPES - TOKENSUSED
      FACE: FACE + TOKENSUSED
    }
    FACE: FACE - (1000 * DOTS)
  }

  if FACE >= 100 {
    POTENTIALS: FACE / 100
    REMAININGPOTENTIALS: POTENTIALS
    if HOPES > 0 & T {
      TOKENSUSED: [lowest of HOPES and REMAININGPOTENTIALS]
      HOPES: HOPES - TOKENSUSED
      FACE: FACE + TOKENSUSED
      REMAININGPOTENTIALS: REMAININGPOTENTIALS - TOKENSUSED
    }
    if BREAKS > 0 & T {
      TOKENSUSED: [lowest of BREAKS and REMAININGPOTENTIALS]
      BREAKS: BREAKS - TOKENSUSED
      FACE: FACE + TOKENSUSED
    }
    FACE: FACE - (100 * POTENTIALS)
  }
  result: FACE
}

function: rolls A:n plus B:n power P:d bonus Q:s tohit H:n usetokens T:n{
  RESULT: 1d0
  ROLL: d10 + B >= H
  if T {
    ROLL: d10 + B + I_OPENINGS >= H
  }
  loop X over {1..A}{
    if X=1 {
      RESULT: RESULT + (ROLL * (P+Q))
    }
    else {
      RESULT: RESULT + (ROLL * P)
    }
  }

  RESULT: [condense RESULT usetokens T]
  
  result: RESULT

}

function: rolls A:n dice plus B:n to hit using P:d power dice per hit with a bonus Q:s dice against to hit H:n that T:n use tokens {
  result: [rolls A plus B power P bonus Q tohit H usetokens T]
}

function: W:d wounds B:s {
  result: W >= 1dB
}

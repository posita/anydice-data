function: shoot WOUND:n AP:n{
  BS: 3
  result: (1d6>=BS)&(1d6>=WOUND)&((1d6+AP)<5)
}

function: multiattack SHOTS:n shots to wound on WOUND:n ap AP:n{
  TOTAL: 0
  loop N over {1..SHOTS} {
    DAMAGE: [shoot WOUND AP]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}
output [multiattack 3 shots to wound on 3 ap 0] named "3 shots with autobolt on geq"
output [multiattack 2 shots to wound on 3 ap -1] named "2 shots with boltrifle on geq"

function: sum X:s less than L with at least K successes {
  R: 0
  S: 0
  loop I over X {
     if I <= L { R: R + I }
     if I > L { S: S + 1 }
  }
  if S >= K { result: R/4 }
  if S < K { result: 0 }

}

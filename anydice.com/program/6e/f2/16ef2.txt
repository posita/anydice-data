function: burn DICE:s and DOCE:s {
 ONE: [count {4..12} in DICE]
 TWO: [count {4..12} in DOCE]
 if ONE + TWO = 2 { result: 2 }
 if ONE + TWO = 1 { result: 1 }
 result: 0
}

loop N over {4,6,8,10,12}{
output [burn 1dN and 1d4] named "d[N]/d4"
}

loop N over {6,8,10,12}{
output [burn 1dN and 1d6] named "d[N]/d6"
}

loop N over {8,10,12}{
output [burn 1dN and 1d8] named "d[N]/d8"
}

loop N over {10,12}{
output [burn 1dN and 1d10] named "d[N]/d10"
}

output [burn 1d12 and 1d12] named "d12/d12"

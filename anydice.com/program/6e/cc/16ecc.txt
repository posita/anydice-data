function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}
function: ROLL:d reroll BAD:s {
  loop I over {1..20} {
    ROLL: [ROLL reroll BAD as ROLL]
  }
  result: ROLL
}

X: [d6 reroll {1,2} as d6]
output [highest 3 of 4dX] named "reroll once"

Y: [d6 reroll {1,2}]
output [highest 3 of 4dY] named "unlimited rerolls"

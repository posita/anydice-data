set "maximum function depth" to 3

BASEDIE: d6

function: target TARGET:n reroll REROLL:n VALUE:n {
  if VALUE = REROLL {
    result: 1 + [target TARGET reroll REROLL BASEDIE]
  }
  if VALUE  >= TARGET {
    result: 1
  }
  result: 0
}

DIE: [target 4 reroll 6 BASEDIE]

output 2dDIE named "2D"
output 4dDIE named "4D"
output 6dDIE named "6D"
output 8dDIE named "8D"
output 10dDIE named "10D"

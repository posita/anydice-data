ENEMYAC: d{16}

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: attackrollb ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGEB } 
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGEB }
 result: 0
}

function: attackrollc ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGEC }
 result: 0
}

ENEMYAC: d{16}

ATTACKMOD: 3
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 50
ATTACKTIMESB: 100
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS
DAMAGEB: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGEB: 2d6 + ATTACKMOD + DMGBONUS
DAMAGEC: 1d6 + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Battleaxe +3 proficiency +3 Attack modifier"
output ATTACKTIMESBd[attackrollb d20 vs ENEMYAC] named "Handaxes x2 +3 proficiency +3 Attack modifier"
output ATTACKTIMESBd[attackroll d20 vs ENEMYAC] named "Battleaxes x2 +3 proficiency +3 Attack modifier"
output ATTACKTIMESd[attackrollb d20 vs ENEMYAC]+ATTACKTIMESd[attackrollc d20 vs ENEMYAC] named "Battleaxes x2 No Dual-Wielding +3 Attack modifier"

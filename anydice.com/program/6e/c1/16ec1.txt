AT: 1d20
ATT: 1d20
AC: 15
CRIT: AT=20
CRITT: ATT=20
BHIT: (AT+5>AC)|(ATT+5>AC)
FHIT: (AT+5>AC)|((AT<3)*(ATT+5>AC))
BCRIT: (AT=20|ATT=20)
FCRIT: (AT>18|ATT>18)

FCDMG: (4d6+3)*FCRIT*FHIT
BCDMG: (4d6+3+2+1d6)*BCRIT*BHIT
FDMG: (2d6+3)*FHIT
BDMG: (2d6+3+2+1d6)*BHIT


output FCDMG
output FDMG
output BCDMG
output BDMG

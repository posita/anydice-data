loop DEFENSE over {7..25} {


function: greatweapon DIE:d {
 result: [greatweapon DIE helper]
}

function: greatweapon N:n helper {
 if N = {1,2} { result: DIE }
 result: N
}



MISS: 0
HIT: 2d[greatweapon 1d6]+3
CRITICALHIT: 4d[greatweapon 1d6]+3
ATTACK: 5



function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

output [attack d20 vs DEFENSE]
}

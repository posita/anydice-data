

function: mexplode SK N:n {
 if N = 1 { result: 1 + [mexplode SK 1d6] }
 else if N <= SK {result: 1}
 result: 0
}

function: roll stat ST for skill SK {
  TOTAL:0
  loop N over  {1..ST} {
     TOTAL: TOTAL + [mexplode SK 1d6]
  }
  result: TOTAL
}

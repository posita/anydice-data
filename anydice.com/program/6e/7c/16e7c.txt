\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: attackrollb ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGEB } 
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGEB }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{19}

ATTACKMOD: 3
DMGBONUS: 3
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
ATTACKTIMESB: 2
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS
DAMAGEB: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGEB: 2d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Battleaxe +3 proficiency +3 Attack modifier"
output ATTACKTIMESBd[attackrollb d20 vs ENEMYAC] named "Handaxes +3 proficiency +3 Attack modifier"

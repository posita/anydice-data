\**** Damage calculator ****\

\global vars\
ATTACKMOD: 5
DICEDMGBONUS: 0
DMGBONUS: 0
BASEWPNDMG: 1d8
ENEMYAC: d{15..20}

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

SNEAKATTACKDICE: 10d6
PROFICIENCY: 6
ACCBONUS: 0
CRITROLL: 20
DAMAGE: BASEWPNDMG + SNEAKATTACKDICE + ATTACKMOD + DMGBONUS + DICEDMGBONUS
CRITDAMAGE: ((BASEWPNDMG + SNEAKATTACKDICE + DICEDMGBONUS) * 2) + ATTACKMOD + DMGBONUS
output [attackroll [highest 1 of 3d20] vs ENEMYAC] named "Rogue"




function: DIEROLL:n dmg SAUSED:n CRNTDMG:n CRNTATTK:n ATKTIMES:n{


if DIEROLL >= CRITROLL {
if SAUSED = 0 {
 CRNTDMG: CRNTDMG + SACRITDAMAGE
 SAUSED: 1
}
else {
 CRNTDMG: CRNTDMG + CRITDAMAGE
}
}

if DIEROLL > 1 {
if DIEROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC {
if CRNTATTK = ATKTIMES {
if SAUSED = 0 {
CRNTDMG: CRNTDMG + SADAMAGE
} else {
CRNTDMG: CRNTDMG + DAMAGE
}
} else {
CRNTDMG: CRNTDMG + DAMAGE
}
}
}


result: {SAUSED, CRNTDMG}
}

function: differentattackroll vs AC:n atknum ATKTIMES:n  {
SAUSED: 0
CRNTDMG: 0

loop CRNTATTK over {1..ATKTIMES}{

DMGFUNCSEQ: [[highest 1 of 3d20] dmg SAUSED CRNTDMG CRNTATTK ATKTIMES]


SAUSED: 1@DMGFUNCSEQ
CRITDAMAGE: 2@DMGFUNCSEQ


}

result: CRNTDMG




}

ATACKSINTURN: 2
SNEAKATTACKDICE: 7d6
PROFICIENCY: 6
ACCBONUS: 2
CRITROLL: 19
DAMAGE: BASEWPNDMG + ATTACKMOD + DMGBONUS + DICEDMGBONUS
CRITDAMAGE: ((BASEWPNDMG + DICEDMGBONUS) * 2) + ATTACKMOD + DMGBONUS
SADAMAGE: BASEWPNDMG + ATTACKMOD + DMGBONUS + DICEDMGBONUS + SNEAKATTACKDICE
SACRITDAMAGE: ((BASEWPNDMG + SNEAKATTACKDICE + DICEDMGBONUS) * 2) + ATTACKMOD + DMGBONUS

output [differentattackroll vs ENEMYAC atknum ATACKSINTURN] named "Champ fighter + Rogue"

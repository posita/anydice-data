function: does D:n hit AC:n {
  result: D >= AC
}

function: MANY:n attacks with MOD:n doing DAMAGE:d hits AC:n killing HP:n {
  result: ([does 1d20+MOD hit AC]*(DAMAGE)*MANY) >= HP
}

function: modifier M:n saves against DC:n {
  result: [does 1d20+M hit DC]
}

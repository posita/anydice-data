function: explode N:n on SIZE:n {

 if N = SIZE & SIZE < 12 {

  result: N + [explode d[emphasis d(SIZE + 2) size SIZE + 2] on SIZE + 2]
 }            
 result: N
}

function: explodenorm N:n on SIZE:n {

 if N = SIZE & SIZE < 12 {

  result: N + [explode d(SIZE + 2) on SIZE + 2]
 }            
 result: N
}

function: emphasis N:n size S:n {
 if N <= 2 {
   result: dS
 }
 result: N
}

output [explodenorm d4 on 4] named "Exp"

output [explode d[emphasis d4 size 4] on 4] named "Exp GWS"

output [explodenorm d4 on 4] + [explodenorm d4 on 4] named "Exp 2 dice"

output [explode d[emphasis d4 size 4] on 4] + [explode d[emphasis d4 size 4] on 4] named "Exp 2 dice GWS"

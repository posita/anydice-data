function: eval longer straight VALUES:s {
 CURRENTVAL: 0
 MAXCOUNT: 0
 COUNT: 0
 loop P over {1..#VALUES} {
  VAL: P@VALUES
  NEXTVAL: CURRENTVAL - 1
  if VAL = NEXTVAL {
   COUNT: COUNT + 1
    if (COUNT > MAXCOUNT) {
     MAXCOUNT: COUNT
    }
  }
  else {
   if VAL != CURRENTVAL{
    COUNT: 1
    if (COUNT > MAXCOUNT) {
     MAXCOUNT: COUNT
    }
   }
  }
  CURRENTVAL: VAL
 }
 result: MAXCOUNT
}

loop N over {1..6}{
output [eval longer straight Nd6] named "[N]D longer straight"
}

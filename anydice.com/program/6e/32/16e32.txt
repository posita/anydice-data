loop N over {1..3}{
 ROLL: Nd6
 SUCCESSES: [count {4..6} in ROLL]
 FAILURES: [count {1..3} in ROLL]
 TRAMA: [count {1} in ROLL]
 EDGE: [count {6} in ROLL]
 output SUCCESSES named "[N]d6 Successes"
 output FAILURES named "[N]d6 Failures"
 output TRAMA named "[N]d6 Trama"
 output EDGE named "[N]d6 Edge"
}

\ count success (4-6) fail (1-3) trama (1) edge (6) \

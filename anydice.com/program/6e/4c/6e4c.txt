MOD: 9
AC: 15
THREAT: 19

function: threat ROLL:n {
 if ROLL + MOD >= AC | ROLL = 20 {
  result: 10
 }
 result: 1
}

function: ROLL:n vs AC:n {
 if ROLL = 1 {result: -1}
 else if ROLL >= THREAT & ROLL + MOD >= AC {
  result: [threat d20]
 }
 else if ROLL + MOD >= AC | ROLL = 20 { result: 1 }
 result: 0
}

output [d20 vs AC]

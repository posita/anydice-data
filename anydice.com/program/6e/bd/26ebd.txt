function: check ROLLA:n with DMGA:d and ROLLB:n with DMGB:d ac AC:n {
  if ROLLA >= AC & ROLLB >= AC {
   result: (DMGA)+(DMGB)
  }
  else if ROLLA >= AC & ROLLB < AC {
   result: DMGA
  }
  else if ROLLA < AC & ROLLB >= AC {
   result: DMGB
  }
  else {
   result: 0
  }
}

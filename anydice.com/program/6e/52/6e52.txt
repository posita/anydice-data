function: maxpool of A:n by B:n {
 V: 0
 loop N over {1..A}{
  V: [highest of V and NdB]
 }
 result: V
}

D: 6

loop M over {1..5}{
L: 4*M
output D + (L-1)dD named "Lv [L] (Rolled)"
output D + (L-1)*(D/2 + 1) named "Lv [L] (Fixed)"
output D + (L-1)*(D/2) named "Lv [L] (Fixed Low)"
output [maxpool of L by D] named "Lv [L] (Pool)"
output D + [maxpool of L-1 by D] named "Lv [L] (Pool Mod)"
}

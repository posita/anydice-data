set "maximum function depth" to 5

function: reroll N:n MIN MAX {
  if N >= MIN {
    if N <= MAX { result: [reroll d10 MIN MAX] }
    result: N
  }
  result: N
}

function: hit N:n MINH MIND {
  if N >= MINH {
    if N >= MIND { result: 2 }
    result: 1
  }
  result: 0
}

output [hit [reroll d10 5 6] 7 10] named "reroll 5 & 6"

loop N over {7..10} {
  output [hit d10 7 N] named "double [N]s"
}

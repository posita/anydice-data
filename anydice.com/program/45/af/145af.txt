CRIT: 20
AC: 11
NORMAL: 1d20
ADV: 1@2d20
DISADV: 2@2d20


function: roll ROLL:n plus MOD:n {
  if ROLL <= 1 { result: 0 }
  if ROLL >= CRIT { result: 2 }
  result: ROLL + MOD >= AC
}

function: multiattack N:n times DIE:d MOD:n NORMDMG:d CRITDMG:d{
  result: [damage [roll DIE plus MOD] N DIE MOD NORMDMG CRITDMG 0 0]
}

function: multiattack N:n times DIE:d MOD:n NORMDMG:d CRITDMG:d FIRSTBONUS:d FIRSTCRITBONUS:d {
  result: [damage [roll DIE plus MOD] N DIE MOD NORMDMG CRITDMG FIRSTBONUS FIRSTCRITBONUS]
}

function: gw multiattack N:n times DIE:d MOD:n X:n x Y:n plus B:n{
  result: [gw damage [roll DIE plus MOD] N DIE MOD X x Y plus B]
}

function: damage HIT:n N:n DIE:d MOD:n NORMDMG:d CRITDMG:d FIRSTBONUS:d FIRSTCRITBONUS:d {
  if N <= 0 { result: 0 }

  if HIT = 1 {
    result: NORMDMG + FIRSTBONUS + [damage [roll DIE plus MOD] N-1 DIE MOD NORMDMG CRITDMG 0 0]
  }
 
  if HIT = 2 {
    result: CRITDMG + FIRSTCRITBONUS + [damage [roll DIE plus MOD] N-1 DIE MOD NORMDMG CRITDMG 0 0]
  }

  result: [damage [roll DIE plus MOD] N-1 DIE MOD NORMDMG CRITDMG FIRSTBONUS FIRSTCRITBONUS]
}

function: gw damage HIT:n N:n DIE:d MOD:n X:n x Y:n plus B:n {
  if N <= 0 { result : 0 }

  if HIT = 1 {
    result: [reroll X x Y less than 3] + B + [gw damage [roll DIE plus MOD] N-1 DIE MOD X x Y plus B]
  }
 
  if HIT = 2 {
    result: [reroll 2*X+1 x Y less than 3] + B + [gw damage [roll DIE plus MOD] N-1 DIE MOD X x Y plus B]
  }

  result: [gw damage [roll DIE plus MOD] N-1 DIE MOD X x Y plus B]
}

function: reroll X:n x Y:n less than Z:n {
  SUM: 0
  loop I over {1..X} {
    SUM: SUM + [reroll 1dY of Y less than Z]
  }
  result: SUM
}

function: reroll W:n of Y:n less than Z:n {
  if W < Z { result: 1dY}
  result: W
}

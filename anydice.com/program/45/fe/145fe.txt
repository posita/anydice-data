function: goaltime SEQ:s plus MOD:n versus TAR:n {
  if {1..#SEQ}@SEQ + MOD >= TAR { result: #SEQ@SEQ }
  result: 0
}

output [goaltime 3d6 plus 0 versus 10]

\loop MODIFIER over {0..6} {
  output [goaltime 3d6 plus MODIFIER versus 10] named "3d6 + [MODIFIER] vs 10]"
}\

loop TARGET over {2..18} {
   output [goaltime 3d6 plus 0 versus TARGET] named "3d6 vs [TARGET]"
}

function: best of DICE:s against TARGETNUMBER:n {
  if 1@DICE <= TARGETNUMBER {
    result: 1
  }

  result: 2@DICE <= TARGETNUMBER
}

loop SKILL over {2, 4, 6, 8, 10, 12, 14, 16, 18} {
  output [best of 2d20 against SKILL] named "2d20, pick best, against [SKILL]"
}

function: flip TEN:n and ONE:n against TARGETNUMBER:n {
  if (TEN-1)*10+ONE <= TARGETNUMBER {
    result: 1
  }
  result: (ONE-1)*10+TEN <= TARGETNUMBER
}

output 0 named "---------------------------------------"

loop SKILL over {10, 20, 30, 40, 50, 60, 70, 80, 90} {
  output [flip d10 and d10 against SKILL] named "d100 against [SKILL]"
}

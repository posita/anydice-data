\ *** SETTINGS *** \
AC: d{12..25}
CRIT: 20
MOD: +4
PROF: +4

SWORDLEVEL: +1

BOOMINGHITDMG: 1d8
SHADOWBLADEHITDMG: 3d8

\ *** DATA *** \

ROLL: 1d20
ADVANTAGE: 1@2d20
SUPERADVANTAGE: 1@3d20

SWORDDAMAGEROLL: 1d6

BOOMINGCRITDMG: 2d(BOOMINGHITDMG)

SWORDATTACKHITBONUS: MOD + PROF + SWORDLEVEL
SWORDHITDMG: SWORDDAMAGEROLL +MOD +SWORDLEVEL
SWORDCRITDMG: 2d(SWORDDAMAGEROLL) +MOD +SWORDLEVEL
SWORDHITDMGTWF: SWORDDAMAGEROLL +SWORDLEVEL
SWORDCRITDMGTWF: 2d(SWORDDAMAGEROLL) +SWORDLEVEL

SHADOWBLADEHITBONUS: MOD + PROF
SHADOWBLADEHITDMG: SHADOWBLADEHITDMG +MOD
SHADOWBLADECRITDMG: 2d(SHADOWBLADEHITDMG) +MOD
SHADOWBLADEHITDMGTWF: SHADOWBLADEHITDMG
SHADOWBLADECRITDMGTWF: 2d(SHADOWBLADEHITDMG)

ANIMATEOBJECTSCOUNT: 10
ANIMATEOBJECTSHITBONUS: +8
ANIMATEOBJECTSHITDMG: 1d4 +4
ANIMATEOBJECTSCRITDMG: 2d4 +4

\ *** FORMULAS *** \

function: attack ROLL:n plus BONUS:d vs AC:n {
  if ROLL <= 1 { result: 0 }
  if ROLL >= CRIT { result: 2 }
  result: (ROLL + BONUS) >= AC
}

function: damage for ATTACKRESULT:n HITDAMAGE:d crit CRITICALDAMAGE:d {
  if ATTACKRESULT = 2 {
    result: CRITICALDAMAGE
  }
  if ATTACKRESULT = 1 {
    result: HITDAMAGE
  }
  result: 0
}

function: attack with ROLL:d plus HITBONUS:d for HITDMG:d crit CRITDMG:d {
  result: [damage for [attack ROLL plus HITBONUS vs AC] HITDMG crit CRITDMG]
}

\ *** OUTPUTS *** \

MAINATTACK: [attack with SUPERADVANTAGE plus SWORDATTACKHITBONUS for SWORDHITDMG + BOOMINGHITDMG crit SWORDCRITDMG + BOOMINGCRITDMG]
EXTRAATTACK: [attack with ROLL plus SWORDATTACKHITBONUS for SWORDHITDMG crit SWORDCRITDMG]
BONUSATTACK: [attack with ROLL plus SWORDATTACKHITBONUS for SWORDHITDMGTWF crit SWORDCRITDMGTWF]
output MAINATTACK + EXTRAATTACK + BONUSATTACK named "Shortsword / Shortsword"

MAINATTACK: [attack with SUPERADVANTAGE plus SWORDATTACKHITBONUS for SWORDHITDMG + BOOMINGHITDMG crit SWORDCRITDMG + BOOMINGCRITDMG]
EXTRAATTACK: [attack with ROLL plus SWORDATTACKHITBONUS for SWORDHITDMG crit SWORDCRITDMG]
BONUSATTACK: [attack with ROLL plus SHADOWBLADEHITBONUS for SHADOWBLADEHITDMGTWF crit SHADOWBLADECRITDMGTWF]
output MAINATTACK + EXTRAATTACK + BONUSATTACK named "Shortsword / Shadowblade"

MAINATTACK: [attack with SUPERADVANTAGE plus SHADOWBLADEHITBONUS for SHADOWBLADEHITDMG + BOOMINGHITDMG crit SHADOWBLADECRITDMG + BOOMINGCRITDMG]
EXTRAATTACK: [attack with ROLL plus SHADOWBLADEHITBONUS for SHADOWBLADEHITDMG crit SHADOWBLADECRITDMG]
BONUSATTACK: [attack with ROLL plus SWORDATTACKHITBONUS for SWORDHITDMGTWF crit SWORDCRITDMGTWF]
output MAINATTACK + EXTRAATTACK + BONUSATTACK named "Shadowblade / Shortsword"

MAINATTACK: [attack with SUPERADVANTAGE plus SWORDATTACKHITBONUS for SWORDHITDMG + BOOMINGHITDMG crit SWORDCRITDMG + BOOMINGCRITDMG]
EXTRAATTACK: [attack with ROLL plus SWORDATTACKHITBONUS for SWORDHITDMG crit SWORDCRITDMG]
BONUSATTACK: [attack with SUPERADVANTAGE plus SHADOWBLADEHITBONUS for SHADOWBLADEHITDMGTWF crit SHADOWBLADECRITDMGTWF]
output MAINATTACK + EXTRAATTACK + BONUSATTACK named "Shortsword / Shadowblade in dim light or darkness"

MAINATTACK: [attack with SUPERADVANTAGE plus SHADOWBLADEHITBONUS for SHADOWBLADEHITDMG + BOOMINGHITDMG crit SHADOWBLADECRITDMG + BOOMINGCRITDMG]
EXTRAATTACK: [attack with SUPERADVANTAGE plus SHADOWBLADEHITBONUS for SHADOWBLADEHITDMG crit SHADOWBLADECRITDMG]
BONUSATTACK: [attack with ROLL plus SWORDATTACKHITBONUS for SWORDHITDMGTWF crit SWORDCRITDMGTWF]
output MAINATTACK + EXTRAATTACK + BONUSATTACK named "Shadowblade / Shortsword in dim light or darkness"

MAINATTACK: [attack with SUPERADVANTAGE plus SWORDATTACKHITBONUS for SWORDHITDMG + BOOMINGHITDMG crit SWORDCRITDMG + BOOMINGCRITDMG]
EXTRAATTACK: [attack with ROLL plus SWORDATTACKHITBONUS for SWORDHITDMG crit SWORDCRITDMG]
BONUSATTACK: ANIMATEOBJECTSCOUNT * [attack with ROLL plus ANIMATEOBJECTSHITBONUS for ANIMATEOBJECTSHITDMG crit ANIMATEOBJECTSCRITDMG]
output MAINATTACK + EXTRAATTACK + BONUSATTACK named "Shortsword / Animate Objects"

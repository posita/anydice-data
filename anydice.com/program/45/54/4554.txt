\ Roll a number of dice, add together all the similar numbers. Take the highest result and remove any ones from that. \

function: highest sum of DICEPOOL:s {
  HIGHEST: 1@DICEPOOL
  ONES: 1*[count 1 in DICEPOOL]
  TWOS: 2*[count 2 in DICEPOOL]
  THREES: 3*[count 3 in DICEPOOL]
  FOURS: 4*[count 4 in DICEPOOL]
  FIVES: 5*[count 5 in DICEPOOL]
  SIXES: 6*[count 6 in DICEPOOL]
  SEVENS: 7*[count 7 in DICEPOOL]
  EIGHTS: 8*[count 8 in DICEPOOL]
  NINES:9*[count 9 in DICEPOOL]
  TENS: 10*[count 10 in DICEPOOL]

  TEMP: [sort {HIGHEST, TWOS, THREES, FOURS, FIVES, SIXES, SEVENS, EIGHTS, NINES, TENS}]
  result: [check highest in TEMP removing ONES]
}

function: check highest in DICE:s removing THEONES {
  result: 1@DICE - THEONES
}

loop I over {1..5} {
  output [highest sum of Id6] named "[I]d6"
}

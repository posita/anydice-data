KEEP: 2
ATTACKS: 5
SKILL: 4
BONUS: 1
TOTALDICE: ATTACKS*SKILL

loop I over {1..ATTACKS}{
   VALUE: 0
   loop J over {0..(KEEP-1)}{
      VALUE: VALUE + (I * KEEP - J) @ (TOTALDICE)d6
   }
   output X + BONUS named "[TOTALDICE] Dice Attack: [I]"
}

output [highest KEEP of SKILLd6] + BONUS named "[SKILL] Dice"

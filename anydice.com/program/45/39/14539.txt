function: rerollone F:n S:n under N:n {
   if F+S < N {
      if F<S { result: 1d6+S } else {result: 1d6+F}
   } else {result: F+S}
}



function: rerolltwo F:n S:n T:n under N:n
{
	if F+S < N 
	{
		if F<S
		{ 
			if T+S < N 
			{
				if T<S 
				{
					result: 1d6+S
				}
					else
				{
					result: 1d6+T
				} 
			}
			else 
			{
				result:T+S
			} 
		}
		else 
		{
			if F+T < N 
			{
				if T<F
				{
					result: 1d6+F
				}
				else 
				{
					result: 1d6+T
				} 
			}
			else 
			{
				result:T+F
			}
		}
	}
	else {result: F+S}
}



output [rerolltwo 1d6 1d6 1d6 under 10] named "reroll two"
output [rerollone 1d6 1d6 under 10] named "reroll one"

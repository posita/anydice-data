function: explode N:n { if N < 3 { result: 2d6 } result: N }

function: infcrit attack A:n damage D:n ac C:n
{
  if A > C
  {
    if A > 18
    {
      result: D + [infcrit attack A damage D ac C]
    }
  }
  result: D
}

set "maximum function depth" to 20

output [infcrit attack 1d20 damage [explode 2d6] ac 16]

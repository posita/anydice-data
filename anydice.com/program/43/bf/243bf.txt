function: compare ROLL:s {
 ONES: 0
 SIXES: 0
 loop D over ROLL{
   if D = 1 {ONES: ONES + 1}
  \ if D = 5 {SIXES: SIXES + 1}\
   if D = 6 {SIXES: SIXES + 1}
 }

if ONES > SIXES {result: -1} \Failure\
if SIXES > ONES {result:  1} \Success\
result: 0 \Conditional Success\
}

}

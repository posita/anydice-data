function: hit DIE:d ATK:n vs AC:n CRIT:n {
  ROLL: [highest 1 of DIE]
  result: (ROLL + ATK >= AC) & ROLL < CRIT
}

function: crit DIE:d vs CRIT:n {
  ROLL: [highest 1 of DIE]
  result: ROLL >= CRIT
}

function: dmg DIE:d ATK:n for DMGD:d DMGF:n vs AC:n CRIT:n {
  CRITDMG: [crit DIE vs CRIT] * (2d(DMGD) + DMGF)
  NORMALDMG: [hit DIE ATK vs AC CRIT] * (DMGD + DMGF)
  result: CRITDMG + NORMALDMG
}

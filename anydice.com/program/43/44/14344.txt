MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL > 17 { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}
function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}
function: attack {
 result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]
}
DEFENSE: d{14..25}
function: reroll DIE:d {
 result: [reroll DIE helper]
}

function: reroll N:n helper {
 if N < 3 { result: DIE}
 result: N
}

ATTACK: 11
HITDAMAGE: d12 + 5
CRITICALDAMAGE: 2d12 + 5
MISSDAMAGE: 0
output [attack] named "Greataxe"

HITDAMAGE: [reroll d12] + 5
CRITICALDAMAGE: 2d[reroll d12] + 5

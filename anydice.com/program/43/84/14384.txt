\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: [saveroll d20 vs SAVE for CRITDAMAGE]} 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: [saveroll d20 vs SAVE for DAMAGE]}
 result: 0
}

function: saveroll ROLL:n vs SAVE:n for DAMAGE:n {
 if ROLL + SAVEMOD >= SAVE { result:DAMAGE }
 result: ( DAMAGE / 2 )
}


\looped against enemy AC from 15-20\
ENEMYAC: d{10..20}
SAVE: 14

SAVEMOD: {0..5}

ATTACKMOD: 2
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 2d8
CRITDAMAGE: 4d8
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Grenade damage per target"

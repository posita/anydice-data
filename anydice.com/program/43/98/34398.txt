function: roll ACC:n {
RES: [count {ACC..6} in 8d6] - [count {4..6} in 6d6]
result: RES
}
function: hitcheck SUC:n DAM:n CRI:n {
  if (SUC >= 1) {
     RES: DAM+(SUC*CRI)
  }  else {
     RES: 0
  }
result: RES
}
function: normal ACC:n DAM:n CRI:n {
ROL: [roll ACC]
RES: [hitcheck ROL DAM CRI]
result: RES
}

output [normal 4 5 2] named "raingun"

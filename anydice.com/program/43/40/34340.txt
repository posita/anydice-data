function: normal SUC:n DAM:n CRI:n {
  if (SUC >= 1) {
     RES: DAM+(SUC*CRI)
  }  else {
     RES: 0
  }
result: RES
}
function: armour SUC:n DAM:n CRI:n {
  if (SUC >= 1) {
     RES: (DAM+(SUC*CRI))/2
  }  else {
     RES: 0
  }
result: RES
}
function: high SUC:n DAM:n CRI:n {
RES: DAM+(SUC*CRI)
result: RES
}
ACC: 4
SUC: [count {ACC..6} in 8d6] - [count {4..6} in 6d6]

output [normal SUC 5 2] named "raingun"

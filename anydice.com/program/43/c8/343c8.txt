I_BREAKS: 0
I_HOPES: 0
RED: {1000, 100, 1, 101, 101, 1002}
BLACK: {1100, 101, 101, 1002, 102, 1202}
WHITE: {1101, 1201, 301, 1102, 302, 1203}
DOES: 1
DOESNT: 0

function: condense FACE:n usetokens T:n{
  BREAKS: I_BREAKS
  HOPES: I_HOPES
  if FACE >= 1000 {
    DOTS: FACE / 1000
    if HOPES > 0 & T {
      HOPES: HOPES - DOTS
      FACE: FACE + DOTS
      if HOPES < 0 {
        FACE: FACE + HOPES
      }
    }
    FACE: FACE - (1000 * DOTS)
  }

  if FACE >= 100 {
    POTENTIALS: FACE / 100
    if HOPES > 0 & T {
      HOPES: HOPES - POTENTIALS
      FACE: FACE + POTENTIALS
      if HOPES < 0 {
        FACE: FACE + HOPES
      }
    }
    if BREAKS > 0 & T {
      BREAKS: BREAKS - POTENTIALS
      FACE: FACE + POTENTIALS
      if BREAKS < 0 {
        FACE: FACE + BREAKS
      }
    }
    FACE: FACE - (100 * POTENTIALS)
  }
  result: FACE
}

function: rolls A:n plus B:n power P:d bonus Q:s tohit H:n usetokens T:n{
  RESULT: 1d0
  ROLL: d10 + B >= H
  loop X over {1..A}{
    if X=1 {
      RESULT: RESULT + (ROLL * (P+Q))
    }
    else {
      RESULT: RESULT + (ROLL * P)
    }
  }

  RESULT: [condense RESULT usetokens T]
  
  result: RESULT

}

function: rolls A:n dice plus B:n to hit using P:d power dice per hit with a bonus Q:s dice against to hit H:n that T:n use tokens {
  result: [rolls A plus B power P bonus Q tohit H usetokens T]
}

function: explode N:n { if N < 3 { result: 2d6+5 } result: N }

function: infcrit attack A:n damage D:n ac C:n
{
  if A > 17
  {
    result: D + 1d6 + [infcrit attack A damage D ac C]
  }
  if A >= C
  {
    result: D
  }
  result: 0
}

set "maximum function depth" to 3

output [infcrit attack 1d20+11 damage [explode 2d6+5] ac 16]

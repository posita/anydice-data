function: mean D:d {
  S:{D}
  A:0
  loop N over {1..#S}
  {
    A : A + N@S
  }
  result: A/#S
}

function: tohit DICE:d ac AC:n
{
  result: d(DICE>=d{AC})
}
function: tohit DICE:d ac AC:n damage DAMAGE:d {
    result: [tohit DICE ac AC]*DAMAGE
}
HIT: 1d20 + 5
HB: -1d4
DAM: 1d4
DB: 1

loop N over {1..20}{
  output [tohit HIT ac N damage DAM] named "ac[N] damage"
  output [tohit (HIT + HB) ac N damage (DAM + DB)] named "ac[N] [HB] tohit bonus"
}

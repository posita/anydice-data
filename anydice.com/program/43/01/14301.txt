
loop I over {1..3} {
  N: 3-I+1
loop D over {1..5}{
  output [lowest 3 of (3+N)d{-1, 0, 0, 1, 1, 2}] >= D named "success at difficulty [D] with malus [N]"
}
}
loop N over {0..3} {
loop D over {1..5}{
  output [highest 3 of (3+N)d{-1, 0, 0, 1, 1, 2}] >= D named "success at difficulty [D] with bonus [N]"
}
}

function: explode N:n { if N < 3 { result: 2d6+5 } result: N }

function: infcrit attack A:n bonus B:n damage D:n ac C:n
{
  if A > 17
  {
    result: D + 1d6
  }
  if A + B >= C
  {
    result: D
  }
  result: 0
}

set "maximum function depth" to 3

function: highest of A:n and B:n and C:n {
  if A > B { 
     if A > C {
        result: A
     }
     result: C
  }
  if B > C {
     result: B
  }
  result: C
}

function: push DICE:d help ROLL:n {

}

function: does ROLL:n beat DIFF:n {
   if ROLL < (DIFF / 2)          { result: 0 }
   if ROLL < DIFF                { result: 0 }
   if ROLL < (DIFF + (DIFF / 2)) { result: DC }
   else                          { result: DC }
}

\ list from lowest to highest so that smaller dice wins ties \
\ e.g. A=d6, B=d8, C=d10, D=d12 \

function: high A:n and B:n and C:n and D:n {
 if A >= [highest of B and [highest of C and D]] { result: 6 }
 else {if B >= [highest of C and D] { result: 8 }
       else {if C >= D { result: 10 }
             else { result: 12 } }}

}

output [high d6 and d8 and d10 and d12]

\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n plus BONUS:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + BONUS + PROFICIENCY + ATTACKMOD >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{17..17}

ATTACKMOD: 8
DMGBONUS: 4
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 4
DAMAGE: 1d10 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d10 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 plus d0 vs ENEMYAC] named "Rajada"


function : dres N:n DC:n
{
 OUT : N > {DC - 10,DC-1,DC+9,DC+9}
 result : OUT
}

function : pf flat DC:n
{

 NORM : [dres 1d{1-10,2..19,20+10} DC]
 result : NORM
}

function : pf basic save POWER:d DC:n
{
 EVALS : [pf flat DC]
 result : (POWER * 1d(EVALS))/2
}

output [dres 16 5]
output [pf basic save 2 5]

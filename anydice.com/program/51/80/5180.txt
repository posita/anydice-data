function: attack A vs B {
  if B >= A {
    result: Ad6
  }
  result: Bd6 + (A-B)d4
}

function: defend A vs B {
  if A >= B {
    result: Bd6
  }
  result: Ad6 + (B-A)d4
}

loop N over {1..5} {
  loop M over {1..5} {
    output [attack N vs M] > [defend N vs M] named "Hit Chance: [N] dice vs [M] dice"
    output [attack N vs M] > [defend N vs M] + 5 named "Big Hit Type 1 Chance: [N] dice vs [M] dice"
    output [attack N vs M] > [defend N vs M] + 4 named "Big Hit Type 2 Chance: [N] dice vs [M] dice"
  }
}

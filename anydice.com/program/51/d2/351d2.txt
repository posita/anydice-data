\ 4+ successo, 1 toglie un successo \
function: ROLL:s roll {
 result: (ROLL >= 4) - (ROLL = 1)
}

\ 4+ successo con il 6 successo doppio, 1 toglie un successo \
function: ROLL:s rollextra {
 result: (ROLL = 6) * 2 + (ROLL = 5) + (ROLL = 4) - (ROLL = 1)
}


output [1d6 roll] named "1 die normal"
output [2d6 roll] named "2 dice normal"
output [3d6 roll] named "3 dice normal"
output [4d6 roll] named "4 dice normal"
output [5d6 roll] named "5 dice normal"

output [1d6 rollextra] named "1 die 6 double"
output [2d6 rollextra] named "2 dice 6 double"
output [3d6 rollextra] named "3 dice 6 double"
output [4d6 rollextra] named "4 dice 6 double"
output [5d6 rollextra] named "5 dice 6 double"

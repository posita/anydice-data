
function: evaluate ROLL:s {
 ASUCCESSES: ROLL >= d12
 BSUCCESSES: ROLL >= d12
 result: ASUCCESSES + BSUCCESSES
}

function: evaluate ROLL:s power POW:s {
 ASUCCESSES: ROLL >= d12
 BSUCCESSES: ROLL >= d12
 result: ASUCCESSES + BSUCCESSES + POW
}

loop POW over {0..2} {
 loop MOD over {1..5}{
 output [evaluate 1d6 + MOD power POW] named "1d6+[MOD] PL[POW]"
 }
}

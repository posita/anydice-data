function: atk R:n B:n D:n{
  if R = 12 {result: 2}
  if R+B >= D {result: 1}
  result: 0
}

A: 0
B: 0

SUMA: 0
SUMB: 0
SUMAB: 0

loop N over {4..13}{
 A: [atk 2d6 1 N]*3
 B: [atk 2d6 0 N]*5
 \output A-B named "[N]"\
SUMA: SUMA+A
SUMB: SUMB+B
SUMAB: SUMAB+(A-B)
\output SUMA named "sumA"\
\output SUMB named "sumB"\
\output SUMAB named "[N]"\
}

\output SUMA named "sum A"
output SUMB named "sum B"\
output SUMA-SUMB named "sum A - sum B"
\output SUMAB named "sum(A-B)"\

DV:6
PUNCHES:22
ARMSTR:16
TRACER:0
LASER:0
STAGGERED:1
FLASHBULB:1
DICE: 6+LASER
KERESIST: 2
NUMOFDODGES:2
DODGEVALUE:3




NUMEROFSUCCESSFULHITS:0
loop P over {1..PUNCHES}{
if NUMOFDODGES>0 {
 NUMOFDODGES:NUMOFDODGES-1
NUMEROFSUCCESSFULHITS: NUMEROFSUCCESSFULHITS+((DV-2*FLASHBULB-TRACER+DODGEVALUE)<= DICEd{0,0,0,1,1,2})
}
else {
 NUMEROFSUCCESSFULHITS: NUMEROFSUCCESSFULHITS+((DV-2*FLASHBULB-TRACER)<= DICEd{0,0,0,1,1,2})
}
}
output NUMEROFSUCCESSFULHITS named "Successful Attack Chances"
output NUMEROFSUCCESSFULHITS*(ARMSTR/2+3*STAGGERED+3*FLASHBULB-KERESIST) named "Likely Damage"

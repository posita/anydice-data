function: DICE:d with gwf {
  X: 0
  loop N over {1..#DICE} {
    X: X + [reroll 1d[maximum of 1@DICE] as 1d[maximum of 1@DICE] under 3]
  }
  result: X
}

function: reroll ROLL:n as REROLL:d under LIMIT:n {
  if ROLL < LIMIT {result: REROLL}
  result: ROLL
}

output 2d6 named "Greatsword"
output [2d6 with gwf] named "Greatsword (GWF)"

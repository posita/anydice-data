function: explode N:n with SIGN:n {
R: 0
if N = 6 {
R: [explode d6 with 1]
}
else if N = 1 {
R: [explode d6 with -1]
}
result: R + N * SIGN
}

ADD: [explode d6 with 1]
SUB: [explode d6 with -1]

function: highest N:n of ROLL:s {
result: {1..N}@ROLL + [lowest of ROLL = 6 and N]dADD + [lowest of ROLL = 1 and N]dSUB
}

output [highest of [lowest of [highest 3 of 4d6] and 60] and 4]

function: most common number in ROLL:s {
  MAX: 1@ROLL
  loop SIDE over ROLL {
    if (SIDE = ROLL) > (MAX = ROLL) {
       MAX: SIDE
    }
  }
  result: MAX 
}

function: count and max in ROLL:s {
  MAX: [most common number in ROLL]
  COUNT: MAX = ROLL
  result: 10*COUNT + MAX
}

loop N over {3..7} {
  output [count and max in Nd{0..9}] named "[N] dice"
}

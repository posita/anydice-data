function: D damage B bonus dice {
 if B<5 {
   result: Dd(12-2*B)
 }
 if B < 8 {
   result: Dd(8-B)
 }
 result: Dd1
}
function: D damage B bonus {
 result: [count 1 in [D damage B bonus dice]]
}

COST: 7
loop B over {0..7}{
 if B < COST {
  D: COST-B
  output [D damage B bonus] named "[D]D [B]B"
 }
}

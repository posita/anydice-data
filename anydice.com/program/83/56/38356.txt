set "maximum function depth" to 12

function: explode N:n on SIZE:n {
 if N = SIZE & N > 1 & SIZE < 20 {
  result: [highest of N and [explode [upgrade SIZE]]]
 }
 result: N
}

function: upgrade DIE:d {
    SIZE: [maximum of DIE]
    if SIZE =  1 { result: d4 }
    if SIZE < 12 { result: d{1..SIZE + 2} }
    result: d20
}

function: downgrade DIE:d {
    SIZE: [maximum of DIE]
    if SIZE <=  4 { result: d1 }
    if SIZE <= 12 { result: d{1..SIZE - 2} }
    result: d12
}

function: explode DIE:d {
 result: [explode DIE on [maximum of DIE]] 
}

output [explode  d1] named "-"
output [explode  d4] named "d4"
output [explode  d6] named "d6"
output [explode  d8] named "d8"
output [explode d10] named "d10"
output [explode d12] named "d12"
output d20 named "d20"

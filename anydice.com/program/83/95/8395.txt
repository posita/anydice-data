function:  DICE:s with heat HEAT:n{
 SUM:HEAT
 loop N over {1..#DICE}{
  if N@DICE<=HEAT{
   SUM:SUM+N@DICE
  }
 }
 result:SUM
}
function:  DICE:s with cold HEAT:n{
 SUM:HEAT
 loop N over {1..#DICE}{
  if N@DICE>=HEAT{
   SUM:SUM+N@DICE
  }
 }
 result:SUM
}
A:3d10
B:1d10
output A
output [A with heat B]
output [A with cold B]

function: roll A:n with M:n plus D:d against DC:n for damage {
    if A = 1 {result:0}
    if A = 20 {result: M+2dD}
    if A >= DC {result: M+D}
    else {result:0}
}

loop DC over {8..21}{
    output 2d[roll 1d20+7 with 4 plus 1d8 against DC for damage] named "2 Attacks 1 Dice against DC [DC]"
    output [roll 1d20+7 with 4 plus 2d8 against DC for damage] named "1 Attacks 2 Dice against DC [DC]"
}

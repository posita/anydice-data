
function: POOL dice with dif DIF {
  SEQ: {}

  loop N over {1..6}{
    if DIF >= N {
      SEQ: {SEQ,0}
    }
    else {
      SEQ:{SEQ,1}
    }
  }
  result: POOLdSEQ
}

function: entropia ENTROPIA e atributo ATRIBUTO{
  result: [highest 1 of ENTROPIAd6] > [highest 1 of ATRIBUTOd6] 
}  



loop DIF over {3..4}{
  loop POOL over {3..5}{
    loop ENTROPIA over {0..3}{
      output [POOL dice with dif DIF] named "Pool de [POOL] dados com dificuldade [DIF]" 
      output [entropia ENTROPIA e atributo POOL] named "chance de caos com [POOL] dados e [ENTROPIA] entropia"
    }
  }
}

function: translate A:n {
 if A >= 12 { result: 2 }
 if A >= 7 { result: 1}
 result: 0
}

function: translates SEQUENCE:s {
 SORTED: {}
 loop P over SEQUENCE {
  SORTED: {SORTED, [translate P]}
 }
 result: SORTED
}

function: SEQUENCE:s easy{
 result: (SEQUENCE >=6 ) > 0
}

function: SEQUENCE:s medium{
 result: (SEQUENCE >=6 ) > 1 | (SEQUENCE >=12 ) > 0
}

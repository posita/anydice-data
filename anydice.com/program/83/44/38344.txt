function: m A:n { \ min function \
 if A>=0{result:0}
 result:A
}

function: sc A:s SP:n B:n C:n D:n { \ skill check {Attributes} Skill Points Dice1 Dice 2 Dice 3 \
  if [count 1 in {B,C,D}] >= 2 {result: 16}
  if [count 20 in {B,C,D}] >= 2 {result: -1} 
  result: SP+[m 1@A-B]+[m 2@A-C]+[m 3@A-D]
}

function: ql A:n { \ quality level \
 result: 1-(A<0)+(A>=4)+(A>=7)+(A>=10)+(A>=13)+(A>=16)
}

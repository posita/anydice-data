function: luck ROLL:s {
  if (ROLL = 1) { result: 1d20 }
  result: ROLL
}

function: advantage {
  result: [highest of [luck 1d20] and [luck 1d20]]
}

function: dpr against AC:n ac with ROLL:s plus ATK_MOD:n and DAMAGE:s plus DAM_MOD:n damage {
  if ROLL = 20 {
    result: 2d(3d6) + DAM_MOD
  }
  if ROLL + ATK_MOD >= AC {
    result: DAMAGE + DAM_MOD
  } else {
    result: 0
  }
}

output [dpr against 14 ac with [advantage] plus 5 and 3d6 plus 3 damage] named "Shortsword + Sneak Attack"

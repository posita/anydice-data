\
 Caltrop (modified)
 1: Things go bad.
 2: Success, but with undesired consequences.
 3: Clean success.
 4: Extreme success.
\

function: caltrop ROLL:s {
if 2@ROLL > 3 {
  result: 4
 }
 else {
  if 1@ROLL > 3 {
  result: 3
 }
 else {
  if 1@ROLL > 2 {
   result: 2
  }
  else {
    result: 1
  }
 }
}
}
output [caltrop 2@2d4] named "disadvantage"
loop P over {1..4} {
 output [caltrop Pd4] named "[P]d4"
}

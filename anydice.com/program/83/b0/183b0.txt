function: minus helper A:n B:n {
  if A * B < 0 { result: 0 }
  result: A
}
function: P:d minus Q:d loop N:n {
  D: P - Q
  loop I over {1..N} { D: [minus helper 1dD 1dD] }
  result: D
}

output [d100 <= 30 minus d100 <= 10 loop 45]
output [d100 <= 30 minus d100 <= 10 loop 50]
output [d100 <= 30 minus d100 <= 10 loop 55]

function: clip X:n {
  if X < 0 { result: d{} }
  result: X
}

function: do turn using sword with damage DIE:n and modifier MOD:n and multiattack penalty PEN:n and deadly DEADLY:n and bonus BONUS:n {
  TOTAL_DAMAGE: 0
  loop I over {1..3}{
    TOTAL_DAMAGE: TOTAL_DAMAGE + [swing sword with damage DIE and tohit (1d20 + MOD - (PEN * I)) and deadly DEADLY and bonus BONUS]
  }
  result: TOTAL_DAMAGE
}
function: swing sword with damage DIE:n and tohit TOHIT:n and deadly DEADLY:n and bonus BONUS:n {
  if(TOHIT >= 22) {
    result: 2 * ((dDIE)+3) + dDEADLY + BONUS
  } else {
    if(TOHIT >= 12) {
      result: (dDIE)+3 + BONUS
    }
  }
  result: 0
}

function: exclude RANGE:s from ROLL:n {
  if ROLL = RANGE { result: d{} }
  result: ROLL
}
function: sign of NUM:n {
  result: (NUM > 0) - (NUM < 0)
}
function: P:d minus Q:d {
  DIFF: P - Q
  loop I over {1..50} {
    TEMP: [exclude 0 from DIFF]
    DIFF: (DIFF != 0) * [sign of TEMP + TEMP]
  }
  result: [exclude -1 from DIFF]
}

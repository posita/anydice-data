function: explode N:n on SIZE:n {
 if N = SIZE {
  if SIZE =  4 { result: [highest of N and [explode  d6 on  6]] }
  if SIZE =  6 { result: [highest of N and [explode  d8 on  8]] }
  if SIZE =  8 { result: [highest of N and [explode d10 on 10]] }
  if SIZE = 10 { result: [highest of N and [explode d12 on 12]] }
  if SIZE = 12 { result: [highest of N and d20] }
 }
 result: N
}

function: explode DIE:d {
 result: [explode DIE on [maximum of DIE]] 
}

output [explode  d4 on  4] named "d4"
output [explode  d6 on  6] named "d6"
output [explode  d8 on  8] named "d8"
output [explode d10 on 10] named "d10"
output [explode d12 on 12] named "d12"
output d20 named "d20"

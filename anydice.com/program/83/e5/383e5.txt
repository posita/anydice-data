function: test A:n against S:s {
  loop N over [reverse {1..#S}]{
    if A >= N@S { result: N@S }
  }
  result: 0
}

function: roll A:n and B:n against S:s {
  loop N over [reverse {1..#S}]{
    if A+B >= N@S { if A >= B { result: N@S } else { result: N@S-1 } }
  }
  result: 0
}

function: roll A:n with C:n and B:n against S:s {
  loop N over [reverse {1..#S}]{
    if [highest of A+B and [highest of A+C and B+C]] >= N@S { if A >= B | C >= B { result: N@S } else { result: N@S-1 } }
  }
  result: 0
}

function: roll A:n and B:n with C:n against S:s {
  loop N over [reverse {1..#S}]{
    if [lowest of A+B and [lowest of A+C and B+C]] >= N@S { if A >= B & A >= C { result: N@S } else { result: N@S-1 } }
  }
  result: 0
}

A: {3,6,9,12}

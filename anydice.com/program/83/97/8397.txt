function:  DICE:s with heat HEAT:s{
 LOW:#HEAT@HEAT
 SUM:LOW*[count LOW in HEAT]
 loop N over {1..#DICE}{
  if N@DICE<=LOW{
   SUM:SUM+N@DICE
  }
 }
 result:SUM
}
function:  DICE:s with cold HEAT:n{
 HIGH:1@HEAT
 SUM:HIGH*[count HIGH in HEAT]
 loop N over {1..#DICE}{
  if N@DICE>=HIGH{
   SUM:SUM+N@DICE
  }
 }
 result:SUM
}
A:3d10
B:1d10
output A
output [A with heat B]
output [A with cold B]

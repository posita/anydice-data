function: ironsworn ROLL:n plus ADD:n over CHALLENGE:s {
  X: ROLL+ADD
  if X > 1@CHALLENGE { result: 2 }
  if X > 2@CHALLENGE { result: 1 }
  result: 0
}

loop N over {0} {
  output [ironsworn d4 plus N over 2d10] named "d4"
  output [ironsworn d6 plus N over 2d10] named "d6"
  output [ironsworn d8 plus N over 2d10] named "d8"
  output [ironsworn d10 plus N over 2d10] named "d10"
  output [ironsworn d12 plus N over 2d10] named "d12"
}

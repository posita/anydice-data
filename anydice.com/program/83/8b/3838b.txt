SIZE: 10 \size of the dice (e.g. 4 = d4)\
AMOUNT: 10 \number of dice to roll\
TARGET: 50 \target difficulty\



function: A:n overflows target B:n {
if (A >= B) {
T: (B / 5)
O: (A - B) / T
result: O
}
result: -1
}

loop N over {1..AMOUNT} {
output [Nd{1..SIZE} overflows target TARGET] named "[N]d[SIZE] (A)"
}

loop N over {1..SIZE} {
output [AMOUNTd{1..N} overflows target TARGET] named "[AMOUNT]d[N] (S)"
}

function: add over VALUE:n in SEQUENCE:s {
 COUNT: 0
 loop P over {1..# SEQUENCE} {
  if P@SEQUENCE > VALUE {
   COUNT: (COUNT + P@SEQUENCE)
  }
 }
 result: COUNT
}

 LOW: 0
 COUNT: 0
 loop P over {1..# SEQUENCE} {
  if LOW = 0{
   LOW: P@SEQUENCE
  }
  else if P@SEQUENCE < LOW {
   if LOW  > VALUE {
    COUNT: (COUNT + LOW )
   }
   LOW: P@SEQUENCE
  }
  else if P@SEQUENCE > VALUE {
   COUNT: (COUNT + P@SEQUENCE)
  }
 }
 result: COUNT
}

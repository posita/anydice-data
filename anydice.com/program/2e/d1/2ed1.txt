function: R:n exp E:n pif P:n {
  if R >= E {result: R + [d100 exp (E + 1) pif P]}
  if R <= P {result: R - d100}
  result: R
}

function: R:n skill S:n df D:n {
  if (R + S) >= D {result: 1}
  result: 0
}

EXP : [d100 exp 90 pif 3]

PRIM: [lowest of 100 and dEXP]

ANIMA: [dPRIM skill 35 df 80]

output dANIMA

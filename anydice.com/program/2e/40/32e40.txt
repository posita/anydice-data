set "maximum function depth" to 7

function: roll ROLL:n threshold T {
  if ROLL >= T {
   result: 1
  }

  result: 0
}

function: rollexploding ROLL:n threshold T {
  if ROLL = 6 {
   result: 1 + [rollexploding 1d6 threshold T]
  }

  result: [roll ROLL threshold T]
}

function: wrapper DICE:n threshold T explode E {
  RESULT: 0

  loop N over {1..DICE} {
    if E {
     RESULT: RESULT + [rollexploding 1d6 threshold T]
    } else {
     RESULT: RESULT + [roll 1d6 threshold T]
    }
  }

  result: RESULT
}

function: p DICE:n {
  result: [wrapper DICE threshold 4 explode 1]
}


output [p 1] named "pool of 1d6"

output [p 2] named "pool of 2d6"

output [p 3] named "pool of 3d6"

output [p 4] named "pool of 4d6"

output [p 5] named "pool of 5d6"

output [p 6] named "pool of 6d6"


function: reroll REROLL:n on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:n on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output 1d12 >= [reroll 10 on d10]
output [reroll 1 on d12] >= [reroll 6 on d6]

function: max A:n B:n {
  if A>B { result: A }
  else { result: B }
}

function: hit N:n D:n {
  result: NdD
}
function: crit N:n D:n {
  result: NdD + NdD
}
function: epiccrit N:n D:n {
  result: N*D + NdD
}
function: keancrit N:n D:n {
  result: [max NdD+NdD N*D]
}
function: redditcrit N:n D:n {
  result: N*(1+D)/2 + NdD
}

N: 2
D: 6
output [hit N D]  named "[N]d[D] hit"
output [crit N D] named "[N]d[D] crit"
output [epiccrit N D] named "[N]d[D] epic crit"
output [keancrit N D] named "[N]d[D] kean crit"
output [redditcrit N D] named "[N]d[D] reddit crit"

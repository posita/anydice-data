function: ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICAL }
  result: DAMAGE
 }
 if ROLL + ATTACK >= DEFENSE { result: DAMAGE }
 result: MISS
}

DEFENSE: d{20..25}
ATTACK: 15
MISS: 0

DAMAGE: d4 + 8 + 2d8
CRITICAL: 4 + 2d6 + 2d8
output [d20 vs DEFENSE] named "Magic dagger"

DAMAGE: d4 + 8 + 2d8
CRITICAL: 4 + 2d12 + 2d8
output [d20 vs DEFENSE] named "Vicious dagger"

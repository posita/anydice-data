function: press BLACK:s black RED:s red {
   if #RED > 0 & #RED@RED = 1 {
       result: -1
   }

   if #BLACK > 0 {
       if 1@BLACK = 6 { result: 2 }
       if 1@BLACK = 5 { result: 1 }
       else { result: 0 }
   }

   result: -6
}

loop R over {0..6} { 
    B: 6 - R
    output [press Bd6 black Rd6 red] named "[B] black, [R] red"
}

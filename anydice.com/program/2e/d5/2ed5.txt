TALENT: 1
REROLL: 5
TARGET: 20

function: reroll D:n if less or equal to N:n {
 if (D <= N) { 
   TALENT: 0
   result: [1d10 openend on 10]
 }
 else { result: D }
}

function: D:s openend on N:n {
 if (D = N) { result: 10 + [1d10 openend on N] }
 else { 
   if (TALENT) { result: [reroll D if less or equal to REROLL] }
   else { result: D }
 }
}

output 3d[1d10 openend on 10] >= TARGET named "higher than [TARGET]"

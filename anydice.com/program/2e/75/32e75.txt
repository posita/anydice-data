DIA: 8 \ damage and attack stat \
DIB: 4 \ secondary attack stat \
HIT: 0 \ accuracy bonus \
DMG: 0 \ damage bonus \
INI: 5 \ target initiative \
ARM: 2 \ target armor \


function: attack ROLLA:n ROLLB:n hit WPH dmg WPD {
if ROLLA = DIA | ROLLA = 6 {
	if ROLLB = DIB | ROLLB = 6 { result: [highest of 0 and (2dDIA + DMG + WPD - ARM) ] }
}
if ROLLA + ROLLB + HIT + WPH < INI | ROLLA = ROLLB = 1 { result: 0 }
result: [highest of 0 and (1dDIA + DMG + WPD - ARM) ]
}

output [attack 1dDIA 1dDIB hit 1 dmg -1] named "Light Blade"
output [attack 1dDIA 1dDIB hit 0 dmg 0] named "Blade"
output [attack 1dDIA 1dDIB hit 0 dmg 1] named "Polearm"
output [attack 1dDIA 1dDIA hit -1 dmg 0] named "Axe"
output [attack 1dDIA 1dDIB hit -2 dmg 0] named "Bow"
output [attack 1dDIA 1dDIB hit 0 dmg -2] named "Unarmed"
output [attack 1dDIA 1dDIB hit 0 dmg -1] named "Improvised"

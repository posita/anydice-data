function: multiattack X:n Y:n Z:d A {
  TOTAL: 0
  loop N over {1..Y} {
    if X = 20 {TOTAL: TOTAL+Z*2}
    else if X < A {TOTAL: TOTAL+0}
    else {TOTAL: TOTAL + Z}
  }
  result: TOTAL
}

ATTACKDICE: 2d10
AC: 0

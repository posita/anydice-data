function: rollskill LVL with POOL {
 X: POOLd6
 ERF: [count 6 in X]
 loop N over {0..LVL}{
  C: [count (5 - N) in X]
  NEWC: {}
  loop E over {C} {
   if E > (LVL - N + 1){
    NEWC: {NEWC, (LVL - N + 1)}
   }
   else {
    NEWC: {NEWC, E}
   }
  }
  ERF: ERF + C
 }

 result: ERF
}

output [rollskill 0 with 6]
output [count {5,6} in 6d6]

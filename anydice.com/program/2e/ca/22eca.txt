function: bladed NAME:s and SKILL:s and ASSET:s {
 NOUT: 0
 PARTIAL: 4
 FULL: 6
 EXPLODE: 0
 if 1@NAME >= PARTIAL { OUT: OUT + 1 }
 if 1@NAME >= FULL { OUT: OUT + 1 }
 if 1@SKILL >= PARTIAL { OUT: OUT + 1 }
 if 1@SKILL >= FULL { OUT: OUT + 1 }
 if 1@ASSET >= PARTIAL { OUT: OUT + 1 }
 if 1@ASSET >= FULL { OUT: OUT + 1 }
 result: OUT
}

loop M over {0,6}{
 loop N over {0,4,8,12}{
  output [bladed 1d6 and 1dN and 1dM] named "1d6/S[N]/A[M]"
 }
}

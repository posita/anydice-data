function: undercut N:n M:n {
	if M = (N + 1)  {
		result: N + M
	}
	if M = (N - 1) {
		result: 0 - (N + M)
	}
	result: N - M
}

function: boom N:n {
 if N = 1 { result: N + [boom d{0,0,1}] }
 if N = -1 { result: N + [boom d{0,0,-1}] }
 result: N
}

output 4d{-1,0,1} - 4d{-1,0,1} named "Standard Fate dice (4DF vs 4DF)"
output (d6 - d6) - (d6 - d6) named "Variant Fate (d6-d6 vs d6-d6)"
output [boom d{-1,0,1}] named "Exploding Fate Die"
output [undercut 1d6-1 1d6-1] named "Undercut"

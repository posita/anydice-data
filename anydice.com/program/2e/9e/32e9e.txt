\ Frei Rollenspiel Dice Pool Mechanics \
\ \
\ 0: Failure \
\ 1: Partial Success \
\ 2: Success \
\ 3: 2+ Successes \

function: pool ROLL:s {
 if 1@ROLL >= 5 {
  result: 2 + (2@ROLL >= 5)
 }
 result: 1@ROLL >= 3
}

output [pool 2@2d6] named "0d"
loop P over {1..6} {
 output [pool Pd6] named "[P]d"
}

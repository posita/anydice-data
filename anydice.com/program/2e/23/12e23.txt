function: multiattack X:n Y:n Z:d A B {
  TOTAL: 0
  loop N over {1..Y} {
    if X = 20 {TOTAL: TOTAL + Z * 2 + B}
    else if X < A {TOTAL: TOTAL + 0}
    else if X = 1 {TOTAL: TOTAL + 0}
    else {TOTAL: TOTAL + Z + B}
  }
  result: TOTAL
}

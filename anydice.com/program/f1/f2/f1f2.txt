function: attack D:n plus ATK:n vs AC:n {
 if D=20 {result: 1}
 if D=1 {result: 0}
 result: D+ATK>=AC
}
 

loop MOD over {0..0} {
 loop AC over {10..19} {
  output [attack 1d20 plus MOD vs AC] named "Classic +[MOD] vs. AC[AC]"
  NEWAC: AC - 10
  output [attack 1d20 plus MOD vs 1d20+NEWAC] named "New +[MOD] vs. AC[NEWAC]"
 }
}

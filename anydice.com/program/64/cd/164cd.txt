SYSTEM: 1d12 + 1d6

function: attack H:d damage D:d block B:d resist A:d {
    HIT: SYSTEM + H
    BLOCK: SYSTEM + B
    HIT_BY: HIT - BLOCK
    result: [hit by HIT_BY with D damage on A armor]
}

function: hit by HB:n with D:d damage on A:d armor {
    if HB >= 0 {
        result: [highest of 0 and HB + D - A]
    } else {
        result: 0
    }
}

output [attack 1d4 damage 1d4 block 0 resist 0] named "Unarmed Human"
output [attack 1d10 damage 1d6 block 0 resist 0] named "Finesse Weapon"
output [attack 1d6 damage 1d12 block 0 resist 0] named "Standard Weapon"
output [attack 1d1 damage 1d20 block 0 resist 0] named "Brutal Weapon"

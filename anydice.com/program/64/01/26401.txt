WILD: [explode d6]
NORMAL: [explode d10]
TN: 4
ATTACK: [highest of WILD and NORMAL]
function: damage from ATTACK:n with target TN:n {
 if ATTACK >= TN + 4 {
  result: [explode 4d6]
 }
 else {
  if ATTACK >= TN {
   result: [explode 3d6]
  }
  else {
   result: 0
  }
 }
}
output [lowest of [damage from ATTACK with target TN] and 24]

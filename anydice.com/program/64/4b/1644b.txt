function: R:s successes {
  result: R > 5
}

function: spirit die with S:n successes {
  if [{1,3,5} contains S] = 1 {
    result: 1d4 = 4
  } else {
    result: 0
  }
}

function: NA:n die SA:n and NB:n die SB:n successes {
  SUC: [NAdSA successes] + [NBdSB successes]
  result: SUC + [spirit die with SUC successes]
}

function: VIRTUE:n and untrained successes {
  result: [3 die VIRTUE and 1 die 6 successes]
}

function: VIRTUE:n and SKILL:n successes {
  result: [3 die VIRTUE and 3 die SKILL successes]
}

output [6 and untrained successes] named "6/untrained"
output [6 and 6 successes] named "6/6"
output [6 and 8 successes] named "6/8 and 8/6"
output [6 and 10 successes] named "6/10 and 10/6"
output [8 and untrained successes] named "8/untrained"
output [8 and 8 successes] named "8/8"
output [8 and 10 successes] named "8/10 and 10/8"
output [10 and untrained successes] named "10/untrained"
output [10 and 10 successes] named "10/10"

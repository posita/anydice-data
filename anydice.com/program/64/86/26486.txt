function: mtasrev COUNT:n dice difficulty DIFFICULTY:n specialization SPECIALIZATION:n willpower WILLPOWER:n {
 STORYTELLER:{1,2:(DIFFICULTY-2),9:(10-DIFFICULTY),10}
 result:[mtasrev COUNT d STORYTELLER difficulty DIFFICULTY specialization SPECIALIZATION willpower WILLPOWER]
}

function: mtasrev ROLL:s difficulty DIFFICULTY:n specialization SPECIALIZATION:n willpower WILLPOWER:n {
 if SPECIALIZATION & STORYTELLER=10 {
  EXPLODE:d STORYTELLER
 }
 SUCCESSES:(STORYTELLER>=9)
 ONES:(STORYTELLER=1)
 if (SUCCESSES+WILLPOWER)>0 & SUCCESSES>ONES {result:SUCCESSES+WILLPOWER-ONES}
 if SUCCESSES=0 & ONES>0 {result:-1}
 result:0
}

output [mtasrev 1 dice difficulty 6 specialization 0 willpower 0]

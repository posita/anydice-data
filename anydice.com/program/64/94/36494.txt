\
F: {-1, 0, 1}

function: base BASE luck LUCK buffer BUFFER
{
  if BUFFER > 0
  {
    result: BASE + [highest LUCK of (LUCK+BUFFER)dF]
  }
  else
  {
    result: BASE + [lowest LUCK of (LUCK-BUFFER)dF]
  }
}

K: 20
loop N over {4..K}{
  output [base 0 luck N buffer 4]
}
\

function: roll N:n keep K:n
{
  result: [lowest of [count {6} in (Nd6)] and K]
}

K: 12
loop N over {1..K}
{
  output [roll N keep N]
}

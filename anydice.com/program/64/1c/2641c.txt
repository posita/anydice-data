function: attack ATTACK:d {
  result: ATTACK
}

function: multiattack X:n Y:d {
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: [attack Y]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

S: 8
B: 2
W: 4

XH: {1,0,0,0,2,2}
JA: {0,0,0,1,1,1}
BL: {0,1,1,2,2,3}

function: pos X:n {
  if X > 0 {
    result: X
  } 
  result: 0
}

function: hit A:n M:n {
  if A = 0 { result: 0 }
  result: A + M
}

function: onhit A:n M:n {
  if A = 0 { result: 0 }
  result: M
}

function: roll H:n {
  result: HdXH / 2
}

function: roll H:n bonus B:n {
  RO: [roll H]
  RES: RO + BdJA
  result: [pos RES]
}

function: attack H:n bonus B:n vs P:n {
  ATT: [roll H bonus B]
  DMG: [hit ATT W-P]
  result: [pos DMG]
}

function: attack H:n bonus B:n fire vs P:n {
  ATT: [roll H bonus B]
  DMG: [hit ATT 1-P]
  FIRE: [onhit ATT (1dBL+1)*2]
  result: [pos DMG+FIRE]
}

function: attack H:n bonus B:n burn vs P:n {
  ATT: [roll H bonus B]
  FIRE: [onhit ATT (1@2dBL+1)*2]
  result: [pos FIRE]
}

output [roll S bonus B] named "base roll"

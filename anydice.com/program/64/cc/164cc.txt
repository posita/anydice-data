function: attack H:d damage D:d block B:d resist A:d {
    HIT: 1d12 + H
    BLOCK: 1d12 + B
    HIT_BY: HIT - BLOCK
    result: [hit by HIT_BY with D damage on A armor]
}

function: hit by HB:n with D:d damage on A:d armor {
    if HB >= 0 {
        result: [highest of 0 and HB + D - A]
    } else {
        result: 0
    }
}

output [attack 1d10 damage 1d06 block 0 resist 0]
output [attack 1d06 damage 1d12 block 0 resist 0]
output [attack 1d01 damage 1d20 block 0 resist 0]

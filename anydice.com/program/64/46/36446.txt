FAIL: 0
MISS: 1
HIT: 2
CRITICALHIT: 3

function: attack ROLL:n vs DEFENSE:n NUMDICE:n {
 if ROLL = NUMDICE { result: FAIL }
 if ROLL = (10 * NUMDICE) {
  result: CRITICALHIT
 }
 if ROLL >= DEFENSE { result: HIT }

 result: MISS
}

function: attackb ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: FAIL }
 if ROLL = 10 {
  result: CRITICALHIT
 }
 if ROLL >= DEFENSE { result: HIT }

 result: MISS
}

output [attack d10 vs d10 1] named "Sin escudo"
output [attack 6d10 vs 6d10+d4 6] named "Con escudo"

output 6d10

function: attr ROLL:n {
  if ROLL < 1 { result: 1 }
  result: ROLL
}

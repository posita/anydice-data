SKILL: 6
STYLE: 5
POOL: SKILL + STYLE

FOCUS: {1..5}
MISS: 0
HIT: 1
CRITICALHIT: 2

function: attempt ROLL:n vs DIFFICULTY:n {
 if ROLL = 20 { result: MISS }
 if ROLL <= FOCUS { result: CRITICALHIT }
 if ROLL <= POOL { result: HIT }
 result: MISS
}

loop FOCUS over {1..5} {
 output 2d[attempt d20 vs POOL] named "2 vs (Crit [FOCUS]) [POOL]"
output 3d[attempt d20 vs POOL] named "3 vs (Crit [FOCUS]) [POOL]"
output 4d[attempt d20 vs POOL] named "4 vs (Crit [FOCUS]) [POOL]"
output 5d[attempt d20 vs POOL] named "5 vs (Crit [FOCUS]) [POOL]"
}

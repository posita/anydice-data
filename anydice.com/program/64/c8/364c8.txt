function: reroll DIE:d COUNT:n times{
  EXP: ([maximum of DIE] + 1) / 2
  result: [reroll DIE COUNT helper]
}

function: reroll N:n COUNT:n helper {
  if N > EXP { result: N }
  if COUNT > 1 { result: [reroll DIE (COUNT-1) helper] }
  result: DIE
}

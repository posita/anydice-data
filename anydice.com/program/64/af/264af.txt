\ 0 is a miss, 1 is a weak hit, 2 is a strong hit \

function: progress CROLL:s PROG:n {
  result: PROG > CROLL
}

function: action AROLL:n AMOD:n CROLL:s {
  result: [lowest of (AROLL + AMOD) and 10] > CROLL
}

loop X over {0..7} {
  output [action 1d6 X 2d10] named "Action 1d6+[X]"
}

loop X over {4..10} {
  output [progress 2d10 X] named "Progress [X]"
}

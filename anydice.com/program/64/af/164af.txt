PROF: 2
STAT: 16
DIE: 1d6
ATTACKS: 1

AC: {10..25}
BONUS: (STAT-10)/2
ATK: BONUS+PROF
DMG: DIE+BONUS

function: normal attack vs AC:n {
ROLL: 1d20
if {(ROLL+ATK)=20} {
DAMAGE: DMG+DIE
}
if {((ROLL+ATK)>=AC)&(ROLL!=1)&(ROLL!=20)} {
DAMAGE: DMG
}
else {DAMAGE: 0}
result: DAMAGE
}

loop N over AC {
DAMAGE: [normal attack vs N]
output DAMAGE named "AC [AC]"
}

function: diceamount ROLL:s {
result: #(ROLL)
}

function: ones ROLL:s {
  result: [count 1 in ROLL]
}

function: left ROLL:s {
  NUM: #(ROLL)
  ONES: [ones ROLL]
  result: NUM - ONES*2
}

function: calc ROLL:s {
  LEFT: [left ROLL]
  if LEFT >= 0 {
    result: {1..LEFT}@ROLL
  }
  result: 0
}

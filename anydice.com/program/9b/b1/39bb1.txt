function: diablo G:n and H:n and I:n {
 if G >= H & H >= I {result: 100*G+10*H+I}
 if G >= I & I >= H {result: 100*G+10*I+H}
 if H >= G & G >= I {result: 100*H+10*G+I}
 if H >= I & I >= G {result: 100*H+10*I+G}
 if I >= G & G >= H {result: 100*I+10*G+H}
 result: 100*I+10*H+G 
 }

function: minordiablo A:n and B:n {

 result: 10*B+A
 }

set "explode depth" to 3

function: nimble N:n of DIE:d {
  result: [nimble N of DIE plus 0]
}

function: nimble N:n of DIE:d plus P:n {
  PRIMARY: [miss [explode DIE]]
  if N = 1 { result: PRIMARY }
  result: [maybe PRIMARY (N-1) d DIE plus P]
}

function: maybe PRIMARY:n REST:d plus P:n {
  if PRIMARY = 0 { result: 0 }
  result: (PRIMARY + REST) + P
}

function: miss DIE:n {
  if DIE = 1 { result: 0 }
  result: DIE
}

output [nimble 2 of d4] named "2d4"
output [nimble 2 of d6 plus 2] named "2d6+2"

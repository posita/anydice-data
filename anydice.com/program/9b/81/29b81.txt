F: {-1,0,1}
X: 4dF
function: stressed Y:s {
 if [Y contains 1]=0 {
  result: Y-[explode [count -1 in Y]d{0,0,1}]
 }
 if [Y contains -1]=0 {
  result: Y+[explode [count 1 in Y]d{0,0,1}]
 }
result: Y
}
function: extreme M:n N:n{
 if (M+N)=0{
  result: [absolute M]
 }
 if [absolute M]>[absolute N]{
  result: M
 }
 result: N
}
output [stressed 4dF]
output 4dF
output [extreme 4dF 4dF]
output [extreme [stressed 4dF] [stressed 4dF]]

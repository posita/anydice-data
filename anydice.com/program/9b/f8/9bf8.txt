function: singleshot SUC:n DMG:n {
 SUCCESS: (RANGEBONUS+SINGLEBONUS+BS-SUC)/10
 if SUCCESS > 0 {result: [damage DMG]}
 result: 0
}

function: autoshot SUC:n DMG:n {
 SUCCESS: (RANGEBONUS+AUTOBONUS+BS-SUC)/10
 if SUCCESS > 0 {result: SUCCESSn[damage DMG]}
 result: 0
}

function: damage V:n {
 ROLL: (1d10 + V)-(TO + AR)
 result: [positive ROLL]
}

function: positive N:n {
if N > 0 { result: N}
result: 0
}

TO: 3
AR: 3
BS: 40
SINGLEBONUS: 10
SEMIBONUS: 10
AUTOBONUS: 20
RANGEBONUS: 30
TEST: 1d100
SUCCESS: (RANGEBONUS+AUTOBONUS+BS-TEST)/10
output [autoshot TEST 2] named "Autopistol"

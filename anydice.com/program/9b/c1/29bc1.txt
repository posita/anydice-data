
function: attack ROLL:s w ATBON ag AC save SVROLL:s ag BONUS {
    if AC < (ROLL+ATBON) {
        if (SVROLL+BONUS) < 13 {
        result: 1d6+1d4+ATBON
        }
        else
        { result: 1d6+ATBON }
    
    }
    else 
    { result: 0 }
}

function: weapon ROLL:s w ATBON ag AC {
    if AC < (ROLL+ATBON) {
        result: 1d10+ATBON
    }
    else 
    { result: 0 }
}

function: bigweapon ROLL:s w ATBON ag AC {
    if AC < (ROLL+ATBON) {
        result: 2d6+ATBON
    }
    else 
    { result: 0 }
}


function: atk thrice FIRST:s SECOND:s THIRD:s normal SWITCH ag ATKBON and ACC with SV:s {
    if SWITCH > 0 {
    result: ([weapon FIRST w ATKBON ag ACC] + [attack SECOND w ATKBON ag ACC save 20 ag 3] + [attack THIRD w ATKBON ag ACC save 20 ag 3])

    }
    else {
    result: ([attack FIRST w ATKBON ag ACC save SV ag 2] + [attack SECOND w ATKBON ag ACC save SV ag 2] + [attack THIRD w ATKBON ag ACC save SV ag 2])

    }

}

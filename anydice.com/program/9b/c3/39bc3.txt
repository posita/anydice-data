function: count successes in ROLL:s vs TN:n {
  result: [count {TN..6} in ROLL]
}
function: count duds in ROLL:s vs TN:n {
  result: [count {1..TN-1} in ROLL]
}
function: push ROLL:s vs TN:n {
  DUDS: [count duds in ROLL vs TN]
  result: [count successes in ROLL vs TN] + [count successes in DUDSd6 vs TN]
}

TN: 4
SKILL: 6
output [count successes in SKILLd6 vs TN] named "Normal"
output [push SKILLd6 vs TN] named "Pushed"

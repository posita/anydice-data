  loop DE over {4, 6, 8, 10, 12} {
   loop NB over {1..5} {
     loop INNE over {0, 1} {
        PLAGE: {SEUIL..12}
        LANCER: NB dDE
        MAX: {DE}
        SUCCES: [count LANCER in PLAGE]
        TEST: [count LANCER in MAX]
        if INNE=1 {SUCCES: SUCCES + TEST}
     
        output SUCCES named "[DE] >= [SEUIL] - [INNE]"
      }
    }
  }
}

ATTACKBONUS: 5
PROFICIENCY: 3
AC: 17

HITS: 6
ELDRITCHBLAST: 1d10+ATTACKBONUS
ELDRITCHBLASTCRIT: 2d10+ATTACKBONUS
HEX: 1d6
HEXCRIT: 2d6

MISS: 0

function: attack ROLL:n{
 if ROLL = 1 {
  result: 0
 }
 if ROLL = 20 {
   result: HITS*(ELDRITCHBLASTCRIT+HEXCRIT)
 }
 if ROLL + ATTACKBONUS + PROFICIENCY >= AC {
   result: HITS*(ELDRITCHBLAST+HEX+ELDRITCHBLAST+HEX)
 }
 result: MISS
}

output [attack 1d20] named "Eldritch Blast"

TN: 20
PASS: 1
AUTOPASS: 1
FAIL: 0
function: roll ROLL:n vs PT:n {
if ROLL = 12 { result: AUTOPASS }
if ROLL >= PT { result: PASS }
if ROLL < PT { result: FAIL }
}
loop N over {0..8} {
output [roll [lowest 1 of 2d12]+Nd6 vs TN] named "ILL [N]"
output [roll [highest 1 of 2d12]+Nd6 vs TN] named "FAV [N]"
output [roll d12+Nd6 vs TN] named "N [N]"
}

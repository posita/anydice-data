A: 1 \ number of dice in the attack roll \
B: 1 \ number of dice in the defense roll \
X: 3 \ difficulty to hit \
Y: 4 \ difficulty to dodge \

function: attack ATTACK:n{
 if ATTACK - X < 0 { result: 0 }
 result: ATTACK
}
function: defense DEFENSE:n{
 if DEFENSE - Y < 0 { result: 0 }
 result: DEFENSE
}
function: absolute NUMBER:n{
 if NUMBER <= 0 { result: 0 }
 result: NUMBER
}
function: custom ROLL:s {
 if 1@ROLL >= X+2 { result: 1@ROLL + (A-1)d[attack d6] }
 if 1@ROLL >= X { result: [absolute 1@ROLL + (A-1)d[attack d6] - Bd[defense d6]] }
result: 0
}
output [custom Ad6]

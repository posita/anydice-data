function: cain roll ROLL:s {
 SUCCESSES: 0
 loop N over {1..#ROLL}
 {
  if N@ROLL >= 4 {
   SUCCESSES: SUCCESSES + 1
  }
 }
 result: SUCCESSES
}

function: cain hard roll ROLL:s {
 SUCCESSES: 0
 loop N over {1..#ROLL}
 {
  if N@ROLL = 6 {
   SUCCESSES: SUCCESSES + 1
  }
 }
 result: SUCCESSES
}

loop R over {1..6} {
 output [cain roll Rd6] named "Straight [R]D"
}
loop H over {1..6} {
 output [cain hard roll Hd6] named "Hard [H]D"
}

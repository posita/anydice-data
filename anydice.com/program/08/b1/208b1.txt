SCALE: 5
DICE: d10
ATR: d{25..50}/10
ARM: d{0..SCALE}
ABS: SCALE+ATR+ARM
MOD: ATR-ABS

function: wounds from A:n {
 result: [highest of 0 and (A+MOD+(SCALE-1))/SCALE]
}

DMG_WEAK: [lowest 1 of 2d[explode DICE]]
DMG_NORMAL: [explode DICE]
DMG_STRONG: [highest 1 of 2d[explode DICE]]

output [wounds from DMG_WEAK] named "Dmg: Weak"
output [wounds from DMG_NORMAL] named "Dmg: Normal"
output [wounds from DMG_STRONG] named "Dmg: Strong"

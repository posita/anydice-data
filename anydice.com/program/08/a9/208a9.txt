
function: closest DICE:n plus RANGE:s to TARGET:n {
    CLOSEST: DICE + 1@RANGE
    loop N over RANGE {
        if [absolute DICE + N - TARGET] < [absolute CLOSEST - TARGET] {
            CLOSEST: DICE + N
        }
    }
    result: CLOSEST
}

loop TARGET over {-3..25} {
    output [closest 2d10 plus {-8..8} to TARGET] named "Closest to [TARGET] of 2d10 + -8 through +8 "

}

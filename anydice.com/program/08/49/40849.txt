AC: 16 \enemy ac\
DEXMOD: 2 \enemy dex mod\
STRMOD: 3 \enemy str mod\

PROF: 3 \pc prof\
MYCHAMOD: 5 \pc cha mod\
MYDEXMOD: 3 \pc dex mod\
DC: 8 + PROF + MYCHAMOD + 2 \difficulty for saving throws\
SPD: PROF + MYDEXMOD \bonus to hit dexterity\
SPC: PROF + MYCHAMOD + 2 \bonus to hit charisma spell\
CRIT: 20 \minimum to crit\


\### FUNCTIONS ###\
set "position order" to "lowest first"


function: atk ROLL:n h MOD:n DICE:d BONUS:n {
 if ROLL >= CRIT { result: 2dDICE+BONUS} 
 if ROLL = 1 { result: 0 }
 if ROLL + MOD >= AC { result: 1dDICE+BONUS}
 result: 0
}


function: atkc ROLL:n h MOD:n DICE:d BONUS:n {
 if ROLL >= 2 { result: 2dDICE+BONUS} 
 if ROLL = 1 { result: 0 }
 if ROLL + MOD >= AC { result: 1dDICE+BONUS}
 result: 0
}

function: absorb ROLL:n BONUS:n {
if ROLL > 0 {result: ROLL + BONUS}
result: 0
}

function: adh ROLL:n MOD:n dc DC:n DMG:n {
if ROLL+MOD >= DC { result : DMG/2}
result : DMG
}

function: adz ROLL:n MOD:n dc DC:n DMG:n {
if ROLL+MOD >= DC { result : 0}
result : DMG
}

function: rerollones NUM:n dice DICE:n {
 COUNT: 0
 loop P over {1..NUM} {
  COUNT: COUNT + [implode 1dDICE]
 }
 result: COUNT
}

function: implode N:n {
 if N = 1 { result: 1dDICE }
 result: N
}

function: empower DICE:s under N:n roll M:n {
 COUNT: 0
 EMPOWERED: 0
 loop P over {1..#DICE} {
  if EMPOWERED < MYCHAMOD & P@DICE < N {
  COUNT: COUNT + 1dM
  EMPOWERED: EMPOWERED+1
  } else {
  COUNT: COUNT + P@DICE
  }
 }
 result: COUNT
}

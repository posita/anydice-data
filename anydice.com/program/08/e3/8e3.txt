set "explode depth" to 1
function: icon N:s {
 if #N = 1 {
  result: [explode d6]
 }
 else {
  if 1@N = 6  {
   result: {1,2}@N
  }
  else {
   result: 1@N
  }
 }
}

output [icon 1d6] named "1 die"
output [icon 2d6] named "2 dice"
output [icon 3d6] named "3 dice"
output [icon 4d6] named "4 dice"

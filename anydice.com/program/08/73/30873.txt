
function: dice D:d mod DM:n{
 R: {#(D),[maximum of 1@D],DM}
 result: ((1@R)d{(1d(2@R)+3@R)})
}

function: sides of D:d {
 result: (#{D}+#(D)-1)/#(D)
}
function: modifier of D:d {
 result: [maximum of 1dD]/#(D)-[sides of D]
}

function: single dice D:d {
 result: 1d{(1d([sides of D]))+[modifier of D]}
}

function: subskill S:n weapon W:n  {
 if S >= W { result: [highest of W and 0] }
 result: 0
}
function: skill S:d weapon W:d {
 result: (#W)d([subskill 1@S weapon [single dice W]])
}

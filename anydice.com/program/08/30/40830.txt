function: bestof NB_HIGHEST:n NUMBER:n DICE:n {
  result: [highest NB_HIGHEST of NUMBERdDICE]
}

function: worstof NB_LOWEST:n NUMBER:n DICE:n {
  result: [lowest NB_LOWEST of NUMBERdDICE]
}


loop DIFFICULTY over {3..24}{
  BESTOF: 2
  TOTAL: 3
  DICE: 12
  output [bestof BESTOF TOTAL DICE] >= DIFFICULTY named "Best [BESTOF] of [TOTAL]d[DICE] - DC[DIFFICULTY]" 
}

  output [worstof 1 3 12] named "Lowest die"
  output [worstof 2 3 12] named "Lowest 2 die"
  output [bestof 1 3 12] named "Highest die"

  output 2d12

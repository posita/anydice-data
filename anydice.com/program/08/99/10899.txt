MIS:0
HIT:1
CRT:2

ATTACK: 4
HITDMG: 2d8
CRTDMG: 2d8*2
MISDMG: 0

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1  { result: MIS }
 if ROLL = 20 { result: CRT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MIS
}

function: damage for ATK:n {
 if ATK = CRT { result: CRTDMG }
 if ATK = HIT { result: HITDMG }
 result: MISDMG
}

loop AC over {10..21} {
 output [damage for [attack d20 vs AC]] named "[AC]"
}

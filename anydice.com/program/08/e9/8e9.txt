function: A:n compare VALUES:s in SEQUENCE:s {
  COUNT: 0
  loop P over {1..#VALUES} {
    loop Q over {A..(A+2)} {
      COUNT: COUNT + (P@VALUES = Q@SEQUENCE)
    }
  }
  result: COUNT
}

loop FV over {1..5} {        \ ~~~~~ HUVUDPROGRAMMET ~~~~~ \

  output [count {1..FV} in DICE] named "~~~~~ [DICE] mot egenskapsvärde [FV] ~~~~~"

  output [(#MODIFICATION-2) compare {1..FV} in MODIFICATION] named "De tre bästa av [MODIFICATION] mot egenskapsvärde [FV]"

  output [1 compare {1..FV} in MODIFICATION] named "De tre sämsta av [MODIFICATION] mot egenskapsvärde [FV]"
}

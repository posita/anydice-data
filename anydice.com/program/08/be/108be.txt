function: giveme ROLL:s with BONUS:s
{
 if ROLL = 1 & ([BONUS contains 1] | [BONUS contains 2] | [BONUS contains 3] ) { result: 6 }
 if ROLL = 2 & ([BONUS contains 4] | [BONUS contains 5] ) { result: 5 }
 if ROLL = 3 & [BONUS contains 6] { result: 4 }
 result: ROLL
}

loop N over {0..5} {
 output [giveme 1d6 with Nd6]
}

function: largest width of DICE:s {
  MAX: 0
  loop X over {1..1@DICE} {
    COUNT: X = DICE
    if COUNT > 1 & COUNT > MAX { MAX: COUNT }
  }
  result: MAX
}

  M: 0
  loop N over {X..1@DICE} {
    if [count N in DICE] >= 2 {
      M: 1
    }
  }
  result: M
}

function: bitd roll NUM:n{
    if NUM = 0
      {result: 2@2d6}
    if NUM = 1
      {result: d6}
    else{
      result:[bitd help NUMd6]
}
}

function: bitd help DICE:s{
if [count 6 in DICE] > 1
{result: 7}
else
{result: 1@DICE}
}

function: bitd interpret NUM:n{
if NUM <4
{result: 0}
if NUM <6
{result: 1}
if NUM =6
{result: 2}
if NUM >6
{result: 3}
}

loop N over {0..5} { 
output [bitd interpret[bitd roll N]] named "[N] dice"
}

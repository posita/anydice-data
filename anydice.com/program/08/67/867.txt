set "maximum function depth" to 3

function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 7
 if (ROLL = 1) { SUCCESSES: SUCCESSES - (ROLL = 1) }
 if (ROLL = 10) { SUCCESSES: SUCCESSES + [evaluate 1d10]}
 result: [filter SUCCESSES]
}
function: filter N:n {
 if N < 0 { result: 0}
 result: N
}
loop POOL over {1..7} {
 output [evaluate POOLd10] named "oWoD roll of [POOL]d10"
}

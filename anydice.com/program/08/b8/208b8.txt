SCALE: 5
DICE: d10
ATR: d{25..50}/10
ABS: SCALE+ATR
WPN: d{0..SCALE}
ARM: d{0..SCALE}

function: wounds from A:n {
 result: [highest of 0 and (A+(SCALE-1))/SCALE]
}

DMG_WEAK: [lowest 1 of 2d[explode DICE]]
DMG_NORMAL: [explode DICE]
DMG_STRONG: [highest 1 of 2d[explode DICE]]

output [wounds from DMG_WEAK+(ATR+WPN)-(ABS+ARM)] named "Dmg: Weak"
output [wounds from DMG_NORMAL+(ATR+WPN)-(ABS+ARM)] named "Dmg: Normal"
output [wounds from DMG_STRONG+(ATR+WPN)-(ABS+ARM)] named "Dmg: Strong"

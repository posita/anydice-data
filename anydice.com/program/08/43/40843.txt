\ Unseen Menace: Advantage, Crit Range +1, 1d10+1+STR\
\ Swordmaster Gloves: +1 attack \
\ DJ Gauntlets: +1d4 Necrotic \
\ Improved Critical Hit: Crit Range +1 \
\ GWF: Reroll 1s and 2s \

STR: 4 \ Level 4 to 
PROF: 4
function: ROLL:n attack AC:n BONUS:n {
  if ROLL>= 19 {
    \ crit \
    result: 2d10+1+STR
  }
  if ROLL+BONUS+STR+PROF+1>= AC {
    result: 1
  }
  result: 0
}
loop AC over {15..16} {
  output [1@2d20 attack AC 0]
}

loop TRAIT over {0..6} {
  loop SKILL over {0..6} {
    if TRAIT = 0 {
      output 0 named "Skill [SKILL] v Trait [TRAIT]"
    }
    else {
      if SKILL = 0 {
        output d{1, 0:5} named "Skill [SKILL] v Trait [TRAIT]"
      }
      else {
        output SKILLd{2, 1:(TRAIT-1), 0:(6-(TRAIT-1))} named "Skill [SKILL] v Trait [TRAIT]"
      }
    }
  }
}

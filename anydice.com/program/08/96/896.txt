DEPTH: 3 \SETS the max number of chained exploding dice permitted\

function: SEQ:s COUNT:n DEPTH:n genseq{
  if DEPTH=0 {
    result: {SEQ, COUNT*10}
  }
  DEPTH: DEPTH-1
  COUNT: COUNT+1
  TEMPSEQ: SEQ
  loop I over{1..9}{
    SEQ: {SEQ,TEMPSEQ}
  }
  SEQ: {SEQ, (COUNT*10+1)..(COUNT*10+9)}
  result: [SEQ COUNT DEPTH genseq]
}

function: ROLLED:n k KEPT:n {
  SEQ: [{1..9} 1 DEPTH genseq]
  result: {1..KEPT}@ROLLEDdSEQ
}

output [4k3] = [3k2]

set "maximum function depth" to 10

function: roll N:n of DIE:d {
 result: [roll N of DIE as DIE plus 0 reset 1]
}

function: roll N:n of DIE:d as X:n plus TOTAL:n reset RESET:n {
 if N < 1 { if RESET != 0 { result: TOTAL } else { result: 0 } }
 result: [roll (N-1) of DIE as DIE plus (TOTAL+X) reset (RESET&(X!=1))]
}

loop T on {1..6} {
 output [roll 2 of d6]
}

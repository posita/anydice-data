function: roll ROLL:s vs TARGET:n {
  result: (ROLL <= TARGET) + (ROLL = 1)
}

loop TRAIT over {0..6} {
  loop SKILL over {0..6} {
    if SKILL = 0 {
      DICE: 1d6
      TARGET: 0
    } else {
      DICE: SKILLd6
      TARGET: TRAIT
    }
    output [roll DICE vs TARGET] named "skill [SKILL], trait [TRAIT]"
  }
}

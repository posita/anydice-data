POWER: 40
SKILL: 5
TARGET: 140

function: popejonsmagicsystem P:n S:n T:n {
if P = T {
  \Got the right number\
  result: T
} else if P > T {
  \Too high, try taking the skill from power\
  if P - S < T {
    \target was within skill roll of power\
    result: T
  } else {
    \power was too high, take all of skill away from power to get as close as possible\
    result: P - S
  }
} else {
  \Too low, try adding skill to power\
  if P + S > T {
    \Target was within skill roll of power\
    result: T
  } else {
    \Power was too low, add all of power to skill\
    result: P + S
  }
}
}
output [popejonsmagicsystem POWERd6 SKILLd6 TARGET]

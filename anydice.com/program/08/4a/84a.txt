STEPS: {4,6,8,10,12}
AP: 1
BP: 1
CP: 2


SUM: 1

A: 1d4
B: 1d6
C: 2d6

function: highest X:n of A:s and B:s {
 result: {1..X}@[sort {A, B}]
}

function: margin of top X:n of A:s and B:s versus C:s {
 result: [highest X of A and B] - 4\{1..X}@C\
}

loop D over STEPS {
 A: 1dD
 output [margin of top SUM of A and B versus C] named "[SUM]@([A],[B]) - 4\[SUM]@([C])\"
}

\loop S over {0..5} {
 output 4d{-1..1} + S named "4dF + [S]"
}\

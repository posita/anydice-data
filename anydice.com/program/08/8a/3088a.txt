function: highest N:n of A:s B:s C:s {
    result: {1..N}@[sort {A, B, C}]
}

ATTR: d8
SKIL: d8
DISC: 0
ASSTA: 0
ASSTB: 0
ASSTC: 0
ASSTT: [highest 1 of ASSTA ASSTB ASSTC]
PWRR: [highest 2 of ATTR SKIL ASSTT]

COND: 0
SPEC: 0
MODF: (SPEC + COND)

ADVT: 0
DSVT: 0
ACCB: (ADVT-DSVT)d6

output [highest 1 of ACCB] named "Accuracy Bonus"

output PWRR named "Power Roll"
output PWRR + MODF named "Final Power"

output (PWRR + MODF) + [highest 1 of ACCB] named "Accuracy Roll"

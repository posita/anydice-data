function: range A:n from X:n to Y:n {
if A < X {result: 0}
if A > Y {result: 0}
result: 1
}

function: thresholds D:n {
if D <= 4 {result: -2}
if [range D from 5 to 7] {result: 0}
if [range D from 8 to 9] {result: 1}
if [range D from 10 to 11] {result: 2}
if D>=12 {result: 3}
}

D: 2d6
E: 1d6+1d8+1
F: 1d4+1d6
G: 2d8+2


output [thresholds D]+[thresholds E]+[thresholds F]+[thresholds G] >= 4
output D >= 8
output E >= 8
output F >= 8
output G >= 8

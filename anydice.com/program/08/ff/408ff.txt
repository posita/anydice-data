function: N:n dice F:n faces success on S:n {
  R: 0
  D: [absolute N]
  if N < 0 {R: [lowest  1 of DdF]}
  if N > 0 {R: [highest 1 of DdF]}
  result: R >= S
}

loop N over {-5..-2,1..5}{
  F: 6
  S: 4
  output [N dice F faces success on S] named "Highest/lowest of [N]d[F] ≥ [S]"
}

loop N over {-4..4}{
  output 4d{-1..1} + N >= 0 named "4dF ≥ 0"
}

function: ws A:n attacks ws B {
  if A > B {
    if A - B > 1 {
      result: [count {2..6} in 1d6]
    }
    else {
      result: [count {3..6} in 1d6]
    }
  }
  else {
    if A = B {
      result: [count {4..6} in 1d6]
    }
    else {
      if B > A {
        if B - A > 1 {
          result: [count {5..6} in 1d6]
        }
        else {
          result: [count {4..6} in 1d6]
        }
      }
    }
  }
}

output [ws 5 attacks ws 4]

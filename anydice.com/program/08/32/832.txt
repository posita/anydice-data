function: splode DIE:d {
  MAX: [maximum of DIE]
  result: [splode DIE helper]
}

function: splode N:n helper {
  if N = MAX {
    result: N + [splode DIE helper]
  }
  result: N
}

function: skill check with N:n dice {
  result: [lowest of 10 and [highest of -1 and Nd[splode d{-1,0,0,1,1,2}]]]
}

loop N over {1..10} {
  output [skill check with N dice] named "Skill check with [N] dice"
}

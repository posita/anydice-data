
function: replace OLD in SEQ:s with NEW {
  NEWSEQ: {}
  loop I over {1 .. #SEQ} {
    if I@SEQ = OLD { NEWSEQ: {NEWSEQ, NEW} }
    else {NEWSEQ: {NEWSEQ, I@SEQ} }
  }
  result: NEWSEQ
}

function: push REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: dicetype N:n {
  if N = 1 { result: {-1, 0:4, 1} }
  if N = 2 { result: {-1, 0:4, 1:3} }
  if N = 3 { result: {-1, 0:4, 1:4, 2} }
  if N = 4 { result: {-1, 0:4, 1:4, 2:3} }
}

\ when specifying D: \
\   1 is a success (2 = 2 successes :) \
\   0 a miss that can be rerolled \
\  -1 a miss that can NOT be rerolled \


loop CI over {1..4}{
  VALI: (CI - 1)*2 + 6
  DI: [dicetype CI]
  BI: [replace -1 in dDI with 0]
  PI: [replace -1 in [push {0} on dDI to dDI] with 0]
  loop CII over {CI..4}{
    VALII: (CII - 1)*2 + 6
    DII: [dicetype CII]
    BII: [replace -1 in dDII with 0]
    PII: [replace -1 in [push {0} on dDII to dDII] with 0]
   \ output dBI + dBII named "Base d[VALI] + d[VALII]"\
    output dBI + dBII >= 1 named "Pass/Fail Base d[VALI] + d[VALII]"
   \ output dPI + dPII named "Push d[VALI] + d[VALII]"\
    \output dPI + dPII >= 1 named "Pass/Fail Push d[VALI] + d[VALII]"\
  }
}

function: check A:s under B:n {
  if 1@A = 1  & 2@A = 1 & 3@A = 1 {result: 1}
  if 1@A = 2  & 2@A = 2 & 3@A = 2 {result: 1}
  if 1@A = 5  & 2@A = 5 & 3@A = 5 {result: 0}
  if 1@A = 6  & 2@A = 6 & 3@A = 6 {result: 0}
  X: [sort A]
  result: (2@A + 3@A) <= B
}

loop B over {1..9}{
 output [check 3d6 under B] named "3d6 Fudge under [B]"
}

loop B over {-4..4}{
 output 2d6 + B >= 7 named "2d6 + [B] ≥ 7"
}

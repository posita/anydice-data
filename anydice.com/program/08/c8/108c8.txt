set "maximum function depth" to 5

EDGE: 3

function: reroll X:n {
 if X < 5 { result: 1d6 > 4 }
 result: 1
}

function: explode X:n {
 if X = 6 { result: 1 + [explode d6] }
 if X = 5 { result: 1}
 result: 0
}

loop N over {3, 5, 7, 10, 15, 20, 30} {
 output Nd[reroll d6] named "reroll [N]d6"
 output (N+EDGE)d[explode d6] named "explode ([N]+[EDGE])d6"
}

\ we're assuming here that the character has an Edge of 3 \
\  feel free to change EDGE at the top if this is untrue  \

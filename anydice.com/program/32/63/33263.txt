A: 0 \Amount of Dice\
B: 0 \Gear Bonus\
C: 8 \Size of Dice\

function: roll DICE:s {
          if A = 0 {
           result: [lowest 1 of 2dC] + B
          }
          Z: A@DICE + B
          if Z >= 1@DICE {
           result: Z}
          else {
           result: 1@DICE
}
}

output [roll Ad8]

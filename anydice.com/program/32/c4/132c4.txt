function: ltdc R B {
  result : Rd{0, 0, 2} + Bd{0, 1, 1} + [lowest of R and B]
}
function: seuil N:n {
  loop I over {0,1,3,5,7,9,12,15,18,21,25,30} {
    if N >= I {S : I}
  }
  result : S
}
\output [ltdc 12 6] named "resultat" \
output [seuil [ltdc 4 4]] named "seuil"

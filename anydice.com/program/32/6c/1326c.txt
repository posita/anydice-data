function: roll X:n skill Y power Z{
A:0
if X < Y+1 {A: A+1}
if X < Z+1 {A: A+1}
result: A
}
\
PDICE:  1d{0,1,1,1,1,1}
SDICE:  1d6
SKILL: 0
POWER: 1
SDIF: -4
PDIF: 1
output [highest of ((1d[roll d6 skill SKILL power POWER ]+POWERdPDICE)
          -(1d[roll d6 skill SKILL-SDIF power POWER-PDIF ]+(POWER-PDIF)dPDICE)) and -1]
\


loop POWER over {0..5} { 
loop SKILL over {0..5} { 
PDICE:  1d{0,1,1,1,1,1}
SDICE:  1d6
output POWERdPDICE+ 1d[roll d6 skill SKILL power POWER ]+0dPDICE  named "Skill: [SKILL], Power: [POWER] " }}
\
loop SUCCESS over {1..7} { 
loop POWER over {0..5} { 
loop SKILL over {0..5} {
PDICE:  1d{0,1,1,1,1,1}
SDICE:  1d6 
output (((POWER d PDICE)  + 1d[roll d6 skill SKILL power POWER]) > (SUCCESS - 1)) named "Skill: [SKILL], Power: [POWER], DC [SUCCESS] " }}}

loop POWER over {0..5} { 
PDICE:  1d{0,1,1,1,1,1}
output (((POWER+1) d PDICE) + 1d[roll d6 skill 0 power (POWER+1)])-(((POWER) d PDICE) + 1d[roll d6 skill 3 power (POWER)])  named "Power: [POWER] " 

}
\

function: fighter ROLL:n vs AC{
 if ROLL = 1 {result: 0}
 if ROLL >= CC {result: CRIT}
 if ROLL + MOD + P >= AC {result: DAMAGE}
 result: 0
}

function: rogue ROLL:n and RR:n vs AC{
 if ROLL = 1 {
  if RR = 1 {result:0}
  if RR = 20 {result: CRIT+CSA}
  if RR + MOD + P >= AC {result: DAMAGE + SA}
  result: 0
 }
 if ROLL = 20 {
  if RR = 1 {result: CRIT+CSA}
  if RR = 20 {result: CRIT+CSA+CRIT}
  if RR + MOD + P >= AC {result: CRIT+CSA+DAMAGE}
  result: CRIT+CSA
 }
 if ROLL + MOD + P >= AC {
  if RR = 1 {result: DAMAGE+SA}
  if RR = 20 {result: DAMAGE+SA+CRIT}
  if RR + MOD + P >= AC {result: DAMAGE+SA+DAMAGE}
  result: DAMAGE + SA
 }
 if ROLL + MOD + P < AC {
  if RR = 1 {result: 0}
  if RR = 20 {result: CRIT+CSA}
  if RR + MOD + P >= AC {result: DAMAGE+SA}
  result: 0
 }
}

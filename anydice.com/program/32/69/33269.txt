function: roll DICE:s and add bonus BONUS {
  result: {1..DICE} + BONUS
}

function: compare ROLL to TARGET {
  if ROLL >= TARGET + 3 { result: "critical success" }
  else if ROLL >= TARGET { result: "success" }
  else if ROLL >= TARGET - 3 { result: "partial success" }
  else { result: "failure" }
}

loop DICE over {1..10} {
  output [compare [roll d8 and add bonus 1] to 5] named "[DICE]d8+1 vs 5"
}



A: 3 \Amount of Dice\
B: 0 \Gear Bonus\
C: 8 \Size of Dice\
D: 5 \Value for Success\
E: 8 \Value for Critical Success\
set "maximum function depth" to 2

function: 
    roll ROLL:s {
        if A@ROLL + B >= 1@ROLL {
            result: A@ROLL + B
        }
        else {
            result: 1@ROLL
        }
}
	  

function: 
    rot DICE:s {        
        if 1@DICE >= E {
            result: 2 + (2@DICE = E)
        }
        else {
            result: 1@DICE >= D
        }
}

output [rot AdC] named "Chances of Success (No Gear) (0 fails, 1 succeeds with hindrance, 2 succeeds, 3 succeeds critically)"
output [rot[roll AdC]] named "Chance of Success (With +[B] Gear)"
output [roll AdC] named "Actual result"

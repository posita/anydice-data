MISS: 0
HIT: 1
CRITICALHIT: 2

CRIT_ROLL: 20

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= CRIT_ROLL { result: CRITICALHIT }
 if ROLL + TO_HIT >= DEFENSE { result: HIT }
 result: MISS
}

TO_HIT: 5 + 3
loop AC over {13..20} {
 output [attack 1d20 vs AC] named "roll vs [AC]"
 output [attack [highest 1 of 2d20] vs AC] named "Adv vs [AC]"
}

\ !! This looks best in the Transposed view !! \
DIE: d20 \ Die type \
COMP: 1 \ The pool we're comparing to \
COMPMOD: 1 \ mod 1 \
B: 5 \ mod 2 \
function: wintielose A:n B:n {
 if A > B { result: 1 } \ A wins \
 if A < B { result: -1} \ B wins \
 result: 0 \ Neither wins: tie \
}

loop N over {1..10} {
 output [wintielose COMPdDIE+COMPMOD COMPdDIE+N] named "[COMP][DIE]+[COMPMOD] v [COMP][DIE]+[N]"
}

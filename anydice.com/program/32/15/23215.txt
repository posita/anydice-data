function: ROLLA:n beatsh ROLLB:n{
 result: (ROLLA >= ROLLB)
}

function: ROLLA:n beatsl ROLLB:n{
 result: (ROLLA > ROLLB)
}

function: ROLLA:n beatse ROLLB:n ELFWEAPON:n{
 result: (ROLLA > ROLLB) | ((ROLLA = ROLLB)*[count {4..7} in d6+ELFWEAPON])
}

function: FIGHTA:n ROLLA:n ELFWEAPONA:n fights FIGHTB:n ROLLB:n ELFWEAPONB:n{
 X: 0
 if FIGHTA > FIGHTB {
    X: [ROLLA beatsh ROLLB]
 } else if FIGHTA < FIGHTB {
    X: [ROLLA beatsl ROLLB]
 }else if FIGHTA = FIGHTB {
    X: [ROLLA beatse ROLLB ELFWEAPONA-ELFWEAPONB]
 }
 result: X
}

output [10 [highest 1 of 4d6] 0 fights 4 [highest 1 of 10d6] 0]

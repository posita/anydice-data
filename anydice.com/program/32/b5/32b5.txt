function: attack ROLL:n vs DC for DAMAGE:n {
  if ROLL >= DC { result: DAMAGE}
  else {result: 0}
}

function: attack ROLL:d vs DC:n for DAMAGE:d on TARGETS:n targets {
  TDAMAGE: 0
  loop N over {1..TARGETS} { 
    TDAMAGE: TDAMAGE+ [attack ROLL vs DC for DAMAGE]
  }
  result: TDAMAGE
}

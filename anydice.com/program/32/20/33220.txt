function: mode ROLL:s {
 NUM_DICE: #(ROLL)
 MAX: 0
 MODE: 0
 loop I over {1..6} {
  COUNT: 0
  loop J over {1..NUM_DICE} {
   if J@ROLL = I {
    COUNT: COUNT + 1
   }
  }
  if COUNT = MAX {
   MODE: 0
  }
  else {
   if COUNT > MAX {
    MAX: COUNT
    MODE: I
   }
  }
 }
 result: MODE
}

output [mode 4d6]

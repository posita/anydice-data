AC:27
ATK:18
NORMAL: 3d6+6+5
CRITBONUS: 2d6+6+10+5
VITAL: 2d6


function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

MAINHAND:[attack [highest 1 of 1d20] plus ATK-2 vs AC for NORMAL+VITAL crit CRITBONUS on 15]
OFFHAND:[attack [highest 1 of 1d20] plus ATK-2 vs AC for NORMAL-1 crit CRITBONUS-1 on 15]
ATTACK:[attack [highest 1 of 1d20] plus ATK vs AC for NORMAL crit CRITBONUS on 15]

function: ripplingblowtotal DAMAGERESULT:n {
    if DAMAGERESULT > 0 {
        result: DAMAGERESULT + ATTACK
    }
    result: DAMAGERESULT 
}

output
[ripplingblowtotal MAINHAND] +
OFFHAND

function: ATTACK:n hit AC:n {
  if ATTACK >= AC {
    result: 1
  }
  else {
    result: 0
  }
}

function: calculate damage ATTACK:n MODIFIER:n DMG_DICE:n DMG_MODIFIER AC:n {
  if ATTACK = 1 {
    result: 0
  }
  else if ATTACK = 20 {
    result: DMG_DICE*2 + DMG_MODIFIER
  }
  else if [(ATTACK+MODIFIER) hit AC] {
    result: DMG_DICE + DMG_MODIFIER
  }
  else {
    result: 0
  }
}

loop AC over {10..30} {
  ATTACK_DICE: 1d20

  HALBERD_HIT_MODIFIER:10
  HALBERD_DMG_DICE: 1d10
  HALBERD_DMG_MODIFIER: 6
  POLEARM_DMG_DICE: 1d4

  GWM_HIT_MODIFIER: HALBERD_HIT_MODIFIER - 5
  GWM_DMG_MODIFIER: HALBERD_DMG_MODIFIER + 10

  ELDRITCH_HIT_MODIFIER:9
  ELDRITCH_DMG_DICE: 1d10
  ELDRITCH_DMG_MODIFIER: 5

  HALBERD_TARGET_DMG: [calculate damage ATTACK_DICE HALBERD_HIT_MODIFIER HALBERD_DMG_DICE HALBERD_DMG_MODIFIER AC]
  HALBERD_TARGET_DMG: HALBERD_TARGET_DMG + [calculate damage ATTACK_DICE HALBERD_HIT_MODIFIER HALBERD_DMG_DICE HALBERD_DMG_MODIFIER AC]
  HALBERD_TARGET_DMG: HALBERD_TARGET_DMG + [calculate damage ATTACK_DICE HALBERD_HIT_MODIFIER POLEARM_DMG_DICE HALBERD_DMG_MODIFIER AC]

  GWM_TARGET_DMG: [calculate damage ATTACK_DICE GWM_HIT_MODIFIER HALBERD_DMG_DICE GWM_DMG_MODIFIER AC]
  GWM_TARGET_DMG: GWM_TARGET_DMG + [calculate damage ATTACK_DICE GWM_HIT_MODIFIER HALBERD_DMG_DICE GWM_DMG_MODIFIER AC]
  GWM_TARGET_DMG: GWM_TARGET_DMG + [calculate damage ATTACK_DICE GWM_HIT_MODIFIER POLEARM_DMG_DICE GWM_DMG_MODIFIER AC]

  ELDRITCH_TARGET_DMG: [calculate damage ATTACK_DICE ELDRITCH_HIT_MODIFIER ELDRITCH_DMG_DICE ELDRITCH_DMG_MODIFIER AC]
  ELDRITCH_TARGET_DMG: ELDRITCH_TARGET_DMG + [calculate damage ATTACK_DICE ELDRITCH_HIT_MODIFIER ELDRITCH_DMG_DICE ELDRITCH_DMG_MODIFIER AC]
  ELDRITCH_TARGET_DMG: ELDRITCH_TARGET_DMG + [calculate damage ATTACK_DICE ELDRITCH_HIT_MODIFIER ELDRITCH_DMG_DICE ELDRITCH_DMG_MODIFIER AC]


  output HALBERD_TARGET_DMG named "H:[AC]"
  output GWM_TARGET_DMG named "GWM:[AC]"
  output ELDRITCH_TARGET_DMG named "EB:[AC]"
}

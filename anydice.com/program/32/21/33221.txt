function: mode ROLL:s {
 NUM_DICE: #ROLL
 NUM_SIDES: [maximum of 1@ROLL]
 MAX: 0
 MODE: 0
 loop I over {1..NUM_SIDES} {
  COUNT: 0
  loop J over {1..NUM_DICE} {
   if J@ROLL = I {
    COUNT: COUNT + 1
   }
  }
  if COUNT = MAX {
   MODE: 0
  }
  else {
   if COUNT > MAX {
    MAX: COUNT
    MODE: I
   }
  }
 }
 result: MODE
}

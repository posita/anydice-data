function: snake_attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n save SAVE:n for PSN:d {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        r: DMG + CRIT_DMG
    }
    elseif ROLL + BONUS >= AC {
        r: DMG
    }

    if SAVE < 11 {
        r: r + PSN
    }
    else{
    r: r + (PSN/2)
    }
    result: r
}

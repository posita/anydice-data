function: tinkertop ROLL:n {
  \ return the number of times damage is applied on an attack roll of ROLL \
  if ROLL <= 1 { result: 0 }     \ natural 1 is always a miss \
  if ROLL >= 20 { result: 2 }  \ critical hit deals damage damage twice \
  result: ROLL + NOTTDEX +1 >= AC
}

function: crossbowexpert N:n times {
  result: [roll [tinkertop d20] to crossbowexpert N times]
}
function: roll FIRSTHIT:n to crossbowexpert N:n times {
  if N <= 0 {
    result: 0
  }
  if FIRSTHIT > 0 {
    TTOP: FIRSTHIT d (TINKERTOP + NOTTSNEAK)
    loop I over {2..N} {
      TTOP: TTOP + [tinkertop d20] d TINKERTOP +1
    }
    result: TTOP
  } else {
    result: [roll [tinkertop d20] to crossbowexpert N-1 times]
  }
}

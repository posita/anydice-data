function: ACTION:s vs CHALLENGE:s min STAKES:s {
  if 1@CHALLENGE < 19 {
   if 1@CHALLENGE < STAKES {HIGH: STAKES} else {HIGH: 1@CHALLENGE}
  } else {HIGH: 0}
  if 2@CHALLENGE < 19 {
   if 2@CHALLENGE < STAKES {LOW: STAKES} else {LOW: 2@CHALLENGE}
  } else {LOW: 0}
 
}

loop MOD over {0..6}{
 loop S over {4, 6, 8}{
  output [(1d12 + MOD) vs 2d20 min 1dS] named "+[MOD] vs [S]"
}}

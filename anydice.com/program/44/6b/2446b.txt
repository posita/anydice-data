function: reroll R:n under N:n {
   if R < N { result: 2d6 } else {result: R}
}

function: rerollone F:n S:n under N:n {
   if F+S < N {
      if F<S { result: 1d6+S } else {result: 1d6+F}
   } else {result: F+S}
}

function: combo F:n S:n P:n Q:n under N:n {
   if F+S < N {
      if P+Q< N { 
         if P<Q { result: 1d6+P } else {result: 1d6+Q}
      } else {result: P+Q}
   } else {result: F+S}
}

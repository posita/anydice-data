ABILITY: 3
PROF: 2
TARGET_AC: 15
GSWD: 2d6
WHIP: 2d4
LSWD: 1d6



function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT on CRIT_RANGE:n {
  if ROLL = 1 {
    result: 0
  }
  if ROLL >= CRIT_RANGE {
    result: DMG + CRIT
  }
  if ROLL + BONUS >= AC {
    result: DMG
  }
  result: 0
}

ROLL: 1d20

TARGET_AC: 10
output [attack ROLL plus ABILITY + PROF vs TARGET_AC for WHIP crit WHIP on 20] named "Fire Whip vs AC10"
output [attack ROLL plus ABILITY + PROF vs TARGET_AC for GSWD crit GSWD on 20] named "+0 Greatsword vs AC10"
output [attack ROLL plus ABILITY + PROF +1 vs TARGET_AC for GSWD+1 crit GSWD+1 on 20] named "+1 Greatsword vs AC10"
\output [attack ROLL plus ABILITY + PROF vs TARGET_AC for LSWD crit LSWD on 20] named "+0 Longsword vs AC10"\

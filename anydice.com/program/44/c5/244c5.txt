function: VALUE:s min MIN {
if VALUE < MIN {result: MIN}
else {result: VALUE}
}

function: dw roll ROLL:s vs SKILL:n for DDICE:n damagedice plus ADDITION:n max MAX:n each TYPE:n {
SUCCESSES: [((SKILL-ROLL)/10)/TYPE min 0]
if SUCCESSES>MAX {ATTACKS:MAX} else {ATTACKS:SUCCESSES}

result: (ATTACKS*DDICE)d10 + (ATTACKS*DDICE)*ADDITION

}

output 1d([dw roll 1d100 vs 48 for 1 damagedice plus 19 max 2 each 1]) named "now: relic blade (pen 9)"
output 1d([dw roll 1d100 vs 71 for 1 damagedice plus 9 max 3 each 2]) named "now: bolt pistol (pen 4)"

output 1d([dw roll 1d100 vs 68 for 1 damagedice plus 19 max 2 each 1]) named "future: relic blade skilled (pen 9)"
output 2d([dw roll 1d100 vs 71 for 1 damagedice plus 9 max 3 each 2]) named "future: double heavy bolt pistols (pen 5)"

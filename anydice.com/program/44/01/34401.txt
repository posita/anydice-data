\ Implementation details \
MISS: 0
NORMAL_HIT: 1
CRITICAL_HIT: 2

function: hit with BONUS:n vs AC:n rolling ROLL:n {
   if ROLL >= CRITICAL_THRESHOLD { result: CRITICAL_HIT }

   if ROLL = 1 { result: MISS }

   if ROLL + BONUS - AC >= 0 { result: NORMAL_HIT }

   result: MISS
}


function: attack with BONUS:n vs AC:n rolling ROLL:n with DICE:d and MOD:n {
   HIT: [ hit with BONUS vs AC rolling ROLL ]

   if HIT = MISS { result: 0 }
   
   if HIT = CRITICAL_HIT { result: DICE + DICE + MOD }
   
   result: DICE + MOD
}
\ End implementation details \

PROFICIENCY: 2
CRITICAL_THRESHOLD: 20

NORMAL_ROLL: 1d20
ADVANTAGE: [ highest 1 of 2d20 ]

LONG: 1d8
GREAT: 2d6
SPEAR: 1d8
PAM: 1d4

loop TARGET_AC over { 12..22 } {

   STR_MOD: 4
   DAMAGE_BONUS: STR_MOD + 2
   HIT_BONUS: STR_MOD + PROFICIENCY 
   output [ attack with HIT_BONUS vs TARGET_AC rolling ADVANTAGE with LONG and DAMAGE_BONUS ] named "Longsword vs AC[TARGET_AC]"

   STR_MOD: 5
   DAMAGE_BONUS: STR_MOD + 2
   HIT_BONUS: STR_MOD + PROFICIENCY 
   output [ attack with HIT_BONUS vs TARGET_AC rolling ADVANTAGE with LONG and DAMAGE_BONUS ] named "Longsword w/ ASI vs AC[TARGET_AC]"

   STR_MOD: 4
   DAMAGE_BONUS: STR_MOD + 2 + 10
   HIT_BONUS: STR_MOD + PROFICIENCY - 5
   output [ attack with HIT_BONUS vs TARGET_AC rolling ADVANTAGE with GREAT and DAMAGE_BONUS ] named "Greatsword w/ GWM vs AC[TARGET_AC]"

   STR_MOD: 4
   DAMAGE_BONUS: STR_MOD + 2
   HIT_BONUS: STR_MOD + PROFICIENCY 
   output [ attack with HIT_BONUS vs TARGET_AC rolling ADVANTAGE with SPEAR and DAMAGE_BONUS ] + [ attack with HIT_BONUS vs TARGET_AC rolling ADVANTAGE with PAM and DAMAGE_BONUS ]  named "Polearm Master vs AC[TARGET_AC]"

   \ Group the results prettier. \
   output {0} named ""
}

set "maximum function depth" to 30
TARGET: 10
NUM_OF_ROLLS: 0
BONUS: 1
function: traintarget of CT:n with roll R:n rollnumber N:n {
if R>=CT {result: N}
else {
N: N + 1
CT: CT - 1d3
result: [traintarget of CT with roll R rollnumber N] 
}
}

output [traintarget of TARGET with roll (1d10+BONUS) rollnumber NUM_OF_ROLLS] named "Number of Rolls to Reach Target"

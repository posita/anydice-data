function: damage from AV:s attack with DAM:n damage against DEF:n defense {
 DAMAGE: 0
 ATTACK: 0
 
 loop P over {1..#AV} {
  if P@AV > 2 {
   ATTACK: ATTACK + 1
  }
  if P@AV > 4 {
   DAMAGE: DAMAGE + 1
  }
 }
 
 if ATTACK >= DEF {
  result: DAMAGE + DAM
 }
 if ATTACK + DAMAGE >= DEF {
  result: DAM
 }
 result: 0

}

loop A over {4..7} {

 AL: A
 if A > 5 {
  AL: 5
 }

 loop B over {1..AL} {
  loop C over {1..5} {
   output [damage from Ad6 attack with C damage against B defense] named "Attack +[A], damage [C] against defense [B]"
  }
 }
}

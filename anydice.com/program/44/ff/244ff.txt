function: roll gwf damage with SIDES_ON_DMG_DIE:n sided die {
    \ create a die that simulates rerolling the first 1 or 2 as a sequence \
    GWF_DIE: {{1..2}:2, {3..SIDES_ON_DMG_DIE}:SIDES_ON_DMG_DIE+2}
    result: dGWF_DIE
}

RAGE_DMG: 3
STR_DMG: 5
ATT_COUNT: 2

output ATT_COUNT * (STR_DMG + RAGE_DMG + 2d[roll gwf damage with 6 sided die]) named "gwf 2d6"


output 2d12 named "gwf 1d12"
output 2d[roll gwf damage with 12 sided die] named "gwf 1d12"

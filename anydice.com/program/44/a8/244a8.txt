function: gwfcheck REROLL_RANGE:s on ROLL_OUTCOME:n to ROLL_AGAIN:d {
    if [REROLL_RANGE contains ROLL_OUTCOME] { 
        result: ROLL_AGAIN }
    else {result: ROLL_OUTCOME}
}

function: gwf on DMG_ROLL:d {
    result: [gwfcheck {1,2} on DMG_ROLL to DMG_ROLL]
}

function: totaldmg ATK_ROLL:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ATK_ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    result: DMG
}

function: ROLL:s count above X {
 result: (ROLL >= X)
}

loop M over {3,4,5,6,7,8,10,12,14,16,18,20,24} {
 loop N over {[highest of 2 and M/2-2]..[lowest of 11 and M/2+2]} {
  loop O over {3,4,5,6,7,8,9,10,12,14,16,18,20,24} {
   P: 2 * N
   output [1dM count above 4]+[1dP count above 4]+[1dO count above 4] named "1d[M]+1d[P]+1d[O]"
  }
 }
}

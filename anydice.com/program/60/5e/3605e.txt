function: attack ROLL:n plus ATTACK:n vs DEFENSE:n ac for DROLL:d plus DMG:n damage {
 if ROLL = 1 { result: 0 }
 if ROLL = 20 { result: DROLL + DROLL + DMG }
 if ROLL + ATTACK >= DEFENSE { result: DROLL + DMG }
 result: 0
}

loop N over {1..35} {
  output [attack d20 plus 11 vs N ac for d8 plus 7 damage] named "[N]"
}

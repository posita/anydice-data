function: ceiling VALUE:n in SEQUENCE:n {
 if (SEQUENCE >= VALUE){
   result: VALUE
 }
 result: SEQUENCE
}
function: limit to zero in SEQUENCE:n {
 if (SEQUENCE < 0){result: 0}
 result: SEQUENCE
}
function: limit to max in SEQUENCE:n {
 if (SEQUENCE > 100){result: 100}
 result: SEQUENCE
}

\Passed Hits : Hits Roll into Save ROll :\
function: HITSROLL:s attacks with save SAVE:n{
 R: #HITSROLL - (#HITSROLLd(dQ>=3))                 \remove 3s\
 result: [limit to zero in (HITSROLL  - (RdQ + SAVE))]
}

\Unsaved Damages : Passed Hits into Damages :\
function: PASSEDHITS:s into D:n damages{
 result: [limit to max in (PASSEDHITS * D)]
}

function: roll NUM {
DICE: 10d10
BOTCH: [count 1 in DICE]
SUCCESS: [count {NUM..10} in DICE]
result: SUCCESS-BOTCH
}
HEALTH: 7
function: drink TOTAL {
ATTEMPT: [roll 9]
if ATTEMPT < 1 {
DMG: 6 - [roll 6]
if DMG > 0 {
HEALTH: HEALTH - DMG
}
} else {
TOTAL: TOTAL + ATTEMPT
}
if HEALTH < 1 {
result: 0
}
if TOTAL > 14 {
result: 1
}
result: [drink TOTAL]
}
output [drink 0]

set "maximum function depth" to 1
M: 2
H: 1
function: bonus A:n dc B:n die C:n {
 if C = 20 { 
 if C + A >= B { result: [bonus A dc B die 1d20] + 2 }
 if C + A < B & C + A >= B - 10 { result: [bonus A dc B die 1d20] + 1 }
 if C + A < B - 10 {
 B : B + M 
 M: M + M + H
 H: 0
 result: [bonus A dc B die 1d20] + 0
}
}
 if C + A >= B & C + A < B + 10 { result: [bonus A dc B die 1d20] + 1 }
 if C + A >= B + 10 { result: [ bonus A dc B die 1d20 ] + 2 }
 if C + A < B - 10 {
 B : B + M 
 M: M + M + H
 H: 0
 result: [ bonus A dc B die 1d20 ] - 1
}
 if C + A < B & C + A >= B - 10 {
 B : B + M 
 M: M + M + H
 H: 0
 result: [bonus A dc B die 1d20] + 0
}
}
loop N over {14, 16, 19, 24} {
 
 CHANCEONE: [bonus d{12, 13:8, 14:10, 15} dc N die 1d20]
 output [highest of 0 and CHANCEONE] named "[N]dcbonus13"
}

function: highest of A:n and B:n and C:n {
  result: [highest of [highest of A and B] and C]
}
function: lowest of A:n and B:n and C:n {
  result: [lowest of [lowest of A and B] and C]
}
function: middle of A:n and B:n and C:n {
  SUM: A+B+C SUM: SUM-[lowest of A and B and C] 
  result: SUM-[highest of A and B and C]
}
function: group ROLL:n with AID:n {
  G: ([highest of ROLL and AID]) - (ROLL<4 | AID<4)
  if G < 1 {result: 1} 
  else {result: G}
}
function: count from I:n in A:n and B:n and C:n {
  L: [lowest of A and B and C]
  M: [middle of A and B and C]
  H: [highest of A and B and C]
  P: {L, M, H}
  COUNT: 0
  if I@P>=4 {loop S over {1..I} {if (I-S+1)@P>=4 {COUNT: COUNT+1}}}
  else {loop F over {I..3} {if F@P<4 {COUNT: COUNT-1}}}
  result: COUNT
}

function: matches in A:n and B:n and C:n {
  if A=B & B=C & C=A {result: 3}
  else if A=B | B=C | C=A {result: 2}
  else {result: 0}
}

RANK: {-3..3}
SIGN: RANK/3

loop RANK over {-3..3}
{
if SIGN<0 {
output (([lowest 4  of 7d6+RANK]/2) + RANK) - d4 named "Rank [RANK]"
}
else if SIGN<0 {
output (([highest 4  of 7d6+RANK]/2) + RANK) + d4 named "Rank [RANK]"
}
else {
output (([middle 4  of 7d6+RANK]/2) + RANK) + d2 named "Rank [RANK]"
}
}

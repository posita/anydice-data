function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT}
 result: MISS
}

function: bite HM:n {
 if HM = MISS { result : 0 }
 if HM = HIT { result : 2d10+6 }
 if HM = CRITHIT { result : 4d10+6 }
 result : 0
}

function: claw HM:n {
 if HM = MISS { result : 0 }
 if HM = HIT { result : 2d6+6 }
 if HM = CRITHIT { result : 4d6+6 }
 result : 0
}

MISS:-1
HIT:-2
CRITHIT:-3

AC : 15
ATTACK : 10
loop AC over {10..20} {
 output [bite [attack 1d20 vs AC]] + [claw [attack 1d20 vs AC]] named "[AC]"
}

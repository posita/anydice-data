function: attack DAMAGE:n DAMAGEDRN:n PROT:n PROTDRN:n{
  if (DAMAGE + DAMAGEDRN) <= (PROT + PROTDRN) { result: 0 }
  result: (DAMAGE + DAMAGEDRN) - (PROT + PROTDRN)
}

function: attackcompare DAMAGE:n DAMAGEBONUS:n DAMAGEDRN:n PROT:n PROTDRN:n{
  if (DAMAGE + DAMAGEDRN + DAMAGEBONUS) <= (PROT + PROTDRN) { result: 0 }
  if (DAMAGE + DAMAGEDRN) >= (PROT + PROTDRN) { result: DAMAGEBONUS }
  result: (DAMAGE + DAMAGEDRN + DAMAGEBONUS) - (PROT + PROTDRN)
}

function: attackstokill DMG:n HP:n DEPTH{

result: DEPTH
}
else{

}

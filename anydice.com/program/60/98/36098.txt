function: force zeros in SEQUENCE:n {
 if (SEQUENCE < 0){result: 0}
 result: SEQUENCE
}

\Passed Hits : Hits Roll into Save ROll :\
function: HITSROLL:s attacks with save SAVE:n{
 R: #HITSROLL - (#HITSROLLd(dQ>=3))                 \remove 3s\
 result: [force zeros in (HITSROLL  - (RdQ + SAVE))]
}

\Unsaved Damages : Passed Hits into Damages :\
function: PASSEDHITS:s into D:n damages{
 result: PASSEDHITS * D
}

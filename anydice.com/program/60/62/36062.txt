function: twilightpush D:s as C:n {
  Z: 0
  if D >= 10 {
    Z: 1
  }
  if D >= 6 {
    Z: Z + 1  
  } else {
    Z: [twilight dC]      
  }
  result: Z
}

function: twilight C:n {
X: 0
  if C >= 6 {
    X: 1
  }
  if C >= 10 {
    X: 2
  }
result: X
}

loop N over {6,8,10,12}{
  loop K over {6,8,10,12}{
    loop M over {6,8,10,12}{
      loop L over {6,8,10,12}{
        output [twilightpush dN as N] + [twilightpush dK as K] - [twilightpush dM as M] - [twilightpush dL as L] named "d[N],d[K]-d[M],d[L]"
      }
    }
  }
}

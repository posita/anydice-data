\ Sumina's Awakened Fables Research Outcome Likelihood Calculator \

PLAYER_SKILL_MOD:9
PLAYER_LIFESTYLE_MOD:5
DM_DC_MOD: 5

PLAYER_ROLL: 1d20+PLAYER_SKILL_MOD+PLAYER_LIFESTYLE_MOD
DM_ROLL: 2d10+DM_DC_MOD

\ output PLAYER_ROLL named "Player Roll" \
\ output DM_ROLL named "DM Roll" \

RESULT_DIFF: (1d20+9+5) - (2d10+5)

\ output RESULT_DIFF named "Player - DM" \

MAJOR_FAIL: RESULT_DIFF <= -10
MINOR_FAIL: [count {-9..-1} in RESULT_DIFF]
MEETS_IT: RESULT_DIFF = 0
MINOR_SUCCESS: [count {1..9} in RESULT_DIFF]
MAJOR_SUCCESS: RESULT_DIFF >= 10

output MAJOR_FAIL named "Major Fail"
output MINOR_FAIL named "Minor Fail"
output MEETS_IT named "Meets It"
output MINOR_SUCCESS named "Minor Success"
output MAJOR_SUCCESS named "Major Success"

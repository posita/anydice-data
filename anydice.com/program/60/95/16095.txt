RANK: {-3..3}
SIGN: RANK/3
N: RANK/2

loop RANK over {-3..3}
{
if SIGN<0 {
output (([lowest 4  of 7d6+RANK]/2) + RANK) - 2dN / 2 + 1 named "Rank [RANK]"
}
else if SIGN<0 {
output (([highest 4  of 7d6+RANK]/2) + RANK) - 2dN / 2 + 1 named "Rank [RANK]"
}
else {
output (([middle 4  of 7d6+RANK]/2) + RANK) - 2dN / 2 + 1 named "Rank [RANK]"
}
}

function: save ROLL:n vs DEECEE:n {
 if ROLL + SAVINGBONUS >= DEECEE { result: SAVE}
 result: FAIL
}

function: firebreath HM:n {
 if HM = FAIL { result : 10d10 }
 if HM = SAVE { result : 10d10/2 }
 result : 0
}

SAVE:-1
FAIL:-2

SAVINGBONUS:1
DC:17
loop DC over {10..20} {
 output [firebreath [save 1d20 vs DC]] named "[DC]"
 output 1 + [save 1d20 vs DC] + 1 named "Save Chance"
}

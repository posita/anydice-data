function: withcrit A:n damage D:d oncrit C:d {
  if (A >= 20) {
    result: C
  }
  if (A <= 2) {
    result: 0
  }
  if (A >= 10) {
    result: D
  }
  result: 0
}

function: roll A:d with damage D:d oncrit C:d {
  result: [withcrit A damage D oncrit C]
}

\output 2d10\

output [roll 2d10 with damage 1d12 oncrit 2d12] named "1d12 dmg"

output [roll 2d10 with damage 2d6 oncrit 3d6] named "2d6 dmg"

output [roll 2d10 with damage 3d4 oncrit 4d4] named "3d4 dmg"

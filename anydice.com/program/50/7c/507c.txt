function: soulsuccess on A upgrade B {
 SUCC: 0
 if B < 3 {
  loop N over {1..(3-B)} {
   SUCC: SUCC + (d6 >= A)
  }
 }
 if B > 0 {
  loop N over {1..B} {
   SUCC: SUCC + (d8 >= A)
  }
 }
 result: SUCC
}

output [soulsuccess on 4 upgrade 0]
output [soulsuccess on 4 upgrade 1]
output [soulsuccess on 4 upgrade 2]
output [soulsuccess on 4 upgrade 3]

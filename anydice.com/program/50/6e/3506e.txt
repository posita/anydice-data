function: any doubles of N:n or above in ROLL:s {
  loop X over ROLL {                  \ Loops through all the possible combinations from 5d6. \
    if X >= N & (ROLL = X) > 1 {      \ If the die's value is >= 5 and it appears at least twice, count it. \
      result: 1
    }
  }
  result: 0
}

output [any doubles of 5 or above in 5d6]

BASE:3d6

function: wild W:n FA:n FB:n D:n M:n {
if FA = 0 {
  if W = 1 {
   if W + D < M { result: -100 } 
   else { result: W + FA + FB + D }
  } else { result: W + FA + FB + D }
 }
 if FA = 1 {
  if FB = 1 {
   if W = 1 {
    if W + FA + FB + D < M { result: -100 }
    else  { result: W + FA + FB + D } 
   } else { result: W + FA + FB + D }
  }
 }
 result: W + FA + FB + D
}

output [wild [explode d6] FA FB BASE MI] = -100 named "Critical Failure"
output [wild [explode d6] FA FB BASE MI] + PIP >= MI named "Minimal"
output [wild [explode d6] FA FB BASE MI] + PIP >= SO named "Solid"
output [wild [explode d6] FA FB BASE MI] + PIP >= GO named "Good"
output [wild [explode d6] FA FB BASE MI] + PIP >= SU named "Superior"

set "explode depth" to 10

F: {-1,0,1}
output 4dF-4dF named "4dF vs"

function: roll A:n vs D:n {
  AR: 4dF
  DR: 4dF
  result: [roll A vs D using AR vs DR]
}
function: roll A:n vs D:n using AR:s vs DR:s {
  AT: A + AR
  DT: D + DR
  if AT != DT { result: AT - DT }

  \ tiebreaker by counting negatives \
  AN: [count -1 in AR]
  DN: [count -1 in DR]
  if AN > DN { result: -1 }
  if DN > AN { result: 1 }

  result: 0
}

output [roll 0 vs 0]

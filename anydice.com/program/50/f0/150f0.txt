function: superadvantage {
  result: [highest of 1d20 and [advantage]]
}

function: advantage {
  result: [highest of 1d20 and 1d20]
}

function: ability check DC MOD {
  result: 1d20 + MOD >= DC
}

function: multiroll ROLL TIMES {
  RET: {}
  loop N over {1..TIMES} {
    RET: RET + ROLL
  }
  result: RET
}

function: lancer MOD {
  RET: 1d20
  DIFF: {}
  loop N over {1..[absolute MOD]} {
    DIFF: [highest of 1d6 and DIFF]
  }
  if MOD > 0 {
    RET: RET + DIFF
  } 
  else {
    RET: RET - DIFF
  }
  result: RET
}

\output [multiroll [ability check 15 2] 7] >= 3\
output [lancer -1]

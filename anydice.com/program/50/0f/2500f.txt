
function: check successes in ROLL:s by APP:n {
 CSC: 0 \final max score current\
 \TSC temporary score holder\
 
 loop I over {1..6} { \I is checker\
  TSC: 0
  
  loop J over {1..#ROLL} {
   if J@ROLL = I { 
    if I=6 {
     TSC: TSC +1
    }
    else {
     if APP>=I {TSC: TSC+1}
    }}
   
  }

  if I=1 {
   CSC: TSC
  }
  else { if CSC < TSC { CSC: TSC} }
 }
 
 result: CSC
}

loop N over {1..6} {
 ROLL: Nd6
 loop T over {1..5} {
  output [check successes in ROLL by T] named "[N] dice rolled with [T] approach"
 }
}


function: spam slot level {
 OUTPUT: 0
 if SORCERER = 2 {OUTPUT: 1}
 if SORCERER = 3 {OUTPUT: 2}
 if SORCERER >= 5 {OUTPUT: SORCERER - 2}
 if OUTPUT > 5 {OUTPUT: 5}
 result: OUTPUT
}
function: slot level {
 result: (SORCERER + 1 + PALADIN/2) / 2
}
function: warlock slot level {
 result: (WARLOCK + 1) / 2
}
function: cantrip level {
 OUTPUT: 1
 if TOTALLEVEL >= 5 {OUTPUT: 2}
 if TOTALLEVEL >= 11 {OUTPUT: 3}
 if TOTALLEVEL >= 17 {OUTPUT: 4}
 result: OUTPUT
}

function: melee FIRST {
 DAMAGE: CHARISMA + [weapon damage]
 if PALADIN >= 2 {DAMAGE: DAMAGE + [smite]}
 if WARLOCK*FIRST >= 5 {DAMAGE: DAMAGE + [eldritch smite]}
 result: DAMAGE
}

function: smite {
 OUTPUT: 0
 SLOTLEVEL: [spam slot level]
 if SLOTLEVEL > 4 {SLOTLEVEL: 4}
 if SLOTLEVEL > 0 {OUTPUT: 1d8 + (SLOTLEVEL)d8}
 result: OUTPUT
}

function: eldritch smite {
 OUTPUT: 0
 SLOTLEVEL: [warlock slot level]
 if SLOTLEVEL > 5 {SLOTLEVEL: 5}
 if SLOTLEVEL > 0 {OUTPUT: 1d8 + (SLOTLEVEL)d8}
 result: OUTPUT
}

function: booming blade FIRST {

 result: OUTPUT
}

function: weapon damage {

}

function: proficiency {

}

function: eldritch blast {
 result: [cantrip level]*(1d10 + CHARISMA + [on hit effects])
}

function: on hit effects {

 result: OUTPUT + HEXBLADE
}

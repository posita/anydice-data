function: superadvantage {
  result: [highest of 1d20 and [highest of 1d20 and 1d20]]
}

function: ability check DC MOD {
  result: 1d20 + MOD >= DC
}

function: multiroll ROLL TIMES {
  RET: {}
  loop N over {1..TIMES}{
    RET: RET + ROLL
  }
  result: RET
}

\output [multiroll [ability check 15 2] 7] >= 3\

PERC: 1d20+9
STEALTH: 1d20+11
output (STEALTH = PERC)

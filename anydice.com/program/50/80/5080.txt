function: soulsuccess on A upgrade B {
 SUCC: 0
 if B < 3 {
  loop N over {1..(3-B)} {
   SUCC: SUCC + (d6 >= A)
  }
 }
 if B > 0 {
  loop N over {1..B} {
   SUCC: SUCC + (d8 >= A)
  }
 }
 result: SUCC
}
function: soulsuccess on A downgrade B {
 SUCC: 0
 if B < 3 {
  loop N over {1..(3-B)} {
   SUCC: SUCC + (d6 >= A)
  }
 }
 if B > 0 {
  loop N over {1..B} {
   SUCC: SUCC + (d4 >= A)
  }
 }
 result: SUCC
}
ONVALUE: 4
\output [soulsuccess on ONVALUE downgrade 3] named "-3"
output [soulsuccess on ONVALUE downgrade 2] named "-2"
output [soulsuccess on ONVALUE downgrade 1] named "-1"\
output [soulsuccess on ONVALUE upgrade 0] named "+0"
output [soulsuccess on ONVALUE upgrade 1] named "+1"
output [soulsuccess on ONVALUE upgrade 2] named "+2"
output [soulsuccess on ONVALUE upgrade 3] named "+3"

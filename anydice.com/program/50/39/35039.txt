\ X is Difficulty Level, Y is the modifier : \
\ This ignores all "trivial" results, where the modifier is bigger than the difficulty, to not bloat the results (doesn't ignore trivially impossible results)\
loop X over {10..25}{
 loop Y over {1..15}{
  if (X > Y){
   output 2d6 + Y > X named "+[Y] modifer vs [X]"
   output {1,2}@3d6 + Y > X named "with advantage"
  }
 }
}

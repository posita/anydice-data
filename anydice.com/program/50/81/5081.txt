function: soulsuccess on A upgrade B {
 SUCC: 0
 if B < 3 {
  loop N over {1..(3-B)} {
   SUCC: SUCC + (d6 >= A)
  }
 }
 if B > 0 {
  loop N over {1..B} {
   SUCC: SUCC + (d8 >= A)
  }
 }
 result: SUCC
}
function: soulsuccess on A downgrade B {
 SUCC: 0
 if B < 3 {
  loop N over {1..(3-B)} {
   SUCC: SUCC + (d6 >= A)
  }
 }
 if B > 0 {
  loop N over {1..B} {
   SUCC: SUCC + (d4 >= A)
  }
 }
 result: SUCC
}
TARGETNUMBER: 5
UPGRADE: 1
output [soulsuccess on 3 upgrade UPGRADE] named "easy"
output [soulsuccess on 4 upgrade UPGRADE] named "medium"
output [soulsuccess on 5 upgrade UPGRADE] named "hard"

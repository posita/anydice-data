function: wild W:n FA:n FB:n D:n B:n M:n {
if FA = 0 {
  if W = 1 {
   if W + FA + FB + D + B < M { result: -100 } 
   else { result: W + FA + FB + D + B }
  } else { result: W + FA + FB + D + B }
 }
 if FA = 1 {
  if FB = 1 {
   if W = 1 {
    if W + FA + FB + D + B < M { result: -100 }
    else  { result: W + FA + FB + D + B } 
   } else { result: W + FA + FB + D + B }
  }
 }
 result: W + FA + FB + D + B
}

output [wild [explode d6] FA FB DICE BASE MI] = -100 named "Critical Failure"
output [wild [explode d6] FA FB DICE BASE MI] >= MI named "Minimal"
output [wild [explode d6] FA FB DICE BASE MI] >= SO named "Solid"
output [wild [explode d6] FA FB DICE BASE MI] >= GO named "Good"

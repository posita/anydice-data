function: withcrit A:n damage D:d oncrit C:d {
  if (A >= 20) {
    result: C
  }
  if (A <= 2) {
    result: 0
  }
  result: D
}

function: roll A:d with damage D:d oncrit C:d {
  result: [withcrit A damage D oncrit C]
}

output 2d10

output [roll 2d10 with damage 1d12 oncrit 2d12]

output [roll 2d10 with damage 2d6 oncrit 3d6]

output [roll 2d10 with damage 3d4 oncrit 4d4]

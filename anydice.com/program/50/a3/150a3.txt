function: stunts DICE:s DRAGON:n TARGET:n {
  if 1@DICE + 2@DICE + DRAGON >= TARGET {
    if 1@DICE = 2@DICE | 1@DICE = DRAGON | 2@DICE = DRAGON {
      result: DRAGON
    }
    result: 0
  }
  result: -1
}

loop X over {5,10,15} {
  output [stunts 2d6 1d6 X] named "Expected SP at target roll [X]"
}

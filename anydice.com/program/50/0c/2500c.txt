
function: check successes in ROLL:s {
 CSC: 0 \final max score current\
 \TSC temporary score holder\
 
 loop I over {1..6} { \I is checker\
  TSC: 0
  
  loop J over {1..#ROLL} {
   if J@ROLL = I {TSC: TSC+1}
   
  }

  if I=1 {
   CSC: TSC
  }
  else { if CSC < TSC { CSC: TSC} }
 }
 
 result: CSC
}

loop N over {1..6} {
ROLL: Nd6
output [check successes in ROLL] named "[N] dice rolled"
}

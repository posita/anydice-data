function: X:n mod Y:n
{
result: X - (X/Y)*Y
}

function: roll X:n       \strictly for evades\
{
if(X<=3){result: 1}
if(X<=5){result:100}
result: 0
}

function: focusdiff X:n Y:n
{
result: [lowest of [highest of 0 and (X-[Y mod 100])] and Y/100]
}

function: evadediff X:n Y:n
{
result: [lowest of [highest of 0 and (X-[Y mod 100])] and 1]
}

function: diff ATK:n DEF:n
{
result: [evadediff ATK DEF] - [focusdiff ATK DEF]
}


ATKDICE: 3d(1d8<=6)
DEFDICE: 3d[roll 1d8]

output [diff ATKDICE DEFDICE] named "amount of damage evading will prevent compared to focusing"

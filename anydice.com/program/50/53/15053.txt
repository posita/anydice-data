MISS:0
HIT:1
CRITICALHIT:2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{12..22}

ATTACK: 2\dex\ + 2\prof\ + 1\inf\
HITDAMAGE: d8 + d6 + 2\dex\ + 1\inf\
CRITICALDAMAGE: 2d8 + 2d6 + 2 + 1
MISSDAMAGE: 0
output [attack] named "Crossbow"

ATTACK: 5\int\ + 2\prof\ + 1\inf\
HITDAMAGE: 1d10 + 1\inf\
CRITICALDAMAGE: 2d10 + 1
output [attack] named "Firebolt"

ATTACK: 5\int\ + 2\prof\ + 1\inf\
HITDAMAGE: 2d6 + 5\int\ + 1\inf\
CRITICALDAMAGE: 4d6 + 5 + 1
output [attack] named "Magic Stone"

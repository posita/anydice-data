\variables - do modify\
LVL: 2
DICE: 3

\calculations - do not modify\
function: corruptdie D:d MOD:n {
    result: [highest of 0 and D-MOD]
}
loop T over {3..18} {
    CORRUPTION: 0
    loop X over {1..DICE} {
         CORRUPTION: CORRUPTION + [corruptdie d6 (LVL/2)]
    }
    output CORRUPTION > T named "Level [LVL] and THT [T] with [DICE]d6 corruption"
}

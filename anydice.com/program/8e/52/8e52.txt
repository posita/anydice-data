function: count dice in D:s {
 result: [count {1d6} in D]
}
function: count hits on D:s {
 result: [count {5,6} in D]
}
function: count ones on D:s {
 result: [count {1} in D]
}
function: tallyroll D:n dice H:n hits O:n ones {
 X: (D+1)/2
 if O > X {
  if H > 0 {
   result: 0 - H
  }
  else {
   result: -999
  }
 }
 else {
  result: H
 }
}
function: roll D:s {
 result: [tallyroll [count dice in D] dice [count hits on D] hits [count ones on D] ones]
}
D: 2d6
output [roll D] named "roll"
output [count hits on D] named "hits"
output [count ones on D] named "ones"
output [count dice in D] named "dice"

\variables - do modify\
THT: 10
\calculations - do not modify\
function: corruptdie D:d MOD:n {
    result: [highest of 0 and D-MOD]
}
loop L over {1..1} {
    loop D over {1..6} {
        CORRUPTION: 0
        loop X over {1..D} {
             CORRUPTION: CORRUPTION + [corruptdie d6 (L/2)]
        }
        output CORRUPTION > THT named "Level [L] with [D]d6"
    }
}

function: more than VALUE:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#SEQUENCE} {
  COUNT: COUNT + (VALUE < P@SEQUENCE)
 }
 result: COUNT
}

ST: 3

output [more than PT in 2d10] + [more than PT in 1d12] = ST named "lvl 1 - PT [PT]"
output [more than PT in 1d10] + [more than PT in 2d12] = ST named "lvl 2 - PT [PT]"
output [more than PT in 3d12] = ST named "lvl 3 - PT [PT]"
}

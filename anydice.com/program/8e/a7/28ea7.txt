\variables - do modify\
THT: 10
LVL: 1

\calculations - do not modify\
function: corruptdie D:d MOD:n {
    result: [highest of 0 and D-MOD]
}
loop D over {1..6} {
    CORRUPTION: 0
    loop X over {1..D} {
         CORRUPTION: CORRUPTION + [corruptdie d6 (LVL/2)]
    }
    output CORRUPTION > THT named "Level [LVL] and THT [THT] with [D]d6 corruption"
}

function: explode N:n {
 if N = 1 {
  result: N - d10
 }
 if N = 10 {
  result: N + d10
 }
 result: N
}

function: damage N:n {
 result: N
}

function: crit N:n {
 if N = 6 {
 result: 1
 }
 result: 0
}

DAMAGE: 5d([damage d6])
CRIT: 5d([crit d6])
ATTACK: [explode 1d10]+16
DODGE: [explode 1d10]+20
DV: 15

output ATTACK > DODGE named "Vs Dodge"
output ATTACK > DV named "Vs DV"
output DAMAGE named "Damage"
output CRIT > 1 named "Crit Chance"

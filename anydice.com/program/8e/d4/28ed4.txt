\ set these to get different results \
N:4 \ number of sides on each die \
L:10 \ max level of (dis)advantage \
\ code \
set "position order" to "lowest first"
loop I over {-L..-1} {
 output 1@(-I+1)dN named "[I]" \ disadvantage \
}
output dN named "0" \ no modifiers \
set "position order" to "highest first"
loop I over {1..L} {
 output 1@(I+1)dN named "+[I]" \ advantage \
}

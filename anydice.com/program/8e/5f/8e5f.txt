MISS: 0 
HIT: 1 
CRITICALHIT: 2 

function: attack ROLL:n vs DEFENSE:n { 
if ROLL = 1 { result: MISS } 
if ROLL = 20 { 
if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT } 
result: HIT 
} 
if ROLL + ATTACK >= DEFENSE { result: HIT } 
result: MISS 
} 

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: uadamage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITUADAMAGE
 }
 if ATTACKRESULT = HIT {
  result: UADAMAGE
 }
 result: MISSDAMAGE
}

function: daggerdamage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITDAGGERDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: DAGGERDAMAGE
 }
 result: MISSDAMAGE
}

function: attack { 
result: [damage for [attack d20 vs DEFENSE]] 
}

}

 result: [daggerdamage for [attack d20 vs DEFENSE]]
}

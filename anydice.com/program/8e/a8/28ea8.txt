\variables - do modify\
THT: 10
DICE: 3

\calculations - do not modify\
function: corruptdie D:d MOD:n {
    result: [highest of 0 and D-MOD]
}
loop L over {1..7} {
    CORRUPTION: 0
    loop X over {1..DICE} {
         CORRUPTION: CORRUPTION + [corruptdie d6 (L/2)]
    }
    output CORRUPTION > THT named "Level [L] and THT [THT] with [DICE]d6 corruption"
}

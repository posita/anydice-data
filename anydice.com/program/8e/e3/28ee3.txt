function: X:n mageroll {
if X >= 6 {result: 1}
if X = 1 {result: -1}
result: 0
}

function: X:n multiroll {
SUM: 0
loop N over {1..X}{SUM: SUM + [d10 mageroll]}
result: SUM
}

function: X:n filter {
if X <= -1 {result: -1}
if X >= 1 {result: 1}
else {result: X}
}

loop N over {1..10}{
 output [[N multiroll] filter] named "[N]d10"
}

function: multiply ROLL:n by AMPLIFIER:n if it is TARGET:n{
    if ROLL = TARGET {
        result: ROLL * AMPLIFIER
    }
    result: ROLL
}
function: weiss ROLL:n on REP_BORDER:n to REPLACE:n{
    if ROLL <= REP_BORDER {
        result: REPLACE
    }
    result: ROLL
}

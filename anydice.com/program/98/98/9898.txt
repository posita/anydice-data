function: double HITS:n {
  if HITS = 3 {
    result: TRIPLE
  }
  if HITS = 2 {
    result: DOUBLE
  }
  if HITS = 1 {
    result: BEST
  }
  result: 0
}

NORMAL: d12 + 10

DOUBLE: BEST + NORMAL
TRIPLE: BEST + 2dNORMAL

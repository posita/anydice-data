TN: 15
KD: 4
BD: 2 
MOD:0
RR: 0 \set to "1" to reroll 1s\

if RR = 1 {
 ROLL: [highest KD of (KD+BD)d{2..6}]+MOD
}
else {
 ROLL: [highest KD of (KD+BD)d6]+MOD
}

output ROLL <= TN-5                  named "crit fail"
output ROLL >  TN-5  & ROLL < TN     named "0"
output ROLL >  TN    & ROLL < TN +5  named "1"
output ROLL >  TN+4  & ROLL < TN +10 named "2"
output ROLL >  TN+9  & ROLL < TN +15 named "3"
output ROLL >= TN+15                 named "4"

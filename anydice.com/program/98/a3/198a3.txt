DC: 13
PROF: 5
EXP: PROF * 2
function: X:n wr W:n{
if X > DC {result: (X + W-DC)}
result: 0
}

function: X:n rep{
if X > DC {result: X-DC}
result: 0
}

function: X:n reduce C:n{
if X > DC-C {result: X-(DC-C)}
result: 0
}

function: X:n reroll{
if X > DC {result: X-DC}
else      {result: [d20 rep]}
}


output [d20 rep]                 named " NORM "
output [d20+PROF rep]            named " PROF "
output [d20+EXP rep]             named " EXPERT "
output [d20 reduce 2]            named " DC - 2 "
output [d20 wr 5]              named " WR + 5 "
output [[highest 1 of 2d20] rep] named " ADV "
 output [d20 reroll] named " REROLL " 
 output [[highest 1 of 3d20] rep] named " SUPER ADV " 
 output [[highest 1 of 2d20]+PROF rep] named " FAVOUR + PROF "

function: attackroll ROLL:n vs AC:n {
if ROLL >= CRITROLL { result: CRITDAMAGE }
if ROLL = 1 { result: 0 }
if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
result: 0
}

loop ENEMYAC over {5..28} {

ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
PROFICIENCY: 3
DMGBONUS: 0

ATTACKMOD: 5
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS

output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Javelin vs. AC [ENEMYAC]"

output ATTACKTIMESd[attackroll [lowest 1 of 2d20] vs ENEMYAC] named "Javelin w/ disadvantage vs. AC [ENEMYAC]"

ATTACKMOD: 0
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS

output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Longbow vs. AC [ENEMYAC]"

output ATTACKTIMESd[attackroll [lowest 1 of 2d20] vs ENEMYAC] named "Javelin w/ disadvantage vs. AC [ENEMYAC]"
}

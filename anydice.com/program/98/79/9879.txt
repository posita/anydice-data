function: attack N:n over P with D{
  if N > P {
    result: D
  }
}

function: gwm N:n and I:n over P with D and B {
  X: 0
  if N > P {
    X: D
  }

  if I > P {
    X: X + B
  }
  result: X
}

output [attack d100 over 54 with d12 + 10] named "Great Weapon Mastery Lv4"
output [attack d100 over 40 with [highest of d12 and d12]] named "Savage Attacks Lv4"

output [gwm d100 and d100 over 54 with d12 and d12] named "Great Weapon Mastery Test"

output [attack d100 over 54 with 2d12 + 10] named "Great Weapon Mastery Lv4"
output [attack d100 over 40 with [highest of 2d12 and 2d12]] named "Savage Attacks Lv5"

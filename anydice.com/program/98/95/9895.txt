\This sets amount of dice that are added to the attribute\
DICE: 3
\This sets maximum amount of explosions\
set "explode depth" to 1


function: add together ROLL:s{
if DICE = 1 {result: [explode 1@ROLL]}
if DICE = 2 {result: [explode 1@ROLL] + 2@ROLL}
if DICE = 3 {result: [explode 1@ROLL] + 2@ROLL + 3@ROLL}
result: 0
}
output [add together 3d12]

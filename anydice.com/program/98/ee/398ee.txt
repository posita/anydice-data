\**** Damage calculator ****\

function: orcattackroll ROLL:n vs AC:n {
 if ROLL >= ORCCRITROLL { result: ORCCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + ORCPROFICIENCY + ORCATTACKMOD + ORCACCBONUS >= AC { result: ORCDAMAGE }
 result: 0
}

function: knightattackroll ROLL:n vs AC:n {
 if ROLL >= KNIGHTCRITROLL { result: KNIGHTCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + KNIGHTPROFICIENCY + KNIGHTATTACKMOD + KNIGHTACCBONUS >= AC { result: KNIGHTDAMAGE }
 result: 0
}

KNIGHTCRITROLL: 20

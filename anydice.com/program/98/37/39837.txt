DIE: d{-1,-1,0,0,2,1}

function: net success for DICE:s {
  BLACK: [count -1 in DICE]
  YELLOW: [count 2 in DICE]
  WHITE: [count 1 in DICE]
  
  NET_WHITE: WHITE - BLACK
  if NET_WHITE < 0 { NET_WHITE: 0 }
  
  result: YELLOW + NET_WHITE
}

output [net success for 1dDIE] named "1 die: net Success"
output [net success for 2dDIE] named "2 dice: net Successes"
output [net success for 3dDIE] named "3 dice: net Successes"

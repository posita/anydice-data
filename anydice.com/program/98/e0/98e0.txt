MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL > 18 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [attack d20 vs DEFENSE]
}

function: notattack {
 result: [damage for [attack d20 vs DEFENSE]]
}

function: doubleattack {
 PRIMARY: [attack d20 vs DEFENSE]
 HITDAMAGE: d6
 CRITICALDAMAGE: 2d6
 OFFHAND: [attack d20 vs DEFENSE]
 result: PRIMARY + OFFHAND
}

function: notdoubleattack {
 PRIMARY: [damage for [attack d20 vs DEFENSE]]
 HITDAMAGE: d6
 CRITICALDAMAGE: 2d6
 OFFHAND: [damage for [attack d20 vs DEFENSE]]
 result: PRIMARY + OFFHAND
}
 
DEFENSE: d{12..22}

MISSDAMAGE: 0
output [attack] named "Talon"

output [attack] named "Magic Rapier"

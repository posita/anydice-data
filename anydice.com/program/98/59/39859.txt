function: explode ROLLEDVALUE:n {
 if ROLLEDVALUE = 2 { result: 2 - 1d9 }
 if ROLLEDVALUE = 3 { result: 3 - 1d9 }
 if ROLLEDVALUE = 18 { result: 18 + 1d9 }
 result: ROLLEDVALUE
}

function: flycheck HEIGHTDIFF:n ATHLETICS:n ROLL:n {
  result: ATHLETICS - [explode ROLL] + (HEIGHTDIFF) / 6
}

function: flydamage CHECK:n {
  if CHECK > 0 {
    result: 0
  }
  else {
    result: -CHECK * 2
  }
}

output [flydamage [flycheck (10 - 0) 8 2d9]] named "Damage taken"

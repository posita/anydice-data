function: attack TGT targets ATK times with advantage ADV {
  if ATK > 1 {
    \ Disadvantage for multiattacks: 3 per extra attack - 1 (feat level), Disadvantage for multitargeting: 1 per extra target \
    ADVANTAGE: ADV - 3*(ATK - 1) + 1 - (TGT - 1)
  }
  else {
    ADVANTAGE: ADV - (TGT - 1)
  }

  if ADVANTAGE > 0 {
    result: ([explode d20] + [highest 2 of (2 + ADVANTAGE)d[explode d6]])
  }
  else {
    result: ([explode d20] + [lowest 2 of (2 - ADVANTAGE)d[explode d6]])
  }
}

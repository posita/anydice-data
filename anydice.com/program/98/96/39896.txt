function: remaining ROLLS:s select SELECT:s vs DC:n {
 if (1@SELECT) != (2@SELECT) & 1@SELECT & 2@SELECT {
  SELECTA: (1@SELECT)@ROLLS
  SELECTB: (2@SELECT)@ROLLS
  SELECTSUM: SELECTA + SELECTB
  if SELECTA & SELECTB & SELECTSUM >= DC {
   result: ROLLS - SELECTSUM
  }
 }
 result: -1
}

function: remaining A:n and B:n and C:n and D:n and E:n vs DC:n {
 result: [maximum of [remaining {A, B, C, D, E} select 2d5 vs DC]]
}

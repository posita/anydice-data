function: roll A:n vs B:n for C:n {
  if A >= B { result: C }
  result: 0
}

function: mod A:n factor B:n {
  if A = 3 {result: -2}
  result: (B-10)/B
}

AB: d{-1..5}+[mod 3d6 factor 4]+d{0..10}/2
AC: d{10..20}+[mod 3d6 factor 4]
DAM: 1

output [roll 2d6+AB vs AC-3 for DAM]
output [roll 1d20+AB vs AC for DAM]

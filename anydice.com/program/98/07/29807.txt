function: roll A:n plus B:n plus C:n plus D:n vs E:n {
 if A+B+C=3 | A+B+C+D<=3 | A+B+C+D+5<=E { result: 100 }
 if A+B+C<=5 { result: 100 }
 if A+B+C=18 | A+B+C+D>=18 | A+B+C+D>E+5 { result: A*10+B }
 if A+B+C+D>E { result: 200 }
 result:100
}


loop X over {12}{
loop N over {-5..5}{
output [roll 1d6 plus 1d6 plus 1d6 plus N vs X] named "+[N] vs [X]"
}}

\This sets amount of dice that are added to the attribute\
DICE: 2
\This sets maximum amount of explosions\
set "explode depth" to 1

function: add together ROLL:s and EXP:s {
if DICE = 1 {result: [highest of 1@ROLL and EXP]}
if DICE = 2 {result: {1,2}@[sort {1@ROLL,2@ROLL,EXP}]}
if DICE = 3 {result: ROLL + EXP}
result: 0
}
output [add together 2d12 and [explode d12]]

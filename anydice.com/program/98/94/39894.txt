function: negative used ROLLS:s select SELECT:s vs T:n {
 if (1@SELECT) != (2@SELECT) & 1@SELECT & 2@SELECT {
  SELECTA: (1@SELECT)@ROLLS
  SELECTB: (2@SELECT)@ROLLS
  SUM: SELECTA + SELECTB
  if SELECTA & SELECTB & SUM >= T {
   result: -SUM
  }
 }
 result: -1000
}

}

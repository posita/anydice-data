\**** Damage calculator duelling Vs archery ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
DMGBONUS: 1
PROFICIENCY: 3
ACCBONUS: 1
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS + 1d8 + 1d8 + ATTACKMOD
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS + 2d8 + 2d8 + ATTACKMOD 
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Mace +1 GWB"

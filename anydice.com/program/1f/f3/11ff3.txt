function: luck ROLLA:n and ROLLB:n{
 RES: 0

 if ([count {1..3} in ROLLA]) {
   RES: RES +1
 }
 if ([count {8..10} in ROLLA]) {
   RES: RES -1
 }

 result: RES
}
LUCK: [luck d10 and d10]
function: managepifia SUCCESSES:n vs DIFF:n{
 if (SUCCESSES = -1) {
   result: -1
 }
 result: SUCCESSES

}

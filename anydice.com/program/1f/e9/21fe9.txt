function: smash {
    result: [hit [highest 1 of 2d6] wound [highest 1 of 2d6] save 4]
         + [hit [highest 1 of 2d6] wound [highest 1 of 2d6] save 4]
         + [hit [highest 1 of 2d6] wound [highest 1 of 2d6] save 4]
         + [hit [highest 1 of 2d6] wound [highest 1 of 2d6] save 4]
         + [hit [highest 1 of 2d6] wound [highest 1 of 2d6] save 4]
         + [hit [highest 1 of 2d6] wound [highest 1 of 2d6] save 4]
}

function: hit HIT:n wound WOUND:n save SAVE:n {
    if HIT > 2 {
        if WOUND > 2 {   
            if HIT = 6 {
                result: [damage SAVE 1d6 1 1d6]
            } else {
                result: [damage SAVE 1d6 0 1d6]
            }
        }
    } else {
        result: 0
    }
}

function: damage SAVE:n SAVEROLL:n RUNE:n RUNEROLL:n {
    if SAVEROLL - 2 < SAVE {
        if RUNE = 1 {
            result: 3 + RUNEROLL
        } else {
            result: 3
        }
    } else {
        result: 0
    }
}

output [smash]

function: Avandrasfavor ROLL:n {
 if ROLL = 7 | ROLL = 12 {
  result: 1
 }
 else if ROLL + 1d6 = 7 | ROLL + 1d6 = 12 {
   result: 2
  }
 else { result: 0 }
}

function: probe ROLL:n {
 if ROLL < 7 {
  result: 0
 }
 else if 9 < ROLL {
  result: 2
 }
 else 
 {
  result: 1
 }
}


function: reroll ROLL:n {
 if ROLL = 1 {
  result: [rerolla d6]
 }
 else {
  result: ROLL
 }
}

function: rerolla ROLL:n {
 if ROLL = 1 {
  result: [rerollb d6]
 }
 else {
  result: ROLL
 }
}

function: rerollb ROLL:n {
 if ROLL = 1 {
  result: d6
 }
 else {
  result: ROLL
 }
}

output [Avandrasfavor 2d6] named "Avandrasfavor"
\output [probe 2d[reroll 1d6]] named "No1"\
output [probe  2d6] named "probe"
output [probe [highest 2 of 3d6]] named "Advantage"
output [probe [lowest 2 of 3d6]] named "Disadvantage"

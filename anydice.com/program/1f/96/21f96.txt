LVL: 10
DEFENSE: d{10..20}
CURSE: 1
ADV: 0

MISS: 0
HIT: 1
CRITICALHIT: 2
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 0 { result: MISS }
 if CURSE!=0 {if ROLL > 18 {result: CRITICALHIT}}
 if CURSE=0 {if ROLL = 20 {result: CRITICALHIT}}
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}
function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}
function: advattack {
 result: [damage for [attack 1@2d20 vs DEFENSE]]
}
function: disattack {
 result: [damage for [attack 2@2d20 vs DEFENSE]]
}

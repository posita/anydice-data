function: attack ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + ACCMOD >= AC { result: DAMAGE }
 result: 0
}

function: pamd ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: 2d4 + ATTACKMOD + DMGBONUS + 2dBONUSDMGDIE + 2 } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + ACCMOD >= AC { result: 1d4 + ATTACKMOD + DMGBONUS + BONUSDMGDIE + 2}
 result: 0
}

function: pamg ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: 2d{1..2,3..4:6} + ATTACKMOD + DMGBONUS + BONUSDMGDIE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + ACCMOD >= AC { result: 1d{1..2,3..4:6} + ATTACKMOD + DMGBONUS + BONUSDMGDIE }
 result: 0
}

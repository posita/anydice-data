function: first A:n second B:n {
  if A >= CRITROLL & B >= CRITROLL {result: ACRIT + BCRIT + CRITTURN}
  if A >= CRITROLL & B = 1 {result: ACRIT + CRITTURN}
  if A >= CRITROLL & B + ATKMOD + PROF + ACCBNS + ACCMOD >= AC {result: ACRIT + BDMG + CRITTURN}
  if A = 1 & B >= CRITROLL {result: BCRIT + CRITTURN}
  if A = 1 & B = 1 {result: 0}
  if A = 1 & B + ATKMOD + PROF + ACCBNS + ACCMOD >= AC {result: BDMG + TURNDMG}
  if A + ATKMOD + PROF + ACCBNS + ACCMOD >= AC & B >= CRITROLL {result: ADMG + BCRIT + TURNDMG}
  if A + ATKMOD + PROF + ACCBNS + ACCMOD >= AC & B = 1 {result: ADMG + TURNDMG}
  if A + ATKMOD + PROF + ACCBNS + ACCMOD >= AC & B + ATKMOD + PROF + ACCBNS + ACCMOD >= AC {result: ADMG + BDMG + TURNDMG}
  if A >= CRITROLL {result: ACRIT + CRITTURN}
  if B >= CRITROLL {result: BCRIT + CRITTURN}
  if A + ATKMOD + PROF + ACCBNS + ACCMOD >= AC {result: ADMG + TURNDMG}
  if B + ATKMOD + PROF + ACCBNS + ACCMOD >= AC {result: BDMG + TURNDMG}
  result: 0
}

\STR/DEX\
ATKMOD: 4
PROF: 3

CRITROLL: 20

\Bonus Damage Die on Hit. Divine Favor, Hunter's Mark, etc. Doubles on Crit.\
BNSDMGDIE: 1d6

\Flat Magic Bonuses. +1/+2/+3\
DMGBONUS: 0
ACCBNS: 0

\Accuracy Bonus. Archery fighting style, Bless, etc.\
ACCMOD: 0

\Enemy AC\
AC: 15

ADMG: 1d8 + ATKMOD + DMGBONUS + BNSDMGDIE
ACRIT: 2d8 + ATKMOD + DMGBONUS + 2dBNSDMGDIE
BDMG: 1d8 + DMGBONUS + BNSDMGDIE
BCRIT: 2d8 + DMGBONUS + 2dBNSDMGDIE

\Once Per Turn Damage. Sneak Attack, Colossus Slayer, etc\
TURNDMG: 0
CRITTURN: 0

output [first d20 second d20] named "Attack"

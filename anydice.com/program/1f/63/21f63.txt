function: checkstress DIFFICULTY:s CURRENT:n {
S:0
loop X over {1..#DIFFICULTY} {
 if (X@DIFFICULTY > CURRENT) {
if (X > 1) {
if ((X-1)@DIFFICULTY != X@DIFFICULTY) { S: S + 1}
} else {
S: S+1
}
}
}
result: S+CURRENT
}

loop X over {1..6} {
loop Y over {0..5} {
output [checkstress Xd6 Y] named "[X]d6 Stress[Y]"
}
}

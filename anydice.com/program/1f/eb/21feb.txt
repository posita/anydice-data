TOHIT: 4
TOWOUND: 3
TOSAVE: 5
REND: 0
DAMAGE: 1
ATTACKS: 2
HITROLL: [highest 1 of 2d6]
WOUNDROLL: 1d6
SAVEROLL: 1d6

function: attack {
    DEALT: 0
    loop N over {1..ATTACKS}{
        DEALT: DEALT + [check hit HITROLL WOUNDROLL]
    }
    result: DEALT
}

function: check hit HIT:n WOUND:n {
    if HIT >= TOHIT & WOUND >= TOWOUND {
        result: [check damage SAVEROLL]
    } else {
        result: 0
    }
}

function: check damage SAVE:n {
    if SAVE - REND < TOSAVE {
        result: DAMAGE
    } else {
        result: 0
    }
}

output [attack]

function: checkstress DIFFICULTY:s CURRENT:n DROP:n{
S:0
D:DROP
loop X over {1..#DIFFICULTY} {
 if (X@DIFFICULTY > CURRENT) {
if (X = 1 | ((X-1)@DIFFICULTY != X@DIFFICULTY)) {
if ((D > 0) & ((X = #DIFFICULTY) |  (X@DIFFICULTY != (X+1)@DIFFICULTY))){
D: D-1
} else {
S: S + 1
}
}
}
}
result: S+CURRENT
}

loop X over {2..8} {
loop Y over {0..5} {
output [checkstress Xd6 Y 1] named "[X]d6 Str[Y]"
}
}

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + ATTACKMOD >= AC { result: DAMAGE }
 result: 0
}

\Set Enemy AC\
AC: d{13..15}
\Set Attack Modifier\
ATTACKMOD: 6
\Set Damage Modifier\
DAMAGEMOD: 4
\Set Critical Range\
CRITROLL: 20
\Set Number of Attacks\
ATTACKTIMES: 1
\Change Damage Dice\
DAMAGEDICE: 10
\Any Critical modifiers\
CRITMOD: 0
\Don't Change This\
DAMAGE: 1dDAMAGEDICE + DAMAGEMOD
CRITDAMAGE: 2dDAMAGEDICE + DAMAGEMOD + CRITMOD
output ATTACKTIMESd[attackroll d20 vs AC] named "Glaive Attack"

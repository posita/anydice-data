set "maximum function depth" to 5

function: gnghelp D:n E:n {
  if E = 1 {
    if D = 6 { result: 1 + [gnghelp d6 E] }
  }

  if D > 3 { result: 1 }
  result: 0
}

function: gng N:n {
  R: 0
  if N > 3 {
    R: Nd[gnghelp d6 1]
  }
  else {
    R: Nd[gnghelp d6 0]
  }
  result: R
}

loop N over {1..12} {
  output [gng N]
}

function: attack ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + ACCMOD >= AC { result: DAMAGE }
 result: 0
}

ENEMYAC: 18

ATTACKMOD: 5
ACCMOD: 0

PROFICIENCY: 4
CRITROLL: 20
EXTRAATTACK: 2
DMGBONUS: 0
ACCBONUS: 0

DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "Normal Attack"

DAMAGE: 2d{{1..2}, {3..6}:4} + ATTACKMOD + DMGBONUS
CRITDAMAGE: 3d{{1..2}, {3..6}:4} + ATTACKMOD + DMGBONUS

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "GWF Greatsword"

DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS

output (1+EXTRAATTACK)d[attack d20 vs ENEMYAC] named "TWF Shortswords"

DAMAGE: 1d8 + ATTACKMOD + DMGBONUS + 2
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS + 2

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "Dueling Longsword"

ACCMOD: 2
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "Archery Longbow"

ACCMOD: -5
DAMAGE: 2d{{1..2}, {3..6}:4} + ATTACKMOD + DMGBONUS + 10

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "FEAT: GWM"

ACCMOD: -3

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "FEAT: Sharpshooter"

ACCMOD: 0

AC: 1
VEXD: d20
VEXHIT: (VEXD+7 > AC | VEXD=20) & VEXD!=1

SCIMDIE: d20
SCIMADVDIE: d20
SCIMD: SCIMDIE + (VEXHIT * (SCIMADVDIE > SCIMDIE)) * (-SCIMDIE + SCIMADVDIE)

SCIMHIT: SCIMD | SCIMD=20

VEXDMG: 1d6
VEX: ((VEXDMG+5) * VEXHIT) + (VEXDMG * VEXD=20)


SCIMDMG:  (1d6 + (VEXHIT * 1d4))
SCIM: (SCIMHIT * (SCIMDMG+5)) + (SCIMDMG * SCIMD=20)


SNEAK: ((VEXHIT | SCIMHIT) * 2d6) + ((VEXD=20 | SCIMD=20) * 2d6)

DMG: VEX + SCIM + SNEAK

output DMG

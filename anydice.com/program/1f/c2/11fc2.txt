STATS: {2, 2, 1, 2}
HIT: {50, 25, 15, 10}
SPEC_HIT: {45, 30, 15, 10}
SHOT: {20, 45, 20, 15}

function: roll F {
  result: (d{1..(17 - F*2)} + F*2 + d4 - d3)
}

function: action A {
  SUM: 0
  loop N over {1..5}{
    SUM: SUM + [roll N@STATS] * N@A
  }
  result: SUM / 100
}

output [action HIT] named "hit"
output [action SPEC_HIT] named "special hit"
output [action SHOT] named "shot"

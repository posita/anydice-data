function: rollies ROLL:n to DICE:n  {
 if DICE < ROLL {
  result: 1
 } else { result: 0 }
}

function: avandra ROLL:n {
 if ROLL = 7 | ROLL = 12 {
  result: 1
 } else { 
  result: [avandrahelp ROLL plus 1d6]
 }
}

function: avandrahelp A plus DICE:n  {
 if A + DICE = 7 | A + DICE = 12 {
  result: 2
 } else { result: 0 }
}

function: probe ROLL:n {
 if ROLL < 7 {
  result: 0
 }
 else if 9 < ROLL {
  result: 2
 }
 else 
 {
  result: 1
 }
}


function: reroll ROLL:n {
 if ROLL = 1 {
  result: [rerolla d6]
 }
 else {
  result: ROLL
 }
}

function: rerolla ROLL:n {
 if ROLL = 1 {
  result: [rerollb d6]
 }
 else {
  result: ROLL
 }
}

function: rerollb ROLL:n {
 if ROLL = 1 {
  result: d6
 }
 else {
  result: ROLL
 }
}

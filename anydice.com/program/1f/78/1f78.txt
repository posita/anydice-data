function: altexplode DIE:d {
 MAX: [maximum of DIE]
 result: [altexplode DIE helper]
}

function: altexplode N:n helper {
 if N = MAX { result: N + [altexplode DIE helper] -1 }
 result: N
}

set "maximum function depth" to 5
set "explode depth" to 3

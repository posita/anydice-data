function: roll TIMES:n die SIDES:n modified MOD:n count SUCC:n explode on EXPL:n {
  SUCCESSES: 0
  DIE: 1dSIDES
  loop N over {1..TIMES} {
    SUCCESSES: SUCCESSES + [DIE modified MOD success on SUCC explode on EXPL]
  }
  result: SUCCESSES
}

function: N:n modified MOD:n success on SUCC:n explode on EXPL:n {
  SUCCESSES: 0
  if N+MOD >= SUCC { SUCCESSES: 1 }
  if N >= EXPL { SUCCESSES: SUCCESSES + [DIE modified N+MOD success on SUCC explode on EXPL] }
  result: SUCCESSES
}

output [roll 2 die 6 modified -4 count 4 explode on 6]

function: o score ROLLEDVALUE:n SKILL:n {
  TOTAL: ROLLEDVALUE
  if TOTAL <= SKILL {
    result: 1
  } else {
    result: 0
  }
}

function: o roll ATTRIBUTE:n SKILL:n {
  TOTAL: 0
  loop N over {1..ATTRIBUTE} {
    TOTAL: TOTAL + [o score 1d6 SKILL]
  }
  result: TOTAL
}

loop ATTRIBUTE over {1..8} {
  loop SKILL over {01..4} {
    output [o roll ATTRIBUTE SKILL] named "Attribute at [ATTRIBUTE] and skill at [SKILL]"
  }
}

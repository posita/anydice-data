function: escalate DIE:d {
MAX: [maximum of DIE]
result: [sizeup DIE up to 8]
}

function: sizeup SIDES:n up to LARGEST:n{
if SIDES >= LARGEST {result: LARGEST}
if SIDES = MAX {result: [check [escalate d(SIDES+2)] against SIDES]}
else {result: SIDES}
}

function: check NEWROLL:n against LOWCLAMP:n{
if NEWROLL > LOWCLAMP {result: NEWROLL}
else {result: LOWCLAMP}
}

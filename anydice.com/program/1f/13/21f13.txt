function:area A:n vs B:n d D:n am AM:n bm BM:n target T:n{
if (A+AA+AM)<(B+BA+BM){
result:[damagearea a A+AA d D target T] 
}
} 

function: damagearea a A:n d D:n target T:n
DI:d
{
if T=ARM{DI:1d4}
if T=SH{DI:2d4}
if T=UK{DI:3d4}
if T=OK{DI:4d4}
if [A >= [lowest 1 of T]]{
ZD:A-[lowest 1 of T] 
if ZD>T{ZD:[highest 1 of T]-[lowest 1 of T]}
result: ASTR+DI+ZD-BSTR
}
}


ASTR:3 BSTR:5
AA:4   BA:5
ARM:{1}
SH:{2..4}
UK:{4..9}
OK:{9..15}
output [area d20 vs d20 dmg 4 am 3 bm 5 target UK]

\**** Damage calculator ****\

function: savingthrow ROLL:n vs DC:n {
 if ROLL + ABILITYMOD <= DC { result: DAMAGE }
 result: 0
}

\Set Spell DC\
DC: 14
\Set Ability Modifier\
ABILITYMOD: {-2..1}
\Set Damage Modifier\
DAMAGEMOD: 0
\Set Number of Targets\
TARGETS: 1
\Change Damage Dice\
DAMAGEDICE: 12
\Don't Change This\
DAMAGE: 1dDAMAGEDICE + DAMAGEMOD
output TARGETSd[savingthrow d20 vs DC] named "Toll The Dead"

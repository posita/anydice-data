INITEIGHTSUM: {0 \6\, 1 \7\, 2 \4\, 3, 4 \2\, 5 \4\, 7}

function: DIE:n plus MOD:n beats EDIE:n plus EMOD:n {
  if DIE+MOD = EDIE+EMOD { result: MOD >= EMOD }
  else { result: DIE+MOD >= EDIE+EMOD }
}
function: DIE:n plus MOD:n improved is better against EDIEA:n plus EMODA:n and EDIEB:n plus EMODB:n {
  A: [DIE plus MOD beats EDIEA plus EMODA]
  AI: [DIE plus (MOD + 4) beats EDIEA plus EMODA]
  B: [DIE plus MOD beats EDIEB plus EMODB]
  BI: [DIE plus (MOD + 4) beats EDIEB plus EMODB]
  \ only better if you wouldn't have beat either of them but now you do \
  result: !A & !B & (AI | BI)
}

\ too slow to do full set; modify sets in brackets to check others \
loop I over {0,2,5} {
  loop J over {2,5} {
    output [1d20 plus 5 improved is better against 1d20 plus I and 1d20 plus J] named "[I], [J]"
  }
}

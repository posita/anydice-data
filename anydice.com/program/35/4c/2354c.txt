\ Psychic Sequence \

function: smite PTESTA:s PTESTB:s {
        if PTESTA + PTESTB >= 11 {result: 1d6}
        if PTESTA + PTESTB >= 5 {result: 1d3}
        result: 0
}

function: shades PTESTA:s PTESTB:s N {
        if PTESTA + PTESTB >= 11 {result: Nd[gate 1d6 5]}
        if PTESTA + PTESTB >= 6 {result: Nd[gate 1d6 6]}
        result: 0
}

function: gate DICE:s N {
        result: DICE >= N
}

output [smite 1d6 1d6] named "Smite"

output [shades 1d6 1d6 6] named "Betraying Shades: 6 Attacks"

output [shades 1d6 1d6 12] named "Betraying Shades: 12 Attacks"

output [shades 1d6 1d6 18] named "Betraying Shades: 18 Attacks"

output [shades 1d6 1d6 24] named "Betraying Shades: 24 Attacks"

output [shades 1d6 1d6 30] named "Betraying Shades: 30 Attacks"

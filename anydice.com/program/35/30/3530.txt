function: roll DIE:n {
 if DIE = 2 { Y:1d10 Z:0}
 if DIE = 3 { Y:1d10 Z:1d10}
 X:1d10 
 result: [explode X Y Z helper]
}

function: explode X:n Y:n Z:n helper {
 if X = Y { if Y = Z {result: X+Y+Z} result: X+Y }
 if X = Z { result: X+Z }
 if Y = Z { result: Y+Z }
 result:[highest of X and [highest of Y and Z]]
}
output [roll 3]

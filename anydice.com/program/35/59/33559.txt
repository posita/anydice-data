function: roll N:n rerolling R:n exploding E:n limit L:n {
  if L < 2 {
    result: N
  }
  if N <= R {
    result: [roll 1d6 rerolling 0 exploding E limit L - 1]
  }
  if N >= E {
    result: N + [roll 1d6 rerolling 1 exploding E limit L - 1]
  }
  result: N
}

SUM: 0
loop I over {1..9} {
  SUM: SUM + [roll 1d6 rerolling 1 exploding 6 limit 10]
}
output SUM

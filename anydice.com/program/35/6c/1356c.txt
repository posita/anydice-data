function: fudge ROLL:n for TARGETS:n {
  if ROLL > 1 { result: TARGETS }  \ all make the save \
  if ROLL < 1 { result: 0 }        \ all fail the save \
  result: 1d(TARGETS-1)            \ some make it, some fail \
}

X: 9  \ number of targets \

loop N over {3, 6, 11, 16, 19} {  \ minimum roll for successful save \
  output Xd(d20 >= N) named "[X] individual saves vs. [N]"
  output [fudge 2d(d20 >= N) for X] named "fudged saves vs. [N] for [X] targets"
}

function: ova ROLL:s {
  ONES: ROLL = 1
  TWOS: {ROLL = 2} * 2
  THREES: {ROLL = 3} * 3
  FOURS: {ROLL = 4} * 4
  FIVES: {ROLL = 5} * 5
  SIXES: {ROLL = 6} * 6
  result: [seqhighest {ONES, TWOS, THREES, FOURS, FIVES, SIXES}]
}

function: seqhighest SEQ:s {
  MAX: 0
  loop POS over {1..#SEQ} {
    if {POS@SEQ} > MAX {
      MAX: {POS@SEQ}
    }
  }
  result: MAX
}

output [ova 5d6]

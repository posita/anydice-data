KPW: 0 \ile k4 z poziomu wyzwania\ 
SZ: 0 \ile k4 ze stanu zdrowia\ 
WO: 6 \ile k6 z Wewnętrznego ognia\
T: 1 \ile k8 z Talentu\
PK: 1 \ile kości k8 z podpalenia, automatycznie odejmie kostke k6\

CZWORKI: KPW+SZ
SZOSTKI: WO-PK
OSEMKI: T+PK

\dodane warunki bo jesli dwie zmienne dla puli wynosily 0 to wtedy psulo wyniki\

if CZWORKI > 0{
 WYNIK_CZWOREK: CZWORKId(d4 = 1)
}
else{
 WYNIK_CZWOREK: 0
}

if OSEMKI > 0{
 WYNIK_OSEMEK_POZYTYWNY: OSEMKId(d8 > 5) 
 WYNIK_OSEMEK_NEGATYWNY: OSEMKId(d8 = 1) 
}
else{
 WYNIK_OSEMEK_POZYTYWNY: 0
 WYNIK_OSEMEK_NEGATYWNY: 0
}

if SZOSTKI > 0{
 WYNIK_SZOSTEK_POZYTYWNY: SZOSTKId(d6 = 6) 
 WYNIK_SZOSTEK_NEGATYWNY: SZOSTKId(d6 = 1) 
}
else{
 WYNIK_SZOSTEK_POZYTYWNY: 0
 WYNIK_SZOSTEK_NEGATYWNY: 0
}


\wyniki\
\=========\

output WYNIK_OSEMEK_POZYTYWNY + WYNIK_SZOSTEK_POZYTYWNY named "Chances for Successes count"
output WYNIK_OSEMEK_NEGATYWNY + WYNIK_SZOSTEK_NEGATYWNY + WYNIK_CZWOREK named "Chances for Failures count"




\brudnopis\
\=========\

\output (PK+T)d(d8 > 5) > 0 + (WO-PK)d(d6 = 6) > 0 named "Total Successes"\
\output (PK + T)d(d8 > 5) > 0 + (WO-PK)d(d6 = 6) > 0 named "Total Successes"\
\output (PK + T)d(d8 = 1) > 0 + (WO-PK)d(d6 = 1) > 0 + (KPW+SZ)d(d4 = 1) named "Total Failures"\

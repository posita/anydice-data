function: dug TARG:n with DICE:n dice
{
    result:  [dughelper TARG with DICE rolling DICEd{0,0,1}]
}

function: dughelper TARG:n with DICE:n rolling ROLL:n
{
    WIN: 0
    if (ROLL >= TARG) {
        result: 1
    }
    else if (ROLL > 0) {
        result: [dug (TARG-ROLL) with DICE dice]
    }
    result: 0
}

output [dug 2 with 2 dice] named "2 dice needing 2 hits"
output [dug 3 with 2 dice] named "2 dice needing 3 hits"
output [dug 2 with 3 dice] named "3 dice needing 2 hits"
output [dug 3 with 3 dice] named "3 dice needing 3 hits"

function: reroll N:n once {
 if N <= 2 { result: d4 }
 result: N
}
function: longbow AR:n MOD:n AC:n{
 if AR+MOD >= AC {result: 1d8+5}
 result: 0
}
function: sharpshoot AR:n MOD:n AC:n{
 if AR+MOD >= AC {result: 1d8+15}
 result:0
}
\output 4d[reroll d4 once]\
AC: 17

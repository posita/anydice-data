function: roll A:n plus B:n vs D:n {
 if A=24 { result: 4 }
 if A=2 { result: 1 }
 if A+B+A+B<=D { result: 1 }
 if A<6 { result: 2 }
 if A+B>=D+D { result: 4 }
 if A+B>=D { result: 3 }
 if A+B<D { result: 2 }
}

loop N over {0}{
output [roll 2d12 plus N vs [lowest 2 of 3d12]]
output [roll 2d12 plus N vs 2d12]
output [roll 2d12 plus N vs [highest 2 of 3d12]]
output [roll 2d12 plus N vs 3d12]
}

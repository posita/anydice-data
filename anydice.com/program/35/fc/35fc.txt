\ adjust threat \

THREAT: 15

\ use transposed graph \

\ leave rest as is \
MISS: 0
NORMALHIT: 1
CRITICALHIT: 2

ATTACK: 22 + 1
CRITPLUS: 8

function: ROLL:n vs AC:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  result: [threat d20]
 }
 if ROLL >= THREAT & ROLL + ATTACK >= AC {
  result: [threat d20]
 }
 if ROLL + ATTACK >= AC { result: NORMALHIT }
 result: MISS
}

function: threat ROLL:n {
 if ROLL = 1 { result: NORMALHIT }
 if ROLL + ATTACK + CRITPLUS >= AC | ROLL = 20 {
  result: CRITICALHIT
 }
 result: NORMALHIT
}

loop AC over {25..40} {
 output [d20 vs AC] named "[AC]"
}

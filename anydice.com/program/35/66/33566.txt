function: checkfumbles N:n {
 if N = 2 {result:0}
 result: N
}

function: edgeroll D:n edge E:n {
 if D <= E {D: D + [edgeroll d10 edge E]}
 result: D
}

\ output [checkfumbles 2d10] \

output 2d[edgeroll d10 edge 1] named "2d10 with an edge of 1"

output 2d[edgeroll d10 edge 2] named "2d10 with an edge of 2"

output 2d[edgeroll d10 edge 3] named "2d10 with an edge of 3"

function: fudge ROLL:n for TARGETS:n {
  if ROLL > 1 { result: TARGETS }  \ all make the save \
  if ROLL < 1 { result: 0 }        \ all fail the save \
  result: 1d(TARGETS-1)            \ some make it, some fail \
}

loop N over {3, 6, 11, 16, 19} {
  output 9d(d20 >= N) named "9 individual saves vs. [N]"
  output [fudge 2d(d20 >= N) for 9] named "fudged saves vs. [N] for 8 targets"
}

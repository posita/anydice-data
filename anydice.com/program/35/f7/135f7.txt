function: getone ROLL:s {
  MODTWOA: [1@ROLL mod 2]
  MODTWOB: [2@ROLL mod 2]
  MODTWOC: [3@ROLL mod 2]
  if MODTWOA = MODTWOB {
    result: 3@ROLL
  }
  if MODTWOA = MODTWOC {
    result: 2@ROLL
  }
  if MODTWOC = MODTWOB {
    result: 1@ROLL
  }
 
  result: 7
}

function: VAL:n mod DIV:n {
  result: VAL - DIV*(VAL/DIV)
}

output [getone 6d6]

\\\*** Base Damage Calc Info ***\\\
ENEMY_AC: d{10..20}  \looped against enemy AC from 10-20\
LEVEL: 10
STAT_MOD: 4

\\\*** Attack A ***\\\
ATTACK_DICE_A: 2d6
BONUS_DAM_MOD_A: 0
BONUS_DICE_A: 0
HIT_BONUS_A: 0
MAGIC_A: 0
N_ATTACKS_A: 1

\\\*** Attack B ***\\\
ATTACK_DICE_B: ATTACK_DICE_A
BONUS_DAM_MOD_B: 10
BONUS_DICE_B: BONUS_DICE_A
HIT_BONUS_B: -5
MAGIC_B: MAGIC_A
N_ATTACKS_B: N_ATTACKS_A

\\\*** General Info ***\\\
CRIT_RANGE: 20
CRIT_BONUS: 0 \Any additional dice used specifically for a critical\
NORMAL_HIT: d20
HIT_ADV: [highest 1 of 2d20]
PROF: (LEVEL+7)/4
REROLL_A: 0
REROLL_B: REROLL_A

\\\*** Functions ***\\\
function: attack ROLL:n vs ENEMY_AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL >= CRIT_RANGE { result: CRITICALDAMAGE }
 if ROLL + ATTACK >= ENEMY_AC { result: HITDAMAGE }
 result: 0
}
function: attackadv ROLL:n vs ENEMY_AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL >= CRIT_RANGE { result: CRITICALDAMAGE }
 if ROLL + ATTACK >= ENEMY_AC { result: HITDAMAGE }
 result: 0
}

function: reroll R:n under N:n {
   if R <= N {result: ATTACK_DICE_A} else {result: R}
}
function: reroll R:n under N:n {
   if R <= N {result: ATTACK_DICE_B} else {result: R}
}

ATTACK: STAT_MOD + PROF + MAGIC_A + HIT_BONUS_A
HITDAMAGE: DICE_NUMBER_Ad[reroll ATTACK_DICE_A under REROLL_A] + STAT_MOD + MAGIC_A + BONUS_DAM_MOD_A + BONUS_DICE_A
CRITICALDAMAGE: 2d(DICE_NUMBER_Ad[reroll ATTACK_DICE_A under REROLL_A]) + STAT_MOD + MAGIC_A + BONUS_DAM_MOD_A + 2dBONUS_DICE_A + CRIT_BONUS
output N_ATTACKS_Ad[attack NORMAL_HIT vs ENEMY_AC] named "Attack A"
output N_ATTACKS_Ad[attackadv HIT_ADV vs ENEMY_AC] named "Adv Attack A"

ATTACK: STAT_MOD + PROF + MAGIC_B + HIT_BONUS_B
HITDAMAGE: DICE_NUMBER_Bd[reroll ATTACK_DICE_B under REROLL_B] + STAT_MOD + MAGIC_B + BONUS_DAM_MOD_B + BONUS_DICE_B
CRITICALDAMAGE: 2d(DICE_NUMBER_Bd[reroll ATTACK_DICE_B under REROLL_B]) + STAT_MOD + MAGIC_B + BONUS_DAM_MOD_B + 2dBONUS_DICE_B + CRIT_BONUS
output N_ATTACKS_Bd[attack NORMAL_HIT vs ENEMY_AC] named "Attack B"
output N_ATTACKS_Bd[attackadv HIT_ADV vs ENEMY_AC] named "Adv Attack B"

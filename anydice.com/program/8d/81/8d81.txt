T: d10
C: d10
FA: 1
FB: d6
FC: d10

function: bonus T:n C:n F:n {
    if T = C { result: T }
    if T = F { result: T }
    result: 0
}

function: chaos DICE:n {
    if DICE = 10 { result: DICE - 10 }
    result: DICE
}
output T+[chaos C]+[bonus T C 0] named "Sans Focus"
output T+[chaos C]+FA+[bonus T C FA] named "Focus 1"
output T+[chaos C]+FB+[bonus T C FB] named "Focus d4"
output T+[chaos C]+FC+[bonus T C FC] named "Focus d6"

L: {1,3,8}
M: {4,6,8}
H: {2,8,11}


function: lo X:d {
  result: [lowest 1 of 2dX]
}

function: hi X:d {
  result: [highest 1 of 2dX]
}

function: S:s at A:n with B:n in Z:n {
  if Z > A {
    result: 1@S
  } else if Z > A-B {
    result: 2@S
  } else {
    result: 3@S
} }

output [M at 14 with d8 in d20]
output [M at 14 with [lo d8] in d20]
output [L at 14 with d8 in [lo d20]]
output [H at 14 with [lo d8] in d20]

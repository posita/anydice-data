\ Exploding Dice \
\ Adventures Beyond 0.3.0 \

\ All Dice Behavior \
set "maximum function depth" to 4
function: roll ROLL:n threshold T{
  if ROLL >=T {result: 1}
  if ROLL =1 {result:0}
  result: 0
}

\ D4 Behavior \
function: rollexplodingfour ROLL:n threshold T{
  if ROLL =4 {result: 1+[rollexplodingfour 1d4 threshold T]}
  result: [roll ROLL threshold T]
}

function: rerollfailuresfour ROLL:n threshold T{
  if ROLL < T {result: [roll 1d4 threshold T]}
  result: [roll ROLL threshold T]
}

function: rerollthenexplodefour ROLL: n threshold T {
  if ROLL < T {result: [rollexplodingfour 1d4 threshold T]}
  result: [rollexplodingfour ROLL threshold T]
}

function: wrapperfour DICE:n threshold T explode E reroll R{
  RES:0
  loop N over {1..DICE} {
    if E & R {RES: RES+[rerollthenexplodefour 1d4 threshold T]}
    else {if E {RES:RES+[rollexplodingfour 1d4 threshold T]}
      else {if R {RES:RES+[rerollfailuresfour 1d4 threshold T]}
        else {RES:RES+[roll 1d6 threshold T]}
      }
    }
  }
  result:RES
}

function: four DICE:n{
  result:[wrapperfour DICE threshold 2 explode 0 reroll 0]
}
function: explodefour DICE:n{
  result:[wrapperfour DICE threshold 2 explode 1 reroll 0]
}
function: rerollfour DICE:n{
  result:[wrapperfour DICE threshold 2 explode 0 reroll 1]
}
function: explodereroll DICE:n{
  result:[wrapperfour DICE threshold 4 explode 1 reroll 1]
}

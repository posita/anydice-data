DC: 15
DMG: 8d8
function: cloakless ROLL:n MOD:n{
 if ROLL + MOD >= DC {
  result: DMG / 2
 } else {
  result: DMG
 }
}
function: cloak ROLL:n MOD:n {
 if ROLL + MOD - 1 >= DC {
  result: 0
 } else {
  result: DMG
 }
}

CON: 2

output [cloakless d20 CON] named "cloakless"

output [cloak d20 CON-1] named "cloak"

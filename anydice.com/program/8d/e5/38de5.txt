TARGET: 5

function: custom die N:n {
  result: d{0:(TARGET-1), TARGET:(N-TARGET), N}
}

SIX: [custom die 6]
TWELVE: [custom die 12]

function: roll SIXES:s TWELVES:s angst P:n {
  SUCCESSES: {SIXES, TWELVES} >= TARGET
  B: [lowest of (TWELVES = 12) and P]
  A: [lowest of (SIXES = 6) and P - B]
  if A + B > 0 { 
    SUCCESSES: SUCCESSES + [roll AdSIX BdTWELVE angst P - A - B]
  }
  result: SUCCESSES
}

loop P over {0..5} {
  output [roll 3dSIX 2dTWELVE angst P] named "Xd6, Yd12, angst [P]"
}

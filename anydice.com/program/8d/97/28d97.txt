ABILITY: 1
MODIFIER: 0


BONUS: SKILL+ABILITY+MODIFIER

function: reroll ROLL:n DICE:n {
  if ROLL < TARGET {
    result: [highest 2 of DICEd6]+BONUS
  } else {
    result: ROLL
  }
}

function: challenge ROLL:n {
  if ROLL >= TARGET {
    result: 1
  } else {
    result: 0
  }
}

M: {2,5,7}
H: {1,6,10}


function: lo X:d {
  result: [lowest 1 of 2dX]
}

function: hi X:d {
  result: [highest 1 of 2dX]
}

function: S:s at A:n with B:n in Z:n {
  if Z > A {
    result: 1@S
  } else if Z > A-B {
    result: 2@S
  } else {
    result: 3@S
} }

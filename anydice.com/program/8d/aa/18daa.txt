\**** Damage calculator duelling Vs archery ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 12-18\
ENEMYAC: d{12..18}

ATTACKMOD: 5
DMGBONUS: 4
PROFICIENCY: 3
ACCBONUS: 1
CRITROLL: 20
ATTACKTIMES: 3
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 1d8 + 8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Orlün gauntlets"

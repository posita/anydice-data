function: resolve HITS:n using STR:n against TOUGH:n with SHARP{
  TOTAL: 0
  loop N over {1..HITS}{
   ROLL:1d10
   ADD: 0
   if SHARP > 0 { ADD: 1d10 }
   TOTAL: TOTAL + ((ROLL + STR + ADD) >= TOUGH)
  }
  result: TOTAL
}

function: bow COUNT:n over TARGET:n with STR:n for TOUGH:n{
 TOTAL: 0
 loop N over {1..COUNT}{
   ROLL:1d10
   TOTAL: TOTAL+(ROLL >= TARGET)
 }
 result: [resolve TOTAL using STR against TOUGH with 0]
}

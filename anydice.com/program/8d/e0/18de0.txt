function: luck ROLLA:n and ROLLB:n{
 RES: 0

 if ([count {1..3} in ROLLA]) {
   RES: RES +1
 }
 if ([count {8..10} in ROLLA]) {
   RES: RES -1
 }

 result: RES
}
LUCK: [luck d10 and d10]

N: 4
DADO: 10
DIFS: [reverse {1..9}]

loop D over DIFS {
  DD: 11-D
  output [count {1..D} in NdDADO] + LUCK named "DIF [DD] ([N]D[DADO])"
}

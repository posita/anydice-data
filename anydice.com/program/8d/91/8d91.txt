\Conan Skill Success probabilities\

FOC:3   \ <---- set your focus\
TN:12   \ <---- set you target number (skill+ability)\

function: ROLL:s target T:n focus F:n {
   if ROLL<=F {result:2}
   result: (ROLL<=T)
}

loop S over {2..5} {
  output [Sd20 target TN focus FOC] named "[S]d20 Target [TN] Focus [FOC]"
}

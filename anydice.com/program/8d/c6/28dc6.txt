\ User-Modifiable variables! \
NUMDICE: {3,4,5,6}  \ Number of d6's to roll (e.g. 5d6), before keeping the 3 lowest (min 3, empty array graphs none!)\
POSITION: {1,2,3}   \ 1 = Lowest of three dice, 2 = Middle Die, 3 = Highest of three dice {1..3} \
DISADVANTAGE: 1     \ Whether to also roll 2 and take lower, thrice (0=no, 1=yes)                \

\ Makes the order of dice results go in incremental order from least to greatest die: \
set "position order" to "lowest first" 


loop INDEX over POSITION{

 if DISADVANTAGE = 1 {
  \ Rolls 2d6, takes the lower, then repeats for a total of 3 results \
  output INDEX@3d(1@2d6) named "Die [INDEX] of 3x DisAdv2d6"
 }

 loop NUM over NUMDICE{
  output INDEX@NUMd6 named "Die [INDEX] of [NUM]d6"
 }
}

function: silhouettish D:s high only HIGHONLY:n {
  HIGHEST: 1@D
  OUTPUT: HIGHEST
  LAST: HIGHEST

  loop I over {2..#D} {
    if I@D = HIGHEST {
       OUTPUT: OUTPUT + 1
    }

    if !HIGHONLY & I@D < HIGHEST & LAST = I@D {
       OUTPUT: OUTPUT + 1
    }

    LAST: I@D
  }

  result: OUTPUT
}

loop N over {1..8} {
   output [silhouettish Nd{0,0,0,1..9} high only 0] named "[N]d"
}

function: attack ROLL:n TARGET:n DAM:n {
  if ROLL = 20 { result: 2 * DAM }
  if ROLL >= TARGET { result: DAM }
  result: 0
}

DAM: 1d10
loop AC over {1..20} {
  output [attack 1d20 AC DAM]/2 named "Attack against [AC] with resistance"
  output [attack 2@2d20 AC DAM] named "Attack against [AC] with disadvantage"
}


function: damage of X:n salvo missiles {
  result: Xd6
}

function: X:n pds rolls on salvo missiles {
  if X<=0 {
    result: 0
  }
  result: [explode d{0,0,0,1,1,2}] + [X-1 pds rolls on salvo missiles]
}

function: salvo missile attack against X:n pds {
  result:[highest of [damage of (1d6 - [X pds rolls on salvo missiles]) salvo missiles] and 0]
}

function: X:n pds rolls on heavy missiles {
  if X<=0 { result: 0 }
  result: d{0,0,1} + [(X-1) pds rolls on heavy missiles]
}

function: heavy missile attack against X:n pds {
  PDS_HITS: [X pds rolls on heavy missiles]
  result: [highest of 1 - PDS_HITS and 0] * 3d6
}

output [salvo missile attack against 0 pds] named "Salvo no PDS"
output [salvo missile attack against 1 pds] named "Salvo 1 PDS"
output [salvo missile attack against 2 pds] named "Salvo 2 PDS"
output [salvo missile attack against 6 pds] named "Salvo 6 PDS"
output [heavy missile attack against 0 pds] named "Heavy no PDS"
output [heavy missile attack against 1 pds] named "Heavy 1 PDS"
output [heavy missile attack against 2 pds] named "Heavy 2 PDS"
output [heavy missile attack against 6 pds] named "Heavy 6 PDS"

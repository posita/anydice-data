function: wild D {
  result: [highest of [1e6] and [explode D]]
}

function: succ D {
  result: D/4
}
function: succ D at T {
  result: (D-T+4)/4
}
function: retry C times T {
  if T <= 0 {
    result: C
  } else {
    result: C | [retry C times T-1]
  }
}
function: retry C {
  result: [retry C times 1]
}
function: XeY {
  result: Xd[explode dY]
}

function: explode DICE:d BOUNCES:n
{
 MAX: [ maximum of DICE ] / #DICE
 result: [ explode DICE BOUNCES 0 helper ]
}

function: explode SEQ:s BOUNCES:n TMP:n helper
{
 if BOUNCES = 0 {result: TMP+SEQ}
 EXTRA_DICE: [lowest of SEQ = MAX and BOUNCES]
 if EXTRA_DICE = 0 {result:TMP+SEQ}
 result: [ explode 1dMAX BOUNCES-EXTRA_DICE SEQ+TMP helper]
}

output [explode 2d8 4] named "Sorcerous Burst ab Level 5"

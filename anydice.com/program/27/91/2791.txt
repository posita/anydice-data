function: interpret ROLL:s {
 if [count {5,6} in ROLL] { result: [count {5,6} in ROLL] } \ success \
 if [count 1 in ROLL] { result: -1 } \ success \
 result: 0
}


function: compare N:n vs M:n {
ATT: [interpret N d6]
DEF: [interpret M d6]
result: ATT-DEF
}

loop DA over {1..10} {
  loop DB over {1..DA} {
    output [compare DA vs DB] named "[DA] vs [DB]"
  }
}

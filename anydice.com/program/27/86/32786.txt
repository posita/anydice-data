set "maximum function depth" to 3
BASEDIE: d6
LIMIT: 3

function: target TARGET:n reroll REROLL:n VALUE:n {
  if VALUE = REROLL {
    if LIMIT > 0 {
    LIMIT: LIMIT -1
    result: 1 + [target TARGET reroll REROLL BASEDIE]
    }
    else {
      result: 1
    }
  }
  if VALUE  > TARGET {
    result: 1
  }
  result: 0
}

DIE: [target 3 reroll 6 BASEDIE]

loop N over {1..10}{
  output NdDIE named "Dicepool [N] Limit [LIMIT]"
}

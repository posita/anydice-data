function: explode DICE:d REPEATS:n {
 MAX: 1@{[ reverse {DICE}]} / #DICE
 result: [explode DICE REPEATS 0 helper]
}

function: explode SEQ:s REPEATS:n CARRY:n helper {
 if REPEATS = 0 { result: SEQ+CARRY }
 EXTRA_DICE: [lowest of (SEQ = MAX) and REPEATS]
 if EXTRA_DICE = 0 { result: SEQ+CARRY }
 result: [explode EXTRA_DICEdMAX REPEATS-EXTRA_DICE SEQ+CARRY helper]
}

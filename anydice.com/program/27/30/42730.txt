DC: 6 \ number of dice \
DF: 7 \ difficulty \

HS:   4 \ critical success \
TS:   2 \ success \
MS:   0 \ partial success \
FF:  -2 \ failure \ 

POOLSIX: (DC)d{0,0,0,1,1,1}
POOLSEX: [explode POOLSIX]

function: check ROLL:n {
       if ROLL >= DF+4 { result: HS }
  else if ROLL >= DF+2 { result: TS }
  else if ROLL >= DF+1 { result: MS }
  else if ROLL >= DF-1 { result: MS }
  else                 { result: FF } 
}

output [check POOLSIX] named "[DC]d6 Pool"
output [check POOLSEX] named "Exploding [DC]d6 Pool"

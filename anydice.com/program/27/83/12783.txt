
function: closest DICE:n plus RANGE:s to TARGET:n {
    CLOSEST: DICE + 1@RANGE
    loop N over RANGE {
        if [absolute DICE + N - TARGET] < [absolute CLOSEST - TARGET] {
            CLOSEST: DICE + N
        }
    }
    result: CLOSEST
}

loop TARGET over {3..18} {
    output [closest 2d6 plus {1..6} to TARGET] named "Closest to [TARGET] of 2d6 + 1 through 6 "

}

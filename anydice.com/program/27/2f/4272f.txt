DC: 1 \ number of dice \
DF: 0 \ difficulty \

CS:   3 \ critical success \
SS:   1 \ success \
MS:   0 \ partial success \
FF:  -1 \ failure \ 

POOLSIX: (DC)d{0,0,0,1,1,1}
POOLSEX: [explode POOLSIX]

function: check ROLL:n {
  if ROLL > DF+2 {result: CS}
  else if ROLL > DF+1 { result: SS }
  else if ROLL > DF { result: MS }
  else if ROLL > DF-2 { result: MS }
  else { result: FF } 
}

output [check POOLSIX] named "[DC]d6 Pool"
output [check POOLSEX] named "Exploding [DC]d6 Pool"

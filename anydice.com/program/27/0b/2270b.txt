function: roll POOL:s keep D:n {
  KEPT: {}
  SPOOL: [sort POOL]
  loop I over {1..D} {
    KEPT: {KEPT, I@SPOOL}
  }
  result: KEPT
}

function: check PAIR:n {
  if PAIR >= 11 {
    result: 1
  }
  else if PAIR >= 6 {
    result: 0
  }
  else {
    result: -1
  }
}

loop STAT over {-1..3} {
  POOL: 2 + STAT
  output [check [roll POOLd6 keep 2]] named "fail/partial/success at stat=[STAT]"
}

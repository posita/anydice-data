function: convert SUM:n {
  if SUM >= 1000 {
    SIXESROLLED: SUM / 1000
    ROLL: SUM - SIXESROLLED * 1000
    result: SIXESROLLED * 1000 
  }
  result: SUM
}

function: multiplySUM:n {
  if SUM >= 1000 {
    SIXESROLLED: SUM / 1000
    ROLL: SUM - SIXESROLLED * 1000
    result: 2 * ROLL 
  }
  result: SUM
}

loop NUM over {1..10} {
 output NUMd{0..3, 0, 0} named "[NUM]d6-3 "
 output [multiply NUMd{0..3, 0, 1000}] named "[NUM]d6-3 multiply sixes"
 output [convert NUMd{0..3, 0, 1000}] named "[NUM]d6-3 explode sixes"
}

function: dtwenty system roll targeting T:n {
  result: d20 >= T
}


function: rollone system roll D:n {
  result: D = 1
}

function: rollone system roll D:d count N:n{
  if N = 1{
    result:[rollone system roll D]
  }
  result: [rollone system roll D] | [rollone system roll D count N-1]
}

output [dtwenty system roll targeting 1]  named "D20 1"

function: save ROLL:n TARGET:n FULLDAM:n SAVEDAM:n {
  if ROLL >= TARGET { result: SAVEDAM }
  result: FULLDAM
}

loop DC over {1..20} {
  output [save 1d20 DC 4 2]/2 named "Save for half against [DC] with resistance"
  output [save 1@2d20 DC 4 2] named "Save for half against [DC] with advantage"
  output [save 1d20 DC 4 0]/2 named "Save for none against [DC] with resistance"
  output [save 1@2d20 DC 4 0] named "Save for none against [DC] with advantage"
}

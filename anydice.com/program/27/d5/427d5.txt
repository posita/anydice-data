function: roll ROLL:n {
 if ROLL = 1 {result: ROLL-10}
 if ROLL > 1 {if ROLL < 20 {result: ROLL}}
 if ROLL = 20 {result: ROLL+10}
}

function: DICE:d NUMBER:n v {
 NUMBER: NUMBER +1
 result: 1@NUMBERdDICE
}
function: DICE:d v {
 result: 1@2dDICE
}

function: ROLL:n dh {
 H: ROLL
 F: ROLL

 if H > F {result: H+F}
 if H < F {result: (H+F)*(-1) }
 if H = F {result: H+F +100}
}

output [1d12 dh]

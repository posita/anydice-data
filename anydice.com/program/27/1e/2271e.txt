function: ROLL:n vs DEFENSE:n ATTACK:n {
 if (ROLL+ATTACK) >= (10+DEFENSE) {
   result: HIT
 }
 result: MISS
}

function: damage for ATTACKRESULT:n with ARMOUR:n {
 if ATTACKRESULT = HIT {
  THIS_DMG: (HITDAMAGE-ARMOUR)
  result: [highest of THIS_DMG and 0]
 }
 result: MISSDAMAGE
}

function: attacks NUMBEROFATTACKS:n {
  TOTAL_DMG: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL_DMG: TOTAL_DMG + [damage for [d20 vs DEF ATK] with AMR]
  }
  result: TOTAL_DMG
}

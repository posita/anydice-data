function: blades ROLL:s{
 if ROLL <= 0{
  result:0
 }
 if ROLL = 1{
  result:1
 }
 if ROLL = 2{
  result:2
 }
 result:3
}





function: max A:n B:n{
    result: 1@[sort {A, B}]
}


loop N over {1..18}{
  A: [count 6 in Nd6]
  B: [count 6 in Nd6]
  AA: [blades A]
  BB: [blades B]
  output [max AA BB] named "[N]d6"
}

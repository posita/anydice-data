FALHA_FEAR: 0
FALHA_HOPE: 1
SUCESSO_FEAR: 2
SUCESSO_HOPE: 3
CRITICO: 4

function: ataque HOPE:n FEAR:n vs DEFESA:n{

 if HOPE = FEAR {result: CRITICO}

 if HOPE + FEAR + ATAQUE >= DEFESA{
  if HOPE > FEAR {result: SUCESSO_HOPE}
  result: SUCESSO_FEAR
 }

 if HOPE > FEAR {result: FALHA_HOPE}
 result: FALHA_FEAR
}

function: dano do RESULTADO:n{
 if RESULTADO = CRITICO {result: DANOCRITICO}
 if RESULTADO = SUCESSO_HOPE {result: DANO}
 if RESULTADO = SUCESSO_FEAR {result: DANO}
 result: 0
}

function: controle RESULTADO:n{
 if RESULTADO = FALHA_FEAR {result: 1}
 if RESULTADO = FALHA_HOPE {result: 1}
 if RESULTADO = SUCESSO_FEAR {result: 1}
 result: 0
}

function: ataque completo {
 result: [dano do [ataque HOPE FEAR vs DEFESA]]
}

function: resultado {
 result: [ataque HOPE FEAR vs DEFESA]
}

function: controle mestre {
 result: [controle [ataque HOPE FEAR vs DEFESA]]
}

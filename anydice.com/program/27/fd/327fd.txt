function: gwf D:d  {
NUM: 1@{D}
DIE: [maximum of D]/1@{D}
ROLL: 1dDIE
ROLLTWO: 1dDIE
X:0
loop NUM over {1..NUM} {
X: X+[ROLL gwf ROLLTWO reroll {1,2}]
}
result: X
}

function: ROLL:n gwf REROLL:n reroll SEQUENCE:s{
  if [SEQUENCE contains ROLL] {
    result: REROLL
  }
  else {
    result: ROLL
  }
}

STR: 4

output [gwf 2d6]+STR named "Greatsword"
output [gwf 1d10]+STR named "Halberd"
output 1d8+2+STR named "Flail"
output 1d4+2+STR named "Whip"

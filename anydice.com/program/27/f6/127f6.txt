WEEKS: 2
ADV: 1

function: convert ROLL:s {
  R: 0
  if ROLL <= 20 {
    R: -3
  } else if ROLL <= 30 {
    R: -2
  } else if ROLL <= 40 {
    R: -1
  } else if ROLL > 90 {
    R: 5*3d10
  } else if ROLL > 80 {
    R: 5*2d8
  } else if ROLL > 60 {
    R: 5*1d6
  }
  result: R    
}

if ADV = 1 {
  NUM: [highest 1 of 2d100]+[lowest of WEEKS*7 and 30]
} else {
  NUM: 1d100+[lowest of WEEKS*7 and 30]
}
SUM: 0
loop I over {1..WEEKS}{
  SUM: SUM + [convert NUM]
}
output SUM

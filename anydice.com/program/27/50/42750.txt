function: maximize A for B {
  result: A + [maximum of A] * (B - 1)
}

function: splitup DIE:n into DIVISOR {
  MULTIDIV: DIE * 10 / DIVISOR
  DIV: DIE / DIVISOR
  DIVMULTI: DIV * 10
  if MULTIDIV - DIVMULTI > 0
    {result: DIV + 1}
  result: DIV
}

W: 2

function: convert ROLL:s {
  R: 0
  if ROLL <= 20 {
    R: -3
  } else if ROLL <= 30 {
    R: -2
  } else if ROLL <= 40 {
    R: -1
  } else if ROLL > 90 {
    R: 5*3d10
  } else if ROLL > 80 {
    R: 5*2d8
  } else if ROLL > 60 {
    R: 5*1d6
  }
  result: R    
}

N: [highest 1 of 2d100]+[lowest of W*7 and 30]
S: 0
loop I over {1..W}{
  S: S + [convert N]
}
output S

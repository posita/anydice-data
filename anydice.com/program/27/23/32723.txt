\ hardcaml \

CHA:     24   \Charisma\
LVL:     12   \Player level\
BONUS:   2    \Critical Strike Bonus\
FIXCRIT: 2    \Fixed Crit #, for Luck of the Realms set to 1\

SNEAK:   1    \Enable full sneak attack abuse\
ROGUE:   8    \Levels of Rogue\
SKUL:    1    \Skullduggery Gloves\

RHAP:    1    \Rhapsody\
SPLSPK:  1    \Spellsparkler\
ALUVE:   1    \Phalar Aluve\
HEX:     1    \Hex\

AGON:    1    \Agonizing Blast\
PO_ROBE: 1    \Potent Robe\
GLOW:    1    \Callous Glow Ring\

VULN:    1    \Wet Vulnerability\
PS:      1    \Perilous Stakes\

if PS > 1 { VULN: 1 }

function: crit D:n with B:n {
   if D >= 20-B { result: 1 } else { result: 0 }
}

function: every D:n {
   if D = 1 { result: 1 } else { result: 0 }
}

function: AB:n mod {
   result: (AB - 10)/2
}

ADV: [highest 1 of 2d20]
CRIT: [crit ADV with BONUS]
SDICE: (ROGUE+1)/2

PROF: 1+(LVL+3)/4
CANT_SCALE: 1 + LVL/5

if FIXCRIT > CANT_SCALE { FIXCRIT: CANT_SCALE }

SR:   RHAP*3
GC:   PO_ROBE*[CHA mod]
AB:   AGON*[CHA mod]
LC:   VULN*SPLSPK
CG:   GLOW*2
HX:   HEX*(1+CRIT)d6
RIDE: SR + CG + HX + LC + GC + AB

PHAL: ALUVE *((1+CRIT)d4 + RIDE)
SD:   SKUL  *((1+CRIT)d4 + RIDE + PHAL)
SA:   SNEAK *(1+CRIT)*SDICEd6 + SD
EB:   PS    *(1+CRIT)*((1+CRIT)d10 + RIDE + SA)
LBL:  SPLSPK*PS*(VULNd8 + 2*(CG + 1) + d4)

function: fixedcrit N:n {
   CRIT: 1
   HX:   HEX*(1+CRIT)d6
   RIDE: SR + CG + HX + LC + GC + AB

   PHAL: ALUVE *((1+CRIT)d4 + RIDE)
   SD:   SKUL  *((1+CRIT)d4 + RIDE + PHAL)
   SA:   SNEAK *(1+CRIT)*SDICEd6 + SD
   EB:   PS    *(1+CRIT)*((1+CRIT)d10 + RIDE + SA)
   
   result: N*EB
}

output [fixedcrit FIXCRIT] + (CANT_SCALE-FIXCRIT)*EB + LBL named "EB | CHA:[CHA] CRITBONUS:-[BONUS] AGON:[AGON] POTENT:[PO_ROBE] ROGUE:[ROGUE] SNK:[SNEAK] STAL:[SKUL] RHAP:[RHAP] SPLSPK:[SPLSPK] GLOW:[GLOW] ALUVE:[ALUVE] HEX:[HEX] WET:[VULN] PERILOUS:[PS] FIXEDCRIT#:[FIXCRIT]"

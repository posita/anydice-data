set "explode depth" to 3
function: explode N:n {
 if N = 1 { result: N - [explode d10] }
 if N = 10 { result: N + [explode d10] }
 result: N
}
function: autofire N:n {
 if N >3 { result: 3}
 if N < 1  { result: 0 }
 result: N
}
set "maximum function depth" to 2

output ([autofire((1d10+14)-(1d10+14))]*2d6 )

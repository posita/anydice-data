\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: 14

ATTACKMOD: 5
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 19
DAMAGE: 1d12 + ATTACKMOD + DMGBONUS +2d8
CRITDAMAGE: 2d12 + ATTACKMOD + DMGBONUS + 6d8
output 4d[attackroll d20 vs ENEMYAC] + 2d6 named "Fgtr1 Greataxe"

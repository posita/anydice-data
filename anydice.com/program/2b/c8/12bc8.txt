CRITICALHIT: 2
HIT: 1
MISS: 0


DAMAGE: 1

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: [power d20] + [power d20]
 }
 if ATTACKRESULT = HIT {
  result: [power d20]
 }
}

function: attack ROLL:n {
 if ROLL = 20 {result: CRITICALHIT}
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: power ROLL:n {
 if ROLL + POWER >= TOUGHNESS { result: DAMAGE }
 result: MISS
}


ATTACK: +2
DEFENSE: 15
POWER: +2
TOUGHNESS: 15

output [attack d20]

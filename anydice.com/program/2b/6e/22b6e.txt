function: sol SEQ:s{
  ONES: [count 1 in SEQ] *1
  TWOS: [count 2 in SEQ] * 2
  THREES: [count 3 in SEQ] * 3
  FOURS: [count 4 in SEQ] * 4
  FIVES: [count 5 in SEQ] * 5
  SIXES: [count 6 in SEQ] * 6
  SEVENS: [count 7 in SEQ] * 7
  EIGHTS: [count 8 in SEQ] * 8
  NINES: [count 9 in SEQ] * 9
  TENS: [count 10 in SEQ] * 10

  result: 1 @ [sort {ONES, TWOS, THREES, FOURS, FIVES, SIXES, SEVENS, EIGHTS, NINES, TENS}]
}

output [sol(1d10)]
output [sol(2d10)]
output [sol(3d10)]

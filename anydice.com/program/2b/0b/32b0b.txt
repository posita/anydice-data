TURRETNUM: 2
TURRETHIT: {0:5,1:1}
TORPEDONUM: 4
TORPEDOHIT: {0:4,1:2}


function: compare A:n B:n {
  if A >= B {
    result: 1
  }
  result: 0
}

function: torpedo survival {
  result: TORPEDONUM - TURRETNUMdTURRETHIT
}

function: eldar torpedoes SURVIVED:n{
  HITS: 0

  loop I over {1..TORPEDONUM} {
    if [compare SURVIVED I] {
      HITS: HITS + [highest of 1dTORPEDOHIT and 1dTORPEDOHIT]
    }
  }

  result: HITS

}

function: torpedoes SURVIVED:n{
  HITS: 0

  loop I over {1..TORPEDONUM} {
    if [compare SURVIVED I] {
      HITS: HITS + 1dTORPEDOHIT
    }
  }

  result: HITS

}

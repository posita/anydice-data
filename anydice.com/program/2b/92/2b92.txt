ACC: {8, 7, 6} \+1 bonus\
DAM: {5,7,10}
HIT: 18
DEF: 12
I: 1
F: 3
loop DEF over {1..36}{
  HITA: [highest of 0 and [lowest of 1@ACC and (HIT)d{0:2,1}] - (DEF)d{0:2,1}]
  HITB: [highest of 0 and [lowest of 2@ACC and (HIT)d{0:2,1}] - (DEF)d{0:2,1}]
  HITC: [highest of 0 and [lowest of 3@ACC and (HIT)d{0:2,1}] - (DEF)d{0:2,1}]
  output HITA>0 named "[DEF]A"
  output HITB>0 named "[DEF]B"
  output HITC>0 named "[DEF]C"
  MDAM: [lowest of F and (3+F)d{0:2,1}]+F
  DAMA: HITA+(1@DAM*(HITA>0))+(MDAM*(HITA>0))
  DAMB: HITB+(2@DAM*(HITB>0))+(MDAM*(HITB>0))
  DAMC: HITC+(3@DAM*(HITC>0))+(MDAM*(HITC>0))
  \output DAMC - DAMA named "[DEF]"\
  \output DAMA named "[DEF]A"\
  \output DAMB named "[DEF]B"\
  \output DAMC named "[DEF]C"\
}

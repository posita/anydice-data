function: check A B {
 if A>B { result:1}
 if A=B { result:0}
 if A<B { result:-1}
}

function: checkdue A:n B:n {
 if A>B { result:1}
 if A=B { result:0}
 if A<B { result:-1}
}


function: sorc A:n B:n C{
 loop N over {1..C}{
  X: [check (N@{A}) (N@{B})]
  if X=1 { result: 1}
  if X=-1 { result: -1}
 }
 result: 0
}

K:10
P:1

output [sorc PdK PdK P]


output [checkdue PdK PdK]

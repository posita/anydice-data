function: sign of N:n {
  if N < 0 { result: -1 }
  if N > 0 { result: 1 }
  result: 0
}

output [highest 1 of 2d4]
output [highest 1 of 2d6]
output [highest 1 of 2d8]
output [highest 1 of 2d10]
output [highest 1 of 2d12]

loop N over {4, 6, 8, 10, 12}{
  loop M over {4, 6, 8, 10, 12} {
    output dN >= dM named "d[N] >= d[M]"
}
}

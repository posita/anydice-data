function: explodetens TENS:n TEMP:n {
if TENS = 0 {result: TEMP}
if TENS = 1 {result: TENS + TEMP}
if TENS > 1 {
result: [explodetens (TENS - 2) (TEMP + 4)]
}
}
function: evaluate ROLL:s {
TENS: [count 10 in ROLL]
SUCCESSES: [count {6,7,8,9} in ROLL]
CRIT: [explodetens TENS 0]
result: SUCCESSES + CRIT
}

loop P over {1..5} {
output [evaluate Pd10] named "[P]d10"
}

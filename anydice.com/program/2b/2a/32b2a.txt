function: tirofuzildecaca ROLL:n {
 if ROLL >= DANTECRITRANGE {
  result: DANTECRIT
 }
 if ROLL < DANTECRITRANGE {
  result: DANTEDMG
 }
}

function: motosserrada ROLL:n {
 if ROLL >= GLEISINCRITRANGE {
  result: GLEISINCRIT + 5
}
 if ROLL < GLEISINCRITRANGE {
  result: GLEISINDMG + 5
 }
}

function: socao ROLL:n {
 if ROLL >= CAIROCRITRANGE {
  result: CAIROCRIT + 5
}
 if ROLL < CAIROCRITRANGE {
  result: CAIRODMG + 5
 }
}

function: acerto ROLL:n to DEF and ACC{
 if ROLL = 20 {
  result: 1
 }
 if ROLL+ACC>= DEF {
  result: 1
 }
 result: 0
}

function: ataquenumbicho ROLL:n plus PROF:n vs DEF:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
 if ROLL = 1 {
  result: 0
 }
 if ROLL >= CRIT_RANGE {
  if ROLL = 20 | ROLL+PROF >= DEF {
   result: CRIT_DMG
  }
 }
 if ROLL + PROF >= DEF {
   result: DMG
 }
 result: 0
}

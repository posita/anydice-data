function: attack ROLL:n and OFFROLL:n and ROCKET:n {
MAINHIT:0
MAINDMG:0
if ROLL + TO_HIT >= AC {MAINDMG: DMG_DICE + DMG_MOD MAINHIT:1}
if ROLL >= CRIT {MAINDMG: 2*DMG_DICE + DMG_MOD MAINHIT:1}
if ROLL = 1  {MAINDMG: 0 MAINHIT: 0}

OFFHIT:0
OFFDMG:0
if OFFROLL + TO_HIT >= AC {OFFDMG: OFF_DMG_DICE + OFF_DMG_MOD OFFHIT:1}
if OFFROLL >= CRIT {OFFDMG: 2*OFF_DMG_DICE + OFF_DMG_MOD OFFHIT:1}
if OFFROLL = 1  {OFFDMG: 0 OFFHIT:0}

ROCKETDMG:0
if MAINHIT = 1 {ROCKETDMG: ROCKET}
if OFFHIT = 1 {ROCKETDMG: ROCKET}

result: MAINDMG + OFFDMG + ROCKETDMG}

DMG_DICE: 1d8
OFF_DMG_DICE: 1d8
DMG_MOD: 5
OFF_DMG_MOD: 0
TO_HIT: 8
OFF_TO_HIT: 8
CRIT: 20

loop AC over {12..23} {
 output [attack d20 and d20 and d4] named "[AC] AC"
}

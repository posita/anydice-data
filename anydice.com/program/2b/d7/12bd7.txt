function: ROLL:n with BONUS against AC:n for DAMAGE:n{
 if ROLL = 20{ result: [CRIT great weapon] + DAMAGE}
 if ROLL = 1 { result: 0 }
 if ROLL + BONUS >= AC { result: [HIT great weapon] + DAMAGE }
 result: 0
}

function: DICE:n great weapon{
 result: DICEd{{3..6}:DICE, {1..2}}
}

STR: 5

ATK: STR + PROF
DMG: STR

ATKGWM: ATK - 5
DMGGWM: DMG + 10

loop N over {1..25}{
  output [d20 with ATK against N for DMG] named "Normal against [N] AC"
  output [d20 with ATKGWM against N for DMGGWM] named "GWM against [N] AC"
}

function: check A:n B:n {
 if A>B { result:1}
 if A=B { result:0}
 if A<B { result:-1}
}

function: checkdue A:s B:s {
 if A>B { result:1}
 if A=B { result:0}
 if A<B { result:-1}
}


function: sorc A:s B:s C{
 loop N over {1..C}{
  X: [check (N@{A}) (N@{B})]
  if X=1 { result: 1}
  if X=-1 { result: -1}
 }
 result: 0
}

function: bladed DICE:s and DACE:s to NUM:n and DOCE:s {
 OUT: 0
 TARGET: 4
 EXPLODE: 0
 if 1@DICE >= TARGET { OUT: OUT + 1 }
 if 1@DICE >= NUM & 1@DACE >= TARGET { OUT: OUT + 1 }
 if 1@DOCE >= TARGET { OUT: OUT + 1 }
 if 1@DOCE >= (TARGET * 2) { OUT: OUT + 1 }
 if 1@DOCE >= (TARGET * 3) { OUT: OUT + 1 } 
 result: OUT
}

loop M over {6,10}{
 loop N over {4,8,12}{
  output [bladed 1dM and 1dM to M and 1dN] named "d[M]/d[N]"
 }
}

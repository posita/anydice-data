\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + CHARISMA - 5 >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 18-25\
ENEMYAC: d{20}

CHARISMA: 6
PROFICIENCY: 6
CRITROLL: 19
ATTACKTIMES: 3
DAMAGE: 1d6 + CHARISMA + CHARISMA + PROFICIENCY + 1d6 + 10
CRITDAMAGE: 2d6 + CHARISMA + CHARISMA + PROFICIENCY + 2d6 + 10


output ATTACKTIMESd[attackroll [highest 1 of 1d20] vs ENEMYAC] named "normal attack"

output ATTACKTIMESd[attackroll [highest 1 of 3d20] vs ENEMYAC] named "attack with super advantage"

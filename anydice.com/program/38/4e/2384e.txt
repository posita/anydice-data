set "maximum function depth" to 5

function: reroll X:n H:n W:n D:n{
 if X>=H {result: [wound 1d6 W D]}
 if X=1 {result: [hit 1d6 H W D]}
 result:0
}

function: hit X:n H:n W:n D:n{
 if X>=H {result: [wound 1d6 W D]}
 result:0
}

function: wound X:n W:n D:n{
 if X>=W {result: D}
 result: 0
}

output [hit 1d6 3 3 1]
output [reroll 1d6 4 3 1]

REROLL: 0
MISSDAMAGE: 1
STRMOD: 5
PROFICIENCY: 6
AC: 16
CRITRANGE: 20
DAMAGE: 2d6

function: roll A:n for damage with reroll B:n {
  if A + STRMOD + PROFICIENCY < AC {result: STRMOD * MISSDAMAGE}
  if A > CRITRANGE - 1 {
    if B = 1 {
      result: [roll d20 for damage with reroll 0] + DAMAGE + ([maximum of DAMAGE]/#(DAMAGE))+ STRMOD
    }
    result: DAMAGE + ([maximum of DAMAGE]/#(DAMAGE))+ STRMOD
  }
  result: DAMAGE + STRMOD
}

output [roll d20 for damage with reroll 1] + [roll d20 for damage with reroll 1] + [roll d20 for damage with reroll 1]

function: explode N:n on SIZE:n {
 if N = SIZE & SIZE > 4 {
  result: N + [explode d(SIZE - 2) on SIZE - 2]
 }
 result: N
}

loop X over {1,2,3,4,5}{
 output (2*[explode 1d((X+1)*2) on ((X+1)*2)])+X named "T[X]"
 output (2*([explode 1d12 on 12]+[explode 1d((X+1)*2) on ((X+1)*2)]))+X+5 named "T[X]-[X]"
}

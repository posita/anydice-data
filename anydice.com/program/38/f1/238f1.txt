PROF: 3
AMOD: 5
DAMR: 2d6
loop AC over {10..20} {
  output (d20 + AMOD + PROF > AC) * (DAMR + AMOD) >=
    (d20 + AMOD + PROF - 5 > AC) * (DAMR + AMOD + 10) named "W[[AC]]"
}

loop AC over {10..20} {
  output ([highest 1 of 2d20] + AMOD + PROF > AC) * (DAMR + AMOD) >=
    ([highest 1 of 2d20] + AMOD + PROF - 5 > AC) * (DAMR + AMOD + 10) named "W[[AC]]"
}

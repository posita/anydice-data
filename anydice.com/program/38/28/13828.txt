function: initiative DIFF:n A:n B:n {
  if (A + DIFF != B) {
    result: A + DIFF > B
  }
  if (DIFF != 0) {
    result: DIFF > 0
  }
  result: d{0,1}
}

loop SIDES over {12,20} {
  loop DIFF over {-SIDES..SIDES} {
    output [initiative DIFF 1d(SIDES) 1d(SIDES)] * 100 named "1d[SIDES], modifier difference: [DIFF]"
  }
}

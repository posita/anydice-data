\ Difficulty of the roll \
DIFF: 8
\ Number of dice to roll \
ND: 10

\ Do not edit below this line \

set "maximum function depth" to 5

function: mbdie N:n {
  result: (N >= DIFF) + (N = 10) - (DIFF = 10)
}

BD: [mbdie 1d10]

function: exdie N:n {
  if N = 2 {
    result: 1 + [exdie BD]
  } else {
    result: N
  }
}

ED: [exdie BD]

output 1dED named "1 exploding die, difficulty [DIFF]"
output NDdED named "[ND] exploding dice, difficulty [DIFF]"

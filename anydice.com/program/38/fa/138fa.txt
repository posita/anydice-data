MISS: 0
NORMALHIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= THREAT {
  result: [threat d20]
 }
 if ROLL + ATTACK >= DEFENSE { result: NORMALHIT }
 result: MISS
}

function: threat ROLL:n {
 if ROLL = 1 { result: NORMALHIT }
 if ROLL + ATTACK + CRITICALCONFIRM >= DEFENSE | ROLL = 20 {
  result: CRITICALHIT
 }
 result: NORMALHIT
}
function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = NORMALHIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}
function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}
function: multiattack NUMATKS:n {
  TOTAL: 0
  loop N over {1..NUMATKS} {
    ATTACK: ATTACK-(5*(N-1))
    DAMAGE: [attack]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

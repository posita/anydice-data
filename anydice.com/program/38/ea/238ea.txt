ATTACKS:5
SKILL:2
CRITRANGE:5
DAMAGE:0
CRIT:0
HIT:0

function: roll DICE:n {
   if DICE >= CRITRANGE{
      DAMAGE:+7
      result:DAMAGE
      }
   if DICE >= SKILL{
      DAMAGE:+5
      result:DAMAGE     
      }
   else {
      result:DAMAGE
      }
   }


function: attacks X{
   TOTAL:0
   loop N over {1..X}{
      TOTAL:TOTAL+[roll 1d6]
      }
      result:TOTAL
   }
   
      


output [attacks ATTACKS]

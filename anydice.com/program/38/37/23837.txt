DICE: 5
TREFFERBONUS: 1
TARGET: d{2..3}
DMGONHIT: 5

function: attackroll ROLL:s vs AC:n {
 COUNTA: 0
 COUNTB: 0
 COUNTC: 0
 loop P over {1..#ROLL} {
  COUNTA: COUNTA + (P@ROLL = 4)
  COUNTB: COUNTB + (P@ROLL = 5)
  COUNTC: COUNTC + (P@ROLL = 6)
 }
 
if COUNTA+ COUNTB + COUNTC + TREFFERBONUS >= AC { result: DMGONHIT+COUNTC }
 result: 0
}

output d[attackroll DICEd6 vs TARGET]

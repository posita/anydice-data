function: sqrt N:n {
  if N <= 0 {
    result: 0
  } else {
    L: #N
    X: 10^((L-1)/2)*(1+2*(L=2*(L/2)))*((1@N)@111222223)
    Y: N / X
    if Y - X <= 1 { result: X }
    if Y * Y - N < N - X * X { X: Y }
    loop K over {1..L} {
      Y: X
      X: (X * X + N) / (2 * X)
      R: N - X * X
      if (R >= 0) & (R <= 2 * X) { result: X }
      if (R < 0) & (R >= 2 - 2 * X) { result: X - 1 }
    }
    if (X * X > N) { X: X - 1 }
    result: X
  }
}

function: hits D:n {
  result: [count {5, 6} in Dd6]
}
FORCE: 14
SUMMON: [hits (6 + 7 + 3 + 5)]
SUMMON: SUMMON + [hits (SUMMON/2)]
SPIRIT: [hits FORCE]
SERVS: [highest of (SUMMON - SPIRIT) and 0]
RES: [hits (8 + 6 + 5)]
RES: RES + [hits ((8 + 6 + 5) - RES)]
RES: [lowest of RES and 10]
DRAIN: [highest of (2*SPIRIT - RES + 3) and 0]
output SERVS named "Services"
output DRAIN named "Drain"

set "explode depth" to 2

function: disadvantage N:n {
  if N<1 {
    result: -1
  } else {
    result: [lowest 2 of (2+N)d[explode d10]]
  }
}

function: normal {
  result: 2d[explode d10]
}

function: advantage N:n {
  if N<1 {
    result: -1
  } else {
    result: [highest 2 of (2+N)d[explode d10]]
  }
}

loop I over {3, 2, 1} {
  output [disadvantage I] named "disadvantage [I]"
}

output [normal] named "normal"

loop I over {1..3} {
  output [advantage I] named "advantage [I]"
}

function: sign of NUMBER:n {
  if NUMBER > 0 { result: +1 }
  if NUMBER <= 0 { result: -1 }
}

ATTACKER: [highest 3 of 3d6]+3
DEFENDER: [highest 3 of 3d6]+3

WINNER: [sign of ATTACKER - DEFENDER]

function: restrict ROLL:n to RANGE:s else REROLL:d {
  if ROLL = RANGE { result: ROLL }
  else { result: REROLL }
}
function: restrict ROLL:d to RANGE:s {
  loop I over {1..21} {
    ROLL: [restrict ROLL to RANGE else ROLL]
  }
  result: ROLL
}

ROUND: [restrict WINNER to {+1,-1}] = 1
output ROUND named "single round with rerolls: 1 = Attacker win, 0 = Defender win"

output 10dROUND named "number of Attacker wins over 10 rounds"

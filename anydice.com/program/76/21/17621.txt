set "position order" to "highest first"

function: D:s negate P:s {
  if 1@P = 1@D {
    result: 0 
  } else {
    result: 1@P
  }
} 
\if all negated, result = 0\
\if some negated, result = next highest\
\if none negated, result = highest\

loop A over {1..6} { \normal dice\
  loop B over {0..3} { \drag/negation dice\

    output [Bd6 negate Ad6] named "[A]d -- [B] Drag"
  }
}

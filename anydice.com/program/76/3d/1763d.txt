\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: first A:n second B:n {
  if A + DEX + PROF >= AC & B + DEX + PROF >= AC {result: 1d8 + 1d6 + DEX + SA}
  if A + DEX + PROF >= AC {result: 1d8 + DEX + SA}
  if B + DEX + PROF >= AC {result: 1d6 + SA}
  result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{12..20}

ATTACKMOD: 4
DMGBONUS: 2

CRITROLL: 20
ATTACKTIMES: 2
SA: 1d6
DAMAGE: 1d4 + ATTACKMOD + DMGBONUS + SA
CRITDAMAGE: 2d4 + ATTACKMOD + DMGBONUS + 2 * SA

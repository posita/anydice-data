DICEMIM: 1 \Starting number of dice rolled\
DICEMAX: 4 \Ending number of dice rolled\
DIEBONUS: 20\Bonus per dice\
CURRENTPROGRESS: 0\Your current project progress\
PROGRESSTARGET: 150\The total progress the project needs\

function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 0
 if SUCCESSES > 0 { result: 1 }
 result: 0
}

loop DICE over {DICEMIM..DICEMAX} {
 output [evaluate (1*DICE)d50+DIEBONUS*DICE+CURRENTPROGRESS-PROGRESSTARGET] named "[DICE]"
}

ATTACK: 8
CRITICAL_AT: 18
HIT_DAMAGE: 1d12+7
CRITICAL_DAMAGE: 2d12+7
MISS_DAMAGE: 0

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS_DAMAGE }
 if ROLL >= CRITICAL_AT { result: CRITICAL_DAMAGE }
 if ROLL + ATTACK >= DEFENSE { result: HIT_DAMAGE }
 result: MISS_DAMAGE
}

function: powerattack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS_DAMAGE }
 if ROLL >= CRITICAL_AT { result: CRITICAL_DAMAGE + 10 }
 if ROLL + ATTACK - 5 >= DEFENSE { result: HIT_DAMAGE + 10 }
 result: MISS_DAMAGE
}

loop AC over {15..25} {
 output [attack d20 vs AC] named "AC[AC]"
 output [powerattack d20 vs AC] named "AC[AC]"
}

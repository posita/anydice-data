function: distinct S:s {
  A: 1@S
  T: 1
  loop I over {2..#S} {
    if I@S != A {
      T: T + 1
    }
    A: I@S
  }
  result: T
}

function: setsize S:s {
  T: 1
  loop N over {1..6} {
    C: [count N in S]
    if C > T {
      T: C
    }
  }
  result: T
}

output [distinct 8d6]
output [setsize 8d6]

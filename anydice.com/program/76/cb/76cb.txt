function: hit A:n R:n {
    if R = 20 {result: 1}
    else{
        if R = 1 {result: 0}
        else {result: R>=A}
     }
}

function: attack A:n R:n D:n{
    if [hit A R]=1 {result: D} else {result: 0}
}

function: hammer A:n{
    result: [attack A 1d20+7 1d8+5]
}

function: javalin A:n{
    result: [attack A 1d20+7 1d6+5]
}

function: axe A:n{
    result: [attack A 1d20+6 1d8+4]
}

function: dmg A:n{
    result: [hammer A]+[javalin A]
}

output [dmg 17]

function: interpret ROLL:s {
 if [count {8..12} in {ROLL, BASE}] { result: 0 } \ miss \
 if [count {7} in {ROLL, BASE}] { result: 1 } \ half damage \
 if [count {1} in {ROLL, BASE}] { result: 4 } \ crit \
 result: 2 \ hit \
}

loop DICE over {8} {
 output [interpret DICE d8 ] named "[DICE]d"
}

output [interpret d12 and 8 d8 ]

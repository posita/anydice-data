function: advantage A:n disadvantage D:n {
  if 0 { result: (D+1)@((A+D+1)d20) }
  result: [[if 1d(A+D)>D then 1 else A+D+1] at (A+D+1)d20]
}

function: A:n at B:s {
  result: A@B
}

function: if A:n then B else C {
  if A { result: B }
  result: C
}

output [advantage 0 disadvantage 0]
output [advantage 1 disadvantage 0]
output [advantage 1 disadvantage 1]
output [advantage 2 disadvantage 0]
output [advantage 2 disadvantage 1]
output [advantage 2 disadvantage 2]

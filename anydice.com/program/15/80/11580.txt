function: gwf ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: GWFCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: GWFDAMAGE }
 result: 0
}

function: gwm ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: GWMCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS - 5 >= AC { result: GWMDAMAGE }
 result: 0
}

ENEMYAC: 20

ATTACKMOD: 5
GWF: 0

PROFICIENCY: 4
CRITROLL: 20
EXTRAATTACK: 2
DMGBONUS: 3
ACCBONUS: 0

GWFDAMAGE: 2d6 + ATTACKMOD + GWF + DMGBONUS
GWFCRITDAMAGE: 5d6 + ATTACKMOD + GWF + DMGBONUS

GWMDAMAGE: 2d6 + ATTACKMOD + GWF + DMGBONUS + 10
GWMCRITDAMAGE: 5d6 + ATTACKMOD + GWF + DMGBONUS + 10

output EXTRAATTACKd[gwf d20 vs ENEMYAC] named "GWF Greatsword"

GWFDAMAGE: 1d12 + ATTACKMOD + GWF + DMGBONUS
GWFCRITDAMAGE: 3d12 + ATTACKMOD + GWF + DMGBONUS

output EXTRAATTACKd[gwf d20 vs ENEMYAC] named "GWF Greataxe"

GWMDAMAGE: 1d12 + ATTACKMOD + GWF + DMGBONUS + 10

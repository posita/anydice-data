\
 Blades in the Dark
 0: Things go bad.
 1: Success, but with undesired consequences.
 2: Clean success.
 3: Critical success!
\

function: blades ROLL:s {
 if 1@ROLL = 6 & 2@ROLL = 6 {
  result: 5
 }
 if 1@ROLL = 6 {
  result: 3
 }
 if 1@ROLL >= 4 {
  result: 2
 }
 result: 1
}

output [blades 2@2d6] named "0d"
loop P over {1..7} {
 output [blades Pd6] named "[P]d"
}

function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

function: crit DIE:d {
  TOTAL: 0
  loop P over {1..(2 * #DIE)} {
    TOTAL: TOTAL + [DIE reroll {1, 2} as d{}]
  }
  result: TOTAL
}

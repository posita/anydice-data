function: roll two DA:n and DB:n {
   if DA = DB {
      result: DA + DB
   } else {
      result: [highest of DA and DB]
   }
}

function: roll three DA:n and DB:n and DC:n {
   if DA = DB & DB = DC {
      result: DA + DB + DC
   } else if DA = DB {
      result: DA + DB
   } else if DB = DC {
      result: DB + DC
   } else if DC = DA {
      result: DC + DA
   } else {
      result: [highest of DA and [highest of DB and DC]]
   }
}

output [roll two d6 and d6]
output [roll three d6 and d6 and d6]

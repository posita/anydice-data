\sneak attack\
S: 1
\full attack\
F: 1
\crit (this is automatic)\
C: 0
\Defending AC\
A: 20
\to hit rolls\
V: 1d20+9
B: 1d20+9
N: 1d20+4
M: 1d20+4

function: rapier damage {
 if S = 1 {
  if C = 1 {
   result: 2d6+14+4d6+7
  }
  else {
   result: 1d6+7+4d6+7
  }
 }
 else {
  if C = 1 {
   result: 2d6+14
  }
  else {
   result: 1d6+7
  }
 }
}


function: kukri damage {
 if S = 1 {
  if C = 1 {
   result: 2d4+12+4d6+7
  }
  else {
   result: 1d4+6+4d6+7
  }
 }
 else {
  if C = 1 {
   result: 2d4+12
  }
  else {
   result: 1d4+6
  }
 }
}

function: ROLL:s rapier a hit {
 U:ROLL
 if U > A {
  if U > 23 {
   C:1
  }
  else {
   C:0
  }
  result:[rapier damage]
 }
 else {
  result:0
 }
 
}

function: ROLL:s rapier b hit {
 if F = 1 {
  U:ROLL
  if U > A {
   if U > 18 {
    C:1
   }
   else {
    C:0
   }
   result:[rapier damage]
  }
 else {
  result:0
 }
 }
 else {
  result:0
 }
}

function: ROLL:s kukri a hit {
 if ROLL > 23 {
  C:1
 }
 else {
  C:0
 }
 if ROLL > A {
  result:[kukri damage]
 }
 else {
  result:0
 }
}

function: ROLL:s kukri b hit {
 if F = 1 {
  if ROLL > 18 {
   C:1
  }
  else {
   C:0
  }
  if ROLL > A {
   result:[kukri damage]
  }
  else {
   result:0
  }
 }
 else {
  result:0
 }
}

output {[V rapier a hit]+[N rapier b hit]+[B kukri a hit]+[M kukri b hit]}

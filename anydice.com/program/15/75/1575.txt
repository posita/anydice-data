function: well of DICE:d luck LUCK:n
{
    if (LUCK <= 10)
    {
        result: [lowest 1 of DICE]
    }
    result: [highest 1 of DICE]
}

output [well of 4d20 luck d20-5] named "-5 Luck"
output [well of 4d20 luck d20-2] named "-2 Luck"
output [well of 4d20 luck d20] named "No Luck"
output [well of 4d20 luck d20+2] named "+2 Luck"
output [well of 4d20 luck d20+5] named "+5 Luck"

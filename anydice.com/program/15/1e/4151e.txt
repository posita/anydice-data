\Lancer damage simulator\

ARMOR: 1 \assume average armor is 1\
TN:  8 \E-Def or Evasion target number - reduce by 1 if Field Analyst and also change 20 to 19 below\
ACC: 0 \Accuracy applied\ 
DIFF: 1 \Difficulty applied - either accuracy or difficulty should be 0\
GRIT: 1 \Increase to 1 if Field Analyst\
RELIABLE: 0

ROLL: d20 + [highest 1 of ACCd6] - [highest 1 of DIFFd6] + GRIT
function: dmgcalc R:n TN:n {
   if R >= TN {
    if R >= 20 {
      result: [highest 2 of 4d6]+4-ARMOR
    } else {
      result: [highest of 2d6+4 and RELIABLE]-ARMOR
    }
  } else {
    result: [highest of 0 and RELIABLE-ARMOR]
  }
}

output [dmgcalc ROLL TN]

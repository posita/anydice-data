function: reroll A:n C:n{ 
 if A <= 2 { 
  A: 1dC 
 }  
 result: A 
}


function: greatweaponmaster B:n A:n {
 X:0
 loop N over {1..B} {
  X: X + [reroll 1dA A]
 }
 result: X
}


output [greatweaponmaster 2 6] named "2d6 Great Weapon Master"
output [greatweaponmaster 1 8] + 1 + 1d6 named "(1d8 + 1 Great Weapon Master) + 1d6"

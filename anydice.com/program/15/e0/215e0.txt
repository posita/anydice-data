MAINTENANCE: 60 / 10

function: cost ROLL:n DAYS {
  if ROLL <= 20 {
    result: - MAINTENANCE * DAYS
  } else if ROLL <= 30 {
    result: - MAINTENANCE * DAYS
  } else if ROLL <= 40 {
    result: - MAINTENANCE * DAYS
  } else if ROLL <= 60 {
    result: 0
  } else if ROLL <= 80 {
    result: 1d6 * 5
  } else if ROLL <= 90 {
    result: 2d8 * 5
  } else {
    result: 3d10 * 5
  }
}

DAYS: 1
ROLL: 1d100 + DAYS
output 30 d [cost ROLL 1] / 30

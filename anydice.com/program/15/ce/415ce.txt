SAVE_DC: 8+2+2
FALLING_DAMAGE: 1
USE_CALLOUS_GLOW: 1
USE_ALCHEMIST_FIRE: 1
MYRKUL_HP: 390
MYRKUL_SAVE: 2
KETHERIC_HP: 218
KETHERIC_SAVE: MYRKUL_SAVE+6

function: throw BOMBS bombs with save DEX_SAVE and ADVANTAGE {
  if ADVANTAGE = 1 {result: ((BOMBS*(1+USE_CALLOUS_GLOW)*(1+FALLING_DAMAGE+USE_ALCHEMIST_FIRE))d((3d4+9)/(1+(([highest 1 of 2d20]+DEX_SAVE)>=SAVE_DC))-2))}
  else {result: ((BOMBS*(1+USE_CALLOUS_GLOW)*(1+FALLING_DAMAGE+USE_ALCHEMIST_FIRE))d((3d4+9)/(1+((2d20+DEX_SAVE)>=SAVE_DC))-2))}
}

KETHERIC_BOMBS: 6
MYRKUL_BOMBS: 9

output [throw KETHERIC_BOMBS bombs with save KETHERIC_SAVE and 0]>=KETHERIC_HP named "Probability of Ketheric kill"
output [throw MYRKUL_BOMBS bombs with save MYRKUL_SAVE and 1]>=MYRKUL_HP named "Probability of Myrkul kill"

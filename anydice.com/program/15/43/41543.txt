function: evaluate ROLL:s {
 CRITICAL: ROLL = 6
 SUCCESSES: ROLL >= 4
 FAILURES: ROLL <= 3
 BOTCH: ROLL = 1
 \if SUCCESSES > 0 { result: SUCCESSES \+ (ROLL = 6)\ }\
 \if BOTCH > 0 { result: -BOTCH }\
 \if [count 1 in ROLL] { result: -1 }\
 if SUCCESSES + CRITICAL - BOTCH < 0 { result: -1 }
 result: SUCCESSES + CRITICAL - BOTCH
 \result: 0\
}

loop DICE over {1..10} {
 output [evaluate DICE d6] named "[DICE]d"
}

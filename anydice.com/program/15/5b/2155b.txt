\ TFC Pit Fighting \

LEVEL: 6
DC: 12 + 1d12  \ steel DC \
SONE: 1d20+6   \ acrobatics \
STWO: 1d20+1d8 \ constitution check \
STHREE: 1d20+6 \ attack \

\ Honest work value table per week \
function: work ROLL:n {
  if ROLL <= 0 { result: 0 }
  if ROLL <= 9 { result: 10 * LEVEL }
  if ROLL <= 14 { result: 40 * LEVEL }
  if ROLL <= 19 { result: 50 * LEVEL }
  if ROLL <= 24 { result: 60 * LEVEL }
  if ROLL <= 29 { result: 80 * LEVEL }
  result: 100 * LEVEL
}

DTVALUE: [work 1d20+6] \ value to character (in gold) of one day of DT \

output DTVALUE named "Honest work"

function: reward SUC:n {
  if SUC = 0 { result: -DTVALUE }
  if SUC = 1 { result: 50 * LEVEL }
  if SUC = 2 { result: 100 * LEVEL }
  if SUC = 3 { result: 150 * LEVEL }
}

SUC: (SONE >= DC) + (STWO >= DC) + (STHREE >= DC)

output SUC named "Successes"

output [reward SUC] named "Pit fighting"

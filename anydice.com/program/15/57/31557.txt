

BASE_SMITE: 1
SPELLSLOT: 4
FIEND_UNDEAD: 1
CRITICAL_HIT: 1


QTY: (BASE_SMITE + SPELLSLOT + FIEND_UNDEAD) * (CRITICAL_HIT + 1)
SIZE: 8

function: roll A:n and reroll if under B {
 if A <= B { result: 1dSIZE }
 result: A
}

output QTY named "rolling [QTY]d[SIZE]"
output QTYd [roll dSIZE and reroll if under 0] named "Smite reroll none"
output QTYd [roll dSIZE and reroll if under 1] named "Smite reroll 1"
output QTYd [roll dSIZE and reroll if under 2] named "Smite reroll 2"

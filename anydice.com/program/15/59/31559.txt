DEMONROLL: d20+2+1d6
DEMONDAMAGE: 2d6
DEMONDEFENSE: d{9..13}
DEMONHEALTH: d{8..11}

function: demon lord attack G:n defense D:n {
 if G>=D {DAMAGE: DEMONDAMAGE} else {DAMAGE: 0}
 if G>=20 {if G>=(D+5) {XDAMAGE: 1d6}} else {XDAMAGE: 0}
 result: DAMAGE+XDAMAGE
}

function: demon lord instakill G:n and D:n and H:n {
 if G>=D {DAMAGE: DEMONDAMAGE} else {DAMAGE: 0}
 if G>=20 {if G>=(D+5) {XDAMAGE: 1d6}} else {XDAMAGE: 0}
 IK: (DAMAGE+XDAMAGE)/H
 result: IK
}

output [demon lord attack DEMONROLL defense DEMONDEFENSE] named "Odds of Damage From Attack"
output [demon lord instakill DEMONROLL and DEMONDEFENSE and DEMONHEALTH] named "Odds of Instakill From Attack"

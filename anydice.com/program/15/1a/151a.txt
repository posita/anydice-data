\ #power dice 1-12 \
NPD: 12

\ allow a single die reroll (ex. Blood of Tzeentch). 1 to allow \
REROLL: 0

\ DO NOT CHANGE ANYTHING BELOW THIS LINE \

function: chance of irresistible force with power dice PD:s {
 if {1,2}@PD = 12 {
  if REROLL > 0 {
   \ if the 3rd die is also a six then rerolling is pointless \
   if 3@PD = 6 {result: 1}
   \ reroll the die \
   result: 1d6 = 6
  }
  result: 1
 }
 result: 0
}

loop N over {1..NPD}{
 PD: Nd6
 if REROLL > 0 {
  output [chance of irresistible force with power dice PD] named "'Irresistible Force' on [N]d6 (with reroll)"
 }
 if REROLL = 0 {
  output [chance of irresistible force with power dice PD] named "'Irresistible Force' on [N]d6"
 }
}

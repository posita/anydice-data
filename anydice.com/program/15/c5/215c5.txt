function: attack ROLL:n plus BONUS:n vs AC:n hit HIT:d miss MISS:d crit CRIT:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: MISS
    }
    if ROLL >= CRIT_RANGE {
        result: CRIT
    }
    if ROLL + BONUS >= AC {
        result: HIT
    }
    result: MISS
}
output [attack 1d20 plus 8 vs 17 hit 1d8+5 miss 0 crit 2d8+5 on 20] named "Standard attack"
output [attack 1d20 plus 8 vs 22 hit 1d8+5 miss 0 crit 2d8+5 on 20] named "Shield"

function: reroll A:n with B:n against C {
  if A < C {
    if B < C { result: 0 }
    else { result: 1 }
  }
  else { result: 1 }
}


output d20 >= 10 named "Classic roll (10)"

output d20 >= 13 named "Hard roll (10 + 3)"

output [reroll d20 with d20 against 10] named "Classic reroll (10)"

output [reroll d20 with d20 against 13] named "Hard reroll (10 + 3)"

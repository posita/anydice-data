 \ 1 is a successful roll  \
  \ 0 is a failed roll    \

  loop K over DIEARRAY {
    output 1dJ > 1dK named "Chance that 1d[J] is rolling HIGHER than 1d[K]"
  }
  output 574127 named "~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~"
}

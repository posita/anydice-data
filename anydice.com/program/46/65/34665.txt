function: dark X:n {
 if X >= 0{
  ROLL: (2+X)d6
  result: 1@ROLL + 2@ROLL
 }
 else {
  ROLL: (2-X)d6
  result: (1-X)@ROLL + (2-X)@ROLL
 }
}

ASKILL: 0
DSKILL: 0
STR: 2
DAMAGE: 4
ARMOR: 0

function: damage SUCCESLEVEL:s{
 if SUCCESLEVEL >= 0 {
  result: [highest of SUCCESLEVEL+[highest 2 of (2+STR)d6]-7-ARMOR+DAMAGE and 0]
 }
 else {
 result: -1
 }
}

output [damage ([dark ASKILL] - [dark DSKILL])] named "[ASKILL] vs [DSKILL]"

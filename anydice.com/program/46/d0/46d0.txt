PX: d6
PY: d6
HX: d6
HY: d6

TIE: ((PX = HX) & (PY = HY)) | ((PX = HY) & (PY = HX))
TIE_BONUS: PX = PY                                               \ player scores doubles \
P_GREATER_SUM:  (PX + PY > HX + HY)                              \ player sum is greater than house sum \
P_GREATER_DIE: ((PX > HX) & (PY > HY)) | ((PX > HY) & (PY > HX)) \ house die is greater than player die \

P_WIN: (TIE & TIE_BONUS) | (!TIE & P_GREATER_SUM & P_GREATER_DIE)
P_LOSE: !P_WIN & !TIE
P_TIE: !P_WIN & TIE

output P_LOSE
output P_WIN
output P_TIE

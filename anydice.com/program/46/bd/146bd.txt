function: chouginga R:n with A:n {
  if A = 5 & R = 6 { result: 1 + [chouginga d6 with A] }
  result: R >= [highest of 6-A and 2]
}

UNSKILLED: [lowest of 3 and [chouginga 1d6 with 0]]
INICIADO: [lowest of 3 and [chouginga 1d6 with 1]]
COMPETENTE: [lowest of 3 and [chouginga 1d6 with 2]]
EXPERTO: [lowest of 3 and [chouginga 1d6 with 3]]
MAESTRO: [lowest of 3 and [chouginga 1d6 with 4]]
LEGENDARIO: [lowest of 3 and [chouginga 1d6 with 5]]

function: melee attack ROLL:n vs REF:n {
  result: (ROLL >= REF) + (ROLL >= REF/2)
}

loop V over {1..12} {
  output [melee attack VdUNSKILLED vs V] named "Ataque CaC / Esgrima: Unskilled / Vigor: [V] / Reflejos: [V]"
  output [melee attack VdINICIADO vs V] named "Ataque CaC / Esgrima: Iniciado / Vigor: [V] / Reflejos: [V]"
  output [melee attack VdCOMPETENTE vs V] named "Ataque CaC / Esgrima: Competente / Vigor: [V] / Reflejos: [V]"
  output [melee attack VdEXPERTO vs V] named "Ataque CaC / Esgrima: Experto / Vigor: [V] / Reflejos: [V]"
  output [melee attack VdMAESTRO vs V] named "Ataque CaC / Esgrima: Maestro / Vigor: [V] / Reflejos: [V]"
  output [melee attack VdLEGENDARIO vs V] named "Ataque CaC / Esgrima: Legendario / Vigor: [V] / Reflejos: [V]"
}

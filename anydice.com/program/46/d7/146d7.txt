\How many d6 are you rolling? Put that number in N.\
N: 6

SIXES: [count 6 in Nd6]

function: count sets in D {
  SETS: 0
  loop S over {1..6} {
    SETS: SETS + ([count S in D] > 1)
  }
  result: SETS
}

SETS: [count sets in Nd6]

output SIXES named "success by rolling 6"
output SETS named "success by rolling sets"

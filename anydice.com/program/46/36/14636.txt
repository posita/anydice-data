function: hit ROLL:n attackmod ATKMOD:n vs ac AC:n crit dice CRIT:n {
  if (ROLL = 20) {
    result: CRIT
  } else {
    if (ROLL + ATKMOD >= AC) {
      result: 1
    } else {
      result: 0
    }
  }
}
function: damage hit H:n damage dice DMG:d damage mod D:n {
  if H = 0 {
    result: 0 
  } else {
    result: HdDMG + D
  }
}

WEAPON_BONUS: 1
PROFICIENCY: 4
STRENGTHMOD: 5
RAGEDAMAGE: 3
GREATAXE: 1d12
TARGET_AC: 14
DAMAGE_MOD: RAGEDAMAGE + STRENGTHMOD + WEAPON_BONUS
BRUTAL_CRITICAL_DICE: 3 \ total damage dice rolled on a crit \

ADVANTAGE: [highest 1 of 2d20]
ATTACK_MODIFIER: WEAPON_BONUS + PROFICIENCY + STRENGTHMOD
output 2d[damage hit [hit d20 attackmod ATTACK_MODIFIER vs ac TARGET_AC crit dice BRUTAL_CRITICAL_DICE] damage dice GREATAXE damage mod DAMAGE_MOD] named "standard greataxe"
output 2d[damage hit [hit ADVANTAGE attackmod ATTACK_MODIFIER vs ac TARGET_AC crit dice BRUTAL_CRITICAL_DICE] damage dice GREATAXE damage mod DAMAGE_MOD] named "reckless greataxe"
output 2d[damage hit [hit d20 attackmod (ATTACK_MODIFIER - 5) vs ac TARGET_AC crit dice BRUTAL_CRITICAL_DICE] damage dice GREATAXE damage mod (DAMAGE_MOD + 10)] named "GWM greataxe"
output 2d[damage hit [hit ADVANTAGE attackmod (ATTACK_MODIFIER - 5) vs ac TARGET_AC crit dice BRUTAL_CRITICAL_DICE] damage dice GREATAXE damage mod (DAMAGE_MOD + 10)] named "GWM Reckless greataxe"

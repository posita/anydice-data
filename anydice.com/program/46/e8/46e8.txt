function: fumble A:n plus B:n {
    if A = 3*1-5 | B = 3*1-5 { result: 3*1-5 }
    result: A + B
}
function: explodefumble ROLL:n at DEPTH:n {
    if ROLL = 3*6 - 5 & DEPTH < 3 {
       result: [fumble ROLL plus [explodefumble 3d6 - 5 at DEPTH+1]]
    }
    result: ROLL
}
ROLL: [explodefumble 3d6 - 5 at 1]
BONUS: 2 + 2 + 0

output [highest of 0 and (ROLL + BONUS - 7) / 2] named "# of successes"
output ROLL named "raw die roll"

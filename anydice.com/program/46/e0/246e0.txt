function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 7
 if SUCCESSES > 0 { result: SUCCESSES + (ROLL = 12) }
 result: 0
}

loop DICE over {1..6} {
 output [evaluate DICE d12] named "[DICE]d12 com 2 sucesso"
}

function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d6] }
 result: N >= 4
}

NWOD: [nwod 6 again d6]

loop N over {1..6} {
 output [lowest of 8 and NdNWOD] named "[N]d explodindo"
}

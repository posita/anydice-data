function: ROLL:n attack vs AC:n
{
	if ROLL = 1 { result: 0 }
	if ROLL <= 20 & ROLL >= (21 - CRITICAL_RANGE)
	{
		if [ROLL threat vs AC] { result: CRITICAL_MULTI d DAMAGE}
	}
	if ROLL >= AC { result: DAMAGE }

	result: 0
}

function: ROLL:n threat vs AC:n
{
	result: ROLL > 1 & (ROLL >= AC | ROLL = 20)
}

CRITICAL_RANGE: 2
CRITICAL_MULTI: 2

loop AC over {1..20}
{
	DAMAGE: 1d8
	output [d20+14 attack vs AC+5] + [d20+14 attack vs AC+5] + [d20+9 attack vs AC+5]
}

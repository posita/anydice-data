function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output [lowest 1 of 2d10] named "0 - Untrained"
output 1d10 named "1 - Novice"
output [highest 1 of 2d10] named "2 - Journeyer"
output [highest 2 of 3d10] named "3 - Expert"
output [highest 2 of 4d[reroll {1} on d10]] named "4 - Professional"
output [highest 2 of 5d[reroll {1,2} on d10]] named "5 - Legendary"

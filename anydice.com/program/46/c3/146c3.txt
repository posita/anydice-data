function: mechanic N:n {
COUNT: 0
loop I over {1..N} {
Y: 1d10
COUNT: COUNT + (Y = 10)
}
result: COUNT
}

function: ROLL:s count tens twice subtract ones {
 result: 2*(ROLL = 10) - (ROLL = 1)
}

function: ROLL:s count tens {
 result: (ROLL = 10)
}

function: clip X:n {
 if X < 0 {result: 0}
 else {result: X}
}

DICE: 10d10
output [DICE count tens] named "first version"
output [DICE count tens twice subtract ones] named "second version"
output [clip [DICE count tens twice subtract ones]] named "second version, no negative total"

\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 8-16\
ENEMYAC: d{8..12}

\ Flipwipwip Lvl 4, DEX 18 \

ATTACKMOD: 4
DMGBONUS: 1
PROFICIENCY: 2
ACCBONUS: 1
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Shortsword, +1"

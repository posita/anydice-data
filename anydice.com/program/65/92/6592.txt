function: rank R:n {
 if R = 1 {result: d4}
 if R = 2 {result: d6}
 if R = 3 {result: d8}
 if R = 4 {result: d10}
 if R = 5 {result: d12}
 if R = 6 {result: d20}
}
function: success DIE:n {
 if [count DIE in {4,5,6,7}] = 1 {result: 1}
 if [count DIE in {8,9,10,11}] = 1 {result: 2}
 if [count DIE in {12,13,14,15}] = 1 {result: 3}
 if [count DIE in {16,17,18,19}] = 1 {result: 4}
 if [count DIE in {20,21,22,23}] = 1 {result: 5}
 result: 0
}
function: pool P:s {
 ACC: 0
 loop N over P {
  ACC: ACC + [success [rank N]]
 }
 result: ACC
}

output [success 1d4] named "d4"
output [success 1d6] named "d6"
output [success 1d8] named "d8"
output [success 1d10] named "d10"
output [success 1d12] named "d12"
output [success 1d20] named "d20"

X: 2d20
output [success 1@X] + [success 2@X] named "2d20"

output [pool {1}] named "pool 1"
output [pool {2,2}] named "pool 2"
output [pool {3,3,3}] named "pool 3"
output [pool {4,4,4,4}] named "pool 4"
output [pool {5,5,5,5,5}] named "pool 5"

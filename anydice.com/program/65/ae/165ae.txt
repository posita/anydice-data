function: attack ATTACK:n AMOD:n AC:n DMG:d CRIT:d {
  if ATTACK = 20 { result: DMG + CRIT }
  if ATTACK + AMOD >= AC { result: DMG }
  result: 0
}

output [attack 1@2d20 10 16 2d6+13 2d6] named "Greatsword normal"
output [attack 1@2d20 5 16 2d6+23 2d6] named "Greatsword GWF"
output [attack 1@2d20 5 16 2d6+23 2d6] - [attack 1@2d20 10 16 2d6+13 2d6] named "difference"

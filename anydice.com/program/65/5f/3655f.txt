ROLLTOHIT: [highest 1 of 2d20]
FIST: 1d8
CRITFIST: 2d8
HOCRITFIST: 3d8
BBFIST: 1d8+2d6
BBCRITFIST: 2d8+4d6


function: fist hit ROLL:n{
  if ROLL = 20 {result: 2}
  if ROLL + 8 >= 20 {result: 1}
  result: 0
}

function: single attack ho ROLL:n{
  if (ROLL=2) {result:HOCRITFIST + 5}
  else if (ROLL=1) {result:FIST + 5}
  else {result:0}
}

function: single attack bb ROLL:n{
  if (ROLL=2) {result:BBCRITFIST + 5}
  else if (ROLL=1) {result:BBFIST + 5}
  else {result:0}
}


output [single attack ho [fist hit ROLLTOHIT]]+[single attack ho [fist hit ROLLTOHIT]]+[single attack ho [fist hit ROLLTOHIT]]+[single attack ho[fist hit ROLLTOHIT]] named "full turn HO"
output [single attack bb [fist hit ROLLTOHIT]]+[single attack bb[fist hit ROLLTOHIT]]+[single attack bb [fist hit ROLLTOHIT]]+[single attack bb [fist hit ROLLTOHIT]] named "full turn BB"

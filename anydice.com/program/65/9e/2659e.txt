function: reroll DICE:n if lt NUM:n {
  if DICE >= NUM {
    result: DICE
  } else {
    result: 4d(d6 > 3)
  }
}

output 3d(1@2d(d6>3)) named "3 dice+reroll"
output 6d(d6>3) named "6 dice base pool"
output 2d(1@2d(d6>3)) named "2 dice+reroll"
output 4d(d6>3) named "4 dice"
output 1d(1@2d(d6>3)) named "1 die+reroll"
output 2d(d6>3) named "2 dice"

MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL >= CRITICALROLL {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

ATTACK: 5 + 3
MISSDAMAGE: 0

DEFENSE: d{12..22}

HITDAMAGE: 2d4
CRITICALDAMAGE: 4d4
CRITICALROLL: 18
output [attack] named "Falchion"

HITDAMAGE: 1d12
CRITICALDAMAGE: 3d12
CRITICALROLL: 20
output [attack] named "Greataxe"

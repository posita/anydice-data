DECAY: 3
ITERATIONS: 20
INITDICE: 2
LOOP: 6

function: usage COUNT:n {
  loop I over {1..ITERATIONS} {
    COUNT: COUNT - [count {1..DECAY} in COUNTd6]
  }
  result: [highest of 0 and COUNT]
}

loop I over {1..LOOP} {
  INITDICE: I
  output [usage INITDICE] named "[INITDICE] init dice"
}
\
loop I over {1..LOOP} {
  ITERATIONS: I
  output [usage INITDICE] named "[ITERATIONS] iterations"
}

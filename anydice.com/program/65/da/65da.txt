function: highest of ROLL:s plus N:n if multiple {
  MAX: 1@ROLL
  if MAX > 0 & (MAX = ROLL) > 1 { MAX: MAX + N }
  result: MAX
}
function: highest of A:d plus N:n if multiple plus M:n times count of B:d excluding EXCL:s {
  ANEW: {}
  loop I over {1@A} {
    if I = EXCL { I: 0 }
    ANEW: {ANEW, I}
  }
  BNEW: {}
  loop I over {1@B} {
    BNEW: {BNEW, !(I = EXCL)}
  }
  result: [highest of (#A) d ANEW plus N if multiple] + M * (#B) d BNEW
}
output [highest of 3d6 plus 10 if multiple plus 100 times count of 2d6 excluding 4d6]

function: hit ROLL:n with BONUS:n against AC:n for DAMAGE:n threat THREAT:n for CRIT:n {
  if (((ROLL + BONUS) > AC) | (ROLL >= 20)) {
    if (ROLL >= THREAT) {
      result: DAMAGE + CRIT
    } else {
      result: DAMAGE
    }
  }
  result: 0
}

function: pbta X:n {
  if X>=10 {result: 2}
  if X>=7 {result: 1}
  result: 0
}
loop N over {-2..4} {
  output [pbta 2d6+N] named "1 Forward to [N]"
}
loop N over {-1..3} {
  output [pbta {1,2}@3d6+N] named "Advantage plus [N]"
}
loop N over {-1..3} {
  output [pbta {2,3}@3d6+N] named "Disadvantage plus [N]"
}

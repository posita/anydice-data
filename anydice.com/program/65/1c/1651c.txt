\ AC less than 18 Heavy Xbow has better mean; AC greater than 18 Hand Xbow +1 has better mean \

AC: 18
EXTRA_DAM: 1d6
WEAPON_ONE_HIT: 10
WEAPON_ONE_DAM: 1d6
WEAPON_ONE_BONUS: 5
WEAPON_TWO_HIT: 9
WEAPON_TWO_DAM: 1d10
WEAPON_TWO_BONUS: 4

function: attack with WEAPON_HIT:n against TARGET_AC:n {
 result: d20+WEAPON_HIT >= TARGET_AC 
}

\ the output curves assumes two attacks with each weapon \
output [attack with WEAPON_ONE_HIT against AC] * (WEAPON_ONE_DAM+WEAPON_ONE_BONUS+EXTRA_DAM) + [attack with WEAPON_ONE_HIT against AC] * (WEAPON_ONE_DAM+WEAPON_ONE_BONUS+EXTRA_DAM) named "Hand Crossbow Damage with Hunter's Mark cast same turn"
output [attack with WEAPON_TWO_HIT against AC] * (WEAPON_TWO_DAM+WEAPON_TWO_BONUS+EXTRA_DAM) + [attack with WEAPON_TWO_HIT against AC] * (WEAPON_TWO_DAM+WEAPON_TWO_BONUS+EXTRA_DAM) named "Heavy Crossbow Damage with Hunter's Mark cast same turn"

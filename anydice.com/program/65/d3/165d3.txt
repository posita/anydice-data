
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 10 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack 2@3d10 vs DEFENSE]]
}



MISS: 0
HIT: 1
CRITICALHIT: 2
MISSDAMAGE: 0

DEFENSE: d{3..20}

ATTACK: 11 + 2
LEVEL: 1
CURSE: [explode 1d6]




HITDAMAGE: [explode 1d6] + 2 + LEVEL + CURSE
CRITICALDAMAGE: HITDAMAGE
output [attack] named "Explosion sobrenatural"


HITDAMAGE: [explode 1d6] + LEVEL + CURSE
CRITICALDAMAGE: [explode 1d6] + LEVEL + CURSE
output [attack] named "Mordedura visual"


HITDAMAGE: [explode 2d6] + LEVEL + CURSE
CRITICALDAMAGE: [explode 2d6] + LEVEL + CURSE
output [attack] named "Fuego de bruja"

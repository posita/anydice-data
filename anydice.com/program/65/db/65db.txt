function: highest of ROLL:s with bonus {
  MAX: 1@ROLL
  BONUS: MAX > 0 & (MAX = ROLL) > 1
  result: 10 * MAX + BONUS
}
function: highest of DICE:d excluding EXCL:s {
  NEW: {}
  loop I over {1@DICE} {
    if I = EXCL { I: 0 }
    NEW: {NEW, I}
  }
  POOL: (#DICE) d NEW
  result: [highest of POOL with bonus]
}
output [highest of 3d6 excluding 4d6]

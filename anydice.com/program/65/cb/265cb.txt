function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: 0 }
 if ROLL = 20 { result: 2 }
 if ROLL + ATTACK >= DEFENSE + 10 { result: 2 }
 if ROLL + ATTACK >= DEFENSE { result: 1 }
 result: 0
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = 2 { result: CRIT }
 if ATTACKRESULT = 1 { result: HIT }
 result: 0
}

ATTACK: 7
HIT: 2d10+4
CRIT: 4d12+8

loop AC over {12, 14, 15, 18} {
 output [damage for [attack d20 vs AC]] named "Damage vs AC [AC]"
}

function: fumble ROLL:s target TARGET:n {
  result: (ROLL = 1) >= 2 & !(ROLL >= TARGET)
}
loop T over {6..8} {
  \ optimization: use a custom d10 that can only roll 1 (fumble), 2 (fail) or 10 (success) \
  DIE: {1, 2:(T-2), 10:(11-T)}
  loop N over {2..10} {
    output [fumble NdDIE target T] named "fumble [N]d10 vs. [T]"
  }
}

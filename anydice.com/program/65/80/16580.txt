MODFORROOKIE: 0
MODFORENEMY: 2

function: fight ROOKIE:n ENEMY:n {
  if ROOKIE > ENEMY { result: 1}
  if ROOKIE < ENEMY { result: 0}
  if ROOKIE = ENEMY { result: [fight 1d20+MODFORROOKIE 1d20+MODFORENEMY] }
}

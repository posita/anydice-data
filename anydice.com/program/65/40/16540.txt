function: attack ROLL:n {
  \ return the number of times damage is applied on an attack roll of ROLL \
  if ROLL <= 1 { result: 0 }     \ natural 1 is always a miss \
  if ROLL >= CRITCHANCE{ result: 2 }  \ critical hit deals damage damage twice \
  result: ROLL + HIT >= AC
}

function: multiattack N:n times {
  result: [roll [attack ATK] to multiattack N times]
}
function: roll FIRSTHIT:n to multiattack N:n times {
  if N <= 0 {
    result: 0
  }
  if FIRSTHIT > 0 {
    DAMAGE: FIRSTHIT d (DMG + BONUS) + MODIFIER
    loop I over {2..N} {
      DAMAGE: DAMAGE + [attack ATK] d DMG + MODIFIER
    }
    result: DAMAGE
  } else {
    result: [roll [attack ATK] to multiattack N-1 times]
  }
}

ATK:        1d20  \ attack roll: use 1@2d20 for advantage or 2@2d20 for disadvantage \
HIT:          9  \ bonus to attack roll \
AC:          15  \ target AC: attack hits if ATK + HIT >= AC \
CRITCHANCE:  21  \ minimum attack roll to crit \
DMG:         2d10  \ damage rolled on successful hit \
BONUS:       0  \ bonus damage on first hit \
MODIFIER:    0  \ damage That is not doubled on crit\

output [multiattack 1 times] named "Fire"

HIT:          5  \ bonus to attack roll \
AC:          15  \ target AC: attack hits if ATK + HIT >= AC \
DMG:         2d12  \ damage rolled on successful hit \

output [multiattack 1 times] named "Toll"

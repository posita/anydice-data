function: highest of ROLL:s plus N:n if multiple {
  MAX: 1@ROLL
  if MAX > 0 & (MAX = ROLL) > 1 { MAX: MAX + N }
  result: MAX
}
function: highest of DICE:d excluding EXCL:s {
  NEW: {}
  loop I over {1@DICE} {
    if I = EXCL { I: 0 }
    NEW: {NEW, I}
  }
  POOL: (#DICE) d NEW
  result: [highest of POOL plus 100 if multiple]
}
output [highest of 3d6 excluding 4d6]

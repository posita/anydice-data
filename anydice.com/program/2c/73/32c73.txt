set "maximum function depth" to 10

function: replace lowest A:s with X:n{
  if #A@A < X & X >= 0 {result: {1..(#A-1)}@A + X}
  result: A
  }

function: jam chance A:n TEN:n{
if OVERHEATS {result: A<91}
else if MODE = 0 & RELIABLE {if  TEN = 1 & A > 96 {result:  0} result: 1}
else if MODE = 0 {result: A<96}
else if RELIABLE {if TEN = 1 & A > 94 {result:  0} result: 1}
result: A<94
}

function: scatter extra hits DOS:n{
if SCATTER {result: DOS/2}
result: 0
}

function: X:n dos A:n Y:n{
if X+TOHITBONUS >= A & Y { result: (X - A)/10 }
result: -1
}

function: hits DOS:n{
SCATTER_EXTRA_HITS: [scatter extra hits DOS]
if MODE = 0 & DOS >= 0 {result: 1*(STORM+1) + SCATTER_EXTRA_HITS}
else if MODE = 1 & DOS >= 0
{
  if ROF <= DOS/2 {result: ROF*(STORM+1) + SCATTER_EXTRA_HITS}
  else {result: (DOS/2 + 1)*(STORM+1) + SCATTER_EXTRA_HITS}
}
else if MODE = 2 & DOS >= 0
{
  if ROF <= DOS {result: ROF + SCATTER_EXTRA_HITS}
  else {result: (DOS + 1)*(STORM+1) + SCATTER_EXTRA_HITS}
}
result: 0
}

function: soak DOS:n{
if (DOS > 2 & RAZORSHARP) | MELTA {result: [highest of AP-(PEN*2) and 0] + (TB*[highest of (UNT-FELLING) and 1])}
else {result: [highest of AP-PEN and 0] + TB*[highest of (UNT-FELLING) and 1]}}

function: ag test X:n{
result: AG < X
}

function: flamer damage{
result: NUMBEROFDTENSd[explode d10] + BONUS
}

function: damage{
if FLAMER{result: [flamer damage]}
N: NUMBEROFDTENS + ACCURATE*([highest of DOS/2 and 2])
DICE:[highest of [explode d10] and PROVEN]
if TEARING{result: [replace lowest [highest N of (N+1)dDICE]with DOS] + BONUS}
else {result: [replace lowest NdDICE with DOS] + BONUS}}

function: flamer wounds caused AG:n{
if AG {result: [highest of [flamer damage] - [soak DOS] and 0]}
result: 0
}

function: wounds caused {
result: [highest of (DAMAGE - [soak DOS]) and 0]
}

function: wounds caused total{
if FLAMER{result: [flamer wounds caused [ag test d100]]}
result: [wounds caused] * HITS}

function: will it kill{
result: (NUMBEROFDTENS*4)@{[wounds caused]} > WOUNDS}

function: kills DOS:s{
if FLAMER{ result: [kills flamer] }
if [will it kill]{result: HITS*(WOUNDS <= [wounds caused])}                   
result: 100*[wounds caused total]/WOUNDS
}

function: kills flamer{
if (NUMBEROFDTENS*4)@{[flamer wounds caused[ag test d100]]}>WOUNDS
{result: (WOUNDS <= [flamer wounds caused[ag test d100]])}
result: 100*[flamer wounds caused[ag test d100]]/WOUNDS
}

function: flamer horde kills{
QUARTER: FLAMERRANGE/4
if FLAMERRANGE != QUARTER*4 {QUARTER: QUARTER + 1}
HITSFLAMER: QUARTER + 1d5
result: HITSFLAMER*([wounds caused] > 0)}

function: horde mag killed{
if FLAMER {result: [flamer horde kills]}
if BLAST {result: (HITS*BLAST + (HITS>0)*EXPLOSIVEDAMAGE)*([wounds caused] > 0)*(DEVISTATING+1)}
result: ((HITS+(HITS>0)*EXPLOSIVEDAMAGE)*([wounds caused] > 0))*(DEVISTATING+1)}

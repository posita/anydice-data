\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{12..16}


ATTACKMOD: 4
DMGBONUS: 1
PROFICIENCY: 3
ACCBONUS: 1
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll 1d20 vs ENEMYAC] named "Mischief"
SNEAK: 3d6
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS + SNEAK
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS + 2 * SNEAK
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "Mischief w/ advantage"
output ATTACKTIMESd[attackroll [highest 1 of 3d20] vs ENEMYAC] named "Mischief w/ Elven Accuracy"

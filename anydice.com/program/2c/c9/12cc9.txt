function: explode N:n{
  if N = 4 {result:4 + [explode d8]}
  result:N
}

function: critroll N:n {
  if N=4 {result: 4 + [critroll d8] + [critroll d8]}
  result:N
}

function: attackroll N:n {
  if N=4{result: [critroll d8] + [critroll d8]}
  if N=20{result: [critroll d8] + [critroll d8]}
  if N>=8{result: [explode d8]}
  result: 0
}

set "maximum function depth" to 5
output [attackroll d20]

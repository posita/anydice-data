LEVEL: 1

ABILITYMOD: 3
DICEPA: 1

WEAPONDICE: 1d8
DAMAGEMOD: 2

/Don't care/
TOHIT: ABILITYMOD + PB
AC: 20

function: attack D:n plus BONUS:n vs AC:n  {
    if D = 20 { result: 1 } 
    else if D = 1 { result: 0}
    else if D + BONUS >= AC { result: 1 } 
    else { result:0}
}

loop AC over {AC} {
    output [attack d20 plus TOHIT vs AC] named "[AC]"
}

output [highest 1 of DICEPAd20]
output ABILITYMOD + WEAPONDICE + DAMAGEMOD

function: reroll A:n when less than B:n with C:d {
    if A<B {
        result: C
    } else {
        result: A
    }
}

BB:     2d8
STR:    4
STAFFD: 1d6
STAFF:  [reroll STAFFD when less than 3 with STAFFD] + STR
PAM:    [reroll 1d4 when less than 3 with 1d4] + STR

output STAFF + PAM named "Full Combo"
output STAFF + BB named "Booming Blade"
output 2dSTAFF + 2dBB named "Booming Blade x2"

output 2dSTAFF + BB + PAM named "Hasted Combo"

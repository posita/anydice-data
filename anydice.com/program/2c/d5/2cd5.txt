function: ceiling M:n and N:n{
 if N > M { result: M }
 result: N
}

function: explode DIE:d {
 MAX: [maximum of DIE]
 result: [explode DIE helper]
}

function: explode N:n helper {
 if N = MAX { result: N - 1 + [explode DIE helper] }
 result: N
}

set "maximum function depth" to 10

WHITE:d{0, 0, 0, 0, 1, 1}
BLUE: d{0, 0, 0, 1, 1, 2}
RED:  d{0, 0, 0, 1, 2, 3}

H:  d{0, 0, 0, 0, 3, 4}
A:  d{0, 0, 1, 1, 1, 2}
R:  d{0, 0, 0, 0, 0, 1}

output  [ceiling 10 and 2dWHITE + 1dBLUE]
output  [ceiling 10 and 2dWHITE + 1d[explode RED]]

loop N over {1, 2, 3, 4, 5, 7, 9, 10} {
  DADOS_BASE: 2
  DIFICULDADE: N + 10
  NIVEL_PERSONAGEM: N
  EXCEDENCIA: 1
  PROFICIENCIA: [highest of 1 and [lowest of 5 and NIVEL_PERSONAGEM]]
  PERICIA: [highest of 0 and [lowest of 5 and PROFICIENCIA+EXCEDENCIA]]
  DP: d(PERICIA*2)

  A: (DADOS_BASE)d(d20 >= DIFICULDADE)
  B: (DADOS_BASE)d([count {20} in d20])
  C: (DADOS_BASE)d([count {DIFICULDADE-PERICIA..DIFICULDADE-1} in d20])
  
  output [lowest of DADOS_BASE*2 and (A + B + C)] named "Desafio [N]"
}

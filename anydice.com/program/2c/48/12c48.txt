function: roll S:n B:n{
  if B > 0 {
    B: B+4
    result: [highest 4 of Bd6 + S]
  }
  B: 4 - B
  result: [lowest 4 of Bd6 + S]
}

function: D:n cat C:n vs E:n{
  if D > C {
    result: C - E
  }
  result: -25
}

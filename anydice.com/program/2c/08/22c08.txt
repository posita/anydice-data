
function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

AC: 16
MAIN_STAT: 4
PROF: 3

DUELLING: 2
SUN_BONUS: 2
SUN_DMG_DICE: 1d8
SPEAR_BONUS: 1
SUN_DMG: 2*[attack [highest 3 of 1d20] plus PROF + MAIN_STAT + SUN_BONUS vs AC for SUN_DMG_DICE+SUN_BONUS+DUELLING crit SUN_DMG_DICE on 20]
SPEAR_DMG: 2*[attack d20 plus (PROF + MAIN_STAT + SPEAR_BONUS) vs AC for 2d6+SPEAR_BONUS+DUELLING crit 2d6 on 20]
ASPEAR_DMG: 2*[attack [highest 3 of 1d20] plus (PROF + MAIN_STAT + SPEAR_BONUS) vs AC for 2d6+SPEAR_BONUS+DUELLING crit 2d6 on 20]

output SUN_DMG named "SUN_DMG"
output SPEAR_DMG named "SPEAR_DMG"
output ASPEAR_DMG named "aSPEAR_DMG"

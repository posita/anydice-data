\WS or BS\
SKILL: 3
S: 4
AP: 0
\Doesn't currently work well with flat values over 1.\
DMG: 1
\Bonus to psychic tests\
PB: 0

T: 4
SV: 3
W: 2
\Bonus to Deny the Witch Tests\
DB: 0

\Number of Shots or Attacks\
DICE: 9

\Units in 9"\
NEARUNITS: 2 
\Number of models in target unit\
MODELS: 10
\Units within 3" of targeted unit for thunderbolt\
FARUNITS: 2

function: hitme NN:n {
  result: [count {SKILL..6} in NN d6]
}

function: woundme HH:n {
 if S >= T*2 { 
  result: [count {2..6} in HH d6] 
 }
 if S > T { 
  result: [count {3..6} in HH d6] 
 }
 if S = T { 
  result: [count {4..6} in HH d6] 
 }
 if S*2 <= T { 
  result: [count {6..6} in HH d6] 
 }
 if S < T {
  result: [count {5..6} in HH d6] 
 }
}

function: saveme MM:n {
 SAVE: SV - AP - 1
 result: [count {1..SAVE} in MM d6]
}

function: damageme DD:n {
 result: DDdDMG
}

function: antime AA:n{
 if AA = 6 {
  result: 1d3+3
 }
 if AA >= 3 {
  result: 3
 }
}

function: fallingst FF:n {
 if FF <= 5 {
  if FF <= MODELS {
   result: 1d3
  }
 }
}

function: smite SS:n vs DN:n {
 if SS >= 11 & SS >= DN {
  result: 1d6
 } 
 if SS >= 5 & SS >= DN{
  result: 1d3
 }
 result: 0
}

function: reroll lowest of ROLL:s as REROLL:d if less than MIN:n {
  LOWEST: (#ROLL)@ROLL       \ the lowest die is sorted last \
  if LOWEST >= MIN {
    result: ROLL + 0
  } else {
    REST: {1..#ROLL-1}@ROLL  \ all but the lowest die \
    result: REST + REROLL
  }
}

function: reroll ROLL:s with REROLL:d if less than MIN:n {
   REST: {MIN..#ROLL}@ROLL
   result: REST + REROLL
}

output [reroll lowest of 10d6 as d6 if less than 2]
output [reroll 1d6 with d6 if less than 1]

output [hitme DICE] named "Expected Hits"
output [woundme [hitme DICE]] named "Expected Wounds"
output [saveme [woundme [hitme DICE]]] named "Expected Unsaved Wounds"
output [damageme [saveme [woundme [hitme DICE]]]] named "Full Expected Damage (Can Overflow)"

output [antime 1d6] named "Antimatter Meteor Expected Mortal Wounds"
output [count {4..6} in NEARUNITSd6]d3 named "Cosmic Fire Expected Mortal Wounds"
output [count {4..6} in 3d6]d3 named "Gaze of Death Expected Mortal Wounds"
output [lowest of 10 and [count {6} in MODELSd6]] named "Sky of Falling Stars Expected Mortal Wounds"
output [fallingst 3d6] named "Sky of Falling Stars Expected Mortal Wounds"
output 1d6 >= W named "Time's Arrow Success Rate"
output [count {2..6} in 1d6]d3 + [count {4..6} in FARUNITSd6] named "Transdimensional Thunderbolt Expected Mortal Wounds"

output [smite 2d6+PB vs 0] named "Smite Expected Mortal Wounds"
output [smite 2d6+PB vs 2d6+DB] named "Smite vs Deny Expected Mortal Wounds"

function: DICE:s count min N {
  NS: 0
  loop D over DICE {
    if D >= N {
      NS: NS + 1
    }
  }
  result: NS
}

function: DICE:d take second highest and add number of N {
  result: 2@DICE + [DICE count min N]
}

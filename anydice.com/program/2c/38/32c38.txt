function: DICE:s take second highest and add number of N {
  NS: 0
  HIGHEST: 0
  SECOND_HIGHEST: 0
  loop D over DICE {
    if D > HIGHEST {
      SECOND_HIGHEST: HIGHEST
      HIGHEST: D
    } else if D > SECOND_HIGHEST {
      SECOND_HIGHEST: D
    }

    if D >= N {
      NS: NS + 1
    }
  }

  if NS >= 2 {
    NS: NS - 1
  }

  result: SECOND_HIGHEST + NS
}

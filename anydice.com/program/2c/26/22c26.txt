ATK:  1d20  \ attack roll: use 1@2d20 for advantage or 2@2d20 for disadvantage \
STR: +5
PROF:   +3
HIT:    PROF+STR  \ bonus to attack roll \

AC:     13  \ target AC: attack hits if ATK + HIT >= AC \
CRIT:   19  \ minimum attack roll to crit \

function: attack ROLL:n CRIT:n DMG:n FLAT:n {
  if ROLL <= 1 { result: 0 }
  if ROLL >= CRIT { result: CRIT + FLAT}
  if ROLL + HIT >= AC { result: DMG + FLAT}
  result: 0
}

\ follow the format of: [attack ATK CRIT_DMG NORMAL_DMG FLAT_DMG] \
output 3d[attack ATK 2d8 1d8 8]

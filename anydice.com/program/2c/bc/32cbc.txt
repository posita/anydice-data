STR: 9
DEX: 6
CON: 2
INT: 3
WIS: 3
CHA: 8

DC: 15

function: check ROLL:n plus BONUS:n vs DIFF:n {
    if ROLL+BONUS>=DIFF {
        result: 1
    }
    result: 0
}

function: dtfeatcheck {
    result : [check 1d20 plus STR vs DC] + 
             [check 1d20 plus DEX vs DC] +
             [check 1d20 plus CON vs DC] +
             [check [highest 1 of 2d20] plus INT vs DC] +
             [check 1d20 plus WIS vs DC] +
             [check 1d20 plus CHA vs DC]
}

output [dtfeatcheck]       named "Number of Successes for Downtime Feat Check (Borrowed Knowledge + Adv. on Int check)"

SIZEA: 12
SIZEB: 0
TN: 3
DICEPOOLA: 24d6
DICEPOOLB: 6d6

function: maximum C:d {
  result: [maximum of C] / #C
}

function: rollera {
  if TN = 1 {
    A: DICEPOOLA
  }
  else {
    A: [count {TN..[maximum DICEPOOLA]} in DICEPOOLA]
  }
  result: A

}

function: rollerb {
  if TN = 1 {
    B: DICEPOOLB
  }
  else {
    B: [count {TN..[maximum DICEPOOLB]} in DICEPOOLB]
  }
  result: B + 12 + (4d3-8)

}
  output [rollera] named "Raw Dicepool"
  output [rollerb] named "Modified Dicepool"

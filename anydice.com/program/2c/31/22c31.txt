function: base cmi { result: 10 }
function: headbands { result: 1 }
function: cursedness { result: 0 }
function: guidance { result: 1d4 }

function: buffed cmi { result: [base cmi]+[headbands]+[cursedness]+[guidance] }


function: adv X {
  result: [highest of X and X]
}
function: collapse nonpositive X:n {
  if X>0 {result: X} else {result: -99}
}

function: excess successes {
  result: [adv d20]+[buffed cmi] - (10+3*0)
}
function: G:n gem {
  result: [collapse nonpositive [G gem helper]/3]
}
function: G:n gem helper {
  if G>1 {
    result: [G-1 gem helper]+[adv d20]-20
  } else {
    result: [excess successes]
  }
}

output [1 gem] named "boundMana,%P(succ) -- 1 gems"
output [2 gem] named "boundMana,%P(succ) -- 2 gems"
output [3 gem] named "boundMana,%P(succ) -- 3 gems"
output [4 gem] named "boundMana,%P(succ) -- 4 gems"

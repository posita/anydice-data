function: from INIT:n rolling X:s {
 result: 4*(INIT + [count 3 in X])
}

function: trunc X:n to MAX:n{
 if X >= MAX { result: [trunc X to MAX] }
 else { result: X}
}

output [from -0 rolling 12d6] named "Original method"
output [trunc [from -1 rolling 12d6] to 20] named "Reroll 20s"
output [from -0 rolling 18d6] named "Lower minimum"
output [trunc [from 2 rolling 18d6] to 20] named "Lower minimum, reroll 20s"

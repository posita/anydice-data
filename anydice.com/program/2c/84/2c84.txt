function: reroll SIDES:n ones ROLL:n {
 if 1 < ROLL { result: ROLL }
 else { result: dSIDES }
}

function: TIMES:n reroll SIDES:n {
 SUM: 0
 loop N over {1..TIMES} {
  SUM: SUM + [reroll SIDES ones 1dSIDES]
 }
 result: SUM
}

function: TIMES:n roll SIDES:n {
 SUM: 0
 loop N over {1..TIMES} {
  SUM: SUM + 1dSIDES
 }
 result: SUM
}

DIESEQ: {}


output [3 reroll 6]
output [3 roll 6]

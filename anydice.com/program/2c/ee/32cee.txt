STR: 4
RAGE: 2
PROF: 3

function: maul MAGIC:n vs AC:n {
  WEAPON: 2d6
  HIT: STR + PROF + MAGIC + 1d20 > AC
  DMG: WEAPON + STR + RAGE + MAGIC
  ATK: HIT * DMG
  result: 2dATK
}

function: handaxes MAGIC:n vs AC:n {
  WEAPON: 1d6
  HIT: STR + PROF + MAGIC + 1d20 > AC
  DMG: WEAPON + STR + RAGE + MAGIC
  DMGOFFHAND: WEAPON + RAGE + MAGIC
  ATK: HIT * DMG
  result: 2dATK + HIT * DMGOFFHAND
}

MAGIC: 1
AC: 16
loop AC over {10, 12, 14, 16} {
  output [maul MAGIC vs AC] named "AC[AC] Maul+[MAGIC]/Extra Attack"
}
loop AC over {10, 12, 14, 16} {
  output [handaxes MAGIC vs AC] named "AC[AC] Hand Axes+[MAGIC]/Extra Attack"
}

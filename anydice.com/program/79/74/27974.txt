function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

ATTACK:0
MISS: 9
HIT: 1d6+1d8 + 4
CRITICALHIT: 2d6 + 2d8 + 4

loop AC over {5..10}{
output 2d[attack [highest 1 of 2d20] vs AC] named "2 hits at [AC]"
output 3d[attack d20 vs AC] named "3 hits at [AC]"
}



CRITRANGE: 20
AC: 15
GUNK_ATT: 1


function: outcome of  ROLL:n at BONUS:n vs AC:n {
 if ROLL >= CRITRANGE {
    result: 1 \Crit\
    }
 if ROLL = 1 | ROLL + 2 < AC - BONUS { \miss by >=3\
    result: 0 
}
 if ROLL >= AC - BONUS { \natural hit\
    result: 2
}
 if ROLL +2 >= AC - BONUS { \miss by < 3, spend ki point to hit\
    result: 3
}
result: 4
}



output [outcome of 1d20 at GUNK_ATT vs AC] named "Roll Outcome"

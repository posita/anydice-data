AC: 14
DEX_MOD: 3
PROF_BONUS: 2
CRITABLEDAM: 1d6
HITDAM: DEX_MOD + 2

    if ROLL >= CRITRANGE {
    result: CRITABLEDAM + CRITABLEDAM + HITDAM
    }
        if ROLL = 1 {
        result: 0
        }
            if ROLL >= AC - ATTACKBONUS {
            result: CRITABLEDAM + HITDAM
            }
                else {
                result: 0
                }
}

F: [roll 1d20 at plus PROF_BONUS + DEX_MOD vs ac AC for CRITABLEDAM plus HITDAM]
output F named "Soul Captcha level 1"

function: hit A:n against AC:n {
  if A = 20 {result: 5d12 + 9 + 4 + 8d6 + 10 + 8}
  if A + 9 + 8 - 5 >= AC & A != 1 {result: 1d12 + 9 + 4 + 4d6 + 10 +8}
  result: 0
}

loop AC over {16..26}{
output [hit [highest 1 of 2d20] against AC] + [hit [highest 1 of 2d20] against AC] + [hit [highest 1 of 2d20] against AC] + [hit [highest 1 of 2d20] against AC] named "[AC]"
}

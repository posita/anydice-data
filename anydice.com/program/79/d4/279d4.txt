X: 3d6
Y: 3d6
Z: 3d6

A: [highest of [highest of X and Y] and [highest of Y and Z]]
B: [lowest of [lowest of X and Y] and [lowest of Y and Z]]

function: compare A:s and B:s to G:s and H:s {
  result: ((A = G) & (B = H))
}
loop I over {3..18}{
    loop J over{3..18}{
    output [compare A and B to I and J] named "[I]x[J]"
    }
}

MISS: 0
HIT: 1
DAMAGE: 2

function: ssb TOHIT:n TODAMAGE:n {
  if TOHIT >= 7 {
    if TODAMAGE >= 7 {result: DAMAGE}
    result: HIT
  }
  result: MISS
}

output [ssb 2d6+1 2d6+1]




loop HITMODIFIER over {-5..5} {
  loop DAMAGEMODIFIER over {-5..5} {
    output [ssb 2d6+HITMODIFIER 2d6+DAMAGEMODIFIER ] named "TEST([HITMODIFIER],[DAMAGEMODIFIER])"
  }
}

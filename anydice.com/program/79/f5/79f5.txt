DIETYPE: d10
DIFFICULTY: 10

function: roll DICE:s against DC:n {
  if [check if 2@DICE and 3@DICE beats DC] { result: 1@DICE }
  if [check if 1@DICE and 3@DICE beats DC] { result: 2@DICE }
  if [check if 1@DICE and 2@DICE beats DC] { result: 3@DICE }
  
  result: 0
}

function: check if ONE and TWO beats DC:n {
  result: ONE + TWO >= DC
}

output [roll 3dDIETYPE against DIFFICULTY] named "Effect with a difficulty of [DIFFICULTY]"

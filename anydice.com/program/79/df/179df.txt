\**** Damage calculator normal Vs advantage ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: attackrollwar ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGEWAR } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGEWAR }
 result: 0
}

function: attackrollgloom ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGEGLOOM } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGEGLOOM }
 result: 0
}

function: count highest dupes in DICE:s {
  MAX: 1
  loop X over {1..1@DICE} {
    COUNT: X = DICE
    if COUNT > 1 & COUNT > MAX { MAX: COUNT }
  }
  result: MAX
}

output [count highest dupes in 5d10] named "Lake 5 Roll"
output [count highest dupes in 7d10] named "Lake 7 Roll"

loop N over {2..12} {
  output [count highest dupes in Nd6] named "[N]d6 pool largest match"
}

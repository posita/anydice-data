function: DIE:n hit AC:n with BON:n for DAM:n crit DAMC:n {
    if DIE = 20 {result: DAMC}
    if DIE = 1 {result: 0}
    if DIE + BON > AC {result: DAM}
    result: 0
}

    if DIE = 20 {result: 2dDAM+BON}
    if DIE = 1 {result: 0}
    if DIE + BON > AC {result: DAM+BON}
    result: 0
}

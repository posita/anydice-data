function: docrit D:n N:n {
 if D = 20 {
  result: 3dN
 }
 else {
  result: 0
 }
}

function: dobonusattack D:n N:n {
 if D > 0 {
  result: D+[docrit [highest 1 of 2d20] N]
 }
 else {
  result: 0
 }
}

function: brutalcrit N:n {
 D: 0
 D: [docrit [highest 1 of 2d20] N]
 D: D+[docrit [highest 1 of 2d20] N]
 D: [dobonusattack D N]
 result: D
}

loop X over {4,6,8,10,12,20} {
 output [brutalcrit X] named "Brutal Critical for d[X] weapon"
}

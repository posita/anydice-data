\
 0: Disaster
 1: Conflict
 2: Triumph
 3: Twist
\

function: action SCORE:n with TRIUMPH:n {
 if SCORE = 6 {
  result: 2 + (TRIUMPH = 6)
 }
 result: SCORE >= 4
}

function: reaction SCORE:n with TRIUMPH:n {
 result: 6 - SCORE
}


loop DIFFICULTY over {0..2} {
 loop POOL over {0..10} {
  SCORE: 0
  TRIUMPH: 0

  if POOL <= DIFFICULTY {
   FUMBLE: ((DIFFICULTY + 1)*2) - POOL
   SCORE: FUMBLE@FUMBLEd6
  } else {
   SCORE: (DIFFICULTY+1)@POOLd6
   
   if POOL > DIFFICULTY+1 {
    TRIUMPH: (DIFFICULTY+2)@POOLd6
   }
  }
  output [action SCORE with TRIUMPH] named "[POOL]d at [DIFFICULTY]"
 }
}

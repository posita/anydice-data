function: best of SEQ:s {
  result: 1 @ [sort SEQ]
}

function: seqs of A:s from SEQ:s {
  MATCH:1
  loop N over A {
    MATCH : MATCH * [SEQ contains N]
  }
  result: MATCH * A
}

function: sets of A:n from SEQ:s {
   NO: [count A in SEQ] > 1
   result: NO * [count A in SEQ] * A
}

function: setcombinations of A:n and B:n from SEQ:s {

   I: [sets of A from SEQ]
   J: [sets of B from SEQ]

   result: I + J
}

function: pokersum A:s {
  result: [best of {
      [sets of 1 from A],
      [sets of 2 from A],
      [sets of 3 from A],
      [sets of 4 from A],
      [sets of 5 from A],
      [sets of 6 from A],
      [setcombinations of 1 and 2 from A],
      [setcombinations of 1 and 3 from A],
      [setcombinations of 1 and 4 from A],
      [setcombinations of 1 and 5 from A],
      [setcombinations of 1 and 6 from A],
      [setcombinations of 2 and 3 from A],
      [setcombinations of 2 and 4 from A],
      [setcombinations of 2 and 5 from A],
      [setcombinations of 2 and 6 from A],
      [setcombinations of 3 and 4 from A],
      [setcombinations of 3 and 5 from A],
      [setcombinations of 3 and 6 from A],
      [setcombinations of 4 and 5 from A],
      [setcombinations of 4 and 6 from A],
      [setcombinations of 5 and 6 from A],
      [seqs of {1,2,3,4,5} from A],
      [seqs of {2,3,4,5,6} from A]
    }]
}

output [pokersum 5d6]

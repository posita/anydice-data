set "maximum function depth" to 10
function: ttk HP:n dmg D:d roll N:n R:d {
 if HP<=0{result:0}
 result:1+[ttk HP-D*N dmg D roll R R]
}
function: N:n plus B:n hits AC:n{
 if N=20|N+B>=AC+10{result:2}
 if N+B>=AC{result:1}
 result:0
}
R:[d20 plus 7 hits 16]
DMG:[highest of 0 and 1d8-1]
output [ttk 4 dmg DMG roll R R] named "Skeleton Guard"

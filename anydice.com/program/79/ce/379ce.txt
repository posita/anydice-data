function: attack A:n B:n roll D:n BONUS:n vs AC:n {
  if D > 17 {
    result: A+A+B
  } else if D = 1 {
    result: 0
  } else if D + BONUS >= AC {
    result: A+B
  } else {
    result: 0
  }
}

function: attack A:d B:d roll D:d BONUS:d vs AC:d hits M:n {
  TOTAL: 0
  loop N over {1..M}{
    DAMAGE: [attack A B roll D BONUS vs AC]
    TOTAL: TOTAL+DAMAGE
  }
  result: TOTAL
} 

AC: 20
ADV: [highest 1 of 2d20]
DIS: [lowest 1 of 2d20]
NORM: d20
D: ADV

output [attack 1d10 9 roll D 15 vs AC hits 4]

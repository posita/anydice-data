function: usts ROLL:s mastery MASTERY:n difficulty DIFFICULTY:n obstacle OBSTACLE:n critical X:n fumble Y:n {
 WIN: ROLL=6
 LOSE: ROLL=1
 if X=1 { CRITICAL: WIN } else { CRITICAL: 0 }
 if Y=1 { FUMBLE: LOSE } else { FUMBLE: 0 } 
 SUCCESS: ROLL>=DIFFICULTY
 if SUCCESS>MASTERY { BONUS: MASTERY } else { BONUS: SUCCESS }
 result: BONUS+MASTERY+CRITICAL-FUMBLE-OBSTACLE
}

output [usts 10d6 mastery 2 difficulty 4 obstacle 0 critical 1 fumble 1]
output [usts 10d6 mastery 4 difficulty 4 obstacle 0 critical 1 fumble 1]
output [usts 10d6 mastery 6 difficulty 4 obstacle 0 critical 1 fumble 1]
output [usts 10d6 mastery 8 difficulty 4 obstacle 0 critical 1 fumble 1]

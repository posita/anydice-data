function: usts ROLL:s mastery MASTERY:n difficulty DIFFICULTY:n obstacle OBSTACLE:n {
 WIN: ROLL=6
 LOSE: ROLL=1
 CRITICAL: WIN
 if WIN>MASTERY { CRITICAL: MASTERY } else { CRITICAL: WIN }
 if LOSE>MASTERY { FUMBLE: MASTERY } else { FUMBLE: LOSE }
 SUCCESS: ROLL>=DIFFICULTY
 if SUCCESS>MASTERY { BONUS: MASTERY } else { BONUS: SUCCESS }
 result: BONUS+MASTERY+CRITICAL-FUMBLE-OBSTACLE
}

output [usts 2d6 mastery 2 difficulty 3 obstacle 0]
output [usts 4d6 mastery 2 difficulty 3 obstacle 0]
output [usts 6d6 mastery 2 difficulty 3 obstacle 0]
output [usts 8d6 mastery 2 difficulty 3 obstacle 0]

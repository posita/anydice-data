function: bas ROLL:s dc B:n {
 X: ROLL>=B
 result: X
}
function: lmt ROLL:s m B:n dc C:n {
 X: ROLL>=C
 if X<B { X: B }
 result: X
}
function: mine ROLL:s skill SKILL:n mastery MASTERY:n difficulty DIFFICULTY:n obstacle OBSTACLE:n {
 SUCCESS: ROLL>=DIFFICULTY
 if SUCCESS<MASTERY { result: 2*MASTERY-SUCCESS-OBSTACLE } else { result: SUCCESS-OBSTACLE }
}

X: d{0,0,1,1,1,1}

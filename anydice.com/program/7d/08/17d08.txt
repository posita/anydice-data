function: curradvance D:n skill N:n {
 if D = 1 { result: 0 }
 if D = 2 { result: N - 1}
 result: (D-2)/2 + N
}

function: newadvance D:n skill N:n {
  if D = 1 { result: 0 }
  result: D + N
} 

output [curradvance d8 skill 4] >= 6 named "Old"
output [newadvance d8 skill 8] >= 14 named "New"

output [curradvance d4 skill 3] >= 4 named "Old"
output [newadvance d4 skill 6] >= 10 named "New"

output [curradvance d10 skill 7] >= 9 named "Old"
output [newadvance d10 skill 14] >= 20 named "New"

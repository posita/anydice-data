function: sign of NUMBER:n {
  if NUMBER > 0 { result: +1 }
  if NUMBER < 0 { result: -1 }
  result: 0
}

HERO: 2d4
ENEMY: 1d12 

WINNER: [sign of HERO - ENEMY] 
output WINNER named "single round: +1 = Hero win, -1 = Enemy win, 0 = tie"

function: restrict ROLL:n to RANGE:s {
  if ROLL = RANGE { result: ROLL }
  else { result: d{} }
}

ROUND: [restrict WINNER to {+1,-1}] = +1
output ROUND named "single round with rerolls: 1 = Hero win, 0 = Enemy win"

output 5dROUND named "number of Hero wins over 5 rounds"

output 5dROUND >= 3 named "probability of Hero winning at least 3 rounds out of 5"

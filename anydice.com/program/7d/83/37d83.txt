FOUR_ABOVE: 2d6*4
THREE_ABOVE: 2d6*3
TWO_ABOVE: 2d6*2
ONE_ABOVE: 2d6*1

function: attackroll for ROLL:n {
 if ROLL > 21 { result: FOUR_ABOVE}
 if ROLL > 20 { result: THREE_ABOVE}
 if ROLL > 19 { result: TWO_ABOVE}
 if ROLL > 18 { result: ONE_ABOVE}
 result: 0
}

loop MOD over {14,15,16,17,18} {
 output [attackroll for 1d10+MOD] named "[MOD]"
}

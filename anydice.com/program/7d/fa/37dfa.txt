function: ROLL:n at BONUS:n crits AC:n {
  if ((ROLL = 20) | (ROLL + BONUS >= AC + 10)) { result: 1 }
  result: 0
}

function: ROLL:n at BONUS:n hits AC:n {
  if (ROLL + BONUS >= AC) { result: 1 }
  result: 0
}

function: ROLL:n at BONUS:n against AC:n dealing DAMAGE:d criting CRIT_DAMAGE:d {
  if ([ROLL at BONUS crits AC]) { result: CRIT_DAMAGE}
  if ([ROLL at BONUS hits AC]) { result: DAMAGE }
  result: 0
}

function: ROLL:n at BONUS:n against AC dealing DAMAGE:d {
  result: [ROLL at BONUS against AC dealing DAMAGE criting 2*DAMAGE]
}

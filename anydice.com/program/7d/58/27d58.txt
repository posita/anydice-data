function: hit A:n E:n N:n I:n B:n R:n M:n{
 Q: 2*N
 if A>=20 {result: [dam [highest N of QdI]+B-M R]}
 if A>E {result: [dam NdI+B-M R]}
 result: R
}

function: dam A:n B:n{
 if A>B {result: A}
 result: B
}

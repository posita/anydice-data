function: usts ROLL:s m MASTERY:n tn TARGET:n ob OBSTACLE:n {
 CRITICAL: ROLL=6
 FUMBLE: ROLL=1
 if CRITICAL>MASTERY { CRITICAL: MASTERY }
 if FUMBLE>MASTERY { FUMBLE: MASTERY }
 SUCCESS: ROLL>=TARGET
 if SUCCESS>MASTERY { BONUS: MASTERY } else { BONUS: SUCCESS }
 result: BONUS+CRITICAL-FUMBLE/2-OBSTACLE
}

output [usts 8d6 m 2 tn 3 ob 0]
output [usts 10d6 m 2 tn 3 ob 0]
output [usts 12d6 m 2 tn 3 ob 0]

function: attack ROLL:n CRIT:n AC:n DAM:d CRITDAM:d {
 if ROLL >= CRIT { result: DAM + CRITDAM }
 if ROLL = 1 { result: 0 }
 if ROLL >= AC { result: DAM }
 result: 0
}

loop AC over {16} {
  loop BRUTAL over {0} {
    output [attack 1@2d20 19 AC 1d12 3d12] named "Barb Focus"
    output [attack 1@2d20 18 AC 1d12 2d12] named "Champ Focus"
  }
}

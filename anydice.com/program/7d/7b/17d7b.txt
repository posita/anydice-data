TYPE:6
COUNT:4
STATS: COUNTdTYPE

BODY: (1@(STATS) - TYPE / 2)
MIND: (3@(STATS) - TYPE / 2)
SPIRIT: (2@(STATS)  - TYPE / 2)

AVG: STATS / #(STATS)
STDD: 0
loop N over {#(STATS)}{
   X: N@(STATS) - AVG
   X: X ^ 2
   STDD: STDD + X
}
STDD: STDD / #(STATS)

output AVG named "Average"
output STDD named "Standard Deviation"

output BODY named "Body"
output MIND named "Mind"
output SPIRIT named "Spirit"

BODY: BODY + 10
output BODY named "DC"
loop N over {1..7}{
   X: 1d20 + N >= BODY
   output X named "+[N] Hit"
}

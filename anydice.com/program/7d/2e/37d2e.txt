set "explode depth" to 50
set "maximum function depth" to 10

function: usagedie D:d keep on TN:n smallest size S:n {
 SIZE: [maximum of D]
 if SIZE <= S {
  result: 1 + [explode D>=TN]
 } else {
  result: 1 + [explode D>=TN] + [usagedie d(SIZE-2) keep on TN smallest size S]
 }
}

output [lowest of [usagedie d6 keep on 3 smallest size 6] and 50] named "# of d6 rolls"

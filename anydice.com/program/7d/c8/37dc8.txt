set "maximum function depth" to 7
function: ttk HP:n dmg D:d roll N:n R:d {
 if HP<=0{result:0}
 result:1+[ttk HP-D*N dmg D roll R R]
}
function: N:n plus B:n hits AC:n{
 if N=20|N+B>=AC+10{result:2}
 if N+B>=AC{result:1}
 result:0
}
loop N over {1..10} {
  output 2d((N+1)d4/2) named "Electric Arc [N]"
  if [{1,3,5,7,9} contains N] {
    output ((N+1)/2)d6 named "Daze [N]"
  }
}

function: armor A B C D{
  R: 0
  loop X over {1..D}{
    T: Bd6 + (C-A)
    N: [tester T]
    R: R + N
  }
  result: R
}

function: tester A:n{
  if A > 0{
    result: A
  }
  else{ 
    result: 0
  }
}

output [armor 21 2 (15+8+2) 5] named "5 attacks w/ no force boost & no boost dmg"

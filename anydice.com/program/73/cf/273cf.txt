PROFICIENCY: 3
ABILITY_BONUS: 4
MAGIC_BONUS: 2
TO_HIT: (PROFICIENCY+ABILITY_BONUS+MAGIC_BONUS)
CRIT_ON: 20
DAMAGE_DICE_MA: 1d10
DAMAGE_DICE_BA: 1d4
DAMAGE_STATIC: (ABILITY_BONUS+MAGIC_BONUS)
DAMAGE_MA: (DAMAGE_DICE_MA+DAMAGE_STATIC)
DAMAGE_BA: (DAMAGE_DICE_BA+DAMAGE_STATIC)
DAMAGE_SMITE: 4d8
function: attack ROLL:n vs AC:n with DAMAGE_DICE:n {
 if ROLL = 1 { result: 0}
 if ROLL >= CRIT_ON { result: DAMAGE_DICE + DAMAGE_DICE + DAMAGE_STATIC + DAMAGE_SMITE + DAMAGE_SMITE}
 if ROLL + TO_HIT >= AC { result: DAMAGE_DICE + DAMAGE_STATIC}
 result: 0
}
EDV_MA: [attack d20 vs 18 with 1d10]
EDV_BA: [attack d20 vs 18 with 1d4]
EDV_A: EDV_MA + EDV_MA + EDV_BA
EDV_B: [attack [highest 1 of 2d20] vs 18 with 1d10] + [attack [highest 1 of 2d20] vs 18 with 1d4]
output EDV_A named "EDV Normal Full Attack"
output EDV_B named "EDV Shove/Attack/BA Attack"

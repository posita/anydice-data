function: pb L {
  if L < 5 {
    result: 2
  } else if L < 9 {
    result: 3
  } else if L < 13 {
    result: 4
  } else if L < 17 {
    result: 5
  } else {
    result: 6
  }
}

function: spelldc MOD level L {
  result: 8 + [pb L] + MOD
}

function: attack bonus B at level L {
  result: B + [pb L]
}

function: attack ATK_DICE:n plus ATK_FIXED level L for DMG_DICE plus DMG_FIXED on crit DMG_CRIT vs AC {
  if ATK_DICE = 1 {
    result: 0
  } else if ATK_DICE = 20 {
    result: (2d(DMG_DICE))+DMG_FIXED+DMG_CRIT
  } else {
    result: ((ATK_DICE+[attack bonus ATK_FIXED at level L])>=AC)*(DMG_DICE+DMG_FIXED)
  }
}

function: attack ATK_FIXED level L for DMG_DICE plus DMG_FIXED on crit DMG_CRIT vs AC {
  result: [attack d20 plus ATK_FIXED level L for DMG_DICE plus DMG_FIXED on crit DMG_CRIT vs AC]
}

function: attack ATK_FIXED level L for DMG_DICE plus DMG_FIXED vs AC {
  result: [attack ATK_FIXED level L for DMG_DICE plus DMG_FIXED on crit 0 vs AC]
}

function: cantrip DC for DMG save SAVE plus SAVE_BONUS {
  result: (SAVE <= DC) * DMG
}

function: cantrip DC for DMG save SAVE_BONUS {
  result: [cantrip DC for DMG save d20 plus SAVE_BONUS]
}

function: once DIE to DAMAGE:n {
  if DAMAGE > 0 {
    result: DIE + DAMAGE
  } else {
    result: DAMAGE 
  }
}

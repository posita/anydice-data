AC: 20
PROFICIENCY: 3
ABILITY_BONUS: 4
MAGIC_BONUS: 1
TO_HIT: (PROFICIENCY+ABILITY_BONUS+MAGIC_BONUS)
DAMAGE_DICE: (1d8+2d6)
DAMAGE_STATIC: (ABILITY_BONUS+MAGIC_BONUS)
DAMAGE: (DAMAGE_DICE+DAMAGE_STATIC)
EDV_A: ((d20+TO_HIT)>=AC)*DAMAGE + ((d20+TO_HIT)>=AC)*DAMAGE  + (2/20)*DAMAGE_DICE
EDV_B: (([highest 1 of 2d20]+TO_HIT)>=AC)*DAMAGE + (1/20)*DAMAGE_DICE
EDV_C: EDV_B + EDV_B
output EDV_A named "EDV Two Attacks"
output EDV_B named "EDV Shove-Attack"
output EDV_C named "EDV, FIGHTING SPIRIT"

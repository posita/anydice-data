\* output [count {1} in 3d6]  *\
output [count {1..1} in 3d6]

function: dupes in DICE:s in RANGE:s {
  D: 0
  if ( [count RANGE in DICE] >= 1) {
   D: 1
   loop X over { 2..#DICE } {
     if ( (X-1)@DICE = X@DICE & [RANGE contains (X-1)@DICE]) { D: D + 1}
   }
  }
  result: D
}

MINFACE: 1
MAXFACE: 6
NDICE: {3..6}

loop ND over NDICE {
 loop V over {MINFACE..MAXFACE} {
  output [dupes in NDd6 in {1..V}] named "Se tiran [ND]D (caras válidas del 1 al [V])"
 }
}

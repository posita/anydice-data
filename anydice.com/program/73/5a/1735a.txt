function: test ROLL:s when TARGET:n {
  if ROLL + 0 != TARGET { result: -1 }
  result: 1@ROLL = 2@ROLL | 2@ROLL = 3@ROLL
}
function: reroll ROLL:n as REROLL:d if negative {
  if ROLL < 0 { result: REROLL }
  result: ROLL
}
function: reroll ROLL:d if negative {
  loop I over {1 .. 20} {
    ROLL: [reroll ROLL as ROLL if negative]
  }
  result: [highest of 0 and ROLL]
}
loop TARGET over {3..18} {
  output [reroll [test 3d6 when TARGET] if negative] named "chance of double when 3d6 = [TARGET]"
}

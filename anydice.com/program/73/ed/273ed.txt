  result: 1@2dDIE

function: reroll DIE:n as REROLL if lower than N {
  if DIE < N { result: REROLL }
  else { result: DIE }
}

function: swing ATK_DICE:n plus ATK_MOD:n for DMG_DICE plus DMG_MOD crit bonus CRIT_DICE plus CRIT_MOD threshold CRIT vs AC {
  if ATK_DICE = 1 { result: 0 }
  else if ATK_DICE >= CRIT { result: 2d(DMG_DICE+CRIT_DICE)+DMG_MOD+CRIT_MOD }
  else if ATK_DICE+ATK_MOD >= AC { result: DMG_DICE+DMG_MOD }
  else { result: 0 }
}

function: resolve DMG:n swing N attack ATK_DICE plus ATK_MOD for DMG_DICE plus DMG_MOD crit bonus CRIT_DICE plus CRIT_MOD threshold CRIT first hit FH_DICE plus FH_MOD vs AC {
  if DMG > 0 {
    result: [swing ATK_DICE plus ATK_MOD for DMG_DICE plus DMG_MOD crit bonus CRIT_DICE plus CRIT_MOD threshold CRIT vs AC]
  } else {
    result: [swing ATK_DICE plus ATK_MOD for DMG_DICE+FH_DICE plus DMG_MOD+FH_MOD crit bonus CRIT_DICE plus CRIT_MOD threshold CRIT vs AC]
  }
}

function: attack T x ATK_DICE plus ATK_MOD flurry FLURRY for DMG_DICE plus DMG_MOD crit bonus CRIT_DICE plus CRIT_MOD threshold CRIT first hit FH_DICE plus FH_MOD vs AC {
  DMG: 0
  loop N over {1..T} {
    DMG: DMG + [resolve DMG swing N attack ATK_DICE plus ATK_MOD for DMG_DICE plus DMG_MOD crit bonus CRIT_DICE plus CRIT_MOD threshold CRIT first hit FH_DICE plus FH_MOD vs AC]
  }
  result: DMG
}

function: reroll DIE:n lower than N {
  if DIE < N { result: [regular] }
  else { result: DIE }
}

function: swing ATK_DICE:n plus ATK_MOD:n for DMG_DICE plus DMG_MOD on crit CRIT vs AC {
  if ATK_DICE = 1 { result: 0 }
  else if ATK_DICE = 20 { result: 2dDMG_DICE+DMG_MOD+CRIT }
  else if ATK_DICE+ATK_MOD >= AC { result: DMG_DICE+DMG_MOD }
  else { result: 0 }
}

  }
}

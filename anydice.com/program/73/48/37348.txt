function: dicepool POOL:s SUCC:s
{
  X: 0
  loop P over POOL 
  {
    if P >= SUCC
    {
      X: X + 1
    }
  } 
  result: X
}
output [dicepool 3d6  4] named "d6"
output [dicepool 3d8  4] named "d8"

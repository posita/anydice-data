function: interpret ROLL:n {
  if [count {6} in ROLL] > 0 { result: 6 }
  if [count {5} in ROLL] > 0 { result: 5 }
  if [count {4} in ROLL] > 0 { result: 4 }
  if [count {3} in ROLL] > 0 { result: 3 }
  if [count {2} in ROLL] > 0 { result: 2 }
  if [count {1} in ROLL] > 0 { result: 1 }
  result: -1
}

output [ interpret 2d6 ]

\loop N over {1..10} {
  output [interpret Nd6]
}\

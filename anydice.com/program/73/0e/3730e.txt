TARGET: 4
WILDDIE: 1d6
VIGORDIE:  1d8
SPIRITDIE:  1d10
IGNOREPEN: 2

WORSTPEN: -5
BESTPEN:  -2+IGNOREPEN

loop P over {WORSTPEN..BESTPEN} {
    VIGOR: [highest of [explode VIGORDIE]+P and [explode WILDDIE]+P]
    SPIRIT: [highest of [explode SPIRITDIE]+P and [explode WILDDIE]+P]
    output VIGOR>=TARGET & SPIRIT>=TARGET named "Penalty of [P]"
}

function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d10] }
 result: N >= 8
}

NWOD: [nwod 10 again d10]

function: nsixwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d6] }
 result: N >= 5
}

NSIXWOD: [nsixwod 7 again d6]


loop N over {1..10} {
 output [lowest of 10 and NdNWOD] - [lowest of 10 and NdNSIXWOD] named "[N]d"
}

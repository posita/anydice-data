\* output [count {1} in 6d6] *\

function: dupes in DICE:s in RANGE:s {
  D: 0
  loop X over {2..#DICE} {
    if ((X-1)@DICE = X@DICE & [RANGE contains (X-1)@DICE]) { D: D + 1}
  }
  result: D
}


NDICE: {3..6}
START: 1

loop ND over NDICE {
 loop V over {START..6} {
  output [dupes in NDd6 in {1..V}] named "Nivel [V] [ND]D"
 }
}

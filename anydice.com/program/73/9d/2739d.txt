function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

function: gwf DICE:d {
    result: [reroll {1,2} on DICE]
}

GWF_GS: 2d[gwf d6]
GWF_GLAIVE: [gwf d10]
GWF_GLAIVE_PAM: [gwf d4]

output 2d6 named "GS" 
output GWF_GS named "GS GWF" 
output 1d10 named "Glaive" 
output GWF_GLAIVE named "Glaive GWF" 
output 1d4 named "PAM" 
output GWF_GLAIVE_PAM named "PAM GWF"

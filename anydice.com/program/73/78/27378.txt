function: pb L {
  if L < 5 {
    result: 2
  } else if L < 9 {
    result: 3
  } else if L < 13 {
    result: 4
  } else if L < 17 {
    result: 5
  } else {
    result: 6
  }
}

function: spelldc MOD lvl L {
  result: 8 + [pb L] + MOD
}

function: attack bonus B lvl L {
  result: B + [pb L]
}

function: attack ATK_DICE:n plus ATK_MOD lvl L for DMG_DICE plus DMG_MOD on crit DMG_CRIT vs AC {
  if ATK_DICE = 1 {
    result: 0
  } else if ATK_DICE = 20 {
    result: (2d(DMG_DICE))+DMG_MOD+DMG_CRIT
  } else {
    result: ((ATK_DICE+[attack bonus ATK_MOD lvl L])>=AC)*(DMG_DICE+DMG_MOD)
  }
}

function: attack ATK_MOD lvl L for DMG_DICE plus DMG_MOD on crit DMG_CRIT vs AC {
  result: [attack d20 plus ATK_MOD lvl L for DMG_DICE plus DMG_MOD on crit DMG_CRIT vs AC]
}

function: attack ATK_MOD lvl L for DMG_DICE plus DMG_MOD vs AC {
  result: [attack ATK_MOD lvl L for DMG_DICE plus DMG_MOD on crit 0 vs AC]
}

function: advantaged ATK_MOD lvl L for DMG_DICE plus DMG_MOD on crit DMG_CRIT vs AC {
  result: [attack 1@2d20 plus ATK_MOD lvl L for DMG_DICE plus DMG_MOD on crit DMG_CRIT vs AC]
}

function: advantaged ATK_MOD lvl L for DMG_DICE plus DMG_MOD vs AC {
  result: [advantaged ATK_MOD lvl L for DMG_DICE plus DMG_MOD on crit 0 vs AC]
}

function: cantrip DMG save SAVE plus SAVE_MOD vs DC {
  result: ((SAVE+SAVE_MOD)<=DC)*DMG
}

function: cantrip DMG save SAVE_MOD vs DC {
  result: [cantrip DMG save d20 plus SAVE_MOD vs DC]
}

function: once DIE to DAMAGE:n {
  if DAMAGE > 0 {
    result: DIE + DAMAGE
  } else {
    result: DAMAGE 
  }
}

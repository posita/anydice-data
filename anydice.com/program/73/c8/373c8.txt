function: dicepool POOL:s
{
  if 1@POOL = 6
  {
    if 2@POOL = 6
    {result: 3}
    else {result: 2}
  }
  if 1@POOL < 4 {result: 0}
  else {result: 1}
}
function: thorn DICE:s
{
   X:0
   loop L over DICE
  {
    if L = 8 | L = 7
    {
      X: X - 1
    }
  } 
  result: X
}
output [dicepool 1d6] named "1d6"
output [dicepool 2d6] named "2d6"

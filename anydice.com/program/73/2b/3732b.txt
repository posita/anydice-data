function: vampire POOL:s TARGET:n {
FAIL: (POOL = 1)
SUCC: (POOL >= TARGET)
CRIT: (POOL = 10)

FAIL: FAIL - CRIT
if FAIL<0 { FAIL: 0 }

result: SUCC-FAIL
}

function: soak POOL:s DIFF:n{
SUCC: POOL>=DIFF
result: SUCC
}

WILLPOWER: 1
EXTRA_ACTION: 1

DEX: 2+4
STR: 2+5

output [vampire (DEX+2-EXTRA_ACTION )d10 7] + WILLPOWER named "Attack"
output [vampire (STR+3-EXTRA_ACTION)d10 6] - [soak 3d10 6] named "Damage"

function: exito D:n TARGET:n{
  if D <= TARGET{ result: 1}
  result: 0
}

function: roll D:n TECNICA:n ESFUERZO:n{
   INCOMPETENCIA: 0
   if TECNICA = 0 { 
      TECNICA: 1 
      INCOMPETENCIA: 1
   }
   RES: [exito D (TECNICA + (ESFUERZO * ACTIVO))]
   
   if RES >= 1 {
      if INCOMPETENCIA= 0{
         RES: RES + [exito 1d6 (TECNICA + ESFUERZO)]
      }
   }
   result: RES
}

function: accion ACTIVO:n VENTAJA:n TECNICA:n ESFUERZO:n{
  X: 0
  loop N over {1..3 + VENTAJA}{
     X: X + [roll 1d6 TECNICA ESFUERZO]
  }

  result: X 
}

loop ESFUERZO over {0..1} {
   loop VENTAJA over {0..3}{
      loop TECNICA over {0..3}{
         loop ACTIVO over {0..3}{
            output [accion ACTIVO VENTAJA TECNICA ESFUERZO ] named "Técnica [TECNICA], ACTIVO [ACTIVO], Esfuerzo [ESFUERZO] y ventaja [VENTAJA]"
         }
      }
   }
}

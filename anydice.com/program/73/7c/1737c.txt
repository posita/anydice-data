function: dupes in DICE:s in RANGE:s {
  D: 1
  loop X over {2..#DICE} {
    if ((X-1)@DICE = X@DICE & [RANGE contains (X-1)@DICE]) { D: D + 1}
  }
  result: D
}

START: 3
NDICE: {3..6}

loop ND over NDICE {
 loop V over {START..6} {
  output [dupes in NDd6 in {1..V}] named "Se tiran [ND]D (caras válidas del 1 al [V])"
 }
}

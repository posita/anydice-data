function: dado D:n{
   result: D
}

function: exito D:n{
  if D = 6{ result: 2}
  if D > 3{ result: 1}
  result: 0
}

function: fallo D:n TECNICA:n{
   if TECNICA = 0 { if D < 6 { result: 1} }
   if TECNICA = 1 { if D < 4 { result: 1} }
   if TECNICA = 2 { if D < 3 { result: 1} }
   if TECNICA = 3 { if D < 2 { result: 1} }
   result: 0
}

TECNICA: 1
ATRIBUTO: 2


   EXITOS: 0
   FALLOS: 0
   
   loop X over {0.. ATRIBUTO}{
      DADO: [dado 1d6]
      EXITOS: EXITOS + [exito DADO]
      FALLOS: FALLOS + [fallo DADO TECNICA]
   }
   
   output EXITOS named "Exitos. Técnica [TECNICA] y atributo [ATRIBUTO]"
   output FALLOS named "Fallos. Técnica [TECNICA] y atributo [ATRIBUTO]"

TARGET:17
FOCUS:5

TARGET_TWO:15
FOCUS_TWO:3

TARGET_THREE:10
FOCUS_THREE:1

\loop TARGET over {10..17}\
\{\
\loop FOCUS over {0..5}\
\{\
ROLL:2d20
SUCCESS:[count {1..TARGET} in ROLL]
SUCCESS_TWO:[count {1..TARGET_TWO} in ROLL]
SUCCESS_THREE: [count {1..TARGET_THREE} in ROLL]

EXTRA_SUCCESS:[count {0..FOCUS} in ROLL]
EXTRA_SUCCESS_TWO: [count {0..FOCUS_TWO} in ROLL]
EXTRA_SUCCESS_THREE: [count {0..FOCUS_THREE} in ROLL]

TOTAL_SUCCESS:SUCCESS+EXTRA_SUCCESS
TOTAL_SUCCESS_TWO:SUCCESS_TWO+EXTRA_SUCCESS_TWO
TOTAL_SUCCESS_THREE:SUCCESS_THREE+EXTRA_SUCCESS_THREE

output TOTAL_SUCCESS named "Best Skill Roll with [FOCUS] Focus and [TARGET] Target"
output TOTAL_SUCCESS_TWO named "Good Skill Roll with [FOCUS_TWO] Focus and [TARGET_TWO] Target"
output TOTAL_SUCCESS_THREE named "Poor Skill Roll with [FOCUS_THREE] Focus and [TARGET_THREE] Target"
\}\
\}\

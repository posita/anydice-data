function: reroll ROLL:n {
    if ROLL < 5 {
        result: 1d6
    }
    result: ROLL
}

function: edge POOL:n {
    result: POOLd[reroll d6]
}

function: roll POOL:n {
    result: POOLd6
}

function: explode POOL:n {
    result: POOLd[explode d6]
}

\output [count {5..6} in [roll 8]] named "example"\

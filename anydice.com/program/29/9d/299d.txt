function: compare X:s against D:n
{
   R:0

   O: 1@X + 2@X
   P: 2@X + 3@X
   Q: 1@X + 3@X

   if O <= D
   {
      R: R + 1
   }
   if P <= D
   {
      R: R + 1
   }
   if Q <= D
   {
      R: R + 1
   }

   result: R
}

loop N over {2..12}
{
   output [compare 3d6 against N] named "target: [N]"
}

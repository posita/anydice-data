//Cheers,//
//- Ludic//

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}


output [reroll {1} on d4 to d4] named "Flames of Phlegethos bonus per d4"
output [reroll {1} on d6 to d6] named "Flames of Phlegethos bonus per d6"
output [reroll {1} on d8 to d8] named "Flames of Phlegethos bonus per d8"
output [reroll {1} on d10 to d10] named "Flames of Phlegethos bonus per d10"
output [reroll {1} on d12 to d12] named "Flames of Phlegethos bonus per d12"

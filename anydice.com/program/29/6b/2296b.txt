function: attack ROLL:n vs DEFENSE:n {
 if ROLL + ATTACKBONUS >= DEFENSE {
   if ROLL >= CRITICALHITTHRESHOLD {
     result: CRITICALHIT
   }
  result: HIT
 }

 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: HITDAMAGE * CRITICALHITMULTIPLIER
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: 0
}

HIT: 1
CRITICALHIT: 2

CRITICALHITTHRESHOLD: 20
HITDAMAGE: 1d3 + 4

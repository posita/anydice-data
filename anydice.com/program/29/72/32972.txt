function: doubles in ROLL:s {
 X:0 
 loop N over {1..#ROLL} {
  I: N + 1
  X: X+(N@ROLL = I@ROLL)
 }
 result: X
}

function: rollunder ROLL:s TN {
 M: [count {TN+1..20} in ROLL]
 result: ( (M = 0) + (M < 2)) + (((M = 0) - (M > 0)) * ([doubles in ROLL] > 0) )
}

loop N over {7..13} {
  output [rollunder 2d20 N] named "[N]"
}

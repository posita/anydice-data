
function: A:n higher B:n{
  if A > B {result: 1}
  else {result:0}
}

function: shot HIT:n a BS:n a WOUND:n a THRESHOLD:n a SAVE:n a ARMOUR:n a DAMAGE:n{
  if HIT >= BS {
    if WOUND >= THRESHOLD {
      if SAVE < ARMOUR { result:DAMAGE}
    }
  }
  result:0
}

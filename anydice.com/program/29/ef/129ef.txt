function: highest R:n again N:n {
 if N >= R { result: 1 + [highest R again d10] }
 result: N > 7
}

function: pbta A:n plus P:s pool {
    X:0
    if A > 7 {
     X:X+1
    }
    if A = 10 {
     X:X + [highest 10 again d10]
    }
    Y: [count {7,8} in P]
    result: X+Y
}

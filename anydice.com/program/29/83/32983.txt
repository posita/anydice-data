function: attack plus A {
  result: d20 + A
}

loop AC over {10..30} {
  BITE_DAMAGE: 0
  BITE_ATTACK_ROLL: [attack plus 4]
  if (BITE_ATTACK_ROLL > AC) {
    BITE_DAMAGE: d4 + 4
  }
  SWORD_DAMAGE: 0
  MOD: 8 - 5 + BITE_DAMAGE
  SWORD_ATTACK_ROLL: [attack plus MOD]
  if (SWORD_ATTACK_ROLL > AC) {
    SWORD_DAMAGE: 2d6 + 8 + 10
  }
  output BITE_DAMAGE + SWORD_DAMAGE
}

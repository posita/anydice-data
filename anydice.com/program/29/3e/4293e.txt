function: convert SUM:n {
 if SUM >= 1000 {
  TENSROLLED: SUM / 1000
  result: SUM - TENSROLLED * 990 + TENSROLLED d [explode d10]
 }
 result: SUM
}

function: ROLLSIZE:n k KEEPSIZE:n {
 result: [convert [highest KEEPSIZE of ROLLSIZE d{1..9, 1000}]]
}

output [highest 1 of 6d10] named "6k1 not exploded"
output [highest 1 of 6d[explode d10]] named "6k1 exploded before keeping"
output [6k1] named "6k1 exploded after keeping"

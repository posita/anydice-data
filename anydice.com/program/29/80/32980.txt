function: bite {
  BITE_DAMAGE: 0
  if d20 + 4 > [armorclass 10] {
    BITE_DAMAGE: d4 + 4
  }
  SWORD_DAMAGE: 0
  if d20 + 8 + BITE_DAMAGE - 5 > [armorclass 10] {
    SWORD_DAMAGE: 2d6 + 8 + 10
  }
  result: BITE_DAMAGE + SWORD_DAMAGE
}

function: armorclass A {
  result: d20 + A
}

output [bite]
output d4

WS: 40
STRENGTH: 8
TOUGHNESS: 5
WOUNDS: 8
loop N over {1..10} 
{
  HITCHANCE: (1d100 < WS)
  DAMAGE: HITCHANCE*([explode d6] + STRENGTH)
  DAMAGETAKEN: [highest of (DAMAGE - TOUGHNESS) and 0]
  WOUNDS: WOUNDS-DAMAGETAKEN
  output WOUNDS<=0 named "dead in [N] round(s)"
}
output DAMAGETAKEN named "Damage taken"

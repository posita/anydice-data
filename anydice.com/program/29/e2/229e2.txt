function: explode highest NUMBER:n of DICE:d {
  SUM: 0
  loop I over {1..NUMBER} { SUM: SUM + [explode and reroll I@DICE on [maximum of I@DICE]] }
  result: SUM
}

function: explode and reroll NUMBER:n on VALUE:n {
  if NUMBER = VALUE { result: [explode and reroll d(VALUE) on VALUE] + [explode and reroll d(VALUE) on VALUE] }
  result: NUMBER
}

set "maximum function depth" to 3

output [explode highest 3 of 3d6] named "Level 0"
output [explode highest 3 of 4d6] named "Level 1"

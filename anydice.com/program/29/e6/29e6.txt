function: roll POOL:s against FIRST and SECOND {
  if FIRST > SECOND {
    result: (FIRST<=1@POOL) + (SECOND<=2@POOL)
  }
  result: (SECOND<=1@POOL) + (FIRST<=2@POOL)
}
               \0 IS FLAT FAILURE, 1 IS SUCCESS, 2 IS FLAT SUCCESS\
output 0 named "0 is flat failure, 1 is success, 2 is flat success"

loop I over {1..10} {
  loop J over {1..10} {
    output [roll 2d10 against I and J] named "Rolling 2d10 against target numbers [I] and [J]"
  }
}

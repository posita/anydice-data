\
 Blades in the Dark
 0: Things go bad.
 1: Success, but with undesired consequences.
 2: Clean success.
 3: Critical success!
\

function: blades ROLL:s {
 O: 1@ROLL >= 4
 if 1@ROLL = 6 {
  O: 2 + (2@ROLL = 6)
 }

 Z: [count {1, 2} in ROLL]
 
 COMP: 0
 if 1@ROLL = 6 {
   COMP: Z
 }



 OUTCOME: O
 if COMP > 0 {
   OUTCOME: 1
 }

result: COMP

\result: OUTCOME\
}

output [blades 2@2d6] named "0d"
loop P over {1..5} {
 output [blades Pd6] named "[P]d"
}

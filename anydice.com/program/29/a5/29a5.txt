\ ~~~~~~~~~~~~~~~~~~~~ PROGRAM ONE ~~~~~~~~~~~~~~~~~~~~~~~~~ \

function: roll N:s and add the pair {
  loop R over {1..6} {
    Q: [count R in N]
      if Q > 1 { result: R }
  }
  result: 0
}

\ ~~~~~~~~~~~~~~~~~~~~ PROGRAM TWO ~~~~~~~~~~~~~~~~~~~~~~~~~ \

function: check ONE:n TWO:n against TN {
  if ONE+TWO <= TN {result: 1}
  result: 0
}

function: roll N:s against TN and the middle is the effect if success {
  Q: [check 1@N 3@N against TN]
  if Q { result: 2@N }
  result: 0
}

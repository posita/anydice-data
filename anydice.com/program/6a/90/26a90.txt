function: repeats in ROLL:s {
   HIGHEST: 0
   C: 0
   loop N over {1d6} {
      C: [count N in ROLL]
      if C > HIGHEST {HIGHEST: C}
   }
   result: HIGHEST
}

loop DICE over {1..5} {
  \ Cap out repeats at 3 for glog:
    1 is no problems, 2 is mishap, 3 is DOOM \
  output [lowest of 3 and [repeats in DICE d6]] named "[DICE]d6"
}

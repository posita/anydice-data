AC: 14
HITMOD: 3
DMGDICE: 1d10
DMGMOD: 2+3+1
DMGBACE:([count {[lowest of 20 and (AC-HITMOD)]..20,20} in 1d20]d DMGDICE)
DMGBACES: ([count {[lowest of 20 and (AC-HITMOD+5)]..20,20} in [highest 1 of 1d20]]d DMGDICE)
DMGADVS: ([count {[lowest of 20 and (AC-HITMOD+5)]..20,20} in [highest 1 of 2d20]]d DMGDICE)
DMGELF: ([count {[lowest of 20 and (AC-HITMOD)]..20,20} in [highest 1 of 3d20]]d DMGDICE)
function: addmod M if D:n {
 if (D > 0) {result: D + M}
 result: D
}
output 1d([addmod DMGMOD+10 if DMGBACES])+ 1d([addmod DMGMOD+10 if DMGADVS])
output 1d([addmod DMGMOD if DMGELF])+ 1d([addmod DMGMOD if DMGBACE])

D: {{1..9,10:4}:4}
H: {{2..9,10:4,11}:4}

function: blackjack S:s {
  T: S + 0
  if S = 1 & T < 12 {
    result: T + 10
  } else {
    result: T
  }
}

loop N over {2..5} {
output [blackjack NdD] named "Blackjack [N]"
}

function: statroll {
 result: [highest 3 of 4d6]
}

function: clamplow NUMBER:n {
 if NUMBER < 72 {
  result: 72
 }
 else {
  result: NUMBER
 }
}

function: clamphigh NUMBER:n {
 if NUMBER > 82 {
  result: 82
 }
 else {
  result: NUMBER
 }
}

function: clamp NUMBER:n {
 result: [clamphigh [clamplow NUMBER]]
}

function: highest of NUMA:n NUMB:n NUMC:n {
 result: 1 @ [sort {NUMA, NUMB, NUMC}]
}

function: fullstatroll {
 result: [statroll]+[statroll]+[statroll]+[statroll]+[statroll]+[statroll]
}

function: fullstatrollclamped {
 result: [clamp [fullstatroll]]
}

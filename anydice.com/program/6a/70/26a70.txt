function: attrib A:n B:n C:n skill S:n {
  SUCC: 0
  FAIL: 0
  loop X over {1..20} {
    loop Y over {1..20} {
      loop Z over {1..20} {
        if [highest of X and A] - A + [highest of Y and B] - B + [highest of Z and C] - C <= S {
          SUCC: SUCC + 1
        } else {
          FAIL: FAIL + 1
        }
      }
    }
  }
  result: {1:SUCC, 0:FAIL}
}

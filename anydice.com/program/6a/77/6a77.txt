\ Attacker Diepool \       A: 8
\ Defender Resist Stat \   D: 3
\ Defender Power Stat \    P: 0
\ Defender Shield Stat \   S: 0

\Attacker's Resisted Roll Result\ ARR: (A-D-S)d(d10 > 7)
\Attacker's Contested Roll Result\ AR: (A-S)d(d10 > 7)
\Defender's Contested Roll Result\ DR: Dd(d10 > 7)
\Attacker Contested Net Hits\       Y: ARR
\Attacker Contested Net Hits\       X: AR - DR

output Y>0 named "Attacker's Resisted Chance to Hit"
output ARR named "Attacker's Resisted Hits Rolled"
output X>0 named "Attacker's Contested Chance to Hit"
output X   named "Attacker's Contested Net hits"
output AR  named "Added Detail: Attacker's Contested Hits Rolled"
output DR  named "Added Detail: Defender's Contested Hits Rolled"

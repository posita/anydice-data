SMITELVL:3
GREATSWORD: 2d{{1..2}, {3..6}:4}
STRMOD: 3
BASESMITE: 1d8
COLOSSUS: 1d8
function: X:n rcrit {
if X = 20 {result: 2*(GREATSWORD + SMITELVLd8 + BASESMITE + COLOSSUS) + STRMOD}
result: (GREATSWORD + SMITELVLd8 + BASESMITE + COLOSSUS) + STRMOD
}

function: X:n fcrit {
if X > 18 {result: 2*(GREATSWORD + (SMITELVL-1)d8 + BASESMITE) + STRMOD}
result: (GREATSWORD + (SMITELVL-1)d8 + BASESMITE) + STRMOD
}

output [d20 rcrit]
output [d20 fcrit]

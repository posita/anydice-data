function: P:s against T:n {
  GOOD: P >= T
  BAD: P = 1
  if GOOD = 0 & BAD > 0 { result: -1 }    \ -1 = botch   \
  if GOOD > BAD { result: GOOD - BAD }    \ +n = success \
  result: 0                               \  0 = failure \
}

loop T over {2 .. 10} {
  X: { 1, 2:(T-2), T:(10-T+1) }  \ <-- custom d10 for optimization \
  loop N over {1..12} {
    output [NdX against T] named "net successes on [N]d10 vs. [T] (-1 = botch)"
  }
}

function: roll A:n with SIZE:n vs B:n {
   if A != SIZE | SIZE = 20 | B <= SIZE{
      result: A != 1 & A >= B
   }
   if SIZE < 12 {
      result: [roll (1d (SIZE + 2)) with (SIZE + 2) vs B]
   }
   result: 1d20 >= B
}

loop P over {4,6,8,10,12,20} {
   loop GM over {4,6,8,10,12,20} {
      output [roll (1d P) with P vs (1 d GM)] named "[P]v[GM]"
     \ C: 1d P
      output C != 1 & C >= (1 d GM) named "[P]v[GM]"\
   }
}

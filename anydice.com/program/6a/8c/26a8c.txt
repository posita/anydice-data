function: buckets ROLL:n {
  if ROLL >= 5 {
    result: 2
  }
  if ROLL >= 3 {
    result: 1
  }
  result: 0
}

loop D over {4,6,8,10,12} {
  output [buckets 1dD] named "d[D]"
}

function: highest N:n of A:s B:s {
    result: {1..N}@[sort {A, B}]
}

loop D over {4,6,8,10,12} {
  output [buckets [highest 1 of 1dD 1d6]] named "d[D] & d6"
}

\ Mini Six BKE (Legends) \
\ Roll a pool of D6, count dice 3 and over. \
\ VERY EASY: 1D \
\      EASY: 2D \
\  MODERATE: 3D - 4D \
\      HARD: 5D - 6D \
\ VERY HARD: 7D - 8D \
\    HEORIC: 9D+ \


function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d6] }
 result: N > 4 \ success criteria \
}

loop N over {2..5} { \ number of dice \
 output [lowest of 10 and NdNWOD] named "[N]d"
}

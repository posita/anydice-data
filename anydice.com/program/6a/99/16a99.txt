function: fact N:n S:n{
  S: N - S + 2
  FP: 0
  loop I over {1..N} {
    FP: FP + I
  }

  loop I over {S..N} {
    FP: FP - I
  }
  result: FP
}

function: odds N:n D:d W:n {
  B: D
  F: [fact N W]
  T: 1
  loop I over {2..W}{
    B: (B=D)*D
  }
  B: B>0
  loop I over {1..F}{
    T: T*(1-B)
  }

  if [maximum of D] < N {
    T: D
  } else {
    T: ((1-T) * 1d10)
  }
  result: T
}

function: cp D:d N:n {
  M: [maximum of D]
  S: N+1
  result: ((D>=N))
}

CHALLENGE_ONE: d10
CHALLENGE_TWO: d10

loop ACTION over {-1..6} {
   output ACTION > CHALLENGE_ONE & ACTION > CHALLENGE_TWO & (CHALLENGE_ONE = CHALLENGE_TWO) named "Strong Hit ([ACTION])"
    output !(ACTION > CHALLENGE_ONE) & !(ACTION > CHALLENGE_TWO) & (CHALLENGE_ONE = CHALLENGE_TWO) named "Miss ([ACTION])"
}

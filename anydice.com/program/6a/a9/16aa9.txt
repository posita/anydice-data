function: checkpair R:s {
 P: 0
 loop N over R {
  if N = P {
   result: 1
  }
  P: N
 }
 result: 0
}

function: checkthrees R:s {
 P: 0
 loop N over R {
  if N = P {
   if N = (N+1)@R {
    result: 1
   }
  }
  P: N
 }
 result: 0
}


function: checkfours R:s {
 P: 0
 loop N over R {
  if N = P {
   if N = (N+1)@R {
    if N = (N+2)@R {
     result: 1
    }
   }
  }
  P: N
 }
 result: 0
}

loop D over {4, 6} {
 loop N over {2..D} {
  output [checkpair NdD] named "[N]d[D] pairs"
 }
 loop N over {3..D} {
  output [checkthrees NdD] named "[N]d[D] triples"
 }
 loop N over {4..D} {
  output [checkfours NdD] named "[N]d[D] quadruples"
 }
}

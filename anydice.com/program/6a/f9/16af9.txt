function: rerollone A:n B:n {  
if A + B < 9 {  
  if A < B {    
    result: B+d6  
  }  
  else {  
    result: A+d6  
  }  
}  
else {
  result: A+B
}  
} 

function: rerolltwo A:n B:n {  
if A + B < 9 {  
  result: d6+d6    
}  
else {
  result: A+B
}  
}   
  
X: d6  
Y: d6  
output d6+d6 named "Normal Charge"
output [rerollone X Y]  named "Reroll one dice"
output [rerolltwo X Y] named "Reroll two dice"

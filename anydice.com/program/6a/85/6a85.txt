\

  NOTES:
  If both the character and the enemy have advantage, subtract the enemy advantage from the characters advanage
  Offensive checks always take precedence (i.e. if someone is trying to defend and it is a tie, the attacker wins)

  "output d{0:5,0:5} > 0" is the same as "output d10 > 5"

  Deal damage equivalent to the amount you roll + any bonuses that your weapons give
  Opponent subtracts any bonuses that their armor gives them?

  Trivial    : Must beat a 3 (i.e. 4 or higher) at least 1 time
  Easy       : Must beat a 4 (i.e. 5 or higher) at least 2 times
  Medium     : Must beat a 5 (i.e. 6 or higher) at least 3 times
  Difficult  : Must beat a 6 (i.e. 7 or higher) at least 4 times
  Impossible : Must beat a 7 (i.e. 8 or higher) at least 5 times

\

STATLVLS: 10

\ UNCONTESTED: \
\
loop STAT over {1..STATLVLS}
{
    output STATd{0:3,1:7} > 0 named "Trivial  with stat of [STAT]"
    output STATd{0:4,1:6} > 1 named "Easy with stat of [STAT]"
    output STATd{0:5,1:5} > 2 named "Medium with stat of [STAT]"
    output STATd{0:6,1:4} > 3 named "Difficult with stat of [STAT]"
    output STATd{0:7,1:3} > 4 named "Impossible with stat of [STAT]"
}

\ CONTESTED: \
\
loop STAT over {1..STATLVLS}
{
    ADVANTAGE: 0
    ENEMYADV: 0

    STATSEQ: {0:5-ADVANTAGE,1:5+ADVANTAGE}
    ENEMYSTATSEQ: {0:5-ENEMYADV,1:5+ENEMYADV}

    loop ENEMY over {1..STATLVLS}
    {
        output STATdSTATSEQ >= ENEMYdENEMYSTATSEQ named "Winning an offensive check against a LVL [ENEMY] (+[ENEMYADV] ATG) with stat LVL [STAT] (+[ADVANTAGE] ATG)"
\        output STATdSTATSEQ - ENEMYdENEMYSTATSEQ named "Damage dealt against enemy with Discipline of [ENEMY] (+[ENEMYADV] ATG) with Strength of [STAT] (+[ADVANTAGE] ATG)"\
    }    
}

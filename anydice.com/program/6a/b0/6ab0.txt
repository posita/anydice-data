function: razor ROLL:s target TAR:n adjust by N:n {
  if (({1..#ROLL}@ROLL + N) < TAR) {
    result: -1
  }
 TWOPAIRS: (1@ROLL = 2@ROLL) & (3@ROLL = 4@ROLL) & (2@ROLL > 3@ROLL)
 PAIRS: [highest of ((ROLL = 2@ROLL) -1) and ((ROLL = 3@ROLL) -1)]
 if TWOPAIRS = 1 {result: 2}
 if PAIRS <= 1 { result: PAIRS }
 result: PAIRS + 1 
}


loop N over {4..20} {
output [razor 4d6 target N adjust by -4] named "<[N]"
}

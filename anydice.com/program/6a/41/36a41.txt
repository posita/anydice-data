I: 2
RES: 1
DMG: 2

function: dsp ROLL:s {
 if 1@ROLL >= I*5 {
  result: 6
 }

 if 1@ROLL >= I*4 {
  result: 5
 }

 if 1@ROLL >= I*3 {
  result: 4
 }

 if 1@ROLL >= I*2 {
  result: 3
 }

 if 1@ROLL >= I {
  result: 2
 }

 if 1@ROLL >= 0 {
  result: 1
 }

 result: 0
}

loop R over {0..RES} {
 loop D over {1..DMG} {
  output [dsp Dd6 - Rd6] named "[D]D Damage - [R]D Resistance"
 }
}

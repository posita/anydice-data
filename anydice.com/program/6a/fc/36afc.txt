function: pchit NUMBER:n{
 if NUMBER < 0 { result: 0}
 result: NUMBER
}


function: npchit NUMBER:n{
 if NUMBER > 0 { result: 0}
 result: -NUMBER
}

function: onlypositive NUMBER:n{
 if NUMBER < 0 { result: 0}
 result: NUMBER
}

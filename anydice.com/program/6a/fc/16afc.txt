function: edgesnag EDGE:s SNAG:s {
  loop X over {1..#EDGE} {
    if X@EDGE > X@SNAG { result: X@EDGE }
 if X@EDGE < X@SNAG { result: 7-X@SNAG }
  }
  if #EDGE < #SNAG { result: 7-X@SNAG }
  result: [lowest of #EDGE@EDGE and #SNAG@SNAG]
}

loop X over {2..5} {
  output [edgesnag 1d6 Xd6] named "1d6 vs [X]d6"
}

loop X over {2..5} {
  output [edgesnag Xd6 1d6] named "[X]d6 vs 1d6"
}

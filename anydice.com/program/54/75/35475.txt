function: count successes from DICE:s {
    if DICE >= 10 { result: 3 }
    if DICE >= 8 { result: 2 }
    if DICE >= 4 { result: 1 }
    result: 0
}

loop X over {4,6,8,10,12} {
    loop Y over {4,6,8,10,12} {
        output [count successes from dX] + [count successes from dY] named "D[X] + D[Y]"
    }
}


function: shootwound SHOT:s on HIT:n to WOUNDROLL:d {
    if [SHOT contains HIT] {
        result: WOUNDROLL
    }
        
    result: -1
}


output [shootwound {6..10} on d10 to (d10+6)] >= (1d10+6) named "Chance for an AIM 4 model to take a DEF 6 model out-of-action by shooting him with a pistol"


function: meleewound STAB:s on HIT:n to WOUNDROLL:d {
    if [STAB contains HIT] {
        result: WOUNDROLL
    }
        
    result: -1
}


output [meleewound {0..10} on ((d10+5)-(d10+4)) to (d10+6)] >= (1d10+6) named "Chance for an melee 4 model to take a DEF 6 model out-of-action by attacking with light weapon"

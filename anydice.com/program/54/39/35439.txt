function: bad ROLLA:n ROLLB:n {
  if (ROLLA < 5) & (ROLLB < 5) & (ROLLA = ROLLB) {result:1}
  result: 0
}

function: roll N:n {
 BAD: 0
 ROLL: Nd6
 loop P over {1..N-1} {
  SCORE: [bad P@ROLL (P+1)@ROLL]
  BAD: BAD + SCORE
 }
 result: (BAD <= 0)
}

loop P over {1..11} {
 F: P - 1
 output [roll P] named "[F]"
}

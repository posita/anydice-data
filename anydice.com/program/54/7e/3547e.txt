\
 Blades in the Dark
 0: Things go bad.
 1: Success, but with undesired consequences.
 2: Clean success.
 3: Critical success!
\

function: blades ROLL:s {
 if 1@ROLL > 16 {
  result: 2 + (2@ROLL > 16)
 }
 result: 1@ROLL >= 12
}

output [blades 2@2d6] named "0d"
loop P over {1..6} {
 output [blades Pd20] named "[P]d"
}

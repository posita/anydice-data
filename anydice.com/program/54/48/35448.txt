function: sucnum{
result: {5..10}
}

function: difnum{
result: {1..4}
}

function: combat A vs B{
result: [count [sucnum] in A] - [count [difnum] in B]
}
output [combat 1d10 vs 2d8] named "Untrained"
output [combat 2d10 vs 2d8] named "Novice"
output [combat 3d10 vs 2d8] named "Proficient"
output [combat 4d10 vs 2d8] named "Expert"
output [combat 5d10 vs 2d8] named "Renowned"
output [combat 6d10 vs 2d8] named "Master"

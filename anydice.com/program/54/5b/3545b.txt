

function: roll DICEONE:n or HALFLINGDICE:n plus DICETWO:n against DC:n {
  if DICEONE = 1 { DICEONE: HALFLINGDICE } \Reroll if first die is a 1\
  if DICEONE = DICETWO { result: 2 }
  if DICEONE + DICETWO >= DC { result: 1 }
  result: 0
}

loop DC over {1..30} {
  output [roll 1d12 or 1d12 plus 1d12 against DC] named "DC [DC]"
}


function: pastriples in DICE:s sauf si V ou plus {
  
  D: 0 
  T: 0
  loop X over {2..#DICE} {
    if ((X-1)@DICE = X@DICE & X@DICE < V) { 
      D: D + 1
      if (D >= 2) {T: 1}
    }
  }
  if (T = 1) { T: 1000} else {T: [count V in DICE] }
  result: T
}

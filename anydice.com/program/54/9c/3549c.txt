
function: pastriples in DICE:s {
  
  D: 0
  loop X over {2..#DICE} {
    if ((X-1)@DICE = X@DICE & X@DICE < 6) { D: D + 1}
  }
  if (D = 2) { D: 1} else {D: 1000+ [count 6 in DICE] }
  result: D
}

loop X over {3..10} {
  output [pastriples in Xd6 ] named "[X] d6  sans faire de triples (à part peut-être des 6) "
}

function: sucnum{
result: {5..10}
}

function: difnum{
result: {1..4}
}

function: combat A vs B{
result: [count [sucnum] in A] - [count [difnum] in B]
}

function: difficulty{
result: 2d8
}
output [combat 1d10 vs [difficulty]] named "Untrained"
output [combat 2d10 vs [difficulty]] named "Novice"
output [combat 3d10 vs [difficulty]] named "Proficient"
output [combat 4d10 vs [difficulty]] named "Expert"
output [combat 5d10 vs [difficulty]] named "Renowned"
output [combat 6d10 vs [difficulty]] named "Master"

HIT:1
MISS:0
function: attack ROLL:n vs DEFENSE:n {
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}


ATTACK: 7
HITDAMAGE: d12 + 3
MISSDAMAGE: 0
loop AC over {12..16} {
output [damage for [attack d20 vs AC]] named "cutter rifle burst vs [AC] AC"
}

ATTACK: 7
HITDAMAGE: 2d8 + 3
loop AC over {12..16} {
output [damage for [attack d20 vs AC]] named "Shear Rifle burst vs [AC] AC"
}

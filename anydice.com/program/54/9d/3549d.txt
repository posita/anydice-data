
function: pastriples in DICE:s sauf si V ou plus {
  
  D: 0
  loop X over {2..#DICE} {
    if ((X-1)@DICE = X@DICE & X@DICE < V) { D: D + 1}
  }
  if (D = 2) { D: 1} else {D: 1000+ [count V in DICE] }
  result: D
}

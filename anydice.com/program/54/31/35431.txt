function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: DICE:d reroll REROLL:s {
    result: [reroll REROLL on DICE to DICE]
}

function: twist DE:n DIE:d {
  if DE <= 1 {
    result: 1@{DIE}
  } 
  if DE = 2 {
    result: [lowest of DIE and DIE]
  }
  if DE = 3 {
    result: [highest of DIE and DIE]
  }
  if DE >= 4 {
    result: 1@[reverse {DIE}]
  }
}

function: ttf DIE:d {
  result: (DIE-1)/5+1
}


function: replace ROLL:n with REPLACEMENT:n on TO_REPLACE:s {
  if [TO_REPLACE contains ROLL] {
        result: REPLACEMENT
  }
  result: ROLL
}

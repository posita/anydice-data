MISS: 0
HIT: 1
CRITICALHIT: 2
MISSDAMAGE: 0

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{12..17}

PRO: 2
DEX: 3
ATTACK: 2 + PRO + DEX
HITDAMAGE: DEX + d6 + d8 + d10 
CRITICALDAMAGE: DEX + 2d6 + 2d8 + 2d10
output [attack] named "Heavy Crossbow"

HITDAMAGE: DEX + d6 + d8 + d8 
CRITICALDAMAGE: DEX + 2d6 + 2d8 + 2d8
output [attack] named "Longbow"

ATTACK: PRO + DEX
HITDAMAGE: DEX + d6 + d6
CRITICALDAMAGE: DEX + 2d6 + 2d6
FIRST: [attack]
HITDAMAGE: d6 + d6
CRITICALDAMAGE: 2d6 + 2d6
SECOND: [attack]
output FIRST + SECOND named "Dual Shortsword"

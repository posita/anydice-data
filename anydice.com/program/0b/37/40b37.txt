function: bin N:n {
  if N < 10 {result: -1}
  if N > 10 {result: 1}
  result: 0
}

function: crit N:n on F:n and S:n {
  if N <= F {result: -1000}
  if N >= S {result: 1000}
  result: N 
}

GURPS: [crit 3d6 on 4 and 17]
PYRAM: [crit 2d10 on 2 and 20]

loop N over {1..10}{
  loop D over {1..10}{
    output [bin 1dGURPS + N - D] named "[N] vs [D]"
  }
}

function: A:n modulo B:n
{
 result: A - A/B * B
}

function: ceiling B:n modulo A:n
{
 result: (B + A - (B - B/A * A))/A
}

function: if A:s then B:s else C:s
{
 R: {}
 loop N over {1..#A}
 {
  if (N@A)
  {
   R: {R, (1 + [(N - 1) modulo #B])@B}
  } else {
   R: {R, (1 + [(N - 1) modulo #C])@C}
  }
 }
 result: R
}

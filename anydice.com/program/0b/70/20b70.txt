function: pbta ROLL:n plus N:n {
  if ROLL + N <= 6 { result: 0 }
  if ROLL + N > 6 & ROLL + N < 10 { result: 1 }
  result: 2
}

function: ROLL:s target T:n D:n {
  A: 1@ROLL
  B: 2@ROLL
  X: T - D
  N: 0
  if A = B { result: 2 }
  if A < X { N: N + 1 }
  if B < X { N: N + 1 }
  result: N
}

S: 20
N: 7
D: 0
output [2dS target N D] named "Roll"
\output [pbta 2d6 plus 0] named "PbtA"\

output [2dS target N D] > 0 named "Success"
\output [pbta 2d6 plus 0] > 0 named "PbtA Success"\

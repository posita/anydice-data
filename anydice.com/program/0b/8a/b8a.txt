function:phenom A:n {
  B: 0
  loop N over {1..A}{
    B: B + (1d10 = 9)
  }
  result: [perils B]
}

function:phenomveil A:n {
  B: 0
  loop N over {1..A}{
    B: B + (1d10 >= 8)
  }
  result: [perils B]
}

function:perils C:n {
  D: 0
  loop N over {1..C}{
    D: D + [lowest 1 of 2d(1d100 >= 75)]
  }
  result: D
}

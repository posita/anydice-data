\ Use normal view for exactly X of a kind, at-least for X-or-more of a kind. \

function: match ROLL:s {
 R: 0
 loop N over {1..6} {
  R: [highest of R and [count N in ROLL]]
 }
 result: R
}

output [match 1d6] named "getting X of a kind"
output [match 2d6] named "getting X of a kind"
output [match 3d6] named "getting X of a kind"
output [match 4d6] named "getting X of a kind"
output [match 5d6] named "getting X of a kind"
output [match 6d6] named "getting X of a kind"

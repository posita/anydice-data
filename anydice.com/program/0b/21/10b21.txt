TAW:7
ATT:13
ATTI:13
ATTII:12
DIF:7
function: hand REN:n{
 if REN>-1 {
PRX:ATT-1d20+DIF
PRXX:ATTI-1d20+DIF
PRXXX:ATTII-1d20+DIF
result: [lowest of PRX and [lowest of PRXX and PRXXX ]]+(REN/2)+(DIF)
} 
result:-1
}
function: mos TAW ATT:s DIFF:n{
MOS: [highest of TAW-DIF and 0]
loop N over {ATT}{
MOS:MOS-[highest of 1d20 +[highest of DIF-TAW and 0] - N and 0]
}
result:[highest of MOS and -1]
}
output [mos TAW {ATT,ATTI,ATTII} DIF]
loop N over {DIF-10..DIF+10}{
DIF:N
output [mos TAW {ATT,ATTI,ATTII} DIF]>-1 named "[N]"
}
output [hand [mos TAW {ATT,ATTI,ATTII} DIF]]
\output (MOS/3)+(MOS/3)+(MOS/3) named "mod3"
output (MOS/3)++(MOS/3)+(MOS/3) named "mod3"
output [hand MOS]
output [highest of [hand MOS] and 0] named "diff"
output [handi MOS]

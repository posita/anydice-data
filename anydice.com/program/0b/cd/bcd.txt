\ custom function for determining success and rerolls \

function: explode N:n {
 if N = 6 { result: 1 + [explode d6] }
 if N >= 4 { result: 1 }
 result: 0
}

\ maximum function depth limits recursion, thus rerolls. \

set "maximum function depth" to 3

\ define custom die for reuse \

DIE: [explode d6]

\ output various pool sizes \

loop X over {1..4} {
 output X d DIE named "[X]d"
}

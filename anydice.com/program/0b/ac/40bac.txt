BASEDIE: d6

function: target TARGET:n reroll REROLL:n VALUE:n {
  if VALUE = REROLL {
    result: 2 + [target TARGET reroll REROLL BASEDIE]
  }
  if VALUE  = 5 {
    result: 1
  }
  if VALUE  = 4 {
    result: 1
  }
  if VALUE  = 1 {
    result: -1
  }
  result: 0
}

DIE: [target 3 reroll 6 BASEDIE]

loop N over {1..10}{
  output NdDIE named "[N]"
}

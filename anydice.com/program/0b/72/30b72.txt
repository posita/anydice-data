function:  N:n of S:s from P:n {
  R:  {}
  loop I over {P..[lowest of P+N-1 and #S]} {
    R:  {R, I@S}
  }
  result:  R
}

function:  explodup FS:s {
  SORTED:  [sort FS]
  R:  1@SORTED * 1000 + 1
  loop I over {2..#SORTED} {
    R:  {R, I@SORTED * 1000}
  }
  result:  [explode d R] / 1000
}
set "explode depth" to 5
output [explodup {1,2,3,4,5,5}]

function: fitness FIT:n
{
 result: FIT - (2d6 >= FIT)
}

function: fitvalue VAL:n FIT:n
{
 result: VAL + (2d6 < FIT)
}

function: damage OFF:n DEF:n
{
 result: (OFF > DEF) * OFF
}

function: vit OFF:n DEF:n VIT:n
{
 result: [highest of (VIT - [damage OFFd6 DEFd6]) and 0]
}

output [fitvalue 2 [fitness 12]]
\output [vit [fitvalue 2 12] 1 14]
output [vit 2 2 48]
output [vit 2 1 14]
output [vit 2 [fitvalue 2 12] 48\

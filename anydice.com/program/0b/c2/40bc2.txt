function: effect ROLL:n MOD:n {
  TOTAL: ROLL + MOD
  DIFF: TOTAL - 10
  if DIFF >= 0 {
    result: DIFF / 2
  } else {
    result: (DIFF - 1) / 2
  }
}

loop PLAYER_RANK over {0..5} {
  loop OPP_RANK over {0..5} {
    PLAYER_EFFECT: [effect 2d8 PLAYER_RANK]
    OPP_EFFECT: [effect 2d8 OPP_RANK]
    output PLAYER_EFFECT - OPP_EFFECT named "Player Rank [PLAYER_RANK], Opp Rank [OPP_RANK]"
  }
}

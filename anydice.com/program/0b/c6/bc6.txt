\ Slår X antal tärningar och räknar de tärningar som visar 4+ som lyckade resultat. \
\ Om någon tärning slår max så slår man ytterligare en tärning. \

SUCCESSES: 0

function: exploding D:d {                  \ Tar in varje enskild tärning. \
  MAX: [maximum of DIE]                    \ Ser vad som är max på tärningen. \
  result: [exploding D helper]
}

function: exploding E:n helper {           \ Gör om en tärning till en variabel. \
  if E > 3 { SUCCESSES: SUCCESSES + 1 }    \ Lägger till 1 till potten om tärningen visar 4+. \
  if E = MAX { result: [exploding 1dDIE] } \ Slår en ny tärning om tärningen visar max. \
  result: SUCCESSES                        \ Returnerar antalet lyckade tärningar. \
}

MINDICE: 1          \ Minsta antalet tärningar i beräkningen. \
MAXDICE: 10         \ Största antalet tärningar i beräkningen. \
DIE: d6             \ Den tärningstyp som du vill använda \

  \ Högre värden ger mer exakta resultat i uträkningen men tar längre tid. Testa själv. \
set "maximum function depth" to 5    \ Default-värdet är 10 men jag minskade det till 5. \

loop N over {MINDICE..MAXDICE} {
  output Nd[exploding DIE] named "Slå 4,5,6 med exploderande [N]d6"
}

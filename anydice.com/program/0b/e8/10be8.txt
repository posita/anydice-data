function: reroll lowest of ROLL:s as REROLL:d if less than MIN:n {
  LOWEST: (#ROLL)@ROLL       \ the lowest die is sorted last \
  if LOWEST >= MIN {
    result: ROLL + 0
  } else {
    REST: {1..#ROLL-1}@ROLL  \ all but the lowest die \
    result: REST + REROLL
  }
}

output 2d12 named "2d12"
loop MIN over {2..12} {
  output [reroll lowest of 2d12 as d12 if less than MIN] named "2d12 replace lowest if < [MIN]"
}
output [highest 1 of 2d12] + 1d12 named "2d12 always replace lowest"

function: attacker A rolls AR:n defender D rolls DR:n {
  if A * AR > D * DR {
    result: A * AR / D
  } else if A * AR < D * DR {
    result: -D * DR / A
  } else {
    result: 0
  }
}

function: stress attacker A defender D {
  result: [attacker A rolls 1d6 defender D rolls 1d6]
}

loop AS over {1..6} {
  loop DS over {1..6} {
    output [stress attacker AS defender DS] named "[AS] attacking [DS]"
  }
}

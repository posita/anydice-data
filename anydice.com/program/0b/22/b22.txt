DD: d{-10..-1,1..10}

function: roll A:n and DA:n and DB:n
{
 if A > 1 {COIN: 1}
 if A<-1{COIN: -1}
 if A = 1 { result: DA}
 if A = -1{ result: DB}
 if A != 0 
 {
  result: ([absolute A]+ [absolute DA-DB])*COIN
 }
}

output [roll d{10,-10,1,1,-1,-1} and d{-9..10} and d{-10..9}]

output [roll d{10,-10,1,1,-1,-1,-1,1} and d{-9..10} and d{-10..9}]

TN: 17
PROF: 3
STR: 4
MAGICWEAP: 0
HITBONUS: STR + PROF + MAGICWEAP
DAMAGEBONUS: STR + MAGICWEAP
BLESSED: 1
ADV: 1 + 0

function: A:n hit odds {
  if A >= TN { result: 1 }
  result: 0
}

function: bless {
  result: BLESSED * 1d4
}

function: stddmg {
  result: [highest 2 of 2d{{3..6}:4, {1..2}}]+DAMAGEBONUS
}

function: superdmg {
  result: [stddmg]+10
}

function: stdattack {
  result: [[highest 1 of ADVd20] + HITBONUS + [bless] hit odds]
}

function: superattack {
  result: [[highest 1 of ADVd20] + HITBONUS - 5 + [bless] hit odds]
}

output [stddmg] * [stdattack] named "Standard Attack EV"
output [superdmg] * [superattack] named "Super Attack EV"

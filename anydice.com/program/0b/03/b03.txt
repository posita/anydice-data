
\ Use normal view for exactly X of a kind, at-least for X-or-more of a kind. \

function: match ROLL:s to HIGH:n{
 R: 0
 loop N over {1..HIGH} {
  R: [highest of R and [count N in ROLL]]
 }
 result: R
}



loop D over {4,6,8,10,12} {
 loop N over {4,6,8,10,12} {
   output [match DdN to [lowest of D and N] ] named "getting X of a kind on [D]d[N]"
  }
}

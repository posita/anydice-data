function: explode N:n {
 if N = 1 { result: 1 + [explode d{DIE}] }
 result: N
}

QUEUE: 5 \ Queue position \
DIE: {0, 0, 1}
A: 0
B: 0
BIRTHS: QUEUE + 1
MAX: (BIRTHS + 1) ^ 2
set "maximum function depth" to BIRTHS

loop M over {1..BIRTHS} {
 A: A + [explode d{DIE}] + 1
 B: [count {BIRTHS..MAX} in A]
 output B named "Mother #[M]"
}

function: lament ROLL:s {
 if 1@ROLL = 1 {
  \ catastrophe \
  result: -1
 }
 ONES: [count 1 in ROLL]
 if ONES >= #ROLL / 2 {
  \ failure \
  result: -0
 }
 if [count 6 in ROLL] >= ONES + 2 {
  \ critical success \
  result: 2
 }
 \ success \
 result: 1
}

loop N over {2..10} {
 output [lament Nd6] named "[N]d"
}

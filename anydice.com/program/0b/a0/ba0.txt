function: sign A:n {
  if A > 0 { result: 1 }
  if A < 0 { result: -1}
  result: 0
}

\ player 1 rolls Ad10 and player 2 rolls Bd10. Both pick the single highest die \
\ 1 indicates that player 1 won, -1 that player 2 won, and 0 indicates a tie \

loop A over {1..10}{
  loop B over {1..A}{
    output [sign ([highest 1 of Ad10] - [highest 1 of Bd10])] named "[A]d10 vs. [B]d10"
  }
}

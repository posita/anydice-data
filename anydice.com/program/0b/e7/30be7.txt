MOD: 0

function: blades ROLL:s {
 if 1@ROLL = 6 {
  result: 2
 }
 result: 1@ROLL >= 4
}

function: pbta AROLL:n AMOD:n{
  if (AROLL+AMOD) >= 10 {result: 2}
  result: (AROLL+AMOD) >= 7
}

function: action AROLL:n AMOD:n {
  result: 2d([lowest of (AROLL+AMOD) and 10] > d10)
}

output 4d{-1..1}
output [blades 2d6] named "Blades"
output [pbta 2d6 MOD] named "PBTA 2d6"
output [action 1d6 MOD] named "Action 1d6"
output 1d20 > 14

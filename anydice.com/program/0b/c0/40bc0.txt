function: effect ROLL:n MOD:n {
  TOTAL: ROLL + MOD
  DIFF: TOTAL - 10
  if DIFF >= 0 {
    result: DIFF / 2
  } else {
    result: (DIFF - 1) / 2
  }
}

loop MOD over {-5..5} {
  loop EFF over {-5..5} {
    LOW: 10 + EFF * 2
    HIGH: LOW + 1
    output [effect 2d8 MOD] = EFF named "2d8 + [MOD], target between [LOW] and [HIGH], Effect [EFF]"
  }
output 1 named "---------------------------------------"
}

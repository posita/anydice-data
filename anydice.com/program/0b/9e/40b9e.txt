\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + MAGICBONUS + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
PROFICIENCY: 4
WEAPON: 1d8
MAGICBONUS: 1
DMGBONUS: 0
EXTRADICE: 2d4
ACCBONUS: 0
CRITROLL: 20
SNEAK: 0
DAMAGE: WEAPON + ATTACKMOD + MAGICBONUS + DMGBONUS + EXTRADICE + SNEAKd6
CRITDAMAGE: 2dWEAPON + ATTACKMOD + MAGICBONUS + DMGBONUS + 2dEXTRADICE + 2d(SNEAKd6)
output [attackroll d20 vs ENEMYAC] named "+1 Weapon, 2d4 extra"

MAGICBONUS: 2
DAMAGE: WEAPON + ATTACKMOD + MAGICBONUS + DMGBONUS + EXTRADICE + SNEAKd6
CRITDAMAGE: 2dWEAPON + ATTACKMOD + MAGICBONUS + DMGBONUS + 2dEXTRADICE + 2d(SNEAKd6)
output [attackroll d20 vs ENEMYAC] named "+2 Weapon, 2d4 extra"

MAGICBONUS: 2
EXTRADICE: 1d8
DAMAGE: WEAPON + ATTACKMOD + MAGICBONUS + DMGBONUS + EXTRADICE + SNEAKd6
CRITDAMAGE: 2dWEAPON + ATTACKMOD + MAGICBONUS + DMGBONUS + 2dEXTRADICE + 2d(SNEAKd6)
output [attackroll d20 vs ENEMYAC] named "+1 Weapon, 1d8 extra"

function: roll vs DANGER:s {
 COUNT: 0
 loop P over {1..3} {
  COUNT: COUNT + (1d6 >= DANGER)
 }
 result: COUNT
}

function: cast at lvl LEVEL:n {
  CAST: 1d6+LEVEL
  RES: 0
  RES: RES + (CAST>1d10)
  RES: RES + (CAST>1d10)
  result: RES
}

output [cast at lvl 1]

output [roll vs 2] named "lvl 2"
output [roll vs 3] named "lvl 3"

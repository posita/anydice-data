function: roll baddie BADDICE:n risk R:n{
 if BADDICE = 1 {
  result: -100
 }
 result: -(R*2)-BADDICE
}
function: wilddice W:n risk R:n{
 if W = 1 {
  result: [roll baddie 1d6 risk R]
 }
 if W = 6 {
  result: [explode 1d6] + 6
 }
 result: W
}
function: dicepool X{
  S: (X-1)d6
  result: S + [wilddice 1d6 risk 1@S]
}
set "maximum function depth" to 4
output [dicepool 3]

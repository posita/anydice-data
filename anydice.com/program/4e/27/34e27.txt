set "explode depth" to 10

RANGE: 16

function: clamp N:n between MIN:n and MAX:n {
  if N > MAX { result: MAX }
  if N < MIN { result: MIN }
  result: N
}

\ Opposed roll \
output [clamp [explode d8]+d8-[explode d8]-d8-d{0..1} between -RANGE and RANGE] named "d8+d8! vs"

\ Dynamic action \
output [clamp [explode d8]+d8-[explode d8]-5 between -RANGE and RANGE] named "d8+d8! vs 5+d8!"

\ Static action \
output [clamp [explode d8]+d8-9 between -RANGE and RANGE] named "d8+d8! vs 9"

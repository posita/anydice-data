function: drill N:n {
  if N = 1 { result: 2d[drill d6] }
  result: N
}

function: drillheat N:n {
  if N = 1 { result: 1 + 2d[drillheat d6] }
  result: 0
}

  if N = 1 { result: 1 + [knifeheat d3] }
  result: 0
}

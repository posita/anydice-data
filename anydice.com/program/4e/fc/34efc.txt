ATT: 50
DEF: 100

function: attacker chance A:n beat C:n {
if A < ATT & ( A > C | C > DEF ) {result: 1}
else {result: 0}
}


function: defender chance C:n beat A:n {
if C < DEF & ( C > A | A > ATT ) {result: 1}
else {result: 0}
}

\ 0 for defender wins, 1 for attacker wins \
output [attacker chance d100 beat d100] named "Attacker Victory"

\ 0 for attacker wins, 1 for defender wins \
output [defender chance d100 beat d100] named "Defender Victory"

function: bonus ROLL:s TN:n INDEX{
  BONUS: 1
  if(INDEX@ROLL>TN){
    BONUS: BONUS+1
  }
  if(INDEX@ROLL=6){
    BONUS: BONUS+1
  }
  result: BONUS
}

T: 4
output [bonus 2d6 T 2] named "Bonus/Penalty on 0d6 and TN [T]"
loop D over {1..6}{
  output [bonus Dd6 T 1] named "Bonus/Penalty on [D]d6 and TN [T]"
}

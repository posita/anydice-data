EIGENSCHAFT: {14,14,14}
TAW:9
MOD: 0

function: max V:s {
 MAX:1@V
 loop N over V{
  if N > MAX {
   MAX: N
  }
 }
 result: MAX
}

function: over zero V:n {
   if V > 0 {
   result: V
  }
 else{
 result: 0
}
}

function: A:n equals N:n than V:n {
   if A =N  {
   result: V
  }
 else{
 result: A
}
}

MISS:d{}
HARDMOD:0
TAW:TAW-MOD
if TAW <0 {
 HARDMOD:-TAW
 output TAW
 TAW:0
}

 loop N over EIGENSCHAFT{
  ROLL:d20 - (N - HARDMOD)
   MISS: MISS + [over zero ROLL]
}


output [[over zero (TAW - MISS + 1)]-1 equals 0 than 1 ] named "TaP*"
\output [over zero (TAW - MISS+1)]-1 >-1 named "Erfolg"\

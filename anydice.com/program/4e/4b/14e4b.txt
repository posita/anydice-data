\******** DEFINITIONS ********\
ARCHERY_STYLE: 2
SHARPSHOOTER_ATTACK_PENALTY: 5
SHARPSHOOTER_DMG_BONUS: 10

\******** ENEMY ********\
AC: d{10..20}

\******** ATTACK FUNCTIONS ********\
function: sharpshooter at level LEVEL:n with DEX:n dex for DMG:d crit CRIT_DMG:d roll ROLL:n against AC:n{
    ATTACK_MODIFIER: (DEX - 10)/2
    PROFICIENCY: ((LEVEL - 1)/4) + 2
    DMG_WITH_BONUS: DMG + ATTACK_MODIFIER + SHARPSHOOTER_DMG_BONUS
	
    if ROLL = 1 {result:0}
    if ROLL = 20 { result: DMG_WITH_BONUS + CRIT_DMG }
    if ROLL + ATTACK_MODIFIER + ARCHERY_STYLE >= AC { result: DMG_WITH_BONUS }

    result: 0
}

function: beast attack plus ATTACK_MODIFIER:n to hit for DMG:d crit CRIT_DMG:d roll ROLL:n against AC:n{
    if ROLL = 1 {result:0}
    if ROLL = 20 { result: DMG + CRIT_DMG }
    if ROLL + ATTACK_MODIFIER >= AC { result: DMG}

    result: 0
}

output 6d[sharpshooter at level 8 with 20 dex for 1d8 crit 1d8 roll 1@3d20 against AC] named "11th Level Samurai Action Surge"
output 16d[beast attack plus 5 to hit for 3d6+3 crit 3d6 roll d20 against AC] named "11th Level Druid (ie Upcast to 5th Level)"
output 8d[sharpshooter at level 8 with 20 dex for 1d8 crit 1d8 roll 1@3d20 against AC] named "20th Level Samurai Action Surge"
output 24d[beast attack plus 5 to hit for 3d6+3 crit 3d6 roll d20 against AC] named "13th Level Druid (ie Upcast to 7th Level)"
output 32d[beast attack plus 5 to hit for 3d6+3 crit 3d6 roll d20 against AC] named "15th Level Druid (ie Upcast to 9th Level)"
output 
    [beast attack plus 5 to hit for 3d6+3 crit 3d6 roll d20 against AC] +
    31d[beast attack plus 5 to hit for 3d6+3 crit 3d6 roll 1@2d20 against AC]
    named "15th Level Druid (ie Upcast to 9th Level), prone after first"

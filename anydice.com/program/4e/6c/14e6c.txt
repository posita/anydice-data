WHITEA:  d{0,0,0,0,0,99,1,1}
BLACKA:  d{0,0,0,99,1,1,1,1}
REDA:    d{0,99,1,1,1,1,1,1}

\ REMOVE: This should always be set to 1. \
\ SURGE: If the model or unit has surge, set this to 1, otherwise set it to 0. \
\ PRECISE: number of rerolls (amount of precision). \
\ AMOUNT: Amount of dice rolled (number of models or dice on weapon). \
\ DICE: Type of dice being rolled. \
\ ROLL: The total amount of dice rolled. \

function: attack REMOVE:n SURGE:n PRECISE:n AMOUNT:n DICE:d ROLL:n{
REMOVAL:99
if SURGE=1 & REMOVE=1 {REMOVAL:98}
loop X over {1..AMOUNT}{
if ROLL >= 99 {ROLL: ROLL - REMOVAL}
}

if ROLL < AMOUNT & PRECISE > 0 {result: [attack 0 SURGE PRECISE-1 AMOUNT DICE ROLL+1dDICE]}
result: ROLL
}

function: remove A:n ROLL:n{
loop X over {1..A}{
if ROLL >= 7 {ROLL: ROLL - 6}
}
result: ROLL
}

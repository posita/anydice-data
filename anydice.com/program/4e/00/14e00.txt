function: haunt check N:n {
  result: 6d{0..2} <= N
}

function: haunt step N:n CHECK:n {
  if CHECK {
    result: N
  } else {
    result: [haunt step (N+1) [haunt check (N+1)]]
  }
}

function: haunt N:n {
  result: [haunt step N [haunt check N]]
}

set "maximum function depth" to 15

output [haunt 0]

\
 Blades in the Dark
 0: Things go bad.
 1: Success, but with undesired consequences.
 2: Clean success.
 3: Critical success!
\

SIDED: 8
MAXDICE: 8

function: blades ROLL:s {
 if 1@ROLL = SIDED {
  result: 2 + (2@ROLL = SIDED)
 }
 result: 1@ROLL >= SIDED/2 + 1
}

output [blades 2@2dSIDED] named "0d"
loop P over {1..MAXDICE} {
 output [blades PdSIDED] named "[P]d"
}

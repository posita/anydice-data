\ 0 is a miss, 1 is a weak hit, 2 is a strong hit \

function: progress CROLL:s PROG:n {
  result: PROG > CROLL
}

function: action AROLL:n AMOD:n CROLL:s {
  result: (AROLL + AMOD) > CROLL
}

output [progress 2d10 5] named "Progress 5"

loop X over {-6..10} {
  output [action 1d6 X 2d10] named "[X]"
}

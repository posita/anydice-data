
function: ROLL:s count sixes and above{
if ROLL>=6
{
 result: (ROLL)
}
}

ABILITIES: 6 d [{1,2,3}@4d6 count sixes and above]
K:6
loop P over {1..K} {

if P-K+6 @ ABILITIES >=16 | P-K+7 @ ABILITIES >=16 | P-K+8 @ ABILITIES >=16 | P-K+9 @ ABILITIES >=16 | P-K+10 @ ABILITIES >=16 | P-K+11 @ ABILITIES >=16
{
output P @ ABILITIES named "Ability [P-K+6]" 
}
else
{
K:K+6
}
}

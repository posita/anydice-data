
\-- Death Save --\


SAVE_DC: 10
BONUS: +0

CONSCIOUS: 2
SUCCEEDED: 1
FAILED: -1
FAILED_TWICE: -2

IS_ALIVE: 1
IS_DEAD: 0

function: save N:n {
  if N  = 20      {result: CONSCIOUS} 
  if N  = 1       {result: FAILED_TWICE}
  if N + BONUS >= SAVE_DC {result: SUCCEEDED} 
  result: FAILED
}


function: roll saves AB:s C:s D:s E:s{
  if [AB contains CONSCIOUS] {result: IS_ALIVE}
  SUCCESSES: [count SUCCEEDED in AB]
  FAILURES:  [count FAILED in AB]  + 2 * [count FAILED_TWICE in AB]
  if FAILURES >= 3 {result: IS_DEAD}

  if [C contains CONSCIOUS] {result: IS_ALIVE}
  SUCCESSES: SUCCESSES + [count SUCCEEDED in C]
  FAILURES:  FAILURES  + [count FAILED in C   ] + 2 * [count FAILED_TWICE in C] 
  if SUCCESSES >= 3 {result: IS_ALIVE}
  if FAILURES  >= 3  {result: IS_DEAD}

  if [D contains CONSCIOUS] {result: IS_ALIVE}
  SUCCESSES: SUCCESSES + [count SUCCEEDED in D]
  FAILURES:  FAILURES  + [count FAILED in D   ] + 2 * [count FAILED_TWICE in D]
  if SUCCESSES >= 3 {result: IS_ALIVE}
  if FAILURES  >= 3  {result: IS_DEAD}

  if [E contains CONSCIOUS] {result: IS_ALIVE}
  SUCCESSES: SUCCESSES + [count SUCCEEDED in E]
  FAILURES:  FAILURES  + [count FAILED in E   ] + 2 * [count FAILED_TWICE in E]
  if SUCCESSES >= 3 {result: IS_ALIVE}
  if FAILURES  >= 3  {result: IS_DEAD}


  result: -1 \ Should be unreachable\
}

function: death save {result: [roll saves 2d[save d20] [save d20] [save d20] [save d20]]}

output [death save] named "Death Save Survival"

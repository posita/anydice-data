set "explode depth" to 6
function: roll X:n Y:n{
 if X = 1 {result: 0}
 result: [rollhelp X+Y]
}

function: rollhelp X:n {
 if X>=65 {result: 12}
 if X>=60 {result: 11}
 if X>=55 {result: 10}
 if X>=50 {result: 9}
 if X>=45 {result: 8}
 if X>=40 {result: 7}
 if X>=35 {result: 6}
 if X>=30 {result: 5}
 if X>=25 {result: 4}
 if X>=20 {result: 3}
 if X>=15 {result: 2}
 if X>=10 {result: 1}
 result: 0
}

output [roll [explode 1d10] 9-4]

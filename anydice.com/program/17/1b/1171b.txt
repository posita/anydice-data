function: gwf N:n{
 result: {1,2,{3..N}:(N/2+1)}
}

function: attack N:n vs AC:n STR:n P:n D:n {
 if N = 1 {result: 0}
 if N = 20 {result: 1d[gwf D] * 2 + 1d[gwf D] + STR}
 if (N+STR+P) < AC {result: 0}
result: 1d[gwf D] + STR
}
SUM:0

function: avg S:s{
 result: ({1..#S}@S)/#S
}

loop AC over {8..20}{
 output (2d[attack 1d20 vs AC 3 3 10] + [attack 1d20 vs AC 3 3 4]) named "[AC]"
}

DIE_SIZES: {20, 12, 10, 8, 6, 4, 2}
NUM_DIE: 4
ADD: 0
SKILL: 5

loop N over {1..#DIE_SIZES}{
  loop M over {0..NUM_DIE-1}{
    if N!=#DIE_SIZES{
      output (NUM_DIE-M)d(N@DIE_SIZES) + Md((N+1)@DIE_SIZES) + ADD
      ADD: ADD+((N@DIE_SIZES)-((N+1)@DIE_SIZES))/2
    }
    if N=#DIE_SIZES & M=0{
      output (NUM_DIE-M)d(N@DIE_SIZES)+ADD
      ADD: ADD+(N@DIE_SIZES)
    }
  }
  ADD: ADD+SKILL
}

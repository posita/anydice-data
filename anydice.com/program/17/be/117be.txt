function: evaluate ROLL:s {
 SUCCESSES: ROLL <= 5
 if SUCCESSES > 0 { result: SUCCESSES}
\if SUCCESSES > 0 { result: SUCCESSES + (ROLL = 1) } to make a 1 count as 2 successes\
 result: 0
}

function: evaluatetwo ROLL:s {
 SUCCESSES: ROLL <= 1
 if SUCCESSES > 0 { result: SUCCESSES}
\if SUCCESSES > 0 { result: SUCCESSES + (ROLL = 1) } to make a 1 count as 2 successes\
 result: 0
}

\loop DICE over {1..10} {\
 output [evaluate 2d6] >= [evaluatetwo 10d6] named "d6"
\}\

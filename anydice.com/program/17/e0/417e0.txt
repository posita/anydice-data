function: hit N:n and T:n and R:n
{
if (N - T- R)>0
{
result: 1
}
else {result: 0}
}
function: multi hits H:s and T:n and R:n
{
COUNT: {0}
loop P over {1..#H}
{
if (P@H - T- R)>0
{
COUNT: COUNT+1
}
}
result: COUNT
}

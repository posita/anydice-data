STEPS: {30,51,66,76,83}

function: astens A:n {
if A = 10 {
result: 0
}
result: A * 10
}

function: asones A:n {
if A = 10 {
result: 0
}
result: A
}

function: percentile X:n Y:n {
if [astens X] = 0 & [asones Y] = 0 {
result: 100
}
result: [astens X] + [asones Y]
}


function: perma A:n B:n SKILL {
if [percentile A B] > SKILL {

if A < B {
result: A
}
result: B
}
}

RED: d10
BLUE: d10

SKILL: 30

loop S over STEPS {
output [perma BLUE RED S] named "[S]"
}

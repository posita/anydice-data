\ Welcome to Kotarou's simple calculator \
\ To work out the probability of your roll, succeeding, copy this program \
\ Then: edit the ACCURACY, MODIFIER, and TARGET values to get your probability of success \
\ A positive accuracy means bonus accuracy, a negative modifier means difficulty \

\ Remember an output value of 0 = miss, a value of 1 = hit \


ACCURACY: 0   \ accuracy is positive, e.g. 1, difficulty is negative, e.g -2 \
TARGET: 10    \ the rough evasion or edef of what you are aiming at. \
MODIFIER: 2   \ targeting or systems + any other bonuses. \

\ DO NOT EDIT BELOW THIS LINE UNLESS YOU KNOW WHAT YOU ARE DOING \

function: target B bonus C against D {
    A: [absolute C]
    COUNT: 0
    MOD: 0
    if C < 0 {
        MOD: [highest 1 of Ad6]
    } else {
        MOD: - [highest 1 of Ad6]
    }
    COUNT: [count {D..50} in 1d20 + MOD + B]
    result: COUNT
}
output[target MODIFIER bonus ACCURACY against TARGET] named "Probability of hitting [TARGET] with a bonus of +[MODIFIER] and [ACCURACY] accuracy/difficulty"



\loop P over {0..6} {
    output [target P bonus 1 against 10] named "Targeting: [P]"
}\

\loop TARGET over {8..18} {
    loop BONUS over {-6..6} {
       output [target 2 bonus BONUS against TARGET] named "[BONUS]"
    }
}\

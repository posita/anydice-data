function: evaluate ROLL:n {
 REGEN: (ROLL >= 13 & ROLL <= 24)
 WISDOM: (ROLL >= 25 & ROLL <= 36)
 STEADFAST: (ROLL >= 37 & ROLL <= 48)
 POWER: (ROLL >= 49 & ROLL <= 60)
 LIFE: (ROLL >= 31 & ROLL <= 72)
 GORMITE: (ROLL >= 73)
 if REGEN > 0 { result: 1 }
 if WISDOM > 0 { result: 2 }
 if STEADFAST > 0 { result: 3 }
 if POWER > 0 { result: 4 }
 if LIFE > 0 { result: 5 }
 if GORMITE > 0 { result: 6 }
 result: 0
}

loop DICE over {1..5} {
 output [evaluate DICE d20] named "[DICE]d"
}

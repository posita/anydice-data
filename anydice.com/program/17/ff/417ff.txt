function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: divine smite nivel A:n {
    result: 1d8 + Ad8
}

function: eldritch smite nivel A:n {
    result: Ad8
}

function: ataque com a marreta {
    result: [reroll {1,2} on d6 to d6] + [reroll {1,2} on d6 to d6] + 4
}

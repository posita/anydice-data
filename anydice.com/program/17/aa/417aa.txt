RANGE: 40
DIE: d12
DIEMAX: [maximum of DIE]
DIEMID: DIEMAX / 2 + 1

DLUCK: d{0,0,0,0,0,1}
DTIE: d{0..1}
DZFIVE: d{0..5}

function: clamp N:n between MIN:n and MAX:n {
  if N > MAX { result: MAX }
  if N < MIN { result: MIN }
  result: N
}

function: extend N:n {
  if N = DIEMAX { result: N + [explode DZFIVE] }
  result: N
}

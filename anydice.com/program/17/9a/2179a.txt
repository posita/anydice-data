function: attackroll ROLL:n vs ENEMY:n {
D: ROLL+STRA+STRB-ENEMY-ENEMYS
if D < 0 { result: 0 }
result: D
}

if A = B { if A+B>M {M: A+B }}

if A = D { if A+D>M {M: A+D }}
if D = B { if D+B>M {M: D+B }}
if C = D { if C+D>M {M: C+D }}

if A = B & B = C & C = D { if A+B+C+D>M {M: A+B+C+D }}

 result: M
}

 result: M

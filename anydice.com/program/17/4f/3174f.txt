function: roll N:s PROF {
RESULT: N
if N = 1 { result: RESULT + PROF * d10 + d4 + d4 }
if N = 2 { result: RESULT + PROF * d10 + d4 + d6 }
if N = 3 { result: RESULT + PROF * d10 + d6 + d6 }
if N = 4 { result: RESULT + PROF * d10 + d6 + d8 }
if N = 5 { result: RESULT + PROF * d10 + d8 + d8 }
if N = 6 { result: RESULT + PROF * d10 + d8 + d10 }
if N = 7 { result: RESULT + PROF * d10 + d10 + d10 }
if N = 8 { result: RESULT + PROF * d10 + d10 + d12 }
if N = 9 { result: RESULT + PROF * d10 + d12 + d12 }
}


loop I over {1..9} {
output [roll I 0] named "[I] level"
output [roll I 1] named "[I] level prof"
output [roll I 1]+[roll dI 0] named "[I] level fusion"
}

output [roll d9 0] named "summary"
output [roll d9 1] named "summary prof"

output [roll d9 0]+[roll d9 1] named "summary fusion"

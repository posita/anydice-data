\ Welcome to Kotarou's simple calculator \
\ To work out the probability of your roll, succeeding, copy this program \
\ Then: edit the modifier and targetcheck values to get your probability of success \
\ A positive modifier means accuracy, a negative modifier means difficulty \
\ If you want to find the probability of a crit, change the checkCrit value to 1, otherwise leave it at 0 \

function: target B bonus C against D {
    A: [absolute C]
    COUNT: 0
    MOD: 0
    if C < 0 {
        MOD: [highest 1 of Ad6]
    } else {
        MOD: - [highest 1 of Ad6]
    }
    COUNT: [count {D..50} in 1d20 + MOD + B]
    result: COUNT
}


\loop P over {0..6} {
    output [target P bonus 1 against 10] named "Targeting: [P]"
}\

loop BONUS over {-6..6} {
    output [target 2 bonus BONUS against 10] named "P(hit against 10) with bonus: [BONUS]"
}

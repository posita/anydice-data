\**** Fighting modifiable Calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}


\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}


\Barb20 gaxe\
ATTACKMOD: 7
DMGBONUS: 14
PROFICIENCY: 6
ACCBONUS: -5
CRITROLL: 20
ATTACKTIMES: 20
DAMAGE: 1d12 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 12 + 4d12 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC]/10 named "B20 gsword: RA+GWM"

\Fighter Champ A+SS\
ATTACKMOD: 5
DMGBONUS: 10
PROFICIENCY: 6
ACCBONUS: -5

ATTACKTIMES: 20
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 8 + 1d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll 1d20 vs ENEMYAC]/10 named "F20 Lbow: A fs + SS"

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
PROFICIENCY: 5
HEX: 1d6
SMITE: 6d8
BOLT: 4d12 + HEX
DMGBONUS: 5
ACCBONUS: 1
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 2d4 + 2 + HEX + ATTACKMOD + DMGBONUS  + HEX + ATTACKMOD + DMGBONUS + SMITE + BOLT
CRITDAMAGE: 2 * (2d4 + HEX + HEX + SMITE + BOLT) + 2 + ATTACKMOD + DMGBONUS + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Celestial Whip"

function: explode N:n on SIZE:n {
 if N = 12 {
  result: [highest of N and 1d20] 
}
 if N = SIZE {
  result: [highest of N and [explode d(SIZE + 2) on SIZE + 2] ]
 }
 result: N
}

FOUR: [explode d4 on 4]
SIX: [explode d6 on 6]
EIGHT: [explode d8 on 8]
TEN: [explode d10 on 10]
TWEL: [explode d12 on 12]

output 1@1dFOUR named "Unskilled"
output 1@2dFOUR named "Dabbling"
output [highest of 1@2dSIX and 1@1dFOUR] named "Trained"
output [highest of 1@1dFOUR and 1@1dEIGHT] named "Skilled"
output 1@2dEIGHT named "Proficient"
output [highest of 1@1dEIGHT and 1@1dTEN] named "Expert"

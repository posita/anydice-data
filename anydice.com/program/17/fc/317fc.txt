\**** Fighting modifiable Calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}


\looped against enemy AC from 15-20\
ENEMYAC: d{13..20}

ATTACKMOD: 7 + 3
DMGBONUS: 4
PROFICIENCY: 6
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 9
DAMAGE: 1d12 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 12 + 4d12 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC]/3 named "B20 gaxe+3: RA"

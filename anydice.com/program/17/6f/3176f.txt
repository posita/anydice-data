AC: 20
MAPFLURRY: {0, 2, 4, 4, 4, 4, 4}
MAP: {0, 5, 10, 10, 10}
MAPAGILE: {0, 4, 8, 8, 8, 8, 8}
HEROCAP: 0
HEROPOINTS: HEROCAP


function: DIE:n BONUS:n MAP:n against AC:n house{
 RESULT: 0
 HIT: DIE+BONUS-MAP
 if HIT >= AC+10 { RESULT:2 }
 if HIT < AC+10 { RESULT:1 }
 if HIT < AC { RESULT:0 }

 if DIE = 20 { RESULT: RESULT+1 }
 if DIE = 1 { RESULT: RESULT-1 }
 
 if RESULT > 2 { result: 2}
 if RESULT < 1 & HEROPOINTS > 0{ 
  HEROPOINTS: HEROPOINTS-1
  result: [d10+10 BONUS MAP against AC house] 
 }
 if RESULT < 0 { result: 0}
 result: RESULT
}

function: DIE:n BONUS:n MAP:n against AC:n normal{
 RESULT: 0
 HIT: DIE+BONUS-MAP
 if HIT >= AC+10 { RESULT:2 }
 if HIT < AC+10 { RESULT:1 }
 if HIT < AC { RESULT:0 }

 if DIE = 20 { RESULT: RESULT+1 }
 if DIE = 1 { RESULT: RESULT-1 }
 
 if RESULT > 2 { result: 2}
 if RESULT < 1 & HEROPOINTS > 1{ 
  HEROPOINTS: HEROPOINTS - 1
  result: [d20 BONUS MAP against AC normal] 
 }
 if RESULT < 0 { result: 0}
 result: RESULT
}

function: hatchet RESULT:n {
 result: (d6+4) * RESULT
}

function: aquebus RESULT:n {
 if RESULT = 2 { result: (1d8 + 3) * 2 + 1d12 }
 else { result: (1d8 + 3) * RESULT }
}

function: ranger hunted ACTIONS{
 DAMAGE: 0
 loop N over {1..1+ACTIONS} {
  DAMAGE: DAMAGE + [hatchet [d20 9 N@{MAPFLURRY} against AC normal]]
 }
 result: DAMAGE
}

function: gunslinger stance ACTIONS{
 DAMAGE: 0
 loop N over {1..ACTIONS-1} {
  DAMAGE: DAMAGE + [aquebus [d20 11 N@{MAP} against AC normal]]
 }
 result: DAMAGE
}


function: ranger agile ACTIONS{
 DAMAGE: 0
 loop N over {1..ACTIONS} {
  DAMAGE: DAMAGE + [hatchet [d20 9 N@{MAPAGILE} against AC normal]]
 }
 result: DAMAGE
}

output [ranger hunted 3] named "Ranger +Hunted 3 Actions "
HEROPOINTS: HEROCAP
output [gunslinger stance 3] named "Gunslinger +Stance 3 Actions"
HEROPOINTS: HEROCAP
output [ranger agile 3] named "Ranger Agile 3 Actions"
HEROPOINTS: HEROCAP

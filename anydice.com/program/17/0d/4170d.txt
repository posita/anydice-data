function: health die DIE:n level LEVEL:n {
 X: [maximum of 1dDIE]
 MIN: ([maximum of dDIE]/2) + 1
 result: [maximum of 1dDIE] + (LEVEL-1)d{MIN..DIE}
}

set "maximum function depth" to 10
loop N over {4,6,8,10,12}{
 loop R over {3..3}{
  output [health die N level R] named "level [R] with hit die d[N]"
 }
}

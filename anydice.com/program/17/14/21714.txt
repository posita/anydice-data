H: 6   \Hero die\
S: 8   \Stat\
C: 8   \Career\
E: 0   \Expertise\
A: 0   \Advantage\
M: 0   \Momentum\
D: 1   \Number of difficulty dice d4's\
TN: 4  \Target number for action dice\
AN: 1  \Activation number for d4\

function: evaluate HERO:s and STAT:s and CAREER:s and EXPERTISE:s and ADVANTAGE:s and MOMENTUM:s vs DIFFICULTY:s {
 HEROSUCCESS: HERO >= TN
 STATSUCCESS: STAT >= TN
 CAREERSUCCESS: CAREER >= TN
 EXPERTISESUCCESS: EXPERTISE >= TN
 ADVANTAGESUCCESS: ADVANTAGE >= TN
 MOMENTUMSUCCESS: MOMENTUM >= TN
 COMPLICATION: 0
 if DIFFICULTY > 0 {COMPLICATION: DIFFICULTY <= AN}
 result: HEROSUCCESS + STATSUCCESS + CAREERSUCCESS + EXPERTISESUCCESS + ADVANTAGESUCCESS + MOMENTUMSUCCESS - COMPLICATION
}

output [evaluate 1dH and 1dS and 1dC and 1dE and 1dA and 1dM vs Dd4]

loop M over {-1..4} {
  W: 6 
  S: 10 

function: blades ROLL:s { 
  H: 1@ROLL + M
  L: 2@ROLL + M

  if L >= W {                  \ if the lower roll is a weak hit \
    if H >= S {                 \ if the higher roll is a strong hit \
       result: 2 + (L >= S)     \ then success, potentially critical \
    }  result: 1                \ else, conditional success \
  } if H >= S {                \ else, if the higher roll is a strong hit \ 
       result: 1                \ then conditional success \
  }   result: 0 + (H >= W)     
}

output [blades 2d10] named "Outcome +[M]"

}

SIZE:6 \size of dice d4,d6,d8,d10,d12,d20\
TARG:3 \target number to beat on dices\
POOL:3 \pool size\
set "maximum function depth" to 4 \iterations when explosive\

function: sixplode N:n
{
if N = SIZE { result: 1 + [sixplode dSIZE]} \for explosive 6's + [sixplode dSIZE]\
if N > TARG { result: 1 }
if N = TARG { result: 1 }
result: 0 
}

loop POOL over {1..5}{
loop TARG over {2..6}{   
 output POOLd[sixplode dSIZE] named "[POOL]d TN:[TARG]"
}
}

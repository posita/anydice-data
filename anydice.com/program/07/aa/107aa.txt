function: focused attack {
  result: [highest 1 of 3d6] >= 4
}
function: attack {
  result: [highest 1 of 3d6] >= 5
}
function: full attack {
  result: [attack] + [attack]
}
R: 0
function: grind A:n rounds {
  loop N over {1..A}{
    R: R + [full attack]
  }
  result: R
}

output ([focused attack]) > [grind 1 rounds] named "Focus/attack twice"
output [full attack] named "Full attack"
output [grind 2 rounds] named "Grind 2 rounds"

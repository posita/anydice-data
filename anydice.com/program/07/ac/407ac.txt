LEVEL: 5

function: hp at level LEVEL:n with con mod CON_MOD:n {
  if LEVEL >= 16 {
    result: LEVELd12 + LEVEL * 3
  }
  if LEVEL >= 11 {
    result: LEVELd10 + LEVEL * 3
  }
  if LEVEL >= 10 {
    result: LEVELd8 + LEVEL * 3
  }
  if LEVEL >= 6 {
    result: LEVELd8 + LEVEL * CON_MOD
  }
  if LEVEL >= 1 {
    result: LEVELd6 + LEVEL * CON_MOD
  }
}

function: easy scaling hp at level LEVEL:n with con mod CON_MOD:n {
  if LEVEL >= 10 {
    result: (5+3) * LEVEL
  }

  if LEVEL >= 1 {
    result: (5+CON_MOD) * LEVEL
  }
}

function: warrior hp at level LEVEL:n with con mod CON_MOD:n {
  result: 10 + LEVEL * CON_MOD + (LEVEL-1)d10
}

output [warrior hp at level LEVEL with con mod 3] named "Level [LEVEL] Warrior HP with +3 Con"
output [hp at level LEVEL with con mod 1] named "Level [LEVEL] HP with +1 Con (CD)"
output [hp at level LEVEL with con mod 2] named "Level [LEVEL] HP with +2 Con (CD)"
output [easy scaling hp at level LEVEL with con mod 1] named "Level [LEVEL] HP with +1 Con (5x)"
output [easy scaling hp at level LEVEL with con mod 2] named "Level [LEVEL] HP with +2 Con (5x)"

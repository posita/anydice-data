function: range on DIE:n between LOW:n and HIGH:n {
  if DIE >= LOW & DIE <= HIGH { result: 1 }
  result: 0
}
set "explode depth" to 1

output [range on 1d20 between 1 and 11] named "TN 12 skill any fail"
output [range on 1d20 between 12 and 23] named "TN 12 skill any success"
output [range on 1d20 between 20 and 20] named "TN 12 skill crit success"

function: attack {
  result: [highest 1 of 3d6] >= 5
}

output [attack] named "Single advantaged attack"

function: full attack {
  result: [attack] + [attack]
}

output [full attack] named "Full advantaged attack"

R: 0

function: grind A:n rounds {
  loop N over {1..A}{
    R: R + [full attack]
  }
  result: R
}

loop O over {1..10}{
  output [grind O rounds] named "Grind [O] rounds"
}

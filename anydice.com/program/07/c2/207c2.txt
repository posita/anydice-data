function: getdouble ROLL:s{
 ENDLOOP:0
 DOUBLE:0
 GRAND:0
 GRAND:[highests 2 of ROLL]
 A:0
 B:0
 POSA:0
 POSB:0
 loop P over {1..#ROLL-1}
 {
  if ENDLOOP=0 {
   A:P@ROLL
   B:(P+1)@ROLL
   if A=B {
    DOUBLE:A+B
    ENDLOOP:1
    POSA:P
    POSB:P+1
   }
  }
 }
 ENDLOOP:0
 if DOUBLE >0 {
  loop P over {1..#ROLL}
  {
   if ENDLOOP=0 {
    if P != POSA {
     if P != POSB {
      ENDLOOP:1
      DOUBLE:DOUBLE+P@ROLL
     }else{
      DOUBLE:DOUBLE+1
     }
    } 
   }
  }
 }
 result:[highest of GRAND and DOUBLE]
}

function: highests NUMBER:n of DICE:s {
 result: {1..NUMBER}@DICE
}

output [getdouble 2d6]
output [getdouble 3d6]
output [getdouble 4d6]
output [getdouble 5d6]

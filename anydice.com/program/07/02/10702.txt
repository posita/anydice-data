function: fate A:n and B:n reroll C:n {
 if A = 6 & B = 1 {
  if C < 6 {
   A: 4
  }
  else {
   A: 5
  }
 }
 if A = 1 & B = 6 {
  if C < 6 {
   B: 4
  }
  else {
   B: 5
  }
 }
 result: A - B
}

output [fate d6 and d6 reroll d6] named "Fate d6"

function: fatediv A:n and B:n reroll C:n {
 if A = 4 & B = 1 {
  if C < 4 {
   A: 3
  }
 }
 if A = 1 & B = 4 {
  if C < 4 {
   B: 3
  }
 }
 result: A - B
}

output [fatediv d4 and d4 reroll d4] named "Fate d4"

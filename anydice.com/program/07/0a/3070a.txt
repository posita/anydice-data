RAISE: 0
loop I over {0..1}{
  if I = 1 {
    RAISE: 6
  }

  loop J over {2..6}{
    STR: 2*J
    loop K over {J..6}{
      WPN: 2*K
      loop N over {11..14}{
        output [explode dSTR] + [explode dWPN] + [explode dRAISE] >= N named "d[STR]+d[WPN]+d[RAISE] shock [N]"
        output [explode dSTR] + [explode dWPN] + [explode dRAISE] >= N+4 named "d[STR]+d[WPN]+d[RAISE] wound [N]"
      }
    }
  }
}

function: sr R:n ros N:n {
 if N >= R { result: 1 + [sr R ros d6] }
 result: N >= 5
}

SR: [sr 6 ros d6]

\ INPUT ARGUMENTS BELOW... \

ROLLS_PREV: 16
ROLLS_POST: 20
THRESHOLDS: 9

\ OUTPUT CALCULATIONS... \

loop N over {ROLLS_PREV..ROLLS_POST} {
 output [lowest of NdSR and THRESHOLDS] named "[N]d - Push the Limit"
}

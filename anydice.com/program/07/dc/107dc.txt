MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

ATTACK: 4
HITDAMAGE: d6+5
CRITICALDAMAGE: 2d6+5
MISSDAMAGE: 0
SKELIN: 6
loop AC over {10..20} {
 output SKELINd[damage for [attack d20 vs AC]] named "[AC]"
}

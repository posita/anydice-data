function: fullround at L with C ac DR dr plus D dam and A atk {
 TOTAL: 0
 INT: [int at L]
 DEX: [dex at L]
 DAM: [dice at L] + INT + D - DR
 BAB: [bab at L]
 loop SING over {1..#BAB}{
  BONUS: SING@BAB
  RESULT: (1d20 + BONUS + DEX + A >= C) * DAM 
  TOTAL: RESULT + TOTAL
 }
 result: TOTAL
}

PROFICIENCY: {3, 4, 5, 6}
DC: 5+5d4
DECEPTION: 1d20+4
INSIGHT: 1d20+2
INTIMIDATION: 1d20+4
BET: 700

function: payout for S:n betting B:n {
  if S = 0 {
    result: -1 * B
  }
  if S = 1 {
    result: -1 * B / 2
  }
  if S = 2 {
    result: B
  }
  if S = 3 {
    result: 2 * B
  }
}

loop P over PROFICIENCY {
  SUCCESSES: (DECEPTION+P>= DC) + (INSIGHT+P>= DC) + (INTIMIDATION+P >= DC)
  output [payout for SUCCESSES betting BET] named "Payout (Prof +[P])"
  output SUCCESSES named "Successes (Prof +[P])"
}

DC: 5+5d4
DECEPTION: 1d20+7
INSIGHT: 1d20+5
INTIMIDATION: 1d20+7
BET: 700
SUCCESSES: (DECEPTION >= DC) + (INSIGHT >= DC) + (INTIMIDATION >= DC)

function: payout for S:n betting B:n {
  if S = 0 {
    result: -1 * B
  }
  if S = 1 {
    result: -1 * B / 2
  }
  if S = 2 {
    result: B
  }
  if S = 3 {
    result: 2 * B
  }
}

output [payout for SUCCESSES betting BET] named "Payout"

output SUCCESSES named "Successes"

MAT: 9
DEF: 12
ARM: 18

function: attack ROLL:n plus ATTACK:n vs DEFENSE:n {
 if ROLL+ATTACK < DEFENSE { result: 0 }
 else { result: 1 }
}

function: damage for ATTACKRESULT:n with weapon WEAPON:n {
 if ATTACKRESULT = 1 {result: WEAPON}
 else {result: 0}
}

function: pow POW:n arm ARM:n dice X:n {
 if (X+POW-ARM <= 0) {result: 0}
 else {result: X+POW-ARM}
}

CAVCHARGE: [attack 3d6 plus MAT vs DEF]
ATTACK: [attack 2d6 plus MAT vs DEF]

LUSTY_TAIL_STRIKE: [pow 14 arm ARM dice 4d6]
LUSTY_TAIL_STRIKE_CHARGE: [pow 14 arm ARM dice 5d6]
LUSTY_GREAT_SWORD: [pow 16 arm ARM dice 3d6]
LUSTY_GREAT_SWORD_CHARGE: [pow 16 arm ARM dice 4d6]

output [damage for CAVCHARGE with weapon LUSTY_TAIL_STRIKE_CHARGE] + [damage for ATTACK with weapon LUSTY_GREAT_SWORD] named "Lusty Tail Strike Charge"
output [damage for CAVCHARGE with weapon LUSTY_GREAT_SWORD_CHARGE] + [damage for ATTACK with weapon LUSTY_TAIL_STRIKE] named "Lusty Great Sword Charge"

function: ROLL:s hits vs DC:n with mastery{
 BADONES: (ROLL = 1)
 SUCCESSES: (ROLL >= DC)
 CRITICAL: (ROLL=12)
 MODS: SUCCESSES+CRITICAL
 if (MODS>8) {result: 8}
 if (MODS<-3) {result: -3}
 result: MODS
}

function: ROLL:s hits vs DC:n{
 BADONES: (ROLL = 1)
 SUCCESSES: (ROLL >= DC)
 CRITICAL: (ROLL=12)
 MODS: SUCCESSES-BADONES+CRITICAL
 if (MODS>8) {result: 8}
 if (MODS<-3) {result: -3}
 result: MODS
}

RANKS:3
ATTR:5
DC:9-RANKS
DICE: ATTR
ADV:0
loop RANKS over {0..5}{
DC:ADV+9-RANKS
output [(DICE)d12 hits vs DC] named "[DICE]d12 with DC [DC] and [RANKS] Ranks" 
}

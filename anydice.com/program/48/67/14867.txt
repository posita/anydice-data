function:  ROLL:n vs DEFENSE:n ATTACK:n {
 if ROLL = 1 {result: MISS}
 if ROLL = 20 {result: CRITICALHIT}
 if ROLL + ATTACK>= DEFENSE {result: HIT}
 result: MISS
}


\ Damage functions \

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: object damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: OBJECTCRITICALDAMAGE
 }

  result: OBJECTDAMAGE
 }
 result: MISSDAMAGE
}

\ Multi attacks \

function: multiattack NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for [d20 vs AC BARDATTACK]]
  }
  result: TOTAL
}


function: multiattackadvantage NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for [[highest 1 of 2d20] vs AC BARDATTACK]]
  }
  result: TOTAL
}

function: multiattackdoubleadvantage NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for [[highest 1 of 3d20] vs AC BARDATTACK]]
  }
  result: TOTAL
}

function:object multiattack NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [object damage for [d20 vs AC OBJECTATTACK]]
  }
  result: TOTAL
}

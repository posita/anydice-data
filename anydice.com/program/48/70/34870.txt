set "explode depth" to 10

RANGE: 40
DIE: d12
DIEMAX: [maximum of DIE]
DIEMID: DIEMAX / 2 + 1

DLUCK: d{0,0,0,0,0,1}
DZFIVE: d{0..5}

function: clamp N:n between MIN:n and MAX:n {
  if N > MAX { result: MAX }
  if N < MIN { result: MIN }
  result: N
}

function: roll {
  result: [roll DIE]
}
function: roll N:n {
  if N = DIEMAX { result: N + [explode DLUCK] }
  if N = 1 { result: N - [explode DLUCK] }
  result: N
}

output [clamp [roll]-DIEMID between -RANGE and RANGE]
\
output 3d8-7
\

output [clamp 2d[explode DZFIVE] between -RANGE and RANGE] named "2z5!"
output [clamp d6+[explode DZFIVE]-4-[explode DZFIVE] between -RANGE and RANGE] named "d6+z5! vs"

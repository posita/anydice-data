
function: ATTACKS:s vs {
  SUM: {0}
  loop ATTACK over ATTACKS {
    SUM: SUM + [1d20 plus]
  }
  result: SUM
}
function: ROLL:n plus {
  if (ROLL >= 19){ result: 1 } 
  if (ROLL < 19) { result: 0 }
}

function: fate ROLL:s {
  result: [count 6 in ROLL] - [count 1 in ROLL] - [count 2 in ROLL]
}

function: skill ROLL:s over TARGET {
  SUCCESSES: ROLL >= TARGET
  result: SUCCESSES - [count 1 in ROLL]
}

loop TARGET over {4, 6, 8} {
  loop ATTR over {1..5} {
    output ([fate (5 - ATTR) d6] + [skill ATTR d10 over TARGET]) > 0 named "[ATTR] vs [TARGET]"
  }
}

AC: {10..20}

function: attack DMG:d crit CRIT_DMG:d penalty PENALTY:n rider RIDER:n stat STAT:n prof PROFICIENCY:n roll ROLL:n{
    MODIFIER: (STAT-10)/2
    DMG_WITH_BONUS: DMG + MODIFIER + RIDER
	
    if ROLL = 1 {
        result: 0
    }
    if ROLL = 20 {
        result: DMG_WITH_BONUS + CRIT_DMG
    }
    if ROLL + MODIFIER + PROFICIENCY - PENALTY >= AC {
        result: DMG_WITH_BONUS
    }
    result: 0
}


output
2d[attack 1d8 crit 1d8 penalty 5 rider 10 stat 20 prof 3 roll [highest 1 of 3d20]] 
named "Sharpshooter Samurai w/ Fighting Spirit, Elven Accuracy, 20 Dex"

2d[attack 1d8 crit 1d8 penalty 0 rider 0 stat 19 prof 1 roll [highest 1 of 2d20]] +

function: roll it against SKILL:n {
  SKILLDIE: [{1..SKILL} contains 1d20]        \ calculating if the first die is successful (1) or failure (0) \
  result: [helper SKILLDIE SKILL]             \ calling another function \
}

function: helper D:n S:n {
  if D { result: 1 + 2d[count {1..S} in 1d20] }     \ if first die is successful \
  else { result: -1 - 2d[count {S+1..20} in 1d20] } \ if first die is a failure  \
}

loop SKILLEVEL over {1..19} {
  output [roll it against SKILLEVEL] named "Rolling against a skill level of [SKILLEVEL]"
}

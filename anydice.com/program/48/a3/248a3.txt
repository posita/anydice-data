function: explode N:n on MAGICNUMBER:n {
 if N = MAGICNUMBER {
  result: N + [explode d4 on MAGICNUMBER]
 }
 result: N
}

set "maximum function depth" to 3
loop X over {1..4} {
 output [explode d4 on X] named "explode on [X]"
}

output [explode d4] named "explode on 4"

function:  ROLL:n vs DEFENSE:n ATTACK:n {
 if ROLL = 1 {result: MISS}
 if ROLL = 20 {result: CRITICALHIT}
 if ROLL + ATTACK>= DEFENSE {result: HIT}
 result: MISS
}


\ Damage functions \

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

 if ATTACKRESULT = CRITICALHIT {
  result: GRIFFONCRITICALDAMAGEBEAK

 result: MISSDAMAGE

\ Multi attacks \

function: griffon claws multiattack NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [griffon claws damage for [d20 vs AC GRIFFONATTACK]]
  }
  result: TOTAL
}

function: griffon beak multiattack NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [griffon beak damage for [[highest 1 of 2d20] vs AC GRIFFONATTACK]]
  }
  result: TOTAL
}

\Enter AC of enemy here\
AC:19
\Number of Entropic points spent on attack\
EP:1
\Scrumpy full attack? 1 = yes, 0 = no\
FA:1

function: X:n viridian {

if X >= AC {result: 1d3+EPd6+3}
result: 0
}

function: X:n scrumpy {

if X >= AC {result: 1d6}
result: 0
}

function: X:n twobek {

if X >= AC {result: 1d8}
result: 0
}

output [1d20+5 viridian] named "Viridian Avg Dmg"
output [1d20-1 twobek] named "Twobek Avg Dmg"

if FA = 0 {
  output [1d20+4 scrumpy] named "Scrumpy Avg Dmg"
}
else {
  output [1d20+1 scrumpy]+[1d20+1 scrumpy] named "Scrumpy Avg Dmg"
}

output 1d3+EPd6+3 named "Viridian Dmg"
output 1d8 named "Twobek Dmg"
output (FA+1)d6 named "Scrumpy Dmg"

output 1d20+5 >= AC named "Viridian Hit"
output 1d20-1 >= AC named "Twobek Hit"
output 1d20+(4-(FA*3)) >= AC named "Scrumpy Hit"

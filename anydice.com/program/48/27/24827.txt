F: {-1,0,1}

function: init DIE:d {
  result: [explode DIE on DIE DIE]
}

function: explode ROLLEDVALUE:n on INIT:n DIE:d {
 if ROLLEDVALUE = INIT { result: INIT + [explode DIE on INIT DIE] }
 result: INIT
}

set "maximum function depth" to 8
output [init dF] named "Open Ended Fudge"
output 4dF named "Standard 4dF"

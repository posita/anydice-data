function: attack ATTACKROLL:n plus ATTACKBONUS:n for DAMAGEDICE:n plus DAMAGEBONUS:n against AC:n{
  if ATTACKROLL = 20 {result: 2*DAMAGEDICE + DAMAGEBONUS}
  if ATTACKROLL + ATTACKBONUS >= AC & ATTACKROLL != 1 {result: DAMAGEDICE + DAMAGEBONUS}
  result: 0
}

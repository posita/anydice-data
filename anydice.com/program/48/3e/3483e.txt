LOOP: {2..8} \ min to max dice to show statistics for \

SUCCESSES: {5, 6} \ change to { 6 } for high pressure or {4, 5, 6} for low pressure \
RRALLOWED: {1, 2, 3, 4} \ which dice can be rerolled, 0 for "none", {1} for training or e.g. {1, 2, 3, 4} for expert \
MAXREROLLS: 4 \ set to DICE if all can be rerolled, or whatever you want \

function: pool DSP:s {
 S: [count SUCCESSES in DSP]
 REROLLS: [lowest of MAXREROLLS and [count RRALLOWED in DSP]]
 S: S + [count SUCCESSES in REROLLSd6]
 result: S
}

loop N over LOOP {
  output [pool Nd6] named "[N]d6"
}

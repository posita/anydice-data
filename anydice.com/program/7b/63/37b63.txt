function: purehits DICE:d against TROUBLE:n TRUMBER:n challenge DIFF:n{
HITS: [count {4..12} in DICE]
TROB: TROUBLE
YIKES: TRUMBER
DIFFICULTY: DIFF
HIGHEST: [highest 1 of DICE]
TRETBACK: HIGHEST < TROUBLE
SETBACKS: [count 1 in DICE]
TOTALSET: SETBACKS + (TRETBACK * YIKES)
MEET: HITS - TOTALSET
result: HITS
}

function: eid DICE:d against TROUBLE:n TRUMBER:n challenge DIFF:n{
HITS: [count {4..12} in DICE]
TROB: TROUBLE
YIKES: TRUMBER
DIFFICULTY: DIFF
HIGHEST: [highest 1 of DICE]
TRETBACK: HIGHEST < TROUBLE
SETBACKS: [count 1 in DICE]
TOTALSET: SETBACKS + (TRETBACK * YIKES)
MEET: HITS - TOTALSET
result: MEET
}

Z: 1

function: roll DICE:s vs TN:n {
  result: (1@DICE >= TN) + (2@DICE >= TN)
}

function: pbta DICE:n {
  result: (DICE >= 7) + (DICE >= 10)
}

loop TN over {10..20} {
  PBTAMOD: 3-((TN-10)/2)
  if (TN/2 = (TN+1)/2) {
    output [pbta 2d6+PBTAMOD] named "pbta [PBTAMOD]"
  }
  output [roll 3d20 vs TN] named "3d20 vs [TN]"
}

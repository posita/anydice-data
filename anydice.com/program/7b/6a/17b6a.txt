\ expects relabeled dice where < 0 is fail and > 0 is success \
function: count fails in ROLL:s if N:n successes {
  FAILS: ROLL < 0
  SUCCESSES: ROLL > 0
  if SUCCESSES != N { result: d{} } \ ignore this case \
  else { result: FAILS }
}

N: 4
POOL: Nd{-1,0,+1}

loop K over {0..N} {
  output [count fails in POOL if K successes] named "fails assuming [K] successes out of [N]"
  M: N - K
  output Md{0,1} named "[M]d{0,1}"
}

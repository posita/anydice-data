MISS: 000
HIT:  111
CRIT: 222

function: roll ROLLTOHIT:n vs DC:n {
 if ROLLTOHIT = 1 
   {result: MISS}
 if ROLLTOHIT = 20 
   {result:CRIT}
 if ROLLTOHIT >= DC
   {result: HIT}  
 else
   {result: MISS}
}

function: damage HITS:n with WEAP:d crit CRITDMG:n {
  if HITS = CRIT {result:CRITDMG}
  if HITS = HIT {result:WEAP}
  else {result:0}
}

loop ARMOR over {8,10,12} {
  output [damage [roll d20 vs ARMOR] with d4 crit 4] named "d4 dmg vs ac[ARMOR]"
  output [damage [roll d20 vs ARMOR] with d6 crit 6] named "d6 dmg vs ac[ARMOR]"
  output [damage [roll d20 vs ARMOR] with d8 crit 8] named "d8 dmg vs ac[ARMOR]"
  output [damage [roll d20 vs ARMOR] with 2d4 crit 8] named "2d4 dmg vs ac[ARMOR]"

  output [damage [roll d20 vs ARMOR] with d10 crit 10] named "d10 dmg vs ac[ARMOR]" 
  output [damage [roll d20 vs ARMOR] with d12 crit 12] named "d12 dmg vs ac[ARMOR]" 

  output [damage [roll d20 vs ARMOR] with 2d6 crit 12] named "2d6 dmg vs ac[ARMOR]"
  output [damage [roll d20 vs ARMOR] with 2d8 crit 16] named "2d8 dmg vs ac[ARMOR]" 
}

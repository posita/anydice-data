function: fractal DIE:n {
if DIE = 1 {
result: -1
} else {
if DIE > 6 {
result: 1
} else {
result: 0
}
}
}

function: reroll DIE:d if OUTCOME:n is less than TARGET:n {
if OUTCOME < TARGET {
result: DIE
} else {
result: OUTCOME
}
}

DICESET: {8, 10, 12, 20}
loop N over {1..#DICESET} {
TOTAL: 0
loop X over {1..N} {
TOTAL: TOTAL + [fractal d(X@DICESET)]
}
output TOTAL named "Level [N]"
output [reroll TOTAL if TOTAL is less than 1] named "Level [N] with reroll"
}

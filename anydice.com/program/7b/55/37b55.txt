


function: shoot bs BS strength S ap AP at toughness T save SV {

    SV_THRESH: SV - AP
    WOUND_THRESH: 0
    if S < T {
        WOUND_THRESH: 5
    }

    if (S * 2) < T {
        WOUND_THRESH: 6
    }

    if S = T {
        WOUND_THRESH: 4
    }
    
    if S > T {
        WOUND_THRESH: 3
    }

    if S > (T * 2) {
        WOUND_THRESH: 2
    }
    
    IS_HIT: d6 >= BS
    IS_WOUND: d6 >= WOUND_THRESH
    IS_NOT_SAVED: d6 < SV_THRESH
    
    
    result: IS_HIT * IS_WOUND * IS_NOT_SAVED
}

output 5d[shoot bs 2 strength 4 ap 0 at toughness 6 save 5]
output 5d[shoot bs 2 strength 4 ap 0 at toughness 6 save 2]

\**** Damage calculator duelling Vs archery ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
DMGBONUS: 2
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 19
ATTACKTIMES: 3
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 8d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Level 8 Hero Pally, Dueling, critfisher"

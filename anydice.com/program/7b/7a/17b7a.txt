function: roll DICE BONUS {
    if (DICE<2){
        result: [explode d6] + BONUS
    } else {
        result: [explode d6] + (DICE-1)d6 + BONUS
    }
}

set "explode depth" to 5

loop D over {1..3} {
  loop O over {0..3} {
    loop DIFF over {6..6}{
      output [roll D O] > DIFF named "[D]D + [O]"
    }
  }
}

DPERHIT: 3
RED: {0,1,0,1,1,0}

WHITE: {1,1,2,2,3,3}

function: avoid hits with X:n rerolls and B:n bonus {
  result: Xd([highest 1 of 2d10] + B >= TOHIT) + (DICE-X)d(1d10 + B >= TOHIT)
}

function: round N:n to zero {
  if N < 0 {
    result: 0
  }
  result: N
}

function: avoid danger with X:n rerolls B:n bonus and A:d armor {
  DANGER: [avoid hits with X rerolls and B bonus] * DPERHIT
  result: [round (DANGER - A) to zero]
}

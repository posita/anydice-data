function: blades ROLL:s DC {
 result: (1@ROLL >= 4) + (1@ROLL >= 6) + (2@ROLL >= 6)
}
function: ogre ROLL:s DC {
 result: (#ROLL@ROLL <= DC) + ((#ROLL-1)@ROLL <= DC) + (#ROLL@ROLL <= 1)
}
output [blades 2@2d6 3] named "blades 0d6"
loop P over {1..6} {
 output [blades Pd6 3] named "blades [P]d6"
}
loop P over {3..16} {
 output [ogre 2d20 P] named "2d20 under [P]"
}

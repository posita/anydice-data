\ the dice you roll \
SIX: d6
SEVEN: d{0..7}
DICE: SIX
COUNT: 1

\ stats \
HP: 4
ATK: 4
DEF: -1
EVD: +2
MOV: 0

\ helper functions \
function: clamp NUM:n to MIN:n {
	if NUM < MIN {
		result: MIN
	} else {
		result: NUM
	}
}
function: clamp NUM:n {
	result: [clamp NUM to 1]
}
function: defend ROLL:s DEF:n ATK:n HP:n {
	NEW: {}
	loop N over ROLL {
		A: [clamp (N + DEF)]
		B: [clamp (ATK - A)]
		C: [clamp (HP - B) to 0]
		NEW: {NEW, C}
	}
	result: NEW
}
function: evade ROLL:s EVD:n ATK:n HP:n {
	NEW: {}
	loop N over ROLL {
		A: [clamp (N + EVD)]
		if (A > ATK) {
			NEW: {NEW, HP}
		} else {
			B: [clamp (HP - ATK) to 0]
			NEW: {NEW, B}
		}
	}
	result: NEW
}

output [evade COUNTdDICE EVD ATK HP] named "evade"
output [defend COUNTdDICE DEF ATK HP] named "defence"
output [clamp (COUNTdDICE + MOV)] named "movement"

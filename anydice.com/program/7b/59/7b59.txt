MISS: 000
HIT:  111
CRIT: 222

function: roll ROLLTOHIT:n vs DC:n {
 if ROLLTOHIT = 1 
   {result: MISS}
 if ROLLTOHIT = 20 
   {result:CRIT}
 if ROLLTOHIT >= DC
   {result: HIT}  
 else
   {result: MISS}
}

function: damage HITS:n with WEAP:d crit CRITDMG:n {
  if HITS = CRIT {result:CRITDMG}
  if HITS = HIT {result:WEAP}
  else {result:0}
}

loop ARMOR over {10,15,18} {
  output [damage [roll d20 vs ARMOR] with d12 crit 12] named "AXE dmg vs ac[ARMOR]" 
  output [damage [roll d20 vs ARMOR] with 2d6 crit 12] named "SWO dmg vs ac[ARMOR]" 
}

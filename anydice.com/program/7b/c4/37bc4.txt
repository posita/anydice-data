\ stats \
HP: 4
ATK: 4
DEF: -1
EVD: +2
MOV: 0

\ the dice you roll \
SIX: d6
SEVEN: d{0..7}
DICE: SIX
COUNT: 1

\ helper functions \
function: minimum of DIE:d {
	result: 1@{DIE}
}
function: clamp NUM:n to MIN:n {
	if NUM < MIN {
		result: MIN
	} else {
		result: NUM
	}
}
function: clamp NUM:n from DIE:d {
	result: [clamp NUM to [minimum of DIE]]
}
function: defend ROLL:d DEF:n ATK:n HP:n {
	NEW: {}
	loop N over {ROLL} {
		A: [clamp (N + DEF) from ROLL]
		B: [clamp (ATK - A) to 1]
		C: [clamp (HP - B) to 0]
		NEW: {NEW, C}
	}
	result: NEW
}
function: evade ROLL:d EVD:n ATK:n HP:n {
	NEW: {}
	loop N over {ROLL} {
		A: [clamp (N + EVD) from ROLL]
		if (A > ATK) {
			NEW: {NEW, HP}
		} else {
			B: [clamp (HP - ATK) to 0]
			NEW: {NEW, B}
		}
	}
	result: NEW
}
function: move ROLL:d MOV:n {
	NEW:{}
	loop N over {ROLL} {
		A: [clamp (N + MOV) from ROLL]
		NEW: {NEW, A}
	}
	result: NEW
}

output [evade COUNTdDICE EVD ATK HP] named "evade"
output [defend COUNTdDICE DEF ATK HP] named "defence"
output [move COUNTdDICE MOV] named "movement"

function: attack D:n plus BONUS:n vs AC:n  {
    if D = 20 { result: 1 } 
    else if D = 1 { result: 0}
    else if D + BONUS >= AC { result: 1 } 
    else { result:0}
}

function: save D:n plus BONUS:n vs DC:n  {
    
    if D + BONUS >= DC { result: 2 } 
    else { result:1}
}

loop AC over {10..18} {
    output [attack d20 plus 6 vs AC]*3d10 named "Defender [AC]"
}

loop BONUS over {0..8} {
    output 4d6/[save d20 plus BONUS vs 14] named "save [BONUS]"
}

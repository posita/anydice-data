\Conan Skill Success probabilities\

FOC:0   \ <---- set your focus\
TN:10   \ <---- set you target number (skill+ability)\

function: D:s target T:n focus F:n {
   if (D<=F) {result:2}
   else if (D<=T){result: 1}
   else {result:0}
}

loop S over {0 .. 3} {
  output 5d[d20 target TN focus FOC] named "Target [TN] Focus [FOC]"
  TN:TN+1
  FOC:FOC+1
}

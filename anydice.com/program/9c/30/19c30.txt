function: count VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: COUNT
}

output [count 10 in ((1d10-1)*10)] + [count 10 in ((1d10-1))] named "At least one 10 PHB"

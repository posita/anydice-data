function: baseline ROLL:s {
 SUCCESSES: ROLL >= 7
 if SUCCESSES > 0 { result: SUCCESSES + (ROLL = 10) }
 if [count 1 in ROLL] { result: -1 }
 result: 0
}

function: double nines ROLL:s {
 SUCCESSES: ROLL >= 7
 if SUCCESSES > 0 { result: SUCCESSES + (ROLL >= 9) }
 if [count 1 in ROLL] { result: -1 }
 result: 0
}

function: reroll ones ROLL:s {
 SUCCESSES: ROLL >= 6
 if SUCCESSES > 0 { result: SUCCESSES + (ROLL = 9) }
 result: 0
}

function: reroll ones and nine ROLL:s {
 SUCCESSES: ROLL >= 6
 if SUCCESSES > 0 { result: SUCCESSES + (ROLL >= 8) }
 result: 0
}


output [baseline 1d10] named "Double 10s"
output [double nines 1d10] named "Double 9s"
output [reroll ones 1d9] named "Reroll 1s"
output [reroll ones and nine 1d9] named "Reroll 1s & Double Nines"

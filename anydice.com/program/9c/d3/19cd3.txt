function: damage DAMAGE:d rolling HIT:n with modifier MODIFIER:n to hit crit on CRIT:n dealing CRIT_DAMAGE:d against ac AC:n{
  if HIT >= CRIT { result: CRIT_DAMAGE}
  if HIT+MODIFIER < AC { result: 0 }
  result: DAMAGE
}

function: multiattack COUNT:n times dealing DAMAGE:n damage {
  result: COUNT * DAMAGE
}

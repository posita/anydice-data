function: is one A:n {
  if A = 1 { result: -100 }
  result: A
}

function: swadejoker DIE:d {
  JOKER: [explode d6]
  TRAIT: [explode DIE]
  HIGH: [highest of TRAIT and JOKER]
  CRIT: [is one HIGH]

  result: CRIT
}

function: swaderesult VALUE:n {
  if VALUE < -70 { result: -1 }
  if VALUE >= 8 { result: 8 }
  if VALUE >= 4 { result: 4 }
  result: 0
}

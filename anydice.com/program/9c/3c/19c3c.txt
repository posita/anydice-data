\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}
\looped against enemy AC from 13-19\
ENEMYAC: d{13..19}

ATTACKMOD: 5
DMGBONUS: 2
PROFICIENCY: 4
ACCBONUS: 2
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 2d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d6 + ATTACKMOD + DMGBONUS
output [attackroll d20 vs ENEMYAC] named "battlesmith lv 11 gswrd"

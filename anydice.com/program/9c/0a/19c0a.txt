\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL <= 9 { result: 10 }
 if ROLL + PROFICIENCY + CHARISMA >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 12-20\
ENEMYAC: d{12..20}

CHARISMA: 5
PROFICIENCY: 5
CRITROLL: 20
ATTACKTIMES: 16
DAMAGE: 2d6 + 1d6
CRITDAMAGE: 4d6 + 2d6


output ATTACKTIMESd[attackroll [highest 1 of 1d20] vs ENEMYAC] named "normal attack"
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "attack with advantage"

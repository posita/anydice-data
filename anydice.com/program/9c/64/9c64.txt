function: singleshot SUC:n DICE:n DMG:n PEN:n DODGED:n MOD:n {
 if SUC > 95 {result: 0}
 if DODGED {result: 0}
 if SUC <= (BS+RANGEBONUS+SIGHTBONUS+MOD) {result: [damage DICE DMG PEN 1]}
 result: 0
}

function: semishot SUC:n DICE:n DMG:n PEN:n DODGED:n AMMO:n{
 if SUC > 93 {result: 0}
 SUCCESS: (BS+RANGEBONUS+SEMIBONUS)-SUC
 if SUCCESS < 0 {result: 0}
 if SUCCESS < 20 {if !DODGED {result: [damage DICE DMG PEN 1]}}
 SUCCESS: (SUCCESS/20)-DODGED
 if SUCCESS > AMMO {result: AMMOd[damage DICE DMG PEN 1]}
 if SUCCESS > 0 {result: SUCCESSd[damage DICE DMG PEN 1]}
 result: 0
}

function: autoshot SUC:n DICE:n DMG:n PEN:n DODGED:n AMMO:n{
 if SUC > 93 {result: 0}
 SUCCESS: (BS+RANGEBONUS+AUTOBONUS)-SUC
 if SUCCESS < 0 {result: 0}
 if SUCCESS < 10 {if !DODGED {result: [damage DICE DMG PEN 1]}}
 SUCCESS: (SUCCESS/10)-DODGED
 if SUCCESS > AMMO {result: AMMOd[damage DICE DMG PEN 1]}
 if SUCCESS > 0 {result: SUCCESSd[damage DICE DMG PEN 1]}
 result: 0
}

function: flamer AGTEST:n AGTESTTWO:n DICE:n DMG:n PEN:n {
 if AGTEST > AG {
  if AGTESTTWO > AG {result: ([damage DICE DMG PEN 0] + [damage 1 0 100 0])}
  result: [damage DICE DMG PEN 0]
 }
 result: 0
}

function: shotgun SUC:n DICE:n DMG:n PEN:n DODGED:n{
 if SUC > 95 {result: 0}
 SUCCESS: (BS+RANGEBONUS+SIGHTBONUS)-SUC
 if SUCCESS < 0 {result: 0}
 if SUCCESS < 20 {if !DODGED {result: [damage DICE DMG PEN 1]}}
 SUCCESS: (SUCCESS/20)-DODGED
 if SUCCESS > 0 {result: SUCCESSd[damage DICE DMG PEN 1]}
 result: 0
}

function: deadshot SUC:n DICE:n DMG:n PEN:n DODGED:n MOD:n {
 if SUC > 95 {result: 0}
 if DODGED {result: 0}
 DEADPEN: PEN + AR
 if SUC <= (BS+RANGEBONUS+SIGHTBONUS-10+MOD) {result: [damage DICE DMG DEADPEN 0]}
 result: 0
}

function: accuratedeadshot SUC:n DICE:n DMG:n PEN:n DODGED:n MOD:n {
 if SUC > 95 {result: 0}
 if DODGED {result: 0}
 DEADPEN: PEN + AR
 SUCCESS: (BS+RANGEBONUS+SIGHTBONUS-10+MOD)-SUC
 if SUCCESS < 0 {result: 0}
 if SUCCESS < 20 {if !DODGED {result: [damage DICE DMG DEADPEN 0]}}
 SUCCESS: (SUCCESS/20)-DODGED
 if SUCCESS > 2 {result: [damage DICE+2 DMG DEADPEN 0]}
 if SUCCESS > 0 {result: [damage DICE+SUCCESS DMG DEADPEN 0]}
 result: 0
}

function: deadshotgun SUC:n DICE:n DMG:n PEN:n DODGED:n MOD:n {
 DEADPEN: PEN + AR 
 if SUC > 95 {result: 0}
 SUCCESS: (BS+RANGEBONUS+SIGHTBONUS)-SUC
 if SUCCESS < 0 {result: 0}
 if SUCCESS < 20 {if !DODGED {result: [damage DICE DMG DEADPEN 1]}}
 SUCCESS: (SUCCESS/20)-DODGED
 if SUCCESS > 0 {result: SUCCESSd[damage DICE DMG DEADPEN 1]}
 result: 0
}

function: damage D:n V:n PEN:n COVER:n {
 if COVER {
  ROLL: (Dd10 + V)- (TO +[positive ((AR + [cover COVERTEST]) - PEN)])
 } else {
  ROLL: (Dd10 + V)- (TO +[positive (AR - PEN)])
 }
result: [positive ROLL]
}

function: positive N:n {
if N > 0 { result: N}
result: 0
}

function: dodgesuccess AGTEST:n {
 if AGTEST > DODGEV {result: 0}
 SUCCESS: DODGEV - AGTEST
 if SUCCESS < 10 {result: 1}
 SUCCESS: SUCCESS/10
 if SUCCESS > 0 {result: SUCCESS}
 result: 0
}

function: cover ROLL:n {
 if INCOVER {
  if ROLL > 30 {result: COVERRATING}
 }
 result: 0
}

\Player\
BS: 40
SIGHTBONUS: 10
RANGEBONUS: 30

\Enemy\

AR: 3
AG: 30
DODGE: 0 \1 or 0\
DODGEBONUS: 10 \0, 10, 20\
if DODGE {DODGEV: AG + DODGEBONUS} else {DODGEV: AG/2}
INCOVER: 1
COVERRATING: 4

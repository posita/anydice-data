function: reroll R:n under N:n {
   if R < N { result: 1d6 } else { result: R }
}

function: crest R:n swap N:n {
   if R = 1 { result: 6 } else { result: R }
}

output [reroll 1d6 under 1]+[reroll 1d6 under 2] named "reroll 1s"
output [crest 1d6 swap 1]+[crest 1d6 swap 1] named "crest"
output 2d6+1 named "+1 to cast"
output 2d6 named "baseline casting"

function: roll X:n empower Y:n {
  result: [roll X empower Y roll d6 total 0]
}

function: roll X:n empower Y:n roll ROLL:n total SUM:n {
  if X = 0        { result: SUM }
  if ROLL = 6     { result: [roll X-1 empower Y roll d6 total SUM+1] }
  if ROLL = {4,5} & Y > 0 { 
    result: [roll X-1 empower Y-1 roll d6 total SUM+1]
  }
  result: [roll X-1 empower Y roll d6 total SUM]
}


output [roll 2 empower 1]

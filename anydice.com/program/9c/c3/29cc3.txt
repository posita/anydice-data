
function: reroll highest of SEQUENCE:s as REROLL:d if total greater than MAX:n {
  HIGHEST: {1} @ [sort SEQUENCE]
  SUM: SEQUENCE + 0
  if SUM <= MAX {
     result: SUM
  } else {
     REMAINING: {2..#SEQUENCE} @ SEQUENCE \ everything but highest \
     result: REMAINING + REROLL
  }
}
output [reroll highest of 2d10 as d10 if total greater than 5] named "TN 5, reroll"
output [reroll highest of 2d10 as d10 if total greater than 10] named "TN 10, reroll"
output [reroll highest of 2d10 as d10 if total greater than 15] named "TN 15, reroll"

function: multiattack X:n times AC:n with BONUS:n deal DAMAGE:d extra CRITBONUSDAMAGE:d with critrange CRITRANGE:n{
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: [d20 attack AC with BONUS deal DAMAGE extra CRITBONUSDAMAGE with critrange CRITRANGE]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

function: ROLL:n attack AC:n with BONUS:n deal DAMAGE:d extra CRITBONUSDAMAGE:d with critrange CRITRANGE:n {
   if(ROLL >= CRITRANGE){ \crit and autohit\
      result: DAMAGE + CRITBONUSDAMAGE 
   }
   if(ROLL = 1) { result:0 } \nat 1 -> miss\
   if(ROLL + BONUS < AC) { result: 0 } \miss\
   result: DAMAGE \hit\
}
\This fixes ROLL to a value and uses it each time - should be the same as *2? \
function: ROLL:n multiattackdirect X:n times AC:n with BONUS:n deal DAMAGE:d extra CRITBONUSDAMAGE:d with critrange CRITRANGE:n{
  TOTAL: 0
  loop N over {1..X} {
     if(ROLL >= CRITRANGE){ \crit and autohit\
        TOTAL: TOTAL + DAMAGE + CRITBONUSDAMAGE 
     }
     if(ROLL = 1) { result:0 } \nat 1 -> miss\
     if(ROLL + BONUS < AC) { result: 0 } \miss\
     TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

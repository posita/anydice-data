
function: target TARGET:n explode EXPLODE:n VALUE:n REROLLS:n {
	if VALUE = EXPLODE {
		result: 1 + [target TARGET explode EXPLODE BASEDIE REROLLS]
	}
	if VALUE  >= TARGET {
		result: 1
	}
	if REROLLS > 0 {
		REROLLS: REROLLS - 1
		result: [target TARGET explode EXPLODE BASEDIE REROLLS]
	}
	result: 0
}

loop C over {1..5}{
  loop R over {1..5}{
    if R <= C {
	DIE: [target d8 explode 8 BASEDIE R]
	output CdDIEA named "[C] v8 R[R]"
    }
  }
}


function: every N:n th number from MIN:n to MAX:n {
  if MAX < MIN {
    result: [reverse [every Nth number from MAX to MIN alt]]
  }

  COUNT: (MAX - MIN)/N + 1
  VALUES: {}
  loop I over {0..COUNT - 1} {
    VALUES: {VALUES, MIN + N*I}
  }
  result: VALUES
}

output [every 2th number from 3 to 17]

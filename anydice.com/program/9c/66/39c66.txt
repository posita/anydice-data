
function: every N:n th number from MIN:n to MAX:n {
  if MAX < MIN {
    result: [reverse [every Nth number from MAX to MIN alt]]
  }

  COUNT: (MAX - MIN)/N + 1
  VALUES: {}
  loop I over {0..COUNT - 1} {
    VALUES: {VALUES, MIN + N*I}
  }
  result: VALUES
}

function: interval (FIRST:n, SECOND:n, LAST:n) {
  N: SECOND - FIRST
  MIN: FIRST
  MAX: LAST
}

loop SIDES over [every 2th number from 2 to 6] {
  loop ROLLS over {1..3} {
    output (ROLLS d SIDES) named "[ROLLS]d[SIDES]"
  }
}

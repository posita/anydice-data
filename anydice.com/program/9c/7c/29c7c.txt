function: attack A:n D:n {
 if A < 10 { result: -1 }
 if A > 0 { result: D+((A+5)/5) }
 result: 0
}

set "maximum function depth" to 3

loop N over {-20..20} {
  output [attack N+([explode d10] - [explode d10]) 10] - ([attack (N-3)+([explode d10] - [explode d10]) 10]+[attack (N-3)+([explode d10] - [explode d10]) 10]) named "[N]"
}

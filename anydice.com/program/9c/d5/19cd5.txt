function: damage DAMAGE:d rolling HIT:n with modifier MODIFIER:n to hit crit on CRIT:n dealing CRIT_DAMAGE:d against ac TO_HIT:n {
  if HIT >= CRIT { result: CRIT_DAMAGE}
  if HIT+MODIFIER < TO_HIT { result: 0 }
  result: DAMAGE
}

function: multiattack COUNT:n times dealing DAMAGE:n damage {
  result: COUNT * DAMAGE
}

loop AC over {16..22}{
  output [multiattack 2 times dealing [ damage 3d6+5 rolling 1d20 with modifier 8 to hit crit on 20 dealing 6d6+5 against ac AC ] damage] named "hex, ac [AC]"
  output [multiattack 2 times dealing [ damage 2d6+5 rolling 1d20 with modifier 11 to hit crit on 19 dealing 4d6+5 against ac AC ] damage] named "curse, ac [AC]"
  output [multiattack 2 times dealing [ damage 3d6+5 rolling 1d20 with modifier 11 to hit crit on 19 dealing 6d6+5 against ac AC ] damage] named "both, ac [AC]"
}

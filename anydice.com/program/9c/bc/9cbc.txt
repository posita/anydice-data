function: do X:n this Y:n please Z:n con {
if (X=10|Y=10)&!(X=10&Y=10)
{
 if Z=1 {result: X+Y-10}
}
result: X+Y
}

function: loaded X:n at high HIGH:n and low LOW:n{
if X<LOW{result: LOW}
if X>HIGH{result: HIGH}
result: X
}

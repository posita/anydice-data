function: roll ROLL:s vs TARGET:n {
  if ROLL+0 > TARGET+0 {
    result: 1 + !(ROLL < 6)   \ success \
  } else {
    result: -1 - !(ROLL > 1)  \ failure \
  }
}

loop CHALLENGE over {1..4} {
  loop SKILL over {1..4} {
    output [roll SKILL d6 vs CHALLENGE d6] named "normal [SKILL]v[CHALLENGE]"
    output [roll (SKILL+1) d6 vs CHALLENGE d6]-1 named "[SKILL]v[CHALLENGE] with spent point"
  }
}

function: season STAT:n DIF:n {
if DIF > 0 {
result: [highest 1 of (DIF+1)dSTAT]
}
if DIF < 0 {
result: [lowest 1 of (1-DIF)dSTAT]
}
result:dSTAT
}
function: sroll STAT:n DIF:n {
result: [highest of [season STAT DIF]-2 and 0]/2-1
}
loop DIF over {0,1,-1,2,-2} {
loop STAT over {4,6,8,10,12} {
output [sroll STAT DIF] named "Stat:[STAT] Dif:[DIF]"
}
}

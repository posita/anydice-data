function: poolexplode N:n on tn TN:n {
 if N = 1 { result: 1 + [poolexplode 1d6 on tn TN] }
 result: N <= TN
}

set "maximum function depth" to 5

loop N over {1..5} {
  loop PARTIAL over {0..2} {
    output Nd[poolexplode 1d6 on tn 3] + ((PARTIAL > 0)*[poolexplode 1d6 on tn PARTIAL]) named "[N].[PARTIAL]"
}}

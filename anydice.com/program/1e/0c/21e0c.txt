function: disad R:n again N:n {
 if N >= R { result: 1 + [disad R again d6] }
 if N <= 1 { result: -1}
 result: N >= 4
}

function: normal R:n again N:n {
 if N >= R { result: 1 + [normal R again d6] }
result: N >= 4
}

function: advant R:n again N:n {
 if N >= R { result: 1 + [normal R again d6] + [normal R again d6] + [normal R again d6]}
result: N >= 4
}

NWODDIS: [disad 6 again d6]
NWODNORM: [normal 6 again d6]
NWODADV: [advant 6 again d6]

loop N over {1..9} {
 output [lowest of 12 and NdNWODDIS] named "[N]d disadvantage"
 output [lowest of 12 and NdNWODNORM] named "[N]d normal"
 output [lowest of 12 and NdNWODADV] named "[N]d advantage"
}

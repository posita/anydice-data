function: disad R:n again N:n {
 if N >= R { result: 1 + [disad R again d6] }
 if N <= 1 { result: -1}
 result: N >= 4
}

function: normal R:n again N:n {
 if N >= R { result: 1 + [normal R again d6] }
result: N >= 4
}

function: advant R:n again N:n {
 if N >= R { result: 1 + [normal R again d6] + [normal R again d6] + [normal R again d6]}
result: N >= 4
}

DIS: [disad 6 again d6]
NORM: [normal 6 again d6]
ADV: [advant 6 again d6]

loop N over {1..9} {
 output [lowest of 12 and NdDIS] named "[N]d disadvantage"
 output [lowest of 12 and NdNORM] named "[N]d normal"
 output [lowest of 18 and NdADV] named "[N]d advantage"
}

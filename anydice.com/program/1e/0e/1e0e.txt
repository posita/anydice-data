function: dam DAM:d at AT:n plasma PLASMA:n master MASTER:n con CON:n prof PROF:n shield SHIELD:n av AV:n JAM:n rolled D:s {
 if JAM {
  result: 0
 }
 ROLL: 1@D + 2@D
 if (!PROF | CON < 10) & (1@D = 2@D & (1@D >= CON | (PLASMA & 1@D >= 8))) {
  result: -1
 }
 if ROLL > AT {
  result: 0
 }
 XS: AT - ROLL
 DAMAGE: DAM
 if SHIELD {
  DAMAGE: [maximum of DAM]
 }
 if MASTER {
  DAMAGE: DAMAGE + XS
 }
 if 1@D = 2@D & (PROF | 1@D = 1 | 1@D <= (CON - 10)) {
  DAMAGE: DAMAGE * 2
 }
 if !PLASMA {
  DAMAGE: DAMAGE - AV
  DAMAGE: [floor DAMAGE]
 }
 result: DAMAGE
}

function: floor DAM:n {
 if DAM < 0 {
  result: 0
 }
 result: DAM
}

function: dam DAM:d at AT:n plasma PLASMA:n master MASTER:n con CON:n prof PROF:n shield SHIELD:n av AV:n rate RATE:n {
 DAMAGE: 0
 JAM: 0
 loop X over {1..RATE} {
  TEMPDAMAGE: [dam DAM at AT plasma PLASMA master MASTER con CON prof PROF shield SHIELD av AV JAM rolled 2d10]
  DAMAGE: DAMAGE + TEMPDAMAGE
 }
 result: DAMAGE
}

function: roll up ROLL:s with bonus BONUS:n {
 M: [count -1 in ROLL]
 if BONUS < 11 {
  if BONUS > M {
   result: ROLL + M
  }
  result: ROLL + BONUS
 }
 
 if BONUS < 21 {
  BB: BONUS - 10
  if BB > M {
   result: ROLL + M * 2
  }
  result: ROLL + M + BB
 }

 else {
  Z: [count 0 in ROLL]
  BB: BONUS - 20
  if BB > Z {
   result: ROLL + M * 2 + Z
  }
  result: ROLL + M * 2 + BB
 }

}


loop P over {0..30} {
F: {-1,0,1}
B: P
output [roll up 10dF with bonus B] named "10dF with a bonus [B]"
}

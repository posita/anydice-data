function: explodeon DIE:d on TARGET:n {
  result: [explodehelper DIE on DIE target TARGET]
}

function: explodehelper R:n on DIE:d target TARGET:n {
  if R = TARGET {
    result: R + [explodehelper DIE on DIE target TARGET]
  } else {
    result: R
  }
}

function: explodepair BIG:d and SMALL:d {
  result: [pairbig BIG given BIG and SMALL]
}

function: pairbig B:n given BIG:d and SMALL:d {
  result: [pairsmall SMALL given B and SMALL]
}

function: pairsmall S:n given B:n and SMALL:d {
  if B = S {
    result: B + S + [explodeon SMALL on S]
  } else {
    result: B + S
  }
}

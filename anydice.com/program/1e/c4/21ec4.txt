function: interpret ROLL:s {
 if [count {1,2} in ROLL] { result: 1 } \ Failure, no change to canon \
 if [count 3 in ROLL] { result: 2 } \ Pass to another player \
 if [count 4 in ROLL] { result: 3 } \ Choice to continue \
 if [count 5 in ROLL] { result: 4 } \ Success with no penalty \

 else {result: 5} \ Success with a catch. \
}

loop P over {2..4} {
 output [interpret P d12] named "[P]d"
}

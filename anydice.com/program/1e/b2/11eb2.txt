function: gwf DIE:d {
  result: [reroll DIE as DIE if under 3]
}
function: reroll ROLL:n as DIE:d if under LIMIT:n {
  if ROLL < LIMIT { result: DIE }
  else { result: ROLL }
}

loop SIDES over {6} {
  GWF: [gwf dSIDES]
  output 2dGWF named "2d[SIDES] with old GWF"
}

output [highest of 2d6 and 2d6] named "2d6 with new GWF"

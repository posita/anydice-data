function: roll DICE:n BONUS {
  ROLL: DICE + BONUS
  if ROLL = 9   { result: 9}
  if ROLL = 10  { result: 9}
  if ROLL = 11  { result: 9}
  if ROLL = 12  { result: 12}
  if ROLL = 13  { result: 12}
  if ROLL = 14  { result: 12}
  if ROLL >= 15 { result: 15}
  result: 0
}

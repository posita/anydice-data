CRITSUCCESS: 21
CRITFAIL: 7

\4d6\

function: rpj R:n {
    if R >= CRITSUCCESS { result: CRITSUCCESS }
    else if R <= CRITFAIL { result: CRITFAIL }
    else { result: R }
}

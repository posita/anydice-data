function: count doubles in S:s {
 DOUBLES: 0
 loop P over {1..6} {
   DOUBLES: DOUBLES + ((S = P)/2)
 }
 result: DOUBLES
}

loop N over {2..4} {
 output [count doubles in Nd6] named "[N]d6"
}

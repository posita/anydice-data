

\ CONFIGURAÇÕES \
set "explode depth" to 5 \ Limita explosões infinitas para não travar o site \

\ DEFINIÇÃO DOS DADOS CUSTOMIZADOS \
\ Regra: O "1" vira "0". O resto se mantém. \
\ A função [explode] do AnyDice já explode o valor máximo automaticamente. \

D_NOVATO: [explode d{0, 2..12}]
D_AVENTUREIRO: [explode d{0, 2..8}]
D_VETERANO: [explode d{0, 2..6}]

\ MONTAGEM DAS RESERVAS (POOLS) \
NOVATO: 2d D_NOVATO
AVENTUREIRO: 3d D_AVENTUREIRO
VETERANO: 4d D_VETERANO

\ SAÍDA 1: VISÃO GERAL DAS PROBABILIDADES \
output NOVATO named "1. Novato (2d12)"
output AVENTUREIRO named "2. Aventureiro (3d8)"
output VETERANO named "3. Veterano (4d6)"

\ SAÍDA 2: CHANCE DE CRÍTICO (Soma > 24) \
output NOVATO > 24 named "Chance de Crítico (>24) - Novato"
output AVENTUREIRO > 24 named "Chance de Crítico (>24) - Aventureiro"
output VETERANO > 24 named "Chance de Crítico (>24) - Veterano"

\ SAÍDA 3: CHANCE DE FALHA CRÍTICA (Soma = 0) \
output NOVATO = 0 named "Chance de Falha Total (0) - Novato"
output AVENTUREIRO = 0 named "Chance de Falha Total (0) - Aventureiro"
output VETERANO = 0 named "Chance de Falha Total (0) - Veterano"

function: add up to BONUS:n for each VAL:n from ROLL:s {
 C: [count VAL in ROLL]
 if BONUS > C {
  result: ROLL + C
 }
 result: ROLL + BONUS
}


loop P over {0..10} {
F: {-1,0,1}
B: P
output [add up to B for each -1 from 10dF] named "10dF replacing [B] values of -1"
}

loop P over {0..10} {
ZZO: {0,0,1}
B: P
output [add up to B for each 0 from 10dZZO] named "10dZZO replacing [B] values of 0"
}

function: add A to B {
 result: A + B
}

function: booleanize NUMBER:n plus VALUE:n for AC:n do DMG plus BONUS{
 if NUMBER=20 { result: [add DMG to DMG]+BONUS }
 else { if (NUMBER+VALUE) > AC { result: DMG+BONUS }
        else { result: 0 }
      }
}

function: beatac AC with DICE add VALUE do DMG plus BONUS {
 Y: DICE
 result: [booleanize Y plus VALUE for AC do DMG plus BONUS]
}

function: dmgavg DMG plus BONUS against AC with DICE add VALUE {
 X: [beatac AC with DICE add VALUE do DMG plus BONUS]
 result : X
}

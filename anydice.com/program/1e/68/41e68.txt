ONE:{-1:9,0:1}
TWO:{-1:7,0:2,1:1}
THREE:{-1:5,0:2,1:3}
FOUR:{-1:3,0:2,1:5}
FIVE:{-1:1,0:2,1:7}

function: evaluate ROLL:s {
 HERO: ROLL = 1
 ALIVE: ROLL = 0
 DEAD: ROLL = -1
 if HERO >= DEAD { result : 0}
 if DEAD > HERO {result : DEAD - HERO}
}

output [evaluate 0dONE]+[evaluate 1dTWO]+[evaluate 1dTHREE]+[evaluate 0dFOUR]+[evaluate 2dFIVE]

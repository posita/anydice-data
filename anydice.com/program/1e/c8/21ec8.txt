function: interpret ROLL:s {
 if [count 1 in ROLL] { result: 1 } \ Failure, no change to canon \
 if [count 2 in ROLL] { result: 2 } \ Pass to another player \
 if [count 3 in ROLL] { result: 3 } \ Success with a catch \

 else {result: 4} \ Success with no penalty. \
}

loop P over {2..4} {
 output [interpret P d6] named "[P]d"
}

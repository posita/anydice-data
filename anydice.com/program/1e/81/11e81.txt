function: attack DICE:n vs AC:n {
if DICE=20 {result: DAMAGE+DAMAGE+STR}
if DICE+PROF+STR>=AC {result: DAMAGE+STR}
else {result: 0}
}

function: great weapon DICE:n {
if DICE={1,2} {result: 1d6}
else {result: DICE}
}

STR: 3
PROF: 2
DAMAGE: 2d[great weapon d6]

loop AC over {10..20} {
output 1d[highest 1 of 2d[attack 1d20 vs AC]]
}

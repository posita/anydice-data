STRENGTH: 2
BAB: 3
AC: 15
ATTACKS: 4
NONCRIT: 0
RESIST: 0
 HITMOD_ONE: BAB + STRENGTH + -4
    WEP_ONE: 1d8
CRITMOD_ONE: 2
 THREAT_ONE: 20
 HITMOD_TWO: BAB + STRENGTH + -4
    WEP_TWO: 1d8
CRITMOD_TWO: 2
 THREAT_TWO: 20
 HITMOD_TRE: BAB + STRENGTH + -4
    WEP_TRE: 1d8
CRITMOD_TRE: 2
 THREAT_TRE: 20
 HITMOD_FOR: BAB + STRENGTH/2 + -9
    WEP_FOR: 1d6
CRITMOD_FOR: 2
 THREAT_FOR: 20
 HITMOD_FIV: BAB + STRENGTH + -6
    WEP_FIV: 1d6
CRITMOD_FIV: 2
 THREAT_FIV: 20
 HITMOD_SIX: BAB + STRENGTH + -6
    WEP_SIX: 1d6
CRITMOD_SIX: 2
 THREAT_SIX: 20
REGINALD: 1
_ONE: 0
_TWO: 0
_TRE: 0
_FOR: 0
_FIV: 0
_SIX: 0

loop LOOP over {REGINALD..ATTACKS} {
 if LOOP=1 {
  HITMOD:   HITMOD_ONE
  WEP:         WEP_ONE 
  CRITMOD: CRITMOD_ONE 
  THREAT:   THREAT_ONE
 }
 if LOOP=2 {
  HITMOD:   HITMOD_TWO
  WEP:         WEP_TWO 
  CRITMOD: CRITMOD_TWO 
  THREAT:   THREAT_TWO
 }
 if LOOP=3 {
  HITMOD:   HITMOD_TRE
  WEP:         WEP_TRE 
  CRITMOD: CRITMOD_TRE 
  THREAT:   THREAT_TRE
 }
 if LOOP=4 {
  HITMOD:   HITMOD_FOR
  WEP:         WEP_FOR 
  CRITMOD: CRITMOD_FOR 
  THREAT:   THREAT_FOR
 }
 if LOOP=5 {
  HITMOD:   HITMOD_FIV
  WEP:         WEP_FIV 
  CRITMOD: CRITMOD_FIV 
  THREAT:   THREAT_FIV
 }
 if LOOP=6 {
  HITMOD:   HITMOD_SIX
  WEP:         WEP_SIX 
  CRITMOD: CRITMOD_SIX 
  THREAT:   THREAT_SIX
 }

NORMHIT: WEP + STRENGTH + NONCRIT - RESIST
CRITHIT: (CRITMOD d (WEP + STRENGTH)) + NONCRIT - RESIST

function: DTWENTY:n vs AC {
 if DTWENTY = 1 { result: 0 }
 if DTWENTY >= THREAT & DTWENTY + HITMOD >= AC {
  result: [threat d20]
 }
 if DTWENTY + HITMOD >= AC | DTWENTY = 20 { result: NORMHIT }
 result: 0
}

function: threat DTWENTY:n {
 if DTWENTY = 1 { result: NORMHIT }
 if DTWENTY + HITMOD >= AC | DTWENTY = 20 {
  result: CRITHIT
 }
 result: NORMHIT
}

REGINALD: REGINALD+1
 if REGINALD=1 { _ONE: [d20 vs AC] }
 if REGINALD=2 { _TWO: [d20 vs AC] }
 if REGINALD=3 { _TRE: [d20 vs AC] }
 if REGINALD=4 { _FOR: [d20 vs AC] }
 if REGINALD=5 { _FIV: [d20 vs AC] }
 if REGINALD=6 { _SIX: [d20 vs AC] }

}

output _ONE+_TWO+_TRE+_FOR+_FIV+_SIX

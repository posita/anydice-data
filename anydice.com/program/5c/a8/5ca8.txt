PSE: d{0,1,1,1,1,1}
PSB: d{0,0,0,0,1,1}
PSP: d{1,0,0,0,0,0}

X: 0

function: energy D:d {
N: 4
J: NdD
result: [energy J N helper]
}

function: energy I:n M:n helper {
if I < M {
result: I+(M-I)dD
}
else {
result: I
}
}

loop K over {1..10} {
X: X+[energy PSE]
}

output X
output [energy PSP]
output [energy PSB]

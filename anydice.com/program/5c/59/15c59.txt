I: 3 \maximum number of dice\
J: 4 \maximum damage modifier\

function: table A:n {
	if A=1{
		result: 0
	}
	else if A>=2&A<=5{
		result: 1
	}
	else if A>=6&A<=9{
		result: 2
	}
	else if A>=10{
		result: 3
	}
}

loop M over {1..I}{
	loop N over {1..5}{
		loop D over {1..J}{
			X: N*2+2
			output [table Md(X)]+[table Md(X)]+D named "[M]d[X]+[D]"
		}
	}
}

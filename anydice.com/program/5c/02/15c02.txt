function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 4
 if SUCCESSES > 0 { result: SUCCESSES + (ROLL = 6) }
 if [count 1 in ROLL] { result: -1 }
 result: 0
}

function: powertrain NUM:n {
 XP: 0
 loop N over {1..NUM}{
  ACTION: d[highest 1 of 2d[explode d6]] / 4
  XP: XP + ACTION
 }
 result: XP
}

output [powertrain 10]

function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d10] }
 result: N >= 8
}

function: calcdamage DAMAGE:n plus HITS:n {
 if HITS < 1 { result: 0}
 result: DAMAGE + HITS
}

N: 3 \Anzahl\
A: 10 \Again\
W: 2 \Weapon damage\


H: Nd[nwod A again d10]

\output [lowest of 10 and H]\

D: [calcdamage W plus H]

\output D\

O: [lowest of (N + W + 2) and D]

output O

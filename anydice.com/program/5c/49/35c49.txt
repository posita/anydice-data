function: NUM:n DIE:d p {
 SUM: 0
 loop I over {1..NUM} {
  SUM: SUM + [DIE p]
 }
 result: SUM
}

function: DIE:d p {
 MAX: [maximum of DIE]
 result: [penetrate DIE helper]
}

function: penetrate N:n helper {
 if MAX = 100 {
  if N = MAX {
   result: N - 1 + [1d20 p]
  }
 }
 else if MAX = 20 {
  if N = MAX {
   result: N - 1 + [1d6 p]
  }
 }
 else if N = MAX {
  result: N - 1 + [penetrate DIE helper]
 }
 result: N
}

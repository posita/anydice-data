  loop N over {1..PROF}{
    PA: PA + [get prof adv N@P]
  }

  PT: 0
  loop N over {1..PROF}{

    BA: BA + [get bons adv N@B]
  }

  D: DIFFd8

    CA: CA + [get chal adv N@C]

    SS: SS + [get setb suc N@S]
  }

  SA: 0
  loop N over {1..SETB}{
    SA: SA + [get setb adv N@S]
  }
 
  TS: AS + PS + BS + DS + CS + SS
  TA: AA + PA + BA + DA + CA + SA

  RESULT: 0

  else if TYPE = 2

    RESULT: PT

    RESULT: CD
  }

  result: RESULT
}

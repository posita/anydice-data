function: from INIT:n rolling X:s {
 result: 10+2*(INIT + [count 1 in X])
}

function: trunc X:n to MAX:n{
 if X >= MAX { result: [trunc X to MAX] }
 else { result: X}
}

output [trunc [from -1 rolling 12d6] to 18] named "Reroll 18s"

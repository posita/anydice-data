function: X check A:s {
 if 3@A >= X { result: 4 }
 if 2@A >= X { result: 3 }
 if 1@A >= X { result: 1 }
 result: 0
}

function: X bonuscheck A:s {
 if 1+3@A >= X { result: 4 }
 if 1+2@A >= X { result: 3 }
 if 1+1@A >= X { result: 1 }
 result: 0
}


loop N over {1..13}{
 output [N check 3d12] named "[N] or more out of 3"
}

loop N over {1..13}{
 output [N check 4d12] named "[N] or more out of 4"
}

loop N over {1..13}{
 output [N bonuscheck 3d12] named "[N] or more with +1"
}

function: opposedhighest of A:n and B:n 

{

if A > B {result: -1 }
if A = B {result: 0}
if A < B {result: 1}
}
output [opposedhighest of 1d3+5d{0,1,2} and 1d3+5d{0,0,1}]
output [opposedhighest of 1d3+5d{0,0,1} and 1d3+5d{0,0,1}]

output [opposedhighest of 2d {0,1} and 2d {0,1}]
output [opposedhighest of 2@4d {0,1} and 2d {0,1}]
output [opposedhighest of 6d{0,1} and 2d{0,1}]
output [opposedhighest of 6@12d{0,1} and 2d{0,1}]


output [opposedhighest of 1@1d8 and 1@1d6]
output [opposedhighest of 1@2d8 and 1@2d6]

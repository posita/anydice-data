function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

H: 6  \ hit die \
L: 5  \ level \
A: H
B: [dH reroll {1} as dH]
C: d(H-1) + 1

output (L-1)dA + H named "hit points without reroll"
output (L-1)dB + H named "hit points reroll 1 once"
output (L-1)dC + H named "hit points reroll 1 always"

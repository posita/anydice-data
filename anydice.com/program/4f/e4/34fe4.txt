function: mydices ADV:n A:n B:n DIS:n D:n
{
    result: ADVd[count {13.. 20} in d20] + Ad[count {3, 4} in d4] + Bd[count {3.. 6, 6} in d6]
          - DISd[count {6, 7, 8} in d8] - Dd[count {2, 4, 6.. 12, 12} in d12]
}

function: mydicea ADV:n A:n B:n DIS:n D:n
{
    result: ADVd[count {1.. 13} in d20] + Ad[count 2 in d4] + Bd[count {1.. 3} in d6]
          - DISd[count {2.. 5} in d8] - Dd[count {1.. 6, 8, 10} in d12]
}

ADV: 0
A:   3
B:   3
DIS: 1
D:   2

output [mydices ADV A B DIS D] named "success"
output [mydicea ADV A B DIS D] named "advantage"

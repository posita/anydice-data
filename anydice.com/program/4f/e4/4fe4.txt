TARGET: 17

ADDDAMAGE: 1d6+2
CRITMULT: 2

CRITDAMAGE:(BASEDAMAGE*CRITMULT+ADDDAMAGE)
THREATNONCRITCHANCE: [count{[highest of MINCRIT and TARGET]..20} in 1d20]*[count{1..(TARGET-1)} in 1d20]
NONCRITCHANCE:[count{TARGET..MINCRIT-1} in 1d20]
NONCRITDAMAGE: BASEDAMAGE+ADDDAMAGE
DAMAGEODDS: (CRITCHANCE * CRITDAMAGE) + (NONCRITCHANCE * NONCRITDAMAGE)

output NONCRITCHANCE named "No Crit Threat"

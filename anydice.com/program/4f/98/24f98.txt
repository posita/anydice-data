function: fate C:n R:n {
 if C >= (10-R) { result: C + d10 }
 if C <= (1+R) { result: C - d10 }
 result: C
}

function: boon C:n R:n {
 if C >= (10-R) { result: C + d10 }
 if C = 1 { result: C - d10 }
 result: C
}

function: doon C:n R:n {
 if C = 10 { result: C + d10 }
 if C <= (1+R) { result: C - d10 }
 result: C
}

function: explode R:n D:n F:n{
 if F = 1 {
  if R = D { result: R + [explode d(D) D F] }  \esplode alla cifra più alta del dado\
  result: R
 }
 if F = 2 {
  if R >= D-1 { result: R + [explode d(D) D F] }  \esplode alle due cifre più alte del dado\
  result: R
 }
 if F = 3 {
  if R = D | R = 1 { result: R + [explode d(D) D F] }  \esplode alla cifra più alta e sul 1\
  result: R
 }
 if F = 4 {
  if R = D { 
   if D=4 { result: R + [explode d(D) D 1] }
   result: R + [explode d(D-2) (D-2) F] 
  }                                       \esplode alla cifra più alta e tira il precedente dado più basso, continua se arriva al d4\
  result: R
 }
 if F = 5 {
  if R >= (D-1) { 
   if D=4 { result: R + [explode d(D) D 2] }
   result: R + [explode d(D-2) (D-2) F] 
  }                                       \esplode alle due cifre più alte e tira il precedente dado più basso, continua se arriva al d4\
  result: R
 }
}

function: double R:n D:n {
 if F=1 {
  if R = D { result: R + d(D) }
  result: R
 }
 if F=2 {
  if R >= D-1 { result: R + d(D) }
  result: R
 }
 if F=3 {
  if R = D | R = 1 { result: R + d(D) }
  result: R
 }
}

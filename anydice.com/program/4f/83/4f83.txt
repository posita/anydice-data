function: ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= 19 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICAL }
  result: DAMAGE
 }
 if ROLL + ATTACK >= DEFENSE { result: DAMAGE }
 result: MISS
}

TOHIT: 12 \+7bab+6dex+1wf+1enhancement+1bracersFalcon-2rapid-2deadlyAim\
BOW: 1d8+8 \+3composite+4deadlyAim+1enhancement\

DEFENSE: d{18..22}

ATTACK: TOHIT
DAMAGE: BOW+BOW
CRITICAL: (BOW)*3+BOW
MISS: 0
PRIMO:[d20 vs DEFENSE]
\output PRIMO named "Primo Attacco"\

ATTACK: TOHIT
DAMAGE: BOW
CRITICAL: (BOW)*3

SECONDO: [d20 vs DEFENSE]
\output SECONDO named "Secondo Attacco"\

ATTACK: TOHIT-5
DAMAGE: BOW
CRITICAL: (BOW)*3

TERZO: [d20 vs DEFENSE]
\output TERZO named "Terzo Attacco"\

output PRIMO + SECONDO + TERZO named "Totale"

function: count with pair TARGETS:s in SEQUENCE:s {
 REVERSED: [reverse SEQUENCE]
 IDX: 1
 COUNT: 0
 loop P over {1..#TARGETS} {
  if P@TARGETS = SEQUENCE {
    COUNT: COUNT + (P@TARGETS = SEQUENCE)
  }if P@TARGETS > SEQUENCE {
    loop Q over {IDX..#REVERSED} {
       if (SEQUENCE + Q@REVERSED) > P@TARGETS {
         COUNT: COUNT + 1
       }
       IDX: IDX + 1
    }
  }else {
    COUNT: COUNT
  }
 }
 result: COUNT
}




output 3d6 named ""

AC: 16
DEX: 5
PROF: 3
SA: 3d6

function: first A:n second B:n weapon C:n weapon D:n {
  \ disabled, we try to only hit once \
  \if A + DEX + PROF >= AC & B + DEX + PROF >= AC {result: C + D + DEX + SA}\
  if A + DEX + PROF >= AC {result: C + DEX + SA}
  if B + DEX + PROF >= AC {result: D + SA}
  result: 0
}

output [first d20 second d20 weapon 1d4 weapon 1d4] named "two daggers"
output [first d20 second d20 weapon 1d6 weapon 1d4] named "short sword and dagger"
output [first d20 second d20 weapon 1d6 weapon 1d6] named "two short swords"
output [first d20 second d20 weapon (1d8-1) weapon 1d4] named "rapier (penalty) and dagger"

BONUS:0

function: advantager A:n and B:n{
 if [highest of A+BONUS and B+BONUS] = 20{
  result: 100
 }
 else if [highest of A+BONUS and B+BONUS] = 1{
  result: 0
 }
 else{
  result: [highest of A+BONUS and B+BONUS]
 }
}
function: disadvantager A:n and B:n{
 if [lowest of A+BONUS and B+BONUS] = 20{
  result: 100
 }
 else if [lowest of A+BONUS and B+BONUS] = 1{
  result: 0
 }
 else{
  result: [lowest of A+BONUS and B+BONUS]
 }
}

output [advantager d20 and d20] named "advantage"

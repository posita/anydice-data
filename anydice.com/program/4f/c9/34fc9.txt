set "explode depth" to 10

RANGE: 20
DLUCK: d{0..9}

function: clamp N:n between MIN:n and MAX:n {
  if N > MAX { result: MAX }
  if N < MIN { result: MIN }
  result: N
}

\ Opposed roll \
output [clamp [explode DLUCK]+DLUCK-[explode DLUCK]-DLUCK-d{0..1} between -RANGE and RANGE] named "z9+z9! vs"

\ Dynamic action \
output [clamp [explode DLUCK]+DLUCK-[explode DLUCK]-5 between -RANGE and RANGE] named "z9+z9! vs 5+z9!"

\ Static action \
output [clamp [explode DLUCK]+DLUCK-10 between -RANGE and RANGE] named "z9+z9! vs 10"

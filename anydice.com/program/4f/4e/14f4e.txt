DIE: 1d10-1           \ Rolling 100 is OK if you got a skill of 100. \

function: roll A:n B:n {
    if (A*10+B) < SKILLVALUE {
    if (B*10+A) < SKILLVALUE { result: +2 }
  }
    if (A*10+B) < SKILLVALUE {
    if (B*10+A) > SKILLVALUE { result: +1 }
  }
    if (A*10+B) > SKILLVALUE {
    if (B*10+A) < SKILLVALUE { result: -1 }
  }
  if (A*10+B) > SKILLVALUE {
    if (B*10+A) > SKILLVALUE { result: -2 }
  }
}

loop SKILLVALUE over {30,40,50,60} {
  output [roll DIE DIE] named "1d100 against skill value [SKILLVALUE]"
}

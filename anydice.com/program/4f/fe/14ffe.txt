
function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output 2d[reroll {1} on d6] named "rr1ere på 2 terninga"

output 3d[reroll {1} on d6] named "rr1ere på 3 terninga"

output 4d[reroll {1} on d6] named "rr1ere på 4 terninga"

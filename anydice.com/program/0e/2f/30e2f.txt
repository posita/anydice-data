\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 40-50\
ENEMYAC: d{40..50}

ATTACKMOD: 6
PROFICIENCY: 20
HOLY: 1d6
ELEMENTAL: 1d6
DMGBONUS: 6 + 3
ACCBONUS: 6
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 4d4 + DMGBONUS + HOLY + ELEMENTAL
CRITDAMAGE: DAMAGE * 2
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "+3 Holy Elemental Greater Striking Whip"

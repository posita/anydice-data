function: highest NUMBER:n of DICE:d {
 result: {1..NUMBER}@DICE+2
}
function: highest NUMBER:n of DICE:d and count VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: COUNT+{1..NUMBER}@SEQUENCE
}

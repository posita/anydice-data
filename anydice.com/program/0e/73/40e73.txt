function: score D:n X:n Y:n {
  if D = 1 {
    result: 2
  }
  if D <= X {
    result: 2
  }
  if D <= X+Y {
    result: 1
  }
  result: 0
}


output 2d[score d12 0 3] named "score 1"


function: advantage STAT:n SKILL:n {
if [lowest of d100 and d100] <= STAT + SKILL {
result: 1
}
result: 0
}

output 1d[advantage  20 30]

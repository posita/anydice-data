function: find modifier of CHAR:n {
 if CHAR < 3 { result: -2 }
 if CHAR = 3 { result: -1 }
 if CHAR = 4 { result: -1 }
 if CHAR = 5 { result: -1 }
 if CHAR = 6 { result: 0 }
 if CHAR = 7 { result: 0 }
 if CHAR = 8 { result: 0 }
 if CHAR = 9 { result: 1 }
 if CHAR = 10 { result: 1 }
 if CHAR = 11 { result: 1 }
 if CHAR > 11 { result: 2 }
}

function: sum of CHARA:n and CHARB:n maximum NUM:n {
 if CHARA+CHARB < NUM { result: CHARA+CHARB }
 else { result: NUM }
}

function: find stat loss of ROLL:n {
 if ROLL > 0  { result: 0 }
 else { result: [absolute ROLL]+1 }
}

\Assumption: The player used a 2d6 roll on END and a 3d6dl1 roll on SOC\
END: 2d6
ENDMOD: [find modifier of END]
SOC: [highest 2 of 3d6]
SOCMOD: [find modifier of SOC]
ENDSOCMOD: [sum of ENDMOD and SOCMOD maximum 2]
MITROLL: (2d6-8)+ENDSOCMOD
MITEFFECT: [highest of MITROLL and 0]

\output ENDMOD named "Endurance Modifier"\
\output SOCMOD named "Social Modifier"\
\output ENDSOCMOD named "Total of End. and Soc. modifiers, max. 2"\
\output MITEFFECT named "Mitigation Effect"\

loop TERMS over {4..8}{
 NEGTERMS: -TERMS
 TOTALMOD: [lowest of (NEGTERMS+MITEFFECT) and 0]
 \output TOTALMOD named "Total Modifier at [TERMS]th Term"\
 AGEROLL: [lowest of (2d6+NEGTERMS) and 1]
 MITAGEROLL: [lowest of (2d6+TOTALMOD) and 1]
 \output AGEROLL named "Vanilla Aging Roll at [TERMS]th Term"\
 \output MITAGEROLL named "Mitigated Aging Roll at [TERMS]th Term"\
 STATLOSS: [find stat loss of AGEROLL]
 MITSTATLOSS: [find stat loss of MITAGEROLL]
 output STATLOSS named "Vanilla Stat Loss at [TERMS]th Term"
 output MITSTATLOSS named "Mitigated Stat Loss at [TERMS]th Term"
}

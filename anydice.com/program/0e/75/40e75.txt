\ 1) Define the custom die \
FATE: d{-1, -1, 0, 0, 1, 1}

\ This function applies specific thresholds for each reroll 'charge' \
function: evaluate ROLL:n rerolls R:n {
  if R = 2 {
    if ROLL < 2 {
      \ Condition 1: Use first reroll if total is less than 2 \
      result: [evaluate 4dFATE rerolls 1]
    }
  }
  if R = 1 {
    if ROLL < 1 {
      \ Condition 2: Use final reroll if total is less than 1 \
      result: [evaluate 4dFATE rerolls 0]
    }
  }
  \ If no conditions are met or R=0, return the final result \
  result: ROLL
}

\ Output the result \
output [evaluate 4dFATE rerolls 2] named "4dFATE (R1 < 2, R2 < 1)"
output 4dFATE named "Standard 4dFATE"

MISS: 0
HIT: 1
CRIT: 2
ATK: 8
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result:CRIT }
 if ROLL + ATK >= DEFENSE { result: HIT }
 result: MISS
}


DMGHIT: 1d8+1d4+5
DMGCRIT:2d8+2d4+5

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRIT {
  result: DMGCRIT
 }
 if ATTACKRESULT = HIT {
  result: DMGHIT
 }
 result: 0
}

output [damage for [attack d20 vs d{0..20}]]

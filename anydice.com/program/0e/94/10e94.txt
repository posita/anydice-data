function: rerollinner CURRENT:s from DICE:d CONDITION:s ATTEMPT:n {
  if ATTEMPT > #CONDITION {
    result: CURRENT
  }
  else if CURRENT <= ATTEMPT@CONDITION {
    result: [rerollinner 1dDICE from DICE CONDITION (ATTEMPT+1)]
  }
  else {
    result: CURRENT
  }
}

function: reroll DICE:d CONDITION:s{
  result: [rerollinner 1dDICE from DICE CONDITION 1]
}


output [reroll d6 {0}]

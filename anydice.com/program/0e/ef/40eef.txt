MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n plus MOD:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + MOD >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + MOD >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRIT_DMG
 }
 if ATTACKRESULT = HIT {
  result: HIT_DMG
 }
 result: MISS_DMG
}

function: attack {
 result: [damage for [attack d20 plus ATK_MOD vs AC]]
}

function: save {
 result: [damage for [attack d20 plus ST_MOD vs DC]]
}

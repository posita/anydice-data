set "position order" to "lowest first"
F: {-1,0,1}

function: pyramid P:n {
 R: {}
 loop N over {1..P} {
  R: {R, N:(P+1-N)}
 }
 result: R
}

function: chars {
 R: {}
 loop M over {-4..4} {
  loop N over {-M..4} {
   loop O over {-M..-N+1} {
    R: {R,O}
   }
  }
 }
 result: R
}

output d[chars] + d[pyramid 3] + 4dF named "Faction"
output d[pyramid 5] + 4dF named "Character"
output d[pyramid 4] + 4dF named "Extra"
output d[pyramid 3] + 4dF named "Goon"

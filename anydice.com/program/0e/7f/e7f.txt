function: ROLL:s reroll lowest {
 result: {1..(#ROLL - 1)}@ROLL + d6
}

function: ROLL:s reroll lowest and keep {
 result: [highest of ROLL and {1..(#ROLL - 1)}@ROLL + d6]
}

loop N over {1..3} {
 output [highest N of (N + 1)d6] named "[N] highest -1"
 output [Nd6 reroll lowest and keep] named "[N] reroll and keep"
 output [Nd6 reroll lowest] named "[N] reroll"
}

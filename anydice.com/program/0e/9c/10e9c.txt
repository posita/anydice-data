function: reroll CURRENT:n DICE:d CONDITION:s ATTEMPT:n {
  if ATTEMPT > #CONDITION | CURRENT > ATTEMPT@CONDITION {
    result: CURRENT
  }
  result: [reroll 1dDICE DICE CONDITION (ATTEMPT+1)]
}

function: reroll DICE:d CONDITION:s{
  result: [reroll 1dDICE DICE CONDITION 1]
}

MILEE:d d{0:3,1,2,3}
loop REROLLVALUE over {0..3} {
  output [reroll MILEE {REROLLVALUE}] named "reroll [REROLLVALUE]"
}

function: basic ROLLEDVALUE:n on MAGICNUMBER:n add LEVEL:n {
 if ROLLEDVALUE = MAGICNUMBER { result: ROLLEDVALUE + LEVEL }
 result: ROLLEDVALUE
}

function: journeyman ROLLEDVALUE:n on MAGICNUMBERA:n MAGICNUMBERB:n add LEVEL:n {
 if ROLLEDVALUE = MAGICNUMBERA { result: ROLLEDVALUE + LEVEL }
 else { if ROLLEDVALUE = MAGICNUMBERB { result: ROLLEDVALUE + LEVEL }}
 result: ROLLEDVALUE
}

 else { if ROLLEDVALUE = MAGICNUMBERB { result: ROLLEDVALUE + LEVEL }
        else { if ROLLEDVALUE = MAGICNUMBERC { result: ROLLEDVALUE + LEVEL }}
      }
 result: ROLLEDVALUE
}

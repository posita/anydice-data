function: find stat loss of ROLL:n {
 if ROLL > 0  { result: 0 }
 else { result: [absolute ROLL]+1 }
}

loop TERMS over {4..8}{
 AGEROLL: 2d6-TERMS
 \output AGEROLL named "Aging Roll at [TERMS]th Term"\
 STATLOSS: [find stat loss of AGEROLL]
 \output STATLOSS named "Stat Loss at [TERMS]th Term"\
 if TERMS = 4 {
  SLFOUR: STATLOSS
  TOTALLOSS: SLFOUR
 }
 if TERMS = 5 {
  SLFIVE: STATLOSS
  TOTALLOSS: SLFOUR + SLFIVE
 }
 if TERMS = 6 {
  SLSIX: STATLOSS
  TOTALLOSS: SLFOUR + SLFIVE + SLSIX
 }
 if TERMS = 7 {
  SLSEVEN: STATLOSS
  TOTALLOSS: SLFOUR + SLFIVE + SLSIX + SLSEVEN
 }
 if TERMS = 8 {
  SLEIGHT: STATLOSS
  TOTALLOSS: SLFOUR + SLFIVE + SLSIX + SLSEVEN + SLEIGHT
 }
 output TOTALLOSS named "Total Stat Loss at [TERMS]th Term"
}

\Error Checking\
\output SLFOUR named "Stat Loss at Fourth Term"\
\output SLFIVE named "Stat Loss at Fifth Term"\
\output SLSIX named "Stat Loss at Sixth Term"\
\output SLSEVEN named "Stat Loss at Seventh Term"\
\output SLEIGHT named "Stat Loss at Eighth Term"\

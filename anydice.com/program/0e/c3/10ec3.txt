function: cstdodec S:n
{
if S=1 {result: 1}
if S=2 | S=3 {result: 2}
if S=4 | S=5 | S=6 {result: 3}
if S=7 | S=8 | S=9 {result: 4}
if S=10 | S=11 {result: 5}
if S=12 {result: 6}
}
output [cstdodec 1@1d12] named "1d6..12"
output [cstdodec 1@2d12] named "2d6..12"
output [cstdodec 1@3d12] named "3d6..12"

 else {
  if R = 3 {result: [higher 2d6 7]}
  else {
   if R = 4 {result: [higher 2d6 6]}
   else {
    if R = 5 {result: [higher 2d6 5]}
    else {
     if R = 6 {result: [higher 2d6 6]}
     else {
      if R = 7 {result: [higher 2d6 7]}
      else {
       if R = 8 {result: [higher 2d6 6]}
       else {
        if R = 9 {result: [higher 2d6 5]}
        else {
         if R = 10 {result: [higher 2d6 7]}
         else {result: [specialduty 1d6]}
        }
       }
      }
     }
    }
   }
  }
 }
}
function: specialduty R:n {
 if R = 1 {result: 1}
 else {
  if R = 2 {result: 1}
  else {
   if R = 3 {result: 1+[higher 1d6 4]}
   else {
    if R = 4 {result: [higher 1d6 5]+[higher 1d6 5]+[higher 1d6 5]+[higher 1d6 5]+[higher 1d6 5]+[higher 1d6 5]}
    else {
     if R = 5 {result: [higher 1d6 5]+[higher 1d6 5]+[higher 1d6 5]+[higher 1d6 5]}
     else {result: 1}
    }
   }
  }
 }
}
function: higher R:n T:n {
 if R < T {result: 0}
 else {result: 1}
}

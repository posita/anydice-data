function: find stat loss of ROLL:n {
 if ROLL > 0  { result: 0 }
 else { result: [absolute ROLL]+1 }
}

loop TERMS over {4..8}{
 AGEROLL: 2d6-TERMS
 \output AGEROLL named "Aging Roll at [TERMS]th Term"\
 STATLOSS: [find stat loss of AGEROLL]
 \output STATLOSS named "Stat Loss at [TERMS]th Term"\
 if TERMS = 4 {
  STATLOSSFOUR: STATLOSS
  TOTALLOSS: STATLOSSFOUR
 }
 if TERMS = 5 {
  STATLOSSFIVE: STATLOSS
  TOTALLOSS: STATLOSSFOUR + STATLOSSFIVE
 }
 if TERMS = 6 {
  STATLOSSSIX: STATLOSS
  TOTALLOSS: STATLOSSFOUR + STATLOSSFIVE + STATLOSSSIX
 }
 if TERMS = 7 {
  STATLOSSSEVEN: STATLOSS
  TOTALLOSS: STATLOSSFOUR + STATLOSSFIVE + STATLOSSSIX + STATLOSSSEVEN
 }
 if TERMS = 8 {
  STATLOSSEIGHT: STATLOSS
  TOTALLOSS: STATLOSSFOUR + STATLOSSFIVE + STATLOSSSIX + STATLOSSSEVEN + STATLOSS
 }
 output TOTALLOSS named "Total Stat Loss at [TERMS]th Term"
}

\Error Checking\
\output STATLOSSFOUR named "Stat Loss at Fourth Term"\
\output STATLOSSFIVE named "Stat Loss at Fifth Term"\
\output STATLOSSSIX named "Stat Loss at Sixth Term"\
\output STATLOSSSEVEN named "Stat Loss at Seventh Term"\
\output STATLOSSEIGHT named "Stat Loss at Eighth Term"\

TEN: {{-9..0},{2..9}:10,{11..20}}

AUTOF: 14
SINGLEF: 16
AUTODV: 17
SINGLEDV:15

function: automult N:n{
if AUTOF+N>AUTODV+3{result:4}
if AUTOF+N<=AUTODV{result:0}
result:AUTOF+N-AUTODV}

function: bodyshot N:n{if N+SINGLEF>SINGLEDV{result: 5d6}
result:0}

output [bodyshot dTEN]

output [automult dTEN]*2d6

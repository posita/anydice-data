\Average Traveller characteristics under various rolling methods\

function: reroll A:n if below B:n {
 if A > B-1 { result: A }
}

V: 2d6+2d6+2d6+2d6+2d6+2d6
X: [highest 2 of 3d6]
Y: X+X+X+2d6+2d6+2d6

\output 12d6 named "Vanilla Total"\
\output [reroll Y if below 42] named "House-Ruled Total"\
output V/6 named "Vanilla Characteristic"
output Y/6 named "Modified Vanilla Char."
output [reroll Y if below 42]/6 named "House-Ruled Char."
\output ([reroll Y if below 42]+X)/7 named "H-R Zhodani Char., Method 1"\
\output ([reroll Y if below 42]+(1d6+6))/7 named "H-R Zhodani Char., Method 2"\

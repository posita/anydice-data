\ sum lowest die plus all other dice that are below THRESHOLD \

THRESHOLD: 4

set "position order" to "lowest first"

function: sum ROLL:s {
 R: 1@ROLL
 loop P over {2..#ROLL} {
  if P@ROLL >= THRESHOLD { result: R }
  R: R + P@ROLL
 }
 result: R
}

loop N over {1..6} {
 output [sum Nd6] named "[N]d"
}

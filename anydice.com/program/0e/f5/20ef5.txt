LEVEL: 5
DC: 15
CRIMINAL: 1d4
REWARD: 100
STEALTH: 9
PERCEPTION: 9
TOOLS: 6

CS: (d20+1d4) + STEALTH + CRIMINAL >= DC
CP: ((d20+1d4) + PERCEPTION + CRIMINAL >= DC)
CT: ((d20+1d4) + TOOLS + CRIMINAL >= DC)

ESC: CS | CP | CT
PARTIAL: (CS & CP) | (CS & CT) | (CP & CT)
SUCCESS: CS & CP & CT  

output ESC named "escape"
output PARTIAL named "partial success"
output (CS & CP & CT) named "success loot"

output ESC + PARTIAL * (REWARD/2)*LEVEL + SUCCESS * REWARD*LEVEL named "expectation"

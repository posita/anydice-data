function: testdif A vs B {
result: A >= B
}
function: disroll plus A {
result: [lowest 1 of 2d20]+A
}
function: advroll plus A {
result: [highest 1 of 2d20]+A
}
function: avgroll plus A {
result: 2d20/2+A
}

output [disroll plus d{-5..5}]
output [avgroll plus d{-5..5}]
output [advroll plus d{-5..5}]

loop N over {1..20}{
output [testdif[avgroll plus d{-5..5}] vs N] named "[N]"
}

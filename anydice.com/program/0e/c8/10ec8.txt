function: csticos S:n
{
if S=1 {result: 1}
if S=2 | S=3 {result: 2}
if S=4 | S=5 {result: 3}
if S=6 | S=7 {result: 4}
if S=8 | S=9 | S=10 {result: 5}
if S=11 | S=12 | S=13 {result: 6}
if S=14 | S=15 {result: 7}
if S=16 | S=17 {result: 8}
if S=18 | S=19 {result: 9}
if S=20 {result: 10}
}
output [csticos 1@1d20] named "1d10..20"
output [csticos 1@2d20] named "2d10..20"
output [csticos 1@3d20] named "3d10..20"

function: tough DICE:n DEF PWR {
 ROLL: DICE + DEF

 CRIT: 0

 if DICE = 1 {
  CRIT: -1
 }
 if DICE = 20 {
  CRIT: 1
 }

 if ROLL >= 15 + PWR {
  result: 2 + CRIT
 }
 if ROLL >= 10 + PWR {
  result: 1 + CRIT
 }
 if ROLL >= 5 + PWR {
  result: -1 + CRIT
 }
 if ROLL >= 0 + PWR {
  result: -2 + CRIT
 }

  result: -2 + CRIT
}

output [tough 1d20 10 10] named "without improved resistance"

output [tough [highest 1 of 2d20] 10 10] named "with improved resistance"

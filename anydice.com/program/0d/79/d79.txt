function: above zero in RANGE:n {
  if RANGE <= 0 {
    result: 0
  } else {
    result: RANGE
  }
}

MAX:15
loop FOO over {MIN..MAX} {
  BAR: 0
  loop BIZ over {1,3,5} {
    output [above zero in ((3d6 + FOO) - (2d6 + BAR + BIZ))] named "Ranged attacker with [FOO] more Agility at range [BIZ]"
  }
}

\Depth 1\
function: A:n is even {
 result: (A / 2) = (A - A / 2)
}

function: S:s to number {
 R: 0
 X: 1
 loop N over {1..#S}
 {
  R: R + N@S * X
  X: X * 2
 }
 result: R
}

function: standard attack CLS:n STR:n WATK:n {
 result: WATK + STR / 2 + (CLS = 6 | CLS = 3 & WATK > 0)
}

function: master attack LV:n STR:n {
 result: LV * 2 + STR
}

function: defense CLS:n VIT:n ADEF:n {
 if (CLS = 3)
 {
  if (ADEF = 0)
  {
   result: VIT / 2
  } else {
   result: VIT / 4 + ADEF
  }
 } else {
  result: ADEF
 }
}

function: static hit percent ACC:n WACC:n {
 result: WACC + ACC
}

function: evasion percent EVA:n AGL:n WGT:n {
 result: EVA + 2 * AGL - WGT
}

function: hit count HITP:n CLS:n WATK:n {
 HITS: (1 + HITP / 32)
 if (CLS = 3 & WATK = 0)
 {
  HITS: HITS * 2
 }
 result: HITS
}

\TODO: Crit rate likely no longer depends on level! (But how to determine?)\
function: critical rate LV:n CLS:n WATK:n WCRIT:n {
 if (WATK = 0)
 {
  if (CLS = 3)
  {
   result: LV * 2
  } else {
   result: 0
  }
 } else {
  result: WCRIT
 }
}

function: static damage ATK:n DEF:n {
 result: [highest of (1d{ATK..(2*ATK)} - DEF) and 1]
}

function: base hit chance WEAK:n {
 result: 168 + WEAK * 40
}

\Depth 2\
function: A:n is odd {
 result: ![A is even]
}

function: A:s sand B:s {
 R: {}
 loop N over {1..[highest of #A and #B]}
 {
  R: {R, N@A & N@B}
 }
 result: R
}

function: accuracy C:s {
 result: [accuracy [class C] [level C]]
}

function: attack for LV:n CLS:n STR:n WATK:n {
 if (CLS = 3 & WATK = 0)
 {
  result: [master attack LV STR]
 } else {
  result: [standard attack CLS STR WATK]
 }
}

function: defense C:s DEFS:s {
 result: [defense [class C] [vitality C] DEFS]
}

function: evasion percent C:s WGTS:s {
 result: [evasion percent [evasion C] [agility C] WGTS]
}

function: critical rate C:s W:s {
 result: [critical rate [level C] [class C] [weapon attack W] [weapon crit W]]
}

function: static damage ATK:n CRIT:n DEF:n WEAK:n {
 if (WEAK)
 {
  ATK: ATK + 4
 }
 result: [static damage ATK ((1 - CRIT) * DEF)]
}

function: static hit chance HITP:n EVA:n WEAK:n {
 result: [base hit chance WEAK] + HITP - EVA
}

function: is hit ROLL:n HITC:n {
 result: ROLL = 0 | ROLL <= [lowest of 199 and HITC]
}

function: is crit ROLL:n CRITR:n {
 result: ROLL = 0 | ROLL <= [lowest of 199 and CRITR]
}

\Depth 3\
function: A:n to bitfield {
 R: {}
 loop N over {1..8}
 {
  R: {R, [A is odd]}
  A: A / 2
 }
 result: R
}

function: attack for C:s W:s {
 result: [attack for [level C] [class C] [strength C] [weapon attack W]]
}

function: defense C:s H:s G:s S:s A:s {
 result: [defense C {[armor defense H],[armor defense G],[armor defense S],[armor defense A]}]
}

function: hit percent C:s W:s {
 result: [static hit percent [accuracy C] [weapon accuracy W]]
}

function: evasion percent C:s H:s G:s S:s A:s {
 result: [evasion percent C {[armor weight H],[armor weight G],[armor weight S],[armor weight A]}]
}

function: static hit chance HITP:n EVA:n WEAK:n DISABLED:n {
 result: [static hit chance HITP ((1 - DISABLED) * EVA) WEAK]
}

function: final damage ROLL:n HITC:n CRITR:n ATK:n DEF:n WEAK:n {
 result: [is hit ROLL HITC] * [static damage ATK [is crit ROLL CRITR] DEF WEAK]
}

\Depth 4\
function: A:n band B:n {
 result: [[[A to bitfield] sand [B to bitfield]] to number]
}

function: hit count C:s W:s {
 result: [hit count [hit percent C W] [class C] [weapon attack W]]
}

function: pain chance C:s H:s G:s S:s A:s DISABLED:n E:s {
 result: [static hit chance [hit percent E] [evasion percent C H G S A] 0 DISABLED]
}

\Depth 5\
function: is weak WT:n ET:n WE:n EW:n {
 result: [WT band ET] | [WE band EW]
}

function: total pain HITS:n C:s H:s G:s S:s A:s DISABLED:n E:s {
 HITC: [pain chance C H G S A DISABLED E]
 CRITR: [critical rate E]
 ATK: [attack for E]
 DEF: [defense C H G S A]
 WEAK: 0
 R: 0
 loop N over {1..HITS}
 {
  R: R + [final damage 1d{0..200} HITC CRITR ATK DEF WEAK]
 }
 result: -R
}

\Depth 6\
function: is weak W:s E:s {
 result: [is weak [weapon type W] [enemy type E] [weapon element W] [enemy weakness E]]
}

function: expected pain C:s H:s G:s S:s A:s DISABLED:n E:s {
 result: [total pain [hit count E] C H G S A DISABLED E]
}

\Depth 7\
function: hit chance C:s W:s E:s DISABLED:n {
 result: [static hit chance [hit percent C W] [evasion percent E] [is weak W E] DISABLED]
}

function: complete pain C:s H:s G:s S:s A:s DISABLED:n ELIST:s {
 ECOUNT: #ELIST / 11
 R: 0
 loop N over {0..(ECOUNT - 1)}
 {
  I: (N * 11) + 1
  E: {}
  loop X over {I..(I + 10)}
  {
   E: {E, X@ELIST}
  }
  R: R + [expected pain C H G S A DISABLED E]
 }
 result: R / ECOUNT
}

\Depth 8\
function: total damage HITS:n C:s W:s E:s DISABLED:n {
 HITC: [hit chance C W E DISABLED]
 CRITR: [critical rate C W]
 ATK: [attack for C W]
 DEF: [defense E]
 WEAK: [is weak W E]
 R: 0
 loop N over {1..HITS}
 {
  R: R + [final damage 1d{0..200} HITC CRITR ATK DEF WEAK]
 }
 result: R
}

\Depth 9\
function: expected damage C:s W:s E:s DISABLED:n {
 result: [total damage [hit count C W] C W E DISABLED]
}

\Depth 10\
function: complete damage C:s W:s ELIST:s DISABLED:n {
 ECOUNT: #ELIST / 11
 R: 0
 loop N over {0..(ECOUNT - 1)}
 {
  I: (N * 11) + 1
  E: {}
  loop X over {I..(I + 10)}
  {
   E: {E, X@ELIST}
  }
  R: R + [expected damage C W E DISABLED]
 }
 result: R / ECOUNT
}

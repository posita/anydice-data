function: effect of ROLL:s with RANK:n {
  result: ROLL + RANK 
}

function: opposed EFFECTA:s vs EFFECTZ:s {
  if EFFECTA <= 0 & EFFECTZ <= 0 {
    result: d{}
  }
  
  MARGIN: EFFECTA - EFFECTZ
  
  if MARGIN = 0 {
    result: d{}
  }
  
  if MARGIN > 0 & MARGIN > EFFECTA {
    result: EFFECTA
  }
  
  if MARGIN < 0 & MARGIN < -EFFECTZ {
    result: -EFFECTZ
  }
  
  result: MARGIN
}

loop RANKC over {0..5} { 
  CHARACTER: [effect of 1d{0..9} - 1d8 with RANKC]
  loop RANKO over {0..5} {
    OPPONENT: [effect of 1d{0..9} - 1d8 with RANKO]
    output [opposed CHARACTER vs OPPONENT] named "Char [RANKC] vs Opp [RANKO]"
  }
}

function: attack D:n plus BONUS:n vs AC:n plus MOD:n {
    if D = 10 & AC != 10 { result: 1 } 
    else if D = 1 & AC != 1 { result: 0}
    else if D + BONUS > AC + MOD & AC != 10 { result: 1 } 
    else if AC = 1 { result:1}
    else if AC = 10 { result:0}
    else { result:0}
}

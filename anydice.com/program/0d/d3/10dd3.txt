function: X:n POOL:s successes {
 ONES: [count 1 in POOL]
 SIXES: [count 6 in POOL]
 if SIXES>0 {
  result: 1
 }
 else {
  if ONES>5 {
   ONES: 5
  }
   result: [(X-ONES-SIXES)d6 contains 6] 
 }
}  

function: X:n POOL:s failures {
 ONES: [count 1 in POOL]
 SIXES: [count 6 in POOL]
 if SIXES>0 {
  result: 0
 }
 else {
   if ONES>5 {
    ONES: 5
   }
   result: [(X-ONES)d6 contains 1] 
 }
}  

loop N over {1..7} {
 output [Nd6 contains 6] named "[N]d6 success"
 output [N Nd6 successes] named "[N]d6 pushed success"
 output [N Nd6 failures] named "[N]d6 pushed fails"
}

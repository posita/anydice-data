function: bushido N:n {
  ROLL: Nd6
  HIGH: 1@ROLL
  R: HIGH

  loop N over {2..#ROLL} {
    R: R + (N@ROLL > 1)
  }

  result: R
}

function: fiftyfifty N:n {
  ROLL: Nd6
  HIGH: 1@ROLL
  R: HIGH

  loop N over {2..#ROLL} {
    R: R + (N@ROLL > 3)
  }

  result: R
}

function: equal N:n {
  ROLL: Nd6
  HIGH: 1@ROLL
  R: HIGH

  loop N over {2..#ROLL} {
    R: R + (N@ROLL = HIGH)
  }

  result: R
}

loop N over {1..6} {
  output [bushido N] named "Bushido [N]"
  output [fiftyfifty N] named "50/50 [N]"
  output [equal N] named "Equal [N]"
}

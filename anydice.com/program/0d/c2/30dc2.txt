function: ThreeRoll A:n B:n C:n
{
    \Find and select the two highest dice\
    if A > B 
    {
         if B > C { result: A+B }
         else { result: A+C }
    }
    else 
    { 
         if A > C {result: A+B}
         else {result: B+C}
    }
}

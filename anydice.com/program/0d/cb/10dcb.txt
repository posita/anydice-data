function: ATKA:n DEFA:n {
	if ATKA >= ATKCRIT {result: DANOCRITA}
	if ATKA = 1 {result: 0}
	if ATKA + MODPRC + MODATR + ATKBONUSA + ATKBONUSG > DEFA {result: DANOA}
	else {result: 0}
}

function: DANOBONUSA:n {
	if ATKA + MODPRC + MODATR + ATKBONUSA + ATKBONUSG > DEFA {result: DANOBONUSA + MODPRC}
	else {result: 0}
}

function: VEZDANOA:n {
	if VEZDANOA > LIMVEZA {result: LIMVEZA}
	if MINVEZA >= VEZDANOA {result: MINVEZA}
	else {result: VEZDANOA}
}


function: ATKB:n DEFB:n {
	if ATKB >= ATKCRIT {result: DANOCRITB}
	if ATKB = 1 {result: 0}
	if ATKB + MODPRC + MODATR + ATKBONUSB + ATKBONUSG > DEFB {result: DANOB}
	else {result: 0}
}

function: DANOBONUSB:n {
        if ATKA + MODPRC + MODATR + ATKBONUSA + ATKBONUSG > DEFB {result: 0}
	if ATKB + MODPRC + MODATR + ATKBONUSB + ATKBONUSG > DEFB {result: DANOBONUSB + MODPRC}
	else {result: 0}
}

function: VEZDANOB:n {
	if VEZDANOB > LIMVEZB {result: LIMVEZB}
	if MINVEZB >= VEZDANOB {result: MINVEZB}
	else {result: VEZDANOB}
}

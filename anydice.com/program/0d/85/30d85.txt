function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

function: multiattack X:n Y:d {
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: Y
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

ATTACK: [attack [highest 1 of 2d20] plus 5 vs 14 for 3d6 crit 5d6 on 20]

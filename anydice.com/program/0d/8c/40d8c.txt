function: effect of ROLL:s with RANK:n and DIFFICULTY:n {
 RESULT: ROLL+RANK - DIFFICULTY
 if RESULT = 0 {result: 1d{-1,1}}
 result: RESULT 
}

loop RANK over {0..5} {
loop DIFFICULTY over {0..5} {
  output [effect of 1d{0..9}-1d8 with RANK and DIFFICULTY ] named "Rank [RANK] vs. Difficulty [DIFFICULTY]"
}
}

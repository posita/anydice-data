\Define Result States\
FAIL_FEAR: 1
FAIL_HOPE: 2
SUCC_FEAR: 3
SUCC_HOPE: 4
CRIT_SUCC: 5

function: daggerheart hope H:n fear F:n bonus B:n target T:n
{
 if H = F { result: CRIT_SUCC }
 else if F > H
 {
  if (H + F + B) < T { result: FAIL_FEAR }
  else { result: SUCC_FEAR }
 }
 else if H > F
 {
  if (H + F + B) < T { result: FAIL_HOPE }
  else { result: SUCC_HOPE }
 }
}

output [daggerheart hope d12 fear d12 bonus 3 target 13]

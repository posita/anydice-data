function: expfour N:n {
 if N = 4 { result: N + [expfour d4] }
 result: N
}
function: expsix N:n {
 if N = 6 { result: N + [expfour d4] }
 result: N
}
function: expeight N:n {
 if N = 8 { result: N + [expfour d4] }
 result: N
}

function: flail A:n p B:n m C:n {
  if A = B {
    if A = 4 { 
      result: A + B + [expfour d4]
    }
    result: A + B
  }
  result: A + B - C
}

set "maximum function depth" to 3
output 0  + [expsix d6]   - [expsix d6]  named "Normal"
output 1  + [expsix d6]   - [expsix d6]  named "Sword/ Stiletto (2G/ 3G)"
output 1  + [expeight d8] - [expsix d6]  named "Greatsword (3G)"
output 2  + [expsix d6]   - [expsix d6]  named "Battle Axe (4G)"
output d4 + [expsix d6]   - [expsix d6]  named "Bow (3G)"
output [flail d4 p d4     m [expsix d6]] named "Flail (5G)"

function: tough DICE:n DEF PWR AFF {
 ROLL: DICE + DEF

 if AFF = 0 {
  DMG: PWR + 5
 }
 else {
  DMG: PWR
 }

 CRIT: 0

 if DICE = 1 {
  CRIT: -1
 }
 if DICE = 20 {
  CRIT: 1
 }

 if ROLL >= 15 + DMG {
  result: 2 + CRIT
 }
 if ROLL >= 10 + DMG {
  result: 2 + CRIT
 }
 if ROLL >= 5 + DMG {
  result: 1 + CRIT
 }
 if ROLL >= 0 + DMG {
  result: -1 + CRIT
 }

  result: -2 + CRIT
}

output [tough 1d20 10 10 0] named "without hardened"

output [tough [highest 1 of 2d20] 10 10 0] named "with hardened"

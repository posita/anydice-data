function: A:n vs B:n {
 \if A = B{
  result: 0
 }
 if A > B{
  result: 1
 }
 else{
  result: -1
 }\
 result: A-B
}

SIZE: 6
MY_DICE: (SIZE+3)d6
WZMACNIACZ: {1..SIZE}@MY_DICE
TRANSMUTER: {2..SIZE+1}@MY_DICE
MANIFESTER: {3..SIZE+2}@MY_DICE

SIZEK: SIZE/2
MY_DICEK: (SIZEK+3)d10
WZMACNIACZK: {1..SIZEK}@MY_DICEK+SIZEK*2 
TRANSMUTERK: {2..SIZEK+1}@MY_DICEK+SIZEK*2
MANIFESTERK: {3..SIZEK+2}@MY_DICEK+SIZEK*2

output WZMACNIACZ named "Wzmacniacz k6"
output WZMACNIACZK named "Wzmackiacz k10"
output TRANSMUTER named "Transmuter k6"
output TRANSMUTERK named "Transmuter k10"
output MANIFESTER named "Manifester k6"
output MANIFESTERK named "Manifester k10"
\
output [WZMACNIACZ vs TRANSMUTER] named "Wzmacniacz vs Transmuter"
output [TRANSMUTER vs MANIFESTER] named "Transmuter vs Manifester"
output [WZMACNIACZ vs MANIFESTER] named "Wzmacniacz vs Manifester"

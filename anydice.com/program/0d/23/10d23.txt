function: A:n vs B:n {
 \if A = B{
  result: 0
 }
 if A > B{
  result: 1
 }
 else{
  result: -1
 }\
 result: A-B
}

SIZE: 3
MY_DICE: (SIZE+3)d6

WZMACNIACZ: {1..SIZE}@MY_DICE
TRANSMUTER: {2..SIZE+1}@MY_DICE
MANIFESTER: {3..SIZE+2}@MY_DICE

output [WZMACNIACZ vs TRANSMUTER] named "Wzmacniacz vs Transmuter"
output [TRANSMUTER vs MANIFESTER] named "Transmuter vs Manifester"
output [WZMACNIACZ vs MANIFESTER] named "Wzmacniacz vs Manifester"

MY_DICE: (SIZE)d10
WZMACNIACZ: MY_DICE+2*SIZE
TRANSMUTER: MY_DICE+SIZE
MANIFESTER: MY_DICE
output [WZMACNIACZ vs TRANSMUTER] named "Wzmacniacz vs Transmuter 2"
output [TRANSMUTER vs MANIFESTER] named "Transmuter vs Manifester 2"
output [WZMACNIACZ vs MANIFESTER] named "Wzmacniacz vs Manifester 2"

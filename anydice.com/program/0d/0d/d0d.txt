MISS: 0
HIT: 1

function: attack AROLL:n mat AMAT:n def ADEF:n {
 if AROLL = 2 { result: MISS }
 else 
   {
   if AROLL + AMAT >= ADEF { result: HIT }
   else 
   { result: MISS }
 }
}


function: damage DDICE:n DPOW:n DARM:n for ATTACKS:s {
 D: 0
 loop A over {1..#ATTACKS} {
  if A@ATTACKS = HIT { D: D + [highest of 0 and DDICEd6 + DPOW - DARM] }
 }
 result: D 
}


function: roll ATTACKS:n attdice ATDICE:n mat MAT:n pow POW:n dmgdice DMGDICE:n vs def DEF:n arm ARM:n
{
result: ATTACKSd[damage DMGDICE POW ARM for [attack ATDICEd6 mat MAT def DEF]]
}

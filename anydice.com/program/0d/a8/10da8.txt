\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

ENEMYAC: d{18..22}

ATTACKMOD: 4
DMGBONUS: 1
PROFICIENCY: 3
ACCBONUS: 1
CRITROLL: 20
ATTACKTIMES: 4
DAMAGE: 3d8 + DMGBONUS + ATTACKMOD
CRITDAMAGE: 6d8 + DMGBONUS + ATTACKMOD
output ATTACKTIMESd[attackroll [ highest 1 of 2d20] vs ENEMYAC] named "Grariar Battleaxe+1 ADV"

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

ENEMYAC: d{15}

ATTACKMOD: 3
DMGBONUS: 0
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 1d6 + DMGBONUS + ATTACKMOD
CRITDAMAGE: 2d6 + DMGBONUS + ATTACKMOD
output ATTACKTIMESd[attackroll [ highest 1 of 1d20] vs ENEMYAC] named "Dualwieldig"

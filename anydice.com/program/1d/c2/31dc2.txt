function: A in B chance {
  if A >= B { result: 1 }
  if A >= 20 {
    if B >= 20 {
      result: d{1:20, 0:(B-20)}
    }
    else {
      result: d{1:A, 0:(B-A)}
    }
  } else {
    result: d{1:A, 0:(B-A)}
  }
}

loop SKILL over {1..20}{
    output (1d2-1)+(1d2-1)+(1d2-1)+(1d2-1)+[SKILL in 5 chance]+[SKILL in 10 chance]+[SKILL in 15 chance]+[SKILL in 20 chance]+[SKILL in 75 chance] named "Skill [SKILL]"
}

function: HEAL:n treatwounds {
  if HEAL = 1 { result: -2d8 }
  else if HEAL >=6 & HEAL < 16 { result: 2d8 }
  else { result: 4d8 }
}
function: CHECKS:n continual {
  TOTAL: 0
  loop N over {1..CHECKS} {
    HEALING: [d20 treatwounds]
    TOTAL: TOTAL + HEALING
  }
  result: TOTAL
}
output [12 continual] named "treating wounds once"

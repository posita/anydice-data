\ pour avoir un resultat en nombre de succes dé blanc ; dé noir \
\ les dés blanc sont mulitplié par 100 pour transformé le vecture nb succes (blanc;noir)\
\ le parmaetre écalat est le nombre d'écalt car avoir éclat 1,2,3 a les même probablité de succes que 2,4,5 par exemple\


function: lance DE:s eclat EC:n {
R:0
loop N over {1..EC} {
 if (DE = N) >= 1 { R:R+1} 
}
result:R

}

function: lance DE:s compte ntime N:n sur de DT:n {
R:0
loop X over {1..DT} {
 if (DE=X )=N { R:R+1} 
}
result:R
}

\en cas de blanc et noir possibles les blancs sont préféré \
function: lance blanc DEBLANC:s noir DENOIR:s eclat EC:n {
R:0
loop X over {1..EC} { \ pour chaque valeur d'écalt on regarde si on trouve un dé avec cet éclat\
 if (DEBLANC=X ) >=1 {R:R+100} \ si on trouve au moins un dé blanc égale a éclat on le prend\
 else if (DENOIR=X )>=1 {R:R+1} \ sinon si on trouve au moins un dé noir égale a éclat on le prend\
}
result:R
}

\en cas de blanc et noir possibles le choix est montré dans les milier\
function: lance blanc DEBLANC:s noir DENOIR:s eclat EC:n avec choix {
R:0
loop X over {1..EC} { \ pour chaque valeur d'écalt on regarde si on trouve un dé avec cet éclat\
 if (DEBLANC=X ) >=1 {
     if (DENOIR=X )>=1 {R:R+1000} \ si on trouve au moins un dé blanc et au moins un dé noir on affiche le choix dans les millier\
     else {R:R+100} \ si on trouve au moins un dé blanc et pas de dé noir on affiche que le dé blanc\
    } 
 else if (DENOIR=X )>=1 {R:R+1} \ sinon si on trouve au moins un dé noir et pas de dé blanc égale a éclat on le prend\
}
result:R
}



DEBLANC:3 \ nombre de dé blanc\
DENOIR:3 \ nombre de dé noir\
DETYPE:6 \le type de dé\

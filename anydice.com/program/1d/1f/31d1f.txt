DC: 15
BONUS: 9

function: HEAL:n treatwounds {
  if HEAL = 1 | HEAL < (DC - BONUS - 10) { result: -2d8 }
  else if HEAL >= (DC - BONUS) & HEAL < (DC - BONUS + 10) { result: 2d8 }
  else { result: 4d8 }
}
function: CHECKS:n continual {
  TOTAL: 0
  loop N over {1..CHECKS} {
    HEALING: [d20 treatwounds]
    TOTAL: TOTAL + HEALING
  }
  result: TOTAL
}
output [12 continual] named "treating wounds continually"

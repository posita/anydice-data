ATT: 4   \Number of Attacks\
RDMG: 4  \Rage Damage\
STR: 5   \STR mod\
HITB: 11  \+ to Hit\
AC: 15   \Targets AC\
BCRIT: 4 \total dice on a crit\
MERCRIT: 15 \merciless assault crit lower bound\
FC: 80   \% chance for frenzy barb to get hit, in increments of 5, other increments are rounded down to an increment of 5\

CRIT: 0
DMG: 0

function: Z:n Y:n reckless
{
L: 0
   if Z > Y
   {
    L: Z
   }
    else
   {
    L: Y
   }
result: L
}

function: X:n attack
{
   if X >= 20
   {
    result: BCRITd8 + STR + RDMG
   }
    else if X + HITB >= AC
   {
    result: 1d8 + STR + RDMG
   }
   else
   {
    result: 0
   }
}

function: X:n mercattack
{
   if X >= MERCRIT
   {
    result: BCRITd8 + STR + RDMG
   }
    else if X + HITB >= AC
   {
    result: 1d8 + STR + RDMG
   }
   else
   {
    result: 0
   }
}

D: 0

   X: [1d20 1d20 reckless]
   D: D + [X attack]
   X: [1d20 1d20 reckless]
   D: D + [X attack]
   result: D
  }

}

}

\DMG: DMG + [1d20 frenzybarbattack]\

DMG: DMG + [1d20 frenzyrevenge]

function: X:n frenzybarbattack
{
   if X >= 20
   {
    result: BCRITd12 + + STR + RDMG +10
   }
    else if X + HITB -5 >= AC
   {
    result: 1d12 + + STR + RDMG + 10
   }
   else
   {
    result: 0
   }
}


function: X:n frenzyrevenge
{
Y: (FC / 5)
if X > 20 - Y {result: [1d20 frenzybarbattack]}
else {result: 0}
}

function: attaque en force avec attaque A:n puissance PUI:n furror FUR:n arme DGT:n vs def DEF:n res RES:n{
 if A+FUR>= 14+DEF { \ Le jet d'attaque A + le bonus de furror doit être supérieur a 14+ la def de l'adveraire\
   result: A+FUR-14-DEF+PUI+DGT-RES
 }else {
   result: 0
 }
}

function: attaque devastatrice  avec attaque A:n puissance PUI:n furror FUR:n arme DGT:n vs def DEF:n res RES:n{
 if A-PUI+FUR>= 14+DEF { \ Le jet d'attaque A + le bonus de furror doit être supérieur a 14+ la def de l'adveraire\
   result: A+FUR-14-DEF+3* PUI+DGT-RES
 }else {
   result: 0
 }
}

function: attaque en force croc du loup avec attaque A:n puissance PUI:n furror FUR:n niveau NIV:n arme DGT:n vs def DEF:n res RES:n{

   result: 0
 }
}

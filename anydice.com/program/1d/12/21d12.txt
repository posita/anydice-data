output [highest 4 of 8d{-1,0,1}] named "Lanzar más dados"
function: R:s with M:n {
 if M > 0 {
  REROLLS: [lowest of R = -1 and M]
  result: R + REROLLS + REROLLSdF
 }
 if M < 0 {
  REROLLS: [lowest of R = 1 and -M]
  result: R - REROLLS + REROLLSdF
 }
 result: R
}

F: d{-1..1}

loop M over {4..4} {
 output [4dF with M] named "Ventaja [M]"
}

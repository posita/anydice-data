\ Para as siglas ATR, PER e IMP, você deve atribuir o valor do dado que vai ser rodado como: 6 para d6, 8 para d8... \
ATR: 6
PER: 6
IMP: 0
VAN: 0 \ Caso tenha vantagem coloque 1; Caso tenha Desvantagem coloque -1; Caso não tenha nada coloque 0. Caso tenha vantagem no dado de atributo e perícia coloque 2 e -2 para desvantagem. Quando roda com vantagem, seu dado referente a vantagem é considerado d12 e d6 caso seja desvantagem  \








if VAN = 0 {
 output 1d(dATR >= 6) + 1d(dATR >= 10) + 1d(dPER >= 6) + 1d(dPER >= 10) + 1d(dIMP >= 6) + 1d(dIMP >= 10) named "d[ATR] + d[PER] + d[IMP] s/vantagem"
}

if VAN = 1 {
 output 1d(dATR >= 6) + 1d(dATR >= 10) + [highest 1 of 2d(d12 >= 6)] + [highest 1 of 2d(d12 >= 10)] + 1d(dIMP >= 6) + 1d(dIMP >= 10) named "d[ATR] + 2d12 + d[IMP] c/vantagem"
}

if VAN = -1 {
 output 1d(dATR >= 6) + 1d(dATR >= 10) + [lowest 1 of 2d(d6 >= 6)] + 1d(dIMP >= 6) + 1d(dIMP >= 10) named "d[ATR] + 2d6 + d[IMP] c/desvantagem"
}

if VAN = -2 {
 output [lowest 1 of 2d(d6 >= 6)] + [lowest 1 of 2d(d6 >= 6)] + 1d(dIMP >= 6) + 1d(dIMP >= 10) named "2d6 + 2d6 + d[IMP] c/desvantagem"
}

if VAN = 2 {
 output [highest 1 of 2d(d12 >= 6)] + [highest 1 of 2d(d12 >= 10)] + [highest 1 of 2d(d12 >= 6)] + [highest 1 of 2d(d12 >= 10)] + 1d(dIMP >= 6) + 1d(dIMP >= 10) named "2d12 + 2d12 + d[IMP] c/vantagem"
}

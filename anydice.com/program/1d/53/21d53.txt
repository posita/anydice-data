function: reroll DIE:d if NUM:n less than NUMBER:n {
  if NUM >= NUMBER {
    result: NUM
  }
  result: [reroll DIE if NUM less than NUMBER]
}

function: reroll DIE:d if less than NUMBER:n {
  result: [reroll DIE if DIE less than NUMBER]
}

function: point cost of NUMBER:n {
  result: NUMBER @ COSTS
}

COSTS: {-10000000, -10000000, -9, -6, -4, -2, -1, 0, 1, 2, 3, 4, 5, 7, 9, 12, 15, 19}
DEFAULTSTATROLL:[highest 3 of 4d6]
REROLLTHREEFOURFIVE: [reroll DEFAULTSTATROLL if less than 6]
SINGLESTAT: REROLLTHREEFOURFIVE
ABILITIES: 6 d SINGLESTAT
ABILITYCOSTS: 6 d [point cost of SINGLESTAT]

output DEFAULTSTATROLL
output REROLLTHREEFOURFIVE

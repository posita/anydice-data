function: strS tT rapid RAPID scatter SCATTER damage DAMAGE miss MISS missroll MISS_ROLL:s rapidroll RAPID_ROLL:s
{
if RAPID
{
RAPID: RAPID_ROLL
if RAPID_ROLL = 6
{X:3}
else if RAPID > 3
{X:2}
else {X:1}
}
else {X:1}

if SCATTER
{X:RAPID_ROLL
MISS:0
}

if (MISS=1 & (MISS_ROLL=1))
{
X:0
}

TO_WOUND:Xd6

if S >= T * 2
{result: DAMAGE*[count {2..6} in TO_WOUND]}

if S > T
{result: DAMAGE*[count {3..6} in TO_WOUND]}
if S = T
{result: DAMAGE*[count {4..6} in TO_WOUND]}
if S <= T / 2
{result: DAMAGE*[count {6} in TO_WOUND]}
if S < T
{result: DAMAGE*[count {5..6} in TO_WOUND]}
}

output [str2 t1 rapid 1 scatter 1 damage 1 miss 0 missroll d6 rapidroll d6] named "+Scattershot+ S2 T1 SV-"

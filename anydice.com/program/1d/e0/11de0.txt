\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 12-15\
ENEMYAC: d{12..15}

ATTACKMOD: 6
DMGBONUS: 0
PROFICIENCY: 0
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d6 + 3 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 1d6 + 12 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Kuo-Toa Archpriest Scepter"

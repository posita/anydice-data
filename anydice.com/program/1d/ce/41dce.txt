WEAPON_DIE: d6
WEAPON_DIE_COUNT: 2
DAMAGE_MODIFIER: 5
ARCANE_SHOT: 1
ARCANE_SHOT_DIE: d8
ARCANE_SHOT_DIE_COUNT: 2
SAVAGE_ATTACKER: 1
PIERCER: 1
PUNCTURE: 0

WEAPON_DAMAGE: WEAPON_DIE_COUNTd(WEAPON_DIE)
ARCANE_SHOT_DAMAGE: 0
if ARCANE_SHOT = 1 { ARCANE_SHOT_DAMAGE: ARCANE_SHOT_DIE_COUNTd(ARCANE_SHOT_DIE) }
else { ARCANE_SHOT_DIE_COUNT: 0 }

PIERCER_BONUS_DAMAGE: 0
if PIERCER = 1 { PIERCER_BONUS_DAMAGE: d(WEAPON_DIE) }

MIN_CRIT_DAMAGE: [maximum of WEAPON_DAMAGE] + [maximum of ARCANE_SHOT_DAMAGE] + DAMAGE_MODIFIER

BEST_WEAPON_DAMAGE: WEAPON_DAMAGE
if SAVAGE_ATTACKER = 1 { BEST_WEAPON_DAMAGE: [highest of WEAPON_DAMAGE and WEAPON_DAMAGE] }

PUNCTURE_DAMAGE: PIERCER_BONUS_DAMAGE
if PUNCTURE = 1 {
  LOWEST_DAMAGE_DIE: [lowest 1 of BEST_WEAPON_DAMAGE and ARCANE_SHOT_DAMAGE and PIERCER_BONUS_DAMAGE]
  output LOWEST_DAMAGE_DIE named "Lowest damage die"
  PUNCTURE_REROLL: {WEAPON_DIE:WEAPON_DIE_COUNT+1, ARCANE_SHOT_DIE:ARCANE_SHOT_DIE_COUNT}
  output PUNCTURE_REROLL named "Puncture Re-Roll"
  PUNCTURE_DAMAGE: dPUNCTURE_REROLL - dLOWEST_DAMAGE_DIE
  output PUNCTURE_DAMAGE named "Puncture Damage"
}  

VARIABLE_DAMAGE: BEST_WEAPON_DAMAGE + ARCANE_SHOT_DAMAGE + PUNCTURE_DAMAGE

output MIN_CRIT_DAMAGE named "Maxed Dice plus Modifier"
output VARIABLE_DAMAGE named "Total Variable Damage"
output MIN_CRIT_DAMAGE + VARIABLE_DAMAGE named "Critical Hit"

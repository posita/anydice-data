function: minusize SKILL {

if SKILL <= 0 {
DIFF: SKILL - 1
result: 1d10 + DIFF
}

result: [highest 1 of SKILL d10]
} 


loop S over {-2..5} {

output [minusize S] = 7 named "Dodge [S], Failure at 7"
output [minusize S] = 5 | [minusize S] = 6 named "Dodge [S], Failure at 5-6"
output [minusize S] <= 4 named "Dodge [S], Failure at 4 or less"

}

LEVEL: 4
CON: 15
ABILITY: 16
DAMAGE: 20

function: save ROLL:d with CON:n{
    DIFFICULTY: DAMAGE/2
    PROFICIENCY: ((LEVEL-1)/4) + 2

    if DIFFICULTY < 10 {
      DIFFICULTY: 10
    }
	
    result: ROLL + PROFICIENCY + ((ABILITY-10)/2) >= DIFFICULTY
   
}

output [save 1d20 with CON + 1] named "Resilient (Con)" 

output [save [highest 1 of 2d20] with CON] named "Warcaster"

function: evaluate LAW:s law and CHAOS:s chaos with target TARGET:n{
 SUCCESSES: (LAW >= TARGET) + (CHAOS >= TARGET) + (CHAOS = 6) - (CHAOS = 1)
 if SUCCESSES < 0 { result: 0}
 result: SUCCESSES
}

NDICE: 8
loop LAW over {1..NDICE} {
 CHAOS: NDICE - LAW
 output [evaluate LAW d6 law and CHAOS d6 chaos with target 3] named "[LAW]/[CHAOS]"
}

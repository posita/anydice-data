ATTRIBUTE: 5     \ Change this to any number between 1 and 5 \
FIRSTTRAIT: 5    \ Change this to any number between 1 and 5 \

function: roll TRAITONE:n and TRAITTWO:n {
  if TRAITONE = 3 { TRAITONE: 4 }
  if TRAITONE = 1 { TRAITONE: 2 }

  if TRAITONE = 5 {
    result: [count {1..ATTRIBUTE} in TRAITTWOd6]
  }
  else if TRAITONE = 4 {
    result: [count {1..ATTRIBUTE} in TRAITTWOd8]
  }
  else if TRAITONE = 2 {
    result: [count {1..ATTRIBUTE} in TRAITTWOd10]
  }
}

loop TWO over {1..5} {
  output [roll FIRSTTRAIT and TWO] named "Rolling trait [FIRSTTRAIT] (dice) and [TWO] (number of dice) against [ATTRIBUTE] (attribute)"
  if FIRSTTRAIT != TWO {
    output [roll TWO and FIRSTTRAIT] named "Rolling trait [TWO] (dice) and [FIRSTTRAIT] (number of dice) against [ATTRIBUTE] (attribute)"
  }
}

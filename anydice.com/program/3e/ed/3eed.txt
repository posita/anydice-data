set "explode depth" to 1

function: icon S {
 if S = 1 {result: [explode d6]}
 else {result: [drama d6]}
}

function: drama D:n {
 if D = 6 {result: 6 + [highest 1 of (S-1)d6]}
 result: [highest of D and [highest 1 of (S-1)d6]]
}

loop S over {1..6}{output [icon S] named "ICON Roll for Attribute [S]"}

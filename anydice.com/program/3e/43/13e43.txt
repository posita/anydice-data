DEXTERITY: 21
FIGHTING: 15
UNARMEDCOMBAT: 21
BODYARMOURPENALTY: 2

EV: 20

BASEHIT: 15+DEXTERITY/2

FIGHTINGHIT: BASEHIT + (1d(FIGHTING+2)-1)

CLAWSHIT: FIGHTINGHIT + 4

UNARMEDCOMBATHIT: CLAWSHIT + (1d(UNARMEDCOMBAT+2)-1)

ARMOURHIT: UNARMEDCOMBATHIT - 1dBODYARMOURPENALTY

function: roll D:n {
result: dD
}

TOHIT: [roll 1d(ARMOURHIT+1)] - 1

output ARMOURHIT named "ARMOURHIT"

output TOHIT named "To-Hit"

output TOHIT > 1dEV named "Hit Rate"

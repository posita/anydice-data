\ wizard level \
WL: 4

\ #power dice 1-12 \
NPD: 6

\ DO NOT CHANGE ANYTHING BELOW THIS LINE \

function: cast spell with wizard level WL:n using power dice PD:s {
 \ critical failure if a natural roll of 1 or 2, ie. 1 or 2 on 1d6 or double 1's on 2d6 \
 if #{PD} = 1 & PD <= 2 {
  result: 0
 }
 if #{PD} = 2 & {1,2}@PD = 2 {
  result: 0
 }
 result: PD + WL
}

\ DISTRIBUTION OF POWER DICE \
loop N over {1..NPD}{
 PD: Nd6
 output [cast spell with wizard level WL using power dice PD] named "Distribution of power dice: [N]d6+[WL]"
}


output 1d10 named "1d10"

function: remy ROLL:n {
 if ROLL = 1 {
  result: [remy 1d10]
 } else {
  result: ROLL
 }
}
output [remy 1d10] named "remy"

function: yan ROLLA:n ROLLB:n vs BET:n {
 if ROLLA = BET | ROLLB = BET {
  result: ROLLA+ROLLB
 } else {
   if ROLLA >= ROLLB {
    result: ROLLA
   } else {
    result: ROLLB
   }
  }
 }

loop BET over {1..10} {
 output [yan 1d10 1d10 vs BET] named "yan b[BET]"
}

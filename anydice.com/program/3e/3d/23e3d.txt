set "explode depth" to 15
set "maximum function depth" to 10

function: riskdie D:d keep over TN:n {
 SIZE: [maximum of D]
 if SIZE <= 4 {
  result: 1 + [explode D>TN]
 } else {
  result: 1 + [explode D>TN] + [riskdie d(SIZE-2) keep over TN]
 }
}

output [lowest of [riskdie d4 keep over 3] and 50] named "d4"
output [lowest of [riskdie d6 keep over 3] and 50] named "d6"

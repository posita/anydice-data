M:10

function: even DICE:n {
   result: (DICE/2)*2 = DICE
}

function: compare X:n and Y:n {
   if(X = Y) {result: -2}
   if([even X+Y]) {result:-1}
   if(X+Y = M+1) {result: 2}
   result: 1
}

function: full compare DICE:s {
  E: -3
  loop X over {1..#DICE-1} {
    loop Y over {X+1..#DICE} {
       E: [highest of E and [compare X@DICE and Y@DICE]]
    }
  }
   result: E
}

loop N over {2..4} {
   output [full compare NdM] named "Rolling [N] Dice" 
}

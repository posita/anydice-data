LEVEL: 4
CONSTITUTION: 15
DAMAGE: 40

function: save ROLL:d with CON:n{
    PROFICIENCY: ((LEVEL-1)/4) + 2
    DIFFICULTY: DAMAGE/2

    if DIFFICULTY < 10 {
      DIFFICULTY: 10
    }
	
    result: (ROLL + PROFICIENCY + ((CON-10)/2)) >= DIFFICULTY
   
}

output [save 1d20 with (CONSTITUTION + 1)] named "Resilient (Con)" 

output [save [highest 1 of 2d20]with CONSTITUTION] named "Warcaster"

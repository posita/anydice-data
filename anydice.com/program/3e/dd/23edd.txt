SA: 4
SB: 4
SC: 3
SD: 3
SE: 2
SF: 2
S: {SA,SB,SC,SD,SE,SF}

function: compare DICE:s with TN:n {
R: 0
loop X over {1..#DICE} {
   D: [absolute X@DICE-TN]
   D: [lowest of D and 6-D]
   R: [highest of R and 3-D]
}
result: R
}

loop N over {1..6} {
loop TN over {1..6} {
   output [compare NdS with TN] named "[N]d{[SA],[SB],[SC],[SD],[SE],[SF]} TN:[TN]"
}
}

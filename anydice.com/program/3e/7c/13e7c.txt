AC: 16
STR: 5
RAGE: 3
BRUTAL_CRIT: 2
CRIT_SCHWELLE: 19


function: gorehowl ROLL:n BONUS:n WAFFEN_DMG:d {
  SINGLE: {1@WAFFEN_DMG}
  NUM: #WAFFEN_DMG
  if ROLL >= CRIT_SCHWELLE {
    result: (2 * NUM + BRUTAL_CRIT)d(SINGLE) + STR + RAGE + BONUS
  } else if ROLL + STR + BONUS >= AC {
    result: (NUM)d(SINGLE) + STR + RAGE + BONUS
  } else { 
    result: 0
  }
}

ADVANTAGE: 1@2d20

output 2d[gorehowl ADVANTAGE 1 2d12] named "Gorehowl +1, 1d12, SV Frenzy #2"
output 3d[gorehowl ADVANTAGE 1 1d12] named "Gorehowl +1, 1d12, PHB Frenzy"

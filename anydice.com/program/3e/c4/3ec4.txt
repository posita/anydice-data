 if (THIS < THAT) { result: -1 }
 if (THIS = THAT) { result: 0 }
 result: 1}

 result: [crit helper DICE]}

function: crit helper RES:n {
 if RES = 1 { result: [implode DICE 0] }
 if RES = 6 { result: [explode DICE 0] }
 result: RES
}

function: implode IR:n C:n {
 if IR = MIN & C < 6{ result: [implode DICE C+1] }
 result: RES - C
}

function: explode IR:n C:n {
 if IR = BON & C < 6{ result: [explode DICE C+1] }
 result: RES + C
}

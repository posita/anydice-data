function: weapondamage ROLL:n vs AC:n {
if ROLL = 20 { result: (2dWD+INT)}
if ROLL < AC { result: 0 }
if ROLL >= AC { result: (1dWD+INT)}
}

function: cantripdamage ROLL:n vs AC:n {
if ROLL = 20 { result: ((2*BD)d10)}
if ROLL < AC { result: 0 }
if ROLL >= AC { result: (BDd10)}
}

INT: 5
CHA: 5
PB: 3
BD: 2
WD: 12
AC: 2d6+10
output [cantripdamage (1d20+CHA+PB) vs AC]
output [weapondamage (1d20+INT+PB) vs AC] + [weapondamage (1d20+INT+PB) vs AC]

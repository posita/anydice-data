function: take NTH:n normal NSEQ:s crit CSEQ:s {
  COMB: [sort {NSEQ, CSEQ}]
  if NTH@COMB = CSEQ {
    result: (NTH-1)@COMB
  }
  result: NTH@COMB
}

output 3@5d10
output [take 3 normal 4d10 crit 1d10]
output [take 3 normal 3d10 crit 2d10]
output [take 3 normal 2d10 crit 3d10]
output [take 3 normal 1d10 crit 4d10]
output 2@5d10

LEVEL:8
CONSTITUTION: 15
DAMAGE: 28


PROFICIENCY: ((LEVEL-1)/4) + 2
function: save ROLL:d with con CON:n and proficiency PROFICIENCY:n {
    DIFFICULTY: DAMAGE/2
    CONSTITUTION_BONUS: (CON-10)/2

    if DIFFICULTY < 10 {
      DIFFICULTY: 10
    }
	
    result: ROLL + PROFICIENCY + CONSTITUTION_BONUS >= DIFFICULTY
}

output [save 1d20 with con (CONSTITUTION + 1) and proficiency PROFICIENCY] named "Resilient (Con)" 
output [save [highest 1 of 2d20] with con CONSTITUTION and proficiency 0] named "Warcaster"
output [save [highest 1 of 2d20] with con CONSTITUTION + 1 and proficiency PROFICIENCY] named "Warcaster + Resilient (Con)"
output [save [highest 1 of 3d20] with con CONSTITUTION + 1 and proficiency PROFICIENCY] named "Warcaster + Resilient (Con) + Lucky"

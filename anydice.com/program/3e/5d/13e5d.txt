STAFFCRITICALDAMAGEPOMMEL: 2d4 + 5 + 3
MISSDAMAGE: 0
AC: 20

function: rolling ROLL:n to hit vs ac of DEFENSE:n with attack of ATTACK:n {
 if ROLL = 1 {result: MISS}
 if ROLL = 20 {result: CRITICALHIT}
 if ROLL + ATTACK>= DEFENSE {result: HIT}
 result: MISS
}

function: damage for staff main ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: STAFFCRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: STAFFHITDAMAGE
 }
 result: MISSDAMAGE
}

function: damage for staff pommel ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: STAFFCRITICALDAMAGEPOMMEL
 }
 if ATTACKRESULT = HIT {
  result: STAFFHITDAMAGEPOMMEL
 }
 result: MISSDAMAGE
}

function: staff damage for ATTACKRESULTMAIN:n and ATTACKRESULTPOMMEL:n {
  TOTAL: 0
  if ATTACKRESULTMAIN = CRITICALHIT {
    TOTAL: TOTAL + STAFFCRITICALDAMAGE
  }
  if ATTACKRESULTMAIN = HIT {
    TOTAL: TOTAL + STAFFHITDAMAGE
  }
  else {TOTAL: TOTAL + MISSDAMAGE}
  if ATTACKRESULTPOMMEL = CRITICALHIT {
    TOTAL: TOTAL + STAFFCRITICALDAMAGEPOMMEL
  }
  if ATTACKRESULTPOMMEL = HIT {
    TOTAL: TOTAL + STAFFHITDAMAGEPOMMEL
  }
  else {TOTAL: TOTAL + MISSDAMAGE}
  result: TOTAL
}

function: multiattack NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for staff main [rolling d20 to hit vs ac of AC with attack of STAFFATTACK]]
  }
  result: TOTAL
}

function: pommelattack {
  result: [damage for staff pommel [rolling d20 to hit vs ac of AC with attack of STAFFATTACK]]
}

function: multiattackdoubleadvantage NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for staff main [rolling [highest 1 of 3d20] to hit vs ac of AC with attack of STAFFATTACK]]
  }
  result: TOTAL
}

function: pommelattackdoubleadvantage {
  result: [damage for staff pommel [rolling [highest 1 of 3d20] to hit vs ac of AC with attack of STAFFATTACK]]
}

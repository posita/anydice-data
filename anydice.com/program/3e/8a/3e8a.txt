function: rollone {
 result: [roll I] + [roll I] + [roll D] + [roll B]
}

function: rolltwo {
 result: [roll (A * D)] + [roll B] + [roll D]
}

function: roll DIE:d {
 result: [rollrec DIE 0 0]
}

function: rollrec DIE:d MOD:n SUM:n {
 MAX: [maximum of DIE]
 if MOD = MAX {result: 0}
 result: [helper DIE MAX SUM]
}

function: helper ROLL:n MAX:n SUM:n {
 if ROLL - MOD < 0 {result: 0}
 if ROLL = MAX {result: (ROLL + [rollrec ROLL MOD + 1 SUM] - MOD)}
 if MOD > 0 {result: ROLL - MOD}
 result: ROLL
}

A: d4

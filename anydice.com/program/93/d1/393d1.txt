ARMOR : 1

function: check X:n target T:n accuracy A:n difficulty D:n bonus B:n {
    if X + A - D + B >= 20 { result : 2 }
    if X + A - D + B >= T { result : 1 }
    result : 0
}

function: reliable X:n target T:n accuracy A:n difficulty D:n grit G:n {
    X : [check X target T accuracy A difficulty D bonus G]
    if X > 0 {
        H : 1@Xd6
        H : [highest of H and 2]
    } else { H : 2 }
    H : [highest of H - ARMOR and 0]
    result : H
}

function: battlerifle X:n target T:n accuracy A:n difficulty D:n grit G:n {
    X : [check X target T accuracy A difficulty D bonus G]
    if X > 0 {
        H : 1@Xd{2..6}
    } else { H : 0 }
    result : H
}

function: hmg X:n target T:n accuracy A:n difficulty D:n grit G:n {
    X : [check X target T accuracy A difficulty D bonus G]
    if X > 0 {
        H : {1..2}@{2*X}d6 + 4
    } else { H : 0 }
    H : [highest of H - ARMOR and 0]
    result : H
}

output [reliable 1d20 target 8 accuracy 0 difficulty 0 grit 1]
output [battlerifle 1d20 target 8 accuracy 0 difficulty 0 grit 1]
output [hmg 1d20 target 8 accuracy 0 difficulty 1d6 grit 1]

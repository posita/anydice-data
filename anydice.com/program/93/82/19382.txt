MISS: 0
HIT: 1
KILL: 1
NOKILL: 0
DAMAGE: 2d6
BONUSSKILL: 3
DEF: 6
HEALTH: 13
BONUSDAMAGE: DAMAGE + BONUSSKILL

function: ROLL:n vs DEFENSE:n {

 if ROLL > DEFENSE { result:ROLL-DEFENSE }
 result: MISS
}


output [BONUSDAMAGE vs DE] named "damage"

function: ROLL:n vss DEFENSE:n {

 if ROLL > DEFENSE { result:HIT}
 result: MISS
}
output [DAMAGE vss DT] named "did hit?"

function: DMG:n hit HP:n {

 if DMG > HP { result:KILL}
 result: NOKILL
}

output [[BONUSDAMAGE vs DT] hit HEALTH] named "one hit KO chance"

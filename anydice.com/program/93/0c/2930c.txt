function: echo A:s {
if (1@A = N@A) { result: A + (N-1)d(S-2)}
if ((1@A = 3@A) | (2@A = N@A)) { result: A + (N-2)d(S-2)}
if ((1@A = 2@A) | (2@A = 3@A) | (3@A = N@A)) { result: A + (N-3)d(S-2)}
result: A
}
N: 4
S: 10
A: NdS
output [echo A] named "echoing [A]"
output 4d[explode dS] named "exploding [A]"
output A named "[A]"

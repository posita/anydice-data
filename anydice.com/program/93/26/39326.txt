X: 6   \ hit point die per level \
DH: dX \ level 1 hit points \
H: dX  \ level 1 hit points \
L: 3   \ level \

function: dwarf A of B {
  S: 0
  loop N over {1..A}{
    S: S + 1@2dB
  }
  result: S+2
}
loop N over {2..L}{
  P: [dwarf N of X]
  DH: [highest of P and DH+1]
  H: [highest of NdX and H+1]
}

output LdX named "[L]d[X]"
output H named "Knave2E Lv[L]"
output [dwarf L of X] named "SD Dwarf Lv[L]"
output DH named "Knave2E Dwarf Lv[L]"

function: ROLL:n damage DEFENSE:n {
	if ROLL > DEFENSE { result:ROLL-DEFENSE }
	else {result:0}
}
function: attack X:n Y:d Z:n {
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: [Y damage Z]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

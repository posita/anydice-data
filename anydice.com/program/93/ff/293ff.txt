function: wound S:n T:n{
  if T*2<=S{
    result: 2
  }
  if T<S{
    result:3
  } 
  if T=S{
    result:4
  }
  if T>S{
    result:5
  }
  if T>S*2{
    result:6
  }
}
function: warhammerattack WS:n HR:n S:n T:n WB:n WR:n AP:n SV:n{
  W: [wound S T]
  H: [highest of [lowest of (1d6)/WS and 1] and HR*[lowest of (1d6)/WS and 1]]
  W: [highest of [lowest of (1d6+WB)/W and 1] and WR*[lowest of (1d6+WB)/WR and 1]]
  SS: [absolute [lowest of ([highest of 1d6-AP and 0])/SV and 1]-1]
  result: H*W*SS
}
function: attack WS:n HR:n S:n T:n WB:n WR:n AP:n SV:n I:n IV:n{
  result: [highest of [warhammerattack WS HR S T WB WR AP SV] - I*[lowest of 1d6/IV and 1] and 0]
}
function: attacks ws WS:n  hit reroll HR:n str S:n tough T:n wound bonus WB:n wound reroll WR:n ap AP:n sv SV:n extra save I:n check IV:n attacks A:n{
  D: 0
  loop N over {1..20}{
    if N<=A {
      D: D + [attack WS HR S T WB WR AP SV I IV]
    }
  }
  result: D
}
function: betterattacks ws WS:n reroll one HRO:n reroll HRR:n bonus HB:n s S:n t T:n reroll one WRRO:n reroll WRR:n bonus WB:n ap AP:n sv SV:n reroll one SVRO:n reroll SVR:n invul save IS:n ignore wound IW:n attacks A:n{
  D: 0
  loop N over {1..3}{
    if N<=A {
      D: D+[betterattack hit roll 1d6 ws WS reroll one HRO*1d6 reroll HRR*1d6 bonus HB s S t T wound roll 1d6 reroll one WRRO*1d6 reroll WRR*1d6 bonus WB ap AP sv SV save roll 1d6 reroll one SVRO*1d6 reroll SVR*1d6 invul save IS ignore wound IW]
    }
  }
  result: D
}
function: bonus A:n{
  D: 0
  loop N over {1..10}{
    if N<=A {
      D: D+1d6/6
    }
  }
  result: D
}
function: manifest warp charge WC:n test bonus TB:n deny the witch DW:n deny bonus DB:n free deny AP:n free deny check DC:n  {
  W: 2d6+TB
  D: (2d6+DB)
  A: 1d6
  R: [highest of ([lowest of W/WC and 1] - DW*[highest of [lowest of ((W-D)*(-1)) and 1] and 0] - AP*[lowest of A/DC and 1]) and 0]
  result: R
}
function: ignore mortal on C:n D:n {
  result: D-[lowest of 1d6/C and 1]-D/2*[lowest of 1d6/C and 1]-D/3*[lowest of 1d6/C and 1]
}



function: betterattack hit roll HR:n ws WS:n reroll one HRO:n reroll HRR:n bonus HB:n s S:n t T:n wound roll WR:n reroll one WRRO:n reroll WRR:n bonus WB:n ap AP:n sv SV:n save roll SR:n reroll one SVRO:n reroll SVR:n invul save IS:n ignore wound IW:n{
  H: [lowest of (HR+HB)/WS and 1]
  WC:[wound S T]
  W: [lowest of (WR+WB)/WC and 1]
  S: -1*([lowest of (SR-AP)/SV and 1]-1)
  if SR=1{
    S: -1*([lowest of (SVRO-AP)/SV and 1]-1)
  }
  if S=0{
    S: -1*([lowest of (SVR-AP)/SV and 1]-1)
  }
  if IS!=0{
    if SV+AP>IS{
      S: -1*([lowest of SR/IS and 1]-1)
    }
  }
  if HR=1{
    H: [lowest of (HRO+HB)/WS and 1]
  }
  if H=0{
    H: [lowest of (HRR+HB)/WS and 1]
  }
  if WR=1{
    H: [lowest of (WRRO+WB)/WC and 1]
  }
  if W=0{
    H: [lowest of (WRR+WB)/WC and 1]
  }
  I:1
  if IW!=0{
    I: -1*([lowest of (1d6)/IW and 1]-1)
  }
  R: H*W*S*I
  result: R  
}
output [attacks ws 2 reroll one 1 reroll 0 bonus 0 s 7 t 6 reroll one 1 reroll 0 bonus 1 ap 3 sv 2 reroll one 0 reroll 0 invul save 4 ignore wound 0 attacks 3]

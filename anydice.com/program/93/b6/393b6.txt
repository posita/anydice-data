\ 3 = crit \
\ 2 = success \
\ 1 = mixed \
\ 0 = fail \
\ X = bonus \

X: 0

function: dark A:s {
 if [count {6..10} in A] > 1 { result: 3 }
 if [count {6..10} in A] > 0 { result: 2 }
 if [count {4,5} in A] > 0 { result: 1 }
 result: 0
}

function: darkzero A:s {
 if [count {-5..3} in A] > 0 { result: 0 }
 if [count {4,5} in A] > 0 { result: 1 }
 if [count {6..10} in A] > 0 { result: 2 }
 result: 0
}


output [darkzero 2d(d6+X)] named "0d6+X"
loop N over {1..4}{
 output [dark Nd(d6+X)] named "[N]d6+X"
}

function: xd ROLL:s COUNT:n MAX:n{
  if COUNT>=2 {
    result: COUNT * MAX
  }
    result: 1@ROLL + 2@ROLL
}

function: testhelper SKILL:n ROLL:s {
  COUNT: [count SKILL in ROLL]
  result: [xd [sort ROLL] COUNT SKILL]
}

function: test cecha D:n skill S:n {
  ROLL: Dd{0, 2..S}
  result: [testhelper S ROLL]
}

CECHA:2
SKILL:3

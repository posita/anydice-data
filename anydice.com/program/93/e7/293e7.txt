function: unique values in S:s {
  U: {}
  loop N over S { if !(N = U) { U: {U, N} } }
  result: U
}  
function: find PATTERN:s in ROLL:s {
  COUNTS: {}
  loop N over [unique values in ROLL] {
    COUNTS: {COUNTS, N = ROLL}
  }
  result: [sort PATTERN] = [sort COUNTS]
}

P:4d10
output [find {4} in P] named "four of a kind"
output [find {3, 1} in P] named "a trio, one single"
output [find {2, 2} in P] named "two pairs"
output [find {2, 1, 1} in P] named "a pair, two singles"
output [find {1, 1, 1, 1} in P] named "all singles"

function: A:s vs B:s {
 loop P over {1..[lowest of #A and #B]} {
  if P@A > P@B { result: [A beats P@B at P] }
  if P@B > P@A { result: -[B beats P@A at P] }
 }

 if [count 1 in {#A, #B}] = 1 {
  if (#A - #B) > 0 {result: 1}
  if (#A - #B) < 0 {result: -1}
 } else {result: 0}
}

function: A:s beats T:n at W:n {
 V: 1
 loop P over {W + 1 .. #A} {
  if P@A > T { V: V + 1 }
  else { result: V }
 }
 result: V
}

output [2d6 vs 1d6]

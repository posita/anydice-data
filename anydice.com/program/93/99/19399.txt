MISS: 0
HIT: 1
KILL: 1
NOKILL: 0
DAMAGE: 2d6
BONUSSKILL: 3
DEF: 5
END: 6
CLAR: 2
DETERM: 3
HEALTH: 13
BONUSDAMAGE: DAMAGE + BONUSSKILL

function: ROLL:n vs DEFENSE:n {

 if ROLL > DEFENSE { result:ROLL-DEFENSE }
 result: MISS
}





function: ROLL:n vss DEFENSE:n {

 if ROLL > DEFENSE { result:HIT}
 result: MISS
}


function: DMG:n hit HP:n {

 if DMG > HP { result:KILL}
 result: NOKILL
}


output [[BONUSDAMAGE vs DEF] hit HEALTH] named "one hit KO chance"
output [DAMAGE vss DEF] named "did hit?"
output [BONUSDAMAGE vs DEF] named "damage"

function: fubar MOD POS {
    D: 5+MOD
    if POS = 1 {
        result: [highest 5 of Dd{0,1}]
    }
    else {
        result: [lowest 5 of Dd{0,1}]
    }
}

output [fubar 0 1] named "0"
output [fubar 1 0] named "-1"
output [fubar 2 0] named "-2"
output [fubar 1 1] named "+1"
output [fubar 2 1] named "+2"

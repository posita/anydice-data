MISS: 0
HIT: 1
CRITICALHIT: 2
SNEAKATTACK: 3d6
SUPDICE: 1d8

function: attack ROLL:n plus BONUS:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  result: CRITICALHIT
 }
 if ROLL + BONUS + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}


function: damage for ATTACKRESULTS:s {
 D: 0
 loop A over ATTACKRESULTS {
  if A = CRITICALHIT {
   D: D + HITDAMAGE + CRITICALDAMAGE
  }
  else if A = HIT {
   D: D + HITDAMAGE
  }
  else {
   D: D + MISSDAMAGE
  }
 }
 result: D + [bonus damage for ATTACKRESULTS]
}


function: bonus damage for ATTACKRESULTS:s {
 result: 0
}

function: attack {
 result: [damage for [attack d20 plus SUPDICE vs DEFENSE]]
}

function: N:n attacks {
 result: [damage for Nd[attack d20 plus SUPDICE vs DEFENSE]]
}

function: bonus damage for ATTACKRESULTS:s {
 if [ATTACKRESULTS contains HIT] {
  result: SNEAKATTACK
 }
else if [ATTACKRESULTS contains CRITICALHIT] {
 result: SNEAKATTACK + SNEAKATTACK
}
result: 0
}

function: supdamage for ATTACKRESULTS:s {
 D: 0
 loop A over ATTACKRESULTS {
  if A = CRITICALHIT {
   D: D + SUPHITDAMAGE + SUPCRITICALDAMAGE
  }
  else if A = HIT {
   D: D + SUPHITDAMAGE
  }
  else {
   D: D + SUPMISSDAMAGE
  }
 }
 result: D + [bonus supdamage for ATTACKRESULTS]
}

function: supattack {
 result: [supdamage for [attack d20 vs DEFENSE]]
}

function: N:n supattacks {
 result: [supdamage for Nd[attack d20 vs DEFENSE]]
}

function: bonus supdamage for ATTACKRESULTS:s {
 if [ATTACKRESULTS contains HIT] {
  result: SNEAKATTACK
 }
else if [ATTACKRESULTS contains CRITICALHIT] {
 result: SNEAKATTACK + SNEAKATTACK
}
result: 0
}

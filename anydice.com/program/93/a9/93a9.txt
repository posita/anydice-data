AC: 11
ATTACK_BONUS: 0
DMG_BONUS: 0
DMG_DIFFERENCE: 33

function: X:n attackwith D:n {
  if X = 20 {result: D + D + DMG_BONUS}
  else if (X + ATTACK_BONUS >= AC) {result: D + DMG_BONUS}
  result: 0
}

TWODSIX: 0
ONEDSIX: 0

loop N over {1..100}{
  TWODSIX: TWODSIX + [1d20 attackwith 2d6]
  ONEDSIX: ONEDSIX + [1d20 attackwith 1d6] + [1d20 attackwith 1d6]
}

output ONEDSIX
output TWODSIX

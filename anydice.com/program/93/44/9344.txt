function: pool of A:s and B:s {
 AHIGHEST: 1@(A)
 ARESULT : AHIGHEST+ [count AHIGHEST in A] - 1

 BHIGHEST: 1@(B)
 BRESULT : BHIGHEST + [count BHIGHEST in B] -1

 if AHIGHEST > BHIGHEST {result : ARESULT}
 if AHIGHEST < BHIGHEST {result : BRESULT}
 result : AHIGHEST + [count BHIGHEST in B] -1

}

M:6     \ die size i.e. sides of the die \
N:2     \ number of dice \

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

output [reroll 1 on d6 to d6] named "d6 reroll 1"

output 1d6 named "d6 straight"

output [reroll (1) on Nd6 to Nd6] named "[N]d6 reroll 1"

output NdM named "[N]d[M] straight"

output [reroll {1,2} on d8 to d8] named "d8 reroll 1,2"

function: pool ROLL:s TARGET:n {
  ACC: 0
  loop I over {1..#ROLL}{
    if I@ROLL >= TARGET {ACC: ACC + 1}
  }
  result: ACC
}

function: combine MAG:n HIT:n ARM:n {
  HITS: [pool MAGd6 HIT]
  ACC: {}
  loop I over {0..#HITS-1} {
    ACC: {ACC, 1d6}
  }
  \result: ACC\
  result: [pool ACC ARM]
}

\TMP: [combine 2 4 4]\

\TMP: {}
loop I over {1..3} {
  TMP: {TMP, 1d6}
}
loop J over {1..#TMP} {
  output J@TMP
}\

\output TMP\

\loop I over {1..#TMP}{
  output I@TMP
}\

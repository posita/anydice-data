function: echo P:s {
if (1@P = N@P) { result: P + (N-1)dES}
if ((1@P = 3@P) | (2@P = N@P)) { result: P + (N-2)dES}
if ((1@P = 2@P) | (2@P = 3@P) | (3@P = N@P)) { result: P + (N-3)dES}
result: P
}

N: 4  \ N for Number of dice in the initial dice pool \
S: 10  \ S for Size of the dice in the initial dice pool \
ES: S-2 \ ES for Echo dice Size \
P: NdS   \ P for initial dice Pool \

output [echo P] named "echoing [P]"
output Nd[explode dS] named "exploding [P]"
output P named "[P]"

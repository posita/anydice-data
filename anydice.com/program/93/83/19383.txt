MISS: 0
HIT: 1
KILL: 1
NOKILL: 0
DAMAGE: 2d6
BONUSSKILL: 3
DEF: 6
END: 6
CLAR: 6
DETERM: 6
HEALTH: 13
BONUSDAMAGE: DAMAGE + BONUSSKILL

function: ROLL:n vs DEFENSE:n {

 if ROLL > DEFENSE { result:ROLL-DEFENSE }
 result: MISS
}


output [BONUSDAMAGE vs DEF] named "damage (with DEF)"
output [BONUSDAMAGE vs END] named "damage (with END)"
output [BONUSDAMAGE vs CLAR] named "damage (with CLAR)"
output [BONUSDAMAGE vs DETERM] named "damage (with DETERM)"

function: ROLL:n vss DEFENSE:n {

 if ROLL > DEFENSE { result:HIT}
 result: MISS
}
output [DAMAGE vss DEF] named "did hit? (with DEF)"
output [DAMAGE vss END] named "did hit? (with END)"
output [DAMAGE vss CLAR] named "did hit? (with CLAR)"
output [DAMAGE vss DETERM] named "did hit? (with DETERM)"

function: DMG:n hit HP:n {

 if DMG > HP { result:KILL}
 result: NOKILL
}

output [[BONUSDAMAGE vs DEF] hit HEALTH] named "one hit KO chance (with DEF)"
output [[BONUSDAMAGE vs END] hit HEALTH] named "one hit KO chance (with END)"
output [[BONUSDAMAGE vs CLAR] hit HEALTH] named "one hit KO chance (with CLAR)"
output [[BONUSDAMAGE vs DETERM] hit HEALTH] named "one hit KO chance (with DETERM)"

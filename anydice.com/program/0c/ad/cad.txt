\
 demonstration of a fast roll-and-keep function for L5R
 10s are kept as-is, they explode after keeping
\

function: ROLL:n g KEEP:n {
 result: [convert [highest KEEP of ROLL d{1..9, 1000}]]
}

function: convert SUM:n {
 if SUM >= 1000 {
  TENSROLLED: SUM / 1000
  result: SUM - TENSROLLED * 990 + TENSROLLED d [explode d10]
 }
 result: SUM
}

set "explode depth" to 1
loop KEEP over {9..10} {
  loop ROLL over {9..10} {
    if KEEP <= ROLL { output [ROLL g KEEP] named "[ROLL]g[KEEP]" }
  }
}

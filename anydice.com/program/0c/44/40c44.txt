\ output [highest 1 of 2d6] + 3 named "Variable Blade Crit" \
function: ROLL:n vs AC:n {
 if ROLL >= THREAT & ROLL + ATTACK >= AC {
  result: [threat d20]
 }
 if ROLL + ATTACK >= AC | ROLL = 20 { result: DAMAGE }
 result: 0
}

function: threat ROLL:n {
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: CRITICAL d DAMAGE
 }
 result: DAMAGE
}

AC: d{14..20}
STRENGTH: 3
ATTACK: 1 + STRENGTH

DAMAGE: 1d8 + STRENGTH
THREAT: 20
CRITICAL: 3

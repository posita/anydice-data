function: nwodrote R:n again N:n {
  if N >= R { result: 1 + [nwodrote R again d10] }
  result: N >= 8
}

function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwodrote R again d10] }
 result: 0 + [nwodrote R again d10]
}

NWOD: [nwod 8 again d10]

loop N over {10..10} {
 output [lowest of 10 and NdNWOD] named "[N]d"
}

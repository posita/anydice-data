function: skillchallenge SUCCESSES:n before FAILURES:n against DC:n {
 ROLLS: SUCCESSES + FAILURES - 1
 TOTAL: 0
 loop N over {1..ROLLS}{
  TOTAL: TOTAL + [skillroll 1d20 vs DC]
 }
 result: TOTAL
}

function: skillroll ROLL:n vs DC:n {
 if ROLL >= CRITROLL { result: CRITRESULT } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ABILITYMOD  >= DC { result: 1 }
 result: 0
}

PROFICIENCY: 6
ABILITYMOD: 0
CRITROLL: 20
CRITRESULT: 2
output [skillchallenge 12 before 3 against 13]

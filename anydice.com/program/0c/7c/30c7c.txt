function: divine ATT:d SKILL:d SET:s {
 FIN: 0
 loop N over SET {
  if N <= [maximum of ATT] {
   FIN: FIN | ATT = N
  }
  if N <= [maximum of SKILL] {
   FIN: FIN | SKILL = N
  }
 }
 result: FIN
}

loop Z over {2,4,6,8,10,12} {
 output [divine 1dZ 1dZ {1}] named "divine 1 2d[Z]"
 output [divine 1dZ 1dZ {1,2}] named "divine 2 2d[Z]"
 output [divine 1dZ 1dZ {1,2,3}] named "divine 3 2d[Z]"
 output [divine 1dZ 1dZ {1,2,3,4}] named "divine 4 2d[Z]"
}

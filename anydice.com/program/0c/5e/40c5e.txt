function: blades ROLL:s {

if 1@ROLL = 24 {
   result: 4
}

if 1@ROLL != 24 & 1@ROLL >= 18 {
   result: 3
}

if 1@ROLL < 18 & 1@ROLL >= 12 {
result:2
}

if 1@ROLL < 12 & 1@ROLL >= 6  {
result:1 
}

result:0 


}
output [blades 2@2d{5..24}] named "0d20+4"
loop P over {1..4} {
 output [blades Pd{5..24}] named "[P]d20+4"
}

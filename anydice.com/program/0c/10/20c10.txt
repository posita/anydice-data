function: attack ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + ARCH >= AC { result: DAMAGE }
 result: 0
}

ENEMYAC: 18

ATTACKMOD: 5

PROFICIENCY: 4
CRITROLL: 20
EXTRAATTACK: 6
DMGBONUS: 0
ACCBONUS: 0
EXTRADMGDICE: 0

ARCH: 0
ACCBONUS: 0
ROUNDS: 1

DAMAGE: 1d8 + ATTACKMOD + DMGBONUS + 2 + EXTRADMGDICE
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS + 2 + 2dEXTRADMGDICE

output (ROUNDS)*EXTRAATTACKd[attack d20 vs ENEMYAC] named "Fighter: Dueling Longsword"

output ROUNDS*2d[attack d20 vs ENEMYAC] named "Rogue + Commander's Strike"

ACCBONUS: -5

DAMAGE: 1d8 + ATTACKMOD + DMGBONUS + EXTRADMGDICE

output (ROUNDS)*EXTRAATTACKd[attack [highest 1 of 3d20] vs ENEMYAC] named "Fighter: Samurai Archer"

function: atk AC A:n M{
 if A=20 {result:2}
 if A+M>=AC{result:1}
 result:0
}

loop N over {5..30}{
 output [atk N 1d20 5+2]*(1d6+5) named "[N] std"
 output [atk N 1@2d20 5+2]*(1d6+5) named "[N] adv"
 output [atk N 2@2d20 5+2]*(1d6+5) named "[N] dis"
 output [atk N 1d20 2]*(1d6+5+10) named "[N] ShS"
 output [atk N 1@2d20 2]*(1d6+5+10) named "[N] SSa"
 output [atk N 2@2d20 2]*(1d6+5+10) named "[N] SSd"
}

set "explode depth" to 3

D:2
REDMIN:2
REDMAX:2
WHITEMIN:1
WHITEMAX:6

loop R over {REDMIN..REDMAX} {
  DAM:D
  loop W over {WHITEMIN..WHITEMAX}{
  NW: W-R
   output Wd{0..5} named "[W]W&0R=[DAM]"
   DAM:DAM+1
   if NW < 0 {
       NR:R+NW
       output NRd[explode d{0..5}] named "0W&[NR]R=[DAM]"
   } else {
      output NWd{0..5} + Rd[explode d{0..5}] named "[NW]W&[R]R=[DAM]"
   }
   DAM:DAM+1
   }
}

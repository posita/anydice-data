function: maximize DIE NUMBER MAX EQUALS helper
{
if NUMBER>0 
{ 
if EQUALS>0 & MAX/EQUALS =DIE 
{
result: [maximize DIE NUMBER-1 MAX+DIE EQUALS+1 helper]
}
else
{
if MAX = DIE 
{
result: [maximize DIE NUMBER-1 MAX+DIE EQUALS+1 helper]
}
else
{
if MAX > DIE
{
result: [maximize DIE NUMBER-1 MAX EQUALS helper]
}
else
{
result: [maximize DIE NUMBER-1 DIE EQUALS helper]
}
}
}
}
result: MAX
}

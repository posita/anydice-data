SUCCES: {6..12, 10,12}
BOTCH: {0}

DES: {6, 8, 10, 12}

function: reroll RESULTAT:n sur DE:n {
  if ![SUCCES contains RESULTAT] & ![BOTCH contains RESULTAT] {result: dDE}
  result: RESULTAT
}


function: roll DE:n relance REROLL:n {
  if REROLL > 0 {
   result: [reroll dDE sur DE]
  }
  result: dDE
}

loop ATTRIBUT over DES {
  loop COMPETENCE over DES {
    output [count SUCCES in [roll ATTRIBUT relance 1]] + 
      [count SUCCES in [roll COMPETENCE relance 1]] 
      named "Attribut D[ATTRIBUT] et Compétence D[COMPETENCE]"
  }
}

\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 3
DMGBONUS: 3
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 1d8 + DMGBONUS + 1d6
CRITDAMAGE: 2d8 + 3 + 2d6
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Longsword + Hex(1d8 + 3 + 1d6) Duration: HOURS less Concentration Loss"

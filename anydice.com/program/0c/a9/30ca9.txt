function: roll pool DFOUR:s and DSIX:s and DEIGHT:s and DTEN:s and DTWELVE:s {
  KEEP: 2
  JINX: 1
  POOL: {DFOUR,DSIX,DEIGHT,DTEN,DTWELVE}
  loop P over {1..#POOL} {
    if P@POOL = JINX {
      POT: {1..(P-1),(P+1)..#POOL}@POOL
    }
  }
  result: {1..KEEP}@[sort POOL]
}

function: jinxes DFOUR:s and DSIX:s and DEIGHT:s and DTEN:s and DTWELVE:s {
  KEEP: 2
  JINX: 1
  JINXES: 0
  POOL: {DFOUR,DSIX,DEIGHT,DTEN,DTWELVE}
  loop P over {1..#POOL} {
    if P@POOL = JINX {
      JINXES: JINXES + 1
    }
  }
  result: JINXES
}

\ Master Included \
output [roll pool 0d4 and 2d6 and 1d8 and 1d10 and 0d12] named "Master Skill"
\ Master Hitches \
output [jinxes 0d4 and 2d6 and 1d8 and 1d10 and 0d12] named "Master Hitches"
\ Expert Included \
output [roll pool 0d4 and 2d6 and 2d8 and 0d10 and 0d12] named "Expert Skill"
\ Expert Hitches \
output [jinxes 0d4 and 2d6 and 2d8 and 0d10 and 0d12] named "Expert Hitches"
\ Trained Only \
output [roll pool 0d4 and 3d6 and 1d8 and 0d10 and 0d12] named "Trained Skill"
\ Trained Hitches \
output [jinxes 0d4 and 3d6 and 1d8 and 0d10 and 0d12] named "Trained Hitches"

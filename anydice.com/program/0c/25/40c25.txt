function: match TARGET:n in SEQUENCE:s {
 loop P over {1..6} {
  TCOUNT: 0
  loop Q over {1..#SEQUENCE} {
   TCOUNT: TCOUNT + (P = Q@SEQUENCE)
  }
  if TCOUNT >= TARGET {
   result: 1
  }
 }
 result: 0
}

function: square root of N:n {
  loop G over {1..N}{
    if G * G > N {
      result: G
    }
  }
}

function: stat {
  result: ((1d256 - 1 + 2 * (1d16 - 1)) * 2 + ([square root of (1d256-1)])/4)
}

X: 0

loop N over {1..5}{
  X: X + [stat]
  if N = 1 {
    X: X + X
  }
}
\output [stat] named "stat"\
output X + 100 + 10 + 5*5 named "mbst"

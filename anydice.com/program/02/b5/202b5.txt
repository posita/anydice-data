function: rolling DICE:n of DIE:n success NUMBER:n {
 V: 0
 if DICE > 0 {
  if DIE > NUMBER {
   V: 1
  }
  result: V + [rolling (DICE - 1) of d6 success NUMBER]
 }
 result: 0
}

function: redo DICE:n for NUMBER:n {
 
 SUCCESS: [rolling DICE of d6 success NUMBER]

 if SUCCESS > NUMBER{
  result: 1
 }
 if SUCCESS = 0{
  result: 0
 }
 result: [redo DICE for (NUMBER - SUCCESS)]
} 

output [redo 4 for 4]

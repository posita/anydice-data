\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n hit HIT:n vs AC:n dmg DMG:n crit CRIT:n {
 if ROLL >= 20 { result: CRIT } 
 if ROLL = 1 { result: 0 }
 if ROLL + HIT >= AC { result: DMG }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

UNARMED:     d[attackroll d20 hit 7 vs ENEMYAC dmg 1d4 + 5 crit 2d4 + 5]
SSWORD:      d[attackroll d20 hit 9 vs ENEMYAC dmg 1d6 + 7 crit 2d6 + 7]
SSWORDLH:    d[attackroll d20 hit 9 vs ENEMYAC dmg 1d6 + 2 crit 2d6 + 2]
DROWSWORD:   d[attackroll d20 hit 7 vs ENEMYAC dmg 1d6 + 9 crit 2d6 + 9]
DROWSWORDLH: d[attackroll d20 hit 7 vs ENEMYAC dmg 1d6 + 4 crit 2d6 + 4]

output UNARMED + UNARMED named "2 unarmed attack"

output SSWORD + SSWORDLH named "two weapon shortsword"

output DROWSWORD + SSWORDLH named "two weapon drow + short"

output DROWSWORD + UNARMED named "drow sword + unarmed"

output SSWORD + UNARMED named "shortword + unarmed"

output DROWSWORD + DROWSWORDLH named "two weapon drow sword"

output SSWORD + DROWSWORDLH named "two weapon short + drow"

output UNARMED + UNARMED + UNARMED named "unarmed w/flurry"

output DROWSWORD + UNARMED + UNARMED named "drow sword w/flurry"

output SSWORD + UNARMED + UNARMED named "shortsword w/flurry"

\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n hit HIT:n vs AC:n dmg DMG:n crit CRIT:n {
 if ROLL >= 20 { result: CRIT } 
 if ROLL = 1 { result: 0 }
 if ROLL + HIT >= AC { result: DMG }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

UNARMED:      d[attackroll d20 hit 7 vs ENEMYAC dmg 1d4 + 5 crit 2d4 + 5]
SSWORD:       d[attackroll d20 hit 9 vs ENEMYAC dmg 1d6 + 7 crit 2d6 + 7]
DROWSWORD:    d[attackroll d20 hit 7 vs ENEMYAC dmg 1d6 + 9 crit 2d6 + 9]
\offhand versions of the two swords for dual wielding\
SSWORDOFF:    d[attackroll d20 hit 9 vs ENEMYAC dmg 1d6 + 2 crit 2d6 + 2]
DROWSWORDOFF: d[attackroll d20 hit 7 vs ENEMYAC dmg 1d6 + 4 crit 2d6 + 4]
\flurry of blows\
FLURRY:       UNARMED + UNARMED

output UNARMED   + UNARMED      named "2 unarmed attack"
output SSWORD    + SSWORDOFF    named "two weapon shortsword"
output DROWSWORD + SSWORDOFF    named "two weapon drow + short (offhand)"
output DROWSWORD + UNARMED      named "drow sword + unarmed"
output SSWORD    + UNARMED      named "shortword + unarmed"
output DROWSWORD + DROWSWORDOFF named "two weapon drow sword"
output SSWORD    + DROWSWORDOFF named "two weapon short + drow (offhand)"
output UNARMED   + FLURRY       named "unarmed w/flurry"
output DROWSWORD + FLURRY       named "drow sword w/flurry"
output SSWORD    + FLURRY       named "shortsword w/flurry"

function: roll S:s default DEFAULT:n {

 \ find 2nd highest value \
 CURRENTH: 1
 CURRENTVALUE: 1@S
 loop P over {1..#S} {
  if CURRENTVALUE != P@S {
   CURRENTH: CURRENTH + 1
   CURRENTVALUE: P@S
  }
  if CURRENTH = 2 { result: P@S }
 }

  result: DEFAULT
}

output [roll 3d20 default 10] named "3d20, pick 2nd highest value"

output [roll 3d8 default 4] named "3d8, pick 2nd highest value"
output [roll 3d6 default 3] named "3d6, pick 2nd highest value"

\Function code\
function: reroll N:n below M:n
{if N < M {result: [highest of d6 and N]} result: N }

\Outputs\
output 6d [{5..6} contains [reroll d6 below 2]] named "A: 5s to hit rerolling 1s"
output 6d [{4..6} contains [reroll d6 below 1]] named "B: 4s to hit no rerolls"
output 6d [{5..6} contains [reroll d6 below 2]] - 6d [{4..6} contains [reroll d6 below 1]] named "A - B"

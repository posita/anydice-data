\Calculates the chance to hit of a monster attack. Accounts for SH and EV.\

\Monster input values\
HD: 13

\Player input values\
SH: 15
PREV_BLOCKS: 0
\end input values\


\functions, yo\
function: rtwo X:n {
    result: 1dX - 1
}

function: rtwoavg X:n across N:n {
    result: (NdX - 1) / N
}


function: shpierce MONS_HD:n{
    result: [rtwo [15 + (3 * MONS_HD / 2)]]
}

function: shblock PL_SH:n {
    result: ([rtwoavg (2 * PL_SH) across 2] / 3) - 1
}


output [] named "SH effectivenss"
output [shpierce HD] named "SH Pierce"
output [shblock SH ]named "SH Block"

ENEMYAC:15
ATKDICE:1d20
ATKBONUS:15
DMGDICE:1d6
DMGBONUS:13

function: damage ATK:n ATKBONUS:n DMG:n DMGBONUS:n { if ATK = 1 { result: 0 } if ATK = 20 { result: 2*DMG+DMGBONUS } if ATK+ATKBONUS < ENEMYAC { result: 0 } result: DMG+DMGBONUS }

output [damage ATKDICE ATKBONUS DMGDICE DMGBONUS] named "No Sharpshooter"
output [damage ATKDICE ATKBONUS-5 DMGDICE DMGBONUS+10] named "Sharpshooter"

function: ATTACKS:s vs DEF:n dmg DAMAGE:n {
  SUM: {0}
  loop ATTACK over ATTACKS {
    SUM: SUM + [3d8 plus ATTACK ac DEF dmg DAMAGE]
  }
  result: SUM
}

function: ROLL:n plus ATTACK:n ac DEF:n dmg DAMAGE:n {
  result: (ROLL + ATTACK > DEF) * (DAMAGE + ((ROLL + ATTACK) - DEF))}

\ Usage
output [{ATK Bonus} vs {Defense} dmg {Base Damage}]
\

\Examples\

output [7 vs 15 dmg 12] named "Greatsword +7 vs 15 DEF (12 DMG)"
output [7 vs 15 dmg 18] named "Greatsword +7 Heavy Strike vs 15 DEF (12 DMG)"
output [7 vs 15 dmg 8] named "Battleaxe +7 vs 15 DEF (8 DMG)"

output [7 vs 20 dmg 12] named "Greatsword +7 vs 20 DEF (12 DMG)"
output [7 vs 20 dmg 8] named "Battleaxe +7 vs 20 DEF (8 DMG)"

function: pick value H:n from S:s {
 CURRENTH: 1
 CURRENTVALUE: 1@S
 loop P over {1..#S} {
  if CURRENTVALUE != P@S {
   CURRENTH: CURRENTH + 1
   CURRENTVALUE: P@S
  }
  if CURRENTH = H { result: P@S }
 }
}

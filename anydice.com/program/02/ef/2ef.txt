function: ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL + ATTACK >= DEFENSE | ROLL = 20 { 
 if ROLL >= THREAT {     
  result: [threat d20]
 }

result: DAMAGE 
}
 result: MISS
}



function: threat ROLL:n {
 if ROLL + ATTACK >= DEFENSE | ROLL = 20 { result: CRITICAL }
 result: DAMAGE
}

result: ATTACKRESULT

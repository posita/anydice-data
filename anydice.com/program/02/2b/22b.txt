function: roll S:s with control number C:n {

 \ find 2nd highest value \

 CURRENTH: 1
 CURRENTVALUE: 1@S
 loop P over {1..#S} {
  if CURRENTVALUE != P@S {
   CURRENTH: CURRENTH + 1
   CURRENTVALUE: P@S
  }
  if CURRENTH = 2 { result: P@S }
 }

 \ no 2nd highest, it's a pair \
 
 \ next is a toss-up between 0 and 6, but 1/6th of the time it's a 3 \

 if 1@S > C { result: 6 }
 if 1@S < C { result: 0 }
 if 1@S = C { result: 3 }
}

output [roll 2d6 with control number d6]

function: ATTACKS:s vs AC:n dmg DAMAGE:n {
  SUM: {0}
  loop ATTACK over ATTACKS {
    SUM: SUM + [3d8 plus ATTACK ac AC dmg DAMAGE]
  }
  result: SUM
}

function: ROLL:n plus ATK:n ac AC:n dmg DAMAGE:n {
  result: (ROLL + ATK >= AC) * (DAMAGE + ((ROLL + ATK) - AC))}

output [7 vs 15 dmg 12] named "Greatsword vs 15 AC (12 DMG)"
output [7 vs 15 dmg 8] named "Battleaxe vs 15 AC (8 DMG)"

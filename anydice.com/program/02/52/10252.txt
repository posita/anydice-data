
FUMBLE: -1
MISS: 0


function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 4
 if [count 1 in ROLL] > SUCCESSES { if SUCCESSES > 0 {result: MISS} if SUCCESSES = 0 {result: FUMBLE }}
 if SUCCESSES = 0 { if [count 1 in ROLL] = 0 {result: MISS }}
 if SUCCESSES > 0 {  if [count 1 in ROLL] <= SUCCESSES {result: SUCCESSES} }
}


loop DICE over {1..10} {
 output [evaluate DICE d6] named "[DICE]d6"
}

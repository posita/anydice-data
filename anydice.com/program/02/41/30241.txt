AC: 12
H: 8
MAGIC: (1d20+2+4-AC)>0
POISON: MAGIC*1d12
THUNDER: MAGIC*3d8
FIRE: MAGIC*1d10
RAY: MAGIC*2d6
DAGGER: ((1d20 + 2 - AC)>0)*(4+1d4)+((1d20 + 2 - AC)>0)*(1d4)
INSECTBITE: ((1d20 +3 +1 - 14)>0) * 2d4
DEPTH: 5

function: turns HP:n TURNS:n ATTACK COST:n{
  if (HP<0) {
    result: TURNS
  }
  if (TURNS>DEPTH) {
    result: TURNS+HP*COST
  }

  result: [turns HP-ATTACK TURNS+COST ATTACK COST]


  
}

output [turns H 0 RAY 1]
output [turns 9 0 INSECTBITE 1]

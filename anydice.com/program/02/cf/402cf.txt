AROLL: 1d20+2+1d6
DAM: 3d6+2
CDAM: [highest 3 of 6d6]+2
EVA: 10

function: damage R:n {
  if (R < EVA) {
    result: 0
  } else if (R < 20) {
    result: DAM
  } else {
    result: CDAM
  }
}

function: outcome S:n {
  if (S < 1) {
    result: 0
  } else if (S < 2) {
    result: d{0,0,1,1,1,2}
  } else {
    result: 2
  }
}

DROLL: [damage AROLL]
output DROLL named "Damage"
output [outcome (DROLL / 10)] named "Outcome (0 = fine, 1 = system trauma, 2 = destroyed)"

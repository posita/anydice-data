\Basic distribution\
output 1d6 named "No rerolls"

\Rerolling 1s\
function: reroll ones N:n 
{if N < 2 {result: [highest of d6 and N]} result: N } 
output 1
d [reroll ones d6] named "Rerolling 1s"

\Rerolling 2s - keeping highest\
function: reroll twos N:n

output 1d [reroll twos d6] named "Rerolling 2s - keeping highest"

\Rerolling 3s - keeping highest\
function: reroll threes N:n

output 1d [reroll threes d6] named "Rerolling 3s - keeping highest"

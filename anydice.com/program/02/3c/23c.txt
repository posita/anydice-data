function: ROLL:n vs AC:n {
 if ROLL >= THREAT & ROLL + ATTACK >= AC { result: [threat d20] }
 if ROLL + ATTACK >= AC { result: DAMAGE }
 result: 0
}

function: threat ROLL:n {
 if ROLL + ATTACK >= AC { result: CRITICAL d DAMAGE }
 result: DAMAGE
}

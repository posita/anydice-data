\Calculates the chance to hit of a monster attack. Accounts for SH and EV.\

\Monster input values\
HD: 13

\Player input values\
SH: 15
PREV_BLOCKS: 3
\end input values\


\functions, yo\
function: rtwo X:n {
    result: 1dX - 1
}

function: rtwoavg X:n across N:n {
    result: (NdX - 1) / N
}


function: shpierce MONS_HD:n prevblocks PBLOCKS:n {
    result: [rtwo 15 + (2 * MONS_HD / 3) + 5*PBLOCKS*PBLOCKS] 
}

function: shblock PL_SH:n {
    result: ([rtwoavg (2 * PL_SH) across 2] / 3) - 1
}


output [shpierce HD prevblocks PREV_BLOCKS] < [shblock SH] named "Did block"
output [shblock SH]named "SH Block"
output [shpierce HD prevblocks PREV_BLOCKS] named "SH Pierce"

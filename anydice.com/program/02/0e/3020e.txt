function: roll ROLL:s vs TARGET:n with optional reroll {
  SUM: ROLL + 0  \ force the sequence to be summed into a single number! \
  if SUM >= TARGET {
    \ no need to reroll, since we've already met the target \
    result: SUM
  } else {
    \ discard and reroll \
    result: 2d6+0
  }
}

output 2d6 named "2d6"
output [highest 2 of 3d6] named "3d6 garder les 2 plus haut"

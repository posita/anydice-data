function: check VALUES:s above THRESHOLD:n dc DC:n {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES > THRESHOLD)
 }
 result: COUNT >= DC
}

output [check 2d6 above 3 dc 2] named "low exertion, untrained PC, standard check"
output [check 2d6 above 2 dc 2] named "low exertion, trained PC, standard check"
output [check 2d6 above 1 dc 2] named "low exertion, expert PC, standard check"
output [check 3d6 above 3 dc 2] named "high exertion, untrained PC, standard check"
output [check 3d6 above 2 dc 2] named "high exertion, trained PC, standard check"
output [check 3d6 above 1 dc 2] named "high exertion, expert PC, standard check"
output [check 3d6 above 3 dc 3] named "high exertion, untrained PC, hard check"
output [check 3d6 above 2 dc 3] named "high exertion, trained PC, hard check"
output [check 3d6 above 1 dc 3] named "high exertion, expert PC, hard check"

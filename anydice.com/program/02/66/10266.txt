
AR: {10}
HR: {0..7}
DIE: 10
N: 4
\DIFS: {0 Fácil, Normal 1, Difícil 2, Muy Difícil 3}\
PLUS: 1

function: count VALUES:s in SEQUENCE:s plus PLUSES:n {
  if PLUSES=0 { 
   result: [count VALUES in SEQUENCE]  
  }   else { 
   result: [count {1..#VALUES+PLUSES} in SEQUENCE]
  }
}

NDF: 2
FUDGE: NDFd{-1..1}

 loop A over AR {
  loop H over HR {
   EA: [absolute ((A-12)/2)] +1
   output [count {1..H} in (NdA)] + FUDGE  named "1a[H] in [N]d[A] [NDF]FDG [PLUS]PLUS"
  }
 }

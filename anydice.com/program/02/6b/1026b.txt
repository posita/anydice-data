\ DIFS: {0 Trivial, Fácil 1, Media 2, Considerable 3, Difícil 4} \
\ Atributos: Pool que te permite por sesión bajar resultados de dados, o retirar dados (anunciando antes de tirar)? \

AR: {10}
HR: {1..9}
DIE: 10
N: 4

PLUS: 0

function: count VALUES:s in SEQUENCE:s plus PLUSES:n {
  if PLUSES=0 { 
   result: [count VALUES in SEQUENCE]  
  }   else { 
   result: [count {1..#VALUES+PLUSES} in SEQUENCE]
  }
}

NDF: 2
FUDGE: NDFd{-1..1}

 loop A over AR {
  loop H over HR {
   EA: [absolute ((A-12)/2)] +1
   output [count {1..H} in (NdA) plus PLUS] + FUDGE named "[H] in [N]d[A] F[NDF] P[PLUS]"
  }
 }

PROF_MOD: 2 * 2
DEX_MOD: 3
OTHER_MOD: 3
BONUS_DICE: 2d4

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}
function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output [highest 1 of 2d[reroll {1} on d20]] + PROF_MOD + DEX_MOD + OTHER_MOD + BONUS_DICE

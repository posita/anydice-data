function: explode VALUE:n {
  if VALUE = 10 {result: 10 + 1d10}
  if VALUE = 1 {result: 1 - 1d10}
  result: VALUE
}

function: average of DIE:d {
  MIN: 1@{DIE}
  R: {}
  loop P over {1..#{DIE}} {
    R: {P@{DIE}, R}
  }
  MAX: 1@{R}

  result: ((MAX + MIN) / 2)
}

function: IN:n floor VALUE:n {
 if IN < VALUE {result: VALUE}
 result: IN
}

function: IN:n ceiling VALUE:n {
 if IN > VALUE {result: VALUE}
 result: IN
}

function: adjust ARMOR:n pen PEN:n bypass BYPASS:n {
  if (ARMOR < BYPASS) {PASS: 0} else {PASS: ARMOR}
  if (PEN = 0) {result: PASS}
  HIGH: ((PASS + 1) / 2)
  LOW: (PASS / 2)
  if (HIGH > LOW) {result: HIGH} else {result: LOW}
}

function: crit DICE:s {
  result: (([count 6 in DICE] > 1) * 5)
}


function: attack DAMAGE:d plus SOLO:n at BASE:n armor ARMOR:d pen PEN:n bypass BYPASS:n range RANGE:s at INCREMENT:n evasion EVASION:d can DODGE:n headshot WEAK:n auto AUTO:n rof ROF:n ap AP:n {
  if ([average of (DODGE * EVASION)] > INCREMENT@RANGE) {DC: EVASION} else {DC: INCREMENT@RANGE}
  if (WEAK = YES) {BONUS: (BASE - 8)} else {BONUS: BASE}
  ATTACK: [explode 1d10] + BONUS
  HITMOD: [[(ATTACK - DC) floor 0] ceiling (AUTO + 1)]
  EFFECTIVE: [adjust ARMOR pen PEN bypass BYPASS]
  THROUGH: [((HITMOD * (DAMAGE + SOLO)) - EFFECTIVE) floor 0]
  if (WEAK = YES) {AMP: (THROUGH * 2)} else {AMP: THROUGH}
  FULL: (AMP + [crit DAMAGE])
  if (ROF > 1) {
    ABLATE: ([THROUGH ceiling 1] * (AP + 1))
    result: FULL + [attack DAMAGE plus SOLO at BASE armor (ARMOR - ABLATE) pen PEN bypass BYPASS range RANGE at INCREMENT evasion EVASION can DODGE headshot WEAK auto AUTO rof (ROF - 1) ap AP]
  }
  result: FULL
}

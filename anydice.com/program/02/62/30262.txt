function: count VALUES:s above THRESHOLD:n {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES > THRESHOLD)
 }
 result: COUNT
}

output [count 1d6 above 3] named "easy check, untrained PC"
output [count 1d6 above 4] named "easy check, trained PC"
output [count 1d6 above 5] named "easy check, expert PC"
output [count 2d6 above 3] named "standard check, untrained PC"

output [count 2d6 above 5] named "standard check, expert PC"

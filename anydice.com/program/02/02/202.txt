function: roll DICE:s against TARGET:n {
 if 0+DICE <= TARGET {
  if DICE = 1 > 1 | DICE = 2 > 1 | DICE = 3 > 1 { result: 2 }
  result: 1
 }
 if DICE = 4 > 1 | DICE = 5 > 1 | DICE = 6 > 1 { result: -2 }
 result: -1
}

POOL: 3d6

loop TARGET over {#POOL..[maximum of POOL]} {
 output [roll POOL against TARGET] named "vs [TARGET]"
}

function: dctwenty ATTACK:n vs DC:n with DAMAGE:n and BONUS:n heavy HEAVY:n{
  if ATTACK = 1 {
    result: 0
  }
  
  else if ATTACK = 20 {
    result: 2 + [strike (DAMAGE) attack result (ATTACK + BONUS) vs (DC) heavy HEAVY]
  }

  else if ATTACK + BONUS >=DC{
    result: [strike (DAMAGE) attack result (ATTACK + BONUS) vs (DC) heavy HEAVY]
  }

  else if ATTACK+BONUS < DC {
    result: 0
  }  
}

function: strike DAMAGE:n attack result OUTCOME:n vs DC:n heavy HEAVY:n{
  if OUTCOME<DC{
    result: 0
  }else if OUTCOME >=DC+5{
    result: DAMAGE + ((OUTCOME - DC) / 5) + HEAVY
  }else{
    result: DAMAGE + ((OUTCOME - DC) / 5)
  }
}

function: roll S:s default DEFAULT:n {

 \ find 2nd highest value \
 CURRENTH: 1
 CURRENTVALUE: 1@S
 loop P over {1..#S} {
  if CURRENTVALUE != P@S {
   CURRENTH: CURRENTH + 1
   CURRENTVALUE: P@S
  }
  if CURRENTH = 2 { result: P@S }
 }

  result: DEFAULT
}

output 4d{1,2,3,4,5,0} named "4d6, 6's are ignored (read as 0)"

output 3d6 named "3d6"

output 2@3d20 named "3d20, pick 2nd highest die"

output [roll 3d20 default 10] named "3d20, pick 2nd highest value"

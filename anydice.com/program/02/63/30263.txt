function: count VALUES:s below THRESHOLD:n {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES < THRESHOLD)
 }
 result: COUNT
}

output [count 1d6 below 4] named "easy check, untrained PC"
output [count 1d6 below 3] named "easy check, trained PC"
output [count 1d6 below 2] named "easy check, expert PC"
output [count 2d6 below 4] named "standard check, untrained PC"
output [count 2d6 below 3] named "standard check, trained PC"
output [count 2d6 below 2] named "standard check, expert PC"
output [count 3d6 below 4] named "hard check, untrained PC"
output [count 3d6 below 3] named "hard check, trained PC"
output [count 3d6 below 2] named "hard check, expert PC"

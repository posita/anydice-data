function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}
output [attack 1d20 plus 13 vs 15 for [highest 1 of 2d8] + 5 crit 2d8 on 18] named "attack regular"
output [attack 1d20 plus 8 vs 15 for [highest 1 of 2d8] + 15 crit 2d8 on 18] named "attack sharpshooter"
output [attack [highest 1 of 3d20] plus 13 vs 15 for [highest 1 of 2d8] + 5  crit 2d8 on 18] named "attack with advantage"
output [attack [highest 1 of 3d20] plus 8 vs 15 for [highest 1 of 2d8] + 15 crit 2d8 on 18] named "attack with advantage and sharpshooter"

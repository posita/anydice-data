function: S:s pick second highest value {

 \ find 2nd highest value \
 CURRENTH: 1
 CURRENTVALUE: 1@S
 loop P over {1..#S} {
  if CURRENTVALUE != P@S {
   CURRENTH: CURRENTH + 1
   CURRENTVALUE: P@S
  }
  if CURRENTH = 2 { result: P@S }
 }

 result: 3
}

output [3d6 pick second highest value] - [3d6 pick second highest value] named "3d6, pick 2nd highest minus 3d6, pick 2nd highest"
output 4d{+1, +1, 0, 0, -1, -1} named "4dF"
output 1d4 - 1d4 named "d4 minus d4"

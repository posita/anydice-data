\******** ENEMY ********\
AC: 15

function: beast attack plus ATTACK_MODIFIER:n to hit for DMG:d crit CRIT_DMG:d roll ROLL:n against AC:n{
    if ROLL = 1 {result:0}
    if ROLL = 20 {result: DMG + CRIT_DMG}
    if ROLL + ATTACK_MODIFIER >= AC {result: DMG}

    result: 0
}

output 1d[beast attack plus 3 to hit for 2d6+1 crit 2d6+1 roll [highest 2 of d20] against AC] named "Giant Owls (Advantage)"
output 1d[beast attack plus 3 to hit for 2d6+1 crit 2d6+1 roll d20 against AC] named "Giant Owls"
output 1d[beast attack plus 3 to hit for 2d6+1 crit 2d6+1 roll [lowest 2 of d20] against AC] named "Giant Owls (Disadvantage)"
output 1d[beast attack plus 4 to hit for 2d4+2 crit 2d4+2 roll [highest 2 of d20] against AC] named "Wolves (Advantage)"
output 1d[beast attack plus 4 to hit for 2d4+2 crit 2d4+2 roll d20 against AC] named "Wolves"
output 1d[beast attack plus 4 to hit for 2d4+2 crit 2d4+2 roll [lowest 2 of d20] against AC] named "Wolves (Disadvantage)"
output 1d[beast attack plus 6 to hit for 3d6+4 crit 3d6+4 roll [highest 2 of d20] against AC] named "Charging Cows (Advantage)"
output 1d[beast attack plus 6 to hit for 3d6+4 crit 3d6+4 roll d20 against AC] named "Charging Cows"
output 1d[beast attack plus 6 to hit for 3d6+4 crit 3d6+4 roll [lowest 2 of d20] against AC] named "Charging Cows (Disadvantage)"
output 1d[beast attack plus 6 to hit for d6+4 crit d6+4 roll [highest 2 of d20] against AC] named "Cows (Advantage)"

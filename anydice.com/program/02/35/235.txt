function: roll S:s with control number C:n {

 \ find 2nd highest value \
 CURRENTH: 1
 CURRENTVALUE: 1@S
 loop P over {1..#S} {
  if CURRENTVALUE != P@S {
   CURRENTH: CURRENTH + 1
   CURRENTVALUE: P@S
  }
  if CURRENTH = 2 { result: P@S }
 }

 \ no 2nd highest value \
 if 1@S = 6 {
  if C = 2 | C = 4 | C = 6 { result: 0}
  result: 6
 }


 if 1@S = C | C = 6 { result: 3 }
 if 1@S > C { result: 6 }
 if 1@S < C { result: 0 }
}

output [roll 3d6 with control number d6]

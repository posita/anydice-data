BONUS_A: 10
BONUS_B: 0
PROGRESS_A: 0
PROGRESS_B: 0

function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 0
 if SUCCESSES > 0 { result: ROLL}
 result: 0
}

loop DICE over {1..2} {
 RESULT: (1d100 + BONUS_A) - (1d100 + BONUS_B)
 PROGRESS_A: PROGRESS_A + [evaluate RESULT]
 PROGRESS_B: PROGRESS_B + [evaluate -RESULT]
}

output [lowest of PROGRESS_A and 120 + BONUS_A]
output [lowest of PROGRESS_B and 120 + BONUS_B]

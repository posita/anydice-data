\Basic distribution\
output 2d6 named "No rerolls"

\Rerolling 1s\
function: reroll ones N:n 
{if N = 1 { result: 1d6 } result: N } 
output 2d [reroll ones d6] named "Rerolling 1s"

\Rerolling 2s - keeping highest\
function: reroll twos N:n 
{if N < 3 {result: [highest of d6 and N]} result: N}
output 2d [reroll twos d6] named "Rerolling 2s - keeping highest"

\Rerolling 3s - keeping highest\
function: reroll threes N:n 
{if N < 4 {result: [highest of d6 and N]} result: N}
output 2d [reroll threes d6] named "Rerolling 3s - keeping highest"

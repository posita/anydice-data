AC: 1
HITMOD: 7
DMGMOD: 5
HITSHARP: -5
DMGSHARP: 10
HBHIT: 2d6+DMGMOD

function: roll A:n {
 if A+HITMOD < AC {
  result: 0
 }
 if A+HITMOD >= AC & A < 20 {
  result: 1
 }
 if A = 1 {
  result: 0
 }
 if A = 20 {
  result: 2
 }
}
function: rollsharp A:n {
 if A+HITMOD+HITSHARP < AC {
  result: 0
 }
 if A+HITMOD+HITSHARP >= AC & A < 20 {
  result: 1
 }
 if A = 1 {
  result: 0
 }
 if A = 20 {
  result: 2
 }
}
function: hbattack {result:
 [roll 1d20]*HBHIT
}
function: hbturn {result:
 [hbattack]+[hbattack]+[hbattack]
}
function: hbattacksharp {result:
 [roll 1d20]*(HBHIT+DMGSHARP)
}
function: hbturnsharp {result:
 [hbattacksharp]+[hbattacksharp]+[hbattacksharp]
}
output [hbturn] named "Normal"
output [hbturnsharp] named "Sharpshooter"

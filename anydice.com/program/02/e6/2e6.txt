function: ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= THREAT {
  if ROLL + ATTACK >= DEFENSE { result: CRITICAL }
  result: DAMAGE
 }
 if ROLL + ATTACK >= DEFENSE { result: DAMAGE }
 result: MISS
}


function: vital strike {
 result: [d20 vs DEFENSE]
}

function: apply sneak attack to ATTACKRESULT:n
{
if 1 > 0 { result: ATTACKRESULT + SNEAKDAMAGE }
result: ATTACKRESULT
}

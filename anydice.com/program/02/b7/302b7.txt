function: get oofs in SEQUENCE:s {
  result: [count {1} in SEQUENCE] 
} 

function: get successes in SEQUENCE:s {
  result: [count {4,5,6} in SEQUENCE]
}

function: get double successes in SEQUENCE:s {
  result: [count {4,5} in SEQUENCE] + ([count {6} in SEQUENCE] * 2)
}

function: get doubles success chance for TN:n in SEQUENCE:s {
  result: [get double successes in SEQUENCE] >= TN
}

function: get success chance for TN:n in SEQUENCE:s {
  result: [get successes in SEQUENCE] >= TN
}

SEQUENCE: 1d6
output [get oofs in SEQUENCE] named "[1d6] # of OOFS"
output [get successes in SEQUENCE] named "[1d6] # of POWS"
output [get double successes in SEQUENCE] named "[1d6] # of POWS (6 = double)"


output [get doubles success chance for 2 in SEQUENCE] named "[1d6] Success chance for TN2 (6 = double)"
output [get success chance for 2 in SEQUENCE] named "[1d6] Success chance for TN2"

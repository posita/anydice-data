function: best or worst NUM:n in A:s {
  SUM: 0
  if( NUM >  #A ) { NUM:  #A }
  if( NUM < -#A ) { NUM: -#A }

  if( NUM > 0 ){
    loop P over {1..NUM}
      { SUM: SUM + P@A }
  }else{
    loop P over {(#A + NUM + 1)..#A}
      { SUM: SUM + P@A }
  }
  result: SUM
}

output [best or worst 10 in 20d6]
output 3 * 10d6 / 2

AC: 10
ATA: {10,10,5}
ATB: {6,6,6,1}
DMG: 1d8
DMGB: 8
CT: 19
CB: 1
CM: 2

function: A:n B:n hit C:n {
result: A=20 | ((A+B)>=C)}

function: A:n B:n atk D:n {
result: [A B hit AC]*(DMG+D)*((A>=CT & [1d20 B+CB hit AC])*(CM-1)+1)
}

function: A iterative B:n {
I:0
X:0
loop I over A{
X:X+[1d20 I atk B]
}
result: X
}

output [ATB iterative DMGB]-[ATA iterative DMGB]

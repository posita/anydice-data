\**** Damage calculator ****\

function: tattackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: TCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + TATTACKMOD + 2 - 2>= AC { result: TDAMAGE }
 result: 0
}

function: rattackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: RCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + 1 >= AC { result: RDAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
AC: d{22}

TATTACKMOD: 6
ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 4
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
TATTACKTIMES: 2
RAGE: 2

TCRITDAMAGE: 2d8 + 10
TDAMAGE: 4d8 + 10
RCRITDAMAGE: 1d12 + RAGE + ATTACKMOD + 1
RDAMAGE: 2d12 + RAGE + ATTACKMOD + 1

output TATTACKTIMESd[tattackroll d20 vs AC] named "Dimitiri"

output ATTACKTIMESd[rattackroll d20 vs AC] named "Fighter"

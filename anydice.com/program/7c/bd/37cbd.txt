MAX_DICE: 6
MIXED: 7
STRONG: 10

function: thresholds X:n {
  if X > STRONG {
    result: 2
  }
  if X > MIXED {
    result: 1
  }
  result: 0
}
  
loop N over {-MAX_DICE..MAX_DICE} {
  if N < -1 {
    M: N * -1
    POWER: M - 1
    output [thresholds [lowest 2 of (1+M)d6]] named "-[POWER] Power: [M]D6"
  }
  if N > -0 {
    POWER: N - 1
    output [thresholds [highest 2 of (1+N)d6]] named "+[POWER] Power: [N]D6"
  }
}

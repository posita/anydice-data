 if A > B { result: A }
 result: B
}

function: highest of A:n and B:n and C:n {
 if A > B & A > C { result: A }
 if B > A & B > C { result: B }
result: C
}

function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d{
    if ROLL = 1 {
        result: 0
    }
    if ROLL = 20 {
        if ROLL + BONUS >= AC{
            result: DMG * 2 + CRIT_DMG  
        }
        result: DMG
    }
    if ROLL + BONUS >= AC + 10 {
        result: DMG * 2 + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}


function: multiattack ROLL:n plus BONUS:n vs AC:n hit HIT:d miss MISS:d crit CRIT:d {
    if ROLL = 1 {
        result: MISS
    }
    if ROLL = 20 {
        if ROLL + BONUS >= AC{
            result: CRIT 
        }
        result: HIT
    }
    if ROLL + BONUS >= AC + 10{
        result: CRIT
    }
    if ROLL + BONUS >= AC {
        result: HIT
    }
    result: MISS
}


function: save ROLL:n plus BONUS:n vs DC:n for DAMAGE:d save SAVE:d {
    if ROLL + BONUS < DC {
         result: DAMAGE
    }
    result: SAVE
}

function: check ROLL:n plus BONUS:n vs DC:n success DAMAGE:d fail FAIL:d { 
    if ROLL + BONUS >= DC{
        result: DAMAGE
    }

function: implode DIE:d with M:n {
 result: [implode rec DIE with M]
}

function: implode rec N:n with MM:n {
  if N = 0 & MM > 1 {
    MM: MM - 1
    result: [
      implode rec DIE with MM
    ]
  }
 result: N*MM
}

SIZE: 8
BIAS: 3
TRAIT: (d(SIZE)-BIAS)
AREA_A: d4
AREA_B: d3+1

output TRAIT*AREA_A named "product-A"
output [implode TRAIT with AREA_A] named "imploded product-A"
output d{(1-BIAS)*[maximum of AREA_A]..(SIZE-BIAS)*[maximum of AREA_A]} named "uniform reference-A"

output TRAIT*AREA_B named "product-B"
output [implode TRAIT with AREA_B] named "imploded product-B"
output d{(1-BIAS)*[maximum of AREA_B]..(SIZE-BIAS)*[maximum of AREA_B]} named "uniform reference-B"

\
 Blades in the Dark
 0: Failure. Dire consequences
 1: Partial success, with consequences
 2: Partial success. Continue at +1 or accept light consequences
 3: Clean success.
 4: Critical success!
\

function: blades ROLL:s {
 if 1@ROLL = 10 {
  result: 3 + (2@ROLL = 10)
 }
 if 1@ROLL >= 8 {
  result: 2
 }
 result: 1@ROLL >= 5
}

output [blades 2@2d10] named "0d"
loop P over {1..7} {
 output [blades Pd10] named "[P]d"
}

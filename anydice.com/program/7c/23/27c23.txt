function: evaluate DICE:d{
 \if DICE=[maximum of {DICE}]{ result: 2 }\
 if {DICE}>10 { result: 1 }
 \if DICE=[minimum of DICE] { result: -1 }\
 result: 0
}

DICE: 2d10
output [evaluate DICE]

\loop MOD over {1..5}{ output [evaluate DICE+MOD] named "[DICE]+[MOD]" }\

\
loop MOD over {1..7}{
loop ARMOR over {6..10}{
 output DICE+MOD>ARMOR named "Lvl[MOD] vs [ARMOR]"
}
}

\ -------------------------------------------------- Probabilties for: 1D8 + 1D12 ------------------------------------------------- \
\ output 1d8 + 1d12 \

\ -------------------------------------------------- Probabilities for: D6 Dice Pool ------------------------------------------------- \
\function: dicepool ROLL:s {
 SUCCESSES: ROLL >= 4
 result: SUCCESSES
}

loop DICE over {1..7} {
 output [dicepool DICE d6] named "[DICE]d"
}\

\ -------------------------------------------------- Probabilities for: D6 Dice Pool with higehst counts double ------------------------------------------------- \
function: doubledicepool ROLL:s {
 SUCCESSES: ROLL >= 4
 if SUCCESSES > 0 { result: SUCCESSES + (ROLL = 6) }
 result: SUCCESSES
}

loop DICE over {1..7} {
 output [doubledicepool DICE d6] named "[DICE]d"
}

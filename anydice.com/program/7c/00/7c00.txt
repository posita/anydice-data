function: count DICE:n A:n B:n {
 result: (DICE >= A & DICE <= B)
}

function: proper DICE:d Q:n COST:n {
 result: [count DICE Q*COST (Q*COST)+COST-1]
}

DICE: 6d6
COST: 15

loop I over {0..[maximum of DICE]/COST} {
 W: I*COST
 Z: [lowest of (I*COST)+COST-1 and [maximum of DICE]]
 output [proper DICE I COST] named "[I]= [W]-[Z] (cost [COST])"
}

set "maximum function depth" to 5

function: expool DIE:d TAR:n EXPL:n {
  result: [expool helper DIE]
}

function: expool helper R:n {
  if R <= EXPL { result: 1 + [expool helper DIE] }
  if R <= TAR { result: 1 }
  result: 0
}

loop N over {1..8} {
  output Nd[expool 1d6 3 1]
}

\loop N over {1..8} {
  output Nd(1d6 <= 3)
}\

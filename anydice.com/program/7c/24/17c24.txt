POOL: 13 \ Dice pool\
EDGE: 4


set "maximum function depth" to 2

function: explode N:n {
 if N = 6 { result: 1 + [explode d6] }
 result: N = 5
}

output POOLd(d6 > 4) named "Basic roll"
output (POOL + EDGE)d[explode d6] named "Edge before roll"
output POOLd(d6 > 4) + (EDGE)d[explode d6] named "Edge after roll"

ROLL: POOLd(d6 > 4)
FAIL: POOL - ROLL
output FAILd(d6 > 4) + ROLL named "Rerolling failures"

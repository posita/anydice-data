\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE }
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}



function: altattackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: ALTCRITDAMAGE }
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: ALTDAMAGE }
 result: 0
}


function: alphaattackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: ALTCRITDAMAGE }
 if ROLL = 1 { result: 0 }
 if ROLL + ATTACKMOD >= AC { result: ALTDAMAGE }
 result: 0
}

function: sneakattack ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: SNEAK_ATTACK }
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: CRIT_SNEAK_ATTACK }
 result: 0
}

function: spiritualattackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: SPCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: SPDAMAGE }
 result: 0
}

function: reroll D:d under N:n{
   result: [rerollcore D die D under N]
}

function: rerollcore R:n die D:d under N:n{
   if R < N { result: D } else {result: R}
}

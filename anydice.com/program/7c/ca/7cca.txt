function: hit TARGET:n with DICE:s plus BONUS:n reroll REROLL:s {

  \ FAIL_RESULT_NUMBER: TARGET-1 \
  FAIL_RESULT_NUMBER: -99

  \~~ If we rolled snake-eyes ~~~~~~~~~~~~~~~~\
  if [count 1 in DICE] >= 2 {
    result: FAIL_RESULT_NUMBER \~~ AutoFail ~~~\
  }
  
  \~~ Calculate the total ~~~~~~~~~~~~~~~~~~~~\
  TOTAL: {1,2}@[sort DICE]

  \~~ If we don't have a re-roll, just roll ~~~\
  if REROLL = 0 {
    if TOTAL + BONUS >= TARGET {
      result: TOTAL + BONUS \~~ Hit ~~~~~~~~~~\
    } else {
      result: FAIL_RESULT_NUMBER \~~ Miss ~~~~\
    }
  }

  \~~ We hit the target with our first roll ~~\
  if TOTAL+BONUS >= TARGET {
    result: TOTAL + BONUS \~~ Hit ~~~~~~~~~~~~\
  } else {
    \~~ We under-rolled... ~~~~~~~~~~~~~~~~~~~\
    \~~ If the re-roll hits ~~~~~~~~~~~~~~~~~~\
    if REROLL + BONUS >= TARGET {
      result: REROLL + BONUS \~~ Hit ~~~~~~~~~\
    } else {
      result: FAIL_RESULT_NUMBER \~~ Miss ~~~~\
    }
  }

}

function: die N:n trait X:n a A:n b B:n c C:n {
if N=1 {if A+X >= 5+2*N {result: 100} else {result: 0}}
if N=2 {if A+B+X >= 5+2*N {result: 100} else {result: 0}}
if N=3 {if A+B+C+X >= 5+2*N {result: 100} else {result: 0}}
else {result: 99}
}


loop N over {1..3} {
loop X over {-3..3} {
output [die N trait X a 1d6 b 1d6 c 1d6] named "[N]d6+[X]"
}}


function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}



output ([reroll {1} on d7]+1) named "reroll 1s once"

output ([reroll {1} on d7]+1)/2 named "reroll, halve result"

output ([reroll {1} on d10]+1) named "reroll 1s once d10"

output ([reroll {1} on d10]+1)/2 named "reroll, weekly d10"

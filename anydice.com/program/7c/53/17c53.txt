\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 13-20\
ENEMYAC: d{10..20}

ATTACKMOD: 5
DMGBONUS: 5
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d8 + DMGBONUS 
CRITDAMAGE: 2d8 + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC]named "Two-handed spear"

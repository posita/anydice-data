function: test X with N {
  result: [highest of [explode 1dX] and [explode 1d6]] + N
}

loop X over {4, 6, 8, 10, 12} {
  if X = 4 { output [test 4 with -2] named "1d4-2" }
  output [test X with 0] named "1d[X]"
  if X = 12 {
    loop N over {1..4} {
      output [test X with N] named "1d[X]+[N]"
    }
  }
}

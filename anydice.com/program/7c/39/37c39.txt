function: iron X:n vs Y:s {
  if X > 1@Y { result: 2 }
  if X > 2@Y { result: 1 }
  result: 0
}
loop N over {0..5} {
output [iron 1d6+N vs 2d10] named "1d6 plus [N] > 2d10"
}
loop N over {-2..4} {
output [iron 2d10+N vs {14, 9}] named "2d10 plus [N] >= 10/15"
}
loop N over {-1..3} {
  output [iron 2d6+N vs {9, 6}] named "pbta 2d6 plus [N]"
}

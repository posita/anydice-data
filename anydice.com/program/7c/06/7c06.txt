function: test X:n with N:n {
  result: [highest of [explode 1dX] and [explode 1d6]] + N
}

loop X over {4, 6, 8, 10, 12} {
  if X = 4 { output [test 4 with -2] >= 4 named "Chance of Success with 1d4-2" }
  output [test X with 0] >= 4 named "Chance of Success with 1d[X]"
  if X = 12 {
    loop N over {1..4} {
      output [test X with N] >= 4 named "Chance of Success with 1d[X]+[N]"
    }
  }
}

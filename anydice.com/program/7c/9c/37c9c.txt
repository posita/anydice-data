function: PRF:s crits over DMG:s{
if PRF >= DMG
{
result: DMG
}
else
{
result: 0
}
}

loop P over {4, 6, 8, 10, 12}{
loop D over {4, 6, 8, 10, 12}{
output [1dP crits over 1dD] named "d[P] proficiency d[D] damage"
loop S over{4, 6, 8, 10, 12}{
output [[highest of 1dP and 1dS] crits over 1dD] named "d[P] proficiency, d[S] superiority, d[D] damage"
}}}

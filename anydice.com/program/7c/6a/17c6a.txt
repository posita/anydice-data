function: floor A:n div B:n {
  if B!=0 {
    if (A*B)>=0 {result: A/B}
    result: -(-A/B+1)
  }
  result: {}
}

function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

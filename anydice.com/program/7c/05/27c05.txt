ROLL: 5

function: doubles X:n {
  if X >= 2 {
    result: 1
  }
  result: 0
}

function: triples X:n {
  if X >= 3 {
    result: 1
  }
  result: 0
}

output [doubles ROLLd{0, 0, 0, 0, 0, 1}] named "Doubles"
output [triples ROLLd{0, 0, 0, 0, 0, 1}] named "Triples"

\
loop I over {2..8} {
  loop J over {1..6} {
    if I >= J {
      output [highest J of Id6] named "[I]k[J]"
    }
  }
}

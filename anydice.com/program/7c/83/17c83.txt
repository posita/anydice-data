\**** Damage calculator normal Vs advantage ****\

function: attackrollblast ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGEBLAST } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMODBLAST + ACCBONUSBLAST >= AC { result: DAMAGEBLAST }
 result: 0
}

function: attackrollmeleew ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGEMELEEW } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMODMELEE + ACCBONUS >= AC { result: DAMAGEMELEEW }
 result: 0
}


function: attackrollmelee ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGEMELEE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMODMELEE + ACCBONUS >= AC { result: DAMAGEMELEE }
 result: 0
}

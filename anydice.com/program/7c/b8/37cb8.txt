MAX_DICE: 6
MIXED: 4
STRONG: 5

function: thresholds X:n {
  if X > STRONG {
    result: 2
  }
  if X > MIXED {
    result: 1
  }
  result: 0
}
  
loop N over {-MAX_DICE..MAX_DICE} {
  if N < -1 {
    M: N * -1
    POWER: M - 1
    output [thresholds [lowest 1 of Md6]] named "-[POWER] Power: [M]D6"
  }
  if N > -0 {
    POWER: N - 1
    output [thresholds [highest 1 of Nd6]] named "+[POWER] Power: [N]D6"
  }
}

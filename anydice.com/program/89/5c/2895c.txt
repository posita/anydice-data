function: teamwork of A:s and B:s
{

MATCH: 0

loop I over {1..#A}
{

loop L over {1..#B}
{

if I@A = L@B 
{
if I@A >= 4 {MATCH:2}
if MATCH != 2 & I@A < 4 {MATCH: 1}
}

}

}

result: MATCH

}


loop POOLONE over {1..3}
{
loop POOLTWO over {1..3}
{
output [teamwork of POOLONEd6 and POOLTWOd6] named "[POOLONE]d6 with [POOLTWO]d6"
}
}

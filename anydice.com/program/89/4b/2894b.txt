function: attack ROLL:n with TOHIT:n to hit and crit on CRITRANGE:n for CRITABLEDAM:d plus HITDAM:d with CRITTIMES:n crit multiple {
  if ROLL = 1 { result: 0 }
  else if ROLL >= CRITRANGE {
    CRIT: 1
    CRITDMG: HITDAM
    loop N over {1..CRITTIMES} {
      CRITDMG: CRITDMG + CRITABLEDAM
    }
    result: CRITDMG
  } else if ROLL + TOHIT >= AC { result: CRITABLEDAM + HITDAM }
  else { result: 0 }
}

function: oneattack ROLLA:n ROLLB:n with TOHIT:n to hit and crit on CRITRANGE:n for CRITABLEDAM:d plus HITDAM:d with CRITTIMES:n crit multiple {
  DMG: [attack ROLLA with TOHIT to hit and crit on CRITRANGE for CRITABLEDAM plus HITDAM with CRITTIMES crit multiple]
  if (GWM) {
    if (ROLLA >= CRITRANGE) {
      DMG: DMG + [attack ROLLB with TOHIT to hit and crit on CRITRANGE for CRITABLEDAM plus HITDAM with CRITTIMES crit multiple]
    }
  }
  result: DMG
}

function: twoattack ROLLA:n ROLLB:n ROLLC:n with TOHIT:n to hit and crit on CRITRANGE:n for CRITABLEDAM:d plus HITDAM:d with CRITTIMES:n crit multiple {

  DMG: DMG + [attack ROLLB with TOHIT to hit and crit on CRITRANGE for CRITABLEDAM plus HITDAM with CRITTIMES crit multiple]
  if (GWM) {
    if (ROLLA >= CRITRANGE) {
      DMG: DMG + [attack ROLLC with TOHIT to hit and crit on CRITRANGE for CRITABLEDAM plus HITDAM with CRITTIMES crit multiple]

      DMG: DMG + [attack ROLLC with TOHIT to hit and crit on CRITRANGE for CRITABLEDAM plus HITDAM with CRITTIMES crit multiple]
    }
  }
  result: DMG
}

function: stress X:n {
    if X = 1 {
        OUT: [explode d10 times 2]
        result: [clamp OUT]
    }
    result: X

}

function: explode X:n times MULTI:n{
    if X = 1 {

        result: [explode d10 times MULTI * 2]
    }
    result: X * MULTI
}

function: clamp X:n {
    if X > 100 {
            result: 100
    }
    result: X
}

output [stress d10-1]

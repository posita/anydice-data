\**** Damage calculator ****\
function: attackroll ROLL:n and BLESS:n vs AC:n {
 if ROLL >= CRIT { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + BLESS + PROF + MOD >= AC { result: DAMAGE }
 result: 0
}

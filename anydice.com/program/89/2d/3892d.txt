function: X:d N:n times {
  A: 0
  loop B over {1..N} {
    A: A + X
  }
  result: A
}

function: reroll SAMPLE:n floor MINIMUM:n from DIE:d {
  if SAMPLE <= MINIMUM {result: DIE}
  result: SAMPLE
}
function: gwf DIE:d {
  result: [reroll DIE floor 2 from DIE]
}
HFL: [reroll ATK floor 1 from ATK]
HFLA: [reroll ADV floor 1 from ADV]

function: crit                           DICE:d {
  result:[crit DICE plus 0]
}
function: crit                           DICE:d plus EXTRA:n {
  result:[crit ATK                   for DICE   plus EXTRA]
}
function: crit ROLL:d                for DICE:d plus EXTRA:n {
  result:[crit ROLL   at 20          for DICE   plus EXTRA]
}
function: crit ROLL:d at THRESHOLD:n for DICE:d plus EXTRA:n {
  result:[crit ROLL   at THRESHOLD   for DICE   plus EXTRA   and 0]
}
function: crit ROLL:n at THRESHOLD:n for DICE:d plus EXTRA:n and BONUS:n {
  if           ROLL  >=  THRESHOLD {
    result:                             [DICE 2 times]+EXTRA  +  BONUS
  }
  result:                                DICE   +    EXTRA
}

function: save X:n plus SAVE DC for half DICE:n {
  if X+SAVE >= DC {result: DICE/2}
  result: DICE
}
function: save X:n plus SAVE DC or take DICE:n plus EXTRA {
  if X+SAVE >= DC {result: 0}
  result: DICE+EXTRA
}
function: save X:n plus SAVE DC or take DICE:n {
  result: [save X plus SAVE DC or take DICE plus 0]
}

function: ROLL:n vs AC:n {
 if ROLL >= THREAT & ROLL + ATTACK >= AC {
  result: [threat d20]
 }
 if ROLL + ATTACK >= AC | ROLL = 20 { result: DAMAGE }
 result: 0
}

function: threat ROLL:n {
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: CRITICAL d DAMAGE
 }
 result: DAMAGE
}

AC: d{14..20}
POWER_ATTACK: d{1..5}
STRENGTH: 2
ATTACK: 5 + STRENGTH

DAMAGE: 1d10 + STRENGTH
THREAT: 19
CRITICAL: 2
output [d20 vs AC] named "War Sword 1H"

DAMAGE: 1d10 + STRENGTH + POWER_ATTACK
THREAT: 19
CRITICAL: 2
output [d20 vs AC] named "War Sword 1H /W Power Attack"

DAMAGE: 1d10 + STRENGTH * 3/2
THREAT: 19
CRITICAL: 2
output [d20 vs AC] named "War Sword 2H"

DAMAGE: 1d10 + STRENGTH + (POWER_ATTACK * 2)
THREAT: 19
CRITICAL: 2
output [d20 vs AC] named "War Sword 2H /W Power Attack"

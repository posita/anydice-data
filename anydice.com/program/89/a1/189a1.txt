function: reroll DICE:n if lt NUM:n {
  if DICE >= NUM {
    result: DICE
  } else {
    result: 4d(d10 > 5)
  }
}

output 4d(d10 > 5) named "base pool"
output [reroll 4d(d10 > 5) if lt 1] named "reroll on failure"
output [reroll 4d(d10 > 5) if lt 3] named "reroll if 2 or fewer successes"
output 4d(1@2d(d10 > 5)) named "reroll individuals"

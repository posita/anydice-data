function: attack ROLL:n with TOHIT:n to hit for BASEDMG:d with crit on CRITRANGE:n for CRITDMG:n {
  if ROLL = 1 { result: 0 }
  else if ROLL >= CRITRANGE { result: BASEDMG + CRITDMG }
  else if ROLL + TOHIT >= AC { result: BASEDMG }
  else { result: 0 }
}

function: firsthit ROLL:n with TOHIT:n to hit for BASEDMG:n with crit on CRITRANGE:n for CRITDMG:n {
  if ROLL = 1 {
    result: 0
  }
  if ROLL < AC {
    result: 0
  }
  result: [attack ROLL with TOHIT to hit for BASEDMG with crit on CRITRANGE for CRITDMG]
}

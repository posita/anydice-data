\**** Damage calculator ****\

function: attackroll WROLL:n vs AC:n {
 if WROLL >= CRIT { result: WCDAM } 
 if WROLL = 1 { result: 0 }
 if WROLL + WPROF + WAM >= AC { result: WDAM }
 result: 0
}

function: attackrolld DROLL:n vs AC:n {
 if DROLL >= CRIT { result: DCDAM } 
 if DROLL = 1 { result: 0 }
 if DROLL + DPROF + DAM >= AC { result: DDAM }
 result: 0
}

function: attackrollp PROLL:n vs AC:n {
 if PROLL >= CRIT { result: PCDAM } 
 if PROLL = 1 { result: 0 }
 if PROLL + PPROF + PAM >= AC { result: PDAM }
 result: 0
}

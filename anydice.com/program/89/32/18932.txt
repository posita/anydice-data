function: swear A:n over X:n and Y:n {
  Z: {X,Y}
  if A > 1@Z { result: 2 }
  if A > 2@Z { result: 1 }
  result: 0
}

loop S over {1..5} { output [ swear 1d6+S over 1d10 and 1d10] named "Vanilla+[S]" }

loop S over {1..5} { output [ swear 1d8+S over 1d10 and 1d10] named "Bigger Action+[S]" }

loop S over {1..5} { output [ swear 1d6+S over 1d8 and 1d10 ] named "Smaller Challenge+[S]" }

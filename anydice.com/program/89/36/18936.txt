\
\\ 6 Piece Polyhedral Dice Pool 
\


\ Variables \
DICE: {20,12,10,8,6,4}

\ Function for the dice roll \
function: roll NUM:n {

  TOTAL: 0

  loop N over {1..NUM} {
    TOTAL: TOTAL + (d(N@DICE)<4)
  }

  result: TOTAL
}

\ Loops to test a variety of combinations \
loop N over {1..#DICE} {
  output [roll N] named "[N]"
}

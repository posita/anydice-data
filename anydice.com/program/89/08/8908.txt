function: rerollone N:n for DICE:d {
  if N > 1 { result: N}
  result: DICE
}

function: rerollones N:n for DICE:d {
  SUM: 0
  loop COUNT over {1..N} {
    SUM: SUM + [rerollone DICE for DICE]
  }
  result: SUM
}

output [rerollones 10 for d8] named "10d8 reroll 1 once"
output 10d8+10 named "10d8 reroll all 1s"
output 10d8 named "10d8"

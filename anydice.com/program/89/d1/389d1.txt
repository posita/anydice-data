function: match SEQUENCE:s with difficulty D:n {
	COUNT:0
	X:0
	Y:0
	MAX:0
	loop P over {1..#SEQUENCE} {
		if X=0 {
			X:P@SEQUENCE
		} else {
			if P@SEQUENCE >= D{
				Y:P@SEQUENCE
				if X=Y {
					COUNT:COUNT+1
				} else {
					if COUNT > MAX {
						MAX: COUNT
					}
					COUNT:0
				}
				X:P@SEQUENCE
			} else {
				if COUNT > MAX {
					MAX: COUNT
				}
				result: MAX+1
			}
		}
	}
	if COUNT > MAX {
		MAX: COUNT
	}
	result: MAX+1
}
output [match 8d10 with difficulty 1]

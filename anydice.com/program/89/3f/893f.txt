function: ROLL:n vs AC:n {
 if ROLL >= THREAT & ROLL + ATTACK >= AC {
  result: [threat d20]
 }
 if ROLL + ATTACK >= AC | ROLL = 20 { result: DAMAGE }
 result: 0
}

function: threat ROLL:n {
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: CRITICAL d DAMAGE
 }
 result: DAMAGE
}

AC: d{14..20}
STRENGTH: 3
ATTACK: 1 + STRENGTH

DAMAGE: 1d10 + STRENGTH
THREAT: 19
CRITICAL: 2
output [d20 vs AC] named "Bardiche"

DAMAGE: 1d12 + STRENGTH
THREAT: 20
CRITICAL: 2
output [d20 vs AC] named "Lucerne Hammer"

THREAT: 20
CRITICAL: 3

function: doubta R:n again N:n {
 if N >= R { result: 1 + [doubta R again d8] }
 result: N >= 6
}

DOUBT: [doubta8 again d8]

loop N over {1..5} {
 output [lowest of 8 and NdDOUBT] named "[N]d"
}

function: doubtb R:n again N:n {

 result: N >= 6
}

DOUBT: [doubtb10 again d10]

loop N over {1..5} {
 output [lowest of 10 and NdDOUBT] named "[N]d"
}

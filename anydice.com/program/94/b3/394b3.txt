function: dupes in DICE:s {
  D: 0
  loop X over {2..#DICE} {
    if ((X-1)@DICE = X@DICE) { D: D + 1}
  }
  result: D
}

function: bounces DICE:s {
  result: [dupes in DICE] > 0
}

function: elemental adept D:n {
  if D = 1 { result: 2 }
  result: D
}

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

DICE: 3
output [bounces DICEd8] named "Base bounce chance"
output [bounces DICEd[elemental adept d8]] named "Elemental Adept"
output [bounces DICEd[reroll {1} on d8 to d8]] named "Flames of Phlegethos"
output [bounces DICEd[elemental adept [reroll {1} on d8 to d8]]] named "FoP followed by EA"

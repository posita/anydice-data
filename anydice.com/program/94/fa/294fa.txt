function: attackroll BAB with ROLL:n vs AC:n {
 if ROLL >= CRITROLL & ROLL + BAB + STRMOD >= AC 
  {    
    result: [critconfirm BAB with d20] 
  } 
 if ROLL = 1 { result: 0 }
 if ROLL + BAB + STRMOD >= AC | ROLL = 20 
  { 
    result: DAMAGE
  }
 result: 0
}

function: perroundroll BAB with ROLL:n vs AC:n {
 if ROLL >= CRITROLL & ROLL + BAB + STRMOD >= AC 
  {    
    result: PERROUND 
  } 
 if ROLL = 1 { result: 0 }
 if ROLL + BAB + STRMOD >= AC | ROLL = 20 
  { 
    result: PERROUND
  }
 result: 0
}

function: multiattackroll {
 FINALDAMAGE: 0
 ONCEPERROUNDDAMAGE: 0
 DAMAGENOTDEALT : 1
 loop N over ATTACKS{ 
   ONCEPERROUNDDAMAGE: [perroundroll N with d20 vs ENEMYAC]     
   FINALDAMAGE: [attackroll N with d20 vs ENEMYAC] + FINALDAMAGE   
  }
  FINALDAMAGE: FINALDAMAGE + ONCEPERROUNDDAMAGE
  result: FINALDAMAGE
}

function: critconfirm BAB with ROLL:n {
 if ROLL = 1 { result: DAMAGE}
 if ROLL + BAB + STRMOD >= AC | ROLL = 20 { result: CRITDAMAGE}
 result: DAMAGE
}

ATT: 6
POW: 11
DEF: 11
ARM: 16
HITDICE: 2d6
DMGDICE: 3d6

function: tohit ROLL:n {result:((ROLL + ATT) >= DEF)}
function: damage ROLL:n {result: ((ROLL + POW) >= ARM)*((ROLL + POW) - ARM)}

function: warmachine HROLL:n DROLL:n {
if ((HROLL + ATT) >= DEF) & ((DROLL + POW) >= ARM){
result: ((DROLL + POW) - ARM)}
else {
result: 0}
}

output [warmachine HITDICE DMGDICE] named "Damage Inflicted"
output [tohit HITDICE] named "Hit Chance"

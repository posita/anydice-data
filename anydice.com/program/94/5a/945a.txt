function: damage from AV:s attack with DAM:n damage against DEF:n defense {
 DAMAGE: 0
 ATTACK: 0
 
 loop P over {1..#AV} {
  if P@AV > 2 {
   ATTACK: ATTACK + 1
  }
  if P@AV > 4 {
   DAMAGE: DAMAGE + 1
  }
 }
 
 if ATTACK >= DEF {
  result: DAMAGE + DAM
 }

 if ATTACK + DAMAGE >= DEF {
  if DAM = 0 {
   result: 0
  }

  result: 1
 }

 result: 0

}

function: damage from AV:s attack with DAM:n damage against DEF:n defense only six{
 DAMAGE: 0
 ATTACK: 0
 GLANCE: 0
 
 loop P over {1..#AV} {
  if P@AV > 2 {
   ATTACK: ATTACK + 1
  }
  if P@AV > 5 {
   GLANCE: GLANCE + 1
  }
  if P@AV > 5 {
   DAMAGE: DAMAGE + 1
  }
 }
 
 if ATTACK >= DEF {
  result: DAMAGE + DAM
 }

 if ATTACK + GLANCE >= DEF {
  if DAM = 0 {
   result: 0
  }

  result: 1
 }

 result: 0

}


function: roll A:n plus S:n vs C:n {
  Q: {1,2}@A
  D: 1@C
  E: 2@C
  X: 0
  if Q+S > D {
    X: X + 1
  }
  if Q+S > E {
    X: X + 1
  }

  result: X
}

function: calculate increments for DICE:s {
  INCREMENTS: 0
  loop N over {1..12} {
    X: [count N in DICE]
    if X = 2 { INCREMENTS: INCREMENTS + 1 }
    if X = 3 { INCREMENTS: INCREMENTS + 3 }
    if X = 4 { INCREMENTS: INCREMENTS + 6 }
    if X >= 5 { INCREMENTS: 100 }
  }
  result: INCREMENTS
}

output [calculate increments for 5d12]

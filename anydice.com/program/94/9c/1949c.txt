function: sign X:s {
A: 0
 if X < 0 {
 A: -1
}
else {if X > 0 {
       A: 1
      }
}
result: A
}

function: uphalf N:n {
S: {}
 loop M over {1..N} {
  S: {S, M*3/2}
}
result: S
}

output [sign 10d[uphalf 6]-10*6] named "Fireball"

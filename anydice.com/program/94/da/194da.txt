function: inbands ROLL:n {
  result: (ROLL >= 7) + (ROLL >= 10)
}

function: reroll lowest of ROLL:d {
  result: 1@ROLL + 1d6
}

function: reroll highest of ROLL:d {
  result: 2@ROLL + 1d6
}

function: reroll lowest two of ROLL:d {
  INIT: 1@ROLL
  FIRST: 1d6
  SECOND: 1d6
  result: [highest of INIT and FIRST] + SECOND
}

function: reroll highest two of ROLL:d {
  INIT: 2@ROLL
  FIRST: 1d6
  SECOND: 1d6
  result: [lowest of INIT and FIRST] + SECOND
}

output [inbands 2d6] named "baseline"
output [inbands 2d6+1] named "+1"
output [inbands [reroll lowest of 2d6]] named "reroll lowest"
output [inbands 2d6+2] named "+2"
output [inbands [reroll lowest two of 2d6]] named "reroll 2 lowest"
output [inbands 2d6-1] named "-1"
output [inbands [reroll highest of 2d6]] named "reroll highest"
output [inbands 2d6-2] named "-2"
output [inbands [reroll highest two of 2d6]] named "reroll 2 highest"

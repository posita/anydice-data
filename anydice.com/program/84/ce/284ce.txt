function: X:n blast {
if X = 20 {result: (1d10 + 5) *2}
else if X + 12 >= 20 {result: 1d10 + 5}
else {result: 0}
}

function: X:n vamp {
if X=20 {result: (6d6) *2}
else if X + 12 >= 20 {result: 6d6}
else {result: 0}
}

function: X:n ener {
if X + 5 <= 20 {result: 4d8}
else {result: 2d8} 
}

output [d20 blast] + [d20 blast] + [d20 blast]
output 4d8
output [d20 ener]
output [d20 vamp]

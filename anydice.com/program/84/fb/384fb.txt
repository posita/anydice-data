function: roll A:n and B:n meets TN:n {
  if (A=12 & B!=12) {
    result: 10 \ Critical Success! \
  }
  if (A>B) {
    if (A>=TN) { 
      result: 3 \ Success with boon \
    } else {
      result: -3 \ Failure with boon? \
    }
  } else if (B>A) {
    if (B>=TN) { 
      result: 1 \ Success with consequences \
    } else {
      result: -1 \ Failure with consequences \
    }
  } else { \ A = B \
    if (A>=TN) { 
      result: 2 \ Success \
    } else {
      result: -2 \ Failure \
    }
  }
}

loop TN over {1..12} {
  output [roll d12 and d12 meets TN] named "Target = [TN]"
}

\ Attack Sequence for AoS - calculates the number of wounds inflicted. \
\ Original by Hammer of Math / "Primaris" Kevin Genson, modified by Dave Hall. \
\ This should work for both combat and shooting. \

\ Attack Values \
A: 2 	 \ # Attacks \
H: 4 	 \ Hit Roll Target \
HRT: 0   \ Hit Re-Roll Type - 0=none, 1=re-roll 1's, 2=re-roll all \
W: 4 	 \ Wound Roll Target \
WRT: 0   \ Wound Re-Roll Type - 0=none, 1=re-roll 1's, 2=re-roll all \
S: 3     \ Save \
R: -1 	 \ Rend (Negative Number) \
D: 2 	 \ Damage \
F: 4	 \ Feel No Pain Roll Target - 7 to disable \
MWHST: 7 \ Mortal Wound on Hit and attack sequence Stops Target - 7 to disable \
MWHSD: 3 \ Mortal Wound on Hit and attack sequence Stops Damage \
MWHT: 6  \ Extra Mortal Wound on Hit Target \
MWHD: 0  \ Extra Mortal Wound on Hit Damage \
MWWT: 6  \ Extra Mortal Wound on Wound Target \
MWWD: 0  \ Extra Mortal Wound on Wound Damage \
EDHT: 6  \ Extra Damage on Hit Target \
EDHD: 0  \ Extra Damage on Hit Bonus \
EDWT: 6  \ Extra Damage on Wound Target \
EDWD: 0  \ Extra Damage on Wound Bonus \

\ Die Values \
HD: 1d6   \ Hit Die \
HRD: 1d6  \ Hit RR Die\
WD: 1d6   \ Wound Die \
WRD: 1d6  \ Wound RR Die \
SD: 1d6   \ Save Die\
FD: 1d6   \ Feel No Pain Die \


\ Select which die to use - original or re-roll \
function: sel ROLL:s REROLL:s TARGET:n RRTYPE:n {
    if ROLL < TARGET &                                  \ roll failed AND \
      (( RRTYPE = 1 & ROLL = 1 ) | RRTYPE > 1 ) {       \ re-roll 1's AND roll = 1 OR re-roll all \
        result: REROLL }                                \ use re-roll dice \
    else { result: ROLL }                               \ use original dice \
}

\ Hit roll with re-rolls \
function: hit HROLL:s {result: HROLL >= H}

\ Wound roll with re-rolls \
function: wound WROLL:s {result: WROLL >= W}

\ Save roll - result of 1 means Failed Save (i.e. the wound goes through to cause damage) \
function: save SROLL:s {
    if SROLL = 1 {result: 1}                  \ unmodified Save roll of 1 always fails \
    result: SROLL < S-R
}

function: gate HR:s WR:s SR:s DMG:s {
	result: HR * WR * SR * DMG
}

\ Feel No Pain roll - applied to all damage inflicted (normal and Mortal Wounds) \
\ Ignore damage (result = 0) if roll >= FNP Target \
function: fnp FROLL:s {
    if FROLL < F {result: 1}
    result: 0
}

\ Attack sequence - the result of this sequence is the number of damage inflicted \
function: attack {
\    if HROLL >= MWHST {  \                     \ Mortal wounds, sequence ends \
\        DAMAGE: MWHSD}\
\    else {       \                             \ Normal wounds \

    DAMAGE: D

    result: Ad([gate [hit HROLL] [wound WROLL] [save SD] DAMAGE]d[fnp  FD])
}

        result: TO_HIT >= H
}

function: wound TO_WOUND:s WREROLL:s {
	if TO_WOUND <= WR {result: WREROLL >= W} 
        result: TO_WOUND >= W
}

function: save TO_SAVE:s {
	if TO_SAVE = 1 {result: 1}
        result: TO_SAVE < S-R
}

function: gate HR:s WR:s SR:s D:s {
	result: HR * WR * SR * D
}

function: fnp DICE:s{
	if DICE < F {result: 1}
	result: 0
} \

\ Bonus Damage Calc: Hit, Hit RR, Wound, Wound RR, Damage \

\function: damage HIT:s HITRR:s WOUND:s WOUNDRR:s{
	DAMAGE: D
	if HIT <= HR {
		if HITRR >= EDHT{DAMAGE: DAMAGE + EDHD}
	}
	else {
		if HIT >= EDHT{DAMAGE: DAMAGE + EDHD}
	}
	if WOUND <= WR {
		if WOUNDRR >= EDWT{DAMAGE: DAMAGE + EDWD}
	}
	else {
		if WOUND >= EDWT{DAMAGE: DAMAGE + EDWD}
	}
	result: DAMAGE
} \

\ Roll D6, ReRoll D6, RR Target, MW Target, MW Damage \
\function: mwonhit TO_HIT:s HREROLL:s RR:n MWT: n MWD: n{
	if TO_HIT <= RR {result: (HREROLL >= MWT) * MWD} 
        result: (TO_HIT >= MWT) * MWD
}

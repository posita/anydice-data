function: damage defence DEF:n armour ARMR:n with ATT:n on miss MISS on hit HIT on crit CRIT {
	if ATT < DEF {result: [highest of 0 and (MISS - ARMR)]}
	if ATT < 20 {result: [highest of 0 and (HIT - ARMR)]}
	result: [highest of 0 and (CRIT - ARMR)]
}

function: attack roll with grit GRIT:n and accuracy ACC:n {
	if ACC < 0 {result: 1d20 + [lowest 1 of (ACC)d6] + GRIT}
	result: 1d20 + [highest 1 of (ACC)d6] + GRIT
}

GRIT: 2
loop ACCURACY over {-1, 0, 1}{
	loop DEF over {6, 8, 10, 12}{
		loop ARMR over {0, 3}{
			BASE_ACC: 0
			RELIABLE: 0
			HIT_DAMAGE: [highest of RELIABLE and 2d6 + 4]
			CRIT_DAMAGE: [highest 1 of 2dHIT_DAMAGE]
			output 2d[damage defence DEF armour ARMR with [attack roll with grit GRIT and accuracy (ACCURACY + BASE_ACC)] on miss RELIABLE on hit HIT_DAMAGE on crit CRIT_DAMAGE] named "2x Auto-Stabilizing HMG with ACC [ACCURACY] against DEF [DEF] ARMR [ARMR]"
			
			BASE_ACC: -1
			RELIABLE: 0
			HIT_DAMAGE: [highest of RELIABLE and 3d6 + 4]
			CRIT_DAMAGE: [highest 1 of 2dHIT_DAMAGE]
			output 2d[damage defence DEF armour ARMR with [attack roll with grit GRIT and accuracy (ACCURACY + BASE_ACC)] on miss RELIABLE on hit HIT_DAMAGE on crit CRIT_DAMAGE] named "2x Insulated HMG with ACC [ACCURACY] against DEF [DEF] ARMR [ARMR]"
			
			BASE_ACC: 2
			RELIABLE: 0
			HIT_DAMAGE: [highest of RELIABLE and 2d6]
			CRIT_DAMAGE: [highest 1 of 2dHIT_DAMAGE]
			output [damage defence DEF armour 0 with [attack roll with grit GRIT and accuracy (ACCURACY + BASE_ACC)] on miss RELIABLE on hit HIT_DAMAGE on crit CRIT_DAMAGE] named "Auto-Stabilizing AMR with ACC [ACCURACY] against DEF [DEF] ARMR [ARMR]"
			
			BASE_ACC: 1
			RELIABLE: 0
			HIT_DAMAGE: [highest of RELIABLE and 3d6]
			CRIT_DAMAGE: [highest 1 of 2dHIT_DAMAGE]
			output [damage defence DEF armour 0 with [attack roll with grit GRIT and accuracy (ACCURACY + BASE_ACC)] on miss RELIABLE on hit HIT_DAMAGE on crit CRIT_DAMAGE] named "Insulated AMR with ACC [ACCURACY] against DEF [DEF] ARMR [ARMR]"
		}
	}

TARGETAC: 16

function: h X:n c Y:n hb HB b B cr CR cd CD {
  if X >= CR {
    if Y+HB >= TARGETAC {result: CD}
    result: B
  }
  if X+HB >= TARGETAC {result: B}
  result: 0
}

function: hitbonus HB base B critrange CR critdmg CD {
result: [h d20 c d20 hb HB b B cr CR cd CD]
}
output [hitbonus 9 base 1d6+4 critrange 18 critdmg 2d6+8] named "Scimitar+1"
output [hitbonus 9 base 1d8+4 critrange 20 critdmg 3d8+12] named "Spear+1"

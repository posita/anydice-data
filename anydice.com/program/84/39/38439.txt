function: rerollminmax ROLL:s {
  if 1@ROLL = 1 | 1@ROLL = 6 {
    result: d6
  }
  result: ROLL
}

output [ rerollminmax d6 ]

output d6 = d6 named "2. Match"

output 1@2d6 named "3. Advantage"
output 2@2d6 named "3. Disadvantage"

set "explode depth" to 3

output [ explode d6 ] named "Single d6 exploding, up to 3 extra rolls"

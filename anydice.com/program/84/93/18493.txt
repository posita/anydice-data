\ roll AdX + BdY, target 4, where X is one size less than Y and A+B = 6 \

DICE: {4,6,8,10,12,20}

loop I over {2..#DICE} {
  X: (I-1)@DICE
  Y: I@DICE

  output 6d(dX >= 4) named "6d[X]"

  loop B over {1..5} {
    A: 6-B
    output Ad(dX >= 4) + Bd(dY >= 4) named "[A]d[X] and [B]d[Y]"
  }
}
output 6d(dY >= 4) named "6d[Y]"

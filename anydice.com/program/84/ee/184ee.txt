function: relabel SIDE:n target TARGET:n {
  if SIDE > TARGET { result: 0 }
  if SIDE = 1 { result: 2 }
  else { result: 1 }
}
TARGET: 3
FOUR: [relabel d4 target TARGET]
SIX: [relabel d6 target TARGET]
EIGHT: [relabel d8 target TARGET]
TEN: [relabel d10 target TARGET]
TWELVE: [relabel d12 target TARGET]

output 2dFOUR named "2d4"
output dFOUR + dSIX named "d4 + d6"
output 2dSIX named "2d6"
output dSIX + dEIGHT named "d6 + d8"
output 2dEIGHT named "2d7"
output dEIGHT + dTEN named "d8 + d10"
output 2dTEN named "2d10"
output dTEN + dTWELVE named "d10 + d12"
output 2dTWELVE named "2d12"

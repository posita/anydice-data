function: iexplodeh SEQUENCE:s M:n {
 result: [count 0 in SEQUENCE]*M+1@SEQUENCE
}

function: iexplode N:n D:s {
 if N < 1 {
    result: N
 }
 SEQUENCE: [lowest of N and 10]dD
 result: [highest of [iexplodeh SEQUENCE 5] and N]
}


loop N over {0..10} {
output ([iexplode N {0..5}])+4 - ([iexplode 2 {0..5}]+4) named "[N]d6"
}

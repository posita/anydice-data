function: total TTL:n roll TIMES:n die SIDES:n count SUCC:n explode on EXPL:n {
  SUCCESSES: 0
  loop N over {1..TTL} {
    SUCCESSES: SUCCESSES + [roll TIMES die SIDES count SUCC explode on EXPL]
  }
  result: SUCCESSES

}
function: roll TIMES:n die SIDES:n count SUCC:n explode on EXPL:n {
  SUCCESSES: 0
  DIE: 1dSIDES
  loop N over {1..TIMES} {
    SUCCESSES: SUCCESSES + [DIE success on SUCC explode on EXPL]
  }
  result: SUCCESSES
}

function: N:n success on SUCC:n explode on EXPL:n {
  SUCCESSES: 0
  if N >= SUCC { SUCCESSES: 1 }
  if N >= EXPL { SUCCESSES: SUCCESSES + [DIE success on SUCC explode on EXPL] }
  result: SUCCESSES
}

output [total 5 roll 7 die 10 count 5 explode on 10] named "player"

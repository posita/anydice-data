function: limit A:s to B:n {
  OUT: {}
  loop RAW over A {
    if RAW > B {
      OUT: {OUT, B}
    } else {
      OUT: {OUT, RAW}
    }
  }
  result: OUT
}

FORMULARATING: 6
BASEDICE: 14
FOCUSRATING: 3

ARTHITS: {[count {1,2} in (BASEDICE+FOCUSRATING)d6]}
output ARTHITS
ARTHITS: {[limit ARTHITS to FORMULARATING]}
output ARTHITS

OPPHITS: [count {1,2} in 9d6]
output ARTHITS-OPPHITS>0

output ARTHITS-OPPHITS

function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

function: ROLL:n explode {
  if ROLL = 6 { result: 2 }
  result: 1
}

T: [d6 reroll {1} as d6]
E: [d6 explode]

output [count {5,6} in 4dT] named "4d6 reroll 1"
output [count {5,6} in 4d6] named "4d6"
output 4dE named "4d6 explode 6"

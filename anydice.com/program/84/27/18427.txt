function: hit TO_HIT:s {
    result: TO_HIT >= WS
}

function: wound TO_WOUND:s {
    if S > T * 2 {
        result: TO_WOUND >= 2
    }
    if S > T {
        result: TO_WOUND >= 3
    }
    if S = T {
        result: TO_WOUND >= 4
    }
    if S < T / 2 {
        result: TO_WOUND >= 6
    }
    if S < T {
        result: TO_WOUND >= 5
    }
}

function: save TO_SAVE:s {
    result: TO_SAVE + AP < AS
}

WS: 2
S: 4
T: 4
D: 2
AS: 3
AP: -2

output [hit 6d6] named "Hit Probability"

output [wound d[hit 6d6]d6] named "Wound Probability"




output [save d[wound d[hit 4d6]d6]d6] named "Single Blade"

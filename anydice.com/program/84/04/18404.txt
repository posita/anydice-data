\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
DMGBONUS: 0
PROFICIENCY: 5
ACCBONUS: 0
CRITROLL: 20
DAMAGE: 4d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 8d6 + ATTACKMOD + DMGBONUS
output [attackroll d20 vs ENEMYAC] named "+5 to hit 4d6"

ATTACKMOD: 5
DMGBONUS: 1
PROFICIENCY: 5
ACCBONUS: 1
CRITROLL: 20
DAMAGE: 2d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d8 + ATTACKMOD + DMGBONUS
output [attackroll d20 vs ENEMYAC] named "+6 to hit 2d8"

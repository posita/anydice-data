function: opposed first dice FIRST:s second dice SECOND:s  {

  loop I over FIRST {
     loop J over SECOND {
        if I     =  J { result: 0 }
        if I * 2 <= J { result: -2 }
        if I     <  J {result: -1 }
        if I     >= J*2 { result: 2 }
        result: 1
     }
  }
}



\output d6-d6 named "baseline"\

loop N over {2..5} {
   output [ opposed first dice d6+N second dice d6+N] named "base [N]"
}

loop N over {2..5} {
   output [ opposed first dice d6+N+1 second dice d6+N] named "base [N] Add 1"
}

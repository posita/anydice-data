WHITE: 1
BLACK: 0
BLOCK: 1

function: die AB:n AW:n {
   result: AWd{0,1,1,1,2,2} + ABd{0,1,1,2,2,3}
}
function: calc D:n B:n {
   if D-B>0 {result: 1} result:0
}
function: attack AB:n AW:n B:n {
   if B>0 {result: [calc [die AB AW] B]} result: [die AB AW]
}


output [die BLACK WHITE] named "Die prediction (without Block)"
if BLOCK>0 {output [attack BLACK WHITE BLOCK] named "Result of attack: 1 = hit"}

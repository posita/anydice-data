function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

function: ROLL:n rerollunder BAD:s as REROLL:d {
  if ROLL < BAD { result: REROLL }
  result: ROLL
}

X: [1d6 reroll {6} as 1d6]
Y: [1dX rerollunder {5} as 1dX]
Z: [1dY reroll {6} as 1dY]

output [count {5,6} in 19dX] named "19 Attacks: Rerolling 6s to Hit w/ 5,6 MW"
output [count {5,6} in 19dY] named "19 Attacks: Reroll 6s to Hit then Reroll <5 w/ 5,6 MW"
output [count {5,6} in 19dZ] named "19 Attacks: Reroll <5 then Reroll 6s to Hit w/ 5,6 MW"

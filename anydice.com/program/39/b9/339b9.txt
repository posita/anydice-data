function: challenge ROLL:n plus MOD:n versus CDICE:s difficulty DIFF:n {
    result: ((ROLL + MOD - DIFF) > CDICE) > 0
}

function: skill ROLL:n plus MOD:n versus DC:n {
  result: (ROLL + MOD) >= DC
}

function: guided ROLL:n plus MOD:n plus GUIDE:n versus DC:n {
  result: (ROLL + MOD + GUIDE) >= DC
}

loop N over {0..10} {
DC: 10 + (N*2)
output [challenge 1d6 plus 2 versus 2d10 difficulty N] named "Challenge for Difficulty +[N]"
output [skill 1d20 plus 2 versus 10 + (N * 2)] named "Skill Check for DC [DC]"
output [challenge 1d6 plus 2 versus 2d10 difficulty N-1d2] named "Guided Challenge for DC [N]"
output [guided 1d20 plus 2 plus 1d4 versus 10 + (N * 2)] named  "Guided Check for DC [DC]"
}

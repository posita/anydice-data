REROLL: 0
MISSDAMAGE: 0
STRMOD: 3
PROFICIENCY: 2
AC: 19
CRITRANGE: 20
DAMAGE: 1d12+2

function: roll A:n for damage with reroll B:n {
  if A + STRMOD + PROFICIENCY < AC {
    result: STRMOD * MISSDAMAGE
  }
  if A > CRITRANGE - 1 {
    result: DAMAGE + ([maximum of DAMAGE]/#(DAMAGE))+ STRMOD
  }
  result: DAMAGE + STRMOD
}

function: roll A:n for damage without strmod {
  if A + STRMOD + PROFICIENCY < AC {
    result: STRMOD * MISSDAMAGE
  }
  if A > CRITRANGE - 1 {
    result: DAMAGE + ([maximum of DAMAGE]/#(DAMAGE))
  }
  result: DAMAGE
}

output [roll d20 for damage with reroll 1] + [roll d20 for damage with reroll 1] + [roll d20 for damage without strmod]

function: compare A:n to B:n {
  D: A - B
  if D < 0 {
    result: -1
  }
  if D > 0 {
    result: 1
  }
  result: 0
}

function: test N:n vs M:n {
  NMAX: [highest 3 of Nd6]
  MMAX: [highest 3 of Md6]
  A: [compare 1@NMAX to 1@MMAX]
  result: A
}

loop N over {0..3} {
  NN: 3 + N * 2
  loop M over {0..N} {
    MM: 3 + M * 2
    output [test NN vs MM] named "[NN] vs [MM]"
  }
}

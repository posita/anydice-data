function: advantage DIE:d {
  result: [highest of [explode DIE] and [explode DIE]]
}
function: disadvantage DIE:d {
  result: [lowest of [explode DIE] and [explode DIE]]
}

function: level N:n {
  if N < 1 {
    result: 0
  } if N < 4 {
    result: 1
  } else if N < 9 {
    result: 2
  } else if N < 16 {
    result: 3
  } else if N < 25 {
    result: 4
  } else if N < 36 {
    result: 5
  } else if N < 49 {
    result: 6
  } else if N < 64 {
    result: 7
  } else if N < 81 {
    result: 8
  } else if N < 100 {
    result: 9
  } else {
    result: 10
  }
}

if 0 > 1 {
loop N over {10,12} {
  DIE: dN
  output [level [explode DIE]-1] named "[DIE]X"
  output [level [advantage DIE]-1] named "[DIE]A"
  output [level [disadvantage DIE]-1] named "[DIE]D"
}
}

output [level [explode d10]-1] named "d10"
output [level [explode d20]]-1 named "d20"

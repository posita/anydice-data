
function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output [reroll {6} on d6] named "r6+d6"
output [reroll {6,5} on d6] named "r5+d6"
output [reroll {6,5,4} on d6] named "r4+d6"
output d6 named "d6"

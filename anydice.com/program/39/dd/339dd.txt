  GWMDMG: 0
  GWMACC: 0
  if GWM = YES {
    GWMDMG: 10

  ACC: HIT > AC - ATTR - PROF + GWMACC
  DMG: WEAPON + ATTR + MAGIC + GWMDMG
  ATK: ACC * DMG
  result: 1dATK
}

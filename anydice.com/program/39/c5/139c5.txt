function: wild DIE:d {
  result: [highest of [explode DIE] and [explode d6]]
}
function: extra DIE:d {
  result: [explode DIE]
}


function: success ROLL:n vs TN:n {
  if ROLL < TN { 
    result: 0
  } else {
    result: 1 + [success (ROLL - TN) vs 4]
  }  
}

loop N over {4,6,8,10,12}{
  output [success [wild dN] vs 4] named "d[N]w"
  output [success [extra dN] vs 4] named "d[N]"
}

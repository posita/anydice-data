function: careful N:n {
 if N = 1 { result: d6 }
 if N = 6 { result: d6 }
 result: N
}
function: risky N:n {
 if N = 1 { result: N-d6 }
 if N = 6 { result: N+d6 }
 result: N
}
output [careful d6]+[careful d6] >= 6 named "Careful Weak Success Chance"
output [careful d6]+[careful d6] >= 10 named "Careful Great Success Chance"
output [risky d6]+[risky d6] >= 6 named "Risky Weak Success Chance"
output [risky d6]+[risky d6] >= 10 named "Risky Great Success Chance"

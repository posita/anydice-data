function: tsc SKILL:n vs DIFFICULTY:n {
  result: (2d6 + DIFFICULTY + SKILL) >= 8
}

function: blades ROLL:s {
  if 1@ROLL = 6 {
    result: 2
  }
  result: 1@ROLL >= 4
}

function: bfour ROLL:s {
 if 1@ROLL = 6 {
  result: 2 + (3@ROLL = 6)
 }
 if 3@ROLL = 5 { result: 2 }
 result: (1@ROLL >= 4)
}

function: bfive ROLL:s {
 if 1@ROLL = 6 {
  result: 2 + (3@ROLL = 6)
 }
 if 3@ROLL = 5 { result: 2 }
 result: (1@ROLL >= 5)
}

function: yze ROLL:s {

  if 3@ROLL = 6 { 
    result: 2
  } 
  if 1@ROLL = 6{
    result: 1 
  }
  if 3@ROLL = 5 {
    result: 1
  }
  result: 0 
}

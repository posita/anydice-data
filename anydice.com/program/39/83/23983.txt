function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

function: ROLLU:n rerollunder BADU:s as REROLLU:d {
  if ROLLU < BADU { result: REROLLU }
  result: ROLLU
}

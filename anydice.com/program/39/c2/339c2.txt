function: evaluate ROLL:s WILD:s{
 
 SUCCESSES: ROLL >= 3
 CRITICAL: ROLL >= 6
 FAIL: 1@WILD = 1
 CW: 1@WILD >=6
 SW: 1@WILD >=4
 
 
 if CRITICAL + CW  >1 { result: 3}
 if CRITICAL + CW = 1 { result: 2 }
 if FAIL=1  { result: 0}
 if SUCCESSES + SW  > 0 { result: 1 }
 
 result: 0

}

loop DICE over {0..5} {
 WILD: 1
 output [evaluate DICE d6 WILD d6] named "[DICE]+[WILD]d"
}

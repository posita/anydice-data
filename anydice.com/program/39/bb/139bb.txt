function: fail A:n {
 if A < 0 { \Failure\
  result: -1
 }
 else { \Success\
  result: 0 \Alternatively, use 0 so you'll know the chances of any success\
 }
}

DC: 15 \Base DC\
MODIFIER: 5 \ Ability Modifier + Proficiency \

output [fail 1d20+MODIFIER-(DC+6)] named "6th Level"
output [fail 1d20+MODIFIER-(DC+7)] named "7th Level"
output [fail 1d20+MODIFIER-(DC+8)] named "8th Level"
output [fail 1d20+MODIFIER-(DC+9)] named "9th Level"

BUFFS: 1d4+1d4 \Examples: 1d4 from Bless, 1d4 up to 1d12 from Bardic Inspiration\

output [fail 1d20+MODIFIER+BUFFS-(DC+6)] named "6th Level w/ Buffs"
output [fail 1d20+MODIFIER+BUFFS-(DC+7)] named "7th Level w/ Buffs"
output [fail 1d20+MODIFIER+BUFFS-(DC+8)] named "8th Level w/ Buffs"
output [fail 1d20+MODIFIER+BUFFS-(DC+9)] named "9th Level w/ Buffs"

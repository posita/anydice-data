\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 5-20\
ENEMYAC: d{15..20}

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20
DAMAGE: 1d12 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d12 + ATTACKMOD + DMGBONUS
output [attackroll d20 vs ENEMYAC] named "stat boost"


ATTACKMOD: 3
DMGBONUS: 0
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20
DAMAGE: [highest 1 of 2d12] + ATTACKMOD + DMGBONUS
CRITDAMAGE: [highest 1 of 2d(2d12)] + ATTACKMOD + DMGBONUS
output [attackroll d20 vs ENEMYAC] named "savage attacker"

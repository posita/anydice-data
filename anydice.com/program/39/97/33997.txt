LANTERN: 10
KNOCK: 10
SH: 10

function: roll BONUS {
 US: [highest 1 of 2d100] + BONUS
 THEM: 1d100
 result: US > THEM
}

LWIN: [roll LANTERN]
KWIN: [roll KNOCK]
SHWIN: [roll SH]

output LWIN named "Lantern"
output LWIN named "Knock"
output SHWIN named "Secret Histories"

output (LWIN & KWIN) | (LWIN & SHWIN) | (KWIN & SHWIN) named "overall result"

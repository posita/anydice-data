function: reroll lowest of ROLL:s as REROLL:d if less than MIN:n {
  LOWEST: (#ROLL)@ROLL       \ the lowest die is sorted last \
  if LOWEST >= MIN {
    result: ROLL + 0
  } else {
    REST: {1..#ROLL-1}@ROLL  \ all but the lowest die \
    result: REST + REROLL
  }
}

output 1d10 named "1d10"
loop MIN over {1..10} {
  output [reroll lowest of 1d10 as d10 if less than MIN] named "1d10 replace lowest if < [MIN]"
}
output [highest 1 of 1d10] + 1d10 named "1d10 always replace lowest"

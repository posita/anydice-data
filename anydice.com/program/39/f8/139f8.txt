function: explode N:s {
 COUNT: 0
 loop P over {1..#{5,6}} {
  if P@{5} = N { COUNT: COUNT + 1 }
  if P@{6} = N { COUNT: COUNT + 1 + [explode 1d6] }
 }
 result: COUNT
}

function: preedge X:n {
 COUNT: 0
 loop P over {1..X} {
  COUNT: COUNT + [explode 1d6]
 }
 result: COUNT
}

output [preedge 1] named "Preedge"

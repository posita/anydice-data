function: cortex A and B and C {
 V: [highest of A+B and B+C]
 result: [highest of V and A+C]
}

function: test DICE:s vs B {
 TOTAL: 0
 loop N over {1..#(DICE)}{
  if N@{DICE} >= B {
   TOTAL: TOTAL + 1
  }
 }
 result: TOTAL
}

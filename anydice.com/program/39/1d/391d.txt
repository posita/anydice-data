function: income ROLL:n CONTROLDC:n ECONOMYBONUS:n CONSUMPTION:n {
  ECONOMYROLL: ROLL + ECONOMYBONUS
  if ECONOMYROLL < CONTROLDC {
   result: -CONSUMPTION
  } else {
   result: (ECONOMYROLL/5)-CONSUMPTION
  }
}

function: explode DIE:d {
  MAX: [maximum of DIE]
  result: [explode DIE helper]
}

function: explode N:n helper {
  if N = MAX {
    result: 6 + [explode DIE helper] }
  result: N
}

output [explode 1d6] named "Average Bandit Attack"

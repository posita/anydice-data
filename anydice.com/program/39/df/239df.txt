function: save  RESULT:n {
 if NOSAVE = 1 { result: 1 }
 if RESULT >= ARMOR{
   if VULNERABLE = 1 & VULNERABLE_APPENED = 0 { 
     VULNERABLE_APPENED : 1
     TORETURN: [save d6]
     VULNERABLE_APPENED : 0
     result: TORETURN
   }
   result: 0
 }
 result: 1
}

function: hit RESULT:n {
 if RESULT >= TOHIT {
   if WEAKEN = 1 & WEAKEN_APPENED = 0{ 
    WEAKEN_APPENED : 1
    REROLL_APPENED : 1
    TORETURN: [hit d6]
    REROLL_APPENED : 0
    WEAKEN_APPENED : 0
    result: TORETURN
   }
   if RESULT = 6 {
     if PRECISION = 1 {
        if CRITBLOW = 1 {result: 1 + [save d6] }
        result: 1
     }
     if CRITBLOW  = 1 { result: [save d6] + [save d6] }
   }
   result: [save d6]
 }
 if REROLL = 1 & REROLL_APPENED = 0 {
   REROLL_APPENED : 1
   TORETURN: [hit d6]
   REROLL_APPENED : 0
   result: TORETURN
 }
 result: 0
}

function: moral DICEONE:n and DICETWO:n damage DAMAGE:d{
  if DICEONE+DICETWO-MORALMALUS >= MORAL {
    if PANIKED = 1 & IMUNETOCKEN = 0 {
      PANIKED: 0
      if DICEONE >= 4 {
        if DICETWO >= 4 {
          if DAMAGE = 1 {
            result: [moral d6 and d6 damage d3]
          }
          result: [moral d6 and d6 damage DAMAGE]
        }

        if DAMAGE = 1 {
          result: [moral d6 and DICETWO damage d3]
        }
          result: [moral d6 and DICETWO damage DAMAGE]
      }
      if DICETWO >= 4 {
        if DAMAGE = 1 {
          result: [moral d6 and DICEONE damage d3]
        }
        result: [moral d6 and DICEONE damage DAMAGE]
      }
      if DAMAGE = 1 {
        result: [moral DICEONE and DICETWO damage d3]
      }
    }
    result: 0
  }
  if PANIKED = 1 {
    PANIKED: 0
    if DAMAGE = 1 {
      result: [moral d6 and DICETWO damage d3]
    }
  }
  result : DAMAGE + MORALWOUND
}

function: takemoral WOUND:n {
   WOUND: WOUND - HITREDUCTION
   if WOUND >=1 {
      result: WOUND + [moral d6 and d6 damage d3]
   }
   result: 0
}

function: attack {
   WOUND: DICEd[hit d6] + BONUSHITd[save d6]
   result : [takemoral WOUND]
}

function: doattack {
  TOHIT : TOHIT + TOHITMALUS
  if TOHIT > 6 {TOHIT: 6}
  if TOHIT < 2 {TOHIT: 2}
  DICE: DICE+ BONUSDICE
  if REAR= 1 {
     ARMORMALUS: ARMORMALUS + 2
     MORALMALUS: MORALMALUS + 2
  }
  else {
    if FLANK = 1 {
       ARMORMALUS: ARMORMALUS + 1
       MORALMALUS: MORALMALUS + 1
    }
  }
  ARMOR: ARMOR + ARMORMALUS
  if ARMOR > 6 {ARMOR: 6}
  if ARMOR < 2 {ARMOR: 2}
  if CHARGE = 1 {
     if DISORDERLY = 0 {
       REROLL : 1
     }
  }
  result: [attack]
}

function: hassurvive WOUND:n health WOUNDTOTAL:n{
    result: WOUND < WOUNDTOTAL
}

function: survival {
  VAR : [hassurvive [doattack] health TOTALWOUND]
  CHARGE: 0
  loop N over {2..10}{
    TOTALWOUND: TOTALWOUND + REGEN
    VAR : VAR + [hassurvive Nd[doattack] health TOTALWOUND]
  }
  result: VAR 
} 

function: survival against ATTACKER:d{
  VAR : [hassurvive 1dATTACKER health TOTALWOUND]
  loop N over {2..10}{
    TOTALWOUND: TOTALWOUND + REGEN
    VAR : VAR + [hassurvive NdATTACKER health TOTALWOUND]
  }
  result: VAR 
} 

function: survival against ATTACKER:d charging CHARGEATT:d {

    TOTALWOUND: TOTALWOUND + REGEN
    VAR : VAR + [hassurvive 1dCHARGEATT+(N-1)dATTACKER health TOTALWOUND]
  }
  result: VAR 
}

  result: [ironmakers]
}

function: survive ironmakers{
  CHARGE: 0
  result: [survival against [ironmakers]]
}

function: survive charging ironmakers{
  CHARGE: 0
  result: [survival against [ironmakers] charging [charging ironmakers]]
}

output [NAME_OF_THE_ATTACK] named "OUTPUT_NAME"

Available attacks :
doattack => attack using variable as define at the moment of the execution
survival chance up to X => replace X by the number of attack max, compute the chance of a unit to survive up to X attack (use default parameter same as doattack)

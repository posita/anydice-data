CRITFAIL: -1
FAIL: 0
SUCCESS: 1
RAISE: 2

function: swroll A:n and B:n against T:n {
if [highest of A and B] = 1 { result: CRITFAIL }
if [highest of [highest of A and B] and [highest of A and B]] <T { result: FAIL}
if [highest of [highest of A and B] and [highest of A and B]] >= T {
 if [highest of [highest of A and B] and [highest of A and B]] > (T+4) { result: RAISE}
 result: SUCCESS}
result: FAIL
}

set "explode depth" to 3
WILD: [explode d6]
SKILL: [explode d6]
TN: 4

\-1 is Crit Fail, 0 is Fail, 1 i Success, 2 is Raise\

output [swroll WILD and SKILL against TN] named "regular roll"
output [highest of [swroll SKILL and WILD against TN] and [swroll SKILL and WILD against TN]] named "Card Sharp"

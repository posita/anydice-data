function: evaluate ROLL:s WILD:s{
 
 SUCCESSES: ROLL >= 4
 CRITICAL: ROLL >= 9
 FAIL: 1@WILD = 1
 CW: 1@WILD >=9
 SW: 1@WILD >=4
 
 if FAIL=1  { result: 0}
 if CRITICAL + CW  >1 { result: 3}
 if CRITICAL + CW = 1 { result: 2 }
 
 if SUCCESSES + SW  > 0 { result: 1 }
 
 result: 0

}

loop DICE over {0..6} {
 WILD: 1
 output [evaluate DICE d10 WILD d10] named "[DICE]+[WILD]d"
}

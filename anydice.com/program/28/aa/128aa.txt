function: normal {
 result: 1d6>4
}

function: hard {
 result: 1d6>5
}

function: easy {
 result: 1d6>3
}

function: standard A:n vs B:n {
 if B >= A {result: 0}
 result: A-B
}

function: boosted A:n vs B:n {
 if B >= A {result: 0}
 result: A-B+1 
}

output [normal] named "normal roll"
output [standard 2d[normal] vs 1d[normal]] named "Damage with $N2vs$N1"
output [boosted 2d[normal] vs 1d[normal]] named "Damage with $N2vs$N1 with AttackUp"

function: blah X:s {
 if #X = 1 { result: 1d{1:3,0:2,-1} }
 ACES: [count -1 in X]
 if ACES < #X { result: [blah [lowest of (#X-ACES) and (#X-1)]d{1:3,0:2,-1}] - ACES }
 result: -ACES
}

output [highest of 5d{1:3,0:2,-1} and 0] named "No Reroll"
output [highest of 5@5d{1:3,0:2,-1} + 4@4d{1:3,0:2,-1} + 3@3d{1:3,0:2,-1} + 2@2d{1:3,0:2,-1} + 1@1d{1:3,0:2,-1} and 0] named "Max Rerolls"
output [highest of [blah 5d{1:3,0:2,-1}] and 0] named "Save Aces"

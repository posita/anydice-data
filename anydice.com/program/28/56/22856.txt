MISS:0
HIT:1
CRITHIT:2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL > 20 - THREAT { result: CRITHIT }
 if ROLL + ATTACK > DEFENSE { result: HIT}
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITHIT {
  result: CRITDAM
 }
 if ATTACKRESULT = HIT {
  result: DAM
 }
 result: MISSDAM
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

function: advattack {
 result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]
}

\ DEFENSE: d{10..20} \

loop DEFENSE over {10..25} {

DEX: 3
PROF: 3
GUNTHREAT:1
CHAMPIONTHREAT:0
FOCUSTHREAT:4
FOCUSATK:4
FOCUSDAM:4
MAXTHREAT:5

 THREAT: MAXTHREAT
}
ATTACK: DEX + PROF
DAM: 2d6 + DEX
CRITDAM: 3d6 + DEX + 6
MISSDAM: 0
\output [advattack] + [advattack] named "Shoot Twice ADV vs AC [DEFENSE]"

}

function: exito D:n TARGET:n{
  if D <= TARGET{ result: 1}
  result: 0
}

function: roll D:n TECNICA:n ESFUERZO:n{
   RES: [exito D TECNICA]
   if RES = 1{
      loop N over {1..ESFUERZO}{
         RES: RES + [exito 1d6 TECNICA]
      }
   }
   result: RES
}

function: accion TECNICA:n ESFUERZO:n AFINIDAD:n{
  X: 0
  loop N over {1..4 + AFINIDAD}{
     X: X + [roll 1d6 TECNICA ESFUERZO]
  }
  result: X
}

loop AFINIDAD over {0..3}{
   loop TECNICA over {1..3}{
      loop ESFUERZO over {0..3}{
         output [accion TECNICA ESFUERZO AFINIDAD] named "Técnica [TECNICA] y Esfuerzo [ESFUERZO] y Afinidad [AFINIDAD]" 
      }
   }
}

function: w of X:n {
  if X > 63 {
    result: (X - 63) / 2
  }
  result: (63 - X) / 5
}

function: final result for R:n {
  if R > 21 {
    H: (R - 21) * 10
    M: (R - 21) * 10
  } else {
    H: (21 - R) * -3
    M: (21 - R) * -4
  }

  X: 6d20 + M
  result: (10d100 / 5) - 1 + H + [w of X]
}

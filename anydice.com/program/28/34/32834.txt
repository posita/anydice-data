function: wild WILDROLL:n N dice PIPS pips{
   if N > 1 {
   OTHERDICE: (N-1)d6 
   if WILDROLL = 6 {
      WILDEFFECT: [explode 1d6] 
      result: OTHERDICE + WILDEFFECT + WILDROLL + PIPS} 
   if WILDROLL < 6 & WILDROLL > 1{
      result: OTHERDICE + WILDROLL + PIPS}
   if WILDROLL = 1 {
      if N > 2 {
         WILDEFFECT: [lowest (N-2) of OTHERDICE]
         result: WILDEFFECT + WILDROLL + PIPS}
      if N = 2 {result: WILDROLL + PIPS}
      }
   }
   else {WILDEFFECT: [explode 1d6] result: WILDEFFECT + PIPS}
}

output [wild 1d6 1 dice 0 pips]

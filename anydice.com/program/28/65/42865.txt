TO_HIT: 12
NORMAL_DAMAGE: 1d10 + 3d8
CRIT_DAMAGE: 2d10 + 6d8

function: attack ROLL:n vs AC:n {
 if ROLL = 20 { result: CRIT_DAMAGE }
 if ROLL = 1 { result: 0 }
 if ROLL + TO_HIT >= AC { result: NORMAL_DAMAGE }
 result: 0
}

loop AC over {15, 18, 20, 22} {
 output 3d[attack d20 vs AC] named "Total Damage vs AC [AC]"
}

function: rollstrat CONDITION:n results A B {
	if CONDITION { result: A } else { result: B }}

function: rerollifunder N {
	result: [rollstrat 1d20>N results 1d(20-N)+N 1d20]}

function: knowread A:n {
	result: (1 + (A > 15) + (A > 25) + (A > 30) + (A > 35)) }

function: knowledge BOOST {
	result: [knowread (BOOST + 1d20)]}

function: know reroll N BOOST {
	result: [knowread ([rerollifunder N] + BOOST)]}

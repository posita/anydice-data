 ROLL: 1@D + 2@D
 if ((!PROF | CON < 10) & (1@D = 2@D & 1@D >= CON)) | (PLASMA & 1@D = 2@D & 1@D >= PLASMA) {
  result: -1
 }
 AT: ATT + SKILL + ( LR * 2 ) + AIM - VITAL
 if ROLL > ( AT ) {
  result: 0
 }
 XS: AT - ROLL
 DAMAGE: DAM
 if SHIELD {
  DAMAGE: [maximum of DAM]
 }
 EXTRADAM: 0
 SINGLE: ( RATE - LR = 1 )
 if MASTER = 1 {
  EXTRADAM: XS
 }
 if MASTER = 2 {
  EXTRADAM:  XS / ( RATE - LR )
 }
 if MASTER = 3 {
  EXTRADAM: [lowest of XS and SKILL]
  if PRECISE = 1 & SINGLE {
   EXTRADAM: [lowest of XS and ( SKILL * 2 )]
  }
  if PRECISE = 2 & SINGLE {
   EXTRADAM: XS
  }
 }
 if AIMEX {
  EXTRADAM: EXTRADAM + AIM
 }
 if VITEX {
  EXTRADAM: EXTRADAM + VITAL
 }
 if PRECISE = 3 & SINGLE {
  EXTRADAM: EXTRADAM * 2
 }
 if !AIMEX {
  EXTRADAM: EXTRADAM + AIM
 }
 if !VITEX {
  EXTRADAM: EXTRADAM + VITAL
 }
 DAMAGE: DAMAGE + EXTRADAM
 if 1@D = 2@D & (PROF | 1@D = 1 | 1@D <= (CON - 10)) {
  DAMAGE: DAMAGE * 2
 }
 if !PLASMA {
  DAMAGE: DAMAGE - AV
 }
 if PLASMA {
  DAMAGE: DAMAGE - [floor (AV - [maximum of DAM])]
 }
 DAMAGE: [floor DAMAGE]
 result: DAMAGE
}

function: floor DAM:n {
 if DAM < 0 {
  result: 0
 }
 result: DAM

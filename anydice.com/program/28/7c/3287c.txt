BASEDIE: d10

function: target TARGET:n reroll REROLL:n VALUE:n {
  if VALUE = REROLL {
    result: 1 + [target TARGET reroll REROLL BASEDIE]
  }
  if VALUE  >= TARGET {
    result: 1
  }
  result: 0
}

TAR: 8
EXP: 10
DIE: [target TAR reroll EXP BASEDIE]

output BASEDIE>=TAR named "raw 1d[BASEDIE] success chance"
output 1d[target EXP reroll EXP BASEDIE] named "taking a chance"

loop N over {1..10}{
  output NdDIE named "exploding success in [N][BASEDIE]"
}

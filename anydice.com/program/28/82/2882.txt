function: dam DAM:d bdam BON:n at AT:n plasma PLASMA:n master MASTER:n con CON:n prof PROF:n shield SHIELD:n av AV:n JAM:n rolled D:s {
 if JAM {
  result: 0
 }
 ROLL: 1@D + 2@D
 if ((!PROF | CON < 10) & (1@D = 2@D & 1@D >= CON)) | (PLASMA & 1@D = 2@D & 1@D >= PLASMA) {
  result: -1
 }
 if ROLL > AT {
  result: 0
 }
 XS: AT - ROLL
 DAMAGE: DAM
 if SHIELD {
  DAMAGE: [maximum of DAM]
 }
 if MASTER = 1 {
  DAMAGE: DAMAGE + XS
 }
 if MASTER = 2 {
  DAMAGE: DAMAGE + ( XS / RATE )
 }
 DAMAGE: DAMAGE + BON
 if 1@D = 2@D & (PROF | 1@D = 1 | 1@D <= (CON - 10)) {
  DAMAGE: DAMAGE * 2
 }
 if !PLASMA {
  DAMAGE: DAMAGE - AV
 }
 if PLASMA {
  DAMAGE: DAMAGE - [floor (AV - [maximum of DAM])]
 }
 DAMAGE: [floor DAMAGE]
 result: DAMAGE
}

function: floor DAM:n {
 if DAM < 0 {
  result: 0
 }
 result: DAM
}

function: dam DAM:d bdam BON:n at AT:n plasma PLASMA:n master MASTER:n con CON:n prof PROF:n shield SHIELD:n av AV:n rate RATE:n {
 DAMAGE: 0
 JAM: 0
 loop X over {1..RATE} {
  TEMPDAMAGE: [dam DAM bdam BON at AT plasma PLASMA master MASTER con CON prof PROF shield SHIELD av AV JAM rolled 2d10]
  DAMAGE: DAMAGE + TEMPDAMAGE
 }
 result: [floor DAMAGE]
}

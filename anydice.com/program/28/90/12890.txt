function: rollatk DIE:n plus BONUS:n vs AC:n {
 if DIE = 20 { result: CRITDMG}
 if DIE = 1 { result: 0}
 if DIE + BONUS >= AC { result: DMG}
 result: 0
}

function: rollmatk NATKS:n atks on DIE plus BONUS:n vs AC:n {
 TOTAL: 0
 loop X over {1..NATKS}{
  DMG: [rollatk DIE plus BONUS vs AC]
  TOTAL: TOTAL + DMG
 }
 result: TOTAL
}

function: ovrcombat ROUNDS:n of combat with NATKS:n atks on a DIE plus BONUS:n vs AC:n{
 TOTAL: 0
 loop X over {1..ROUNDS}{
  DMG: [rollmatk NATKS on DIE plus BONUS vs AC]
  TOTAL: TOTAL + DMG
 }
 result: TOTAL
}

\ === UNIQUE PASSIVE STATS === \
\ Passives that can only roll once across all slots \
\ Each gets a sentinel (identity) and a ranges (real stat-range count) \
\ To add a new unique passive: \
\   1. Define XX_SENTINEL and XX_RANGES \
\   2. Add an entry to UNIQUE_STATS \
\   3. Add an if-line to passive, decode, and cap uniques \
DS_SENTINEL: 101
DS_RANGES:   3
UNIQUE_STATS: {
  3  \ Double Strike \
}
TOTAL_PASSIVES: #PASSIVE_STATS + #UNIQUE_STATS

function: passive R:n {
  if R = 1 { result: DS_SENTINEL } \ Double Strike \
  result: PASSIVE_DIE
}

function: decode P:n {
  if P = DS_SENTINEL { result: DS_RANGES } \ Double Strike \
  result: P
}

function: cap uniques A:n B:n C:n {
  if UNIQUE_PASSIVES_BUGGED = 1 { result: [decode A] + [decode B] + [decode C] }
  DECODED: [decode A] + [decode B] + [decode C]
  REROLLS: 0
  REFUND: 0
  \ Double Strike \
  DS_EXTRAS: (A = DS_SENTINEL) + (B = DS_SENTINEL) + (C = DS_SENTINEL) - 1
  if DS_EXTRAS > 0 {
    REROLLS: REROLLS + DS_EXTRAS
    REFUND: REFUND + DS_EXTRAS * DS_RANGES
  }
  \ Add more unique passives here following the same pattern \
  if REROLLS = 0 { result: DECODED }
  if REROLLS = 1 { result: DECODED - REFUND + PASSIVE_DIE }
  result: DECODED - REFUND + PASSIVE_DIE + PASSIVE_DIE
}

function: W:n weapon {
  P_A: [passive d(TOTAL_PASSIVES)]
  P_B: 0
  P_C: 0
  if W = 0    { P_B: [passive d(TOTAL_PASSIVES)] }
  if BOSS = 1 { P_C: [passive d(TOTAL_PASSIVES)] }
  result: [roll W + [cap uniques P_A P_B P_C] stats] - 1
}

function: X:n worn {
  if      X  = 1  { result: 5 }
  else if X <= 6  { result: 3 }
  else if X <= 16 { result: 1 }
  else { result: 0 }
}

function: roll X:n stats {
  if SHOP = 1 { result: 40 + (Xd61/X) + [d100 worn] }
  result: (Xd101/X) + [d100 worn]
}

function: dismantle X:n {
  if      X <= 20  { R: 1 }
  else if X <= 40  { R: 3 }
  else if X <= 60  { R: 5 }
  else if X <= 80  { R: 25 }
  else if X <  95  { R: 300 }
  else if X < 100  { R: 1000 }
  else { R: 5000 }
  if BOSS = 1 { result: R * 3 / 2 }
  result: R
}

output [dismantle [WEAPON_DIE weapon]]

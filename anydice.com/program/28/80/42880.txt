function: unique values in SEQ:s {
  N: 0
  loop M over {#SEQ@[sort SEQ]..1@[sort SEQ]}{
    if [SEQ contains M] {
      N: N + 1
    }
  }
  result: N
}

function: highest value not in SEQ:s max M:n {
  loop N over [reverse {1..M}]{
    if ![SEQ contains N] {
      result: N
    }
  }
}

output [unique values in 5d6] named "Count of unique values"
output [highest value not in 5d6 max 6] named "Highest value not on 5d6"

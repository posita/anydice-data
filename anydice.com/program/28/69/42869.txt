TO_HIT: 8
NORMAL_DAMAGE: 1d4 + 4
CRIT_DAMAGE: 2d4 + 4

\ 2. Single attack logic \
function: attack ROLL:n vs AC:n {
 if ROLL = 20 { result: CRIT_DAMAGE }
 if ROLL = 1 { result: 0 }
 if ROLL + TO_HIT >= AC { result: NORMAL_DAMAGE }
 result: 0
}

\ 3. Running 10 attacks vs various ACs \
loop AC over {15, 18, 20} {
 output 10d[attack d20 vs AC] named "10 Attacks vs AC [AC]"
}

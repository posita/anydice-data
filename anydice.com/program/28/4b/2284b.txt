    HIT: 20 - (AC - H) + 1
    MISS: 20 - HIT
    HIT: HIT - #C

    \The more damage dice, the worse this issue becomes\

    \result: 1d{0:MISS, (DAM+MOD):HIT, (DAM+DAM+MOD):#C}\

    \This simpler formula will only account for misses, not crits\
    \However it also assumes a nat-20 will miss when you have a +0 against a 21 AC\
    \Even though by the rules crits should always hit\

    result: (1d20 + H >= AC) * (DAM + MOD)

function: hit H:n for DAM:d plus MOD:n {
    result: [hit H for DAM plus MOD crit {20}]
}

function: hit H:n for DAM:d {
    result: [hit H for DAM plus 0 crit {20}]
}

function: save vs DC:n for DAM:d {
    X: 20 - (DC - SAVE) + 1
    result: 1d{0:X, 1:(20-X)} * DAM
}

function: savehalf DC:n for DAM:d {
    X: 20 - (DC - SAVE) + 1
    result: (1d{1:X, 2:(20-X)} * DAM) / 2
}

function: reroll S:s of D:d {
    CNT: #D
    MAX: [maximum of D] / CNT
    RES: 0
    loop N over {1..CNT} {
        RES: RES + 1d{{(#S + 1)..MAX}:(MAX + #S), S:(#S)}
    }
    result: RES
}

function: X times Y {
    Z: 0
    loop N over {1..X} {
        Z: Z + Y
    }
    result: Z
}

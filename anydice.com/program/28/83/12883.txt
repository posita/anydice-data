AC: 15
PROFICIENCY: 4

function: power attack DMG:d crit CRIT_DMG:d with attack mod MOD:n penalty PENALTY:n roll ROLL:n{
    DMG_WITH_BONUS: DMG + MOD + 10
	
    if ROLL = 1 {
        result: 0
    }
    if ROLL = 20 {
        result: DMG_WITH_BONUS + CRIT_DMG
    }
    if ROLL + MOD + PROFICIENCY - PENALTY >= AC {
        result: DMG_WITH_BONUS
    }
    result: 0
}

output
4d[power attack 1d6 + 1d6 crit 1d6 with attack mod 5 penalty 3 roll 1d20] named "Sharpshooting XBE Gloomy"

output
2d[power attack 1d{3..10} + 5d8 crit 1d{3..10} + 5d8 with attack mod 4 penalty 5 roll [highest 1 of 2d20]] + 
[power attack 1d{3..4} + 5d8 crit 1d{3..4} + 5d8 with attack mod 4 penalty 5 roll [highest 1 of 2d20]] named "Vengeance Pally with GWF/GWM/PAM"

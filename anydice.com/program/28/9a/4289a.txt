function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}


\output d12 named "Greataxe"\
\output [reroll {1,2} on d12 to 3] named "Greataxe with Great Weapon Fighting"\
output [highest of [reroll {1,2} on d12 to 3] and [reroll {1,2} on d12 to 3]] named "Greataxe with Great Weapon Fighting and Savage Attacker"



\output 2d6 named "Greatsword"\
\output [reroll {1,2} on d6 to 3] + [reroll {1,2} on d6 to 3] named "Greatsword with Great Weapon Master"\

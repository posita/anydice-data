function: forcar A:s {
 if #A<3{result:A}
 if 1@A=6 & 2@A=6{result:12}
 T:#A-[count 1 in A]-2
 if T<=0{result:{1,2}@A}
 result:[comparar Td6 com 1@A e 2@A]
}
function: comparar A:s com B:n e C:n{
 if 2@A>=B {result:{1,2}@A}
 if 1@A>=B | 1@A>=C {result:1@A+B}
 result:B+C
}

loop X over {1..7}{
 \output {1,2}@Xd6 named "[X]d"\
 output [forcar Xd6] named "[X]d forçando"
}

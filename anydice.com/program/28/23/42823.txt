set "maximum function depth" to 6

function: count even rolls ROLL:n depth DEPTH:n {
  if DEPTH > 20 { result: 0 } \- Zastavení rekurze -\
  
  if ROLL = 6 { result: 1 } \- Našli jsme šestku, započítáme ji a končíme -\
  
  if ROLL = 2 | ROLL = 4 { 
    \- Je to sudé číslo, započítáme ho a házíme znovu -\
    result: 1 + [count even rolls 1d6 depth DEPTH + 1]
  }
  
  \- Je to liché číslo, nezapočítáme ho, ale házíme znovu -\
  result: [count even rolls 1d6 depth DEPTH + 1]
}

output [count even rolls 1d6 depth 1] named "Počet sudých hodů než padne 6"

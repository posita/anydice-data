set "position order" to "lowest first"

function: A:s lower than B:n {
C: 0
loop N over A {
if N <= B {
C: C + 1
}
}
result: C
}

function: sim ROLL:s STR:n {
SORTED: [sort ROLL]
if 1@SORTED < STR {
result: STR - 1@SORTED + [ROLL lower than STR]
}
else {
result: 1
}
}

function: roll HP:n with STR:n str {
ROLL: HPd6
result: [sim ROLL STR]
}
loop S over {1..6} {
loop H over {1..6} {
output [roll H with S str] named "[H]d6 ([S])"
}
}

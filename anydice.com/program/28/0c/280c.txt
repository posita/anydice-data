\ FUNCTIONS \
function: get successes ROLL:s {
   result: (ROLL>=5)
}

function: check LIMIT:n AROLL:s {
   if ([get successes AROLL])>LIMIT {result: LIMIT}
   result: [get successes AROLL]
}

function: check DPOOL:s with AUTO:n{
   result: [get successes DPOOL]+AUTO
}

\ SR5E DICEPOOL COMPARISON \
\ Step 1: Replace the values below as needed. \
\ Step 2: Calculate and explore results \

\ Aggressor Details : Aggressor is the one making the test \
AGRESSORDICEPOOL: 24d6 \ <-- This is the aggressor's dice pool with nd6 \
AGRESSORLIMIT: 7 \ <-- This is the applicable limit to the test, ie. accuracy, set to n if you want no limits \

\ Defender Details : Defender is the one opposing the test \
DEFENDERDICEPOOL : 12d6 \ <-- This is the defender's dice pool with nd6 \
DEFENDERAUTOSUCCESSES : 0 \ <-- Use this for immunity and hardened armor \

\output [check AGRESSORLIMIT AGRESSORDICEPOOL] named "Aggressor Results" \
\output [check DEFENDERDICEPOOL with DEFENDERAUTOSUCCESSES] named "Defender Results" \
output [highest of 0 and ([check AGRESSORLIMIT AGRESSORDICEPOOL]-[check DEFENDERDICEPOOL with DEFENDERAUTOSUCCESSES])] named "Net Successes of Aggressor versus Defender"

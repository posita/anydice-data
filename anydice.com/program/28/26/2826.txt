function: count VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: COUNT
}

function: double S:n {
 if S>1 {
  result: S*2
 }
 result: S
}

function: legend A:s {
 B: 0
 loop N over {1..6} {
  B: B + (N * [double [count N in A] ])
 }
 result: B
}

output [legend 5d6]

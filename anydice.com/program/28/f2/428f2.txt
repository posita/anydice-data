function: compare FIRST:s and SECOND:s {
  USED: 0  \ how many dice from the second pool have been used? \ 
  loop ROLL over FIRST {
    if ROLL > (USED + 1)@SECOND {
      result: ROLL
    } else {
      USED: USED + 1
    }
  }
  result: 0  \ all dice in the first pool were eliminated \
}

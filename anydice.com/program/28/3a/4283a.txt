set "position order" to "highest first" \ the default behavior \

MISS:0
HIT:1
CRIT:2

function:attack ROLL:n vs DEFENSE:n {
if ROLL = 1 {result: MISS}
if ROLL = 20 {result:CRIT}
if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

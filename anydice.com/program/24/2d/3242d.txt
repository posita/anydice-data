function: attack ATTACK:d {
  result: ATTACK
}

function: reroll R:n N:n D:d{ 
  if R < N { result: D } else {result: R}
}

function: rollhit R:n M:n AC:n C:n {
  ROLL: R + M
  if R >= C { result: 2}
  if ROLL >= AC { result:  1}
  else { result: 0}
}

function: rolldamage H:n RE:n R:d M:n {
  DAMAGE: 0
  if H >= 1 {
    if RE = 1 {DAMAGE: [reroll R 3 R]}
    else {DAMAGE: [attack R]}
    if H = 2 {DAMAGE: DAMAGE + DAMAGE + 1d6}
    DAMAGE: DAMAGE + M
  }
  result: DAMAGE
}

function: gwm H:n RH:d MH:n AC:n C:n RD:d MD:n RE:n {
  DAMAGE: 0
  if H = 2{ 
    HIT: [rollhit RH MH AC C]
    DAMAGE: [rolldamage HIT RE RD MD]
  }
  result: DAMAGE
}

function: roll X:n RH:d MH:n AC:n C:n RD:d MD:n RE:n GW:n{
  TOTAL: 0
  BONUS: 1
  loop N over {1..X} {
    HIT: [rollhit RH MH AC C]
    DAMAGE: [rolldamage HIT RE RD MD]
    TOTAL: TOTAL + DAMAGE
    if GW = 1 { if BONUS = 1{
      DAMAGE: [gwm HIT RH MH AC C RD MD RE]
      TOTAL: TOTAL + DAMAGE
      BONUS: 0
    }}
  }
  result: TOTAL
}

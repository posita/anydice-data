AC: 15
PROFICIENCY: 3

function: gwmattack ROLL:n plus DEX:n for DMG:d crit CRIT_DMG:d {
   
    if ROLL = 1 {
        result: 0
    }
    if ROLL = 20 {
        result: 10 + DMG + CRIT_DMG
    }
    if ROLL + DEX + PROFICIENCY - 5 >= AC {
        result: 10 + DMG
    }
    result: 0
}

output
2d[gwmattack [highest 1 of 2d20] plus 4 for (2d{3..6} + 4 + 2d8) crit (2d{3..6} + 3d8)] named "Great Sword + GWM + (+2 Str L4 ASI) + GWF Style"

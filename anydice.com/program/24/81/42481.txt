MISS: 0
WEAK_HIT: 1
STRONG_HIT: 2
CRIT: 3

function: task HOPE:n and FEAR:n vs TARGET:n with STRESS:n {
 ROLL: HOPE + FEAR
 if ROLL < STRESS { result: MISS }
 if HOPE = FEAR { result: CRIT }
 if ROLL <= TARGET {
   if HOPE > FEAR { result: STRONG_HIT }
   result: WEAK_HIT
 }
 result: MISS
}

loop STRESS over {0..10} {
  output [task d10 and d10 vs 13 with STRESS] named "Stress [STRESS]"
}

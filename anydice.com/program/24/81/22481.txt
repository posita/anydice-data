P:2
S:9
MA:1d20+5
MDN:1
MDR:12
MDD:3
MD:MDNdMDR+MDD
MDC:[maximum of MD]+MDR
PD:8+P+D
PDC:PD+10
A:[count {PD..PDC-1} in MA]
AC:[count {PDC..[maximum of MA]} in MA]
MDA:[count {S..[maximum of MD]} in MD]
MDAC:[count {S..[maximum of MDC]} in MDC]
if PD<=[maximum of MA]{
   if S<[maximum of MD]{
      if PDC<=[maximum of MA]{
         if S<[maximum of MDC]{
            output A*MDA named "вероятность нанести урон"
            output A*MDA*(([maximum of MD]+MDD+MDN)/2)
            output AC*MDAC named "вероятность нанести крит урон"
         }
      }
      else{
         output A*MDA named "вероятность нанести урон"
      }
   }
}

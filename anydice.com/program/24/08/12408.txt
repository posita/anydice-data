AC: 14
DAMAGE_BONUS: 3
HP: 21
loop N over {1..10} 
{
  STRIKE_ROLL: (1d20)
  HITCHANCE: (STRIKE_ROLL >= AC)
  NATURAL_TWENTY: (STRIKE_ROLL = 20)
  DAMAGE: NATURAL_TWENTY*1d6 + HITCHANCE*(1d6 + DAMAGE_BONUS)
  HP: HP - DAMAGE
  output HP <= 0 named "dead in [N] round(s)"
}
output DAMAGE named "Damage taken"

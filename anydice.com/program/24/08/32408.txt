function: reroll R:n under N:n {
   if R < N { result: 1d6 } else {result: R}
}
GWF: 2 d [reroll 1d6 under 3]
NORMALDAMAGE: GWF + 1d6 + 1d4 +13
CRITDAMAGE: GWF + GWF + 2d6 + 2d4 + 13
GWMANDELIXIR: GWF + 1d6 + 1d4 + 27
GWMEXLIXIRCRIT: GWF + GWF + 2d6 + 2d4 + 27

output NORMALDAMAGE named "Normal Damage" 
output CRITDAMAGE named "Normal Critical Damage"
output [highest of NORMALDAMAGE and NORMALDAMAGE] named "Savage Attacker Damage"
output [highest of CRITDAMAGE and CRITDAMAGE] named "Savage Attacker Critical Damage"
output GWMANDELIXIR named "With Great Weapon Master and Strength Elixir"
output GWMEXLIXIRCRIT named "Great Weapon Master and Elixir Crits"

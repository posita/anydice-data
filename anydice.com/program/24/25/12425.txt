function: election DIE:n BONUS:n OPENSEATS:n {

X: DIE + BONUS + OPENSEATS

if X < 11{
  RESULT : -1
}

if X > 10 {
  RESULT : 0
}
if X > 16{
  RESULT : 1
}
 result: RESULT
}

function : seatcalc RESULTS:n{

 if RESULTS = -1{
   result : OPENSEATS + 2
 }
 if RESULTS = 0{
   result : OPENSEATS 
 }
 if RESULTS = 1{
   result : OPENSEATS - 2 
 } 
}

OPENSEATS : 0
CN : -3
PC : 0
NI : 0
FA : 0
UCL : 0
M : 0
FP : 0

CNR :[election 1d20 CN OPENSEATS] 
OPENSEATS : [seatcalc CNR]
PCR :[election 1d20 PC OPENSEATS] 
OPENSEATS : [seatcalc PCR]
NIR :[election 1d20 NI OPENSEATS] 
OPENSEATS : [seatcalc NIR]
FAR : [election 1d20 FA OPENSEATS] 
OPENSEATS : [seatcalc FAR]
UCLR :[election 1d20 UCL OPENSEATS] 
OPENSEATS : [seatcalc UCLR]
MR : [election 1d20 M OPENSEATS] 
OPENSEATS : [seatcalc MR]
FPR :[election 1d20 FP OPENSEATS] 

output CNR named "Constitutional Nationalists"
output PCR named "Purity Club"
output NIR named "New Independents"
output FAR named "Fairness Association"
output UCLR named " United Communist League"
output MR named "Monarchists"
output FPR named "Fringe party"

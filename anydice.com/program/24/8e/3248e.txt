function: explode N:n {
 if N > 6 { result: N + [explode d8] }
 result: N
}

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}
function: rereroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

set "maximum function depth" to 4
output [rereroll {1,2} on [explode [rereroll {1,2} on d8]]]
output [rereroll {1,2} on 1d12]

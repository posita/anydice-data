E_A : 1d10
E_B : 2d10
E_C : 3d10
P : 1d8

function: best of DICE:s {
 MAX: 0
 loop N over DICE{
  if N > MAX { MAX: N }
 }
 result: MAX
}


output [highest 1 of E_A] named "1 x Effort"
output [highest 1 of E_B] named "2 x Effort"
output [highest 1 of E_C] named "3 x Effort"
output [highest of [highest 1 of E_A] and [highest 1 of P]] named "1 x Effort + Power"
output [highest of [highest 1 of E_B] and [highest 1 of P]] named "2 x Effort + Power"
output [highest of [highest 1 of E_C] and [highest 1 of P]] named "3 x Effort + Power"

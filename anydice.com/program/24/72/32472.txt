function: splitpoolhelper A:n BH:n BL:n MA:n MB:n {
 if A = 6 & BH = 6 { result: MA+MB }
 result: A+BH
}
function: splitpool A:d B:d {
 result: [splitpoolhelper A [highest 1 of B] [lowest 1 of B] [maximum of A] [maximum of 1@B]]
}

loop S over {1..5} {
 output [splitpool d4 Sd6]
}

function: suitpoolhelper SUITS:n MAIN:n HIGH:n {
 SUITMAX: 12/SUITS  \6\
 MAINSUIT: (MAIN-1)/SUITMAX   \0..1\
 MAINROLL: MAIN-(MAINSUIT*SUITMAX)  \1..6\
 HIGHSUIT: (HIGH-1)/SUITMAX   \0..1\
 HIGHROLL: HIGH-(HIGHSUIT*SUITMAX)  \1..6\
 if MAINROLL = 1 & MAINSUIT = HIGHSUIT { result: 1 }
 else if MAINROLL = SUITMAX & MAINSUIT = HIGHSUIT { result: (SUITMAX*2)+1 }
 result: MAINROLL + HIGHROLL
}
function: suitpool SUIT:n ROLL:n POOLSUITS:s POOLROLLS:s {
 HIGHSUIT: 1@POOLSUITS
 HIGHROLL: 1@POOLROLLS
 if ROLL = 1 {
  if ![POOLSUITS contains SUIT] { result: 2 }
  else { result: 21 }
 }
 result: ROLL + HIGHROLL
}

loop S over {1..3} {
 output [suitpool d4 1d10 Sd4 Sd10]
}

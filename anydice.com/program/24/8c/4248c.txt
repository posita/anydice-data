\Get a breakdown of 2dX + 0 to Y group by boundaries that represent the below

0 = Mishap 
1 = Success With Mishap (Total roll >= 3:THRESHHOLDS)
2 = Success (Total roll >= 2:THRESHHOLDS)
3 = Critical Success (Total roll >= 1:THRESHHOLDS)
\

function: tieredresolution ROLL:s THRESHHOLDS:s{
TOTAL:1@ROLL + 2@ROLL 

    if TOTAL >= 1@THRESHHOLDS{
        result: 3
    }
    
    if TOTAL >= 2@THRESHHOLDS{
        result: 2
    }
    
    if TOTAL >= 3@THRESHHOLDS{
        result: 1
    }
    
    result: 0
}

LOWESTMOD:0
HIGHESTMOD:0

loop N over {LOWESTMOD..HIGHESTMOD}{
    output [tieredresolution 2d6+N {12,10,7}] named "2d6+[N]"
    output [tieredresolution 2d8+N {16,12,8}] named "2d8+[N]"
    output [tieredresolution 2d10+N {20,15,10}] named "2d10+[N]"
    output [tieredresolution 2d12+N {24,16,12}] named "2d12+[N]"
    output [tieredresolution 2d20+N {40,30,20}] named "2d20+[N]"
}

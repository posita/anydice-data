\
 This example counts each 6 as a success and each 5 as half a success
 (rounded down).
\
function: score ROLL:s{
 result: [count 6 in ROLL] + [count 5 in ROLL] / 2
}

output [score 1d6]
output [score 2d6]
output [score 3d6]
output [score 4d6]
output [score 5d6]
output [score 6d6]
output [score 7d6]
output [score 8d6]
output [score 9d6]
output [score 10d6]

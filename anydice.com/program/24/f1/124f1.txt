function: attack ROLL:n vs AC:n {
	if ROLL = 20 { result: CRIT_DAMAGE }
	if ROLL = 1 { result: 0 }
	if ROLL + TO_HIT >= AC { result: BASE_DAMAGE }
	result: 0
}

ENEMY_AC: d{10..20}
STR_MOD: 4
PROF: 4
TO_HIT: STR_MOD + PROF
ATTACKS: 3

BASE_DAMAGE: 1d8 + STR_MOD + 2
CRIT_DAMAGE: 2d8 + STR_MOD + 2
ATTACK_COUNT: ATTACKS
output ATTACK_COUNTd[attack d20 vs ENEMY_AC] named "Longsword /w Duelist ([ATTACKS] attacks)"

BASE_DAMAGE: 2d6 + STR_MOD
CRIT_DAMAGE: 4d6 + STR_MOD
ATTACK_COUNT: ATTACKS
output ATTACK_COUNTd[attack d20 vs ENEMY_AC] named "Greatsword Fighting([ATTACKS+1] attacks)"

BASE_DAMAGE: 1d4 + STR_MOD
CRIT_DAMAGE: 2d4 + STR_MOD
ATTACK_COUNT: ATTACKS + 1
output ATTACK_COUNTd[attack d20 vs ENEMY_AC] named "Daggers /w Two-Weapon ([ATTACKS] attacks)"

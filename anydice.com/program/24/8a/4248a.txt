\Get a psobibilty of getting a success or a mishap of 2dX + 0 to Y. 

0 = Mishap 
1 = Success With Mishap (Total roll >= 3:THRESHHOLDS)
2 = Success (Total roll >= 2:THRESHHOLDS)
3 = Critical Success (Total roll >= 1:THRESHHOLDS)
\

function: mishapsuccessratio ROLL:s THRESHHOLDS:s TYPE:n{
TOTAL:1@ROLL + 2@ROLL 

	if TOTAL >= 1@THRESHHOLDS{
		result: 1
	}
	
	if TOTAL >= 2@THRESHHOLDS{
		result: 1
	}
	
	if TOTAL >= 3@THRESHHOLDS{
		result: TYPE
	}
	
	result: 0
}


LOWESTMOD:0
HIGHESTMOD:0

loop N over {LOWESTMOD..HIGHESTMOD}{
 output [mishapsuccessratio 2d10+N {20,15,10} 0] named "2d10+[N] Mishap Chance"
 output [mishapsuccessratio 2d10+N {20,15,10} 1] named "2d10+[N] Success Chance"
}

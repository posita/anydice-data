function: attack ROLL:n vs AC:n {
	if ROLL = 20 { result: CRIT_DAMAGE }
	if ROLL = 1 { result: 0 }
	if ROLL + TO_HIT >= AC { result: BASE_DAMAGE }
	result: 0
}

ENEMY_AC: d{10..20}
STR_MOD: 5
PROF: 6
TO_HIT: 9
ATTACKS: 1

BASE_DAMAGE: 1d8 + 7
CRIT_DAMAGE: 2d8 + 7
ATTACK_COUNT: ATTACKS
output ATTACK_COUNTd[attack d20 vs ENEMY_AC] named "Longsword ([ATTACKS] attacks)"

BASE_DAMAGE: 2d6 + 5
CRIT_DAMAGE: 4d6 + 5
ATTACK_COUNT: ATTACKS
output ATTACK_COUNTd[attack d20 vs ENEMY_AC] named "Daggers ([ATTACKS] attacks)"

BASE_DAMAGE: 1d4 + 5
CRIT_DAMAGE: 2d4 + 5
ATTACK_COUNT: ATTACKS + 1
output ATTACK_COUNTd[attack d20 vs ENEMY_AC] named "Daggers ([ATTACKS] attacks)"

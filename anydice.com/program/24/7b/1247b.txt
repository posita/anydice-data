                                                \****************************************************************************\
EAC: 17                                         \ Enemy AC                                                                   \
GHM: 10                                         \ Global Hit Modifier                                                        \
GCT: 20                                         \ Global Crit Threshold                                                      \
HIT: 1d20                                       \ Hit Dice (Don't Change)                                                    \
                                                \****************************************************************************\
function: acc A:n B:n C:n D:n E:n G:n I:n{      \ Begin ACC function                                                         \
if A=1 {H: 0}                                   \ if 1d20 equals 1, miss                                                     \
else if A>1 & A<C & A+B < EAC {H: 0}            \ if 1d20 plus attack mod is less than enemy AC, miss                        \
else if A>1 & A<C & A+B > EAC {H: D+E}          \ if 1d20 != 20 & added attack dice is more than enemy AC, calculate damage  \
else if A>=C {H: D*2+E}                         \ if 1d20 equals 20, double dice and add damage modifiers.                   \
  TOTAL: 0                                      \ Multiattack loop                                                           \
  loop N over {1..G} {                          \ Multiattack loop                                                           \
    if I>=1 {J: ((N*3)-3)} else {J: 0}          \ Multiattack loop (Dobor Sword Consideration)                               \
    DAMAGE: [attack H]                          \ Multiattack loop                                                           \
    TOTAL: TOTAL + DAMAGE + J                   \ Multiattack loop                                                           \
  }                                             \ Multiattack loop                                                           \
  if H=0 {result: 0} else {result: TOTAL}       \ Multiattack loop                                                           \
                                                \ Multiattack loop                                                           \
}                                               \ Multiattack loop                                                           \
                                                \****************************************************************************\
function: attack ATTACK:d {
  result: ATTACK
}
                                                \***********************\
function: gwf A:n B:n{                          \ Great Weapon Fighting \
   if A < 3 {result: B} else {result: A}        \ Great Weapon Fighting \
}                                               \ Great Weapon Fighting \
                                                \***********************\

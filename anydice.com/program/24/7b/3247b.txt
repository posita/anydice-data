
PROFICIENCY_BONUS: 5
EXPERTISE: PROFICIENCY_BONUS * 2
DEX_BONUS: 5
WIS_BONUS: 1

ITEM_ALL_SKILLS: 1 \ Stone of Good Luck \
ITEM_SLIGHT: 5 + ITEM_ALL_SKILLS \ Gloves of Thievery \
ITEM_STEALTH: 0 + ITEM_ALL_SKILLS
ITEM_PERCEPTION: 0 + ITEM_ALL_SKILLS

STEALTH: EXPERTISE + DEX_BONUS + ITEM_STEALTH
PERCEPTION: EXPERTISE + WIS_BONUS + ITEM_PERCEPTION
SLIGHT: EXPERTISE + DEX_BONUS + ITEM_SLIGHT

FIXED_DC: 18
RANDOM_DC: 2d10 + d4 + 5

\ CRIME: stealth, perception and slight of hand \
output 1@2d20 + STEALTH       >= RANDOM_DC &
       d20    + PERCEPTION    >= RANDOM_DC &
       d20    + SLIGHT        >= RANDOM_DC
       named "CRIME Random DC"
       
output 1@2d20 + STEALTH       >= FIXED_DC &
       d20    + PERCEPTION    >= FIXED_DC &
       d20    + SLIGHT        >= FIXED_DC
       named "CRIME Static DC"


\ BUY MAGIC ITEM: slight of hand \
output d20    + SLIGHT        >= RANDOM_DC named "BUY MAGIC ITEM Random DC"
output d20    + SLIGHT        >= FIXED_DC named "BUY MAGIC ITEM Static DC"

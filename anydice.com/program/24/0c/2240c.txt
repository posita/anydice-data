function: roll D:n {
    if D = 1  { result: D-1d10 }
    if D = 10 { result: D+1d10 }
    result: D
}

DIE: [roll 1d10]

function: ranged ROLL:n vs DV:n for DAMAGE:d {
    if ROLL > DV {result: DAMAGE}
    result: 0
}

function: autofire ROLL:n vs DV:n max MAX {
    if ROLL <= DV { result: 0 }
    result: [lowest of ROLL - DV and MAX] * 2d6
}

loop DV over {-3..5} {
    output [autofire DIE vs DV max 3] named "SMG Autofire vs [DV]"
}
\
loop DV over {-3..5} {
    output [autofire DIE vs DV max 4] named "AR Autofire vs [DV]"
}
\
loop DV over {-3..5} {
    output [ranged DIE vs DV for 4d6] named "VH Pistol vs [DV]"
}

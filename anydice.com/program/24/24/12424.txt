function: election DIE:n BONUS:n {

X: DIE+BONUS+OPENSEATS

if X < 11{
  RESULT : -1
  OPENSEATS: OPENSEATS + 2
}

if X > 10 {
  RESULT : 0
}
if X > 24 {
  RESULT : 1
  OPENSEATS: OPENSEATS - 2
}
 result: RESULT
}

OPENSEATS : 0
CN : -3
PC : 0
NI : 0
FA : 0
UCL : 0
M : 0
FP : 0

output [election 1d20 CN] named "Constitutional Nationalists"
output [election 1d20 PC] named "Purity Club"
output [election 1d20 NI] named "New Independents"
output [election 1d20 FA] named "Fairness Association"
output [election 1d20 UCL] named " United Communist League"
output [election 1d20 M] named "Monarchists"
output [election 1d20 FP] named "Fringe party"

function: xpls DS:n {
  S: 0
  if (DS=6) {
    S: 1+[xpls d6]
  }
  if (DS=5) {
    S: S+1
  }
  result: S
}

function: dten DT:n {
  ST:0
  if (DT=10) {
    ST: 2 + [xpls d6]
  } else {
    if (DT=5) {
      ST:1
    } else { 
      if (DT>=6) {
        ST: 1 + [xpls d6]
      }
    }
  }
  result: ST
}

output 2d[xpls d6] named "2d6 exploding successes"
output [dten d10]  named "just a d10 with exploding successes"
output 3d[xpls d6] + [dten d10] named "3d6 and a d10 with exploding successes"

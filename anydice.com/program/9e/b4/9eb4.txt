function: add helper RES:n {
  if RES = MAX { result: 1 + [add helper DIE] }
  if RES <= TAR { result: 1 }
  result: 0
}

function: add DIE:d on TAR:n trig MAX:n {
  result: [add helper DIE]
}

loop N over {1..10} {
  output [lowest of (Nd[add 1d10 on 5 trig 1]) and 12]
  loop B over {1..4} {
    output [lowest of (Nd[add 1d10 on 5 trig 1] + [add 1d10 on B trig 1]) and 12]
  }
}

DC: 10
PROF: 2
EXP: PROF * 2
function: X:n wr W:n{
if X > DC {result: (X + W-DC)}
result: 0
}

function: X:n wrf W:n{
if X > DC {result: (X + W-DC)}
result: X + W-DC
}

function: X:n rep{
if X > DC {result: X-DC}
result: 0
}

function: X:n repf{
if X > DC {result: X-DC}
result: X-DC
}


function: X:n reduce C:n{
if X > DC-C {result: X-(DC-C)}
result: 0
}

function: X:n reroll{
if X > DC {result: X-DC}
else      {result: [d20 rep]}
}


output [d20 rep]                 named " NORM "
output [d20+PROF rep]            named " PROF "
output [d20+EXP rep]             named " EXPERT "
output [d20 reduce 2]            named " DC - 2 "
output [d20 wr 5]              named " WR + 5 "
output [[highest 1 of 2d20] rep] named " ADV "
output [d20 reroll] named " REROLL " 
output [[highest 1 of 3d20] rep] named " SUPER ADV " 
output [[highest 1 of 2d20]+PROF rep] named " FAVOUR + PROF " 

output [d20+1 repf] named "tier 1 normal"
output [d20+1+PROF repf] named "tier 1 ideal"
output [d20+3+PROF repf] named "tier 2 normal"
output [d20+3+EXP repf] named "tier 2 ideal"
output [d20+5+PROF repf] named "tier 3 normal"
output [d20+5+EXP wrf 10] named "tier 3 ideal"
output [ [highest 1 of 2d20] +5+PROF repf] named "tier 4 normal"
output [ [highest 1 of 2d20] +5+EXP wrf 10] named "tier 4 normal"

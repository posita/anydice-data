function: attack W:n damage X:n versus grace Y:d brawn Z:d armor U:n V:n {
E: [maximum of Y] + U
MDT: [maximum of Z] + V
HDT: ([maximum of Z]+V)*2
 if W >= E {
  if X < MDT {
   result: 1
  }
  else if X >= MDT & X < HDT {
   result: 2
  }
  else {
   result: 3
  }
 }
 else {
  result: 0
 }
}

function: attack W:d damage X:d versus grace Y:d brawn Z:d armor U:n V:n until ko {
 H: [maximum of Z]
 result: [attack W damage X versus grace Y brawn Z armor U V until ko helper H]
}

function: attack W:d damage X:d versus grace Y:d brawn Z:d armor U:n V:n until ko helper G:n{
 if G > 0 {
  G: G - [attack W damage X versus grace Y brawn Z armor U V]
  result: 1 + [attack W damage X versus grace Y brawn Z armor U V until ko helper G]
 }
 else {
  result: 1
 }
}

function: hard by N:n {
  MAX: 20-N
  result: d{0:N, 1..MAX}
}

REGULAR: d20
ADVANTAGE: 1@2d20
DISADVANTAGE: 2@2d20
EASY: d20+3
HARD: [hard by 3]
 
output REGULAR named "d20"
output ADVANTAGE named "advantage"
output EASY named "easy"
output DISADVANTAGE named "disadvantage"
output HARD named "hard"

\
TARGET: 12
output REGULAR > TARGET named "d20 vs [TARGET]"
output ADVANTAGE > TARGET named "advantage vs [TARGET]"
output EASY > TARGET named "easy vs [TARGET]"
output DISADVANTAGE > TARGET named "disadvantage vs [TARGET]"
output HARD > TARGET named "hard vs [TARGET]"

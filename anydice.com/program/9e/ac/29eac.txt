function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

loop AC over {14..20} {
output [attack 1d20 plus 3+5 vs AC for 1d8+5 crit 2d8+5 on 20] +
       [attack 1d20 plus 3+5 vs AC for 1d8+5 crit 2d8+5 on 20] +
       [attack 1d20 plus 3+5 vs AC for 1d8+5 crit 2d8+5 on 20] +
       [attack 1d20 plus 3+5 vs AC for 1d8+5 crit 2d8+5 on 20] named "Whirlwind (+mod) vs AC[AC]"
}

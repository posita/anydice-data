MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs EVASION:n {
  if ROLL < EVASION {
    result: MISS
  }
  if ROLL >= 20 {
    result: CRITICALHIT
  }
  result: HIT
}

function: damage for ATTACKRESULT:n from HITDAMAGE:n and CRITDAMAGE:n{
  if ATTACKRESULT = CRITICALHIT {
    result: CRITDAMAGE
  }
  if ATTACKRESULT = HIT {
    result: HITDAMAGE
  }
  result: 0
}

ORACLE: 2+1d20+2d6
ORACLEHIT: 1d3
ORACLECRIT: [highest 1 of 2d3]
function: oracle vs EVASION:n {
  result: [damage for [attack ORACLE vs EVASION] from ORACLEHIT and ORACLECRIT]
}

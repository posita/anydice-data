function: explode ROLLEDVALUE:n {
 if ROLLEDVALUE = 6 { result: 6 + [explode d4] }
 result: ROLLEDVALUE
}

output [explode d6] named "explode on 6"

function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 5
 
 if SUCCESSES > 0 { result: SUCCESSES}
 if [count 1 in ROLL] { result: -1 }
 result: 0
}

loop DICE over {1..5} {
 output [evaluate DICE d10] named "[DICE]d"
}

function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

loop AC over {14..20} {
output [attack 1d20 plus 3+5-5 vs AC for 2d6+5+10 crit 4d6+5+10 on 20] + 
       [attack 1d20 plus 3+5-5 vs AC for 2d6+5+10 crit 4d6+5+10 on 20] named "GWF vs AC[AC]" 
}

function: damage MISS DAMAGE:n AS STAT{
if DAMAGE <= MISS {result: 0}
result: [highest of DAMAGE+STAT-AS and 0]
}

function: hit ATTACK:n AC MODIFIER DAMAGE:d STAT{
if ATTACK = 20{result: DAMAGE+DAMAGE+STAT}
if ATTACK = 1{result: 0}
if ATTACK+MODIFIER < AC{result: 0}
result: [highest of DAMAGE+STAT and 0]
}

AS: 0
COUNT: 1
DIE: 4
PROF: 2
STAT: 0

output [damage COUNT COUNTd[explode dDIE] AS STAT] named "Damage to hit"
output [hit 1d20 AS+10 PROF+STAT COUNTdDIE STAT] named "Roll to hit"

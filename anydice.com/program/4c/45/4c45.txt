function: hit above AC:n for TOHIT:n plus TOHITADD:n and do DAMAGE plus DAMAGEADD:n
{
  if TOHIT = 1
  {
    result: 0
  }
  else if TOHIT = 20
  {
    result: DAMAGE + DAMAGE + DAMAGEADD + 6d8
  }
  else if AC > TOHIT + TOHITADD
  {
    result: 0
  }
  else
  {
    result: DAMAGE + DAMAGEADD
  }
}

function: reroll ROLL:n on a N:n then ROLLAGAIN:n
{
  if ROLL <= N
  {
    result: ROLLAGAIN
  }
  else
  {
    result: ROLL
  }
}

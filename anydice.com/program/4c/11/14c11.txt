SSIX: {-1, 0, 0, 0, 0, 1}
STEN: {-1, 0, 0, 0, 0, 1, 1, 1, 1, 1}
SPSIX: {0, 0, 0, 0, 1}
SPTEN: {0, 0, 0, 0, 1, 1, 1, 1, 1}
SLTEN: {-1, 0, 0, 0, 0, 1, 1, 1, 1, 2}
SLPTEN: {0, 0, 0, 0, 1, 1, 1, 1, 2}

function: peripheral N {
  result: [N check six siders for botches (5-N)@(5-N)d6]
}

function: N:n check six siders for botches MIN:n {
 if MIN = 1 {
  result: (5-N-1)dSSIX + NdSTEN + 1
 }
 else {
  result: [N now check ten siders for botches N@Nd10]
 }
}

function: N:n now check ten siders for botches MIN:n {
 if MIN = 1 {
  result: (5-N)dSPSIX + (N-1)dSTEN + 1
 }
 else {
  result: (5-N)dSPSIX + NdSPTEN
 }
}


function: alternate peripheral N {
  result: [N check ten siders for botches N@Nd10]
}

function: N:n check ten siders for botches MIN:n {
 if MIN = 1 {
  result: (5-N)dSSIX + (N-1)dSTEN + 1
 }
 else {
  result: [N now check six siders for botches (5-N)@(5-N)d6]
 }
}

function: N:n now check six siders for botches MIN:n {
 if MIN = 1 {
  result: (5-N-1)dSSIX + (N)dSPTEN + 1
 }
 else {
  result: (5-N)dSPSIX + NdSPTEN
 }
}



loop N over {0..5}{
	output (5-N)dSSIX + NdSTEN named "Skill [N]"
	output [peripheral N] named "Skill [N] plus peripheral"
	output [alternate peripheral N] named "Skill [N] plus peripheral ALT"
	
}

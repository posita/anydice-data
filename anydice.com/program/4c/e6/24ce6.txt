SIZE: 20

function: risk N:n of S:s ignore P:n failure {
  if [count 1 in S] > P { result: 1 }
  if [S contains SIZE] { result: SIZE }
  R: 0
  loop X over {1..N} {
    R: R + X@S
  }
  result: R
}

output 1dSIZE named "1d[SIZE]"

loop P over {2..5} {
  output [risk 1 of PdSIZE ignore 0 failure] named "[P]d[SIZE]"
}

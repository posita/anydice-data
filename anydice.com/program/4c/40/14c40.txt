function: hits A B {
 X: 0
 loop N over {1..A}{
  X: X + (1dB / 4)
 }
 result: X
}

output [hits 1 8] >= 1 named "Success with 1d8"
output [hits 2 8] >= 2 named "Success with 2d8"
output [hits 3 8] >= 3 named "Success with 3d8"
output [hits 4 8] >= 4 named "Success with 4d8"
output [hits 5 8] >= 5 named "Success with 5d8"

AC: 16
PROFICIENCY: 2
DEX: 3
CRITRANGE: 20
SNEAK: 0

function: lbow ROLL:n vs AC:n {
 if ROLL >= CRITRANGE { result: LBOWCRIT } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + DEX >= AC { result: LBOW }
 result: 0
}

function: ssraw ROLL:n vs AC:n {
 if ROLL >= CRITRANGE { result: LBOWCRIT+10 } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + DEX -5 >= AC { result: LBOW+10 }
 result: 0
}

function: ssnew ROLL:n vs AC:n {
 if ROLL >= CRITRANGE { result: LBOWCRIT+2dPROFICIENCY } 
 if ROLL = 1 { result: 0 }
 if ROLL + DEX >= AC { result: LBOW+2dPROFICIENCY }
 result: 0
}

LBOW: 1d8+DEX+SNEAK
LBOWCRIT: 2d8+DEX+2dSNEAK

output [lbow [highest 1 of 2d20] vs AC] named "longbow"
output [ssraw [highest 1 of 2d20] vs AC] named "sharpshooter raw"

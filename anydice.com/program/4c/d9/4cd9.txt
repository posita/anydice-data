MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 ATTACK: PROF_MOD + STR_MOD + MAGIC_WEP_HIT_MOD
 if ROLL = 1 { result: MISS }
 if ROLL = 60 { result: CRITICALHIT*3 }
 if ROLL > 39 { result: CRITICALHIT*2 }
 if ROLL >= CRIT_ON { result: CRITICALHIT }
  
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}



function: damage for ATTACKS:s {
 D: 0
 DAMAGE: (DAMAGE_DIE * DAMAGE_DICE)
 loop A over {1..#ATTACKS} {
  if A@ATTACKS = CRITICALHIT { D: D + (DAMAGE * 2) + STR_MOD + MAGIC_WEP_DMG_MOD }
  else if A@ATTACKS = (CRITICALHIT*2) { D: D + (DAMAGE * 4) + STR_MOD + MAGIC_WEP_DMG_MOD }
  else if A@ATTACKS = (CRITICALHIT*3) { D: D + (DAMAGE * 10) + STR_MOD + MAGIC_WEP_DMG_MOD }

  else if A@ATTACKS = HIT { D: D + DAMAGE + STR_MOD + MAGIC_WEP_DMG_MOD }
 }
 result: D
}

function: reroll DIE:d {
 result: [reroll DIE helper]
}

function: reroll N:n helper {
 if N < 3 { result: [highest of N and DIE] }
 result: N
}

function: damage with great weapon master for ATTACKS:s {
 D: 0
 loop A over {1..#ATTACKS} {
  if A@ATTACKS = CRITICALHIT { D: D + (DAMAGE_DICE*2)d[explode DAMAGE_DIE] + STR_MOD + MAGIC_WEP_DMG_MOD }
  else if A@ATTACKS = HIT { D: D + (DAMAGE_DICE)d[explode DAMAGE_DIE] + STR_MOD + MAGIC_WEP_DMG_MOD }
 }
 result: D
}

MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 ATTACK: PROF_MOD + STR_MOD + MAGIC_WEP_HIT_MOD
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKS:s {
 D: 0
 loop A over {1..#ATTACKS} {
  if A@ATTACKS = CRITICALHIT { D: D + (DAMAGE * 2) + STR_MOD + MAGIC_WEP_DMG_MOD }
  else if A@ATTACKS = HIT { D: D + DAMAGE + STR_MOD + MAGIC_WEP_DMG_MOD }
 }
 result: D
}

function: strike {
 result: [damage for [attack d20 vs DEFENSE]]
}
function: strike with advantage {
 result: [damage for [attack [highest of d20 and d20] vs DEFENSE]]
}
function: strike with disadvantage {
 result: [damage for [attack [lowest of d20 and d20] vs DEFENSE]]
}


\BASICS OF CHARACTER\
DEFENSE: 17 \SET TO AC of MOB\
PROF_MOD  : 4

\LONGSWORD\
DAMAGE  : d10
STR_MOD: 5
MAGIC_WEP_HIT_MOD: 0
MAGIC_WEP_DMG_MOD: 0

output [strike] named "Non-Magic Weapon and Shield"
output [strike with advantage] named "Non-Magic Weapon and Shield (ADVANTAGE)"
output [strike with disadvantage] named "Non-Magic Weapon and Shield (DISADVANTAGE)"

function: ROLL:s difficulty D:n {
  result: (ROLL >= D) - (ROLL = 1)
}

function: ROLL:s difficulty D:n wp{
  result: (ROLL >= D) - (ROLL = 1)
}

\ Set dice pool and difficulty \
N: 6
D: 6

\ Roll with and without willpower \
output [Nd10 difficulty D] named "[N]d10, diff [D]"
output [Nd10 difficulty D+1 wp] named "[N]d10, diff [D]+1 + wp"

function: EXP dice of at least THRESHOLD reroll REROLL TOREROLL  {
  ROLL: EXPd6
  SUCCESSES: [count {THRESHOLD..6} in ROLL]
  if REROLL = 1 {
    UNDERTHRESHOLD: THRESHOLD - 1
    FAILURESCOUNT: EXP-SUCCESSES
    REROLLED: [lowest of FAILURESCOUNT and TOREROLL]
    result: SUCCESSES + [REROLLED dice of at least 5]
  }
  result: SUCCESSES
}

function: EXP dice of at least THRESHOLD {
  result: [EXP dice of at least THRESHOLD reroll 0 0]
}

function: EXP dice of at least THRESHOLD reroll NUMBER failures {
  result: [EXP dice of at least THRESHOLD reroll 1 NUMBER]
}

output [5 dice of at least 5 reroll 1 failures]
output [5 dice of at least 5 reroll 2 failures]

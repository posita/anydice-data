function: weary ROLL:n {
 if ROLL = 1 { result: 0}
 if ROLL = 2 { result: 0}
 if ROLL = 3 { result: 0}
 if ROLL > 3 { result: ROLL}
}


function: pierce ROLL:n with WEAPON:s {
 SUCCESSES: WEAPON = 6
if ROLL + SUCCESSES >= KEEN { result: 1}
result: 0
}

KEEN: 11
\output 1d12 + [ weary(1d6) ] +[weary(1d6)] \
loop DICE over {1..5}
output [ pierce (d12) with d6 ]

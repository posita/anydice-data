function: attack ATTACK:d {
  result: ATTACK
}

function: multiattack X:n Y:d {
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: [attack Y]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

STR: 5
DEX: 5

M_QUARTERSTAFF: 1d8 + DEX 
M_PUNCH: 1d4 + DEX

E_PUNCH: 1d10+ STR

output [multiattack 1 M_QUARTERSTAFF] + [multiattack 1 M_PUNCH] named "Monk quarterstaff plus unarmed strike at lv1."
output [multiattack 1 E_PUNCH] named "Empowered unarmed strike at lv1"

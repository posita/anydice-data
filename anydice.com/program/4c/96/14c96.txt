set "position order" to "lowest first"

function: roll ROLL:s vs HAZARD:s {
 SUCCESS: 1@ROLL  \ lowest roll \
 CRIT:    2@ROLL  \ second-lowest roll \
 if SUCCESS <= HAZARD {
   if CRIT <= HAZARD { result: CRIT*10000 + SUCCESS }
   result: SUCCESS
 }
 if CRIT <= HAZARD { result: CRIT*10000 }
 result: 0
}

loop H over {1..8} { output [roll 2d10 vs H] named "Hazard [H]" }

function: compound check A:n and B:n against C {
    if [highest of A and B] >= C { result: 2 }
    if (A + B) > C { result: 1 }
    result: 0
}

loop N over {4, 6, 8, 10, 12}{
    loop M over {4, 6, 8, 10, 12}{
        output [highest of dN and dM] named "Individual roll, d[N] and d[M]"
        output dN + dM named "Total d[N] and d[M]"
    }
}

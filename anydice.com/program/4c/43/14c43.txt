AC: d{19..22}
LEVEL: 10

function: attack DMG:d crit CRIT_DMG:d penalty PENALTY:n rider RIDER:n stat STAT:n superiority SUP:n roll ROLL:n against AC:n{
    PROFICIENCY: ((LEVEL-1)/4) + 2
    MODIFIER: (STAT-10)/2
    DMG_WITH_BONUS: DMG + MODIFIER + RIDER
	
    if ROLL = 1 {
        result: 0
    }
    if ROLL = 20 {
        result: DMG_WITH_BONUS + CRIT_DMG
    }
    if ROLL + MODIFIER + PROFICIENCY - PENALTY >= AC {
        result: DMG_WITH_BONUS + SUP
    }
    
    if ROLL + SUP + MODIFIER + PROFICIENCY - PENALTY >= AC {
        result: DMG_WITH_BONUS
    }
    result: 0
}
    


output 4d[attack 1d8 crit 1d8 penalty 0 rider 0 stat 20 superiority 0 roll 1@3d20 against AC] named "Samurai w/ Fighting Spirit & Elven Accuracy"

output 4d[attack 1d8 crit 1d8 + 2d10 penalty 0 rider 0 stat 20 superiority 1d10 roll 1d20 against AC] named "Battlemaster"

function: attack ATTACK:d {
  result: ATTACK
}

function: multiattack X:n Y:d {
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: [attack Y]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

STR: 5
DEX: 5

M_WEAPON: 1d4
M_ATTACK: M_WEAPON + DEX

E_WEAPON: 1d10
E_ATTACK: E_WEAPON + STR

output [multiattack 2 M_ATTACK] named "2 lv1 Monk Attacks"
output [multiattack 1 E_ATTACK] named "1 lv1 Empowered Attack"

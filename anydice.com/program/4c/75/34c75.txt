set "explode depth" to 10

RANGE: 20
ZNINE: d{0..9}

function: clamp N:n between MIN:n and MAX:n {
  if N > MAX { result: MAX }
  if N < MIN { result: MIN }
  result: N
}

\ Opposed roll \
output [clamp [explode ZNINE]+d12-[explode ZNINE]-d12-d{0..1} between -RANGE and RANGE] named "d12+z9! vs"

\ Dynamic action \
output [clamp [explode ZNINE]+d12-[explode ZNINE]-7 between -RANGE and RANGE] named "d12+z9! vs 7"

\ Static action \
output [clamp [explode ZNINE]+d12-11 between -RANGE and RANGE] named "d12+z9! vs 11"

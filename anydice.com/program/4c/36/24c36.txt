SIZE: 13

function: risk N:n of S:s ignore P:n failure {
  if [count 1 in S] > P { result: -10 }
  if [S contains SIZE] { result: SIZE + 10 }
  R: 0
  loop X over {1..N} {
    R: R + X@S
  }
  result: R
}

output 1dSIZE named "1d[SIZE]"
output [highest 1 of 2dSIZE] named "2d[SIZE] with talent to ignore 1 failure"

loop P over {2..5} {
  output [risk 1 of PdSIZE ignore 0 failure] named "[P]d[SIZE]"
}

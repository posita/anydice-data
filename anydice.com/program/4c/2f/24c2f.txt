SIZE: 10

function: risk N:n of S:s ignore P:n patzer {
  if [count 1 in S] > P { result: -10 }
  if [S contains SIZE] { result: 20 }
  R: 0
  loop X over {1..N} {
    R: R + X@S
  }
  result: R
}

output 1dSIZE named "1d[SIZE]"

loop P over {2..5} {
  output [risk 1 of PdSIZE ignore 1 patzer] named "[P]d[SIZE]"
}

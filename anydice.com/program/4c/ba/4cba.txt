{
  if TOHIT = 1
  {
    result: 0
  }
  else if TOHIT = 20
  {
    result: CRITICAL
  }
  else if AC > TOHIT + TOHITADD
  {
    result: 0
  }
  else
  {
    result: DAMAGE
  }
}

\**** Damage calculator ****\

function: boons BOON:n {
if BOON >= 1 {result: [highest 1 of BOONd6] }
if BOON <= 0 {result: -[highest 1 of -BOONd6] }

}

function: attackroll ROLL:n vs DEF:n {
if ROLL + ATTACKMOD >= 20 { result: CRITDAMAGE }
if ROLL + ATTACKMOD >= DEF { result: DAMAGE }
result: RELIABLE
}

set "explode depth" to 10

RANGE: 20
ZFIVE: d{0..5}

function: clamp N:n between MIN:n and MAX:n {
  if N > MAX { result: MAX }
  if N < MIN { result: MIN }
  result: N
}

\ Opposed roll \
output [clamp [explode ZFIVE]+d6-[explode ZFIVE]-d6-d{0..1} between -RANGE and RANGE] named "d6+z5! vs"

\ Dynamic action \
output [clamp [explode ZFIVE]+d6-[explode ZFIVE]-4 between -RANGE and RANGE] named "d6+z5! vs 6"

\ Static action \
output [clamp [explode ZFIVE]+d6-6 between -RANGE and RANGE] named "d6+z5! vs 6"

function:  ROLL:n vs DEFENSE:n ATTACK:n {
 if ROLL = 1 {result: MISS}
 if ROLL = 20 {result: CRITICALHIT}
 if ROLL + ATTACK>= DEFENSE {result: HIT}
 result: MISS
}

\ Damage functions \

function: damage for big ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: BIGCRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: BIGHITDAMAGE
 }
 result: MISSDAMAGE
}

\ Multi attacks \

function: bigmultiattack NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for big [d20 vs AC BIGATTACK]]
  }
  result: TOTAL
}

function: bigmultiattackadvantage NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for big [[highest 1 of 2d20] vs AC BIGATTACK]]
  }
  result: TOTAL
}

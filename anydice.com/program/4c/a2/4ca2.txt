function: ROLL:s difficulty D:n {
  result: (ROLL >= D) - (ROLL = 1)
}

function: simplified ROLL:s difficulty D:n {
  result: (ROLL >= D)
}

\ The number of dice and the base difficulty \
N: 5
D: 6

\ Output the old system and the simplified system \
output [Nd10 difficulty D] named "[N]d10, diff [D]"
output [simplified Nd10 difficulty D+1] named "simplified [N]d10, diff [D]+1"

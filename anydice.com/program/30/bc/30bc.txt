function: repeat lowest NUMBER:n of DICE:d REPEAT:n times
{
   if NUMBER >= #DICE { result: { 1..#DICE:REPEAT}@DICE }
   result: {1..NUMBER:REPEAT, 1..#DICE}@DICE
}

function: repeat highest NUMBER:n of DICE:d REPEAT:n times
{
   if NUMBER >= #DICE { result: { 1..#DICE:REPEAT}@DICE }
   result: {1..#DICE,(#DICE-NUMBER+1)..#DICE:REPEAT}@DICE
}

output 1@4d20
output 2@4d20
output 3@4d20
output 4@4d20

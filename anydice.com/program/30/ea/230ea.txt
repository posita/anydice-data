function: roll A:n B:n C:n {
 D: 0
 if A = 10 { \if the tens die comes up 00\
  if B = 10 {D: 100} \because 00 and 0 is 100\
  D: B \otherwise it's just what the units die says\
 }
 else{ 
  if B = 10 {
   D: A*10
  }
  D: A*10+B
 }
 if D < (C+1) {result: C}
 result: A+B
}

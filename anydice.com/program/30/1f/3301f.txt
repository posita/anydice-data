
\** Attack Roll Function **\
function: attackrollevade ROLL:n vs EVADE:n {
 if ROLL = 1 & EVADE != 1 { result: CRITDAMAGE } 
 if ROLL <= GREAT & EVADE > OPGREAT { result: CRITDAMAGE } 
 if ROLL <= GREAT & EVADE <= OPGREAT { result: 0 } 
 if ROLL <= GOOD & EVADE > OPGOOD { result: DAMAGE } 
 if ROLL <= GOOD & EVADE <= OPGOOD { result: 0 } 
 if ROLL <= NORMAL & EVADE > OPNORMAL { result: DAMAGE } 
 if ROLL <= NORMAL & EVADE <= OPNORMAL & EVADE != 100 { result: 0 } 
 if ROLL = 100 {result: 0}
 result: 0
}

function: attackrollcounter ROLL:n vs EVADE:n {
 if ROLL = 1 & EVADE != 1 { result: CRITDAMAGE } 
 if ROLL <= GREAT & EVADE > 1 { result: CRITDAMAGE } 
 if ROLL <= GREAT & EVADE = 1 { result: -10 } 
 if ROLL <= GOOD & EVADE > OPGREAT { result: DAMAGE } 
 if ROLL <= GOOD & EVADE <= OPGREAT { result: -10 } 
 if ROLL <= NORMAL & EVADE > OPGOOD { result: DAMAGE } 
 if ROLL <= NORMAL & EVADE <= OPGOOD & EVADE != 100 { result: -10 } 
 if ROLL > NORMAL & EVADE > OPNORMAL {result: 0}
 if ROLL = 100 {result: -10}
 result: -10
}

function: attackrollsave ROLL:n vs EVADE:n { \** Ataque mágico com crit, crita 1.5 se der extremo ou alvo falhar. Conjurador ganha em tie**\
if ROLL = 1 & EVADE != 1 { result: SPELLCRITDAMAGE } 
 if ROLL <= NORMAL & EVADE < OPNORMAL {result: SPELLCRITDAMAGE}
 if ROLL <= GREAT & EVADE > 1 { result: SPELLCRITDAMAGE } 
 if ROLL <= GREAT & EVADE = 1 { result: (DAMAGE/2) } 
 if ROLL <= GOOD & EVADE > OPGREAT { result: DAMAGE } 
 if ROLL <= GOOD & EVADE <= OPGREAT { result: (DAMAGE/2) } 
 if ROLL <= NORMAL & EVADE > OPGOOD { result: DAMAGE } 
 if ROLL <= NORMAL & EVADE <= OPGOOD & EVADE != 100 { result: (DAMAGE/2) } 
 if ROLL > NORMAL & EVADE > OPNORMAL {result: (DAMAGE/2)}
 if ROLL = 100 {result: (DAMAGE/2)}
 result: (DAMAGE/2)
}

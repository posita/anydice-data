TARGET_AC: {18}

DEX: 4
PROF: 3
DUELING: 2
WEAPON_DIE_SIZE: 6
WEAPON_DIE_NUMBER: 1
MOD: 1
CRIT: 2

CANTRIP: 1
SHADOWBLADE_LEVEL: 3
EXTRA_ATTACK: 2
CHA: 5
DEVOTION: 1
VENGEANCE: 1 - DEVOTION
SPIRIT_GUARD: 3d8 * 2 / 3

WEAPON_ATTACK: {DEX + PROF + MOD + {CHA * DEVOTION}}
SHADOWBLADE_ATTACK: {DEX + PROF + {CHA * DEVOTION}}


WEAPON_ROLL: {WEAPON_DIE_NUMBER}d{WEAPON_DIE_SIZE}
WEAPON_ROLL_CRIT: {CRIT * WEAPON_DIE_NUMBER}d{WEAPON_DIE_SIZE}
SHADOWBLADE_ROLL: {SHADOWBLADE_LEVEL}d8
SHADOWBLADE_ROLL_CRIT: {CRIT * SHADOWBLADE_LEVEL}d8

CANTRIP_WEAPON_ROLL: {WEAPON_DIE_NUMBER}d{WEAPON_DIE_SIZE} + {CANTRIP}d8
CANTRIP_WEAPON_ROLL_CRIT: {CRIT * WEAPON_DIE_NUMBER}d{WEAPON_DIE_SIZE} + {CRIT * CANTRIP}d8
CANTRIP_SHADOWBLADE_ROLL: {SHADOWBLADE_LEVEL}d8 + {CANTRIP}d8
CANTRIP_SHADOWBLADE_ROLL_CRIT: {CRIT * SHADOWBLADE_LEVEL}d8 + {CRIT * CANTRIP}d8


WEAPON_DAMAGE: WEAPON_ROLL + DEX + MOD + DUELING
WEAPON_DAMAGE_CRIT: WEAPON_ROLL_CRIT + DEX + MOD + DUELING
SHADOWBLADE_DAMAGE: SHADOWBLADE_ROLL + DEX + DUELING
SHADOWBLADE_DAMAGE_CRIT: SHADOWBLADE_ROLL_CRIT + DEX + DUELING

CANTRIP_WEAPON_DAMAGE: CANTRIP_WEAPON_ROLL + DEX + MOD + DUELING
CANTRIP_WEAPON_DAMAGE_CRIT: CANTRIP_WEAPON_ROLL_CRIT + DEX + MOD + DUELING
CANTRIP_SHADOWBLADE_DAMAGE: CANTRIP_SHADOWBLADE_ROLL + DEX + DUELING + {CANTRIP}d8
CANTRIP_SHADOWBLADE_DAMAGE_CRIT: CANTRIP_SHADOWBLADE_ROLL_CRIT + DEX + DUELING + {CRIT * CANTRIP}d8


MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{TARGET_AC}

ATTACK: WEAPON_ATTACK
HITDAMAGE: CANTRIP_WEAPON_DAMAGE + SPIRIT_GUARD
CRITICALDAMAGE: CANTRIP_WEAPON_DAMAGE_CRIT + SPIRIT_GUARD
MISSDAMAGE: SPIRIT_GUARD
output [attack] named "Cantrip Weapon"

ATTACK: SHADOWBLADE_ATTACK
HITDAMAGE: CANTRIP_SHADOWBLADE_DAMAGE
CRITICALDAMAGE: CANTRIP_SHADOWBLADE_DAMAGE_CRIT
MISSDAMAGE: 0
output [attack] named "Cantrip Shadowblade"

ATTACK: WEAPON_ATTACK

CRITICALDAMAGE: WEAPON_DAMAGE_CRIT * 2  + SPIRIT_GUARD
MISSDAMAGE: SPIRIT_GUARD
output [attack] named "Extra Attack Weapon"

HITDAMAGE: SHADOWBLADE_DAMAGE * 2
CRITICALDAMAGE: SHADOWBLADE_DAMAGE_CRIT * 2
MISSDAMAGE: 0
output [attack] named "Extra Attack Shadowblade"

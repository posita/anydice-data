function: good D:n TN:n DB:n {
    if D >= DB { result: 2 }
    if D >= TN { result: 1 }
    result: 0
}

function: neutral D:n TN:n {
    if D >= TN { result: 1 }
    result: 0
}

function: evil D:n TN:n {
    if D = 1 { result: -1 }
    result: [neutral D TN]
}

function: both D:n TN:n DB:n {
    if D = 1 { result: -1 }
    if D >= DB { result: 2 }
    if D >= TN { result: 1 }
    result: [good D TN DB]
}


function: nn N:n{ \nn = not negative \
    if N < 0 {result : 0} 
    result: N
}

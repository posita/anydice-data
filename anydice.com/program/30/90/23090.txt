function: rolledhighdicecrit A:s {
  result: 1@A >= 9 & 2@A >= 9
}
function: rolleddoubledicecrit A:s {
  result: (1@A = 2@A | 2@A = 3@A) & 2@A >= 6
}

output [rolledhighdicecrit 2d10] named "roll 9+ twice"
output [rolleddoubledicecrit 2d10] named "roll doubles of 6+"
output 2d10 >= 18 named "roll 18+"

output [rolledhighdicecrit 3d10] named "adv roll 9+ twice"
output [rolleddoubledicecrit 3d10] named "adv roll doubles of 6+"
output [highest 2 of 3d10] >= 18 named "adv roll 18+"

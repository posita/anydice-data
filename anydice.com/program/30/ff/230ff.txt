function: exp DIE:d {
 MAX: [maximum of DIE]
 MIN: -MAX
 result: [exphelp DIE]
}

function: exphelp DIE:n {
 if DIE = MAX { result: DIE + [maxhelp 1d{-1,0,1}]}
 if DIE = MIN { result: DIE + [minhelp 1d{-1,0,1}]}
 result: DIE
}

function: minhelp M:n {
 if M = -1 { result: M + [minhelp 1d{-1,0,1}] }
 result: M
}

function: maxhelp N:n {
 if N = 1 { result: N + [maxhelp 1d{-1,0,1}] }
 result: N
}

set "maximum function depth" to 10

output [exp 3d{-1,0,1}]

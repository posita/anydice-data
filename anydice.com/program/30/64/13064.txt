HITS: 0
DICELEFT: 5

function: roll A:s for B:s{ 
   COUNT: [count B in A]
   HASHIT: COUNT != 0
   if HASHIT {
      HITS: HITS + COUNT
      HASHITTHREE: [check for three hits] = 0
      if HASHITTHREE {         
         DICELEFT: DICELEFT - HITS
         HITS: HITS + [roll DICELEFTd6 for B]
         result: HITS
      }
      else {
         result: HITS
      }
   }
   else {      
      result: 0
   }
}

function: check for three hits {
   if HITS = 3 {
      result: 1
   }
   else {
      result: 0
   }
}

output [roll DICELEFTd6 for 1]

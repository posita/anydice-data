function: escalatingexplode DIE:d {
 FACES: [maximum of DIE]
 result: [escalatingexplode DIE FACES]
}

function: escalatingexplode N:n FACES:n {
 if N = FACES & FACES < 12 {
  result: N + [escalatingexplode d{1..FACES+2}]
 }
 result: N
}

output [escalatingexplode d4] named "escalating explode starting with d4"

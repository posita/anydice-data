function: ROLL:s count fours and above{
 result: (ROLL >= 4) 
}

loop N over {1..15} {
output [Nd6 count fours and above] named "[N]d"

}


function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d6] }
 result: N >= 4
}

NWOD: [nwod 6 again d6]

loop N over {1..10} {
 output [lowest of 10 and NdNWOD] named "[N]d"
}

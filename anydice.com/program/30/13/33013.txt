\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ABLSCORE + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\Static variables\
ENEMYAC: {15..22}  \ range of target AC\
PROFICIENCY: 4
ABLSCORE: 5
ADV: 2 \Advantage, 1 = normal, 2=adv, 3=elven\

\ Pistol +2 \
DMGBONUS: 2
ACCBONUS: 2
CRITROLL: 20
DAMAGE: 1d10 + ABLSCORE  + DMGBONUS
CRITDAMAGE: 2d10 + ABLSCORE  + DMGBONUS
loop AC over ENEMYAC {
 output [attackroll [highest 1 of ADVd20] vs AC] named "Pistol +2 vs. AC [AC]|"
}
output [attackroll [highest 1 of ADVd20] vs dENEMYAC] named "Pistol +2 MEAN|"


\ Pistol +2 SS\
DMGBONUS: 2 + 10
ACCBONUS: 2 - 5
CRITROLL: 20
DAMAGE: 1d10 + ABLSCORE  + DMGBONUS
CRITDAMAGE: 2d10 + ABLSCORE  + DMGBONUS
loop AC over ENEMYAC {
 output [attackroll [highest 1 of ADVd20] vs AC] named "Pistol +2 SS vs. AC [AC]|"
}
output [attackroll [highest 1 of ADVd20] vs dENEMYAC] named "Pistol +2 SS MEAN|"

\ DW Pistol \
DMGBONUS: 1
ACCBONUS: 1
CRITROLL: 20
DAMAGE: 1d10 + ABLSCORE  + DMGBONUS + 1d6          \+1d6 on normal hit\
CRITDAMAGE: 2d10 + ABLSCORE  + DMGBONUS + +2d6 + 5 \ +2d6+5 on a crit\
loop AC over ENEMYAC {
 output [attackroll [highest 1 of ADVd20] vs AC] named "DW Pistol vs. AC [AC]|"
}
output [attackroll [highest 1 of ADVd20] vs dENEMYAC] named "DW Pistol MEAN|"

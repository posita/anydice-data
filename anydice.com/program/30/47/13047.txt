\**** Jesse's Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL >= HARDHIT { result: HARDDAMAGE }
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
result: 0
}

\looped against enemy AC from d{14..18}\

\Unused but called\
PROFICIENCY: 0
ATTACKMOD: 0

\My Variables\
ENEMYAC: 18
ACCBONUS: 5
DMGMOD: 10
DMGMIT: -1d4  
DMGBONUS: 1d4 
CRITBONUS: 1d4

CRITROLL: 20
HARDHIT: 16

ATTACKTIMES: 3

DAMAGE: DMGMOD + DMGMIT
HARDDAMAGE: DMGMOD + DMGBONUS
CRITDAMAGE: 17

output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Jesse Orc +/- 1d4 (16) vs AC: 18"

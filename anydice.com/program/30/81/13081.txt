MISS: 0
HIT: 1
CRITICALHIT: 2
GREATSWORDATTACK: 4 + 3 + 1
GLAIVEATTACK: 3 + 3 + 1
HITDAMAGE: 2d6 + 4
CRITICALDAMAGE: 4d6 + 4
MISSDAMAGE: 0
GLAIVEDAMAGEMAIN: 1d10 + 4
GLAIVECRITICALDAMAGEMAIN: 2d10 + 4
GLAIVEDAMAGEPOMMEL: 1d4 + 4 
GLAIVECRITICALDAMAGEPOMMEL: 2d4 + 4
AC: 18

function:  ROLL:n vs DEFENSE:n ATTACK:n {
 if ROLL = 1 {result: MISS}
 if ROLL = 20 {result: CRITICALHIT}
 if ROLL + ATTACK>= DEFENSE {result: HIT}
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: glaive damage for ATTACKRESULTMAIN:n and ATTACKRESULTPOMELL:n {
  TOTAL: 0
  if ATTACKRESULTMAIN = CRITICALHIT {
    TOTAL: TOTAL + GLAIVECRITICALDAMAGEMAIN
  }
  if ATTACKRESULTMAIN = HIT {
    TOTAL: TOTAL + GLAIVEDAMAGEMAIN
  }
  else {TOTAL: TOTAL + MISSDAMAGE}
  if ATTACKRESULTPOMELL = CRITICALHIT {
    TOTAL: TOTAL + GLAIVECRITICALDAMAGEPOMMEL
  }
  if ATTACKRESULTPOMELL = HIT {
    TOTAL: TOTAL + GLAIVEDAMAGEPOMMEL
  }
  else {TOTAL: TOTAL + MISSDAMAGE}
  result: TOTAL
}


function: multiattack NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for [d20 vs AC GREATSWORDATTACK]]
  }
  result: TOTAL
}

output [multiattack 2] named "Greatsword"

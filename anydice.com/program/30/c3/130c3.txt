\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 10-20\
ENEMYAC: d{10..20}


ATTACKMOD: 2
DMGBONUS: 3
PROFICIENCY: 2 
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 10
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "10 skeleton archers"

ATTACKMOD: 4
DMGBONUS: 10
PROFICIENCY: 3 
ACCBONUS: -3
CRITROLL: 20
ATTACKTIMES: 3
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of 3d20] vs ENEMYAC] named "Darkness/Devil's Sight, Elven Accuracy, archery fighting style, crossbow expert, sharpshooter"

\ 0 failure, 1 partial success, 2 success, 3 critical. \

SIDED: 6
PARTIALBOTTOM: 4

function: blades ROLL:s {
 if 1@ROLL = SIDED {
  result: 2 + (2@ROLL = SIDED)
 }
 result: 1@ROLL >= PARTIALBOTTOM
}

output [blades 2@2dSIDED] named "0d"
loop P over {1..SIDED} {
 output [blades PdSIDED] named "[P]d"
}

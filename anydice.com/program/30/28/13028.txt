function: blades DICE:s {
   HIGHEST: 1@DICE

   if #DICE > 1 {
      NEXT: 2@DICE
      if HIGHEST = NEXT & NEXT = 6 {
         result: 3
      }
   }
   
   if HIGHEST = 6 { result: 2 }
   if HIGHEST > 3 { result: 1 }
   else { result: 0 }
}

loop N over {1..6} {
   output [blades Nd6] named "blades [N]d"

   M: N - 3
   output 2d6 >= (7 - (M)) named "pbta +[M]"
}

function: pickhelper SEQ:s {
  CRITS: 0
  CRITS: CRITS + (SEQ = 20)
  CRITS: CRITS - (SEQ = 1)
  result: CRITS
}

function: critmiddle DICE:s {
  CRITTING: [pickhelper DICE]
  if CRITTING > 0 {result: 1@DICE }
  if CRITTING < 0 {result: 3@DICE}
  result: 2@DICE}

function: pick NTH:n of SEQ:s {
  NTH: #SEQ - NTH + 1
  result: NTH@SEQ}

function: splodehelper SEQ:s target TAR:n{
  CRITS: 0
  CRITS: CRITS + (SEQ < TAR)
  \CRITS: CRITS - (SEQ = 1)\
  result: CRITS
}

function: critsplode DICE:s middle MID:n adder ADD:n crit on TAR:n{
  CRITTING: [splodehelper DICE target TAR]
  if CRITTING > 0 {result: MID@DICE + ADD@DICE}
  \if CRITTING < 0 {result: 2@DICE - 3@DICE}\
  result: MID@DICE
}

BASEDIE: d6

set "maximum function depth" to 5

function: target TARGET:n reroll REROLL:n VALUE:n {
  if VALUE = REROLL {
    result: 1 + [target TARGET reroll REROLL BASEDIE]
  }
  if VALUE  > TARGET {
    result: 1
  }
  result: 0
}

function: target TARGET:n VALUE:n {
  if VALUE  > TARGET {
    result: 1
  }
  result: 0
}

DIEEXPLODE: [target 4 reroll 6 BASEDIE]
DIE: [target 4 BASEDIE]

loop N over {1..8}{
  output NdDIE named "[N]"
  output NdDIEEXPLODE named "[N] Explode"
}

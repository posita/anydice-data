function: Success S:n Favored F:n Illfavored I:n Miserable M:n Weary W:n {
if M>0 {
 FD:{-100,1..10,100}
} else {
 FD:{0..11,100}
}
if W>0 {
 SD:{0,0,0,4..6}
} else {
 SD:{1..6}
}
if (F>0 & I>0) | (F<1 & I<1) { 
 result: 1dFD + SdSD
} else if F>0 {
 result: 1@2dFD + SdSD
} else {
 result: 2@2dFD + SdSD
}
}

loop S over {0..6} {
F:0
I:0
M:1
W:0

output [Success S Favored F Illfavored I Miserable M Weary W] named "S[S] F[F] I[I] M[M] W[W]"
}

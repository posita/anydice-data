function: attack ROLL:n CRIT:n AC:n DAM:d CRITDAM:d {
 if ROLL >= CRIT { result: [maximum of DAM] + CRITDAM }
 if ROLL = 1 { result: 0 }
 if ROLL >= AC { result: DAM }
 result: 0
}

loop AC over {1,6,11,16,21} {
  loop BRUTAL over {0,1,2,3} {
    output [attack 1@2d20 20 AC 1d12+1d4 (1+BRUTAL)d12] named "Reckless Brute against AC [AC], brutal [BRUTAL]"
    output [attack 1@2d20 19 AC 1d12 (1+BRUTAL)d12] named "Reckless Champion against AC [AC], brutal [BRUTAL]"
  }
}

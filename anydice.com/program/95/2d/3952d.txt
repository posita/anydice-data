\DAMAGE MODIFIERS AND COMBINATIONS\

DMGMOD: 4
PROFBONUS: 3
ATKMOD: DMGMOD + PROFBONUS
MISSDMG: 0 \This allows support for the "Graze" Weapon mastery property. Simply change 0 to the appropriate modifier\
WEAPDMG_A: 1d6 
WEAPDMG_B: 1d6 
BASICDMG_A: WEAPDMG_A + DMGMOD
BASICDMG_B: WEAPDMG_B + DMGMOD
NICKDMG: WEAPDMG_A + WEAPDMG_B + DMGMOD 
TWFNICK: NICKDMG + DMGMOD 
CRITBASIC_A: 2dWEAPDMG_A + DMGMOD 
CRITBASIC_B: 2dWEAPDMG_B + DMGMOD  

\Bonus Damage Modifiers [Hunter's Mark, Smites, etc.]\
BONUSDMG_A: 1d6
BONUSDMG_B: 0

MISS:0  NO:0
HIT:1   YES:1
CRIT:20 

ADVANTAGE: NO

\MAKE SURE YOU SET YOUR AC TARGETS CORRECTLY!\
 
AC: d{15} 


function: attack ROLL:n vs AC:n {
 if ROLL + ATKMOD >= AC { result: HIT}
 if ROLL + ATKMOD < AC { result: MISS}
}

 if ADVANTAGE = YES {
  output [highest of [attack d20 vs AC] and [attack d20 vs AC]] named "Hit chance with advantage vs AC"
}
 else if ADVANTAGE = NO {
 output [attack d20 vs AC] named "Hit chance vs AC"
}


\This damage function supports output for Abilities that are NOT limited to the number of uses per round [e.g. Hunter's Mark]. It will apply the bonus damage with every hit\

function: damage for ATTACKRESULT:s {
D: 0
  loop A over ATTACKRESULT {
   if A = HIT 
     { D: BASICDMG_A } \ADJUST AS NEEDED\
   else if A = MISS 
     { D: D + MISSDMG }
   if !A = MISS {D: D + [bonusdmg for ATTACKRESULT]}
}
result: D
}
function: bonusdmg for ATTACKRESULT:d {
  result: BONUSDMG_A + BONUSDMG_B
}
if ADVANTAGE = NO {
 function: attack {
  result: [damage for [attack d20 vs AC]]
 }
  function: N:n attack {
 result: [damage for Nd[attack d20 vs AC]]
 }
}
if ADVANTAGE = YES {
 function: attack {
  result: [damage for [highest of [attack d20 vs AC] and [attack d20 vs AC]]]
 }
  function: N:n attack {
 result: [damage for Nd[highest of [attack d20 vs AC] and [attack d20 vs AC]]]
 }
}

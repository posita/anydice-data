W: 2d12
X: 0
Y: 0
Z: 0

WC: W = [maximum of W]
XC: X = [maximum of X]
YC: Y = [maximum of Y]
ZC: Z = [maximum of Z]
 
T: [count 1 in X]+[count 1 in Y]+[count 1 in Z]+[count 1 in W]
M: T < 2

function: highest N:n of A:s B:s C:s D:s{
	result: {1..N}@[sort {A, B, C, D}]
}

CR: [highest 2 of W X Y Z]
CRT: WC*XC*YC*ZC*1d6+[maximum of CR]


output CR*M 
output CRT*M
output T

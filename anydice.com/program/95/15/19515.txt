SURVIVE: 1 \Constant; you always have a 100% chance to survive before your first roll\
FORT: 0    \Variable; your character's Fortitude value\
DC: 10     \Variable; the DC of the death roll\
ROUND: 10  \Variable; number of rounds since dropping to 0 health\

loop X over {1:ROUND}{
  ROLL: 3d6+FORT >= DC
  SURVIVE: ROLL & SURVIVE
  output SURVIVE
}

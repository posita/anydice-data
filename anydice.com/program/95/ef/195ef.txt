AC:15
ATTACKS:1
function: calc ROLL:n ATTACKBONUS:n DMGBONUS:n {
 if ROLL = 20 { result:2d8+DMGBONUS}
 if ROLL >=(AC-ATTACKBONUS) {result:1d8+DMGBONUS}
 result: 0
}
function: atc ATTACKS ATTACKBONUS:n DMGBONUS:n{
 RES:0
 loop N over {1..ATTACKS}{
  ROLL: [highest 1 of 3d20]
  RES: RES+[calc ROLL ATTACKBONUS DMGBONUS]
 }
 result:RES
}
output [atc 4 0 0]

function: degreeofsucess ROLL:s {
 if 1@ROLL = 6 {
  result: 2
 }
 result: 1@ROLL >= 4
}

function: trophy LIGHT:s DARK:s {

 result: #{LIGHT,DARK}
}

loop L over {0..2} {
 loop D over {0..4} {
  if D = 0 {
   output [degreeofsucess Ld6] named "[L] light 0 dark"
  }
  else {
   output [trophy Ld6 Dd6] named "[L] light [D] dark"
  }
 }
}

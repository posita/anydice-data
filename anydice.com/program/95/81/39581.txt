
function: action roll A:n M:n C:s {
  if A + M > 1@C { result : 2 }
  if A + M > 2@C { result : 1 }
  result : 0
}

function: loyalist A:n B:n C:n {
  if C = 10 { result : 1 @ [sort {A,B,C}] }
  result : 2 @ [sort {A,B,C}]
}

loop P over {0..9} {
  output [action roll 1d6 P 2d10] named "Action +[P]"
}

W: 2d6
X: 0
Y: 0
Z: 0

WC: W = [maximum of W]
XC: X = [maximum of X]
YC: Y = [maximum of Y]
ZC: Z = [maximum of Z]

function: highest N:n of A:s B:s C:s D:s{
    result: {1..N}@[sort {A, B, C, D}]
}

CR: [highest 2 of W X Y Z]
CRT: WC*XC*YC*ZC*CR+[maximum of CR]

RS: [highest 2 of W X Y Z]

output RS
output CRT

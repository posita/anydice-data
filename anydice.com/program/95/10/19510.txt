SURVIVE: 1 \Do not change; you always have a 100% chance to survive before your first roll\
FORT: 0    \Your character's Fortitude value\
DC: 10     \The DC of the death roll\
ROUND: 10  \Number of rounds since dropping to 0 health\

loop X over {1:ROUND}{
  ROLL: 3d6+FORT >= DC
  SURVIVE: ROLL & SURVIVE
  output SURVIVE
}

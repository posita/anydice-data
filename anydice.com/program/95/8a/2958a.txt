function: kill HP DMG SAVE DC {
  CRITDMG: 2*DMG
  CRITFAIL: d20 + SAVE <= (DC - 10)
  FAIL: d20 + SAVE < DC & d20 + SAVE > (DC - 10)
  result: (FAIL & (DMG >= HP) + CRITFAIL & (CRITDMG >= HP))
}

DMG: 6d6+3
output [kill 29 DMG 3 21] named "armorL2"
output [kill 30 DMG 9 21] named "banditL2"

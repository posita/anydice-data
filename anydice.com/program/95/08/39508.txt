function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}
output 1d6 named "regular"
output [reroll {1,2} on d6 to d6] named "GWF'14"
output [reroll {1,2} on d6 to 3] named "GWF'24"
output 1d6+2 named "dueling"
output 1@2d6 named "'adv'"

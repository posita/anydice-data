\ point buy cost of a single ability score using extended D&D 5e cost table \
\ from https://chicken-dinner.com/5e/5e-point-buy.html via https://rpg.stackexchange.com/a/185268 \
function: cost of SCORE:n {
  \if SCORE < 3 { result: -1000 } \   \ safety check 1: rolling below 3 should be impossible \
  \if SCORE > 18 { result: 1000000 }\ \ safety check 2: rolling above 18 should also be impossible \
  result: (SCORE - 2) @ { -8, -6, -4, -2, -1, 0, 1, 2, 3, 4, 5, 7, 9, 11, 13, 15 }
}

set "maximum function depth" to 20

function: if CONDITION:n then A else B {
  if CONDITION { result: A } else { result: B }
}  

function: reroll X:n outside MIN:n and MAX:n {
  result: [if (X < MIN | X > MAX) then [reroll X outside MIN and MAX] else X]
}

\ optimized alternative to 1@(NdD) from https://rpg.stackexchange.com/a/176183 \
function: highest of N:n x D:d {
  if N = 0 { result: 0 }
  if N = 1 { result: D }
  RES: 1@2d[highest of N/2 x D]
  if N - (N/2)*2 = 1 { RES: [highest of D and RES] }
  result: RES
}

\ point cost distributions for 6 stat arrays rolled with different methods \
STANDARD: 6d[cost of [highest 3 of 4d6]]  \ standard (4d6 drop lowest) \
CLASSIC: 6d[cost of 3d6] \ classic (3d6) \

output [reroll STANDARD outside 0 and 60] named "5e point cost of standard 4d6 drop lowest array"
output [reroll [highest of 8 x CLASSIC] outside 0 and 60] named "highest 5e point cost of 8 classic 3d6 arrays"
output [reroll STANDARD outside 27 and 60] named "standard but reroll when lower than 27"
output [reroll CLASSIC outside 27 and 60] named "classic but reroll when lower than 27"
output [highest of 2 x [reroll STANDARD outside 0 and 60]] named "standard 4d6k3 w/ advantage"

\ User Data \
\ Change these to suit your scenario \
\ To hit and damage data \
STRENGTHBONUS: 3
ANYOTHERTOHITBONUS: 0
ANYOTHERDAMAGEBONUS: 0
WEAPONDAMAGE: 1d12
CRITICALDAMAGE: 12

\Barbarian Details \
BARBARIANLEVEL: 1
RAGING: 1 \ If you are raging, set this to 1, if not set it to 0 \

\Fighter Details \
FIGHTERLEVEL: 3

\ Advantage/Disadvantage \
ADVANTAGE: 0 \ 0 is Niether, 1 is Advantage, -1 is Disadvantage \
\ End of user settings \

\ Functions \
\ Work out the attack matrix \
function: attack with CRITRANGE and ADVANTAGE {
FAIL: [lowest of 19 and [highest of 0 and TARGET-1]]
SUCCESS: [lowest of 20-CRITRANGE and [highest of 0 and 21-TARGET]]
ATTACKRESULT: d{0, 0:[highest of 0 and FAIL-1], 1:[highest of 0 and SUCCESS-CRITRANGE], 2:CRITRANGE}
if ADVANTAGE > 0 {
result: [highest 1 of 2dATTACKRESULT]
}
if ADVANTAGE < 0 {
result: [lowest 1 of 2dATTACKRESULT]
}
result: ATTACKRESULT
}

\ Work out the damage \
function: damage with ATTACK:s from NORMALDMG:d and CRITDMG:d {
if ATTACK = 1 {
 result: NORMALDMG
}
if ATTACK = 2 {
 result: CRITDMG
}
result: 0
}
\ end functions \

\ Fixed Data - Don't change these \
PROFICIENCY: {2:4,3:3,4:4,5:4,6:4}
CHAMPIONIMPROVEDCRITRANGE: {0:2,1:12,2:6}
BRUTEEXTRADAMAGEDIE: {0:2,4:7,6:6,8:4,10:1}
BRUTEDEVASTATINGCRITICAL: {0:14,15,16,17,18,19,20}
BARBARIANRAGEDAMAGE: {2:8,3:7,4:5}
BARBARIANEXTRAATTACK: {0:4,1:16}
BARBARIANBRUTALCRITICAL: {0:8,1:4,2:4,3:4}
\ End Fixed Data \

\Begin calculations \

loop TARGETAC over {13..18} {
\ Work out the number needed on the d20 to hit \
TARGET: TARGETAC - STRENGTHBONUS - (BARBARIANLEVEL + FIGHTERLEVEL)@PROFICIENCY - ANYOTHERTOHITBONUS
TARGET: [highest of 2 and [lowest of 20 and TARGET]]

\ Work out the to hit and critical possibilities for the Champion \
CHAMPIONCRITRANGE: 1 + FIGHTERLEVEL@CHAMPIONIMPROVEDCRITRANGE
CHAMPIONCRITRANGE: [lowest of CHAMPIONCRITRANGE and (21 - TARGET)]
CHAMPIONATTACK: [attack with CHAMPIONCRITRANGE and ADVANTAGE]
CHAMPIONNORMALDAMAGE: WEAPONDAMAGE + STRENGTHBONUS + ANYOTHERDAMAGEBONUS + RAGING*BARBARIANLEVEL@BARBARIANRAGEDAMAGE
CHAMPIONCRITICALDAMAGE: WEAPONDAMAGE + STRENGTHBONUS + ANYOTHERDAMAGEBONUS + RAGING*BARBARIANLEVEL@BARBARIANRAGEDAMAGE + CRITICALDAMAGE + BARBARIANLEVEL@BARBARIANBRUTALCRITICAL

\ Work out the to hit and critical possibilities for the Brute \
BRUTEATTACK: [attack with 1 and ADVANTAGE]
BRUTENORMALDAMAGE: WEAPONDAMAGE + STRENGTHBONUS + ANYOTHERDAMAGEBONUS + RAGING*BARBARIANLEVEL@BARBARIANRAGEDAMAGE  + 1d(FIGHTERLEVEL@BRUTEEXTRADAMAGEDIE)
BRUTECRITICALDAMAGE: WEAPONDAMAGE + STRENGTHBONUS + ANYOTHERDAMAGEBONUS + RAGING*BARBARIANLEVEL@BARBARIANRAGEDAMAGE + CRITICALDAMAGE + 2d(FIGHTERLEVEL@BRUTEEXTRADAMAGEDIE) + BARBARIANLEVEL@BARBARIANBRUTALCRITICAL

output [damage with CHAMPIONATTACK from CHAMPIONNORMALDAMAGE and CHAMPIONCRITICALDAMAGE] named "Champion [FIGHTERLEVEL]/Barbarian [BARBARIANLEVEL] v AC[TARGETAC]"
output [damage with BRUTEATTACK from BRUTENORMALDAMAGE and BRUTECRITICALDAMAGE] named "Brute [FIGHTERLEVEL]/Barbarian [BARBARIANLEVEL] v AC[TARGETAC]"
}

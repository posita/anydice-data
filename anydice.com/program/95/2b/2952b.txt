function: partial advantage diff based with A:n and B:n {
  H: [highest of A and B]
  L: [lowest of A and B]
  if H-L > 6 { result: H } else { result: L}
}

function: if CONDITION:n then A else B {
  if CONDITION { result: A } else { result: B }
}

function: partial advantage half time with A:n and B:n {
  result: [if 1d2 = 1 then [highest of A and B] else A]
}

function: N:n is even { result: N = 2*(N/2) }

function: partial advantage signs match with A:n and B:n {
  if ([A is even] & [B is even]) | (![A is even] & ![B is even]) {
    RES: [highest of A and B]
  } else {
    RES: A
  }
  result: RES
}

function: partial advantage signs dont match with A:n and B:n {
  if ([A is even] & [B is even]) | (![A is even] & ![B is even]) {
    RES: A
  } else {
    RES: [highest of A and B]
  }
  result: RES
}

function: highest of ROLL:s prefer even {
  if [2@ROLL is even] & ![1@ROLL is even] { result: 2@ROLL }
  result: 1@ROLL
}

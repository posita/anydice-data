function: hit ac A:n using dice B:n with hit bonus C:n and damage dice D:d and damage bonus E:n {
if B = 20{
result: 2dD + E
} if B+C >= A{
result: D+E
} else {
result:0
}
}
loop N over {7..22}{
output [hit ac N using dice d20 with hit bonus 6 and damage dice 2d6 and damage bonus 3+1] named "Normal AC [N]"
output [hit ac N using dice d20 with hit bonus 1 and damage dice d6 and damage bonus 14] named "Sharpshooter AC [N]"
}

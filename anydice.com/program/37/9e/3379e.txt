function: attack ROLL:s against DEFENSE:s {
  ATTACK: [count 1 in ROLL]
  BLOCK: [count -1 in DEFENSE]
  result: ATTACK - BLOCK
}

HERO_DIE: {1,1,1,0,-1,-1}
MONSTER_DIE: {1,1,-1,0,0,0}
HERO_DICE: 2
MONSTER_DICE: 1

output [attack HERO_DICE d HERO_DIE against MONSTER_DICE d MONSTER_DIE] named "Damage to Monster"
output [attack MONSTER_DICE d MONSTER_DIE against HERO_DICE d HERO_DIE] named "Damage to Player"
output [attack HERO_DICE d HERO_DIE against MONSTER_DICE d MONSTER_DIE] named "Net Damage to Monster (Hero Victory Probability)"

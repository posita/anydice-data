FIGHT: 5
SWORDS: 3
SHIELDS: 2
EXPLODE: 2

set "explode depth" to EXPLODE

function: TYPE count VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: (TYPE+COUNT)
}

output [SWORDS count {4,5,6} in FIGHTd [explode d6]] named "Exploding Swords"
\output [SHIELDS count {1} in FIGHTd [explode d6]] named "Exploding Shields"\

output [SWORDS count {4,5,6} in FIGHTd6] named "Static Swords"
\output [SHIELDS count {1} in FIGHTd6] named "Static Shields"\

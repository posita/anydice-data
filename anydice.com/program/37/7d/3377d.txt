
\ Fate Functions \
function: fate roll FATE:n {
 result: [count {1..FATE} in 2d6]
}

function: fate results ROLL:n {
 if ROLL = 0 { result: -1 }
 if ROLL = 1 { result: 0 }
 if ROLL = 2 { result: 1 }
}

function: fate FATE:n {
 ROLL: [fate roll FATE] 
 result: [fate results ROLL]
}


\ Luck Functions \
\remember luck only rolls when they get a 6!\


\ Roll Functions \
function: basic roll STAT:n { \A basic roll without mods\
 ROLL: 1d6
 result: [count {1..STAT} in ROLL]
}

function: basic full A:n B:n C:n { \a full roll of three stats, w/o mods\
 ROLLA: [basic roll A]
 ROLLB: [basic roll B]
 ROLLC: [basic roll C]

 result: ROLLA + ROLLB + ROLLC
}

\ Stats \
A_ONE: 2
A_TWO: 2
A_THR: 2
A_LUCK: 2
A_FATE: 3

B_ONE: 2
B_TWO: 2
B_THR: 2
B_LUCK: 2
B_FATE: 3


\ Outputs \
output [fate A_FATE] named "Fate"

output [basic roll A_ONE] named "basic roll w/o mods"

output [basic full A_ONE A_TWO A_THR] named "basic full roll w/o mods"

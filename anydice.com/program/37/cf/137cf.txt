function: A:n cmp  D:n {
  if A < D {result: -1} 
  if A > D { result: 1}
  result: 0
}
function: max X:n Y:n {
  if X > Y {result: X}
  result: Y
}
function: A:n vs D:n {
  R: 0
  AR: Ad6
  DR: Dd6
  loop N over {1..[max A D]} {
    R: R + [N@AR cmp N@DR]
  }
  result: R

}
D: 3
loop A over {1..6} {
    output [A vs D] named "[A] vs [D]"
}

function: attack ROLL:s against DEFENSE:s {
  ATTACK: [count 1 in ROLL]
  BLOCK: [count -1 in DEFENSE]
  result: ATTACK - BLOCK
}

HERO_DIE: {1,1,1,0,-1,-1}
MONSTER_DIE: {1,1,-1,-1,0,0}

loop HERO over {2..5} {
  loop MONSTER over {1..3} {
    output [attack HERO d HERO_DIE against MONSTER d MONSTER_DIE] named "[HERO] Hero Dice vs [MONSTER] Monster Dice"
  }
}

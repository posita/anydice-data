FIGHT: 5
SWORDS: 0
SHIELDS: 0
EXPLODE: 1

set "explode depth" to EXPLODE

function: TYPE count VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: (TYPE+COUNT)
}

\output [SWORDS count {4,5,6} in FIGHTd [explode d6]] named "Swords"\
\output [SHIELDS count {1,2} in FIGHTd [explode d6]] named "Shields"\

\output [SWORDS count {4,5,6} in FIGHTd6] named "Swords"\
\output [SHIELDS count {1} in FIGHTd6] named "Shields"\

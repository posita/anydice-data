MISS: 0
HIT: 1
CRITICALHIT: 2
CRITTRESHOLD: 20

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= CRITTRESHOLD {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKS:n {
 if ATTACKS = CRITICALHIT { result: CRITICAL }
 else if ATTACKS = HIT { result: DAMAGE }
 result: 0
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

function: attack oath {
 result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]
}

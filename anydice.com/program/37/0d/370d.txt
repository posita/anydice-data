function: sr N {
  result: Nd{0,0,1}
}

function: sr N limit L {
  result: [highest L of [sr N]]
}

loop A over {1,3,6,9} {
  output [sr A] named "[A] dice"
}

output [sr 6] - [sr 6] named "6 dice vs 6 dice"
loop A over {1..6} {
  output [sr 6 limit A] - [sr 6] named "6 dice [A] limit vs 6 dice"
}

output [sr 6] - 2 named "6 dice vs threshold 2"
loop A over {1..6} {
  output [sr 6 limit A] - 2 named "6 dice [A] limit vs threshold 2"
}

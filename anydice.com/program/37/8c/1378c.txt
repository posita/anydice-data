function: roll gwf damage with SIDES_ON_DMG_DIE:n sided die {
  \ create a die that simulates rerolling the first 1 or 2 as a sequence \
  GWF_DIE: {{1..2}:2, {3..SIDES_ON_DMG_DIE}:SIDES_ON_DMG_DIE+2}
  result: dGWF_DIE
}

output 3d[roll gwf damage with 4 sided die] named "GWF 3d4"
output 2d[roll gwf damage with 6 sided die] named "GWF 2d6"
output 1d[roll gwf damage with 12 sided die] named "GWF 1d12"

\------Standard Dice------\\

function: attackroll ROLL:n vs AC:n { 
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 } 
 if ROLL + PROFICENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result:0 
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 7
DMGBONUS: 12
PROFICENCY: 4
ACCBONUS: - 3
CRITROLL: 20
ATTACKTIMES: 3
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMES d[attackroll d20 vs ENEMYAC] named "two darts w/ BA dart "

DMGBONUS: 1
PROFICENCY: 4
ACCBONUS: 1

ATTACKTIMES: 4
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS

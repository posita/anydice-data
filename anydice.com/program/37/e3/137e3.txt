AC_A: 28
AC_B: 25
CRIT_RANGE: 20
MOD: 15
SHIELD: 0


function:attack ROLL:n vs AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL >= CRIT_RANGE { result: 2 }
 if ROLL + MOD >= AC { result: 1 }
 result: 0
}

AC: AC_A+SHIELD
HIT: [highest 1 of 3d20]
output [attack HIT vs AC] named "3d20d2+Mod"
AC: AC_B+SHIELD
HIT: 1d20
output [attack HIT vs AC] named "1d20+Mod"

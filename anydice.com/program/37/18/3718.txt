\ To Modify \
DICE: 3
HITCHANCE: 5
HITCHANCEFORBONUSDICE: {2}
RRSUCCESS: 0
RRFAILED: 0
DAMAGE: {3,5}
CRITIC: {12}

\ Start Process \
PROBATOHIT: 1
loop N over HITCHANCEFORBONUSDICE{
  PROBATOHIT: PROBATOHIT + (d12 <= N)
}
SUCCESSHIT: (d12 <= HITCHANCE)
SUCCESSHITRRS: (d12 <= HITCHANCE)*(d12 <= HITCHANCE)
SUCCESSHITRRF: [absolute ((d12 <= (12-HITCHANCE))*(d12 <= (12-HITCHANCE)) - 1)]
NBSUCCESS: SUCCESSHIT * PROBATOHIT
NBSUCCESSRRS: SUCCESSHITRRS * PROBATOHIT
NBSUCCESSRRF: SUCCESSHITRRF * PROBATOHIT
RESULT: (DICE-RRSUCCESS-RRFAILED)dNBSUCCESS + RRSUCCESSdNBSUCCESSRRS + RRFAILEDdNBSUCCESSRRF

output RESULT named "number of hits on [DICE]d12"

loop N over DAMAGE{
  DSUCCESS : RESULT/N
  output DSUCCESS named "number of damages on DR[N]"
}

loop N over CRITIC{
  CSUCCESS : RESULT/N
  output CSUCCESS named "number of critics on CR[N]"
}

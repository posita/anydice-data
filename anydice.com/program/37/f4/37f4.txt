function: roll D:s under T:n{
 if D <= T {result:D} else {result:0}
}
function: roll A:n skill B:n{
 S:0
 loop N over {1..A}{
  S:S + [roll d12 under B]
 }
 result:S
}

output [roll 2 skill 4] named "Average Beginner"
output [roll 3 skill 8] named "Average Expert"
output [roll 2 skill 10] named "Highly skilled but Weak"
output [roll 6 skill 3] named "Potent but Unskilled"

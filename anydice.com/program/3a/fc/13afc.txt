\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 10-18\
ENEMYAC: d{10..18}

ATTACKMOD: 2
DMGBONUS: 2
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Staff of the Woodlands (One-Handed)"

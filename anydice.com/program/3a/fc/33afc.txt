MIN_DICE: 0
MAX_DICE: 5
DICE_AMOUNT: 4
MIN_OUTCOME: MIN_DICE * DICE_AMOUNT
MAX_OUTCOME: MAX_DICE * DICE_AMOUNT
AVERAGE_OUTCOME: ( MIN_OUTCOME + MAX_OUTCOME ) / 2

loop LEVEL over {-10..10} {
  if LEVEL < 0 {
    output [lowest DICE_AMOUNT of (DICE_AMOUNT + [absolute LEVEL])d{MIN_DICE..MAX_DICE}] named "[LEVEL]"
  } else {
     output [highest DICE_AMOUNT of (DICE_AMOUNT + [absolute LEVEL])d{MIN_DICE..MAX_DICE}] named "[LEVEL]"
  }
}

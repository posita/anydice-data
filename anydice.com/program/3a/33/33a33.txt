function: explode N:n {
 if N > 18 { result: N + [explode d20] }
 result: N
}

function: explodehigh N:n {
 if N > 19 { result: N + [explodehigh d20] }
 result: N
}

function: limitedexplode L:n and H:n {
 A: 0
 loop N over {1..L} {
  A: A + [explode d20]
 }

 loop N over {1..H} {
  A: A + [explodehigh d20]
 }
 result: A
}

output [limitedexplode 4 and 0]

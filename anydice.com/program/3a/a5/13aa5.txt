ATK: 10
TRGTAC: 20
DMG: 1d10 + 5
PWR: 2d10 + 5

function: attack ATK:n target TRGTAC:n penalty P:n{
  result: [highest of [lowest of d20 + ATK - P - TRGTAC + 1 and 1] and 0] * (DMG)
}

function: pwratk ATK:n target TRGTAC:n penalty P:n{
  result: [highest of [lowest of d20 + ATK - P - TRGTAC + 1 and 1] and 0] * (PWR)
}

output [attack ATK target TRGTAC penalty 0] + [attack ATK target TRGTAC penalty 5] named "Regular attacks"
output [pwratk ATK target TRGTAC penalty 0] named "Power attack"

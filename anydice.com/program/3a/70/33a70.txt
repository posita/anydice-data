function: roll R:n under N:n {
  if R <= N {
    result: 1
  }
  else {
    result: 0
  }
}

function: compare ROLLS:n to EXPECTED:n {
  if ROLLS = EXPECTED {
    result: 1
  }
  else {
    result: 0
  }
}

function: iterate T roll R under N {
  COND: 0
  loop X over {1..T} {
    COND: COND + [roll R under N]
  }
  result: [compare COND to T]
}
output [iterate 10 roll 3d6 under 14]

ATK: 10
TRGTAC: 20
DMG: 1d8 + 4

function: attack ATK:n target TRGTAC:n penalty P:n{
  result: [highest of [lowest of d20 + ATK - P - TRGTAC + 1 and 1] and 0] * (DMG)
}

function: advattack ATK:n target TRGTAC:n penalty P:n{
  result: [highest of [lowest of [highest 1 of 2d20] + ATK - P - TRGTAC + 1 and 1] and 0] * (DMG)
}

output [attack ATK target TRGTAC penalty 0] + [attack ATK target TRGTAC penalty 5] named "Regular attacks (2)"
output [advattack ATK target TRGTAC penalty 0] named "Advantage attack"

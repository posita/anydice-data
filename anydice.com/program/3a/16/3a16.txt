function: income CONTROLDC:n ECONOMYBONUS:n STABILITYBONUS:n CONSUMPTION:n {
 result: [income helper 1d20 1d20 CONTROLDC ECONOMYBONUS STABILITYBONUS CONSUMPTION]
}

function: income helper ECONOMYROLL:n STABILITYROLL:n CONTROLDC:n ECONOMYBONUS:n STABILITYBONUS:n CONSUMPTION:n {
  ECONOMYROLLTOTAL: ECONOMYROLL + ECONOMYBONUS
  STABILITYROLLTOTAL: STABILITYROLL + STABILITYBONUS
  if STABILITYROLLTOTAL < CONTROLDC {
   UNRESTINCOME: 0
  } else {
   UNRESTINCOME: 1
  } 
  if ECONOMYROLLTOTAL < CONTROLDC {
   result: -CONSUMPTION+UNRESTINCOME
  } else {
   if ECONOMYROLL = 1 {
    result: -CONSUMPTION+UNRESTINCOME
   } else {
     result: (ECONOMYROLLTOTAL/5)-CONSUMPTION+UNRESTINCOME
   }
  }
}

function: explode DIE:d {
  MAX: [maximum of DIE]
  result: [explode DIE helper]
}

function: explode N:n helper {
  if N = MAX {
    result: 6 + [explode DIE helper] }
  result: N
}

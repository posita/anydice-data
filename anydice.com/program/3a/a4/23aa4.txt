function: rerollone A:n B:n C:n under N:n {
    \Only reroll if less than N\
    if A + B + C < N {
        \Find the lowest dice\
             if A < B & A < C { result: B + C + 1d6 }
        else if B < A & B < C { result: A + C + 1d6 }
        else                  { result: A + B + 1d6 }
    }
    else { result: A + B + C }
}

function: rerollall R:n under N:n {
   \Only reroll if less than N\
   if R < N { result: 3d6 }
   else { result: R }
}

output [rerollone 1d6 1d6 1d6 under 9]
output [rerollall 3d6 under 9]

function: count highest dupes in DICE:s {
  MAX: 0
  loop X over {1..1@DICE} {
    COUNT: X = DICE
    if COUNT > 1 & COUNT > MAX { MAX: COUNT }
  }
  result: MAX
}

output [count highest dupes in 2d6]

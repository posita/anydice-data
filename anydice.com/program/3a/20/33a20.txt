function: roll ROLL:s locked LCK:n {

   if [count {TGT} in ROLL] = 0 { 
      \ if not a six is rolled, compromise (mark XP, USAGE, STRESS, or risk increased trouble like 1s add to Doom Track) and reroll \
      REROLL: [count {TGT} in Nd6]
      OUTCOME: [lowest of REROLL and 1] \ caps the result \
   }    
   else { 
      OUTCOME: [lowest of (2 + (2@ROLL = TGT)) and 2] \ if first roll is SIX (Success), if DOUBLE SIX (Critical) \
   }

   result: OUTCOME
}

loop N over {1..3} {
  output [roll Nd6 locked 1] named "[N]d6 + reroll"
}

\ outcomes    \
\ fail: 0     \
\ pushed success: 1  \
\ clean success: 2  \

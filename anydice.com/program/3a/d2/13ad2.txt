HIT: 1d8 + 6
CRITICALHIT: 2d8 + 6

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: 0}
 if ROLL = 20 {
  if ROLL >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL >= DEFENSE { result: HIT }
 result: 0
}

DAMAGE: 0
loop N over {1..100}{
 DAMAGE: DAMAGE + [attack 1d20 + 8 vs 15]
}
output DAMAGE

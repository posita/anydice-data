DAYS: 10
ADS: 30
COST: 5+1
DOWN: 0

function: X:n roll {
if X <= 20 {result: -3*COST*DAYS/2}
if X <= 30 {result: -COST*DAYS}
if X <= 40 {result: -COST*DAYS/2}
if X <= 60 {result: 0}
if X <= 80 {result: 1d6*COST}
if X <= 90 {result: 2d8*COST}
else {result: 3d10*COST}
}

output [1d100+DAYS+ADS+DOWN roll]-ADS

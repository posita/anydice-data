function: ROLL:n vs AC:n {
 if ROLL >= THREAT { result: CRITICAL + DAMAGE }
 if ROLL + PROF + STRENGTH + GWF + MAGIC >= AC { result: DAMAGE }
 if ROLL = 1 { result: 0 }
 result: 0

}

AC: d{10..20}
STRENGTH: 5
PROF: 4
MAGIC: 0
GWF: 0
ATTACK: PROF + STRENGTH + GWF + MAGIC

DAMAGE: 1d8 + STRENGTH + MAGIC
THREAT: 20
CRITICAL: 1d8
output 2d[d20 vs AC] named "SwashbuckerRogueDW"

DAMAGE: 5d6
THREAT: 20
CRITICAL: 5d6
output [[highest 1 of 2d20] vs AC] named "SneakAttack"

DAMAGE: 2d6 + STRENGTH + MAGIC
THREAT: 19
CRITICAL: 2d6
output 3d[d20 vs AC] named "Greatsword"

GWF: -5
BONUS: 10
DAMAGE: 2d6 + STRENGTH + MAGIC + BONUS
THREAT: 19
CRITICAL: 2d6
output 3d[d20 vs AC] named "GWFsword"

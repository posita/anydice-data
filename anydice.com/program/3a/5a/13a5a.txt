function: noir DICE:d {
  NUMBEROF: #DICE
  HIGHESTDIE: 1@DICE
  LOWESTDIE: NUMBEROF@DICE
  DIETYPE: [maximum of HIGHESTDIE]

  result: [check HIGHESTDIE LOWESTDIE NUMBEROF 1dDIETYPE]
}

function: check HIGHEST:n LOWEST:n NUMBEROF:n DIETYPE:d {
  if LOWEST = 1 {
    result: HIGHEST + LOWEST
  }

  if LOWEST = 10 {
    result: LOWEST + HIGHEST + NUMBEROFd[explode DIETYPE]
  }

  if HIGHEST = 10 {
    result: LOWEST + HIGHEST + 1d[explode DIETYPE]
  }

  result: HIGHEST + LOWEST
}

output [noir 2d10]

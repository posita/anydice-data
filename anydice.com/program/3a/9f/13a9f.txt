ATK: 10
TRGTAC: 20
DMG: 1d8 + 4

function: iscrit ATK:n target TRGTAC:n penalty P:n{
  X: [highest of [lowest of d20 + ATK - P - (TRGTAC + 10) + 1 and 1] and 0]
  result: X
}

function: hitmiss ATK:n target TRGTAC penalty P:n {
  Y: [highest of [lowest of d20 + ATK - P - TRGTAC+ 1 and 1] and 0]
  if [iscrit ATK target TRGTAC penalty 0] = 0 {Y: 2}
  result: Y
}

output [hitmiss ATK target TRGTAC penalty 0 ] named "Hit or Miss"

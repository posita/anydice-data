\ --- SAVAGE WORLDS ------------------------------------------------------------------------------ \
loop M over { 4, 6, 8, 10, 12 } {
  output [ highest of 1d6 and 1dM ] named "Savage Worlds d[M]"
  output [ highest of [ explode 1d6 ] and [ explode 1dM ] ] named "Savage Worlds Exploded d[M]"
}

\ --- GURPS -------------------------------------------------------------------------------------- \
function: gurps M:n {
  if M < 0 { result: -1 }
  result: M
}

loop M over { 3..18 } {
  output [ gurps (M - 3d6) ] named "GURPS x [M]"
}
\ ------------------------------------------------------------------------------------------------ \

\ --- EABA --------------------------------------------------------------------------------------- \
function: eaba R:s {
  result: 1@R + 2@R + 3@R
}

loop M over { 3..15 } {
  N: M / 3
  B: M - N * 3
  output [ eaba Nd6 ] + B named "EABA [N]d+[B]"
}
\ ------------------------------------------------------------------------------------------------ \

\ --- 2D20 --------------------------------------------------------------------------------------- \
function: success R:s at T:n {
  if R > T { result: 0 }
  if R = 1 { result: 2 }
  result: 1
}

loop M over { 2..19 } {
  D: [ success 1d20 at M ]
  output 2dD named "2d20 vs [M]"
}

\ --- SCOP --------------------------------------------------------------------------------------- \
function: cut R:s at T:n {
  if R > T { result: 0 }
  result: R
}

function: decide R:s {
  if 1@R <= 0 { result: 0 }
  result: 1@R + (R > 0) - 1
}

F: [ cut 1d10 at 4 ]
loop M over { 1..10 } {
  output [ decide MdF ] named "Scop [M]d10x4"
}

G: [ cut 1d6 at 3 ]
loop M over { 1..10 } {
  output [ decide MdG ] named "Scop [M]d6x3"
}

function: damage of ATTACK:n vs DEFENSE:n {
  DIFF: ATTACK - DEFENSE
  if (DIFF < 0) {
    DIFF: 0
  }
  result: DIFF
}

function: damage of ATTACK:n vs DEFENSE:n with defense buff {
  if (DEFENSE < 10) {
    DEFENSE: 10
  }
  result: [damage of ATTACK vs DEFENSE]
}

output [damage of 3d6 vs 3d6 with defense buff] named "3d6 damage (defense >= 10)"
output [damage of 4d{-1..1} vs 4d{-1..1}] named "Fate damage"

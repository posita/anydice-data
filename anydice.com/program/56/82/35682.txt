function: lowversion ROLL:s X:n{
   A: 3@ROLL
   B: 2@ROLL
   result: [highest of A and X] + B
}

output [lowversion 3d6 3]

function: highversion ROLL:s X:n{
   A: 2@ROLL
   B: 1@ROLL
   result: [highest of A and X] + B
}

output [highversion 3d6 3]

function: midversion ROLL:s X:n{
   A: 3@ROLL
   B: 1@ROLL
   result: [highest of A and X] + B
}

output [midversion 3d6 3]

function: dualhopefear on ROLLHOPE:n and ROLLFEAR:n {
 if ROLLHOPE = ROLLFEAR {
  result: 1
 }
 if ROLLHOPE > ROLLFEAR {
  result: 1
 }
 if ROLLHOPE < ROLLFEAR {
  result: 0
 }
}

output [dualhopefear on 1d12 and 1d12] named "Normal Duality Dice Hope"
output [dualhopefear on 1d20 and 1d12] named "Orderborne d20 Hope"

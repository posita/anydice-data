function: rangeband DICE:n between LOW:n and HIGH:n {
RESULT: 0

 if (DICE <= HIGH) {
  if (DICE >= LOW) {
   RESULT: 1
  }
 }
 result: RESULT
}

output [rangeband 2d8 between 0 and 2] named "Total Failure"
output [rangeband 2d8 between 3 and 8] named "Failure"
output [rangeband 2d8 between 9 and 13] named "Success"
output [rangeband 2d8 between 14 and 100] named "Total Success"

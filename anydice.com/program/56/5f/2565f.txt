function: ROLL:n range RANGE:n damage bonus DBONUS:n {
  AC: 15
  BONUS: 5 
  DAMAGE: 1d10
  if ROLL >= RANGE {
    result: 2dDAMAGE + DBONUS
  }
  else if ROLL = 1 {
    result: 0
  }
  else if ROLL + BONUS >= AC {
    result: DAMAGE + DBONUS
  }
  else {
    result: 0
  }
}

output [1d22 range 20 damage bonus 0] named "custom blast"
output [1d20 range 20 damage bonus 3] named "agonising blast"
output [1d20 range 20 damage bonus 0] named "normal blast"

function: rangeband DICE:n between LOW:n and HIGH:n {
 RESULT: 0
 
 if (DICE <= HIGH) {
  if (DICE >= LOW) {
   RESULT: RESULT + 1
  }
 }

 result: RESULT
}

loop X over {-2..4} {
 output [rangeband 2d6+X between 1 and 5] named "[X] Modifer: 1-5"
 output [rangeband 2d6+X between 6 and 8] named "[X] Modifer: 6-8"
 output [rangeband 2d6+X between 9 and 11] named "[X] Modifer: 9-11"
 output (2d6+X >= 12) named "[X] Modifer: 12+"
}

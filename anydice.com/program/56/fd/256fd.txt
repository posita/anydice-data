function: ROLL:n replace FILTER:s with REROLL:d {
  if ROLL = FILTER { result: REROLL }
  result: ROLL
}

function: DIE:d keep K:n {
  result: [highest K of DIE]
}

function: DIE:d lose L:n {
  result: [lowest (#(DIE)-L) of DIE]
}

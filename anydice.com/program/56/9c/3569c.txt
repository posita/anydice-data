function: countdices without VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: #SEQUENCE - COUNT
}

function: highest non zeros from DICE:d {
 NONZEROS: #DICEd(dQ>=1)
 result: {1..(#NONZEROS)}@DICE
}

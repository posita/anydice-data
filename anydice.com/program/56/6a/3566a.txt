function: lowversion ROLL:s X:n{
   A: [highest of 3@ROLL and X]
   B: 2@ROLL
   C: 1@ROLL
   result: A + B + C - [highest of A and C]
}

output [lowversion 3d6 3]

function: highversion ROLL:s X:n{
   A: [highest of 3@ROLL and X]
   B: 2@ROLL
   C: 1@ROLL
   result: A + B + C - [lowest of A and C]
}

output [highversion 3d6 3]

\**** Damage calculator Martial Arts ****\

function: attackroll ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + (WPNDAMAGEA*2)} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS - 3 >= DEF { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + WPNDAMAGEA + (WPNDAMAGEB*3)}
 if ROLL + ATTACKBONUS - 3 >= DEF { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + WPNDAMAGEA + (WPNDAMAGEB*2)}
 if ROLL + ATTACKBONUS - 1 >= DEF { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + WPNDAMAGEA + WPNDAMAGEB}
 if ROLL + ATTACKBONUS >= DEF { result: ROLL + ATTACKBONUS - DEF + RAGEBONUS + WPNDAMAGEA}
 result: 0
}

\looped against enemy DEF\
ENEMYDEF: d{11}

ATTACKBONUS: 6
RAGEBONUS: 5
CRITROLL: 12
WPNDAMAGEA: 9
WPNDAMAGEB: 9
output [attackroll 2d6 vs ENEMYDEF] named "Martial arts combo 3 Final"
output 2d6

function: gwf A:s as B:n
{
   RES: 0

   loop C over {1..#(A)}
   {
      if C@A < 3
      {
         RES: RES + 1dB
      }
      else
      {
         RES: RES + C@A
      }
   }

   result: RES
}

output [gwf 2d6 as 6]+3
output [gwf 2d6 as 6]+3+10
output ([gwf 2d6 as 6]+3+10)*2
output ([gwf 2d6 as 6]+4+10)*2
output ([gwf 2d6 as 6]+5+3+10)*2
output ([gwf 2d6 as 6]+5+4+10)*2

\**** Damage calculator Martial Arts Type B ****\

function: attackmain ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + (WPNDAMAGEA*2)} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS - 0 >= DEF { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + WPNDAMAGEA}
 result: 0
}

function: attackcombone ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: (WPNDAMAGEB*2)} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS - 1 >= DEF { result: WPNDAMAGEB}
 result: 0
}

\looped against enemy DEF\
ENEMYDEF: d{8}

ATTACKBONUS: 2
RAGEBONUS: 2
CRITROLL: 12
WPNDAMAGEA: 4
WPNDAMAGEB: 4
output [attackmain 2d6 vs ENEMYDEF] + [attackcombone 2d6 vs ENEMYDEF] named "Martial arts Type B combo 1"
output 2d6

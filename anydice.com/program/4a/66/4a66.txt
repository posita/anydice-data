function: gwf A:s as B:n
{
   RES: 0

   loop C over {1..#(A)}
   {
      if C@A < 3
      {
         RES: RES + 1dB
      }
      else
      {
         RES: RES + C@A
      }
   }

   result: RES
}

output [gwf 2d6 as 6]
output [gwf 1d10 as 10]
output [gwf 1d12 as 12]
output [gwf 2d6 as 6]+3
output [gwf 1d10 as 10]+3
output [gwf 1d12 as 12]+3
output [gwf 2d6 as 6]+4
output [gwf 1d10 as 10]+4
output [gwf 1d12 as 12]+4
output [gwf 2d6 as 6]+5
output [gwf 1d10 as 10]+5
output [gwf 1d12 as 12]+5

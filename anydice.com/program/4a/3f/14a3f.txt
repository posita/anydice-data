function: cast X:n {
 result: X
}

function: roll to hit {
 result: d6 >= 3
}

function: roll to wound {
 result: d6 >= 5
}

function: roll to unsave {
 result: d6 < 3}

function: roll X:n shots {
 R: 0
 loop N over {1..X}{
  R: R + ([roll to hit] & [roll to wound] & [roll to unsave])
 }
 result: R
}


output [roll 90 shots]

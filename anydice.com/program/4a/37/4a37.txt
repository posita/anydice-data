\ Battled Reckoner, Assault v Colossal \
MAT: 8
RAT: 7
DEF: 10
ARM: 19
HEALTH: 56
DAMAGE: 0

\ Assault Shot \
ATTACKS: 1
POW: 15
TOHIT: [highest of DEF - RAT and 3]
loop N over {1..ATTACKS}{
 DAMAGE: DAMAGE + (2d6 >= TOHIT) * [highest of 2d6 + POW - ARM and 0]
}
\ Charge Attack \
ATTACKS: 1
POW: 19
TOHIT: [highest of DEF - MAT and 3]
loop N over {1..ATTACKS}{
 DAMAGE: DAMAGE + (2d6 >= TOHIT) * [highest of 3d6 + POW - ARM and 0]
}
\ Bought Attacks \
ATTACKS: 2
POW: 19
TOHIT: [highest of DEF - MAT and 3]
loop N over {1..ATTACKS}{
 DAMAGE: DAMAGE + (2d6 >= TOHIT) * [highest of 2d6 + POW - ARM and 0]
}

output DAMAGE >= HEALTH named "Kill"
output DAMAGE named "Damage"

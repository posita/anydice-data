\**** Damage calculator Martial Arts Type A & B ****\

function: attackroll ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + (WPNDAMAGEA*2)} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS - 1 >= DEF { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + WPNDAMAGEA + WPNDAMAGEB}
 if ROLL + ATTACKBONUS >= DEF { result: ROLL + ATTACKBONUS - DEF + RAGEBONUS + WPNDAMAGEA}
 result: 0
}

function: attackmain ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + (WPNDAMAGEA*2)} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS - 0 >= DEF { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + WPNDAMAGEA}
 result: 0
}

function: attackcombone ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: (WPNDAMAGEB*2)} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS - 1 >= DEF { result: WPNDAMAGEB}
 result: 0
}

\looped against enemy DEF\
ENEMYDEF: d{8}

ATTACKBONUS: 2
RAGEBONUS: 0
CRITROLL: 12
WPNDAMAGEA: 5
WPNDAMAGEB: 5
output [attackroll 2d6 vs ENEMYDEF] named "Martial arts Type A combo 1"
output [attackmain 2d6 vs ENEMYDEF] + [attackcombone 2d6 vs ENEMYDEF] named "Martial arts Type B combo 1"
output 2d6

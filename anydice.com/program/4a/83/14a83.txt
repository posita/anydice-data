TARGET: 15
NORMAL: d20 + 5 >= TARGET
ADVANTAGE: 1@2d20 + 5 >= TARGET
SAVE: d20 >= 10
DAMAGE: 1

function: attack N:n times HIT:n {
 if N = 0 {
  result: 0
 }
 if HIT {
  result: DAMAGE + [trip SAVE with N - 1 attacks]
 }
 result: [attack N - 1 times NORMAL]
}

function: trip S:n with N:n attacks {
 if S { result: [attack N times NORMAL] }
 result: [attack N times with advantage]
}

function: attack N:n times with advantage {
 result: (NdADVANTAGE)dDAMAGE
}

output [attack 3 times NORMAL]

set "explode depth" to 15

function: POOL:n p NPICKED:n {
 DELTA: POOL - NPICKED
 SIZE: NPICKED - DELTA
 if DELTA < 0 {
  result: [explode [lowest NPICKED of SIZE d6]]
 }
 else {
  result: [explode [highest NPICKED of POOL d6]]
 }
}

output [3p1] named "3p1"
output [3p2] named "3p2"
output [3p3] named "3p3"

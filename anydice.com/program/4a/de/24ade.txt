\**** Damage calculator Dual Wield Rogue ****\

function: attackmain ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + (WPNDAMAGEA*2)} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS - 0 >= DEF { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + WPNDAMAGEA}
 result: 0
}

function: dualwield ROLL:n vs DEF:n {
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS >= DEF { result: WPNDAMAGEB}
 result: 0
}

\looped against enemy DEF\
ENEMYDEF: d{9}

ATTACKBONUS: 3
RAGEBONUS: 0
CRITROLL: 12
WPNDAMAGEA: 7
WPNDAMAGEB: 7
COMBOTIMES: 1
output [attackmain 2d6 vs ENEMYDEF] + ([dualwield 2d6 vs ENEMYDEF] * COMBOTIMES) named "Trickster Rogue Dual Wield"

output 2d6

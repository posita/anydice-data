function: check S:n vs N:n repeat RP:n mod M:n {
if (S <= N) {
R: 1
}
else {
R: -1
}
if (R = RP) {
result: R + [check 1d20 vs N+M repeat RP mod M]
}
else {
result: 0
}
}

function: check S:s vs N:n {
R: S <= N
if (R = #S) {
result: R + [check 1d20 vs N-4 repeat 1 mod -4]
}
else if (R = 0) {
result: R + [check 1d20 vs N+4 repeat -1 mod 4]
} 
else {
result: R
}
}

 set "maximum function depth" to 2

loop N over {4} {
output [check 3d20 vs N] named "vs [N]"
}

function: reroll lowest of ROLL:s as REROLL:d if less than MIN:n {
  LOWEST: (#ROLL)@ROLL       \ sortuje kości ustawiając najniższą jako pierwszą\
  if LOWEST >= MIN {         \ jeżeli pierwsza kość jest mniejsza niż MIN:n\
    result: ROLL + 0         \ rzuć jeszcze raz\
  } else {
    REST: {1..#ROLL-1}@ROLL  
    result: REST + REROLL    \ wynik, dwie najwyższe kości i przerzut najniższej\
  }
}

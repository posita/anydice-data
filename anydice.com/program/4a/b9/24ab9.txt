\FONCTIONS\

\fonction d'attaque\
function: sum N:n times D:d {
 if N > 1 {result: D + [sum N-1 times D]}
 result: D
}

\reroll de Ciryis Great Weapon Master\
function: reroll X:n {
   if X < 3 {
       result: 1d4
   }
   else { result: X}
}

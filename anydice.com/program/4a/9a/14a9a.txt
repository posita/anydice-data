\ Angry Wizard \

MISS: 0
HIT: 1
CRITICALHIT: 2
WIZARDATTACK: 5
BIGHITDAMAGE: 1d6 + 3
BIGCRITICALDAMAGE: 2d6 + 3
MISSDAMAGE: 0
AC: 10

function:  ROLL:n vs DEFENSE:n ATTACK:n {
 if ROLL = 1 {result: MISS}
 if ROLL = 20 {result: CRITICALHIT}
 if ROLL + ATTACK>= DEFENSE {result: HIT}
 result: MISS
}

\ Damage functions \

function: damage for big ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: BIGCRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: BIGHITDAMAGE
 }
 result: MISSDAMAGE
}

function: damage for small ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: SMALLCRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: SMALLHITDAMAGE
 }
 result: MISSDAMAGE
}

\ Multi attacks \

function: bigmultiattack NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for big [d20 vs AC WIZARDATTACK]]
  }
  result: TOTAL
}

function: smallmultiattack NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for small [d20 vs AC WIZARDATTACK]]
  }
  result: TOTAL
}

function: bigmultiattackadvantage NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for big [[highest 1 of 2d20] vs AC WIZARDATTACK]]
  }
  result: TOTAL
}

function: smallmultiattackadvantage NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for small [[highest 1 of 2d20] vs AC WIZARDATTACK]]
  }
  result: TOTAL
}

function: smallmultiattackdoubleadvantage NUMBEROFATTACKS:n {
  TOTAL: 0
  loop N over {1..NUMBEROFATTACKS} {
    TOTAL: TOTAL + [damage for small [[highest 1 of 3d20] vs AC WIZARDATTACK]]
  }
  result: TOTAL
}

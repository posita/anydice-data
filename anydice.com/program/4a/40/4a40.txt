function: roll DMG:d twice on crit ROLL:n {
 if ROLL = CRIT { result: DMG+DMG }
 else { result: DMG }
}

function: ATK:n hits if gte AC:n {
 if ATK = CRIT | ATK >= AC { result: [roll D twice on crit ATK] }
 else { result: 0 }
}

CRIT: 20
D: 1d6
R: 1d20
AC: 21

output [roll D twice on crit R]
output [R hits if gte AC]

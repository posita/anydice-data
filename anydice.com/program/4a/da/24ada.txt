\**** Damage calculator Martial Arts Type C ****\

function: attackmain ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + (WPNDAMAGEA*2)} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS - 0 >= DEF { result: ROLL + ATTACKBONUS - DEF + (RAGEBONUS) + WPNDAMAGEA}
 result: 0
}

function: attackcombo ROLL:n vs DEF:n {
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS >= DEF { result: WPNDAMAGEB}
 result: 0
}

\looped against enemy DEF\
ENEMYDEF: d{9}

ATTACKBONUS: 4
RAGEBONUS: 0
CRITROLL: 12
WPNDAMAGEA: 7
WPNDAMAGEB: 7
COMBOTIMES: 2
output [attackmain 2d6 vs ENEMYDEF] + ([attackcombo 2d6 vs ENEMYDEF] * COMBOTIMES) named "Martial arts Type c combo extra"

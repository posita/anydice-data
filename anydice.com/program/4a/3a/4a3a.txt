ATTACKS: 1

loop N over {1..ATTACKS}{
 DAMAGE: DAMAGE + (2d6 >= TOHIT) * [highest of 3d6 + POW - ARM and 0]
}

ATTACKS: 4

TOHIT: [highest of DEF - MAT and 3]
loop N over {1..ATTACKS}{
 DAMAGE: DAMAGE + (2d6 >= TOHIT) * [highest of 2d6 + POW - ARM and 0]
}

output DAMAGE >= HEALTH named "Kill"

function: gwf A:s as B:n
{
   RES: 0

   loop C over {1..#(A)}
   {
      if C@A < 3
      {
         RES: RES + 1dB
      }
      else
      {
         RES: RES + C@A
      }
   }

   result: RES
}

output (1d10+1d6)
output (1d10+1d6+3)
output (1d10+1d6+3)*2
output (1d10+1d6+3)*3
output (1d10+1d6+4)*4
output (1d10+1d6+5)*4

set "position order" to "lowest first"

function: partial ROLL:n ATTRIBUTE:n SKILL:n {
  if SKILL < ATTRIBUTE {
    if ROLL <= SKILL {
      result: 2
    }
    if ROLL <= ATTRIBUTE {
      result: 1 
    } 
    result: 0
  }   
  if SKILL = ATTRIBUTE {
    if ROLL <= SKILL {
      result: 2
    }
    result: 0
  }     
  if SKILL > ATTRIBUTE {
    if ROLL <= ATTRIBUTE {
      result: 2
    }
    if ROLL <= SKILL {
      result: 1 
    } 
    result: 0
  }   
  \ this should never happen, using 10 as error code \
  result: 10
} 
  
\ Schleife über Attributswerte von 1 bis 10 \
loop SKL over {1..10} {
  ATT: 5 \ fixer Fertigkeitswert für überschaubare Graphen \

  \ Ausgaben, nur eine zur Zeit aktivieren \
  output [partial d10 ATT SKL] named "Standard ATT [ATT] SKL [SKL]"
  \output [partial [lowest 1 of 2d10] ATT SKL] named "Vorteil ATT [ATT] SKL [SKL]"\
  \output [partial [highest 1 of 2d10] ATT SKL] named "Nachteil ATT [ATT] SKL [SKL]"\
}

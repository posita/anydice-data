\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL = CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{14..20}


ATTACKMOD: 5
PROFICIENCY: 3
CRITROLL: 20
ATTACKTIMES: 2

CRITBONUS: 10
DAMAGE: 1d6 + ATTACKMOD
CRITDAMAGE: 2d6 + ATTACKMOD + CRITBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Shortsword of Lifestealing"
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "Shortsword of Lifestealing (ADV)"


CRITBONUS: 0
DAMAGE: 1d6 + ATTACKMOD
CRITDAMAGE: 2d6 + ATTACKMOD + CRITBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Shortsword"
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "Shortsword (ADV)"

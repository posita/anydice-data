set "maximum function depth" to 8

function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d10] }
 result: N >= 8
}

function: nwod R:n again N:n rote {
 if N >= R { result: 1 + [nwod R again d10] }
 if N < 8 { result: [nwod R again d10] }
 result: N >= 8
}

NWOD: [nwod 10 again d10 rote]

loop N over {1..10} {
 output [lowest of 10 and NdNWOD] named "[N]d"
}

function: reroll R:n under N:n again D:d {
   if R < N { result: D } else {result: R}
}

function: roll R:n minimum N:n {
   if R < N { result: N} else {result: R}
}

loop N over {10, 12} {
   output [reroll 1dN under 2 again 1dN] named "1d[N] dmg- old GWF"
   output [roll 1dN minimum 3] named "1d[N] dmg- new GWF"
}

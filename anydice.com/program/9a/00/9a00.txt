function: roll N:n DIE:d {
 VALUE: 0
 loop I over {1..N}{
  VALUE: VALUE + DIE}
 result: VALUE
}

NUMBER: 3 \ number of dice to roll\
output [roll NUMBER (d6 > 2)] named "standard rules"
output [roll NUMBER (d5 > 2)] named "rerolling weak successes"
output [roll NUMBER (d6 > 3)] named "weak success = 0"
output [roll NUMBER d{0, 0, 1, 1, 2}] named "strong success = 2"

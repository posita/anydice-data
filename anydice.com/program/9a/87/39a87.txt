C: {-6, -5, -4, -3, -2, -1, 1, 2, 3, 4, 5, 6}
N: 12
set "maximum function depth" to 33

function: highest DIE:d in N:n rolls {
  if N = 1 { result: DIE }
  else { result: [highest of DIE and [highest DIE in (N-1) rolls]] }
}

loop K over {1, 2, 4, 8, 16, 32} {
  output [highest Nd10 in K rolls] named "Real highest of [K] rolls"
}
loop K over {1, 2, 3, 4, 5} {
  POWER: 2^K
  output [lowest of N*11/2 and K*11]+[highest of N-2*K and 0]d10+Kd6 named "Combined highest of [POWER] rolls"
}

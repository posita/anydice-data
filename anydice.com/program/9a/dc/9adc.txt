DIE: 1d{-1..1}
LOWESTAMOUNT: 1
HIGHESTAMOUNT: 15
TARGETNUMBER: 1

function: check SINGLEDIE:n against target number {
  result: SINGLEDIE >= TARGETNUMBER
}

loop AMOUNT over {LOWESTAMOUNT..HIGHESTAMOUNT} {
  output AMOUNTd[check DIE against target number] named "Rolling [AMOUNT]df against [TARGETNUMBER]"
}

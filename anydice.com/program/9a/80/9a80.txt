function: two if one A:n {
  if A = 1 { result:2 }
  result: A
}

function: elem NUMDICE:n of DICE:d{
 RES: 0
 loop N over {1..NUMDICE}{
   RES: RES + [two if one 1dDICE]
 }

 result:RES
}

function: roll A:n { result: A }

function: reroll one two A:n of B:d {
  if A = 1 { result:[roll B] }
  result: A
}

function: elemroll NUMDICE:n of DICE:d{
 RES: 0
 loop N over {1..NUMDICE}{
   RES: RES + [reroll one two 1dDICE of 1dDICE]
 }

 result:RES
}

output 2d8
output d[elemroll 2 of d8]
output 2d12
output d[elemroll 2 of d12]

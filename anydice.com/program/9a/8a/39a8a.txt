
function: roll dice{
 X:(R+S)d6
 V:[check [count 6-L/2 in X] and [count 1+L/2 in X]]
 result: [getval V]
}

function: getval C:n{
L:L+1
if C = 0{result:[roll dice]}
result: C
}

function: check A:n and B:n{
 T:A
 if (B+DIF)>S {T:T-(B+DIF)+S}
 result:T
}

loop DIF over {0..3}{
 loop N over {1..5} {
  loop M over {0..N}{
   R:N
   S:M
   L:0
   output [roll dice] named "attribute [N] and skill [M] difficulty [DIF]"
  }
 }
}

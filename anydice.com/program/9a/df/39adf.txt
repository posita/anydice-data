function: highestorSumOfMultiples {
  # Roll 6d6
  R: 6d6

  # Get the highest two rolls
  H: highest 2 of R

  # Count occurrences of each number
  C: count {1, 2, 3, 4, 5, 6} in R

  # Calculate the sum of multiples
  S: sum {i * C[i] | i in {1, 2, 3, 4, 5, 6} if C[i] > 1}

  # Compare the sum of multiples with the highest two rolls
  result: maximum(H, S)

  # Return the result
  result
}

output [highest2orSumOfMultiples]

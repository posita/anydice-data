\
 demonstration of a fast roll-and-keep function for L5R
 10s are kept as-is, they explode after keeping
\

function: ROLLSIZE:n k KEEPSIZE:n {
 result: [convert [highest KEEPSIZE of ROLLSIZE d{1..9, 1000}]]
}

function: convert SUM:n {
 if SUM >= 1000 {
  TENSROLLED: SUM / 1000
  result: SUM - TENSROLLED * 990 + TENSROLLED d [explode d10]
 }
 result: SUM
}

RING: 2

loop POOL over {RING..10} {
output [POOL kRING] named "[POOL]k[RING]"

}

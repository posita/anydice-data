function: to hit N:n {
 if N > 30 {result: 0}
 if N > 20 {result: d{0:19, 1}}
 if N = 20 {result: d{0:19, 2}}
 C: [highest of 11 - N and 1]
 M: N - 1
 H: 20 - C - M
 result: d{0:M, 1:H, 2:C}
}

function: true strike N:n { result: [highest 1 of 2d[to hit N]] }

MAP: 5
loop N over {2..20} {
 TS: [true strike N] + [to hit MAP + N]
 NORMAL: [to hit N] + [to hit MAP + N] + [to hit (MAP * 2) + N]
 output TS - NORMAL named "true strike margin @[N] to hit"
}

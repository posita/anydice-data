function: crit MULT:n DAMAGE:d {
  SUM: {0}

  loop N over {1..MULT} {
    SUM: SUM + DAMAGE
  }

  result: SUM
}

\ TODO: confirm crits \
function: attack AC:n ac roll ROLL:n plus BONUS:n for DAMAGE:d and EXTRA_DAMAGE:d crit CRIT_MULTI:n on CRIT_THREAT:n {
  TO_HIT: ROLL + BONUS

  if (ROLL = 1 | TO_HIT < AC) {
    result: 0
  } else if (ROLL >= CRIT_THREAT) {
    result: [crit CRIT_MULTI DAMAGE] + EXTRA_DAMAGE
  } else {
    result: DAMAGE + EXTRA_DAMAGE
  }
}

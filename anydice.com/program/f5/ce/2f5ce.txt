W: {1:3, 2:3}
D: {0:4, 1:2}

function: W:n outcomes {
  MAJOR: 0
  MINOR: 0
  SUCCESSES: 0
  if W = 2 { MAJOR: 1 }
  if W = 1 { MINOR: 1 }
  SUCCESSES: MAJOR + MINOR
  result: SUCCESSES
}



loop DICE over {1..3} {
  TOTAL: 0
function: D:n cancels {
  NIX: 0
  if D = 0 { NIX: 0 }
  if D = 1 { NIX: 1 }
  result: NIX

  loop THROW over {1..DICE} { TOTAL: TOTAL + [W dW outcomes] }

  if [count 1 in TOTAL > 0] & [count 1 in NIX = 1] { TOTAL: -1 }
}

  output TOTAL named "[DICE] D"
}

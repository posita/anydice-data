function: rerollgwf A:n{
 if(A=1 | A=2){result:1d6}
 result:A
}

function: greatsword damage plus D:n{
 A:[rerollgwf 1d6] + [rerollgwf 1d6] + D
 result:A
}

output [greatsword damage plus 3] named "Greatsword GWF"

output [greatsword damage plus 3] + 
       [greatsword damage plus 3] named "2x Greatsword GWF"

output [greatsword damage plus 3] + 1d8 named "Greatsword GWF + Booming Blade"

output [greatsword damage plus 3] +
       [greatsword damage plus 3] + 
       [greatsword damage plus 3] + 1d8 named "2x Greatsword GWF + Booming Blade BA"

output [rerollgwf 1d6] + [rerollgwf 1d6] + [rerollgwf 1d6] + [rerollgwf 1d6] +
       [rerollgwf 1d6] + [rerollgwf 1d6] + [rerollgwf 1d6] + [rerollgwf 1d6] + 
       [rerollgwf 1d6] + [rerollgwf 1d6] + 1d8 + 15 named "4x Greatsword GWF + Booming Blade BA"

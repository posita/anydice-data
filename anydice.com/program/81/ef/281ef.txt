D: {0,0,0,0,0,1,1,1,2}

function: roll A:n B:n C {
 if A - B >= 2 {
  result: 2
 }

 if A - B >= 1 {
  result: 1
 }
 if A - B >= 0 {
  result: 0
 }
 result: -1
}

loop M over {1..6}{
 loop N over {1..6}{
  output [roll NdD ((M)dD) M] named "[N]-[M]d6"
 }
}
output [roll 10dD (10dD) 10] named "10-10d6"

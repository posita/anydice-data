
function: action roll A:n M:n X:n Y:n {
  HITS : 0
  if A+M > 10 { M : 10 - A }
  if A+M > X { HITS : HITS + 1 }
  if A+M > Y { HITS : HITS + 1 }
  if X = Y {
    if HITS = 2 { HITS : HITS + 0 }
    if HITS = 0 { HITS : HITS - 0 }
  }
  result : HITS 
}

function: loyalist A:n B:n C:n {
  if C = 10 { result : 1 @ [sort {A,B,C}] }
  result : 2 @ [sort {A,B,C}]
}

output [loyalist 1d10 1d10 1d10]
output 1@2d10

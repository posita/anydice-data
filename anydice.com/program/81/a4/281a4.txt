\**** Damage calculator AOE ****\

function: attackroll ROLL:n vs FORT:n {
 result: ROLL + SPELLATTACKBONUS + SPELLDAMAGE - FORT
}

ENEMYFORT: 6

SPELLATTACKBONUS: 2
SPELLDAMAGE: 8
output [attackroll 2d6 vs ENEMYFORT] named "AOE ATTACK SINGLE"
output [attackroll 2d6 vs ENEMYFORT]*5/2 named "AOE ATTACK x 2.5"

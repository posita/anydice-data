\\ Polyhedrals Roll Below Target Number
\


\ Variables, you can add a list inside the {} \
DICE: {4,6,8}
NUMBERS: {1}
VULNERABILITY: {2}
BONUS: {1}



\ Recursive function for the dice roll \
function: for NUM:n of ROLL:n reroll VUL:n as DIE:d if BNS:n {
  \ Handle rerolls if theres bonus damage \
  if ROLL >= VUL & BNS > 0 {
   result: [for NUM of DIE reroll VUL as DIE if (BNS-1)] 
  }
  \ Move on to next die in pool \
  if NUM > 1 {
   result: (ROLL < VUL) + [for NUM-1 of DIE reroll VUL as DIE if BNS]
  }

  \ Return damage result, remaining bonus damage included \
  result: (ROLL < VUL) + BNS
}

\ Loops to test a variety of combinations \
loop DIE over DICE{
 loop N over NUMBERS {
  loop B over BONUS{
   loop V over VULNERABILITY{
    output [for N of dDIE reroll V as dDIE if B] named "Damage Dice: [N]d[DIE] + Bonus Damage:[B] vs [V]"
   }
  }
 }
}

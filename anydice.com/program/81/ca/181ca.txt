function: pro M:n of D:n and L:n
{
  X:0
  loop I over {1..M} {X:X+(1dD >= L)}
  result:X
}

D:5
loop S over {7,6,5,4,3,2}
{
  output [pro (D+2) of ((((S-1)/2)*2)+6) and (6-(S/2))] named "#[S]-[D]-1"
  loop J over {2..7}
  {
    if J <= S
    {
      X:S-J+1
      output [pro (D+J) of ((((X-1)/2)*2)+6) and (6-(X/2))] named "[X]-[D]-[J]"
    }
  }
}

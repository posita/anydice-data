function: owl ROLL:n vs AC:n {
    if ROLL + 3 >= AC {
        result: 2d6+1
    }    
    if ROLL = 20 {
        result: 2d6+1+2d6
    }
    
    result: 0

}
function: multi X:n Y:d vs AC:n {
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: [owl Y vs AC]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}


output [multi 8 1d20 vs 14] named "multiple"
output [owl 1d20 vs 14] named "owl"

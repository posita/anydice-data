SKILL: 3
OPPSKILL: 3
REROLL: 2
OPPREROLL: 2
DICE: 5
TYPE: 6

function: rolling C:n {
 result: (DICE d (d TYPE <= C))
}
function: rerolling A:n by B:n with S:n {
 E: [rolling S]
 result: A + [lowest of (DICE-A) and B] d (d TYPE <= S)
}
function: rerollingb E:n by B:n with S:n {
 A: [rolling S]
 result: A + [lowest of (DICE-A) and B] d (d TYPE <= S)
}

output [rerolling [rolling SKILL] by REROLL with SKILL] - [rerolling [rolling OPPSKILL] by OPPREROLL with OPPSKILL]
output [rerollingb [rolling SKILL] by REROLL with SKILL] - [rerollingb [rolling OPPSKILL] by OPPREROLL with OPPSKILL]

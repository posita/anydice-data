HITROLL: 2
HITDAMAGE: d10 + 11
CRITROLL: 15
CRITDAMAGE: 2d10 + 17
ATTACKS: 6
HITPOINTS: 190

function: attackroll ROLL:n
{
 if ROLL = 1 { result: 0 }
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL >= HITROLL { result: HITDAMAGE }
 result: 0
}

loop ROUNDS over {1..10} 
{
  loop N over {1..ATTACKS} 
    {
      HITPOINTS: HITPOINTS-[attackroll d20]
    }
  output HITPOINTS<=0 named "dead in [ROUNDS] round(s)"
}

output [attackroll d20] named "Damage taken"

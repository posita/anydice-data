function: emphasis DICE:d {
  MEDIAN: [maximum of DICE]/2
  DICE: {DICE}
  MYRESULT: {}
  loop I over DICE{
    loop J over DICE{
      if [absolute I@DICE - MEDIAN] > [absolute J@DICE - MEDIAN] {
        MYRESULT: {MYRESULT, I@DICE}
      }
      else {
        MYRESULT: {MYRESULT, J@DICE}
      }
    }
  }
  result: MYRESULT
}
DIETYPE: d20
output [emphasis DIETYPE] named "Roll with Emphasis"

output DIETYPE named "Straight Roll"
\

function: R:s with emphasis {
 A: [absolute 1@R - 10]
 B: [absolute 2@R - 10]
 if A > B {
  result: 1@R
 }
 if A < B {
  result: 2@R
 }
}

output [2d20 with emphasis]

function: attack D:n plus BONUS:n vs AC:n  {
   if D = 20 { result: 1 } 
   else if D = 1 { result: 0}
   else if D + BONUS >= AC { result: 1 } 
   else { result:0}
}

loop AC over {21..24} {
   output [attack d20 plus 13 vs AC] named "[AC]"
}


loop AC over {21..24} {
   output [attack 1@2d20 plus 13 vs AC] named "with Advantage"
}

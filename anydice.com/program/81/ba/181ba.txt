OFFHAND_DMG_MOD: 0
DMG_MOD: 3
CRIT_THRESHOLD: 20
MISS_THRESHOLD: 10
CRITICAL_MOD: 2

function: R:n attack with DMG:n and MOD:n{
if R >= CRIT_THRESHOLD {result: CRITICAL_MOD d DMG + MOD}
if R >= MISS_THRESHOLD {result: DMG + MOD}
if R < MISS_THRESHOLD {result: 0}
}

DIS_ROLL: [lowest 1 of 2d20]
ADV_ROLL: [highest 1 of 2d20]
ROLL: 1d20

DMG: 1d6
output [ROLL attack with DMG and DMG_MOD] + [ROLL attack with DMG and DMG_MOD] + [ROLL attack with DMG and OFFHAND_DMG_MOD] named "short swords"
\output [ADV_ROLL attack with DMG] + [ADV_ROLL attack with DMG] + [ADV_ROLL attack with DMG] named "adv short swords"\

RAPIER: 1d8
DAGGER: 1d4
output [ROLL attack with RAPIER and DMG_MOD] + [ROLL attack with RAPIER and DMG_MOD] + [ROLL attack with DAGGER and OFFHAND_DMG_MOD] named "rapier&dagger"
\output [ADV_ROLL attack with RAPIER] + [ADV_ROLL attack with RAPIER] + [ADV_ROLL attack with DAGGER] named "adv rapier&dagger"\

DAGGER: 1d2
output [ROLL attack with RAPIER and DMG_MOD] + [ROLL attack with RAPIER and DMG_MOD] + [ROLL attack with DAGGER and OFFHAND_DMG_MOD] named "rai"
\output [ADV_ROLL attack with RAPIER] + [ADV_ROLL attack with RAPIER] + [ADV_ROLL attack with DAGGER] named "adv rai"\

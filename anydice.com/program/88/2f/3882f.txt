function: explode on one three X:n {
  if X = 1 { result: 1 + [explode on one three d6] }
  result: X
}

function: explode on three and four X:n {
  if X = 3 { result: 3 + [explode on three and four d4] }
  if X = 4 { result: 4 + [explode on three and four d4] }
  result: X
}

function: explode on high and low X:n {
  if X = 1 {result: 1 + [explode on high and low d6] }
  if X = 6 {result: 6 + [explode on high and low d6] }
  result: X
}

function: highest X:n of A:s B:s C:s {
  result: {1..X}@[sort {A, B, C}]
}

function: lowest X:n of A:s B:s C:s {
  Y: {1..X}@[sort {A, B, C}]
  result: Y @ [reverse Y]
}

function: reroll X:n {
  if X = 1 {result: [reroll d6]}
  result: X
}

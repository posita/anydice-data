

\Basic Attacks in 1 roll combat system\
\Attack & Defender have no modifiers from traits or items\

N: 2d6

function: semiexplode N:n {
 if  N = 12 {result:12 + d6}
 else {result: N}
}

HANDICAPW: ([semiexplode N]+1) >= ([semiexplode N]+1+2)
HANDICAPM: ([semiexplode N]+1) >= ([semiexplode N]+1+4)
HANDICAPS: ([semiexplode N]+1) >= ([semiexplode N]+1+6)
EXECUTEW: ([semiexplode N]+1) >= ([semiexplode N]+2+2)
EXECUTEM: ([semiexplode N]+1) >= ([semiexplode N]+2+4)
EXECUTES: ([semiexplode N]+1) >= ([semiexplode N]+2+6)
KILL: ([semiexplode N]+1) >= ([semiexplode N]+2+8)

output HANDICAPW named "Weak Handicap"
output HANDICAPM named "Mid Handicap"
output HANDICAPS named "Strong Handicap"
output EXECUTEW named "Weak Execute"
output EXECUTEM named "Mid Execute"
output EXECUTES named "Strong Execute"
output KILL named "InstaDeath"

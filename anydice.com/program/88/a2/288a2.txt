function: check D:n against M:n and N:n {
  if D < M & D < N {
    result: 0
  } else if D >= M & D >= N {
    result: 2
  } else {
    result: 1
  }
}

loop N over {-4..+7} {
  output [check 1d6+N against 1d10 and 1d10] named "Ironsworn (mod [N])"
  output [check 2d6+N against 7 and 10] named "PbtA (mod [N])"
  output [check 1d20+N against 12 and 17] named "Entira RPG (mod [N])" 
}

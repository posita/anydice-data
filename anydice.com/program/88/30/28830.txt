function: test ATTRIBUTE{
 result: [roll 2d10]
}

function: testadv ATTRIBUTE with ADV{
 result: [roll [highest 2 of (2+ADV)d10]]
}

function: testdisadv ATTRIBUTE with DISADV{
 result: [roll [lowest 2 of (2+DISADV)d10]]
}

function: roll YY:s {
 if YY + ATTRIBUTE < 12  { result: 1 }
 if YY + ATTRIBUTE <= 17 { result: 2 }
 else { result: 3}
}

ATTRIBUTE: 0

 output [testdisadv ATTRIBUTE with 1] named "+[ATTRIBUTE] disadv"
 output [test ATTRIBUTE] named "+[ATTRIBUTE]"
 output [testadv ATTRIBUTE with 1] named "+[ATTRIBUTE] adv"
 output [testadv ATTRIBUTE with 2] named "+[ATTRIBUTE] adv 2"
 output [testadv ATTRIBUTE with 3] named "+[ATTRIBUTE] adv 3"

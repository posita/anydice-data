
function: attack with hands AC TYPEOFATTACK {
   
   if TOHITROLL  > AC {
      result: TYPEOFATTACK
   }
}

STR: 5

PROF: 5

STREETFIGHT: 1

RAGEDAMAGE: 3

UNARMEDTOHITMOD: STR + PROF + STREETFIGHT

AXETOHITMOD: STR + PROF + 1

function: unarmed attack ROLL:n ENEMYAC {
   if ROLL = 20 { 
      result: [explode 12d4] + STR + RAGEDAMAGE
   }
   else {
      if ROLL + UNARMEDTOHITMOD > ENEMYAC { 
         result: [explode 5d4] + STR + RAGEDAMAGE
      }
      else { 
         result: 0 
      }
   }
}

function: axe attack ROLL:n ENEMYAC {
   if ROLL = 20 { 
      result: 4d12 + 4d4 + STR + RAGEDAMAGE
   }
   else {
      if ROLL + AXETOHITMOD> ENEMYAC { 
         result: 1d12 + 1d4 + STR + RAGEDAMAGE
      }
      else { 
         result: 0 
      }
   }
}

function: advantage unarmed attack ROLLA:n ROLLB:n ENEMYAC {
   if ROLL = 20 { 
      result: [explode 12d4] + STR + RAGEDAMAGE
   }
   else {
      if ROLL + UNARMEDTOHITMOD > ENEMYAC { 
         result: [explode 5d4] + STR + RAGEDAMAGE
      }
      else { 
         result: 0 
      }
   }
}

function: advantage axe attack ROLLA:n ROLLB:n ENEMYAC {
   if ROLLA = 20 { 
      result: 4d12 + 4d4 + STR + RAGEDAMAGE
   } else if ROLLB = 20 {
      result: 4d12 + 4d4 + STR + RAGEDAMAGE
   } else if ROLLA + AXETOHITMOD > ENEMYAC {
      result: 1d12 + 1d4 + STR + RAGEDAMAGE
   } else if ROLLB + AXETOHITMOD > ENEMYAC {
      result: 1d12 + 1d4 + STR + RAGEDAMAGE
   } else {
      result: 0
   }
}

function: advantage unarmed attack ROLLA:n ROLLB:n ENEMYAC {
   if ROLLA = 20 { 
      result: [explode 12d4] + STR + RAGEDAMAGE
   } else if ROLLB = 20 {
      result: [explode 12d4] + STR + RAGEDAMAGE
   } else if ROLLA + AXETOHITMOD > ENEMYAC {
      result: [explode 5d4] + STR + RAGEDAMAGE
   } else if ROLLB + AXETOHITMOD > ENEMYAC {
      result: [explode 5d4] + STR + RAGEDAMAGE
   } else {
      result: 0
   }
}

function: advantage new axe attack ROLLA:n ROLLB:n ENEMYAC {
   if ROLLA = 20 { 
      result: 4d6 + 4d10 + STR + RAGEDAMAGE
   } else if ROLLB = 20 {
      result: 4d6 + 4d10 + STR + RAGEDAMAGE
   } else if ROLLA + AXETOHITMOD > ENEMYAC {
      result: 2d6 + 1d10 + STR + RAGEDAMAGE
   } else if ROLLB + AXETOHITMOD > ENEMYAC {
      result: 2d6 + 1d10 + STR + RAGEDAMAGE
   } else {
      result: 0
   }
}

output [unarmed attack 1d20 18] named "Unarmed attack"

output [axe attack 1d20 18] named "Axe attack"

output [advantage axe attack 1d20 1d20 18] named "Axe attack with advantage"

output [advantage unarmed attack 1d20 1d20 18] named "Unarmed attack with advantage"

output [advantage new axe attack 1d20 1d20 18] named "New Axe attack with advantage"

function: hit? ATK vs AC {
  result: 1d20 + ATK >= AC
}

function: average damage ATK vs AC with DMG {
  result: [hit? ATK vs AC] * DMG
}

output [average damage 8 vs 14 with 1d6+5] named "Avg Damage with +8 atk vs AC 14"
output [average damage 8 vs 18 with 1d6+5] named "Avg Damage with +8 atk vs AC 18"

function: check event RED:n and GREEN:n with MOD:n using TYPE:n {
  SUM: RED + GREEN + MOD
  if 10 <= SUM & TYPE = 2 {
    result: RED
  } else if 7 <= SUM & SUM <= 9 & TYPE = 1 {
    result: GREEN
  } else {
    result: 0
  }
}

output [check event 1d6 and 1d6 with +0 using 1] named "2d6, event"
output [check event 1d6 and 1d6 with +1 using 1] named "2d6+1, event"
output [check event 1d6 and 1d6 with +2 using 1] named "2d6+2, event"
output [check event 1d6 and 1d6 with +3 using 1] named "2d6+3, event"

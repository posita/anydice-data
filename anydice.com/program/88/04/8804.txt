
function: xplode N:n {
 if N = 6 { result: 1 + [xplode d6]}
 if N = 5 { result: 1}
 result: 0
}

function: kount VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {

  COUNT: COUNT + (P@VALUES = SEQUENCE)

  if SEQUENCE = 6{
    COUNT: COUNT + [xplode d6]
  }
 }
 result: COUNT
}


output [kount {5, 6} in 9d6]

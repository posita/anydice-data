function: attack VALUES:s in ATTACK:s against DVALUES:s in DEFENSE:s{
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = ATTACK)
 }
 loop P over {1..#DVALUES} {
  COUNT: COUNT - (P@DVALUES = DEFENSE)
 } 
 if COUNT < 0{ result: 0}
 else { result: COUNT }
}

\output [count {3..8} in [highest 1 of 2d8]]

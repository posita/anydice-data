DIE: {1,2,3,1,2,3,1,2,4}
DV: 3	\ "die virtual", what the die is acting as \
DA: 9	\ "die actual", what the die is and explodes at \

function: explode S:s {
  if S = 4 { result: 3 + [explode dDIE] }
  result: S
}
  set "maximum function depth" to 5

loop N over {1..6} {

  output Nd[explode dDIE]+N named "[N]d[DV]EX[DA]+[N]"

}

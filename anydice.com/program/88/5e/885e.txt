function: sum all I in ROLL:s{
 result: 1*( ROLL=I )
}

function: myroll ROLL:s{

 A: [sum all 1 in ROLL]

 loop I over {2..6}{
  B: [sum all I in ROLL]
  A: [highest of A and B] 
 }

 result: A
}

function: check D:n plus MOD:n against M:n and N:n with failure on F:n {
  if D=F | (D+MOD < M & D+MOD < N) {
    result: 0
  } else if D+MOD >= M & D+MOD >= N {
    result: 2
  } else {
    result: 1
  }
}

loop M over {-4..+7} {
  if M < 0 {
    output [check 3d6 plus M against 10 and 15 with failure on 1] named "2d6[M]"
  } else {
    output [check 3d6 plus M against 10 and 15 with failure on 1] named "2d6+[M]"
  }
}

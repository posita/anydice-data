PROFICIENCY: 2
STAT: 4
BASE: 3d6
RPROF: 3d6 + PROFICIENCY
RSTAT: 3d6 + STAT
RPAS: 3d6 + STAT + PROFICIENCY
RS: 3d6 + STAT + PROFICIENCY + PROFICIENCY
RSPELL: 3d6 + STAT + PROFICIENCY + STAT
output BASE named "Base Roll"
output RPROF named "With Proficiency"
output RSTAT named "With Stat"
output RPAS named "With Stat + Proficiency"
output RS named "As Specialty"
output RSPELL named "As Spell"
AVERAGE: (BASE + RPROF + RSTAT + RPAS + RS + RSPELL)/6
\output AVERAGE named "Average"\

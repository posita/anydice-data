\ Implementation details \
MISS: 0
NORMAL_HIT: 1
CRITICAL_HIT: 2

function: hit with BONUS:n vs AC:n rolling ROLL:n {
   if ROLL >= CRITICAL_THRESHOLD { result: CRITICAL_HIT }

   if ROLL = 1 { result: MISS }

   if ROLL + BONUS - AC >= 0 { result: NORMAL_HIT }

   result: MISS
}


function: attack with BONUS:n vs AC:n rolling ROLL:n with DICE:d and MOD:n {
   HIT: [ hit with BONUS vs AC rolling ROLL ]

   if HIT = MISS { result: 0 }
   
   if HIT = CRITICAL_HIT { result: DICE + DICE + MOD }
   
   result: DICE + MOD
}
\ End implementation details \

PROFICIENCY: 2
CRITICAL_THRESHOLD: 20

NORMAL_ROLL: 1d20
ADVANTAGE: [ highest 1 of 2d20 ]

LS: 1d8
LS_SAVAGE: 1@2d8
GS: 2d6
GS_SAVAGE: 1@2d6 + 1@2d6
LS_SMITE: 1d8 + 2d8
GS_SMITE: 2d6 + 2d8
LS_SMITE_SAVAGE: 1@2d8 + 1@2d8 + 1@2d8
GS_SMITE_SAVAGE: 1@2d6 + 1@2d6 + 1@2d8 + 1@2d8

loop TARGET_AC over { 12..20 } {

   STR_MOD: 3
   DAMAGE_BONUS: STR_MOD
   HIT_BONUS: STR_MOD + PROFICIENCY + 1d4
   output [ attack with HIT_BONUS vs TARGET_AC rolling NORMAL_ROLL with GS_SMITE_SAVAGE and DAMAGE_BONUS ] named "GS Savage vs AC[TARGET_AC]"

   STR_MOD: 4
   DAMAGE_BONUS: STR_MOD
   HIT_BONUS: STR_MOD + PROFICIENCY + 1d4
   output [ attack with HIT_BONUS vs TARGET_AC rolling NORMAL_ROLL with GS_SMITE and DAMAGE_BONUS ] named "GS ASI vs AC[TARGET_AC]"

   STR_MOD: 3
   DAMAGE_BONUS: STR_MOD + 2
   HIT_BONUS: STR_MOD + PROFICIENCY + 1d4
   output [ attack with HIT_BONUS vs TARGET_AC rolling NORMAL_ROLL with LS_SMITE_SAVAGE and DAMAGE_BONUS ] named "Longsword Savage vs AC[TARGET_AC]"

   STR_MOD: 4
   DAMAGE_BONUS: STR_MOD + 2
   HIT_BONUS: STR_MOD + PROFICIENCY + 1d4
   output [ attack with HIT_BONUS vs TARGET_AC rolling NORMAL_ROLL with LS_SMITE and DAMAGE_BONUS ] named "Longsword ASI vs AC[TARGET_AC]"

   \ Group the results prettier. \
   output {0} named ""
}

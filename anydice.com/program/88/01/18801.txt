TRUE: 1
FALSE: 0
ADD_STATS: FALSE
STATS: {1..6}
AS_IS: {6, 9, 12, 15, 18, 21}
SUGGESTED: {9, 10, 11, 12, 13, 14, 16}

DC_STEPS: SUGGESTED
    loop DC over {DC_STEPS} {
        output 3d6 >= DC named "3d6 vs DC[DC]"
        if ADD_STATS = TRUE {
            loop STAT over {STATS} {
                output 3d6 + STAT >= DC named "3d6 + [STAT] vs DC[DC]"
            }
        }
    }

function: chance N:s T:n {
 L: (#N)@N
 H: 1@N
 if L = 1 { result: 0 }
 if H < T { result: 0 }
 result: #N
}

function: two N:s T:n {
L: (#N)@N
 if L = 1 { result: 0 }
 result: [count {T..6} in N]
}
function: three N:s T:n {
 L: (#N)@N
 if L < T + #N -1 {result: 0 }
 result: #N
}

loop T over {2..6} {
 output [highest 1 of 1d6] >= T named "T=[T], 1d6, normal"
loop D over {2..4} {
 output [chance Dd6 T] named "T=[T], [D]d6, fail if any ones"
output [two Dd6 T] named "T=[T], [D]d6, fail if any ones, independent successes"
output [three Dd6 T] named "T=[T], [D] targets, all or none"
}
}

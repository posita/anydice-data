function: chance N:s T:n {
 L: (#N)@N
 M: T + #N
 if L = 1 {result: 0 }
 result: [count {M..20} in N]
}

loop T over {5,8,11,14,17} {
 output [highest 1 of 1d20] >= T named "T=[T], normal"
 loop D over {2..4} {
  M: T+D
  output [chance Dd20 T] named "T=[T], [D] targets, independent with T'=[M]"
 }
}

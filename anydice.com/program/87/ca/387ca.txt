DAMAGE_MODIFIER: 0

MAX_CRIT_GUARD: 17

STRIKE_COST: 3

CRIT_GUARD: MAX_CRIT_GUARD - (STRIKES_USED * STRIKE_COST)
IS_CRIT: 0

IS_CRIT: TOTAL_DAMAGE > CRIT_GUARD

output IS_CRIT named "crit chance"
output TOTAL_DAMAGE named "damage"

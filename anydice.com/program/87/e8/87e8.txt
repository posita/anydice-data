SPEED: 2
ACCURACY: 8
STRENGTH: 0
LUCK: 1

PLAYERSPEED: 0
PLAYERACCURACY: 0
PLAYERSTRENGTH: 0
PLAYERLUCK: 0

TOUGHNESS: 8


HITCHANCE: (1d10 >= ACCURACY+PLAYERACCURACY)

WOUNDROLL: 1d10
WOUNDRESULT: (WOUNDROLL + STRENGTH + PLAYERSTRENGTH) >= TOUGHNESS

WOUNDCHANCE: HITCHANCE*WOUNDRESULT
output WOUNDCHANCE named "Chance to Wound per Attack"

CRITICALCHANCE: (WOUNDROLL+LUCK+PLAYERLUCK) >= 10
CRITICALHIT: CRITICALCHANCE*HITCHANCE
output CRITICALHIT named "Chance to Crit per Attack"

WOUNDS: 0
CRITS: 0
loop N over {1..(SPEED+PLAYERSPEED)}
{
  WOUNDS: WOUNDS+WOUNDCHANCE
  CRITS: CRITS+CRITICALHIT
}
output WOUNDS named "Chance to Wound"
output CRITS named "Chance to Crit"

function: roll A:s vs armor B:n {
  if (A > B) { result: 1 }
  result: 0
}

function: damage A:s with bonus B:n vs armor C:n {
  Y: [roll 1d20 + B vs armor C]
  result: Y * A
}
output [damage 1d8+3 with bonus 5 vs armor 11]
output [damage 2d8 with bonus 3 vs armor 11]

output [damage 1d8+3 with bonus 5 vs armor 14]
output [damage 2d8 with bonus 3 vs armor 14]

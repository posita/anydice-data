SPEED: 2
ACCURACY: 8
STRENGTH: 0
TOUGHNESS: 8
LUCK: 1


HITCHANCE: (1d10 >= ACCURACY)

WOUNDROLL: 1d10
WOUNDRESULT: (WOUNDROLL + STRENGTH) >= TOUGHNESS

WOUNDCHANCE: HITCHANCE*WOUNDRESULT
output WOUNDCHANCE named "Chance to Wound per Attack"

CRITICALCHANCE: (WOUNDROLL+LUCK) >= 10
output CRITICALCHANCE named "Chance to Crit per Attack"

WOUNDS: 0
CRITS: 0
loop N over {1..SPEED}
{
  WOUNDS: WOUNDS+WOUNDCHANCE
  CRITS: CRITS+CRITICALCHANCE
}
output WOUNDS named "Chance to Wound"
output CRITS named "Chance to Crit"

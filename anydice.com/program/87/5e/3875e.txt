PROGRESS: 150
PROGRESS_TOTAL:450

DICE_BONUS: 27
DICE:3
AA_DICE:0

ROLLS: (DICE + AA_DICE)*(2d50)
PROGRESS_NEEDED: PROGRESS_TOTAL-(PROGRESS + DICE*DICE_BONUS)
R:{ROLLS}
X:{}
loop N over R {
  if N@R >= PROGRESS_NEEDED {
     X: {N@R, X}
  }
}
output #X
output 2@ROLLS

output PROGRESS_NEEDED
ROLLS_SUCCEED: ROLLS > PROGRESS_NEEDED
output ROLLS_SUCCEED

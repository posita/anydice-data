 X:d dFACES-1
 result: [kersplode helper X M times]

function: kersplode helper N:n M:n times {
  if M = 0 {
    if N = FACES-1 { result: FACES-2 }
    else {result: N}
  }
  else if N = FACES-1 { result: FACES-2 + [kersplode M-1 times] }
  else {result: N}
}

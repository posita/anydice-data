\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..22}

ATTACKMOD: 4
DMGBONUS: 2
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: {20}
ATTACKTIMES: 2
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS 
output ATTACKTIMESd[attackroll [highest 1 of 1d20] vs ENEMYAC] named "EA"

ATTACKMOD: 4
DMGBONUS: 2
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: {20}
ATTACKTIMES: 2
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS 
PMS: ATTACKTIMESd[attackroll [highest 1 of 1d20] vs ENEMYAC]

ATTACKMOD: 4
DMGBONUS: 2
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: {20}
ATTACKTIMES: 1
DAMAGE: 1d4 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d4 + ATTACKMOD + DMGBONUS 
PME: ATTACKTIMESd[attackroll [highest 1 of 1d20] vs ENEMYAC] 

output PMS named "PMS"
output PMS+PME named "PMT"

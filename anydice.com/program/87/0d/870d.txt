function: tiecount VALUES:n in SEQUENCE:s {
 COUNT: 0
 loop P over {1..VALUES} {
  if P = 1@SEQUENCE
  {
   COUNT: COUNT + (P = SEQUENCE)
  }
 }
 result: COUNT
}

function: dupes in DICE:s {
  D: 0
  loop X over {2..#DICE} {
    if ((X-1)@DICE = X@DICE) { 
      if (X@DICE > 6){
        D: D + 1}
      }
  }
  result: D
}

function: rollc ROLL:s  {
 result: (ROLL > 6) + ((ROLL = 10)/2)*2
}
function: roll ROLL:s  {
 result: (ROLL > 6) 
}



loop X over {2..4} {
output [roll Xd10] named "[X] Success"
output [dupes in Xd10] named "[X] Crit"
}

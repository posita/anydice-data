function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: 0 }
 if ROLL >= 20 {
  if ROLL + ATTACK >= DEFENSE { result: 2 }
  result: 1
 }
 if ROLL + ATTACK >= DEFENSE { result: 1 }
 result: 0
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = 2 {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = 1 {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

ATTACK: 6
HITDAMAGE: d10 + 4 + 2
CRITICALDAMAGE: 1d10 + 10
MISSDAMAGE: 0
output [damage for [attack d20+6 vs d{5..30}]] * 2

MISS: 0
HIT: 1
CRITICALHIT: 2
MISSDAMAGE: 0

DEFENSE: d{15..19}
MINCRIT: 20
ROLL: d20

function: attack ROLL:n vs DEFENSE:n {
 ATTACK: ATKBONUS + PROF + DEX
 if ROLL = 1 { result: MISS }
 if ROLL >= MINCRIT {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULTS:s {
 D: 0
 loop A over ATTACKRESULTS {
  if A = CRITICALHIT {
   D: D + CRITDAMAGE
  }
  else if A = HIT {
   D: D + HITDAMAGE
  }
  else {
   D: D + MISSDAMAGE
  }
 }
 result: D + [bonus damage for ATTACKRESULTS]
}

function: bonus damage for ATTACKRESULTS:s {
 if [ATTACKRESULTS contains CRITICALHIT] {
  result: 2dBONUS
 }
 if [ATTACKRESULTS contains HIT] {
  result: BONUS
 }
 result: 0
}

function: dualwield prof P:n dex D:n {
 PROF: P
 DEX: D
 ATKBONUS: 0
 HITDAMAGE: WEAPON + MARK + DEX
 CRITDAMAGE: 2dWEAPON + 2dMARK + DEX
 D: [damage for [attack ROLL vs DEFENSE]]
 HITDAMAGE: WEAPON + MARK
 CRITDAMAGE: 2dWEAPON + 2dMARK
 result: D + [damage for [attack ROLL vs DEFENSE]]
}

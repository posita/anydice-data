\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL = 1 { result: CRITDAMAGE } 
 if ROLL = CRITROLL  { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS <= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15}

ATTACKMOD: 0
DMGBONUS: 0
PROFICIENCY: 0
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 9d6
CRITDAMAGE: 18d6
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Shotgun"

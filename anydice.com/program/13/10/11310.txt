D: 16
E: 7

LA: 6
LB: 6
XMAX: 20

function: push helper D:n {
 if D=6 {result: 1 + [push helper 1d6]}
 if D=5 {result: 1}
 result: 0
}

function: push D:n {
 T: 0
 loop N over {1..D}{
  T: T + [push helper 1d6]
 }
 result: T
}

function: push opposed D limit L bonus B against R {
 NH: [highest of 0 and [push D] - Rd{0,0,1}]
 D: [net hits NH bonus B]
 result: D
}

function: reroll opposed D limit L bonus B against R {
 NH: [highest of 0 and [lowest of Dd{0,0,1,0,0,1,1,1,1} and L] - Rd{0,0,1}]
 D: [net hits NH bonus B]
 result: D
}

function: opposed D limit L bonus B against R {
 NH: [highest of 0 and [lowest of Dd{0,0,1} and L] - Rd{0,0,1}]
 D: [net hits NH bonus B]
 result: D
}

function: net hits NH:n bonus B {
 if NH > 0 {result: NH+B}
 result: NH
}

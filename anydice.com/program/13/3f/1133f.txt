function: plusone ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE  }
 result: 0
}

ENEMYAC: d{12..20}

ATTACKMOD: 5
PROFICIENCY: 3
CRITROLL: 20
ATTACKTIMES: 3
DAMAGEBONUS: 0
ACCBONUS: -3

DAMAGE: 1d6 + ATTACKMOD + DAMAGEBONUS + 10
CRITDAMAGE: 2d6 + ATTACKMOD + DAMAGEBONUS + 10

output ATTACKTIMESdDAMAGE named "SS CE"
output ATTACKTIMESd[plusone [highest 1 of 1d20] vs ENEMYAC] named "CE v 12-20 AC"

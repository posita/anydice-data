function: esplode ROLL:s bone BONE:s btwo BTWO:s on SUCCESS:n {
  if (#ROLL < 2) { result: ROLL }
  SUM: 1@ROLL + 2@ROLL
  if (1@ROLL = 2@ROLL){
    SUM: SUM + BONE
    if (BONE = 1@ROLL) {
      SUM: SUM + BTWO
    }
  }
  result: SUM
}

function: ecplode ROLL:s bone BONE:s btwo BTWO:s on SUCCESS:n {
  if (#ROLL < 2) { result: ROLL }
  COUNT: [count {SUCCESS..6} in ROLL]
  if (1@ROLL = 2@ROLL){
    COUNT: COUNT + (BONE >= SUCCESS)
    if (BONE = 1@ROLL) {
      COUNT: COUNT + (BTWO >= SUCCESS)
    }
  }
  result: COUNT
}

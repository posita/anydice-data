function: primary_roll {
  P: d2
  if P = 1 {
    result: [explode_value d4 4]
  } else {
    result: [explode_value d6 6]
  }
}

function: nimble_damage STR ARMOR {
  R: [primary_roll]

  if R = 1 {
    result: 0
  }

  if ARMOR = 0 {
    result: R + STR
  }

  if ARMOR = 1 {
    result: R
  }

  if ARMOR = 2 {
    result: R / 2
  }
}

output [nimble_damage 2 0]
output [nimble_damage 3 1]
output [nimble_damage 4 2]

HIGHONE: 0
HIGHTWO: 0
BONUSONE: 0
BONUSTWO: 0
SUCCESS: 0

loop N over {1} {
  RESULT: d6
  
  if RESULT > HIGHONE {
    if HIGHONE > HIGHTWO {
      HIGHTWO: HIGHONE
    }
    HIGHONE: RESULT
  } else if RESULT > HIGHTWO {
    HIGHTWO: RESULT
  }

  if RESULT >= 4 {
    SUCCESS: SUCCESS + 1
  }
}

if HIGHONE = HIGHTWO {
  BONUSONE: d6

  if BONUSONE >= 4 {
    SUCCESS: SUCCESS + 1
  }

  if BONUSONE = HIGHONE {
    BONUSTWO: d6
    if BONUSTWO >= 4 {
      SUCCESS: SUCCESS + 1
    }
  }
}

TOTRESULT: HIGHONE + HIGHTWO + BONUSONE + BONUSTWO

output TOTRESULT
output SUCCESS

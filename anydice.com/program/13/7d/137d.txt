\ Tally how many rolls are needed to get a required number of successes. \

set "maximum function depth" to 20

function: tally SUCCESS:n {
 REQUIRED: REQUIRED - SUCCESS
 if REQUIRED {
  if !SUCCESS {
   THRESHOLD: [lowest of 9 and [highest of 1 and THRESHOLD + ADJUST]]
  }
  result: 1 + [tally d10 > THRESHOLD]
 }
 result: 1
}

REQUIRED: 4
THRESHOLD: 3
ADJUST: 0
output [tally d10 > THRESHOLD] named "fixed threshold"
ADJUST: 1
output [tally d10 > THRESHOLD] named "increasing threshold"

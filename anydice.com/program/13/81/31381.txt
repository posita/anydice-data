\
D&D ability scores
3d6, reroll 1s
\

function: reroll ONE:s and TWO:s and THREE:s with REROLL:s {
  DIEONE: ONE
  DIETWO: TWO
  DIETHREE: THREE

  if DIEONE = 1 {
   DIEONE: REROLL
  } else {
   if DIETWO = 1 {
    DIETWO: REROLL
    } else {
     if DIETHREE = 1 {
      DIETHREE: REROLL
     }
    }
   }

  R: DIEONE + DIETWO + DIETHREE
  result: R
}


output [reroll d6 and d6 and d6 with d6]

function: nimble X:n {
   if X = 1 {
      result: 0
   } else {
      result: X
   }
}

loop N over {4, 6, 8, 10, 12} {
  output [nimble [explode [highest 1 of 3dN]]] named "d[N] with double advantage"
  output [nimble [explode [highest 1 of 2dN]]] named "d[N] with advantage"
  output [nimble [explode dN]] named "normal d[N]"
  output [nimble [explode [lowest 1 of 2dN]]] named "d[N] with disadvantage"
  output [nimble [explode [lowest 1 of 3dN]]] named "d[N] with double disadvantage"
}

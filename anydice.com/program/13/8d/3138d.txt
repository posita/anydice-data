function: roll TIMES:n {
  EXPLODED: 0
  SUCCESSES: 0
  loop N over {1..TIMES} {
    DIEROLL: 1d6
    SUCCESSES: SUCCESSES + [helper DIEROLL ge 5]
    EXPLODED: EXPLODED + [helper DIEROLL ge 6]
  }
  result: SUCCESSES + [roll EXPLODED]
}

function: helper N:n ge GE:n {
  if N >= GE { result: 1 }
  result: 0
}

output [roll 1]

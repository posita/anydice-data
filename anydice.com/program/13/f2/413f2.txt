\Lancer damage calculator\

ARMOR: 0 \I assume average armor is 1, put 0 for AP\
TN:  8 \Evasion or E-def target number\
ACC: 3 \Accuracy applied\
DIFF: 0 \Difficulty applied - either accuracy or difficulty should be 0\
GRIT: 1 \Increase by 1 if ally in LOS has Field Analyst\
RELIABLE: 5

ROLL: d20 + [highest 1 of ACCd6] - [highest 1 of DIFFd6] + GRIT
function: dmgcalc ROLLTOTAL:n TN:n {
   if ROLLTOTAL >= TN {
    if ROLLTOTAL >=20 {
      result: [highest 3 of 6d6]+3-ARMOR
  } else {
      result: [highest of 3d6+3 and RELIABLE]-ARMOR
      }
  } else {
    result: [highest of 0 and RELIABLE-ARMOR]
  }
}

output [dmgcalc ROLL TN]

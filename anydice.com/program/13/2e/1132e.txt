function: attackroll ROLL:n vs AC:n {
if ROLL >= CRITROLL { result: CRITDAMAGE }
if ROLL = 1 { result: 0 }
if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 4
DMGBONUS: 12
PROFICIENCY: 3
ACCBONUS: -3
CRITROLL: 1
ATTACKTIMES: 1
DAMAGE: 1d8 + 4d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + 8d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Level 8 Assassin Surprise Longbow Attack w/Sharpshooter"

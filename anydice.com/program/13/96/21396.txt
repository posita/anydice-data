function: shadowrun N:n {
 result: N >= 5
}
S: [shadowrun 1d6]

loop N over {2..20} {
 output NdS>2 named "[N]d easy"
}
loop N over {2..20} {
 output NdS>4 named "[N]d medium"
}
loop N over {2..20} {
 output NdS>6 named "[N]d hard"
}
loop N over {-10..10} {
 output 1d20+N>=10 named "Lancer + [N]"
}
loop N over {-10..10} {
 output 1d20+N+1d6>=10 named "Lancer + [N] Advantage"
}
loop N over {-10..10} {
 output 1d20+N-1d6>=10 named "Lancer + [N] Disadvantage"
}
loop N over {-5..10} {
 output 1d20+N>=20 named "Risky + [N]"
}
loop N over {-5..10} {
 output 1d20+N+1d6>=20 named "Risky + [N] Advantage"
}
loop N over {-5..10} {
 output 1d20+N-1d6>=20 named "Risky + [N] Disadvantage"
}

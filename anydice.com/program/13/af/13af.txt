\ complex skill checks as described in WotC's Unearthed Arcana p85 \

function: complex check SUCCESS:n r SR a AF {
 if SUCCESS {
  if SR <= 1 { result: 1 }
  result: [complex check ROLL r SR - 1 a AF]
 }
 if AF {
  result: [complex check ROLL r SR a AF - 1]
 }
 result: 0
}

SUCCESSESREQUIRED: 5
ALLOWEDFAILURES: 2
TARGETNUMBER: 6

ROLL: d20 >= TARGETNUMBER

output [complex check ROLL r SUCCESSESREQUIRED a ALLOWEDFAILURES]

set "maximum function depth" to 10

function: botch N:n {
 if N = 10 {result: -1}
 result: 0
}

function: stress ROLL:n and B:n {
  if ROLL = 10 {result: [botch Bd10]}
  if ROLL = 1 {result: 2* [stress d10 and 0] }
  result: ROLL
}

output [botch 1d10]
output [botch 2d10]
output [botch 3d10]
output [botch 4d10]
output [stress d10 and 1]

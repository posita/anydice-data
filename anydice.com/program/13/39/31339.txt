CRYSTAL:2

DEX:2
AGI:2
MIG:2
ENEMYDODGE:6
ENEMYARMOR:2
ARMORPIERCE:2

DAGGER:3
SWORDHIT:2
SWORDDAM:2
TWOHANDED:8

function: swordroll ROLL:n vs ENEMY:n {
 if ROLL > ENEMY { result: (ROLL-ENEMY)+MIG+SWORDDAM-ENEMYARMOR}
 if ROLL <=  ENEMY { result: 0 }
}

function: axeroll ROLL:n vs ENEMY:n {
 if ROLL > ENEMY { result: (ROLL-ENEMY)+MIG+TWOHANDED-ENEMYARMOR+ARMORPIERCE}
 if ROLL <=  ENEMY { result: 0 }
}

function: dualdaggerroll ROLL:n vs ENEMY:n {
 if ROLL > ENEMY { result: (ROLL-ENEMY)+MIG-ENEMYARMOR}
 if ROLL <=  ENEMY { result: 0 }
}

function: dualswordroll ROLL:n vs ENEMY:n {
 if ROLL > ENEMY { result: (ROLL-ENEMY)+MIG+SWORDDAM+SWORDDAM-ENEMYARMOR}
 if ROLL <=  ENEMY { result: 0 }
}

output [swordroll 1d10+DEX+SWORDHIT+CRYSTALd{0,0,0,1,1,2} vs ENEMYDODGE] named "1H"
output [axeroll 1d10+DEX+CRYSTALd{0,0,0,1,1,2} vs ENEMYDODGE] named "2H"
output [dualdaggerroll 1d10+DEX+DAGGER+DAGGER+CRYSTALd{0,0,0,1,1,2} vs ENEMYDODGE] named "Dual Light"
output [dualswordroll 1d10+DEX+SWORDHIT+SWORDHIT+CRYSTALd{0,0,0,1,1,2} vs ENEMYDODGE] named "Dual Medium"
output [swordroll 1d10+DEX+SWORDHIT+DAGGER+CRYSTALd{0,0,0,1,1,2} vs ENEMYDODGE] named "1H light + 1H Medium"

function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1           { result: 0 }
    if ROLL >= CRIT_RANGE { result: DMG + CRIT_DMG }
    if ROLL + BONUS >= AC { result: DMG }
    result: 0
}
function: lucky ROLL:n with LUCK:d  {
    if ROLL = 1 { result: LUCK }
    result: ROLL
}

AC: 12
WS: 0
DS: 0
DC: 14
HIT: 6
WD: 1d8  \ main weapon damage \
MA: 1d4  \ martial arts die \
DDB: 3   \ damage bonus (from DEX) \
WDB: 3   \ damage bonus (from WIS) \

function: lat DMG:d crit CRIT_DMG:d {
    result: [attack [lucky d20 with d20] plus HIT vs AC for DMG crit CRIT_DMG on 20]
}

function: harm DMG:n {
    if DMG > 0 { result: DMG + MA+WDB }
    result: DMG
}

W: [lat WD+DDB crit WD] \ main weapon damage\
P: [lat MA+DDB crit MA] \ punch damage \

output P + P named "punch + punch, 0 ki"
output W + P named "staff + kick, 0 ki"
output [harm P] named "punch + harm, <1 ki"
output W + [harm P] named "staff + harm kick, <1 ki"
output [harm P + P] named "punch + punch + harm, <1 ki"
output P + P + P named "punch + flurry, 1 ki"
output W + P + P named "staff + flurry, 1 ki"
output W + [harm P + P] named "staff + flurry + harm, <2 ki"
output [harm P + P + P] named "punch + flurry + harm, <2 ki"

output [lat 1d4+5 crit 1d4] named "sling"
output [lat 1d6+5 crit 1d6] named "shortbow"
output (d20 + WS < DC) * 2d12 named "toll"
output (d20 + DS < DC) * 2d8 named "flame"

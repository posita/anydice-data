\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5
DMGBONUS: 21
PROFICIENCY: 6
ACCBONUS: 1
CRITROLL: 19
ATTACKTIMES: 3
DAMAGE: 1d6 + 1d4 + 1d8 + DMGBONUS
CRITDAMAGE: 2d6 + 2d4 + 2d8 + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of 3d20] vs ENEMYAC] named "Shadow of Moil"

ATTACKMOD: 5

ATTACKTIMES: 3

CRITDAMAGE: 2d6 + 4d4 + 2d4 + 2d8 + DMGBONUS

function: roll DICE:n adv ADV:n {
 if ADV>0 {
  result: {1..DICE}@(DICE+ADV)d6
 }
 else if ADV<0 {
  result: {(-1*ADV)..DICE-ADV}@(DICE-ADV)d6
 }
 else {
  result: DICEd6
 }
}

function: calculate bonus of N:n {
 if N > 10 { result: (N-10)/2 }
 if N < 10 { result: (N-11)/2 }
 result: 0
}

output 6d[calculate bonus of [highest 3 of 4d6]]
output 6d[calculate bonus of 6d3]
output 6d[calculate bonus of 3d6]
output 6d[calculate bonus of (1d10+1d5+1d2)]
output 6d[calculate bonus of (1d12+6)]

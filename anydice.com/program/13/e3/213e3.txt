function: roll A:n and B:n and C:n vs D:n {
 if A+B=2 | A+C=2 { result: 1 }
 if A+B+C<D { result: 2 }
 if A+B+C=D | A+B+C=D+1 { result: 3}
 if A=B | A=C { result: 5 }
 result: 4
}

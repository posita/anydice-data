\ Heart, The City Beneath \
\ Results: 0 = failure; 1 = Success at a cost; 2 = Success; 3 = Full Success \

NORMAL: 1
RISKY: 2
DANGEROUS: 3

function: heart RISK:n ROLL:s {
 L: RISK@ROLL 
 if L > 7 {
  result: 2 + (L>9)
 }
 result: L >= 6
}

\ Normal D10 Skill Rolls \

loop P over {2..4} {
 output [heart NORMAL Pd10] named "[P]d10"
}

\ D12 based rolls in The Timeless Arena \

loop P over {2..4} {
 output [heart NORMAL Pd12] named "[P]d12"
}

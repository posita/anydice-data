\******** ENEMY ********\
AC: d{15}

\******** ATTACK FUNCTIONS ********\
function: beast attack plus ATTACK_MODIFIER:n to hit for DMG:d crit CRIT_DMG:d roll ROLL:n against AC:n{
    if ROLL = 1 {result:0}
    if ROLL = 20 {result: DMG + CRIT_DMG}
    if ROLL + ATTACK_MODIFIER >= AC {result: DMG}

    result: 0
}

output 2d[beast attack plus 8 to hit for 1d8+5 crit 1d8 roll d20 against AC] named "Schwidder"

output 2d[beast attack plus 8 to hit for 1d8+1d6+2d6+5 crit 1d8+1d6+2d6 roll d20 against AC] named "Schwidder2 with everything"
output 2d[beast attack plus 3 to hit for 1d8+1d6+1d6+15 crit 1d8+1d6+1d6 roll d20 against AC] named "Schwidder with Sharpshooter + everything"


function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

function: multiattack X:n {
  TOTAL: 0
  loop N over {1..X} {
    DAMAGE: [attack d20 plus 8 vs 1 for 1d8+6 + 1d8 + 1d6 crit 2d8+6 + 2d8 + 2d6 on 20]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}
MULTIATTACK: 4

output [multiattack MULTIATTACK] named "[MULTIATTACK] attacks"

set "position order" to "lowest first"

function: partial ROLL:s OFFSET:n ATTRIBUTE:n SKILL:n {
  LOW: OFFSET
  HIGH: OFFSET + 1

  LOW_VALUE: 1@[sort {SKILL, ATTRIBUTE}]
  HIGH_VALUE: 2@[sort {SKILL, ATTRIBUTE}]

  if LOW@ROLL > HIGH_VALUE {
    result: 0
  }

  if LOW_VALUE >= HIGH@ROLL {
    result: 2
  }

  result: 1
}

\ Schleife über Attributswerte von 1 bis 10 \
loop SKL over {0..10} {
  ATT: 5
  output [partial 2d10 1 ATT SKL] named "Standard ATT [ATT] SKL [SKL]"
  \output [partial 3d10 1 ATT] named "Vorteil ATT [ATT]"\
  \output [partial 3d10 2 ATT] named "Nachteil ATT [ATT]"\
}

function: partial ROLL:n ATTRIBUTE:n SKILL:n {
  RESULT: 0

  if ROLL <= ATTRIBUTE {
    RESULT: 1
  }

  if ROLL <= SKILL {
    RESULT: RESULT + 1
  }

  result: RESULT

}

loop SKL over {0..10} {
  ATT: 5
  \output [partial [highest 1 of 2d10] ATT SKL] named "Nachteil ATT [ATT] SKL [SKL]"\
  output [partial 1d10 ATT SKL] named "Standard ATT [ATT] SKL [SKL]"
  \output [partial [lowest 1 of 2d10] ATT SKL] named "Vorteil ATT [ATT] SKL [SKL]"\
}


output 4d6-3 named "CurveD20"

function: curve adv with modifier N {
  result: [highest 1 of 2d(4d6-3 + N)]
}

loop N over {1, 2, 3} {
  output 4d6-3+N named "CurveD20 + [N]"
  output [curve adv with modifier N] named "Advantage on CurveD20 + [N]"
}

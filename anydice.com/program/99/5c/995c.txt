function: shock DIE:n BONUS:n {
  if DIE = 1 { result: 0 }
  if DIE + BONUS >= 6 { result: d6 }
  if DIE + BONUS >= 4 { result: 1 }
}
function: hits on HIT:n DIE:n helper {
  if DIE = 0 { result: 0 }
  if DIE >= HIT { result: 1 }
  result: 0
}
function: hits on HIT:n {
  result: [hits on HIT d6 helper]
}
loop N over {0..5} {
  output (3*4)d[shock d6 N]d[hits on 3]*2 named "Shock +[N]"
}

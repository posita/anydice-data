function: insegale A:n et B:n et C:n test TEST spe N:n{
 if (A=B)&(C=B)&(A<=TEST)&(N=1) {result: -3}
 if (A=B)&(C=B)&(A>=TEST)&(N=1) {result: 6}
 if (A=B)&(C=B)&(N=0) {result: 6}
 result: (A<=TEST)+(B<=TEST)+(C<=TEST)
}

function: testinsobjectif TEST:n spe N:n{
 A:1d6
 B:1d6
 C:1d6
 result: [insegale A et B et C test TEST spe N]
}


function: deins DE:n objectif TEST:n{
 if DE <= TEST { result: 1 }
 result: 0
}

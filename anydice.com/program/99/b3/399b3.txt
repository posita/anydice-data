function: sqrt X:n {
    if X < 2 {
        result: X
    }

    Y: X
    Z: (Y + (X / Y)) / 2

    loop N over {0..10000}{
        if [absolute (Y - Z)] = 0 {
            if Z*Z<X {
                result: Z+1
            }
            result: Z
        }
        A: Y

        Y: Z
        Z: ((Y + (X / Y)) / 2)

        if A=Z {
            result: [sqrt (X-1)]
        }
    }
}

output [sqrt 1d(20^2)]
output [highest 1 of 2d20]

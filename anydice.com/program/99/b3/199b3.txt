set "position order" to "lowest first"

function: partial ROLL:s ATTRIBUTE:n SKILL:n {
  if ATTRIBUTE < SKILL {
    if 1@ROLL <= ATTRIBUTE {
      if 2@ROLL <= SKILL {
        result: 2
      } else {
        result: 1
      }
    } else {
      if 1@ROLL <= SKILL {
        result: 1
      } else {
        result: 0
      }
    }
  } else {
    if 1@ROLL <= SKILL {
      if 2@ROLL <= ATTRIBUTE {
        result: 2
      } else {
        result: 1
      }
    } else {
      if 1@ROLL <= ATTRIBUTE {
        result: 1
      } else {
        result: 0
      }
    }
  }
  \ this should never trigger, but: \
  result: 0
}

\ Schleife über Attributswerte von 1 bis 10 \
loop ATT over {1..10} {
  SKL: 10 \ fixer Fertigkeitswert für überschaubare Graphen \

  \ Ausgaben, nur eine zur Zeit aktivieren \
  \output [partial 2d10 ATT SKL] named "Standard ATT [ATT] SKL [SKL]"\
  \output [partial [lowest 2 of 3d10] ATT SKL] named "Vorteil ATT [ATT] SKL [SKL]"\
  output [partial [highest 2 of 3d10] ATT SKL] named "Nachteil ATT [ATT] SKL [SKL]"
}

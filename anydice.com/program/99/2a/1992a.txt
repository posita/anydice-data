function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}
function: ROLL:n failsless THRESHOLD:n{
  if ROLL < THRESHOLD { result: 0}
  result: ROLL
}
function: ROLL:n failsmore THRESHOLD:n{
  if ROLL >= THRESHOLD { result: 0}
  result: ROLL
}
function: hits on HIT:n wounds on WOUND:n saves on SAVE:n damage DMG{
  result: 1d[[[[d6 failsless HIT] reroll {HIT..6} as [d6 failsless WOUND]] reroll {WOUND..6} as [d6 failsmore SAVE]] reroll {1..(SAVE-1)} as DMG]
}

function: SHOTS:n shots hits on HIT:n wounds on WOUND:n saves on SAVE:n damage DMG{
  result: SHOTSd([[[[d6 failsless HIT] reroll {HIT..6} as [d6 failsless WOUND]] reroll {WOUND..6} as d{0:(7-SAVE),1..(SAVE-1)}] reroll {1..(SAVE-1)} as DMG])
}
function: SHOTS:n shots hits on HIT:n rerolling BAD:s wounds on WOUND:n saves on SAVE:n damage DMG{
  result: SHOTSd([[[[[d6 reroll BAD as d6] failsless HIT] reroll {HIT..6} as d{0:(WOUND-1),WOUND..6}] reroll {WOUND..6} as d{0:(7-SAVE),1..(SAVE-1)}] reroll {1..(SAVE-1)} as DMG])
}

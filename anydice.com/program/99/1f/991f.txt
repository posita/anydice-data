MISS: 0 
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

ATTACK: 5+2
loop AC over {15..30} {
 output [attack [highest 1 of 2d20]+1d4 vs AC] named "[AC]"
}

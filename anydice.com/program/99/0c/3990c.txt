\******** ENEMY ********\
AC: d{10..15}

\******** ATTACK FUNCTIONS ********\
function: beast attack plus ATTACK_MODIFIER:n to hit for DMG:d crit CRIT_DMG:d roll ROLL:n against AC:n{
    if ROLL = 1 {result:0}
    if ROLL = 20 {result: DMG + CRIT_DMG}
    if ROLL + ATTACK_MODIFIER >= AC {result: DMG}

    result: 0
}

output 1d[beast attack plus 7 to hit for 1d8+3 crit 1d8 roll d20 against AC] named "Folsted"
output 1d[beast attack plus 6 to hit for 2d8+4 crit 2d8 roll d20 against AC]  named "Giant Constrictor Snake"

output 2d[beast attack plus 8 to hit for 1d10+5 crit 1d10 roll d20 against AC] named "Tim"

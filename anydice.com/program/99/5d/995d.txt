function: shock DIE:n BONUS:n {
  if DIE = 1 { result: 0 }
  if DIE + BONUS >= 6 { result: d6 }
  if DIE + BONUS >= 4 { result: 1 }
}
function: wounds on T:n DIE:n helper {
  if DIE = 0 { result: 0 }
  if DIE >= T { result: 1 }
  result: 0
}
function: wounds on T:n {
  result: [wounds on T 1d6 helper]
}
loop N over {0..5} {
  output (3*4)d[shock d6 N]d[wounds on 3]*2 named "Shock +[N]"
}

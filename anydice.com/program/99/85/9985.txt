function: attack A:s vs D:s {
  SUM: 0
  loop N over {1..([lowest of #A and #D])} {
    SUCCESS: N@A > N@D
    SUM: SUM+SUCCESS
  }
  result: SUM
}
function: statroll {
  result: {1, 2, 3}@(4d6)
}
\ output [attack 1d6 vs 1d6] named "1v1"
output [attack 2d6 vs 1d6] named "2v1"
output [attack 2d6 vs 2d6] named "2v2"
output [attack 3d6 vs 1d6] named "3v1"

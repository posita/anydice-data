function: ROLL:n vs AC:n {
 if ROLL >= THREAT & ROLL + ATTACK >= AC {
  result: [threat d20]
 }
 if ROLL + ATTACK >= AC | ROLL = 20 { result: DAMAGE +1d6 +1d6}
 result: 0
}

function: threat ROLL:n {
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: CRITICAL d DAMAGE +1d6 +1d6
 }
 result: DAMAGE +1d6 +1d6
}

AC: 30
POWERATTACK: 5
ATTACK: 9 + 20 - POWERATTACK

DAMAGE: (24d8 + 10)*2 + POWERATTACK
THREAT: 20
CRITICAL: 2
output [d20 vs AC] named "Naruls Fist"

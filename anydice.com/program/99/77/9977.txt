function: ROLL:s limited success roll {
 \ if [count {6..10} in ROLL] = 0 { result: 0 } \
 \ if [count {10} in ROLL] > 0 { result: [lowest of [count{6..10} in ROLL] and 5] } \
 \ if [count {9} in ROLL] > 0 { result: [lowest of [count{6..10} in ROLL] and 4] } \
 \ if [count {8} in ROLL] > 0 { result: [lowest of [count{6..10} in ROLL] and 3] } \
 \ if [count {7} in ROLL] > 0 { result: [lowest of [count{6..10} in ROLL] and 2] } \
 \ if [count {6} in ROLL] > 0 { result: [lowest of [count{6..10} in ROLL] and 1] } \
 if (ROLL > 5) = 0 { result: 0 }
 result: [lowest of (ROLL > 5) and 1 @ [sort ROLL]-5]
}

function: count VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: COUNT
}

function: cast D:n target N:n{
  if D < N {
    result: 0
  }
  result: 1
}

function: D:n greater N:n{
  if D > N{
    result: 1
  }
  result: 0
}

function: D:n smite plus N:n cast M:n damage{
  if (D+N) < 5 {
    result: 0
  }
  if (D+N) < 11{
    result: (1d3+M)
  }
  result: (1d6+M)
}

function: all target smites{
  result: ([cast 2d6 target 7] * (1d3 + 1)) + ([cast 2d6 target 8] * 1d3) + ([cast 2d6 target 4] * 3 * [1d6 greater 3]) + ([cast 2d6 target 7] * 9 * [1d6 greater 5])
}

function: all smites{
  result: 2 * [2d6 smite plus 0 cast 0 damage] + 2 * [2d6 smite plus 1 cast 0 damage] + 1 * [2d6 smite plus 1 cast 1 damage]
}

function: full focus{
  result: [all smites] + ([cast 2d6 target 5] * (1d3 + 1)) + ([cast 2d6 target 4] * 3 * [1d6 greater 3]) + ([cast 2d6 target 8] * 1d3) + ([cast 2d6 target 7] * (1d3 + 1))
}
output [full focus]

function: ROLL:s dark fated with bonus BONUS:n and target TARGET:n {
  if ROLL + BONUS < TARGET{
    result: ROLL + BONUS >= TARGET
  } else {
    result: {2..#ROLL}@ROLL + d6 + BONUS >= TARGET
  }
}
function: ROLL:s light fated with bonus BONUS:n and target TARGET:n {
  if ROLL + BONUS >= TARGET {
    result: ROLL + BONUS >= TARGET
  } else {
    result: {1..#ROLL-1}@ROLL + d6 + BONUS >= TARGET
  }
}

BONUS: 0
TARGET: 1d6
DICE: 1d6
ADVANTAGE_DICE: 2d6

output [lowest #DICE of ADVANTAGE_DICE] + BONUS >= TARGET named "disadvantage"
output [DICE dark fated with bonus BONUS and target TARGET] named "dark fated"
output DICE + BONUS >= TARGET named "normal"
output [DICE light fated with bonus BONUS and target TARGET] named "light fated"
output [highest #DICE of ADVANTAGE_DICE] + BONUS >= TARGET named "advantage"


NORMAL: {2}@3d20
ADVANTAGE: {3}@4d20
DISADVANTAGE: {2}@4d20
\
function: degrees ROLL:n v TARGET:n {
 if ROLL <= TARGET {
  if (((TARGET-ROLL)/2)+1) > 5 {
   result: 5
  }
  else {
   result: (((TARGET-ROLL)/2)+1)
  }
 }
 else {
  if -(((ROLL-(TARGET+1))/2)+1) < -5 {
   result: -5
  }
  else {
   result: -(((ROLL-(TARGET+1))/2)+1)
  }
 }
}

output [degrees NORMAL v 10] named "Normal"\
output NORMAL

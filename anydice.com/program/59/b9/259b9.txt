function: ROLL:s dark fated with bonus BONUS:n {
  if (#1)@ROLL < 4{
    result: ROLL + BONUS
  } else {
    result: {2..#ROLL}@ROLL + d6 + BONUS
  }
}
function: ROLL:s light fated with bonus BONUS:n {
  if (#ROLL)@ROLL >= 4 {
    result: ROLL + BONUS
  } else {
    result: {1..#ROLL-1}@ROLL + d6 + BONUS
  }
}
function: ROLL:s light then dark fated with bonus BONUS:n {
  if (#ROLL)@ROLL >= 4 {
    result: {2..#ROLL}@ROLL + d6 + BONUS
  } else {
    result: {1..#ROLL-1}@ROLL + d6 + BONUS
  }
}

BONUS: 0
DICE: 2d6
ADVANTAGE_DICE: 3d6

output [lowest #DICE of ADVANTAGE_DICE] + BONUS  named "disadvantage"
output [DICE dark fated with bonus BONUS] named "dark fated"
output DICE + BONUS named "normal"
output [DICE light fated with bonus BONUS] named "light fated"
output [highest #DICE of ADVANTAGE_DICE] + BONUS named "advantage"
output [DICE light then dark fated with bonus BONUS] named "light then dark fated"

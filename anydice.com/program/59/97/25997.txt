function: mutagen ROLL:n MODIFIER:n {
  TOTAL: (ROLL + MODIFIER)
  if (TOTAL <= 6) {
    result: 0
  }
  if (TOTAL >= 7) & (TOTAL <= 9) {
    result: 1
  }
  if (TOTAL >= 10) & (TOTAL <= 11) {
    result: 2
  }
  if (TOTAL >= 12)  {
    result: 3
  }
}

loop M over {-1..3} {
  output [mutagen 2d6 M] named "[M]"
}

function: count VALUES:s in SEQUENCE:s subtracting NEGATIVES:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 loop P over {1..#NEGATIVES} {
  COUNT: COUNT - (P@NEGATIVES = SEQUENCE)
 }
 result: COUNT
}

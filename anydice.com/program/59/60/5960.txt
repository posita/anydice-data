function: owod N:n tn TN:n {
 if N = 1 { result: -1 }
 if N < TN { result: 0 }
 if N < 10 { result: 1 }
 \*** result: 1 + [owod d10 tn TN] Sans explosive dice*\
}

OWOD: [owod d10 tn 6]

loop N over {3, 6} {
 output [highest of [lowest of NdOWOD and 10] and -4] named "[N]d"
}

output [highest 3 of 3d{-1,0,0,0,0,1,1,1,1,1}] named "3d drop lowest"
output [highest 3 of 6d{-1,0,0,0,0,1,1,1,1,1}] named "6d drop lowest"

function: downtime D:n {
 B: 0
 if D = 1 {
  B: -1
 }
 if D = 20 {
  B: +1
 }
 if D <= -5 {
  result: B-2
 }
 if D <= 0 {
  result: B-2
 }
 if D <= 5 {
  result: B-1
 }
 if D <= 10 {
  result: B+0
 }
 if D <= 15 {
  result: B+1
 }
 if D <= 20 {
  result: B+2
 }
 if D <= 25 {
  result: B+3
 }
 if D <= 30 {
  result: B+4
 }
 if D <= 35 {
  result: B+5
 }
 if D <= 40 {
  result: B+6
 }
}

loop G over {1..20}{
 T: 0
 loop N over {1..8}{
  T: T + [downtime 1d20+G]
 }
 output T named "[G]"
}

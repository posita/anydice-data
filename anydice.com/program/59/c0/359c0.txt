ADR: 3  \* Adrenaline *\
ADRENALINE: 3
\* output 1dSTY > 1dORK | 1dADR > 1dORK named "STY T[STY] eller ADR T[ADR] över UTMATTNING 1d[ORK]" *\

SKILL: 1

function: die LEVEL {
 result: (2 * LEVEL + 2)
}

loop UTMATTNING over {2,3,4,5,6} {
  UTM: UTMATTNING - ADRENALINE
  output 1d[die SKILL] > 1d[die UTMATTNING - ADRENALINE] named "NYA: SKILL [SKILL] över UTM [UTM] med ADR [ADRENALINE]"
}

loop ORK over {4,6,8,10,12} {
output 1dSTY > 1dORK | 1dADR > 1dORK named "@ GAMLA: STY T[STY] eller ADR T[ADR] över UTMATTNING 1d[ORK]"
}

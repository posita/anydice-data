function: swadie D:n M:n {
 if D = M {
  result: M + [swadie 1dD M]
 }
 result: D
}

function: wild D:n W:n {
 if D = 1 & W = 1 {
  result: -4
 }
 result: [highest of D and W]
}

function: extra D:n W:n {
 if D = 1 & W = 1 {
  result: -4
 }
 result: D
}

loop N over {4, 6, 8, 10, 12} {
 output [lowest of 5 and [extra [swadie 1dN N] [swadie 1d6 6]] / 4] named "Extra d[N]"
 output [lowest of 5 and [wild [swadie 1dN N] [swadie 1d6 6]] / 4] named "Wild d[N]"
}

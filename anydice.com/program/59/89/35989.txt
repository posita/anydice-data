DIE: d12
DIEMAX: [maximum of DIE]

function: roll R:n vs TN:n {
  PART: TN
  FULL: (TN+1) / 2

  if PART > (DIEMAX - 1) { PART: DIEMAX - 1 }
  if FULL >= PART { FULL: PART - 1 }

  if R <= FULL { result: 2 }
  if R <= PART | R = 1 { result: 1 }
  result: 0
}

loop TN over {2,3,6,8,12,14,16,20} {
  output 2d[roll DIE vs TN] named "[TN]"
}

loop TN over {2,3,6,8,12,14,16,20} {
  output [highest 2 of 3d[roll DIE vs TN]] named "[TN]+"
}

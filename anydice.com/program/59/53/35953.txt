function: dual X:n and Y:n {
  HIGHEST: [highest of X and Y]
  LOWEST: [lowest of X and Y]
  if HIGHEST = LOWEST {
    result: (A + A + B)-2
  }
  result: (A + B)-2
}

function: damage DIE:n {
  if DIE >= 20 { result: 6 }
  if DIE >= 17 { result: 5 }
  if DIE >= 10 { result: 4 }
  if DIE >= 7 { result: 3 }
  if DIE >= 4 { result: 2 }
  if DIE >= 1 { result: 1 }
  result: 0
}

B: d4

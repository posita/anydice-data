\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: CRITDAMAGE }
 result: 0
}

\looped against enemy AC from 15-25\
ENEMYAC: d{15..20}

ATTACKMOD: 10
DMGBONUS: 5
PROFICIENCY: 4
ACCBONUS: 0
CRITROLL: 20
BOW: 1d6
SNEAK: 2d6
CRITDAMAGE: 2 * (2 * BOW + ATTACKMOD + DMGBONUS + (2 * SNEAK) + (2 * BOW) + ATTACKMOD + DMGBONUS + (2 * BOW) + ATTACKMOD + DMGBONUS + 2d8)
output [attackroll d20 vs ENEMYAC] named "Whisper Assassin Bow"

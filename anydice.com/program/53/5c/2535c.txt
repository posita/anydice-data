BASE_R: 116
MAX_S: 5

function: hit with S:n stacks {
 RATE: BASE_R+([lowest of S and MAX_S]*160)
 ROLL: 1d1000
 result: ROLL>RATE
}

function: explode N:n with H:n {
 if N {
  result: [explode [hit with (H+1) stacks] with H+1]
 }
 result: H+1
}

output [explode [hit with 0 stacks] with 0]
output [hit with 0 stacks]

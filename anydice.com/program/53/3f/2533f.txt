function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + ATTACKMOD >= AC { result: DAMAGE }
 result: 0
}

function: fullturn ENEMYAC:n {
 result: [attackroll 1d20 vs ENEMYAC] + [attackroll 1d20 vs ENEMYAC]
}

EAC: 14

CRITROLL: 20

ATTACKMOD: 7

DAMAGE: 1d8+4+3d8
CRITDAMAGE: 2d8+4+6d8

output [fullturn EAC]

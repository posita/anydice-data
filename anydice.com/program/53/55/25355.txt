\Enemy AC\
ENEMYAC: 14

\Character Stats\
STATBONUS: 3
PROF: 3
HITBONUS: 1

\Character Damage\
DAMAGE: 1d8+4+3d8
CRITDAMAGE: 2d8+4+6d8

\Additional Values\
ATTACKMOD: STATBONUS+PROF+HITBONUS
CRITROLL: 20
NUMOFATTACKS: 2

\Functions to calculate stuff\
function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + ATTACKMOD >= AC { result: DAMAGE }
 result: 0
}

function: attackaction vs ENEMYAC:n {
 result: NUMOFATTACKSd[attackroll 1d20 vs ENEMYAC]
}

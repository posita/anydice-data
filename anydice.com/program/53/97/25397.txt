ENEMY_AC: 17
DEX: 4

function: X:n first hit {
  if (X = 20) { result: (1d8 * 2) + 1d6 + DEX + 2 }
  if (X + DEX + 4 > ENEMY_AC) { result: 1d8 + 1d6 + DEX + 2 }
  result: 0
}

function: X:n second hit {
  if (X = 20) { result: (1d8 * 2) + 1d8 + 1d6 + DEX + 2 }
  if (X + DEX + 4 > ENEMY_AC) { result: 2d8 + 1d6 + DEX + 2 }
  result: 0
}

output [d20 first hit] + [d20 second hit] named "normal hit"
output [1@2d20 first hit] + [1@2d20 second hit] named "advantage hit"

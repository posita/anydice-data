\**** Damage calculator normal Vs advantage ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 20-25\
ENEMYAC: d{20..25}

ATTACKMOD: 5
DMGBONUS: 2
PROFICIENCY: 4
ACCBONUS: 2
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 1d10 + 1d + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d10 + 2d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "normal attack"

ATTACKMOD: 5
DMGBONUS: 2
PROFICIENCY: 4
ACCBONUS: 2
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 1d10 + 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d10 + 2d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "attack with advantage"

\ To roll for wild magic, roll a d20 + half sorcerer level (rounded down). A surge happens if the roll is less than 7 + spell level. A natural 1 on this roll always causes a surge, regardless of the result. \

loop CASTERLEVEL over {1,5,10,15,20} {
 loop SPELLLEVEL over {1,5,9} {
  \output CASTERLEVELd4 named "[CASTERLEVEL]d4"\
  output d20 > [highest of (7 + SPELLLEVEL) - (CASTERLEVEL / 2) and 1] named "Caster level [CASTERLEVEL] casting spell level [SPELLLEVEL]"
 }
}

function: highest of A:n and B:n {
 if A > B { result: A }
 result: B
}

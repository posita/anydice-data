function: attack ROLL:n CRIT:n AC:n DAM:d CRITDAM:d {
 if ROLL >= CRIT { result: DAM + CRITDAM }
 if ROLL = 1 { result: 0 }
 if ROLL >= AC { result: DAM }
 result: 0
}

loop AC over {6,16,18,21,25} {
    output [attack d20+5 19 AC 1d10+5 15] named "[AC] crit 19"
    output [attack d20+5 20 AC 1d10+5 15] named "[AC] crit 20"
}

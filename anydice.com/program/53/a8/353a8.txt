function: min LOW:n {
  result: (1d6>=LOW)
}

function: min LOW:n on DIE:n {
  if DIE >= LOW { result: 1 }
  result: 0
}

function: ceaseless ROLL:n {
  if ROLL = 1 {
  result: 1d6
  }
  result: ROLL
}

function: relentless ROLL:n min MIN:n {
  if ROLL < MIN {
  result: 1d6
  }
  result: ROLL
}

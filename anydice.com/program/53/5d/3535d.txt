function: basedice EX:n again N:n against DIF:n {
 if N >= EX { result: 1 + [explosion d10 against DIF] }
 if N >= DIF { result: 1 }
 if {N = 1} & {DIF != 1} { result: -1 }
 else {result: 0}
}

function: explosion N:n against DIF:n {
 if {N = 1} & {DIF != 1} { result: -1 }
 result: N >= DIF
}

function: cap N:n {
 if N <= -1 { result: 0}
 else {result: N}
}

DIE_SIZE: d10
DIFFICULTY: 11
EXPLODE_ON: 10
FULLDIE: [basedice EXPLODE_ON again DIE_SIZE against DIFFICULTY ]
ADVANTAGE: 1@2dFULLDIE
DOUBLE_ADVANTAGE: 1@3dFULLDIE



\-- For finding optimal trade off of difficulty vs dice loss \
loop REALM over {0..12} {

 START_DICE: 24
 START_DIF: 0
 loop DROP over {0..4} {
  POOL: START_DICE - 2*DROP
  VAL: START_DIF + REALM - DROP 
  output [cap POOLd[basedice EXPLODE_ON again DIE_SIZE against VAL]] named "R[REALM] M[DROP]"
 }
}


\-- For finding average roll as realm increases \ \
loop REALM over {0..10} {
 loop T over {3} {
  POOL: T + 6
  VAL: 10 - T + REALM
  output [cap POOLd[basedice EXPLODE_ON again DIE_SIZE against VAL]] named "R[REALM] T[T]"
 }
}
\



\-- For checking different difficulties \ \
 loop N over {1..10} {
  output [basedice EXPLODE_ON again DIE_SIZE against N] named "Difficulty [N]" 
}
\

\-- For checking different numbers of dice \ \
loop N over {1..4} {
 output [cap NdFULLDIE] named "capped [N]d10"
}

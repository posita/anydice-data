MELEE: {0,0,1,1,1,2}
RANGE: {0,0,0,1,1,2}
DEFENSE: {0,0,0,0,1,2}

MAXEXPLODECOMBO: 2
set "maximum function depth" to (2 + MAXEXPLODECOMBO)

function: POOL:n DIE:s with up to REROLLS:n rerolls {
  result: [lowest of 10 and [reroll up to REROLLS misses for (POOL d DIE) DIE]]
}

function: reroll up to R:n misses for ROLL:s DIE:s{
  MISSES: ROLL = 0
  HITS: ROLL > 0
  CRITS: ROLL > 1
  result: HITS + [reroll up to [highest of (R - MISSES) and 0] misses for (([highest of [lowest of MISSES and R] and 0] + CRITS)d DIE) DIE]
}

output [1 MELEE with up to 0 rerolls] named "1dMELEE"
output [1 MELEE with up to 1 rerolls] named "1dMELEE, 1 reroll"

output [4 MELEE with up to 3 rerolls] named "DIVA Melee"
output [5 DEFENSE with up to 1 rerolls] named "DIVA Defense"

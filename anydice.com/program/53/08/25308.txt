function: rankcompare LEFT:s vs RIGHT:s {
    LIMIT: [lowest of #LEFT and #RIGHT]
    SLEFT: [sort LEFT]
    SRIGHT: [sort RIGHT]
    HITS: 0
    loop N over {1..LIMIT} {
        if (N@SLEFT > N@SRIGHT) { HITS: HITS+1 }
    }
    result: HITS
}

output [rankcompare {1, 4, 6} vs {5, 5}] named "one hit"
output [rankcompare 3d6 vs 2d6] named "Risk battle"

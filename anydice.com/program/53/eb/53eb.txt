function: fate dice {
 result: 4d{-1..1}
}

function: owod roll DICE:d vs DIFFICULTY:n {
 
  MAXSUCCESS: 10 
  SUCCESSROLLS: {DIFFICULTY..MAXSUCCESS}
  BOTCHROLLS: {1}

  SUCCESSES: [count SUCCESSROLLS in DICE]
  BOTCHES: [count BOTCHROLLS in DICE]

  result: SUCCESSES
}

output [fate dice] named "Fate dice"

function: savage ROLL:d {
    result: [highest of [explode 1d6] and [explode ROLL]]
}
function: reroll REROLL:s from ROLL:n to REPLACEMENT:d {
    if ROLL = 1 {result: 1}
    if [REROLL contains ROLL] {result: REPLACEMENT}
    result: ROLL
}
function: reroll REROLL:s from DICE:d {
    result: [reroll REROLL from DICE to DICE]
}

output [savage 1d6] named "savage d6"
output [reroll {1..3} from [savage 1d6]] named "d6 reroll failures"
output [reroll {1..7} from [savage 1d6]] named "d6 reroll non-raise"

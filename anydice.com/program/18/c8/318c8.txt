
function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

\ Great Weapon Fighting: Re-Roll 1-2 Damage Dice \
\ Savage Attacker: Roll Damage Twice, keep highest \
\ Savage Attacks: Extra Damage di on Critical Hit \

output 2d[reroll {1,2} on d6] named "Great Sword, Normal Hit, GWF, Savage Attacks"
output 1d[reroll {1,2} on d12] named "Great Axe, Normal Hit, GWF, Savage Attacks"

\ Critical Hits, including Savage Attack (extra damage di) \
output 4d[reroll {1,2} on d6]+1d6 named "Great Sword Critical, GWF, Savage Attacks"
output 2d[reroll {1,2} on d12]+1d12 named "Great Axe Critical Hit, GWF, Savage Attacks"

\ With Savage Attacker (Roll Twice, keep Highest Result) \

output [highest of 2d[reroll {1,2} on d6] and 2d[reroll {1,2} on d6]] named "Great Sword, Normal Hit, GWF, Savage Attacks, Savage Attacker"
output [highest of 1d[reroll {1,2} on d12] and 1d[reroll {1,2} on d12]] named "Great Axe, Normal Hit, GWF, Savage Attacks, Savage Attacker"

output [highest of 4d[reroll {1,2} on d6]+1d6 and 4d[reroll {1,2} on d6]+1d6] named "Great Sword Critical, GWF, Savage Attacks, Savage Attacker"
output [highest of 2d[reroll {1,2} on d12]+1d12 and 2d[reroll {1,2} on d12]+1d12] named "Great Axe Critical Hit, GWF, Savage Attacks, Savage Attacker"

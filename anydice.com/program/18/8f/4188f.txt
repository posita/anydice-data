
function: tohit DIE:n plus B:n vs dv DV:n {
    if DIE + B <= DV {result: 0}
    else if DIE + B = DV + 2 {result: DAM * 2}
    else if DIE + B > DV + 2 {result: DAM * 3}
    else {result: DAM}
}

function: tohit DIE:n plus B:n tohurt DAM:n vs dv DV:n and sp SP:n {
    TEST : [tohit DIE plus B vs dv DV]
    if TEST <= SP {result: 0}
    else {result: TEST - SP}
}

loop SP over {4..11} {
output [tohit d10 plus 14 tohurt 2d6 vs dv 17 and sp SP] named "[SP] Stopping Power"
}

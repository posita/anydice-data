function: modulo LHS by RHS {
    result: LHS - (RHS * (LHS/RHS))
}

function: power BASE to EXP {
    if EXP = 0 {
        result: 1
    }

TEMP: [power BASE to EXP/2]

if([modulo EXP by 2] = 0) {
     result: TEMP*TEMP
    }
    else {
        result: BASE*TEMP*TEMP
    }
}

set "maximum function depth" to 10

loop I over {1..100}
{
    output [power I to 2]
}

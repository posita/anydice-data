X: 3
Y: 6

function: advantage X dice Y sides {
  RESULT: 0
  loop I over {1..X} {
    RESULT: RESULT + [highest 1 of 2dY]
  }
  result: RESULT
}

output 3d6 named "no advantage"
output [highest X of (2*X)dY] named "(2X)dY"
output [highest X of (1+X)dY] named "(X+1)dY"
output [highest of XdY and XdY] named "Roll XdY twice"
output [advantage X dice Y sides] named "Roll 2dY X times"

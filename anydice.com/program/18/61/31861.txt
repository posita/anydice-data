
function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output 2d[reroll {1,2} on d6] named "reroll 1-2 on 2d6 (Great Sword Normal Hit)"
output 1d[reroll {1,2} on d12] named "reroll 1-2 on a d12 (Great Axe Normal Hit)"

output 4d[reroll {1,2} on d6] named "reroll 1-2 on 4d6 (Great Sword Critical)"
output 3d[reroll {1,2} on d12] named "reroll 1-2 on a 3d12 (Great Axe Critical Hit)"

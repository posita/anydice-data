
function: tohit DIE:n plus B:n vs dv DV:n {
    if DIE + B <= DV {result: 0}
    else {result: DAM}
}

function: tohit DIE:n plus B:n tohurt DAM:n vs dv DV:n and sp SP:n {
    SPTEST: [tohit DIE plus B vs dv DV]
    if SPTEST <= SP {result: 0}
    else {result: SPTEST - SP}
}

function: rof ROF:n tohit DIE:n plus B:n tohurt DAM:n vs dv DV:n and sp SP:n {
    result: [tohit DIE plus B tohurt DAM vs dv DV and sp SP]
}


loop SP over {4..11} {
output [rof 2 tohit d10 plus 14 tohurt 3d6 vs dv 15 and sp SP] named "[SP] Stopping Power"
}

function: checknumberedsequence A:s {
 X: [sort A]
 XN: -9
 NBORD: 0
 NBORDMAX:0
 loop N over X{
  if N = XN-1 {
   NBORD:NBORD+1
  }else {
   NBORD:0
  }
  NBORDMAX: [highest of NBORD and NBORDMAX]
  XN:N
 }
 result:NBORDMAX
}

function: checkpairormore A:s {
 X: [sort A]
 NBORD: 0
 NBORDMAX:0
 loop N over X{
  NBORD:[count N in X]
  NBORDMAX: [highest of NBORD and NBORDMAX]
 }
 result:NBORDMAX
}


output [checknumberedsequence 6d10] named "ord"


output [checkpairormore 6d10] named "fam"


function: tohit DIE:n plus B:n vs dv DV:n {
    if DIE + B <= DV {result: 0}
    else {result: DAM}
}

function: tohit DIE:n plus B:n tohurt DAM:n vs dv DV:n and sp SP:n {
    TEST : [tohit DIE plus B vs dv DV]
    if TEST <= SP {result: 0}
    else {result: TEST - SP}
}

loop SP over {4..11} {
output [tohit d10 plus 14 tohurt 3d6 vs dv 15 and sp SP] named "[SP] Stopping Power"
}

H: 12   \Hero die\
S: 12   \Skill die\
C: 12   \Career die\
E: 12   \Expertise die\
A: 12   \Advantage die\
TN: 5   \Target number for action dice\
D: 5    \Number of difficulty dice\
AN: 3   \Activation number for difficulty dice\


function: evaluate HERO:d and STAT:d and CAREER:d and EXPERTISE:d and ADVANTAGE:d vs DIFFICULTY:n {
  result: (HERO >= TN) + (STAT >= TN) + (CAREER >= TN) + (EXPERTISE >= TN) + (ADVANTAGE >= TN) - DIFFICULTYd(d4<=AN)
}

output [evaluate 1dH and 1dS and 1dC and 1dE and 1dA vs D]

\**** Damage calculator Weapon Master ****\

function: attackroll ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: ROLL + ATTACKBONUS - DEF + DMGBONUS + (WPNDAMAGEA*3) + (WPNDAMAGEB*2)} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS - 3 >= DEF { result: ROLL + ATTACKBONUS - DEF + DMGBONUS + (WPNDAMAGEA*2) + (WPNDAMAGEB*2)}
 if ROLL + ATTACKBONUS - 2 >= DEF { result: ROLL + ATTACKBONUS - DEF + DMGBONUS + (WPNDAMAGEA*2) + (WPNDAMAGEB*1)}
 if ROLL + ATTACKBONUS - 1 >= DEF { result: ROLL + ATTACKBONUS - DEF + DMGBONUS + (WPNDAMAGEA*2) + (WPNDAMAGEB*0)}
 if ROLL + ATTACKBONUS >= DEF { result: ROLL + ATTACKBONUS - DEF + DMGBONUS + WPNDAMAGEA}
 result: 0
}

\looped against enemy DEF\
ENEMYDEF: d{11}

ATTACKBONUS: 5
DMGBONUS: 0
CRITROLL: 12
ATTACKTIMES: 1
WPNDAMAGEA: 8
WPNDAMAGEB: 5
output ATTACKTIMESd[attackroll 2d6 vs ENEMYDEF] named "Weapon Master"
output 2d6

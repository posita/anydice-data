X: 1
Y: 20

function: advantage X dice Y sides {
  RESULT: 0
  loop I over {1..X} {
    RESULT: RESULT + [highest 1 of 2dY]
  }
  result: RESULT
}

output XdY named "no advantage"
output [highest X of (2*X)dY] named "highest X of (2X)dY"
output [highest X of (1+X)dY] named "highest X of (X+1)dY"
output [highest of XdY and XdY] named "Roll XdY twice, tale highest"
output [advantage X dice Y sides] named "Roll 2dY X times, summing the higher reaults from each pair"

loop SKILL over {1..4}{
function: ROLL:s with rerolls {
 BUMPS: SKILL
 R: 0
 loop P over {1..#ROLL} {
  if P@ROLL=4 { R: R + P@ROLL }
  else if BUMPS > 0 {
   R: R + P@ROLL + 1
   BUMPS: BUMPS - 1 }
  else { R: R + P@ROLL }
 }
 result: R
}

output [4d4 with rerolls] named "[SKILL] bumps"
}
output 4d4 named "4d4"

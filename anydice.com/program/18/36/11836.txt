Q_SIXES : 3
S_SIXES : 1d6
Q_FOURS : 2
S_FOURS : 1d4

function: pool POOL:n of DIE:d {
  NEW: {}
  loop J over {1 .. POOL} {
    if J <= POOL { NEW: { NEW , DIE} }
  }
 result : NEW
}

output ({[pool Q_SIXES of S_SIXES],[pool Q_FOURS of S_FOURS]})
output [count {1..2} in {[pool Q_SIXES of S_SIXES],[pool Q_FOURS of S_FOURS]}]
output ({1d6,1d6})

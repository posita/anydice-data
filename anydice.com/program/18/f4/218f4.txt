\**** Damage calculator ****\

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 10-20\
ENEMYAC: d{10..20}

ATTACKMOD: 4
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20

DAMAGE: 1d8 + ATTACKMOD + 2d6
CRITDAMAGE: 2d8 + ATTACKMOD + 4d6

output d[attackroll 1d20 vs ENEMYAC] named "Zélphar niveau 4 avec Elven Accuracy sans avantage"

output d[attackroll [highest 1 of 3d20] vs ENEMYAC] named "Zélphar niveau 4 avec Elven Accuracy avec avantage"

PIERCERDAMAGE: [reroll {1..4} on d8]

DAMAGE: PIERCERDAMAGE + ATTACKMOD + 2d6
CRITDAMAGE: PIERCERDAMAGE + 2d8 + ATTACKMOD + 4d6
output d[attackroll 1d20 vs ENEMYAC] named "Zélphar niveau 4 avec Piercer sans avantage"

output d[attackroll [highest 1 of 2d20] vs ENEMYAC] named "Zélphar niveau 4 avec Piercer avec avantage"

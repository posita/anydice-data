function: pool N:n prof P:n setback S:n {
  R: [count {5,6} in Pd6] + [count {6} in Nd6] - [count {1} in Sd6]
  result: R
}

loop STAT over {1,3} {
loop SKILL over {1,3,5} {
loop W over {1..3} {
  P: [lowest of SKILL and STAT]
  N: [absolute (SKILL-STAT)]
  output [pool (N+W) prof P setback 0] named "[STAT] Sk[SKILL] W[W]"
}
}
}

\ Enchantment magnitude: starts at 1 if enchanted, then +1 repeatedly while a 1/3 check succeeds.\
function: ench {
  MAG: 1
  loop N over {1..5}{
    if [d3] = 1 { MAG: MAG + 1 } else { result: MAG }
  }
}

\ Signed enchantment:\
\ 1/10 negative, 1/10 positive, otherwise 0.\
function: enchant {
  if d10 = 1 { result: -[ench] }   \ negative 10% \
  if d9 = 1 { result: [ench] }    \ positive 10% \
  result: 0                          \ 80% \
}

output [d6 + [enchant]]

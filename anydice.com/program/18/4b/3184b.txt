DAMAGEDICE: 1d4 + 5d6
DAMAGEBONUS: 7
TOCRIT: 17
ADVANTAGE: 1

function: damage from ROLL:n {
  if ROLL >= TOCRIT {result: DAMAGEDICE*2 + DAMAGEBONUS}
  else {result: DAMAGEDICE + DAMAGEBONUS}
}

if ADVANTAGE = 1 {ROLLDICE: 1@2d20}
if ADVANTAGE = 0 {ROLLDICE: d20}
output [damage from ROLLDICE] named "Total Damage"

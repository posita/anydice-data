ATTACK : 2d6
DEFENSE : 2d6
MAXDIE : 6
THITS : 0

function: is S:n over zero {
 if S >0 {
  T : 1
}
 else {
  T : 0
}
 result : T
}


function: hits A through D {
 loop N over {1..MAXDIE} {
  AHITS : [count {N} in A]
  DHITS : [count {N} in D]
   SUCCESS : (AHITS - DHITS)
    THITS : THITS + SUCCESS * [is SUCCESS over zero]
}  
result : THITS
}

output [hits ATTACK through DEFENSE]

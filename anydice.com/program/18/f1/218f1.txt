\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 10-20\
ENEMYAC: d{10..20}

ATTACKMOD: 4
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20
DAMAGE: 1d8 + ATTACKMOD + 2d6
CRITDAMAGE: 2d8 + ATTACKMOD + 4d6
output d[attackroll 1d20 vs ENEMYAC] named "Zélphar niveau 4 avec Elven Accuracy sans avantage"

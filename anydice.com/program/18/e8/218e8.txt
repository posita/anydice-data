STR_MOD: 5
PROF_MOD: 4 + 2 - 5
ATTACK: STR_MOD + PROF_MOD

TARGET_AC: 15
HIT: 1d6 + 10 + STR_MOD
CRIT: 2d6 + 10 + STR_MOD

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: 0 }
 if ROLL = 20 {
  result: CRIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: 0
}

DPR: 0

loop N over {1..4}{
 DPR: DPR + [attack d20 vs TARGET_AC]
}

output DPR

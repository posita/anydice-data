function: explode N:n on SIZE:n {
 if N = 12 {
  result: N
}
 if N = SIZE {
  result: [highest of N and [explode d(SIZE + 2) on SIZE + 2] ]
 }
 result: N
}

FOUR: [explode d4 on 4]
SIX: [explode d6 on 6]
EIGHT: [explode d8 on 8]
TEN: [explode d10 on 10]
TWEL: [explode d12 on 12]

output FOUR
output SIX
output EIGHT
output TEN
output TWEL
output 1@2dFOUR named "Mediocre"
output [highest of 1dFOUR and 1dSIX] named "Average"
output 1@2dSIX named "Fair"
output [highest of 1dSIX and 1dEIGHT] named "Good"
output 1@2dEIGHT named "Great"
output [highest of 1dEIGHT and 1dTEN] named "Superb"
output 1@2dTEN named "Fantastic"
output [highest of 1dTEN and 1dTWEL] named "Epic"
output 1@2dTWEL named "Legendary"

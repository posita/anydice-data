function: hit DICE:n vs RESI:n {
   if DICE > RESI { result: 1 }
   else { result: 0 }
}

function: damage DICE:n vs RESI:n {
   loop R over {0..RESI}{
      if DICE > RESI { result: DICE - RESI } 
      else { result: 0}
   }
   if DICE > RESI { result: 1 }
   else { result: 0 }
}

output 1d{0,1,1,1,2,2}+1d{1,1,2,2,2,3}

output [hit 1d{0,1,1,1,2,2}+1d{1,1,2,2,2,3} vs 2]

output [damage 1d{0,1,1,1,2,2}+1d{1,1,2,2,2,3} vs 4]

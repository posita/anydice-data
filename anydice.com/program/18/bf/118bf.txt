function: luck ROLL:s {
 RES: 0

 if ([count {8,9,10} in ROLL] >= 1) {
   RES: RES +1
 } 

 if ([count {1,2,3} in ROLL] >= 1) {
   RES: RES -1
 }

 result: RES
}

output [luck 1d10]

output 2d{-1, -1, 0, 0, 1, 1}

function: even A:s and B:s {
  result: [count {2, 4, 6, 8, 10} in {A,B}]
}
output [even d10 and d10]

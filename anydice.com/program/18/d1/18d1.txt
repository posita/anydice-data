function: svgw R:n again N:n {
 if N = R { result: N + [svgw R again d10] }
 result: N
}

function: N:n R:n beats T:n {
 if (N + R) = 2 { result: -1}
 D: [highest of N and R]
 if D < T {result: 0}
 if D < (T*2) {result: 1}
 result: 2
}

loop I over {2..6} {
 D: I*2
 N: [svgw D again dD]
 R: [svgw 6 again d6]
 output [N R beats 4] named "d[D]"
}

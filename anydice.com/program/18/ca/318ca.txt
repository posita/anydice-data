CRITICALMISS: 0
MISS: 1
HIT: 2
CRITICALHIT: 3

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: CRITICALMISS }
 if ROLL = 20 { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

ATTACK: 5 + 3
ARMORCLASS: 15

output [attack d20 vs ARMORCLASS] named "Normal"
output [attack [highest 1 of 2d20] vs ARMORCLASS] named "Advantage"
output [attack [lowest 1 of 2d20] vs ARMORCLASS] named "Disadvantage"

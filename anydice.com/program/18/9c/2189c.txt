H: 6   \Hero die\
S: 6   \Skill die\
C: 6   \Career die\
E: 0   \Expertise die\
A: 0   \Advantage die\
TN: 4   \Target number for action dice\
D: 3    \Number of difficulty dice\

function: evaluate HERO:d and STAT:d and CAREER:d and EXPERTISE:d and ADVANTAGE:d vs DIFFICULTY:n {
  result: (HERO >= TN) + (STAT >= TN) + (CAREER >= TN) + (EXPERTISE >= TN) + (ADVANTAGE >= TN) - DIFFICULTYd(d4>=TN)
}

output [evaluate 1dH and 1dS and 1dC and 1dE and 1dA vs D]

function: attack ROLL:n MOD:n AC:n DAM:d CRITRNG:n CRITDAM:d {
  if ROLL = 1 { result: 0 }
  if ROLL >= CRITRNG { result: DAM + CRITDAM }
  if ROLL + MOD >= AC { result: DAM }
  result: 0
}

BASEDAM: 1d8+3
CRITDAM: 1d8
HITMOD: 5
loop AC over {5,10,15,20,25} {
output [attack 1d20 HITMOD AC BASEDAM 20 CRITDAM] named "Normal attack AC [AC]"
output [attack 1@2d20 HITMOD AC BASEDAM 20 CRITDAM] named "Advantage attack (vow of enmity) AC [AC]"
output [attack 1d20 HITMOD AC BASEDAM+2 19 CRITDAM] named "Hexblade's curse attack AC [AC]"
}

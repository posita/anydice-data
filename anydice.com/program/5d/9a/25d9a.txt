function: compare DIE:n TARGET:n {
  if DIE >= TARGET {
    result: 1
  }
  result: 0
}

function: roll MAG:n HIT:n {
  HITS: 0
  loop _ over {1..MAG} {
    HITROLL: 1d6
    HITS: HITS + [compare HITROLL HIT]
  }
  result: HITS
}

output [roll 3 4]
output [roll 4 4]
output [roll 8 4]
output [roll 12 4]

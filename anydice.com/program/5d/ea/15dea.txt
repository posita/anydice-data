AC: 15
PROFICIENCY: 3

function: attack DMG:d crit CRIT_DMG:d penalty PENALTY:n rider RIDER:n stat STAT:n superiority SUP:n style STYLE:n roll ROLL:n{
    MODIFIER: (STAT-10)/2
    DMG_WITH_BONUS: DMG + MODIFIER + RIDER
    
    if ROLL = 1 {
        result: 0
    }
    if ROLL = 20 {
        result: DMG_WITH_BONUS + CRIT_DMG
    }
    if ROLL + MODIFIER + PROFICIENCY + STYLE - PENALTY >= AC {
        result: DMG_WITH_BONUS + SUP
    }
    
    if ROLL + SUP + MODIFIER + PROFICIENCY + STYLE - PENALTY >= AC {
        result: DMG_WITH_BONUS
    }
    result: 0
}

function: dc DC:n damage D:d save SAVE:n r ROLL:n{
	if ROLL + SAVE < DC{
		result: D
	}
	if ROLL + SAVE >= DC{
                result: D/2
	}
}

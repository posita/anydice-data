function: survive ROLL:s {
  FAILS: 0
  SUCCESSES: 0

  loop N over {1..5} {
    if N@ROLL = 20 { SUCCESSES: SUCCESSES + 3}
    else if N@ROLL = 1 { FAILS: FAILS + 2 }
    else if N@ROLL >= 10 { SUCCESSES: SUCCESSES + 1 }
    else { FAILS: FAILS + 1 }
    
    if SUCCESSES >= 3 { result: N}
    if FAILS >= 3 { result: -N}
  }
  result: 0
}

output [survive 5d20]

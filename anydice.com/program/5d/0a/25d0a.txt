CHECKDIE: 6
SKILLDIE: 6
CHECKTN: 6
SKILLTN: 6
ATLEAST: 1

function: check ROLL:s {
   result: (ROLL >= CHECKTN) >= ATLEAST
}

function: skill ROLL:s {
  result: (ROLL >= SKILLTN) >= ATLEAST
}

function: test CHECKROLL:s {
  if [check CHECKROLL] = 1 {
    result: 1
  } 
  result: 0
}


function: test CHECKROLL:s with SKILLROLL:s {
  if 1@[check CHECKROLL] = 1 {
    result: 1
  } 
  if [skill SKILLROLL] = 1 {
    result: 1
  } 
  result: 0
}

\ loop N over {1..3} {
  loop I over {0..5} {
    output [test NdCHECKDIE with IdSKILLDIE] named "test [N]d[CHECKDIE] check with [I]d[SKILLDIE] skill"
  }
} \

loop N over {1..10} {
  \ output [check NdCHECKDIE] named "check [N]d[CHECKDIE]" \
  output [skill NdCHECKDIE] named "test [N]d[CHECKDIE]"
}

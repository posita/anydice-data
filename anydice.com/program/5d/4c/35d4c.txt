function: count VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..6VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: COUNT
}

output COUNT

function: count VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: COUNT
}

function: explode N:n {
 if N > 3 { result: N + [explode d6] }
 result: N
}

set "maximum function depth" to 5
output [explode d6]

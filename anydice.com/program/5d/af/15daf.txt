function: curtail DIE:s {
 T: {}
 loop N over DIE {
  if N < 1 {
   N: 1
  }
  if N > 10 {
   N: 10
  }
  T: {T, N}
 }
 result: d{T}
}

loop D over {4,6,8,10,12} {
 output [curtail 3dD-(D+1)] > 2d10/2 & [curtail 3dD-(D+1)] > (2d10+1)/2 named "Partner take token on d[D]"
 output [curtail 3dD-(D+1)] < 2d10/2 & [curtail 3dD-(D+1)] < (2d10+1)/2 named "Player take token on d[D]"
}

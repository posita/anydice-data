PSA: d{0,1,1,1,1,1}
PSB: d{0,0,0,0,1,1}
PSC: d{1,0,0,0,0,0}


function: energy D:d {
N: 4
J: NdD
result: [energy J N helper]
}

function: energy I:n M:n helper {
if I < M {
result: I+(M-I)dD
}
else {
result: I
}
}

output [energy PSA] named "(a) 에너지가 나올 경우"
output [energy PSB] named "(b) 원하는 에너지가 나올 경우"
output [energy PSC] named "(c) 사이드 킥이 나올 경우"

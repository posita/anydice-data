KEEP: 3

loop SKILL over {0..10}
{
     if(SKILL<5){
          DICEPOOL: 7-SKILL
          output [lowest 3 of (7-SKILL)d6] named "Rank [SKILL] (Lowest 3 of [DICEPOOL]d6)"
     }else{
          DICEPOOL: SKILL-1
          output [highest 3 of (SKILL-1)d6] named "Rank [SKILL] (Highest 3 of [DICEPOOL]d6)"
     }
}

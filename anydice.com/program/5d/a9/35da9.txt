\
 Blades in the Dark
 -1: Things go worse than you thought.
 0: Things go bad.
 1: Success, but with undesired consequences.
 2: Clean success.
 3: Critical success!
\

function: blades ROLL:s {
 if 1@ROLL = 6 | 1@ROLL = 5
 {
    if 2@ROLL = 6 | 2@ROLL = 5
    {
      result: 3
    }
  result: 2
 }
if 1@ROLL = 3 | 1@ROLL = 4
{
  result: 1
}
if 1@ROLL = 1 | 1@ROLL = 2
{
  result: -1
}
  result: 0
}

output [blades 2@2d6] named "0d"
loop P over {1..7} {
 output [blades Pd6] named "[P]d"
}

function: reroll helper DIER:n DIE:d N:n {
 if DIER < N { result: DIE}
 result: DIER
}

function: reroll DIE:d bellow N:n {
 DIENUMBER: #(DIE)
 SUM: 0
 loop I over {1..DIENUMBER}{
 SUM: SUM + [reroll helper I@DIE I@DIE N]
}
 result: SUM
}

output [reroll 4d6 bellow 3] named "4d6 reroll 1 and 2"

function: X check A:s {
 if 4@A >= X { result: 5 }
 if 3@A >= X { result: 4 }
 if 2@A >= X { result: 3 }
 if 1@A >= X { result: 1 }
 result: 0
}

function: X bonuscheck A:s {
 if 1+3@A >= X { result: 4 }
 if 1+2@A >= X { result: 3 }
 if 1+1@A >= X { result: 1 }
 result: 0
}

function: X checkwithfreesuccess A:s {
 if 3@A >= X { result: 5 }
 if 2@A >= X { result: 4 }
 if 1@A >= X { result: 3 }
 result: 1
}


loop N over {1..13}{
 output [N checkwithfreesuccess 3d12] named "free success plus [N] or more out of 3"
}

loop N over {1..13}{
 output [N check 4d12] named "[N] or more out of 4"
}

loop N over {1..13}{
 output [N bonuscheck 3d12] named "[N] or more with +1"
}

function: gwf A:n B:n C:n D:n {
  if B < 3 { 
    BB: 1dA 
  } else { 
    BB: B
  }
  if C < 3 { 
    CC: BB + 1dA
  } else { 
    CC: BB + C
  }
  if D < 3 { 
    result: CC + 1dA
  } else { 
    result: CC + D
  }
}

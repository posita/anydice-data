SIDESONDIE: 6

function: roll ROLL:s adding one for any extra sixes {
  NUMBEROFMAXRESULTS: [count SIDESONDIE in ROLL]
  result: 1@ROLL + NUMBEROFMAXRESULTS - (NUMBEROFMAXRESULTS>0)
}

}

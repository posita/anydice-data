\ output  2d6 + 5 \
\ output [highest of 2d6 and 2d6] + 5 \

loop AC over {10, 12, 15, 18, 20} {
  output (1d20 + 7 >= AC) * (2d6 + 5)  named "Normal AC[AC]"
  output (1d20 + 2 >= AC) * ([highest of 2d6 and 2d6] + 5 + 10) named "Savage AC[AC]"
  output ([highest of 1d20 and 1d20] + 7 >= AC) * (2d6 + 5) named "Lucky AC[AC]"
}

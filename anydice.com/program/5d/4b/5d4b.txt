function: check ACTIVE:n against PASSIVE:n {
  if ACTIVE > PASSIVE { result: 1  }
  if ACTIVE < PASSIVE { result: -1 }
  if ACTIVE = PASSIVE { result: 0  }
}

DICESTRING: {4, 6, 8, 10, 12}

loop ACTIVE over DICESTRING {
  output 0 named "~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~"
  loop PASSIVE over DICESTRING {
    output [check 1dACTIVE against 1dPASSIVE] named "1d[ACTIVE] against 1d[PASSIVE] (1 = win, -1 = loss)"
  }
}

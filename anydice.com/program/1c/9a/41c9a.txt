function: truecheck R:n B:n DC:n{
X: 0
if R + B >= DC {X: 1}
if (R + B >= DC + 10) + (R = 20) {X: X + 1}
if (R + B <= DC - 10) + (R = 1) {X: X - 1}
result: X
}
function: check B:n DC:n {result: [truecheck 1d20 B DC]}
function: rita P:n S:n DC:n{
A: [check S DC]
result: [highest of ([check (P + (2 * (A = 2)) - (4 * (A = 0))) (DC+5)] - (A = -1)) and -1]
}
function: attack DC:n{result: [highest of [check 0 DC] and 0]}
loop P over {19,22,24,27}{
output [rita 21 21 P]
}

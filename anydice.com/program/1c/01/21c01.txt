function:check ROLL:n XROLL:n w BONUS:n vs DIFFICULTY:n {
 C: ROLL
 if ROLL = 1 {
  C: ROLL - XROLL 
 }
 if ROLL = 10 {
  C: ROLL + XROLL 
 }
 result: C + BONUS > DIFFICULTY
}

loop BONUS over {1..12}{
  loop DIFF over {13,15,16,17,20,25,30}{
    output [check 1d10 1d10 w BONUS vs DIFF] named "Bonus [BONUS] vs Difficulty [DIFF]"
  }
}

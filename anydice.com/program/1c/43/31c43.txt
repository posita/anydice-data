AC:25
ATK:22+2
MIN:[lowest of [highest of AC-ATK-2 and 1] and 19]
HIT:[highest of 19-MIN and 0]
CRITCHANCE:{0:MIN,1:HIT,2}
function: N:n punches{
X:0
loop L over {1..N}{
X:X+[punch damage 1dPUNCH]
}
result:X
}
function: adv N:n punches{
X:0
loop L over {1..N}{
X:X+[punch damage [highest 1 of 2dPUNCH]]
}
result:X
}
function: dis N:n punches{
X:0
loop L over {1..N}{
X:X+[punch damage [lowest 1 of 2dPUNCH]]
}
result:X
}
function: punch damage D:n{

if D=2{result: 6d10+10+4d6}
result: 0
}
loop N over {16}{
output [N punches] named "[N] Punches"
}

AW: -1
AH: 3
BW: -1
BH: 3
N: 2
function: what W:n how H:n{
  ABW: [absolute W]
  if W < 0{
    R: [lowest N of (ABW+N)d6]+H
  } else {
    R: [highest N of (ABW+N)d6]+H
  }
  result: R   
}
WHA: [what AW how AH]
WHB: [what BW how BH]
output WHA named "w[AW]h[AH]"
output WHB named "w[BW]h[BH]"
output WHA >= WHB named "w[AW]h[AH] vs w[BW]h[BH]"

\**** Damage calculator normal Vs advantage ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d10 + ATTACKMOD + DMGBONUS + 1d4 + ATTACKMOD + 1d6 
CRITDAMAGE: 2d10 + ATTACKMOD + DMGBONUS + 2d4 + ATTACKMOD + 2d6
output ATTACKTIMESd[attackroll [highest 1 of 1d20] vs ENEMYAC] named "no advantage w/hex "

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 2d10 + 2*ATTACKMOD + DMGBONUS + 1d4 + ATTACKMOD
CRITDAMAGE: 2d10 + ATTACKMOD + DMGBONUS + 2d4 + ATTACKMOD
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "advantage w/o hex"

function: attack ROLL:n BONUS:n AC:n DMG:d {
  if ROLL + BONUS >= AC { result: DMG }
  result: 0
}

function: attack with advantage ROLL:d BONUS:n AC:n DMG:d {
  result: [attack [highest 1 of ROLL] BONUS AC DMG]
}

output [attack d20 5 13 1d10 + 5] + [attack d20 5 13 1d10 + 5] named "Scenario 1: Two Normal Attacks"

output [attack with advantage 2d20 5 13 1d10 + 5] named "Scenario 2: True Strike"

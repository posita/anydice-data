
function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

set "explode depth" to 2
output [explode [reroll {1,2} on d4 to d4]] + [explode [reroll {1,2} on d4 to d4]] + [explode [reroll {1,2} on d4 to d4]]
output [explode [reroll {1,2} on d6 to d6]] + [explode [reroll {1,2} on d6 to d6]]
output [explode [reroll {1,2} on d12 to d12]]

TARGET_AC: {20}

STR: 5 
PROF: 4
RAGE: 3
WEAPON_DIE: 6
WEAPON_DIE_AMOUNT: 2
CRIT: 2 * WEAPON_DIE_AMOUNT
WEAPON_BONUS: 3
FELL_HANDED_A: 1
SAVAGE_ATTACKS: 1
BRUTAL_CRIT: 2
BRUTAL_FIST_DIE: 4

VOID_BLADE_ATTACK: {FELL_HANDED_A + STR + PROF + WEAPON_BONUS}

BRUTAL_FIST_ROLL: 1d{{1..3}, {BRUTAL_FIST_DIE}:3}
BRUTAL_FIST_ROLL_CRIT: CRITd{{1..3}, {BRUTAL_FIST_DIE}:3}
VOID_BLADE_ROLL: WEAPON_DIE_AMOUNTd{{1..3}, {4..WEAPON_DIE}:WEAPON_DIE / 2 + 1}
VOID_BLADE_ROLL_CRIT: {CRIT + SAVAGE_ATTACKS + BRUTAL_CRIT}d{{1..3}, {4..WEAPON_DIE}:WEAPON_DIE / 2 + 1}

GREAT_WEAPON_MASTER_A: -5
GREAT_WEAPON_MASTER_D: 10

VOID_BLADE_DAMAGE: VOID_BLADE_ROLL + BRUTAL_FIST_ROLL + STR + WEAPON_BONUS
VOID_BLADE_DAMAGE_CRIT: VOID_BLADE_ROLL_CRIT + BRUTAL_FIST_ROLL_CRIT + STR + WEAPON_BONUS

MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{TARGET_AC}

ATTACK: VOID_BLADE_ATTACK
HITDAMAGE: VOID_BLADE_DAMAGE
CRITICALDAMAGE: VOID_BLADE_DAMAGE_CRIT
MISSDAMAGE: 0
output [attack] named "VB"

ATTACK: VOID_BLADE_ATTACK + GREAT_WEAPON_MASTER_A
HITDAMAGE: VOID_BLADE_DAMAGE + GREAT_WEAPON_MASTER_D
CRITICALDAMAGE: VOID_BLADE_DAMAGE_CRIT + GREAT_WEAPON_MASTER_D
output [attack] named "VB GWM"

ATTACK: VOID_BLADE_ATTACK
HITDAMAGE: VOID_BLADE_DAMAGE + RAGE
CRITICALDAMAGE: VOID_BLADE_DAMAGE_CRIT + RAGE
output [attack] named "VB Rg"

ATTACK: VOID_BLADE_ATTACK + GREAT_WEAPON_MASTER_A
HITDAMAGE: VOID_BLADE_DAMAGE + GREAT_WEAPON_MASTER_D + RAGE
CRITICALDAMAGE: VOID_BLADE_DAMAGE_CRIT + GREAT_WEAPON_MASTER_D + RAGE
output [attack] named "VB Rg GWM"

function: attack {
 result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]
}

ATTACK: VOID_BLADE_ATTACK
HITDAMAGE: VOID_BLADE_DAMAGE
CRITICALDAMAGE: VOID_BLADE_DAMAGE_CRIT
output [attack] named "Adv VB"

ATTACK: VOID_BLADE_ATTACK + GREAT_WEAPON_MASTER_A
HITDAMAGE: VOID_BLADE_DAMAGE + GREAT_WEAPON_MASTER_D
CRITICALDAMAGE: VOID_BLADE_DAMAGE_CRIT + GREAT_WEAPON_MASTER_D
output [attack] named "Adv VB GWM"

ATTACK: VOID_BLADE_ATTACK
HITDAMAGE: VOID_BLADE_DAMAGE + RAGE
CRITICALDAMAGE: VOID_BLADE_DAMAGE_CRIT + RAGE
output [attack] named "Adv VB Rg"

ATTACK: VOID_BLADE_ATTACK + GREAT_WEAPON_MASTER_A
HITDAMAGE: VOID_BLADE_DAMAGE + GREAT_WEAPON_MASTER_D + RAGE
CRITICALDAMAGE: VOID_BLADE_DAMAGE_CRIT + GREAT_WEAPON_MASTER_D + RAGE
output [attack] named "Adv VB Rg GWM"

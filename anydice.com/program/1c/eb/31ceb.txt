MAGIC: {0, 1, 1, 2, 2, 3, 3, 4}
PHYSI: {0, 1, 1, 2, 2, 3}

function: criticals ROLL:s {
 result: ROLL = 4
}

function: continuous ROLL:s {
 result: ROLL = 0
}

loop DICE over {1..3} {
 output [criticals DICE dMAGIC] named "[DICE]d Magical Criticals"
}

loop DICE over {1..6} {
 output [continuous DICE dPHYSI] named "[DICE]d Physical Continuous"
}

loop DICE over {1..3} {
 output [continuous DICE dMAGIC] named "[DICE]d Magical Continuous"
}

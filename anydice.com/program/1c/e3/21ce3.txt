function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d10] }
 if N = 8 { result: 1 }
 if N = 6 { result: 1 }
if N = 4 { result: 1 }
if N = 2 { result: 1 }
if N = 1 { result: -1 }
 result: 0
}

NWOD: [nwod 10 again d10]

loop N over {1..10} {
 output [lowest of 10 and NdNWOD] named "[N]d"
}

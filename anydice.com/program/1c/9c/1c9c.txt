function: match VALUES:s in A:s and B:s and C:s {
COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = {A,B,C})
 }
 result: COUNT
}

loop N over {-2,-1,0,1,2,3,9}{
 output 3d(d{0,-2,1,1,2,3}=N) named "Roll like [N]"
 output [match N in d{0,-2,-2,1,3,3} and d{0,-2,1,1,2,2} and d{0,1,1,1,2,3}] named "Roll diff [N]"
}

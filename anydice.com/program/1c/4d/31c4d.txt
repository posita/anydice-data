\ FUNCIONES \
function: advantage A{
  result: [highest of A and A]
}

function: hit A:n against enemy B:n with damage C:n{
 if A > B { result: A - B + C}
 result: 0
}

function: hit A:n against boss B:n with damage C:n{
 if A > B { result: C}
 result: 0
}

\ DADOS \
FATE:4d{-1..1}
ADVANTAGE_FATE:[advantage FATE]


\ VARIABLES \
ATTACK:5
DAMAGE:4
DEFENSE:12
FORCE:3

\ Resultados de ataques \
\output FATE+ATTACK named "Normal"\
\output ADVANTAGE_FATE+ATTACK named "Ventaja"\
\output FATE+ATTACK+FORCE named "Fuerza"\
\output ADVANTAGE_FATE+ATTACK+FORCE named "Fuerza y Ventaja"\


\ Resultados contra enemigo \

\ Enemigo simple \
\output [hit FATE+ATTACK against enemy FATE+DEFENSE with damage DAMAGE] named "Normal against enemy"\
\output [hit ADVANTAGE_FATE+ATTACK against enemy FATE+DEFENSE with damage DAMAGE] named "Advantage against enemy"\
\output [hit FATE+ATTACK+FORCE against enemy FATE+DEFENSE with damage DAMAGE] named "Force against enemy"\
\output [hit ADVANTAGE_FATE+ATTACK+FORCE against enemy FATE+DEFENSE with damage DAMAGE] named "Advantage and force against enemy"\


\ Enemigo boss \
\output [hit FATE+ATTACK against boss FATE+DEFENSE with damage DAMAGE] named "Normal against enemy"\
\output [hit ADVANTAGE_FATE+ATTACK against boss FATE+DEFENSE with damage DAMAGE] named "Advantage against enemy"\
\output [hit FATE+ATTACK+FORCE against boss FATE+DEFENSE with damage DAMAGE] named "Force against enemy"\
output [hit ADVANTAGE_FATE+ATTACK+FORCE against boss FATE+DEFENSE with damage DAMAGE] named "Advantage and force against enemy"

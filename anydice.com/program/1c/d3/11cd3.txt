\--- Variables ---\
D: 5
B: 3
PD: 0
FLATMOD: 0
TN: 10
DOSVALUE: 5
FINALMOD: 10

\--- DoS Function ---\
DIE: [highest (D-PD) of (D+B)d6]+FLATMOD
function: dos ROLL:n TN:n {
 if ROLL <= TN-5 {
    result: -1
 }
 if ROLL > TN-5 & ROLL < TN {
    result: 0
 }
 if ROLL >= TN & ROLL < TN+5 {
    result: 1
 }
 if ROLL >= TN+5 & ROLL < TN+10 {
    result: 2
 }
 if ROLL >= TN+10 & ROLL < TN+15 {
    result: 3
 }
 if ROLL >= TN+15 {
    result: 4
 }
}

output ([dos DIE TN]*DOSVALUE)-FINALMOD

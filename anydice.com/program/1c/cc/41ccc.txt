set "maximum function depth" to 8
KARMA: 3

function: blowup ROLLED:n SIZE:n on THRESH:n {
if ROLLED >= THRESH {
  if SIZE = 20 {
    result: ROLLED + [blowup d20 20 on 20]
  }
  if SIZE = 12 {
    result: ROLLED + [blowup d(SIZE+8) (SIZE+8) on (THRESH+8)]
  }
  result: ROLLED + [blowup d(SIZE+2) (SIZE+2) on (THRESH+2)]
 }
result: ROLLED
}

function: blowupkarma ROLLED:n SIZE:n on THRESH:n karma K:n {
if ROLLED >= (THRESH - K) {
  if SIZE = 20 {
    result: SIZE + [blowupkarma d20 20 on 20 karma (ROLLED - THRESH + K)]
  }
  if SIZE = 12 {
    result: SIZE + [blowupkarma d(SIZE+8) (SIZE+8) on (THRESH+8) karma (ROLLED - THRESH + K)]
  }
  result: SIZE + [blowupkarma d(SIZE+2) (SIZE+2) on (THRESH+2) karma (ROLLED - THRESH + K)]
 }
result: ROLLED
}

function: blowkarma ROLLED:n SIZE:n on THRESH:n karma K:n {
if ROLLED >= (THRESH - K) {
  result: SIZE + [blowkarma d(SIZE) SIZE on THRESH karma (ROLLED - THRESH + K)]
 }
result: ROLLED + K
}

output [blowupkarma d2 2 on 2 karma KARMA] named "2s"

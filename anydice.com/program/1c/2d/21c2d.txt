MISS: 0
HIT: 1
CRITICALHIT: 2
ATTACK: 3+6+5
HITDAMAGE: 1d8+3+6+5+5
CRITICALDAMAGE: 2d8+3+6+5+5
MISSDAMAGE: 0
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= 19 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}
function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}
function: attackadv {
 result: [damage for [attack 1@2d20 vs DEFENSE]]
}

DEFENSE: d{12..22}

PB: 6
CHA: 5
DEX: 3
PW: 3
LD: 5

ATTACK: PB + CHA + PW
HITDAMAGE: d8 + ATTACK + LD
CRITICALDAMAGE: 2d8+ATTACK
MISSDAMAGE: 0
output [attack] named "Pact Weapon"

ATTACK: PB+DEX
HITDAMAGE: 4d8+ATTACK
CRITICALDAMAGE: 8d8+ATTACK
output [attackadv] named "Shadowblade"

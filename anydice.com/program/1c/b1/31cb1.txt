function: comparison A:s B:s {
  R: -10
  loop N over {1..[lowest of #A and #B]} {
   if (N@A > N@B) { result: 1 }
   if (N@B > N@A) { result: -1 }
   if (N@A = N@B) { R: 0 }
  }
  result: R
}

output [comparison 3d6 1d6]
output [comparison 3d6 2d6]
output [comparison 3d6 3d6]
output [comparison 3d6 4d6]

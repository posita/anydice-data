function: doubles FIRST:n SECOND:n {
  if FIRST = SECOND {result: 1}
  result: 0
}
function: critsuccess FIRST:n SECOND:n {
  if FIRST = SECOND | (FIRST + SECOND) >= 18 {result: 1}
  result: 0
}
function: tens FIRST:n SECOND:n {
  if FIRST = 10 | SECOND = 10 {result: 1}
  result:0
}

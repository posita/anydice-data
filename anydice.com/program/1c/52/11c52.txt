function: DIE:n plus MOD:n hits AC:n{
  if(DIE = 20){ result: 2 }
  if(DIE+MOD >= AC+10){ result: 2 }
  result: DIE+MOD >= AC
}

function: damage DIE:d plus MOD:n on ATTACK:n plus CRITDIE:d on crit{
  if(ATTACK = 2){ result: 2*(DIE+MOD) + CRITDIE }
  else{ result: (DIE+MOD) * ATTACK }
}

function: restrict ROLL:n to RANGE:s else REROLL:d {
  if ROLL = RANGE { result: ROLL }
  else { result: REROLL }
}
function: restrict ROLL:d to RANGE:s {
  loop I over {1..20} {
    ROLL: [restrict ROLL to RANGE else ROLL]
  }
  result: ROLL
}

loop P over {0..20} {
F: {-1,0,1}
A: P+4
BASE: AdF
SPEC: {0}
LL: [highest 4 of (P+4)dF]
output LL+ [restrict [count 1 in BASE]dSPEC to {0}]+P named "[A]dF + [P]"
}

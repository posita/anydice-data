\Compares all Aptitude die d4, d6, d8, d10, d12, d20, d100 to Skill Level\
\Change SKILL: to whatever number you wish to compare all die to (0-10)\
\Change DEPTH: to however many times you want the Aptitude die to potentially Explode (1-10)\

loop N over {4, 6, 8, 10, 12, 20, 100}{
SKILL: 1
DEPTH: 1
 set "explode depth" to DEPTH
output [explode dN] + d20 + SKILL named "d[N] and Skill of [SKILL] Exploded at most [DEPTH] time(s)"
}

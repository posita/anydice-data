STAT: 8
SKILL: 6
OTHER: 4
DV: 17

BASE: STAT + SKILL + OTHER

function: redroll BASE:n DICE:s {
  if DICE=10 { result: BASE + DICE + 1d10}
  if DICE=1 { result: BASE + DICE - 1d10}
  result: BASE + DICE
}

output [redroll BASE 1d10] named "Distribution"

output [redroll BASE 1d10] >= DV named "Does it beat DV"

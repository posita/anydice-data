function: successes DIE:n BONUS:n BREAKPOINTS:s {

X: DIE + BONUS

if X < 1@BREAKPOINTS{
  RESULT : -1
}

if X >= 1@BREAKPOINTS {
  RESULT : 0
}

if X >= 2@BREAKPOINTS {
  RESULT : 1
}
if X >= 3@BREAKPOINTS {
  RESULT : 2
}
 result: RESULT
}

TWO: {6,11,16}
NEGTWO: -2
NEGONE: -1

output [successes 2d10 NEGTWO TWO] named "2d10 - 2"
output [successes 2d10 NEGONE TWO] named "2d10 - 1"
output [successes 2d10 0 TWO] named "2d10"
output [successes 2d10 1 TWO] named "2d10 + 1"
output [successes 2d10 2 TWO] named "2d10 + 2"
output [successes 2d10 3 TWO] named "2d10 + 3"
output [successes 2d10 4 TWO] named "2d10 + 4"
output [successes 2d10 5 TWO] named "2d10 + 5"
output [successes 2d10 6 TWO] named "2d10 + 6"

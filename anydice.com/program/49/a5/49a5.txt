function: matching dice on VALUES:s in ROLL:s {
  DUBS: 0
  NEWVAL: 0
  loop P over VALUES {
   NEWVAL: [count P in ROLL]
   if NEWVAL > 1 {
    DUBS: DUBS + NEWVAL - 1
   }
  }
  result: DUBS
}

output d6+1 named "1a) Standard GUMSHOE d6+1"
output [highest 1 of 2d6] named "1b) Choose best die of 2d6"
output [highest 1 of 2d6]+[matching dice on {1..6} in 2d6] named "1c) best die of 2d6, +1 for each matching die"

output d6+2 named "2a) Standard GUMSHOE d6+2"
output [highest 1 of 3d6] named "2b) Choose best die of 3d6"
output [highest 1 of 3d6]+[matching dice on {1..6} in 3d6] named "2c) best die of 3d6, +1 for each matching die"

function: attackdamage ATTACKROLL:n ATTACKMOD:n AC:n DAMAGE_DICE:d DAMAGE_MOD:n  {
  if (ATTACKROLL = 20) { result: DAMAGE_DICE + DAMAGE_DICE + DAMAGE_MOD }
  else if (ATTACKROLL = 1) { result: 0 }
  else if ((ATTACKROLL + ATTACKMOD) >= AC) { result: DAMAGE_DICE + DAMAGE_MOD }
  else { result: 0 }
}

function: save SAVE:n SAVEROLL:n ROLLMOD:n DAMAGE_DICE:d FAIL_DAMAGE_DICE:d {
  if (SAVEROLL + ROLLMOD) >= SAVE { result: DAMAGE_DICE }
  else { result: FAIL_DAMAGE_DICE }
}

ROLL: 10

function: VALUE:s replace MIN:n {
if VALUE <= MIN {result: 0}
else {result: VALUE}
}

DISADVANTAGE: -2
NOADVANTAGE: 1
ADVANTAGE: 2
ELVENACCURACY: 3

function: REPEAT attacks at TOHIT:n vs DC:n with ADV:n for DDICE:d plus DMODS:n critical on CRIT:n {
TOTAL: 0
loop N over {1..REPEAT}{
if ADV > 0 {
  SIDES: 20^ADV
  CRITS: SIDES-19^ADV
  HITS: SIDES-(DC-TOHIT)^ADV-CRITS
  MISSES: SIDES-CRITS-HITS
  } else {
  SIDES: 20^(0-ADV)
  MISSES: SIDES-(DC-TOHIT)^(0-ADV)
  CRITS: 0-ADV
  HITS: SIDES-MISSES-CRITS
  }
CHECKSUM: SIDES-CRITS-HITS-MISSES
DAMAGE: [(1d{0:MISSES,1:HITS,2:CRITS})dDDICE+DMODS replace DMODS]
TOTAL: TOTAL + DAMAGE
}
result: TOTAL
}

function: attack A B C D E F G {
result: [A attacks at B vs C with D for E plus F critical on G]
}

output [1 attacks at +7 vs 15 with NOADVANTAGE for 1d8 plus 3 critical on 20] named "example rapier"

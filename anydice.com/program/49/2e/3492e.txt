# Define the number of sides on the die
sides = 6

# Define the total number of outcomes
total_outcomes = sides ** 2

# Define the events
event_A_outcomes = 0
event_B_outcomes = 0

# Iterate through all possible outcomes of rolling two dice
for roll1 in range(1, sides + 1):
    for roll2 in range(1, sides + 1):
        # Compute the sum of the two rolls
        roll_sum = roll1 + roll2
        # Check Event A
        if roll_sum > 9:
            event_A_outcomes += 1
        # Check Event B
        if roll_sum % 4 == 0 or roll_sum % 6 == 0:
            event_B_outcomes += 1

# Calculate the probabilities
probability_event_A = event_A_outcomes / total_outcomes
probability_event_B = event_B_outcomes / total_outcomes

# Print the results rounded to two decimal places
print("Probability of Event A (sum > 9): {:.2f}".format(probability_event_A))
print("Probability of Event B (sum divisible by 4 or 6): {:.2f}".format(probability_event_B))

stat STAT:n

roll ROLL:n

against dc POISON_SAVE_DC:n {
    MODIFIER: STAT-10/2
    DMG_WITH_BONUS: DMG + MODIFIER + RIDER

    }

        result: DMG_WITH_BONUS+POISON_DAMAGE

}

function: interpret ROLL:s SKILL:n ABILITY:n {
  S: [sort ROLL]
  SKILLT: SKILL
  RES: 1
  if 2@ROLL - SKILL > ABILITY { RES: 0 }
    else { if {2,3}@ROLL - SKILL > ABILITY * 2 { RES: 0 } }
  result: RES
}

loop SKILL over {0..5} {
  output [interpret 3d20 SKILL 10] named "Odds for skill + [SKILL]"
}

function: matching dice on VALUES:s in ROLL:s {
  DUBS: 0
  NEWVAL: 0
  loop P over VALUES {
   NEWVAL: [count P in ROLL]
   if NEWVAL > 1 {
    DUBS: DUBS + NEWVAL - 1
   }
  }
  result: DUBS
}

output d6+1 named "Standard +1"
output [highest 1 of 2d6] named "House 2d6"
output [highest 1 of 2d6]+[matching dice on {1..6} in 2d6] named "House 2d6+#doubles"

output d6+2 named "Standard +2"
output [highest 1 of 2d6] named "House 3d6"
output [highest 1 of 2d6]+[matching dice on {1..6} in 3d6] named "House 3d6+#doubles"

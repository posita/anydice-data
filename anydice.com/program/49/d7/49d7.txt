\ Savage Attacker Feat \
function: savage DMG:d {
 result: [highest of d12 and d12]
}


function: attack ATT:n crange CRNG:n cmod CMOD:n dmg DMG:d ac AC:n{
 if ATT < AC { result: 0 }
 DMG_RESULT: DMG
 if ATT >= CRNG {
  loop P over {2..CMOD} {
   DMG_RESULT: DMG_RESULT+DMG
  }
 }
 result: DMG
}

loop AC over {2..25} {
  loop ATTACK over {1..20} {
    output [attack ATTACK crange 20 cmod 2 dmg d12 ac AC] named "Greatsword"
    output [attack ATTACK crange 20 cmod 2 dmg [savage d12] ac AC] named "Greatsword with Savage Attacker"
    output [attack ATTACK crange 20 cmod 2 dmg d8 ac AC] named "Longsword"
    output [attack ATTACK crange 19 cmod 2 dmg d10 ac AC] named "Bastard Sword"
    output [attack ATTACK crange 19 cmod 2 dmg [savage d10] ac AC] named "Bastard Sword with Savage Attacker"
  }
}

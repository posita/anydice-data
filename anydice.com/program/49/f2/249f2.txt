function: roll D:n plus N:n {
  if D+N >= 1 { result: 2 }
  if D+N = 0 { result: 1 }
  result: 0
}

function: explofate D:n M:n {
  if D <= -M { result: [explofate 1d{0,0,-1} 1] + D }
  if D >=  M { result: [explofate 1d{0,0, 1} 1] + D }
  result: D
}

output [explofate 1d{-1..1} 1]

loop N over {-5..5} {
output [roll [explofate 1d{-1..1} 1] plus N] named "Explofate: [N]"
}

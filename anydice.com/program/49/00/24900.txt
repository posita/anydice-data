function: evaluate RED:s WHITE:s and GREEN:s for SEUIL:n {
 result: (WHITE >= SEUIL) + (WHITE >= SEUIL) + (WHITE >= SEUIL)
 \loop P in {WHITE, RED, GREEN}
 {
  if P >= SEUIL { result: result+1}
 }\
}

loop BASE over {4..4} {
   loop BONUS over {1..1} {
      loop MALUS over {1..1} {
         WHITE : 1 + BASE - BONUS - MALUS
         loop SEUIL over {6..8} {
      output [evaluate MALUS d8 WHITE d10 and BONUS d12 for SEUIL ] named "[BASE] base, [BONUS] bonus, [MALUS] malus, SR [SEUIL]"
}}}}

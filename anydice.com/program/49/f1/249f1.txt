function: monohackinner D:n M:n {
  \ Reroll 1dF if the result is +/- the maximum \
  \ but count only matching symbols \
  if D <= -M { result: [monohackinner 1d{0,0,-1} 1] + D }
  if D >=  M { result: [monohackinner 1d{0,0, 1} 1] + D }
  result: D
}

function: monohack {
  \ Basic roll of 2dF, with a max result of 2 \
  F: {-1..1}
  result: [monohackinner 2dF 2]
}

output [monohack]

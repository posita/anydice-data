function:oppose A:n to B:n{
 if A-B<0 {result:0 } result:A-B
}
function:limit A:n to B:n{
 if A>B {result:B} result:A
}
function:roll A B:n {
 if B=0{result:d(A)} if B<0 {result:(1+(B*-1))@(1+(B*-1))d(A)} result:1@(1+B)d(A)
}

function: delta A:n against S:n {
  \ Failed roll. Return a negative. \
  if A > S {result: -1}

  \ Successful roll.  Return the delta. \
  result: S-A
}

function: compare A:n and B:n {
  \ Player 1 has failed \
  if A < 0 {
    if B < 0 {result: 3}
    result: 2
  }

  \ Player 2 has failed \
  if B < 0 {
    if A < 0 {result: 3}
    result: 1
  }

  \ Neither player failed. Compare their deltas \
  if A < B {result: 1}
  result: 2
}

output [compare [delta d100 against 50] and [delta d100 against 70]]

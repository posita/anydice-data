function: check A:n vs B:n {
 if (A >= B) {result: 1}
 result: 0
}

function: hitchance N tohitv AC {
 result: [check 1d20+N vs AC]
}

function: hitchanceadv N tohitv AC{
 result: [check ([highest 1 of 2d20]+N) vs AC]
}

\-output [hitchance 6tohitv 20] named "+6 hitchance vs AC 20"
output [hitchanceadv6tohitv20] named "+6 hitchance w/ adv. vs AC 20"
-\
function: N tohit DMG:n dmgperturnv AC{
 result: DMG*[hitchance N tohitv AC]
}
function: N advtohit DMG:n dmgperturnv AC{
 result: DMG*[hitchanceadv N tohitv AC]
}

output [6 tohit 2d6+4 dmgperturnv 20] named "+6, 2d6+4 dmg/t vs AC 20"
output [6 advtohit 8d6 dmgperturnv 20] named "adv +6, 8d6 dmg/t vs AC 20"

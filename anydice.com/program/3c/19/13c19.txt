PROF: 3
ATK:  5
DIE:  1d12
MOD:  0
AMOD: -5
DMOD: 12
ADV:  1
AC:   17
function: hit R:n vs T:n{
 if R = 20 {result: DIE + DIE + ATK + MOD + DMOD}
 if R + PROF + ATK + MOD + AMOD >= T & R != 1 {result: DIE + ATK + MOD + DMOD}
 result: 0
}
function: atk T:n{
 if ADV = 1 {result: [hit [highest 1 of 2d20] vs T]}
 if ADV = -1 {result: [hit [lowest 1 of 2d20] vs T]}
 result: [hit d20 vs T]
}
output [atk AC] + [atk AC]

\ attack result definitions \
MISS: 0
HIT: 1
CRITICAL_HIT: 2

\ attack result vs defense with multiple attack penalty \
function: attack ROLL:n vs DEFENSE:n with MAP:n {
 if ROLL= 1 { result: MISS }
 if ROLL + ATTACK_MOD + MAP >= DEFENSE + 10 {
   result: CRITICAL_HIT
 }
 if ROLL = 20 {
  if ROLL + ATTACK_MOD + MAP >= DEFENSE { result: CRITICAL_HIT }
  result: HIT
 }

 if ROLL + ATTACK_MOD + MAP >= DEFENSE { result: HIT }
 result: MISS
}



\ damage result \
function: damage for ATTACK_RESULT:n {
 if ATTACK_RESULT= CRITICAL_HIT {
  result: CRITICAL_DAMAGE
 }
 if ATTACK_RESULT= HIT {
  result: HIT_DAMAGE
 }
 result: MISS_DAMAGE
}

\ damage result double slice \
function: double slice damage for ATTACK_RESULT_ONE:n and ATTACK_RESULT_TWO:n {
 DAMAGE: [damage for ATTACK_RESULT_ONE] + [ATTACK_RESULT_TWO]
 if ATTACK_RESULT_ONE= CRITICAL_HIT & ATTACK_RESULT_TWO= CRITICAL_HIT {
  DAMAGE: DAMAGE - SNEAK_DAMAGE
 }
 result: DAMAGE
}



\ define AC range of monster \
MONSTER_AC_MIN: 27
MONSTER_AC_MAX: 29
MONSTER_AC: 29
MONSTER_MAX_HP: 155




\MAGUS FIGHTER (inexorable iron 2H)\

\ define attack modifier \
ATTACK_ATTRIBUTE_BONUS: +4
ATTACK_PROFICIENCY_BONUS: +13
ATTACK_ITEM_BONUS: +1
ATTACK_MOD: ATTACK_ATTRIBUTE_BONUS + ATTACK_PROFICIENCY_BONUS + ATTACK_ITEM_BONUS

\ define physical damage \
DAMAGE_WEAPON_SPECIALIZATION: +3
DAMAGE_ATTRIBUTE_BONUS: +4
STRIKING_RUNE: +1
WEAPON_DAMAGE: (1+STRIKING_RUNE)d12
PHYSICAL_DAMAGE: WEAPON_DAMAGE + DAMAGE_ATTRIBUTE_BONUS + DAMAGE_WEAPON_SPECIALIZATION

\ define spell damage \
SPELL_LEVEL: 4
SPELL_DAMAGE: (1+SPELL_LEVEL)d6

\ define spell_strike damage \
SPELL_STRIKE_DAMAGE: PHYSICAL_DAMAGE + SPELL_DAMAGE




\ calculate first action damage \
HIT_DAMAGE: SPELL_STRIKE_DAMAGE
CRITICAL_DAMAGE: 2*HIT_DAMAGE
MISS_DAMAGE: 0
ATTACK_MAP: -0

FIRST_ACTION_DAMAGE: [damage for [attack 1d20 vs MONSTER_AC with ATTACK_MAP]]

\ calculate second action damage \
HIT_DAMAGE: PHYSICAL_DAMAGE
CRITICAL_DAMAGE: 2*HIT_DAMAGE
MISS_DAMAGE: 0
ATTACK_MAP: -10
SECOND_ACTION_DAMAGE: [damage for [attack 1d20 vs MONSTER_AC with ATTACK_MAP]]

TOTAL_TURN_DAMAGE: FIRST_ACTION_DAMAGE + SECOND_ACTION_DAMAGE
output TOTAL_TURN_DAMAGE named "M/F inexiron vs [MONSTER_AC]"

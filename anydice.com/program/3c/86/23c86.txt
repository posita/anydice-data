function: savethrow for DC:n and MOD:n {
result: 1d20+MOD > DC
}

function: dissave for DC:n and MOD:n {
result: [lowest 1 of 2d20]+MOD >= DC
}

function: advsave for DC:n and MOD:n {
result: [highest 1 of 2d20]+MOD >= DC
}

function: survive DAMAGE:n with SAVE:n {
if SAVE = 0 {
 result: HEALTH >= DAMAGE }
else {
 result: HEALTH >= (DAMAGE/2) }
}

HEALTH: 2+3d8+7
SAVE: [advsave for 14 and 1]
output [survive 4d8 with SAVE] named "Chances of Shatter Survival"
output [survive 8d6 with SAVE] named "Chances of Fireball Survival"

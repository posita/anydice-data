function: customroll D:n accum A:n
{
  if D = 6 { result: [customroll 1dDRDICE accum D+A-3] }
  else { result: D + A }
}

DRDICE: {0, 1, 2, 2, 3, 6}

set "maximum function depth" to 5
output [customroll 1dDRDICE accum 0] named "Single Roll Result"

SUM: 0
loop N over {1..4}{
 SUM: SUM + [customroll 1dDRDICE accum 0]
}
output SUM named "4 Rolls Result"
output 4d[customroll 1dDRDICE accum 0] named "another 4 Rolls Result"

function: fifth edition attack ROLL:n AC:n DMG_DICE:d DMG_MOD:n ATTACK_BONUS:n {
if ROLL = 1  {result: 0}
if ROLL >= 20 {result: 2*DMG_DICE + DMG_MOD}
if ROLL + ATTACK_BONUS >= AC {result: DMG_DICE + DMG_MOD}
result: 0}

loop AC over {8..23} {
 output 2*[fifth edition attack d20 AC 1d8 2 7] named "[AC] AC, Longbow +1"
 output [fifth edition attack d20 AC 4d6 0 6] named "[AC] AC, Guiding Bolt"
}

function: fifth edition save ROLL:n DC:n SAVE_MOD:n DMG_DICE:d {
if ROLL+SAVE_MOD >= DC {result: 0}
result: DMG_DICE}

loop SAVE_MOD over {-2..10} {
 DC: 14
 output [fifth edition save d20 DC SAVE_MOD 2d8] named "+[SAVE_MOD] DEX, Sacred Flame 2d8"
}

function: keep NUMBER:n out of DFOUR:s and DSIX:s and DEIGHT:s and DTEN:s and DTWELVE:s {
  POT: {1@DFOUR,2@DFOUR,1@DSIX,2@DSIX,1@DEIGHT,2@DEIGHT,1@DTEN,2@DTEN,1@DTWELVE,2@DTWELVE}
  result: {1..NUMBER}@[sort POT]
}

function: keeps NUMBER:n out of DFOUR:s and DSIX:s and DEIGHT:s and DTEN:s and DTWELVE:s {
  POT: {1@DFOUR,2@DFOUR,1@DSIX,2@DSIX,1@DEIGHT,2@DEIGHT,1@DTEN,2@DTEN,1@DTWELVE,2@DTWELVE}
  loop P over {1..#POT} {
    if P@POT = 1 {
\      POT: {1..(P-1),(P+1)..#POT}@POT\
      POT: {{0..(P-1)}@POT,{(P+1)..#POT}@POT}
    }
  }
  result: {1..NUMBER}@[sort POT]
}

function: keepss NUMBER:n out of DFOUR:s and DSIX:s and DEIGHT:s and DTEN:s and DTWELVE:s {
  POT: {1@DFOUR,2@DFOUR,1@DSIX,2@DSIX,1@DEIGHT,2@DEIGHT,1@DTEN,2@DTEN,1@DTWELVE,2@DTWELVE}
  loop P over {1..#POT} {
    if P@POT = 1 {
      POT: {1..(P-1),(P+1)..#POT}@POT
\      POT: {{0..(P-1)}@POT,{(P+1)..#POT}@POT}\
    }
  }
  result: {1..NUMBER}@[sort POT]
}

function: drop all VALUE:n in DFOUR:s and DSIX:s and DEIGHT:s and DTEN:s and DTWELVE:s {
  POT: {1@DFOUR,2@DFOUR,1@DSIX,2@DSIX,1@DEIGHT,2@DEIGHT,1@DTEN,2@DTEN,1@DTWELVE,2@DTWELVE}
  loop P over {1..#POT} {
    if P@POT = VALUE {
      POT: {{0..(P-1)}@POT,{(P+1)..#POT}@POT}
    }
  }
  result: POT
}

FOUR: {0,2..4}

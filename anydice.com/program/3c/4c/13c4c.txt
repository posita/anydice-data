function: DICE:d advantage ADV:n {
  S: [explode d20]
  M: [maximum of DICE]
  R: (#DICE+ADV)d(M)
  loop P over {1 .. (#DICE)} {
     S: S+[exploder P@R]
  }
  result: S
}

function: exploder V:n {
  if V = M {result: V + [explode d(M)]}
  result: V
}

set "explode depth" to 3
output [1d6 advantage 0]

DYST:[explode d{0,0,0,1,1,2}]           \ normal die \
DYSTHARD:[explode d{0,0,0,0,1,2}]       \ 5+ die \

function: roll N:n on H:n {
 if H = 4 {result: NdDYST}
 if H = 5 {result: NdDYSTHARD}
}

function: shield N:n on H:n {
  S: [roll N on H]
  result: S+[inventivescientists [lowest 1 of S] on H ]
}

function: inventivescientists N:n on H:n  {
  if N = 0 { result: [roll 1 on H] }
  result: 0 
}

\ simple trim function \
function: D:d between L:n and H:n {
  result: [lowest of H and [highest of D and L]]
}

output [[roll 11 on 4] - [shield 2 on 4] between 5 and 20] named "Shooting 11AD vs. an Epicurus (DR 6)"
output [[roll 9 on 4] - [roll 5 on 5] - [shield 2 on 5] between 3 and 12] named "Shooting 10AD rockets vs. a Plato squad (DR 4)"

TFTS: [5 - 7d{0,0,1} between 0 and 5]   \a full wing of CoA drones gets shot by 7 AA \
output TFTS named "this many drones survive"
output [[roll TFTS*2 on 4] between 5 and 12] named "CoA drone wing makes an attack run on a Tsukuyomi"

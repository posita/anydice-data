function: ROLL:n vs EVAROLL:n {
 if EVAROLL = 20 {
 result: 0
 }
 if ROLL >= CRIT & ROLL + ATTACK >= EVAROLL + EVA {
  result: CRITDAMAGE
 }
 if ROLL + ATTACK >= EVAROLL + EVA | ROLL = 20 {
 result: DAMAGE
 }
 result: 0
}

EVA: 32
EVAROLL: d20

ATTACK: 32+12
NUMDICE: 2
DAMAGE: 15d{2..8}+24
CRITDAMAGE: 30d[explode d{2..8}]+24

CRIT: 14
NAUGHTY: [[highest 1 of NUMDICE d 20] vs EVAROLL]
output NAUGHTY named "Cometfall Naughty"

CRIT: 15
NICE: [[highest 1 of NUMDICE d 20] vs EVAROLL]
output NICE named "Cometfall Nice"

output NAUGHTY+NICE named "Cometfall Total"

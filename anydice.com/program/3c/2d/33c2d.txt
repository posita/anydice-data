function: rerollc REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}



function: rerollb REROLL:s on ROLL:n to REPLACEMENT:d with ADJUST {
    if [REROLL contains ROLL] {
        result: REPLACEMENT +1d6
    }
    result: ROLL +ADJUST
}



function: rerolla REROLL:s on ROLL:n to REPLACEMENT:d with ADJUST {
    if [REROLL contains ROLL] {
        result: REPLACEMENT +2d6
    }
    result: ROLL + ADJUST
}

Y: [rerollc {1} on 1d6 to 1d6]
X: [rerollb {1} on 1d6 to 1d6 with Y]
N: [rerolla {1} on 1d6 to 1d6 with X]

H: [rerollc {1,2} on 1d6 to 1d6]
J: [rerollb {1,2} on 1d6 to 1d6 with H]
K: [rerolla {1,2} on 1d6 to 1d6 with J]


output 3d6 named "base roll"

output N named "reroll one one"

output K named "reroll one one or two"

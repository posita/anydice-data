function: savethrow for DC:n and MOD:n {
result: 1d20+MOD > DC
}

function: dissave for DC:n and MOD:n {
result: [lowest 1 of 2d20]+MOD >= DC
}

function: advsave for DC:n and MOD:n {
result: [highest 1 of 2d20]+MOD >= DC
}

function: survive DAMAGE:n with SAVE:n {
if SAVE = 0 {
 result: HEALTH > DAMAGE }
else {
 result: HEALTH > (DAMAGE/2) }
}

HEALTH: 14
SAVE: [savethrow for 14 and 6]
output [survive 4d8 with SAVE] named "Oloc Chances of Shatter Survival"
HEALTH: 14*2
SAVE: [savethrow for 14 and 0]
output [survive 8d6 with SAVE] named "Oloc Chances of Fireball Survival"

HEALTH: 14+5
SAVE: [savethrow for 14 and 6]
output [survive 4d8 with SAVE] named "Oloc Chances of Shatter Survival with Temp HP"
HEALTH: (14+5)*2
SAVE: [savethrow for 14 and 0]
output [survive 8d6 with SAVE] named "Oloc Chances of Fireball Survival with Temp HP"


HEALTH: (14+5 + 3 + 1d8)*2
SAVE: [savethrow for 14 and 6]
output [survive 8d6 with SAVE] named "Oloc Chances of Fireball Survival with Temp HP + cure wounds"

HEALTH: 2 + 2 + 1d8
SAVE: [advsave for 14 and 1]
output [survive 4d8 with SAVE] named "Tah Chances of Shatter Survival with 1st level cure wounds"
HEALTH: 2 + 3 + 1d8
SAVE: [advsave for 14 and 1]
output [survive 8d6 with SAVE] named "Tah Chances of Fireball Survival with 1st level cure wounds"


HEALTH: 2 + 2 + 3d8
SAVE: [advsave for 14 and 1]
output [survive 4d8 with SAVE] named "Tah Chances of Shatter Survival with 3rd level cure wounds"

SAVE: [advsave for 14 and 1]

function: explode N:n S:d {
  if d[N] = N {
    result: d[N] + [explode [N 1] ]
  } else {
    result: d[N]
  }
}

set "maximum function depth" to 10


output [explode 6d4]  named "6d4"
output [explode 5d6]  named "5d6"

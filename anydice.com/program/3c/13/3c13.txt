SPELLSINGERMODIFIER: 10

MONSTERSAVEBONUS: 3
MONSTERHP: 2d12
MONSTERHPAVERAGE: 13

SPELLSONGITERATIONS: (1d20+SPELLSINGERMODIFIER)/5
SPELLSONGDC: 10+SPELLSONGITERATIONS
SPELLSONGDAMAGE: ((2*SPELLSONGITERATIONS)d4)

MONSTERSAVECHANCE: (1d20+MONSTERSAVEBONUS) >= SPELLSONGDC
MONSTERDAMAGETAKEN: (SPELLSONGDAMAGE + SPELLSONGDAMAGE*MONSTERSAVECHANCE)/2
MONSTERSURVIVES: MONSTERHP > MONSTERDAMAGETAKEN
MONSTERAVERAGEHPSURVIVES: MONSTERHPAVERAGE > MONSTERDAMAGETAKEN

output SPELLSONGITERATIONS named "iterations w/ +10 modifier"
output SPELLSONGDC named "save DC"
output SPELLSONGDAMAGE named "damage"
output MONSTERSAVECHANCE named "Skeleton Wolf chance to succeed on Reflex save"
output MONSTERSURVIVES named "Skeleton Wolf chance to survive (randomly rolled HP)"
output MONSTERAVERAGEHPSURVIVES named "Skeleton Wolf chance to survive (average HP)"

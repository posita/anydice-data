function: fifth edition attack ROLL:n {
if ROLL = 1  {result: 0}
if ROLL >= 20 {result: 2*DMG_DICE + DMG_MOD}
if ROLL+TO_HIT >= AC {result: DMG_DICE + DMG_MOD}
result: 0}


loop AC over {8..23} {
 DMG_DICE: 1d8
 DMG_MOD: 2
 TO_HIT: 7
 output [fifth edition attack d20] named "[AC] AC, Longbow +1"
 DMG_DICE: 1d6
 DMG_MOD: 1
 TO_HIT: 4
 output [fifth edition attack d20] named "[AC] AC, Shortsword"
 DMG_DICE: 1d12
 DMG_MOD: 0
 TO_HIT: 3
 output [fifth edition attack d20] named "[AC] AC, Greataxe"
}

function: roll X:n and Y:n against D:n with favor F:n {
  Z: 0
  G: F
  if X > D {
    Z: Z+1
  }
  else if X+G > D {
    Z: Z+1
    G: F-(D-X+1)
  }

  if Y > D {
    Z: Z+1
  }
  else if Y+G > D {
    Z: Z+1
  }

  result: Z
}

loop N over {1..5}{
  output [roll 1d6 and 1d6 against N with favor 1]
}

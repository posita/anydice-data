function: pick NTH:n of SEQ:s and BON:s {
 ALL: [sort {SEQ, BON}]
 if NTH@ALL = BON { result: (NTH-1)@ALL}
 result: NTH@ALL
}

function: pick NTH:n of SEQ:s on BON:n {
 if BON = 1 { result: (NTH-1)@SEQ }
 result: NTH@SEQ
}

DIE: 20

\output 1@4dDIE named "1"\
loop N over {2} {
output N@4dDIE named "[N]"
output [pick N of 3dDIE and 1dDIE] named "[N]+"
output [pick N of 2dDIE and 2dDIE] named "[N]++"
output [pick N of 1dDIE and 3dDIE] named "[N]+++"
}

output 1@4dDIE named "4"

function: sign A:n {
  if A > 0 { result: 1 }
  if A < 0 { result: -1}
  result: 0
}

\ player 1 rolls 3d6 and player 2 rolls 3d6. \
\ 1 indicates that player 1 won, -1 that player 2 won, and 0 indicates a tie \

loop A over {3}{
  loop B over {3}{
    output [sign 3d6 - 3d6] named "[A]d6 vs. [B]d6"
  }
}

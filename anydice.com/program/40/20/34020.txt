\ Situational inputs \
AC: 14
SACRED_WEAPON: 0
GREAT_WEAPON: 0

\ Character attributes \
STR: 2
CHAR: 3
PROF: 3

\ Hit calculation \
MOD: STR + PROF

if SACRED_WEAPON {
  MOD: MOD + CHAR
}

if GREAT_WEAPON {
  MOD: MOD - 5
}

HIT: 1d20 + MOD > AC

\ Damage calculation \
function: if CONDITION:n then A else B {
  if CONDITION { result: A } else { result: B }
}

DMGA: 1d6
DMGB: 1d6
DMG: [if DMGA <= 2 then 1d6 else DMGA] + [if DMGB <= 2 then 1d6 else DMGB]

\ Outputs \
output HIT named "hit"
output DMG named "damage"

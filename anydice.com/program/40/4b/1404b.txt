function: HV:n to hit vs DV:n {
 HIT: 1d20 + HV
 DODGE: 1d20 + DV
 result: HIT > DODGE
}
function: backswing on HIT:n {
 if HIT {result: 1}
 else {result: (1d4 = 1)*[HV to hit vs DV]}
}
function: HV:n to hit vs DV:n with backswing {
 BACK: [backswing on [HV to hit vs DV]]
 result: BACK
}

\just calculate for one condition\
HV: 2
DV: 0
output [HV to hit vs DV] named "normal [HV] [DV]"
output [HV to hit vs DV with backswing] named "backswing [HV] [DV]"

\loop over different conditions\
\
loop HV over {0..4}{
 loop DV over {0..4}{
  output [HV to hit vs DV] named "normal [HV] [DV]"
  output [HV to hit vs DV with backswing] named "backswing [HV] [DV]"
 }
}
\

\if you want all the backswings together, put them in their own loop\
\
loop HV over {0..4}{
 DV: 0
 output [HV to hit vs DV] named "normal [HV] [DV]"
}
loop HV over {0..4}{
 DV: 0
 output [HV to hit vs DV with backswing] named "backswing [HV] [DV]"
}

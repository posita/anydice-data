function: defense of A:s {
   DOUBLES: 0
   COUNT: 0
   loop I over A {
       if I = 6 {
           DOUBLES: DOUBLES + 1
       }
       if I = 5 {
           COUNT: COUNT + 2
       }
       if I = 4 | I = 3 {
           COUNT: COUNT + 1
       }
   }
   result: COUNT * (2^DOUBLES)
}

output [defense of 4d6]

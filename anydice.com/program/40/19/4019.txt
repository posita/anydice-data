function: extended pick NTH:n of SEQ:s k K:n m M:n{
 if NTH@SEQ = M { result: M+(SEQ = M) - NTH }
 if NTH@SEQ = 1 { result: 1-(SEQ = 1) + (1+K-NTH) }
 result: NTH@SEQ
}

K: 6
M: 10
loop N over {1..K} {
 output [highest of [extended pick N of KdM k K m M] and [extended pick N of KdM k K m M]] named "[N]R"
 output [extended pick N of KdM k K m M] named "[N]"
}
 output [lowest of [extended pick K of KdM k K m M] and [extended pick K of KdM k K m M]] named "[N]D"
 output [lowest of [extended pick K of KdM k K m M] and [lowest of [extended pick K of KdM k K m M] and [extended pick K of KdM k K m M]]] named "[N]D"

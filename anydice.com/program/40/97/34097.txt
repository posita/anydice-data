\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-25\
ENEMYAC: d{15..25}

ATTACKMOD: 0
DMGBONUS: 8
PROFICIENCY: 4
ACCBONUS: 0
CRITROLL: 20
DAMAGE: 2d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 12 + ATTACKMOD + DMGBONUS
output [attackroll d20+d6 vs ENEMYAC] named "Standard shot"

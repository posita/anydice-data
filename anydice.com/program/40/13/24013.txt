MISS:0
HIT:1
CRITHIT:2
if ISCHAMPION=NO{
 CHAMPIONTHREAT:0
}

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL > 20 - THREAT { result: CRITHIT }
 if ROLL + ATTACK > DEFENSE { result: HIT}
 result: MISS
}
function: strcontest ROLLATK:n vs ROLLDEF:n{
 if ROLLATK = 1 {result: MISS}
 if ROLLATK = 20 { result: HIT}
 if ROLLATK + DEX > ROLLDEF + ENEMYSTRMOD {result: HIT}
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITHIT {
  result: CRITDAM
 }
 if ATTACKRESULT = HIT {
  result: DAM
 }
 result: MISSDAM
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

function: advattack {
 result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]
}

function: knockdownattackonce STRCONTESTRESULT:n{
 if STRCONTESTRESULT = HIT {result: [advattack]} 
 result: [attack]
}
function: knockdownattacktwice STRCONTESTRESULT:n{
 if STRCONTESTRESULT = HIT {result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]+[damage for [attack [highest 1 of 2d20] vs DEFENSE]]} 
 result: [damage for [attack d20 vs DEFENSE]]+[damage for [attack d20 vs DEFENSE]]
}
function: knockdownattackthrice STRCONTESTRESULT:n{
 if STRCONTESTRESULT = HIT {result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]+[damage for [attack [highest 1 of 2d20] vs DEFENSE]]+[damage for [attack [highest 1 of 2d20] vs DEFENSE]]} 
 result: [damage for [attack d20 vs DEFENSE]]+[damage for [attack d20 vs DEFENSE]+[damage for [attack d20 vs DEFENSE]]]
}


function: knockdownattacksix STRCONTESTRESULT:n{
 if STRCONTESTRESULT = HIT {result: [advattack]+[advattack]+[advattack]+[advattack]+[advattack]+[advattack]} 
 result: [attack]+[attack]+[attack]+[attack]+[attack]+[attack]
}

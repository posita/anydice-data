A: 15 \ AC \
M: 7 \ To hit \
N: 4 \ Number of attacks to be made \

function: roll DICE:n to hit {
if (DICE = 1) {
result: 0
} else if (DICE = 20) {
result: 1
} else {
result: DICE + M >= A
}
}

output Nd([roll 1d20 to hit]) named "chances for N attacks to hit"
output Nd([roll 1@2d20 to hit]) named "chances for N attacks to hit with advantage"
output Nd([roll 2@2d20 to hit]) named "chances for N attacks to hit with disadvantage"

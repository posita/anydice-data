     \ Adding the two highest dice after removing X of the highest dice, where X is the difficulty. (Removing 4 dice from 5d6 will only result in one die.) \
      \ CHANGE THESE VARIABLES BELOW \

DIETYPE: d6               \ What kind of dice to be rolled \

LOWESTDIFFICULTY: 1       \ The lowest possible difficulty in the system \
HIGHESTDIFFICULTY: 5      \ The highest possible difficulty in the system \

LOWESTSKILL: 5            \ The lowest number of dice possible to be rolled \
HIGHESTSKILL: 10          \ The highest number of dice possible to be rolled \

function: remove highest X:n dice from Y:d rolled {
  if X >= #Y       { result: 0 }
  else if X = #Y-1 { result: [lowest 1 of Y] }
  else             { result: [highest 2 of {X..X+1}@Y] }
}

loop DIFFICULTY over {LOWESTDIFFICULTY..HIGHESTDIFFICULTY} {
  loop DICEPOOL over {LOWESTSKILL..HIGHESTSKILL} {
    output [remove highest DIFFICULTY dice from DICEPOOLdDIETYPE rolled] named "Rolled [DICEPOOL][DIETYPE] against difficulty [DIFFICULTY]"
  }
  output 1337 named "~~~~~~~~~~~~~~~~~~~~~ NEW DIFFICULTY ~~~~~~~~~~~~~~~~~~~~~"
}

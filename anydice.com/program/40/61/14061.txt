MISS: 0
HIT: 1
CRITICALHIT: 2
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}
function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

MISSDAMAGE: 0
AC: d{12..15}

ATTACK: 5
HITDAMAGE: 1d8
CRITDAMAGE: 2d8
output [damage for [attack d20 vs AC]] named "Longsword"

ATTACK: 5
HITDAMAGE: 1d8
CRITDAMAGE: 2d8
output [damage for [attack [highest 1 of 2d20] vs AC]] named "Longsword w/ Advantage"

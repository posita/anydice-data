function: extended pick NTH:n of SEQ:s k K:n m M:n{
 MIN: 1
 if NTH@SEQ = M { result: M+(SEQ = M) - NTH }
 if NTH@SEQ = MIN { result: MIN-(SEQ = MIN) + (1+K-NTH) }
 result: NTH@SEQ
}

K: 6
M: 6

output [extended pick 5 of KdM k K m M]+M named "[vi++]"
output [extended pick 6 of KdM k K m M]+M named "[vi+]"
loop N over {1..K} {
 NAM: K+1-N
 output [extended pick N of KdM k K m M] named "[NAM]"
}
output [extended pick 1 of KdM k K m M]-M named "[i-]"
output [extended pick 2 of KdM k K m M]-M named "[i--]"

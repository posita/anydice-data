function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d10] }
 result: N >= 8
}

function: clash A:n vs B:n {
  if A > B { result: 1 }
  if A < B { result: -1 }
  if A = B { result: [clash A vs B] }
}

NWOD: [nwod 10 again d10]

output [clash 6dNWOD vs 6dNWOD] named "5 dice pool clash"
output 6dNWOD >= 6dNWOD named "5 dice pool clash"

loop N over {1..10}  {
  output [lowest of 10 and 3dNWOD] > N named "3d10 >= [N]"
}

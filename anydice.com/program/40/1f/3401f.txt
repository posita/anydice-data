\ Situational inputs \
AC: 14
SACRED_WEAPON: 0
GREAT_WEAPON: 0

\ Character attributes \
STR: 2
CHAR: 3
PROF: 3

\ Calculation \
MOD: STR + PROF

if SACRED_WEAPON {
  MOD: MOD + CHAR
}

if GREAT_WEAPON {
  MOD: MOD - 5
}

HIT: 1d20 + MOD > AC

DMG: 2d6

\ Outputs \
output HIT named "hit"
output DMG named "damage"

function: checkammo ROLL:n SIZE:n THRESHOLD:n MIN_SIZE:n {
	if SIZE < MIN_SIZE {result: 0}
	if ROLL <= THRESHOLD {result: 1 + [checkammo d{1..SIZE-2} SIZE-2 THRESHOLD MIN_SIZE]}
	result: 1 + [checkammo d{1..SIZE} SIZE THRESHOLD MIN_SIZE]
}

function: start SIZE:n THRESHOLD:n MIN_SIZE:n {
	result: [checkammo d{1..SIZE} SIZE THRESHOLD MIN_SIZE]
}

set "maximum function depth" to 8
output [start 6 2 4]

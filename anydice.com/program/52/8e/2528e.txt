function: ability N:n {
 if N >= 8 { result: 1 + [ability  d8] }
 result: N >= 5
}

function: skill N:n {
 if N >= 6 { result: 1 + [skill  d6] }
 result: N >= 5
}

AB: [ability d8]
SK: [skill d6]

loop N over {1..5} {
  loop M over {0..5} {
   output [lowest of 20 and NdAB + MdSK] named "[N]a[M]s"
  }
}

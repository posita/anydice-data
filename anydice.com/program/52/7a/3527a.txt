\
 Results:
 0: Things go bad.
 1: Success, but with undesired consequences.
 2: Clean success.
 3: Critical success!
\

function: pool ROLL:s {
 if 1@ROLL = 6 {
  result: 2 + (2@ROLL = 6)
 }
 result: 1@ROLL >= 4
}

output [pool 2@2d6] named "0d"
loop P over {1..7} {
 output [pool Pd6] named "[P]d"
}

\ helper function: count the number of times each of ITEMS occurs in SEQ, skipping zeros \
function: summarize ITEMS:s in SEQ:s {
  R: {}
  loop I over ITEMS { if I = SEQ { R: {R, I = SEQ} } }
  result: R
}
function: test A:n and B:n and C:n and D:n {
  S: [summarize {1..6} in {A, B, C, D}]
  result: [sort S] = [sort {2, 1, 1}]  \ one pair, two singles \
}
output [test d6 and d6 and d6 and d6]

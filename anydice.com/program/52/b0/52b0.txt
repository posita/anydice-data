EC: 1d6 \easy challenge\
HC: 2d6 \hard challenge\
LC: 3d6 \legendary challenge\

N: 0 \loop iterator\
MIN_BOLDS: 1 \minimum number of kobolds\
MAX_BOLDS: 6 \maximum number of kobolds\
MIN_PERKS: 0 \minimum number of perks\
MAX_PERKS: MAX_BOLDS \maximum number of perks\
KS: 0 \instantiate "kobolds"\
PS: 0 \instantiate "perks"\
DS: 0 \instantiate "dice"\
TB: 0 \instantiate "team bonus"\
RESULT: 0 \instantiate "result" variable\
CHAL: EC \challenge level\
loop N over {MIN_BOLDS..MAX_BOLDS}{
 KS: N
 loop N over {MIN_PERKS..N}{
  PS: N
  DS: KS + PS
  if KS > 1{
   TB: KS
  }else{
   TB: 0
  }
  RESULT: (DSd6 + TB)
  RESWINS: RESULT > CHAL
  output RESWINS named "[KS]k[PS]p, DS = [DS], so I will roll [DS]d6+[TB] and compare to [CHAL]"
 }
}

\KKPP_LC: KKPP > LC
output KKPP_LC named "two perky greater than leg chal"

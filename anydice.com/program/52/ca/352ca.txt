TARGET_AC:15

DEX:2  \dex modifier\
AFS:2  \archery fighting style modifier\
WEP:1  \+1 magic weapon modifier\
SSP:5  \sharpshooter attack penalty\
SSB:10 \sharpshooter damage bonus\

function: handcrossbow AC HIT_PENALTY DMG_BONUS {
  HIT: 1d20 + DEX + WEP + AFS - HIT_PENALTY >= AC
  DMG: 1d6 + DEX + WEP + DMG_BONUS
  result: HIT * DMG
}

function: heavycrossbow AC HIT_PENALTY DMG_BONUS {
  HIT: 1d20 + DEX + WEP + AFS - HIT_PENALTY >= AC
  DMG: 1d10 + DEX + WEP + DMG_BONUS
  result: HIT * DMG
}

output [handcrossbow TARGET_AC 0 0] + [handcrossbow TARGET_AC 0 0] + [handcrossbow TARGET_AC 0 0] named "Normal +1 Hand Crossbow"
output [handcrossbow TARGET_AC 5 10] + [handcrossbow TARGET_AC 5 10] + [handcrossbow TARGET_AC 5 10] named "Sharpshooter +1 Hand Crossbow"
output [heavycrossbow TARGET_AC 0 0] + [heavycrossbow TARGET_AC 0 0] named "Normal +1 Heavy Crossbow"
output [heavycrossbow TARGET_AC 5 10] + [heavycrossbow TARGET_AC 5 10] named "Sharpshooter +1 Heavy Crossbow"

function: max of D:n and V:n {
if D > V {
result: V
}
else {
if D < 0 {
result: 0
}
else {
result: D
}
}
}

function: full auto N:n dice of D:n damage F:n rate of fire A:n accuracy C:n recoil against R:n reflex {
HITS: [max of 2d10+A+F-2*C-R and F]
DICE: HITS * N
result: DICEdD
}

function: full auto mul dice dmg rof acc rec ref M:n N:n D:n F:n A:n C:n R:n {
HITS: [max of 2d10+A+F-2*C-R and F]
DICE: HITS * N * M
result: DICEdD
}

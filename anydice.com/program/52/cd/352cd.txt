TARGET_AC:15

DEX:2  \dex modifier\
AFS:2  \archery fighting style modifier\
WEP:1  \+1 magic weapon modifier\

function: attack TIMES DMG_DIE AC HIT_MOD DMG_MOD {
  TOTAL: 0
  loop N over {1..TIMES}{
    HIT: 1d20 + DEX + WEP + AFS + HIT_MOD >= AC
    DMG: DMG_DIE + DEX + WEP + DMG_MOD
    TOTAL: TOTAL + HIT * DMG
  }
  result: TOTAL
}

output [attack 3 1d6 TARGET_AC 0 0] named "Normal +1 Hand Crossbow"
output [attack 2 1d10 TARGET_AC 0 0] named "Normal +1 Heavy Crossbow"

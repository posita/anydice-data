set "explode depth" to 5
set "maximum function depth" to 5

function: poolplode DIE:d TN:n EN:n {
 result: [poolplode helper DIE]
}

function: poolplode helper RES:n {
 if RES >= EN { result: 1 + [poolplode helper DIE] }
 if RES >= TN { result: 1 }
 result: 0
}

loop N over {1..16} {
 output Nd[poolplode 1d6 4 6]
}

output (16d[explode (1d6 >= 4)])

\loop N over {1..8} {
  output (Nd[explode (1d6 >= 5)]) 
}\

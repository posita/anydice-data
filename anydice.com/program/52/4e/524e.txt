set "explode depth" to 5

function: dexplode N RES:n {
 if RES = 0 { result: 1- (N@Nd4) }
 result: RES
}

\output [dexplode 1d{0,0,0,0,0,-1}]\


\output 1d[explode (2@2d6 >= 5)]

output 1d[explode (1d6 >= 5)]\

output 1d[explode (1@2d6 >= 5)]



loop N over {1..16} {
  output (Nd[explode (1d6 >= 5)] + 1d[explode (1@2d6 >= 5)])
}

set "explode depth" to 10

RANGE: 20
DIE: d12
DIEMAX: [maximum of DIE]
DIEMID: DIEMAX / 2 + 1
DTIE: d{0,1}

function: clamp N:n between MIN:n and MAX:n {
  if N > MAX { result: MAX }
  if N < MIN { result: MIN }
  result: N
}

function: extend {
  result: [extend DIE]
}
function: extend N:n {
  if N = 1 { result: N - [luck] }
  if N = DIEMAX{ result: N + [luck] }
  result: N
}

function: luck {
  result: [luck d6]
}
function: luck N:n {
  if N = 6 { result: 5 + [luck d6] }
  result: N - 1
}

function: ez {
  result: [ez DIE]
}
function: ez N:n {
  if N = DIEMAX { result: N + [luck] }
  result: N
}

output [clamp [extend]-DIEMID between -RANGE and RANGE] named "[DIE]X-[DIEMID]"
output [clamp [luck] between -RANGE and RANGE] named "luck"
output [clamp [ez]-[ez]-DTIE between -RANGE and RANGE] named "[DIE]L vs"

output [clamp d6+[luck]-d6-[luck]-DTIE between -RANGE and RANGE] named "d6+d6L vs"

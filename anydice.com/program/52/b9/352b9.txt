
function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: vargas {
    result: [highest 1 of [reroll {1,2} on 1d6 to 1d6]]
}

\ Vargas \
output [vargas] + [vargas] + 7 named "Vargas Melee"
output 1d8+2 named "Vargas Ranged"

\ Anastasia \
output 1d8+5 named "Anastasia Shillelagh"


function: skill roll S:d {
  result: [highest of d20 and S]
}

function: roll D:d skill S:n {
  result: [roll D skill S helper]
}
function: roll N:n skill S:n helper {
  if N <= S { result: [highest of N and D] }
  result: N  
}
function: roll D:d skill once S:n {
  result: [roll D skill once S helper]
}
function: roll N:n skill once S:n helper {
  if N <= S { result: D }
  result: N
}

TEST_RANGE_MAX: {1, 2, 4, 6, 8, 10, 12, 20}
TEST: {4}

loop N over TEST {
 output [skill roll dN] named "Skill Roll d[N]"
 N_HALF: N / 2
 output [roll d20 skill N_HALF] named "Skill Adv [N_HALF]"
 output [roll d20 skill once N_HALF] named "Skill reroll [N_HALF]"
 output d20 + N_HALF named "Roll + [N_HALF]"
}

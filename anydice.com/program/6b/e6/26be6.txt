\reroll a given roll if the result is a given number\
function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}
\check if a given roll passes a given dc\
function: pass ROLL:n dc DC:n{
  if ROLL >= DC {result: 1}
  result:0
}
\d&d attack roll sequence with varying to hit bonus, ac, crit threshold, damage, crit damage\
function: attack ROLL:n bonus BONUS:n crits CRIT:n ac AC:n dmg DMG:d critdamage DMGCRIT:d {
  if ROLL >= CRIT {result: DMGCRIT}
  else if ROLL <= 1 {result: 0}
  else if (ROLL + BONUS) >= AC {result: DMG}
  result: 0
}
\D&D ATTACK\

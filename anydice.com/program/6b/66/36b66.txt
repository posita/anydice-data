DIE_ONE: 1d10
DIE_ONE_ATTACKS: 2
DIE_TWO: 2d6
DIE_TWO_ATTACKS: DIE_ONE_ATTACKS - 1
NEEDED_ROLL: 19

function: attack NA:n times against DC:n dealing DS:d {
 TV: 0
 loop X over {1..NA}{
  CV: (1d20 > DC)*DS
  TV: TV + CV
 }
 result: TV
}

output !([attack DIE_ONE_ATTACKS times against NEEDED_ROLL dealing DIE_ONE] > [attack DIE_TWO_ATTACKS times against NEEDED_ROLL dealing DIE_TWO]) +1 named "which attack is better (ties favor the first)"

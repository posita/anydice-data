function: implode DIE:d {
 MIN: [minimum of DIE]
 result: [implode DIE helper]
}

function: implode N:n helper {
 if N = MIN { result: N - [explode DIE] }
 result: N
}

function: explode DIE:d {
 MAX: [maximum of DIE]
 result: [explode DIE helper]
}

function: explode N:n helper {
 if N = MAX { result: N + [explode DIE helper] }
 result: N
}

function: plode DIE:d {
  MIN: [minimum of DIE]
  MAX: [maximum of DIE]
  result: [plode helper DIE]
}

function: plode helper R:n {
 if R = MIN { result: R - [explode DIE] }
 if R = MAX { result: R + [explode DIE] }
 result: R
}

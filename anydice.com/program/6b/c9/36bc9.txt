function: A beats high B {
result: [1@Ad6 beats high num 1@Bd6]
}

function: A:n beats high num B:n {
result: A >= B
}

function: A beats tie B {
result: [1@Ad6 beats tie num 1@Bd6]
}

function: A:n beats tie num B:n {
X: A - B
if X > 0 {
result: 1
}
else if X = 0{
result: d2 - 1
}
else {
result: 0
}
}

function: A beats low B {
result: [1@Ad6 beats low num 1@Bd6]
}

function: A:n beats low num B:n {
result: A > B
}

function: A wounds B {
result: [[highest 1 of Ad6] wounds num B]
}
function: A:n wounds num B:n {
result: A >= B
}

\**** Damage calculator normal Vs GWM ****\
\Maximum AC = attack bonus - damage/2 + 16\


function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: DAMAGE + DAMAGE + DAMBONUS} 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE +DAMBONUS}
 result: 0
}

ATTACKMOD: 4
PROFICIENCY: 5
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d10+1d8

\looped against enemy AC\
\ENEMYAC: d{13..20}\
\ENEMYAC:14\
ENEMYAC: d{16..20}

ACCBONUS: 3
DAMBONUS: 7

output ATTACKTIMESd[attackroll [highest 1 of 1d20] vs ENEMYAC] named "Att"
output ATTACKTIMESd[attackroll [highest 1 of 1d20+1] vs ENEMYAC] named "Att+1"
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "VoEAtt"

ACCBONUS: -2
DAMBONUS: 17
output ATTACKTIMESd[attackroll [highest 1 of 1d20] vs ENEMYAC] named "GWMAtt"
output ATTACKTIMESd[attackroll [highest 1 of 1d20+1] vs ENEMYAC] named "GWMAtt+1"
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "VoEGWMAtt"

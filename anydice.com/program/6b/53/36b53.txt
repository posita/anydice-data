\\\*** Functions ***\\\
NORMAL: 1d20
ADV: [highest 1 of 2d20]
CRIT_RANGE: 20
CRIT_BONUS: 0 \Any additional dice used specifically for a critical\

function: attack ROLL:n vs ENEMY_AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL >= CRIT_RANGE { result: CRITICALDAMAGE }
 if ROLL + ATTACK >= ENEMY_AC { result: HITDAMAGE }
 result: 0
}
function: single ROLL:n vs ENEMY_AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL + ATTACK >= ENEMY_AC { result: SINGLEDAMAGE }
 result: 0
}
function: spell ROLL:n vs SPELL_DC:n {
 if ROLL + SAVE >= SPELL_DC {result: SUCC_SAVE}
 if ROLL + SAVE < SPELL_DC {result: FAIL_SAVE}
}

\\\ WEAPON DMG \\\
ATTACK: P_STAT_MOD + PROF + MAGIC_BONUS + HIT_BONUS
if P_DMG_RESIST = 0 {
HITDAMAGE: ATTACK_DICE + P_STAT_MOD + MAGIC_BONUS + BONUS_DAM_MOD + BONUS_DICE
CRITICALDAMAGE: 2dATTACK_DICE + P_STAT_MOD + MAGIC_BONUS + BONUS_DAM_MOD + 2dBONUS_DICE + CRIT_BONUS
SINGLEDAMAGE: SINGLE_HIT
}
else { 
if P_DMG_RESIST = 1 {
HITDAMAGE: (ATTACK_DICE + P_STAT_MOD + MAGIC_BONUS + BONUS_DAM_MOD + BONUS_DICE)/2
CRITICALDAMAGE: (2dATTACK_DICE + P_STAT_MOD + MAGIC_BONUS + BONUS_DAM_MOD + 2dBONUS_DICE + CRIT_BONUS)/2
SINGLEDAMAGE: (SINGLE_HIT)/2
}
}

  output N_ATTACKSd[attack NORMAL vs ENEMY_AC] + [single NORMAL vs ENEMY_AC] named "Weapon"

  output N_ATTACKSd[attack ADV vs ENEMY_AC] + [single ADV vs ENEMY_AC] named "Adv Weapon"
}
else {
  if P_ADV = 2 {
  output N_ATTACKSd[attack NORMAL vs ENEMY_AC] + [single NORMAL vs ENEMY_AC] named "Weapon"
  output N_ATTACKSd[attack ADV vs ENEMY_AC] + [single ADV vs ENEMY_AC] named "Adv Weapon"
}
}

function: attackroll ROLL:n {
if ROLL >= 20 { result: CRIT_DMG }
if ROLL = 1 { result: 0 }
if ROLL + 8 - HIT_PENALTY >= 16 { result: BASE_DMG }
result: 0
}

AC: 16
HIT_PENALTY: 5
DMG_BONUS: +10
BASE_DMG: 1d6 + 1d10
CRIT_DMG: 2d6 + 2d10
output ([highest 1 of 3d20] + 8 - HIT_PENALTY >= 16)*(BASE_DMG + DMG_BONUS) named "Elven Accuracy"
output ([highest 1 of 2d20] + 9 - HIT_PENALTY >= 16)*(BASE_DMG + DMG_BONUS + 1) named "DEX 20"
output [attackroll [highest 1 of 2d20] + 9 - HIT_PENALTY] + DMG_BONUS + 1 named "DEX 20"
output [attackroll [highest 1 of 3d20] + 8 - HIT_PENALTY] + DMG_BONUS named "Elven Accuracy"

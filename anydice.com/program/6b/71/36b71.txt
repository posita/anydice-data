function: ROLL:s count double sixes or triples from four up {
 if #ROLL < 3 {
   result: 0
 }
 else {
  X: 0
  Y: 0
  Z: 0
  loop N over {1..#ROLL} {
   if N@ROLL = 6 {
    X: X + 1
   }
   if N@ROLL = 5 {
    Y: Y + 1
   }
   if N@ROLL = 4 {
    Z: Z + 1
   }
  }
  result: (((X > 1) + (Y> 2) + (Z>2)) >0) 
 }
}

output [8d6 count double sixes or triples from four up]

function: champattack ROLL:n plus TOHIT:n vs AC:n for DICEDMG:n plus BONUSDMG:n {
 if ROLL = 1 { 
  result: 0
 }
 if ROLL >= 19 {
  result: 2 * DICEDMG + BONUSDMG
 }
 if ROLL + TOHIT >= AC {
  result: DICEDMG + BONUSDMG
 }
 result: 0
}

function: attack ROLL:n plus TOHIT:n vs AC:n for DICEDMG:n plus BONUSDMG:n {
 if ROLL = 1 { 
  result: 0
 }
 if ROLL >= 20 {
  result: 2*DICEDMG + BONUSDMG
 }
 if ROLL + TOHIT >= AC {
  result: DICEDMG + BONUSDMG
 }
 result: 0
}

loop AC over {10..25} {
 output (2d[champattack d20 plus 10 vs AC for 1d6 plus 8]+[champattack d20 plus 10 vs AC for 1d5 plus 8] -
         2d[attack d20 plus 10 vs AC for 1d6 plus 8]-[attack d20 plus 10 vs AC for 1d5 plus 8])
}

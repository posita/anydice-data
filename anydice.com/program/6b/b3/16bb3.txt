function: roll TRAIT:s against DIFFICULTY:n specialty SPECIALTY:n {
  SUCCESSES: 0
  loop ROLL over TRAIT {
    if SPECIALTY & ROLL = 10 { SUCCESSES: SUCCESSES + 2 }
    else if ROLL >= DIFFICULTY { SUCCESSES: SUCCESSES + 1 }
    else if ROLL = 1 { SUCCESSES: SUCCESSES - 1 }
  }
  result: SUCCESSES
}

output [roll 4d10 against 7 specialty 0] named "frenzy"

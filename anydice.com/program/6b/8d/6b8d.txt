function: aw A:n {
   if A >= 10 { result: 2 }
   if A >= 7 { result: 1 }
   result: 0
}

function: hl S:s N:n {
  H: 1@S
  L: 2@S
  T: 9 - N
  if H >= T & L >= T { result: 2 }
  if H >= T & L < T { result: 1 }
  result: 0
}

loop N over {-1..3} {
 output [aw 2d6+N] named "AW + [N]"
}

loop N over {-1,0,2,4,6} {
 output [hl 2d12 N] named "HL + [N]"
}

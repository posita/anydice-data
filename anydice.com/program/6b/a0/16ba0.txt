function: flip ROLL{
result: 7 - ROLL
}

function: flip ROLL:n negative{
if ROLL = {4..6}{result: [flip ROLL]}
result: ROLL
}

function: flip ROLL:n positive{
if ROLL = {1..3}{result: [flip ROLL]}
result: ROLL
}

function: reroll A:n if less than N:n{
 if A < N { result: 1d6 }
 result: A
}

function: roll DICE dice rerolling under THRESHOLD up to REROLLS times posflip POSFLIP negflip NEGFLIP{
SUCCESS: 0
loop N over {1..DICE}{
   ROLL: 1d6
   loop F over {1..NEGFLIP}{
      ROLL: [flip ROLL negative]
   }
   loop F over {1..POSFLIP}{
      ROLL: [flip ROLL positive]
   }
   loop M over {1..REROLLS}{
      ROLL: [reroll ROLL if less than THRESHOLD]
   }
   SUCCESS: SUCCESS + [count{5,6} in ROLL] - [count{1,2} in ROLL]
}
result: SUCCESS
}

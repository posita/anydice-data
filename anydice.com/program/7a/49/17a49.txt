function: explode NUMBER:n highest of DICE:d {
  SUM: 0
  loop I over {1..NUMBER} {
    SUM: SUM + [explode and reroll I@DICE on [maximum of I@DICE]]
  }
  result: SUM
}

function: explode and reroll NUMBER:n on VALUE:n {
  if NUMBER = VALUE { result: [explode and reroll d(VALUE) on VALUE] + [explode and reroll d(VALUE) on VALUE] }
  result: NUMBER
}

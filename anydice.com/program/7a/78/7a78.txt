function: countmax VALUES:s in SEQUENCE:s max MAX:n exclude EXC:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  if (COUNT < MAX)
   {
    COUNT: COUNT + (P@VALUES = SEQUENCE)
   }
  if (P@VALUES = EXC & COUNT > -1)
   {
    COUNT: COUNT - 1
   }
 }
 result: COUNT
}
function: character CARAC:n COMP:n {
 X : CARACd10
 Y : [countmax X in {7..10} max COMP exclude {1}]
 result: Y
}

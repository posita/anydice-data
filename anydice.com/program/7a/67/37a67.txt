function: ROLL:n vs AC:n {
 if (ROLL >= THREAT & ROLL + ATTACK >= AC) | ROLL = 20 {
  result: [threat d20]
 }
 if ROLL + ATTACK >= AC { result: DAMAGE }
 result: 0
}

function: threat ROLL:n {
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: CRITICAL d DAMAGE + EXDAMAGE
 }
 result: DAMAGE + EXDAMAGE
}

function: roll D:s with skill S:n against T:n {
   REMAINING: S
   SUCCESSES: 0
   loop I over {1..#D} {
      R: I@D
      if R >= T {
         SUCCESSES: SUCCESSES + 1
      }
      else {
         MARGIN: T - R
         if REMAINING >= MARGIN {
            SUCCESSES: SUCCESSES + 1
            REMAINING: REMAINING - MARGIN
         }
      }
   }

   result: SUCCESSES
}

loop DICE over {1..4} {
   loop SKILL over {1..3} {
      output [roll DICEd6 with skill SKILL against 5] named "[DICE]d6 + [SKILL]"
   }
}

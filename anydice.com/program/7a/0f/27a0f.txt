CRITRANGE: 20
AC: 17
DEX_MOD: 4
PROF_BONUS: 3
SNEAK: 4d6
HITDAM: DEX_MOD 
LEVEL: 9
BOOMING: 3d8

function: roll ROLL:n at plus ATTACKBONUS:n vs ac AC:n for CRITABLEDAM:d plus HITDAM:d {
    if ROLL >= CRITRANGE {
    result: CRITABLEDAM + CRITABLEDAM + HITDAM
    }
        if ROLL = 1 {
        result: 0
        }
            if ROLL >= AC - ATTACKBONUS {
            result: CRITABLEDAM + HITDAM
            }
                else {
                result: 0
                }
}

function: attack ROLL:n plus BONUS:n vs AC:n hit HIT:d miss MISS:d crit CRIT:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: MISS
    }
    if ROLL >= CRIT_RANGE {
        result: CRIT
    }
    if ROLL + BONUS >= AC {
        result: HIT
    }
    result: MISS
}



DART: [roll 1d20 at plus PROF_BONUS + DEX_MOD vs ac AC for 1d4 plus DEX_MOD+2]

output DART+HCB named "Soul Captcha level [LEVEL] [Dart + Hand crossbow]"

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: attackrolloh ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGEOH } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMODOH + ACCBONUSOH >= AC { result: DAMAGEOH }
 result: 0
}

function: attackrollnocrit ROLL:n vs AC:n {
 if ROLL = 1 { result: 0 }

 result: 0
}

 if ROLL + PROFICIENCY + ATTACKMODOH + ACCBONUSOH >= AC { result: DAMAGEOH }
 result: 0
}

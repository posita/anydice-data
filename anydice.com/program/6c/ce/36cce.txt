function: pass ROLL:n plus MOD:n {
  if ROLL = 20 { result: 0 }
  else if ROLL + MOD >= 10 { result: 1 }
  result: 0
}

function: fail ROLL:n plus MOD:n {
  if ROLL = 1 { result: 0 }
  else if ROLL + MOD < 10 { result: 1 }
  result: 0
}

ROLL: d20
MOD: 0

output [pass ROLL plus MOD] named "pass without crit"
output [fail ROLL plus MOD] named "fail without crit"
output ROLL = 20 named "crit pass"
output ROLL = 1 named "crit fail"

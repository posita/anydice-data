D_SIX: {-1,0:4,1:4,2}
D_SEVEN: {-1,0:5,1:3,2}
D_EIGHT: {-1,0:6,1:2,2}
D_NINE: {-1,0:7,1,2}

function: vampire POOL:s {
FAIL: (POOL = -1)
SUCC: (POOL = 1)
CRIT: (POOL = 2)

FAIL: FAIL - CRIT
if FAIL<0 { FAIL: 0 }

result: SUCC+CRIT-FAIL
}

function: soak POOL:s DIFF:n{
SUCC: POOL>=DIFF
result: SUCC
}

WILLPOWER: 1
EXTRA_ACTION: 1

output [vampire (6+2+2-EXTRA_ACTION )dD_NINE] + WILLPOWER named "Attack"
output [vampire (10+3-EXTRA_ACTION)dD_SIX] - [soak 5d10 6] + WILLPOWER named "Damage"

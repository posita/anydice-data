    result: ( A - 10 ) / 2
}

function: check A B {
    if B > 0 {
        roll1: 1d20
        roll2: 1d20
        if roll1 > roll2 {
            result: roll1 + A
        } else {
            result: roll2 + A
        }
     } else {
        result: 1d20 + A
     }
}

    roll: [check A false]
    if B > 0 {
        result: roll + PROF
    } else {
        result: roll
    }
}

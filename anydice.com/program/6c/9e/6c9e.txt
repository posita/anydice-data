function: dicepool SEQ:s with target T:n and botching at C:n {
  RES: SEQ >= T
  if RES = 0 {
    if (SEQ <= C) > 0 { result: -1 }  \ -1 = "Critical fail" \
    \ If no successes and you get results less than or equal to C, crit-fail \
  }
                        
  result: RES
}


loop N over {1..10} { \ Number of dice \
  output [dicepool Nd10 with target 6 and botching at 1]
}

\ Critical fail = "-1" \
\ No successes = "0" \
\ The rest are just number of successes \

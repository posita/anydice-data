
function: attr A {
    result: ( A - 10 ) / 2
}

function: check A with B {
    X: 1d20
    Y: 1d20
    if !B | X > Y {
        result: X + A
    }
    else {
        result: Y + A
    }
}

function: skill A plus B {
    if B {
        result: [check A with 0] + P
    }
    else {
        result: [check A with 0]
    }
}

function: dicepool SEQ:s with target T:n and critting at C:n {
  RES: SEQ >= T
  if RES = 0 {
    if (SEQ <= C) > 0 { result: -1 } 
                        \ -1 = "Critical fail" \
  }
  result: RES
}


loop N over {1..10} { \ Number of dice \
  output [dicepool Nd10 with target 8 and critting at 1]
}

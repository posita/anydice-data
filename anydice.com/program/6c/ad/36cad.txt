function: wodexp E:n again D:n {
 if D >= E { result: 1 + [wodexp E again d10] }
 result: D >= 7
}

function: nwod R:n again N:n {
 if N >= R { result: 2 + [wodexp R again d10] }
 if N = 1 { result: -1 }
 result: N >= 7
}

NWOD: [nwod 10 again d10]

loop N over {1..19} {
 output [lowest of 10 and NdNWOD] named "[N]d"
}

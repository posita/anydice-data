function: pass ROLL:n {
  if ROLL = 20 {
    result: 0
  }
  else if ROLL >= 10 {
    result: 1
  }
  result: 0
}

function: fail ROLL:n {
  if ROLL = 1 { result: 0 }
  else if ROLL < 10 { result: 1 }
  result: 0
}

ROLL: d20

output [pass ROLL] named "pass without crit"
output [fail ROLL] named "fail without crit"
output ROLL = 20 named "crit pass"
output ROLL = 1 named "crit fail"

\**** Damage calculator normal Vs advantage ****\

function: pwrattackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: PWRCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: PWRDAMAGE }
 result: 0
}
function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD >= AC { result: DAMAGE }
 result: 0
}

ATT: 1d20
ADV: [highest 1 of 2d20]

ATTACKTIMES: 1

CRITDAMAGE: 4d6 + ATTACKMOD + ATTACKMOD
PWRDAMAGE: 2d6 + ATTACKMOD + DMGBONUS
PWRCRITDAMAGE: 4d6 + ATTACKMOD + ATTACKMOD + DMGBONUS

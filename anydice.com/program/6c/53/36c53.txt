SEI:  1d{1,2,3,4,0,0}
QUAT: 1d{0,2,3,4}

function: N:n piu alti di A:s B:s C:s {
 SORTED: [sort {A,B,C}]
 R: 0
 loop M over {1..N}{
  R: R + M@SORTED
 }
 result: R
}

function: sfiga al VAL:n o meno su un DADO:n {
 R: 0
 if DADO <= VAL { R: DADO }
 if R > 0 { if R < 3 { R: 3 } }
 result: - R
}

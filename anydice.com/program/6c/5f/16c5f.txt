function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD >= AC { result: DAMAGE }
 result: 0
}

function: athleticroll ROLL:n vs ATHLETIC:n {
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD >= ATHLETIC { result: 1 }
 result: 0
}

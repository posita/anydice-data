function: attack ROLL:n plus TOHIT:n vs AC:n for DICEDMG:n plus BONUSDMG:n {
 if ROLL = 1 { 
  result: 0
 }
 if ROLL >= 20 {
  result: 2 * DICEDMG + BONUSDMG
 }
 if ROLL + TOHIT >= AC {
  result: DICEDMG + BONUSDMG
 }
 result: 0
}


loop AC over {10..25} {
 output [attack d20 plus 10 vs AC for 2d10+2d8 plus 2*8+2] named "[AC]"
}

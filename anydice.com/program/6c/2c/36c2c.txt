function: reroll D:d under N:n {
    result: [rerollcore D die D under N]
}

function: rerollcore R:n die D:d under N:n{
    if R < N {result: D} else {result: R}
}

GAXE: d12
GWGAXE: [reroll d12 under 3]
GSWORD: 2d6
GWGSWORD: [reroll d6 under 3] + [reroll d6 under 3]
LSWORDV: d10 + 1
GWLSWORDV: [reroll d10 under 3] + 1

RITE: d6
STR: 4

output 2*(GAXE+RITE+STR) named "Double Attack Axe"
output 2*(GSWORD+RITE+STR)named "Double Attack Sword"
output 2*(LSWORDV+RITE+STR) named "Double Attack Longsword"
output 2*(GWGAXE+RITE+STR) named "Double Attack Axe - GW"
output 2*(GWGSWORD+RITE+STR)named "Double Attack Sword - GW"
output 2*(GWLSWORDV+RITE+STR) named "Double Attack Longsword - GW"

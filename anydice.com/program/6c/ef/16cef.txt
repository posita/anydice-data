function: drop VAL:n from DICE:s at most MAXTIMES:n times {
 COUNT: DICE = VAL
 if COUNT < MAXTIMES {
  result: DICE - VAL*COUNT
 }
 result: DICE - VAL*MAXTIMES
}

function: fudge with N:n custom advantages {
 result: [drop -1 from (4+N)d{-1..1} at most N times]
}

function: DICE:d with N:n custom disadvantages {
 result: [drop 1 from (4+N)d{-1..1} at most N times]
}


output 4d{-1..1} named "Fudge die"

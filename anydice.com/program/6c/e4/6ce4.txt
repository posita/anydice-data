  if T < RR {      \ Countdown \
   SUCS: SEQ = MAX \ Count successes \

   result: SUCS + [keep helper ((#SEQ - SUCS)dDIE) (T+1)]
                   \ Successes + the remainder of the dice rerolled, increase the timer \
   }
  result: 0
}

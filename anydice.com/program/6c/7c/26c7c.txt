function: reroll lowest of ROLL:s as REROLL:d if less than MIN:n {
    LOWEST: (#ROLL)@ROLL    \sort lowest die last\
    if LOWEST >= MIN {
      result: ROLL + 0}
    else {
      REST: {1..#ROLL-1}@ROLL   \all but lowest die \
      result: REST + REROLL
    }
}   
output 1d4 named "1d4"    
output [reroll lowest of 1d4 as d4 if less than 3] named "1d4 reroll 1 and 2, take reroll"

output 1d10 named "1d10"
output [reroll lowest of 1d10 as d10 if less than 3] named "d10 reroll 1 and 2, take reroll"

function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 8
 if SUCCESSES > 0 { result: SUCCESSES + (ROLL = 10) }
 if [count 1 in ROLL] { result: -1 }
 result: 0
}

loop DICE over {1..10} {
 output [evaluate DICE d10] named "[DICE]d"
}

loop DICEE over {1..10} {
 output [DICEE d{-1..1} + 0] named "[DICEE]d"
}

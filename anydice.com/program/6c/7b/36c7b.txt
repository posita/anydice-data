\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}
function: savedmg ROLL:n {
 if ROLL + SAVE < 8 + PROFICIENCY + ATTACKMOD + DCBONUS { result: FULLDMG } 
 result: SAVEDMG
}

DIE: d4
PROF: 4
STAT: 3
MOD: 2
EXTRA: d8 + 2d6
function: hit D:n{
 if D = 20 {result: DIE + DIE + STAT + MOD + EXTRA + EXTRA}
 result: DIE + STAT + MOD + EXTRA
}

function: ac T:n vs D:n{
 if D = 1 | D + STAT + PROF + MOD < T {result: 0}
 result: [hit D]
}
loop I over {13..18}{
 output [ac I vs d20] named "No Mod, AC [I]"
 MOD: MOD + 1
 output [ac I vs d20] named "+1 Mod, AC [I]"
 MOD: MOD - 1
}

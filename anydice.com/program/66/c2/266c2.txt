function: fu P:s pos N:s neg {
  \ for each positive die... \
  loop PD over {1..#P} {
    \ check to see if there are more of this die in the positive pool than in the negative pool \
    if [count PD@P in P] > [count PD@P in N] {
      \ if there is, this is the result of the roll. Else, move on to the next die \
      result: PD@P
    }
  }
  \ if all dice have been cancelled out... \
  result: 1
}

output [fu 3d6 pos 2d6 neg]

function: petrify SAVE:d {
  result: [petrifyhelper SAVE SAVE 0 0]
}

function: petrifyhelper SAVE:d ROLL:n S:n F:n {
  if S & !F { result: 1 }
  if S = 3 { result: 1 }
  if F = 3 { result: 0 }
  if ROLL { result: [petrifyhelper SAVE SAVE S+1 F] }
  result: [petrifyhelper SAVE SAVE S F+1]
}

loop DC over {1..20} {
  output [petrify 1d20>=DC] named "[DC] required"
}

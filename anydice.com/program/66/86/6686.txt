function: highest NUMBER:n of DICE:d {
 result: {1..NUMBER}@DICE
}

function: ROLL:n saves against DC with DAMAGE {
 if ROLL >= DC {
  result: DAMAGE / 2
 }

 result: DAMAGE
}

function: sign of NUMBER:n {
  if NUMBER > 0 { result: +1 }
  if NUMBER < 0 { result: -1 }
  result: 0
}
function: ROLL:n reroll zero as REROLL:d {
  if ROLL != 0 { result: ROLL }
  else { result: REROLL }
}
function: ROLL:d reroll zero {
  loop I over {1..20} {
    ROLL: [ROLL reroll zero as ROLL]
  }
  result: ROLL
}

ROOKIE: d20 + 0
ENEMY: d20 + 2

WINNER: [sign of ROOKIE - ENEMY] 
output WINNER named "single round: +1 = Rookie win, -1 = Enemy win, 0 = tie"

ROUND: [WINNER reroll zero] > 0
output ROUND named "single round with rerolls: 1 = Rookie win, 0 = Enemy win"

output 5dROUND named "total Rookie wins over 5 rounds"

output 5dROUND >= 3 named "probability of Rookie winning at least 3 rounds out of 5"

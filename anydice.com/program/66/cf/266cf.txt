D: d{-1, 0, 0, 0, 1, 2}
AD: d{-1, 0,0, 1, 1, 2}
DA: d{-1, 0, 0, 0, 0, 2}

function: stage A:n {
  if A < 1 {
     result: 0
  }
  if A < 4 {
     result: 1
  }
  if A < 6 {
     result: 2
  }
  result: 3
}

loop N over {2..12} {
   output [stage NdDA] named "[N] Disadvantage"
   output [stage NdD] named "[N] Normal"
   output [stage NdAD] named "[N] Advantage"
}

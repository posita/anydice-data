set "maximum function depth" to 5
function: limit N:n {
 if N >= 15 {result: 15}
 if N <= 7 {result: 7}
 result: N
}
function: rolllimit N:n {
 if N <= 15 & N >= 7 {result: N}
 result: [rolllimit 3d{2..6}]
}

output [rolllimit 3d6 + 3] named "3d6 + 3, rolllimit"
output [limit 3d6 + 2] named "3d6 + 2, limit"
output [highest 3 of 4d6] named "4d6, drop 1"

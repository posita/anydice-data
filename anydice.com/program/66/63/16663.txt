MISSDAMAGE: 0

\ adjust threat range \

THREAT: 17

\ set confirm bonus from feats like "Critical Focus" \

CRITICALCONFIRM: 0

\ Attack Roll logic \
MISS: 0
NORMALHIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS } \ Auto miss on nat 1 \
 if ROLL >= THREAT { \ If crit threat, need to confirm \
  result: [threat d20]
 }
 if ROLL + ATTACK >= DEFENSE { result: NORMALHIT } \ If not crit threat or auto miss, compare to AC to see if hit \
 result: MISS
}

function: threat ROLL:n {
 if ROLL = 1 { result: NORMALHIT } \ Nat 1 on confirmation roll doesn't confirm \
 if ROLL + ATTACK + CRITICALCONFIRM >= DEFENSE | ROLL = 20 { \ Nat 2 on confirmation roll auto confirm \
  result: CRITICALHIT
 }
 result: NORMALHIT
}
function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT { \ if you crit, deal extra damage \
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = NORMALHIT { \ otherwise deal normal damage \
  result: HITDAMAGE
 }
 result: MISSDAMAGE \ no damage if you miss \
}
function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}
function: multiattack NUMATKS:n {
  TOTAL: 0
  loop N over {1..NUMATKS} { \ Attack N times and calculate total damage \
    ATTACK: ATTACK-(5*(N-1))
    DAMAGE: [attack]
    TOTAL: TOTAL + DAMAGE
  }
  result: TOTAL
}

\ magus spellstrike logic \
function: spellstrike { 
 result: [ssdamage for [attack d20 vs DEFENSE]]
}
function: ssdamage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT { 
  result: CRITICALDAMAGE+2*SPELLDAMAGE 
  \ Spellstrike uses the crit range of your weapon but never does more than 2x spell damage \
 }
 if ATTACKRESULT = NORMALHIT {
  result: HITDAMAGE+SPELLDAMAGE
  \ On a normal hit, spellstrike does weapon damage + spell damage \
 }
 result: MISSDAMAGE
}

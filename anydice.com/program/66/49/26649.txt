H: 2*12 + 7
W: 35

H_MOD: 2d4
W_MOD: 1

H_MIN: #(H_MOD)
H_MAX: [maximum of H_MOD]
H_AVG: ([maximum of H_MOD]+#(H_MOD))/2

W_MIN: H_MIN * #(W_MOD)
W_MAX: H_MAX * [maximum of W_MOD]
W_AVG: H_AVG * ([maximum of W_MOD]+W_MIN)/2

output H + H_MIN named "Min Height"
output H + H_MAX named "Max Height"
output H + H_AVG named "Average Height"

output W + W_MIN named "Min Weight"
output W + W_MAX named "Max Weight"
output W + W_AVG named "Average Weight"

EVASION: 10
CRITDMG: [ highest 1 of 2d(2d6) ] + 4
NORMALDMG: 2d6+4
MISSDMG: 0
ATTACKDIE: 1d20
DISADVANTAGE: 1
ADVANTAGE: 0


function: attack with ROLL:n DISADVANTAGEMOD:n ADVANTAGEMOD:n {
  FINALROLL: ROLL - DISADVANTAGEMOD + ADVANTAGEMOD
  if (FINALROLL >= 20) {
    result: CRITDMG
  }
  else if (FINALROLL = 20)
  {
    result: CRITDMG
  }
  else if (FINALROLL >= EVASION)
  {
    result: NORMALDMG
  }
  result: MISSDMG
}

function: allpossibleattackresults {
  DISADVANTAGEVALUE: [highest 1 of (DISADVANTAGE)d6]
  ADVANTAGEVALUE: [highest 1 of (ADVANTAGE)d6]
  result: [attack with ATTACKDIE DISADVANTAGEVALUE ADVANTAGEVALUE]
}

output [allpossibleattackresults]
output 1d20 - 1d6

set "maximum function depth" to 5

function: limit N:n {
 if N >= 15 {result: 15}
 if N <= 7 {result: 7}
 result: N
}
function: rolllimit N:n {
 if N <= 15 & N >= 7 {result: N}
 result: [rolllimit 3d{2..6}]
}

output [highest 3 of 4d6] named "4d6, drop 1 lowest"
output [highest 2 of 3d{2..6}] +4 named "3d6 + 4, reroll 1s, drop 1 lowest"
output [limit 3d{2..6}] named "3d6, reroll 1s, limit"
output [rolllimit [highest 3 of 4d{2..6}]] named "4d6D1, reroll 1s, rerolllimit"

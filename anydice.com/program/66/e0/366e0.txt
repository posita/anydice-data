function: size of ROLL:s without TN:n plus {          \ Remove Xs & above from the Hits roll \
 COUNT: 0
 loop N over {1..#ROLL} {
   if(N@ROLL < TN){ result: #ROLL -N+1 }
 }
 result: 0
}

function: compare X:n and Y:n {          \ Find nbr of dices for Saves Roll from comparison table using -1/0/+1 \
 if(X > Y){ result: 1 }
 if(X < Y){ result: -1 }
 result: 0
}

function: compare doubled X:n and Y:n {   \ Find nbr of dices for Saves Roll from comparison table using -1/0/+1 \
 if(X > 2*Y){ result: -1 }
 if(X < Y/2){ result: 1 }            \ !warning! in AnyDice its rounded down ! \
 result: 0
}

\*****************************************************************************\
function: find saves type SV:n from HITSROLL:s using U_S:s {             \find nbr of dices used to roll the saves\
 if(SV=0){result: #HITSROLL}                                      \ 0: reuse same nbr of dices than Hits Roll \
 if(SV=1){result: [size of HITSROLL without 3 plus]}              \ 1: remove 3s from the Hits Roll \
 if(SV=2){result: [size of HITSROLL without 2 plus]}              \ 2: remove 2+s from the Hits Roll \
 if(SV=3){result: #HITSROLL-[size of HITSROLL without 3 plus]}    \ 3: use only 3s from the Hits Roll \
 if(SV=4){result: [window crop 1@U_S - 4@U_S]}                    \ 4: M-BV (=0 if under 0) \
 if(SV=5){                                                     \ 5: Compare SizeUnit vs SizeUnit & Body \
   SOURCES: [compare 6@U_S and 7@U_S]
   WEIGHT: [compare doubled 2@U_S and 3@U_S]
   result: (#HITSROLL)+SOURCES+WEIGHT}
 result: 0
}

function: find saves modifier type SV:n using U_S:s {     \find the save modifier\
 if(SV=0){result: 0}                                      \ 0: None \
 if(SV=1){result: [window crop 4@U_S - 5@U_S]}                           \ 1: Velocity Defender - Velocity of the Attacks \
 result: 0
}

function: window crop SEQUENCE:n {                  \ limit the display range between [-1;100] \
 MIN: 0
 MAX: 100
 if (SEQUENCE < MIN){result: MIN}
 if (SEQUENCE > MAX){result: MAX}
 result: SEQUENCE
}

function: roll hits HITS:d {       \ roll Hits Roll \
 result: HITS
}
function: roll saves SAVES:d {
 result: SAVESdQ
}
function: get unsaved hits from HITSROLL:s and SAVESROLL:s {                 \ get Unsaved Hits \
 result: HITSROLL-[window crop SAVESROLL]
}
function: apply modifiers HITSMOD:n and SAVESMOD:n to SEQ:s {          \ apply All modifiers \
 result: SEQ+HITSMOD+SAVESMOD
}

function: roll HITSROLL:n {
 result: 4@HITSROLL
}

function: do combat up to step NBR:n with DATAS:s and UNIT_STATS:s {
 S_UN: [roll hits (1@DATAS)dQ]
 S_DEUX: [find saves type (4@DATAS) from S_UN using UNIT_STATS]
 S_TROIS: [roll saves S_DEUX]
 S_QUATRE: [get unsaved hits from S_UN and S_TROIS]
 S_CINQ: [apply modifiers (2@DATAS) and [find saves modifier type 1 using UNIT_STATS] to S_QUATRE]
 S_SIX: S_CINQ*((5@DATAS)dD+(6@DATAS))
 if(NBR=1){result: S_UN}                          \roll hits-roll\
 if(NBR=2){result: S_DEUX}                        \find save\
 if(NBR=3){result: S_TROIS}                       \roll save\
 if(NBR=4){result: [window crop S_QUATRE]}        \get the Unsaved Hits (limited from 0 to 100)\
 if(NBR=5){result: [window crop S_CINQ]}          \apply modifiers\
 if(NBR=6){result: [window crop S_SIX]}           \add bonus Damages\
 result: 0
}

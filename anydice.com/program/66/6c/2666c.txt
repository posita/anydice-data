function: special R:s mastery M:n {
  \ First, sum the dice \
  BASE: R
  \ Now, if there was at least 1 Skull, add 1 back into cancel its -1 \
  if #R@R = -1 {
    BASE: BASE+1
  } 
  \ Now, if there are more than M 2's in the list, subtract 1 for each of them so they count as regular successes \
  BASE: BASE - [highest of ([count 2 in R]-M) and 0]
  result: BASE
}

output [special 8d{-1, 0, 1, 1, 2, 2} mastery 2]

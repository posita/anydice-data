\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 16-19\
ENEMYAC: d{16..19}
SNEAKATK: 5d6

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 4
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: (1d8)*2 + ATTACKMOD + DMGBONUS
DMGRESULT: [attackroll d20 vs ENEMYAC]
output ATTACKTIMESdDMGRESULT named "normal attack"

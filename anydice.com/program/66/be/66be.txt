POOL: 10
BOTCHES: 0
TOTAL: 0
function: roll{
  loop N over {1..[POOL]}
  {
    X: d{-1,0:5,1,1,1,2}
    if X=-1{BOTCHES: BOTCHES+1}
    if X=1{TOTAL: TOTAL+1}
    if X=2{TOTAL: TOTAL+2}
  }
  if TOTAL>0 {
    TOTAL: TOTAL
  }
  else {
    if BOTCHES=0{
      TOTAL: 0
    }
    else{
      TOTAL: -1
    }
  }
  TOTAL: TOTAL
}

output TOTAL

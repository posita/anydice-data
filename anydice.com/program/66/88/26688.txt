function: rolldw S:s dw N:n {
  if(1@S-N<3@S) { result: 3@S }
   if(1@S-N<2@S) { result: 1@S-N }
 result: 2@S
}

function: rollup S:s up N:n {
  if(3@S+N>1@S) { result: 1@S }
   if(3@S+N>2@S) { result: 3@S+N }
 result: 2@S
}

loop N over {10,8,6,4,2}{
 output [rolldw 3d20 dw N] named "Modificador de atributo -[N]"
}

loop N over {0,2,4,6,8,10}{
 output [rollup 3d20 up N] named "Modificador de atributo +[N]"
}

function: fractalize DIE:s {
    if 1@DIE = 2@DIE {
        result: DIE + [fractalize helper 1d4 1@DIE]
    }
    result: DIE
}

function: fractalize helper DIE:s X {
    if DIE = X {
        result: DIE + [fractalize helper 1d4 X]
    }
    result: DIE
}

output 1d4 named "1st Level"
output 1@2d4 named "4th Level"
output [fractalize 2d4] named "8th Level"

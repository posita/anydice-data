\ The longest match starting with triples, otherwise the number of pairs. \

function: match ROLL:s {
 P: 0
 L: 0
 LONGEST: 1
 PAIRS: 0
 loop N over ROLL {
  if N = P {
   L: L + 1
  }
  else {
   if L > LONGEST {
    LONGEST: L
   }
   if L = 2 {
    PAIRS: PAIRS + 1
   }
   L: 1
   P: N
  }
 }
 
 if L > LONGEST {
  LONGEST: L
 }
 if LONGEST > 2 {
  result: LONGEST
 }
 
 if L = 2 {
  PAIRS: PAIRS + 1
 }
 result: PAIRS
}

output [match 6d6]

function: roll N:s base D:d{
COUNT: 0
loop P over {1..#N} {
COUNT: COUNT + (P@N > 5)
}
EXPLOSIONS: [count {6..20} in COUNTdD]
result: COUNT + EXPLOSIONS
}

output [roll 1d6 base d6] named "1d6 "

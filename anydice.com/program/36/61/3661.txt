function: ROLL:s blocked by POWER:s {
R: 0
PP: 1
loop P over {1..#ROLL} {
if P@ROLL < 4 { result: R }
if P@ROLL > PP@POWER { R: R + 1 }
if P@ROLL = PP@POWER { R: R + 1 }
else { PP: PP + 1 }
}
result: R
}

function: A:s p PA:s vs B:s p PB:s {
result: [A blocked by PB] + (PA > 3)
}

\output [d6 p 2d6 vs 2d6 p d6]\

output [2d6 p 0 vs d6 p 0]

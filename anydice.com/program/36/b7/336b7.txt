function: eadv {result: [highest of 1d20 and [adv]]}
function: gwf N:n {result: 1d{1dN,1dN,3..N:N}}

DICEDAMAGE: 1d6
FLATDAMAGE: 5+2+10
HITBONUS: 5+4+2+6-5
AC: 17
ATTACKS: 3
OPROHB: 0
CRIT: 20
CRITB: 0



function: attack B:n HIT:n DD:d FD:n {
  if HIT >= CRIT {
    \on crit\
    result: DD + FD + DD + CRITB
  }
  if HIT + B >= 9 {
    \on hit\
    result: DD + FD
  } else {
    \on miss\
    result: 0
  }
}



function: attackac B:n AC:n HIT:n DD:d FD:n {
  result: [attack B-AC+9 HIT DD FD]
}

function: oproh OHB:n ATK:n {
  if ATK = 0 {
    result: 0
  }
  result: ATK + OHB * (1 + (1d20>=CRIT))
}

function: atk {
   result: [attack HITBONUS 1d20 DICEDAMAGE FLATDAMAGE]
}

function: atkh H:n HIT:d {
   result: [attack HITBONUS+H 1d20 DICEDAMAGE FLATDAMAGE]
}

function: atkd D:d HIT:d {
   result: [attack HITBONUS HIT DICEDAMAGE+D FLATDAMAGE]
}

function: aatk {
   result: [attack HITBONUS [adv] DICEDAMAGE FLATDAMAGE]
}


function: eatk {

}

function: atkacd D:d HIT:d {

   CRITB: 0

\function: rr one D:n DD:d {
  if D = 1  {
    result: DD
  } 
  result: D
}
function: two one D:n {
  if D = 1  {
    result: 2
  } 
  result: D
}\

\function: piercercorrection A:n B:n AT:d BT:d {
  if A < B {
    result: AT-A
  }
  result: BT-B
}

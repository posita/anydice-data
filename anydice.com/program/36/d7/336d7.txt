\**** Damage calculator ****\

function: bless N:n {
 result:Nd4
}

function: attackroll ROLL:n vs AC:n with BLESS:n{
 
 if ROLL >= CRITROLL { result: DAMAGE + CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS+ BLESS >= AC { result: DAMAGE }
 result: 0
}

function: dragonattackroll ROLL:n vs AC:n with BLESS:n{
 
 if ROLL >= DRAGONCRITROLL { result: DRAGONDAMAGE  + DRAGONCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + DRAGONATTACKMOD + BLESS >= AC { result: DRAGONDAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
\ENEMYAC: d{15..20}\
ENEMYAC: 15

CRITROLL: 20
DRAGONCRITROLL: 20 


\-----\
ATTACKMOD: 3
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
BLESSDICE: 0
DAMAGE: 2d8 + 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + 1d6
DRAGONATTACKMOD: 3
DRAGONDAMAGE: 1d6 + PROFICIENCY
DRAGONCRITDAMAGE: 1d6
output [attackroll d20 vs ENEMYAC with [bless BLESSDICE]] + [dragonattackroll d20 vs ENEMYAC with [bless BLESSDICE]]named "Mark + Feat"

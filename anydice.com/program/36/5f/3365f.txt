function: newsys PROF:s {

  TPROF: [count {3,6,6} in PROF]
  TOTAL: TPROF

  if TOTAL > 2 {LEVEL: 3}
  if TOTAL = 2 {LEVEL: 2}
  if TOTAL = 1 {LEVEL: 1}
  if TOTAL <= 0 {LEVEL: 0}

  result: [lowest of LEVEL and 3]

}

loop N over {1..4} {
  output [newsys Nd6] named "[N]d6"
}

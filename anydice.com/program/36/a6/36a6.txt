SIDES: 8
NOOFDICE: 6
NOOFSUCCESSES: 0
RATING: 4

set "maximum function depth" to 3

function: explode ROLLEDVALUE:n {
  if ROLLEDVALUE = SIDES {
    TEMPB: dSIDES
    NOOFSUCCESSES: [explode TEMPB]
    NOOFSUCCESSES: NOOFSUCCESSES + (TEMPB >= RATING)
  }
  result: NOOFSUCCESSES
}

loop N over {1..NOOFDICE}{
  TEMP: dSIDES
  NOOFSUCCESSES: [explode TEMP]
  NOOFSUCCESSES: NOOFSUCCESSES + (TEMP >= RATING)
}

output NOOFSUCCESSES

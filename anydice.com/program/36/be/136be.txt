MAXREROLL: 3
NUMDICE: 3

function: withreroll ROLL:s {
  if NUMDICE@ROLL <= MAXREROLL {
    result: ROLL + d6 - NUMDICE@ROLL
  } else {
    result: ROLL
  }
}

function: rerollall ROLL:n {
  THRESHOLD: (NUMDICE*7)/2
  if ROLL <= THRESHOLD{
    result: NUMDICEd6
  } else {
    result: ROLL
  }
}

output [withreroll NUMDICEd6] named "Reroll one"
output [rerollall  NUMDICEd6] named "Reroll all"

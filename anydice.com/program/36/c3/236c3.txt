function: roll A:n plus B:n vs D:n {
 if A=16 { result: 4 }
 if A=2 { result: 1 }
 if A+B+A+B<=D { result: 1 }
 if A<5 { result: 2 }
 if A+B>=D+D { result: 4 }
 if A+B>=D { result: 3 }
 if A+B<D { result: 2 }
}

}}

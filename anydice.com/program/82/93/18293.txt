function: ROLL:n gwf REROLL:d reroll SEQUENCE:s{
  if [SEQUENCE contains ROLL] {
    result: REROLL
  }
  else {
    result: ROLL
  }
}

STR: 3
DIE_ONE: [d6 gwf d6 reroll {1,2}]
DIE_TWO: [d6 gwf d6 reroll {1,2}]
DIE_THREE: [d10 gwf d10 reroll {1,2}]
DIE_FOUR: [d4 gwf d4 reroll {1,2}]

output (DIE_ONE+DIE_TWO+STR)*2 named "Greatsword"
output (DIE_FOUR+DIE_FOUR+STR)*2 named "Double-Scimitar"
output (DIE_THREE+STR)*2 named "Halberd"
output DIE_FOUR+STR named "Bonus Attack"
output (1d8+2+STR)*2 named "Flail"
output (1d4+2+STR)*2 named "Whip"

function: mod NUMBER:n MODULUS:n {
 if NUMBER >= MODULUS {
  result: [mod (NUMBER - MODULUS) MODULUS]
 }
 else {
  result: NUMBER+1
 }
}

function: verl ONE:n TWO:n THREE:n {
 TOTAL: ONE+TWO+THREE
 if ONE >= TWO {
  if THREE >= TWO {
   result: [mod TOTAL-ONE 6]+[mod TOTAL-THREE 6]
  }
  else {
   result: [mod TOTAL-ONE 6]+[mod TOTAL-TWO 6]
  }
 }
 else {
  if THREE >= ONE {
   result: [mod TOTAL-TWO 6]+[mod TOTAL-THREE 6]
  }
  else {
   result: [mod TOTAL-TWO 6]+[mod TOTAL-ONE 6]
  }
 }
}

output [verl 1d6 1d6 1d6] named "Verl"
output 2d6 named "Normal 2d6"


loop SK over {1..9}{
UP:SK+1 UQ:SK+2 SJ:SK-1
  output (d10<=SK)+SJd(d12<=SK)   named "Sk[SK] -1D"
  output (d10<=SK)+SKd(d12<=SK)   named "Sk[SK] reg"
  output (d10<=SK)+SKd(d12<=UP)   named "Sk[SK] +1L"
  output (d10<=SK)+UPd(d12<=SK)   named "Sk[SK] +1D"
  output (d10<=SK)+SKd(d12<=UQ)   named "Sk[SK] +2L"
  output (d10<=SK)+SKd(d12<=SK)+1 named "Sk[SK] +1S"
}

output 14 named ""

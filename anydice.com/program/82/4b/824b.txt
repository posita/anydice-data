function: mod NUMBER:n MODULUS:n {
 if NUMBER >= MODULUS {
  result: [mod (NUMBER - MODULUS) MODULUS]
 }
 else {
  result: NUMBER
 }
}

function: reroll MOD:n {
 if MOD = 1 {
  result: [reroll [mod 3d6 12]+1]
 }
 else {
  result: MOD
 }
}

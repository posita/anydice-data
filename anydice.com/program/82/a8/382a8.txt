function: sign of NUMBER:n {
  if NUMBER > 0 { result: +1 }
  if NUMBER <= 0 { result: -1 }
  result: 0
}

ROOKIE: 2d10 + 2
ENEMY: 2d10 + 0

WINNER: [sign of ROOKIE - ENEMY] 
output WINNER named "+1 = ATQ | -1 = DEF"

function: restrict ROLL:n to RANGE:s {
  if ROLL = RANGE { result: ROLL }
  else { result: d{} }
}

ROUND: [restrict WINNER to {+1,-1}] = +1
output ROUND named "+1 = ATQ | -1 = DEF"

output 5dROUND named "number of ATQ wins over 5 rounds"

output 5dROUND >= 3 named "probability of ATQ winning at least 3 rounds out of 5"

function: damage from AV:s attack with DAM:n damage against DEF:n defense {
 DAMAGE: 0
 ATTACK: 0
 
 loop P over {1..#AV} {
  if P@AV > 2 {
   ATTACK: ATTACK + 1
  }
  if P@AV > 4 {
   DAMAGE: DAMAGE + 1
  }
 }
 
 if ATTACK >= DEF {
  result: DAMAGE + DAM
 }

 result: 0

}

C: 1

loop A over {1..7} {
 loop B over {0..A} {
  output [damage from Ad6 attack with C damage against B defense] named "Attack +[A], damage [C] against defense [B]"
 }
}

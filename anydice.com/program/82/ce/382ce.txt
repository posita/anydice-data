
function: POOL dice with dif DIF {
  SEQ: {}

  loop N over {1..6}{
    if DIF >= N {
      SEQ: {SEQ,0}
    }
    else {
      SEQ:{SEQ,1}
    }
  }
  result: POOLdSEQ
}

loop DIF over {3..5}{
  loop POOL over {3..9}{
    output [POOL dice with dif DIF] named "Pool de [POOL] dados com dificuldade [DIF]"
  }
}

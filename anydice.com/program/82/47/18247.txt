function: ROLL:n gwf REROLL:d reroll SEQUENCE:s{
  if [SEQUENCE contains ROLL] {
    result: REROLL
  }
  else {
    result: ROLL
  }
}

STR: 3
DIE_ONE: [d6 gwf d6 reroll {1,2}]
DIE_TWO: [d6 gwf d6 reroll {1,2}]
DIE_THREE: [d10 gwf d10 reroll {1,2}]
DIE_FOUR: [d4 gwf d4 reroll {1,2}]

output DIE_ONE+DIE_TWO+STR named "Greatsword"
output DIE_FOUR+DIE_FOUR+STR named "Double-Scimitar"
output DIE_ONE+DIE_TWO+STR+DIE_ONE+DIE_TWO+STR named "2x Greatsword"
output DIE_FOUR+DIE_FOUR+STR+DIE_FOUR+DIE_FOUR+STR+DIE_FOUR+STR named "3x Double-Scimitar"
output DIE_THREE+STR named "Halberd"
output 1d8+2+STR named "Flail"
output 1d4+2+STR named "Whip"

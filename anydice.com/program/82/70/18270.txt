\Max Damage Calculator for 5th lvl Vash With cursed staff\
TOTAL:0
ACTOBEAT:14
ATTB:5
SLOT:3

function: orb {
  STAFF:d4 \staff effect\
  ATT:d20 \roll\
  CRIT:ATT=20
  ATT:ATT+ATTB \plus bonus\
  ATT:ATT+(3*(STAFF=3)) \if staff adds 3, add 3\
  ATT:ATT-(STAFF=4) \if staff subs 1, sub 1\
  ATT:ATT*(STAFF=1) \if staff rolls one, miss\
  DAM:(ATT > ACTOBEAT) \did it even hit\
  DAM:DAM*((2+SLOT)d8) \if it hit, do damage\
  DAM:DAM+((STAFF=2)*((2+SLOT)d8)) \if the staff twinned, do more damage\
  result:DAM*(CRIT+1) \multiply by 2 if crit\
}

TOTAL: TOTAL+[orb]+[orb] \twin chromatic orb\
output 2*TOTAL

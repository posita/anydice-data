function: SEQUENCE:s contains NUMBER:n{
 result: (SEQUENCE = NUMBER) > 0
}

function: translate A:n {
 if A >= 12 { result: 2 }
 if A >= 6 { result: 1}
 result: 0
}

function: translates SEQUENCE:s {
 SORTED: {}
 loop P over SEQUENCE {
  SORTED: {SORTED, [translate P]}
 }
 result: SORTED
}

function: SEQUENCE:s easy{
 result: (SEQUENCE >=6 ) > 0
}

function: SEQUENCE:s easyb{
 result: ([translates SEQUENCE]) > 0
}


function: SEQUENCE:s medium{
 result: (SEQUENCE >=6 ) > 1 | (SEQUENCE >=12 ) > 0
}

function: SEQUENCE:s hardunfinished{
 result: (SEQUENCE >=6 ) > 3 | (SEQUENCE >=12 ) > 1
}


function: SEQUENCE:s epicunfinished{
 result: (SEQUENCE >=6 ) >= 4 | (SEQUENCE >=12 ) > 1
}

function: min A:n {
  if A>=0 {
    result: 0
  } else {
    result: A
  }
}

function: skill check A:n AB:n B:n BB:n C:n CB:n SP:n {
  DICE: {AB,BB,CB}
  SC: [count 1 in DICE]
  SV: [count 20 in DICE]
  if SC >= 2 {
    result: 16
  } else if SV >= 2 {
    result: -1
  } else {
    result: SP+[min A-AB]+[min B-BB]+[min C-CB]
  }
}

function: quality level A:n {
  if A >= 16 {
    result: 6
  } else if A >=13 {
    result: 5
  } else if A >=10 {
    result: 4
  } else if A >=7 {
    result: 3
  } else if A >=4 {
    result: 2
  } else if A >=0 {
    result: 1
  } else {
    result: 0
  }
}

B: d20
SKL: [skill check 14 B 14 B 14 B 12]
output [quality level SKL]

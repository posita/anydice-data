\ count the dice over N, defined as 7 in the last line \
\ modified from original code so you don't reroll 10s \ 
\ they just count as two successes \
function: nwod R:n again N:n {
 if N >= R { result: 2 }
 result: N >= 7
}

\ call the nwod function.
\ change to 9 or 8 for 9-again or 8-again\
NWOD: [nwod 10 again d10]

NWODDMG: [nwod 10 again d10]

\ roll 1 to 20 dice, each as a set \ 
\ max out at 20 successes though \
loop N over {1..20} {
 output [lowest of 20 and NdNWOD] named "[N]d"
}

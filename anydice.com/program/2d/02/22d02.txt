D: 12
S: 6

function: succ D vs S{
 R: 0
 loop N over {0..S-1}{
    
  R: R | (([count {6} in Dd6] = N) & ([count {4,5} in (D-N)d6] >= ((S-N)*2)))

 }
 
 R: R | ([count {6} in Dd6] >= S)

 result: R 
}

function: fail D vs S{
 result: [count {1,2,3} in Dd6] >= (D-S+1)
}

SUCC: [succ D vs S]
FAIL: [fail D vs S]
MSUC: !SUCC & !FAIL

output FAIL named "Fallimento"

output SUCC named "Successo"

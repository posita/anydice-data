\ This is just to coerce variables. Don't refactor! https://rpg.stackexchange.com/questions/56358/how-can-i-execute-code-depending-on-a-die-value-in-anydice \
function: busted ONES:n N:n EXPLODED:d {
  \ We don't have a way to do real division and round up, so hack things! \
  if (N - ONES) < ((N+1)/2) { result: 0 }
  else { result: [highest 1 of EXPLODED] }
}
function: trait N:n S:n {
  EXPLODED: Nd [explode dS]
  result: [busted [count 1 in EXPLODED] N EXPLODED]
}

output [trait 1 4] named "1d4"
output [trait 2 4] named "2d4"
output [trait 3 4] named "3d4"
output [trait 4 4] named "4d4"
output [trait 2 6] named "2d6"
output [trait 4 6] named "4d6"

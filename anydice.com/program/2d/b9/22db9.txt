function: reroll ROLL:s NUMBER:n if less than MIN:n {
    SUM: 0
    loop N over {1..NUMBER} {
        CURRENTDIE: N@ROLL
        if CURRENTDIE < MIN {
            SUM: SUM + d6
        } else {
            SUM: SUM + CURRENTDIE
        }
    }
    result: SUM
}

MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 | ROLL + ATTACK < DEFENSE { result: MISS }
 if ROLL = 20 | ROLL + ATTACK >= DEFENSE + 10 { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE & ROLL + ATTACK <= DEFENSE + 9 { result: HIT }
}

DICEMAX: 10 \How many dice you roll\
DIEBONUS: 7 \Bonus per die\
CURRENTPROGRESS: 8 \Your current project progress\
PROGRESSTARGET: 250 \The total progress the project needs\
OMAKEBONUS: 15 \Omake bonus points\

function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 0
 if SUCCESSES > 0 { result: 1 }
 result: 0
}

loop DICE over {1..DICEMAX} {
 output [evaluate DICEd100+DIEBONUS*DICE+CURRENTPROGRESS-PROGRESSTARGET+OMAKEBONUS]
}

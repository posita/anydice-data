set "explode depth" to 3
function: dexplode N:n {
 if N = 4 { result: N + [explode 4d{0,0,1}] }
 if N = -4 { result: N + [explode 4d{-1,0,0}] }
 result: N
}
function: explode N:n {
 if N = 4 { result: N + [explode 4d{-1,0,1}] }
 if N = -4 { result: N + [explode 4d{-1,0,1}] }
 result: N
}


output 4d{-1,0,1}
output [explode 4d{-1,0,1}]
output [dexplode 4d{-1,0,1}]
output 4d{-1,0,1}-4d{-1,0,1}
output [explode 4d{-1,0,1}]-[explode 4d{-1,0,1}]
output [dexplode 4d{-1,0,1}]-[dexplode 4d{-1,0,1}]

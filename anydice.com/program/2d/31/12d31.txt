\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{14}

ATTACKMOD: 5
DMGBONUS: 0
PROFICIENCY: 4
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
GWF: 0
NUMBEROFDIE: 2
DAMAGEOFDIE: 6
DAMAGE: 2d6 + ATTACKMOD + DMGBONUS  + (GWF * (NUMBEROFDIE * ((DAMAGEOFDIE/2) - 1))/(DAMAGEOFDIE/2))
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS + 12 + GWF * (NUMBEROFDIE * (DAMAGEOFDIE/2 - 1))/(DAMAGEOFDIE/2)
output ATTACKTIMESd[attackroll [highest 1 of 1d20] vs ENEMYAC] named "Fighter lvl 10 both"

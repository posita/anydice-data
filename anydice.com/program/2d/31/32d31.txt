function: ccviejo DETALLE:n ATAQUE:s ATAQUESUPERAR:n DEFENSA:s DEFENSASUPERAR:n HERIDAS:n{
 CA: ATAQUE = 6
 CD: DEFENSA = 6

 RETAO: 0
 CRITICOS: 0
 NORMALES: 0

 if CA > 0 {
   if CD > 0 {
      if CA > CD {
         CRITICOS: (CA - CD)
         
         A: (ATAQUE > ATAQUESUPERAR) - (CA - CD)
         D: (DEFENSA > DEFENSASUPERAR) - CD
      } else {
         A: (ATAQUE > ATAQUESUPERAR) - CA
         D: (DEFENSA > DEFENSASUPERAR)- CD
      }
   } else {
      CRITICOS: CA
      A: (ATAQUE > ATAQUESUPERAR) - CA
      D: (DEFENSA > DEFENSASUPERAR)
   }
 } else {
   A: (ATAQUE > ATAQUESUPERAR)
   D: (DEFENSA > DEFENSASUPERAR)
 }

 if A > D {
    NORMALES: (A - D)
 }
 
 RA: ((CRITICOS + NORMALES) * HERIDAS) + CRITICOS
 RB: CRITICOS * 1000 + NORMALES

 if DETALLE = 0 {
   result: RA
 } else {
   result: RB
 }
}

   NA: 0

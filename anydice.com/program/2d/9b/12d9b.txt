function: silhouettish D:s lower LOWER:n {
  HIGHEST: 1@D
  OUTPUT: HIGHEST
  LAST: HIGHEST
  LASTBONUS: HIGHEST

  loop I over {2..#D} {
    if I@D = HIGHEST {
       OUTPUT: OUTPUT + 1
    }

    if LOWER > 0 & LAST < LASTBONUS & LAST = I@D {
       OUTPUT: OUTPUT + 1
       LASTBONUS: LAST
    }

    LAST: I@D
  }

  result: OUTPUT
}

loop N over {1..6} {
   output [silhouettish Nd6 lower 0] named "straight [N]"
   output [silhouettish Nd6 lower 1] named "modified [N]"
}

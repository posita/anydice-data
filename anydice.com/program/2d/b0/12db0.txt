\DMG CALC\
 function: attack ROLL:n vs AC:n
{
 if ROLL = 1 { result: 0 }
 if ROLL >= THREAT { result: CRIT } 
 if ROLL + PROF + ABIMOD + TOHIT >= AC { result: DMG }
 result: 0
}
THREAT: 20
TOHIT: 0
TODMG: 0
ABIMOD: 3
DMG: 2d6 + ABIMOD + TODMG
CRIT: DMG*2 + ABIMOD + TODMG
NUM_OF_ATTACKS: 1
PROF: 2
AC: d{12}
output NUM_OF_ATTACKSd[attack d20 vs AC] named "damage"

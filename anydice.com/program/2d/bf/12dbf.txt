\gwf reroller\
function: gwf DIE:n {GWFDIE: {{1..2}:2, {3..DIE}:DIE+2} result: dGWFDIE }
THREAT: 20
TOHIT: 0
TODMG: 0
ABIMOD: 3
DMG: 2d[gwf 6] + ABIMOD + TODMG
CRIT: 4d[gwf 6] + ABIMOD + TODMG
NUMATTACKS: 1
PROF: 2
AC: 16

\damage calculator\
function: attack ROLL:n vs AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL >= THREAT { result: CRIT } 
 if ROLL + PROF + ABIMOD + TOHIT >= AC { result: DMG }
 result: 0
}

output NUMATTACKSd[attack d20 vs AC] named "[NUMATTACKS] attacks vs [AC] AC"

function: silhouettish D:s higher HIGHER:n lower LOWER:n {
  HIGHEST: 1@D
  OUTPUT: HIGHEST
  LAST: HIGHEST
  LASTBONUS: HIGHEST

  loop I over {2..#D} {
    if HIGHER & I@D = HIGHEST {
       OUTPUT: OUTPUT + 1
    }

    if LOWER > 0 & LAST < LASTBONUS & LAST = I@D {
       OUTPUT: OUTPUT + 1
       LASTBONUS: LAST
    }

    LAST: I@D
  }

  result: OUTPUT
}

loop N over {6..6} {
   output [silhouettish Nd6 higher 1 lower 0] named "high bonus [N]"
   output [silhouettish Nd6 higher 0 lower 1] named "low bonus [N]"
   output [silhouettish Nd6 higher 1 lower 1] named "both [N]"
}

function: busted ONES:n N:n EXPLODED:d {
  \ We don't have a way to do real division and round up, so hack things! \
  if (N - ONES) < ((N+1)/2) { result: 0 }
  else { result: [highest 1 of EXPLODED] }
}
set "explode depth" to 5
N: 3
S: 8
EXPLODED: Nd [explode dS]
ONES: [count 1 in EXPLODED]
output (N-ONES) named "N-ONES"
output (N+1)/2 named "MID POINT"
output ((N - ONES) < ((N+1)/2)) named "COND"
output [busted ONES N EXPLODED] named "busted"

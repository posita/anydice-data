DIE: 1d12                              \ Change this die type to anything \
SKILLVALUE: 5                          \ Change the player's skill value to anything \
set "maximum function depth" to 2      \ Anything higher than 2 will generate a timeout \

function: realoutput F:n S:n {
  if (F = S) & (F<=SKILLVALUE = S<=Q) {  \ An excessiv if statement. If (1 or 0) AND (1 or 0) \
    result: [realoutput DIE DIE]
  }
  result: (F<=SKILLVALUE)*100+F > (S<=Q)*100+S
}

loop Q over {1..13} {                  \ Loops the opponent's skill value between 1 and 13 \
  output [realoutput DIE DIE] named "Skill value [SKILLVALUE] vs skill value [Q]"
}

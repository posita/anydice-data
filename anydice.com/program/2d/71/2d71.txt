function: best of SEQ:s {
  result: 1 @ [sort SEQ]
}

function: seqs of A:s from SEQ:s {
  MATCH:1
  loop N over A {
    MATCH : MATCH * [SEQ contains N]
  }
  result: MATCH * A
}

function: sets of A:n from SEQ:s {
   result: [count A in SEQ] * A
}

function: ctech A:s {
  result: [best of {
      [sets of 1 from A],
      [sets of 2 from A],
      [sets of 3 from A],
      [sets of 4 from A],
      [sets of 5 from A],
      [sets of 6 from A],
      [sets of 7 from A],
      [sets of 8 from A],
      [sets of 9 from A],
      [sets of 10 from A],
      [seqs of {1,2,3} from A],
      [seqs of {2,3,4} from A],
      [seqs of {3,4,5} from A],
      [seqs of {4,5,6} from A],
      [seqs of {5,6,7} from A],
      [seqs of {6,7,8} from A],
      [seqs of {7,8,9} from A],
      [seqs of {8,9,10} from A]
    }]
}

output [ctech 4d10]

\Fire Bolt (level 1-4), Chromatic Orb vs True Strike, Chromatic Orb\

BOLT: 1d20 + 5
ORB: 1d20 + 5
ADVORB: [highest 1 of 2d20] + 5

loop AC over {12..23} {
  output (1d(BOLT >= AC) * 1d10) + (1d(ORB >= AC) * 3d8) named "Fire Bolt, Chromatic Orb (AC: [AC])"
  output (1d(ADVORB>= AC) * 3d8) named "True Strike, Chromatic Orb (AC: [AC])"
}

function: reroll R:n under N:n this D {
   if R < N { result: D } else {result: R}
}

function: hit H:n mod M:n crit C dmg D bonus B extra E ac A {
   if H=1 {result: 0}
   else if H >= C {result: D+D+B+E}
   else if H+M >= A {result: D+B} 
   else {result: 0}
}

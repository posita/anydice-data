\DMG CALC\
 function: attack ROLL:n vs AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL >= THREAT { result: CRIT } 
 if ROLL + PROF + ABIMOD + TOHIT >= AC { result: DMG }
 result: 0
}
\GWF reroller\
function: explode N:n {
 if N < 3 { result: N + [explode d6] }
 result: N
}
set "maximum function depth" to 2
THREAT: 20
TOHIT: 0
TODMG: 0
ABIMOD: 3
DMG: 2d[explode d6] + ABIMOD + TODMG
CRIT: 4d[explode d6] + ABIMOD + TODMG
NUMATTACKS: 1
PROF: 2
AC: 16
output NUMATTACKSd[attack d20 vs AC] named "[NUMATTACKS] attacks vs [AC] AC"

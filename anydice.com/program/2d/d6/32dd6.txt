loop M over {1..10}{output Md(d12<=M)+1d(d10<=M) named "Skill [M]" }

loop N over {1..10} {
 loop SN over {2,1} {
  output Nd(d12<=N)+1d(d10<=N)-SN named "Skill [N] -[SN]S" }
 loop HN over {2,1} {
  output (N-HN)d(d12<=N)+1d(d10<=N) named "Skill [N] -[HN]d" }
 output Nd(d12<=N)+1d(d10<=N) named "Skill [N]"
 loop HP over {1,2} {
  output (N+HP)d(d12<=N)+1d(d10<=N) named "Skill [N] +[HP]d" }
 loop SP over {1,2} {
  output Nd(d12<=N)+1d(d10<=N)+SP named "Skill [N] +[SP]S" }
}

\**** Damage calculator duelling Vs archery ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{20..20}

ATTACKMOD: 19
DMGBONUS: 11
PROFICIENCY: 0
ACCBONUS: 0
CRITROLL: 15
ATTACKTIMES: 4
DAMAGE: 1d6 + DMGBONUS
CRITDAMAGE: 2d6 + 2*DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Ataque mano habil"

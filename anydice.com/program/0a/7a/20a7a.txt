function : dice map PLUS:n vs AC:n
{
 CUTOFF : AC - PLUS - 1
 MISS : [lowest of [highest of 1 and CUTOFF] and 19]
 result : 1d{0:MISS,1:(19-MISS),2}
}

function : save map PLUS:n vs DC:n
{
 CUTOFF : DC - PLUS - 1
 MISS : [lowest of [highest of 1 and CUTOFF] and 19]
 result : 1d{2:MISS,1:(19-MISS),1}
}

function : attack MAP:s with DICE:d DPLUS:n
{
 O : 0
 if MAP > 0
 {
  O : DPLUS + MAPdDICE
 }
 result : O
}

function : save MAP:s with DAMAGE:d
{
 result : (MAP * DAMAGE)/2
}

function : hp HP:s MHP:s vs DAMAGE:s
{
 if HP < 0
 {
  if (DAMAGE > 0)
  {
   result : [highest of -3 and HP - 0]
  }
  else
  {
   result : HP
  }
 }
 else
 {
  if DAMAGE < HP + MHP
  {
   result : [highest of 0 and HP - DAMAGE]
  }
  else
  {
   result : -3
  }
 }
}

function : attack APLUS:n with DAMAGE:d DPLUS:n vs AC:n
{
 result : [attack [dice map APLUS vs AC] with DAMAGE DPLUS]
}

function : paralyse PMAP:s on DAMAGE:s
{
 if DAMAGE > 0
 {
  result : PMAP = 1
 }
 else
 {
  result : 0
 }
}

function : paralyse map MAP:d PMAP:d PAR:s
{
 if PAR > 0
 {
  result : PMAP
 }
 else
 {
  result : MAP
 }
}

function : carrion crawler vs HP:d MHP:n AC:n CON:n PARNOW:d TIMES:n
{
 
 MAP : [dice map 8 vs AC]
 MAPQ : 2 * (1@2dMAP > 0)
 
 BITE : [dice map 4 vs AC]
 BITEQ : 2 * (1@2dBITE > 0) 
 
 CONQ : [save map CON vs 13]/2

 J : MAP
 JB : BITE
 P : PARNOW

 A : HP
 loop N over {1..TIMES}
 {
  D : [attack J with 1d4 2]
  P : P | [paralyse CONQ on D]
  J : [paralyse map MAP MAPQ P]
  JB : [paralyse map BITE BITEQ P]
  D : D  + [attack JB with 2d4 2]
  A : [hp A MHP vs D]
 }

 result : A
 
}

MAP : [dice map 8 vs 16]
MAPQ : 2 * (1@2dMAP > 0)

HP : {27,18,17,25}
AC : {16,10,14,15}
CON : {6,1,2,5}

loop N over {1..4}
{
 loop J over {1..#AC}
 {
  Q : [carrion crawler vs J@HP J@HP J@AC J@CON 0 N]
  output Q named "Player [J] attack [N]"
 }
}

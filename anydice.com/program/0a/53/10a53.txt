D: 4

function: ROLL:s count succes{
 result: ((ROLL >= 4) - (ROLL = 6)) + ((ROLL = 6)*2)
}

function: ROLL:s count exalted {
 result: (ROLL = 6) * 2
}

 output [Nd6 count succes] named "sucess icons for [N]d6"
 output [Nd6 count exalted] named "succes from only exalted icons [N]d6"
}

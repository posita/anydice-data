function: highstat
{
	if STR>=DEX
	{result: STR}
	else {result: DEX}
}

function: ATTEMPTS:n attacks ROLL:n
{ 
 DAMAGE: 0
  loop X over {1..ATTEMPTS}
 {
  if ROLL > 0
    {
     DAMAGE: DAMAGE+[damageroll]
    }
 }
result:DAMAGE
}

function: simulate NUMATTACKS
{
DAMAGE: 0
loop X over {1..NUMATTACKS}
{
DAMAGE:DAMAGE+[damageroll]
}

result: DAMAGE
}

function: damageroll
{
result: [floor ((WEAPON+ATTSTAT)-[floor (DR/DRMOD-(VSDR+DEX))])]
}

function: attackroll
{
result: ((1d20+HITSTAT+ACC+BAB)-(1d20+REFDEFENSE))
}

function: floor DICE:n
{

if DICE<0
{
result:0
}
else
{
result:DICE
}
}

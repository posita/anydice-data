


function: standard roll A:s pick NTH:n highest {

 result: NTH@[sort {A}]
}

function: wild roll A:s pick NTH:n highest {

 SIXES: 0
 SIXES: [count 6 in {A}] + SIXES

 if SIXES > 0 {
   MATCHES: SIXES - 1
 }
 else {
   MATCHES: 0
 }

 result: NTH@[sort {A}] + MATCHES
}

function: unskilled roll A:s pick NTH:n lowest {

 ONES: 0
 ONES: [count 1 in {A}] + ONES

 if ONES >= #{A} {
   \ result: -1 \
   result: 1
 }

 result: NTH@[reverse[sort {A}]]

}

output [unskilled roll 2d6 pick 1 lowest] named " 0|  2d6L"
output [wild roll  1d6 pick 1 highest]    named " 1|  1d6H"
output [wild roll  2d6 pick 1 highest]    named " 2|  2d6H"
output [wild roll  3d6 pick 1 highest]    named " 3|  3d6H"
output [wild roll  4d6 pick 1 highest]    named " 4|  4d6H"
output [wild roll  5d6 pick 1 highest]    named " 5|  5d6H"
output [wild roll  6d6 pick 1 highest]    named " 6|  6d6H"

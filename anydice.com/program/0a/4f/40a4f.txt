DANOTESTECRIT: 5d6+3
function: atk ROLL:n {
 ROLLING: ROLL
 if ROLLING >= CRITMAGI {
  result: DANOTESTECRIT
 }
 if ROLL < CRITMAGI {
  result: DANOTESTE
 }
}

 DANOTOTAL: 0
 if ATK = 1 {
  if ROLL >= CRITMAGI {
   DANOTOTAL: DANOTOTAL + DANOCRIT
  }
  if ROLL < CRITMAGI {
   DANOTOTAL: DANOTOTAL + DANO
  }
 }
 if MANA = 0 {
 
 }
 if MANA = 1 {
  DANOTOTAL: DANOTOTAL + DANOTOQUE
 }
 if MANA > 1 {
  DANOTOTAL: DANOTOTAL + DANOTOQUE + (1d8+1)*(MANA-1)
 }
 result: DANOTOTAL
}

function: blades ROLL:s DC {
 result: (1@ROLL >= 4) + (1@ROLL >= 6)
}
function: knave ROLL:s DC {
 result: 2*(1@ROLL+DC >= 15)
}
function: ogre ROLL:s DC {
 result: (#ROLL@ROLL <= DC) + ((#ROLL-1)@ROLL <= DC)
}
output [blades 2@2d6 3] named "blades 0d6"
loop P over {1..6} {
 output [blades Pd6 3] named "blades [P]d6"
}
loop P over {0..10} {
 output [knave 1d20 P] named "d20+[P]"
}
loop P over {2..9} {
 output [ogre 2d12 P] named "2d12 under [P]"
}
loop P over {2..9} {
 output [ogre 3d12 P] named "2d12 under [P] push"
}

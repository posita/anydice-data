\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 5 + 1
PROFICIENCY: 5
HEX: 1d6
DMGBONUS: ATTACKMOD + PROFICIENCY
ACCBONUS: 5 + 1
CRITROLL: 19
ATTACKTIMES: 2
DAMAGE: 1d8 + DMGBONUS + HEX
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS + HEX + HEX
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Pact Sword"

ROLL: 1d20  \ attack roll: use 1@2d20 for advantage or 2@2d20 for disadvantage \
HIT:    +0  \ bonus to attack roll \
AC:      8  \ target AC: attack hits if ATK + HIT >= AC \
CRIT:   20  \ minimum attack roll to crit \
DMG:   1d4  \ damage rolled on successful hit \
BONUS: 1d6  \ bonus damage on first hit \
MULTI:   2  \ number of attacks to make \

function: ROLL:n plus HIT:n vs AC:n {
    CRITS: {20}
    \ return the number of times damage is applied on an attack roll of ROLL \
    if ROLL = 1 | ROLL + HIT < AC {
        result: 0
    }
    if ROLL = 20 | ROLL + HIT >= AC {
        if [CRITS contains ROLL] {result: 2} else {result: 1}
    }
    \if ROLL + HIT = AC {
        if [CRITS contains ROLL] {result: 1} else {result: 1/2}
    }\
}

function: roll ATTACK:d to multiattack N:n times {
  result: [ ATTACK and then roll ATTACK to multiattack N times]
}
function: FIRSTHIT:n and then roll ATTACK:d to multiattack N:n times {
  if N <= 0 {
    result: 0
  }
  if FIRSTHIT > 0 {
    result: FIRSTHIT d (DMG + BONUS) + (N-1) d (ATTACK d DMG)
  } else {
    result: [roll ATTACK to multiattack N-1 times]
  }
}

function: hit twice with ATTACK:n and ATTACK_TWO:n {
    if ATTACK > 0 { 
        result: ATTACK d (DMG + BONUS) + ATTACK_TWO d (DMG)
    } else { 
        result: ATTACK_TWO d (DMG + BONUS)
    }
}

output [roll [ROLL plus HIT vs AC] to multiattack MULTI times] named "[MULTI]x Multi-hit"
output [hit twice with [ROLL plus HIT vs AC] and [ROLL plus HIT vs AC]] named "2x hit"
output [ROLL plus HIT vs AC] d (DMG + BONUS) named "1x hit"

output [ROLL plus HIT vs AC] d (DMG) named "1x hit (no bonus)"

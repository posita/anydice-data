function: doubleendedexplode ROLLEDVALUE:n {
  if ROLLEDVALUE = 1 {result: [doubleendedexplode d{-1..1}] + 1}
  if ROLLEDVALUE = -1 {result: [doubleendedexplode d{-1..1}] - 1}
  result: ROLLEDVALUE
}
output 4d[doubleendedexplode d{-1..1}] named "Double-ended exploding 4dF"

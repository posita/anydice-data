MISS: 0
HIT: 1
CRIT: 1

function: attack ROLL:n with MOD:n against AC:n {
  if (ROLL = 1) { result: MISS }
  if (ROLL = 20) { result: CRIT } 
  if (ROLL + MOD >= AC) { result: HIT }
  result: MISS
}

function: damage ROLL:n atk HITS:n {
  if (HITS = 0) { result: 0 }
  if (HITS = 1) { result: ROLL }
  if (HITS = 2) { result: ROLL }
  result: 0
}

output [damage 2d8+2 atk [attack 1d20 with 5 against {10..14}]]

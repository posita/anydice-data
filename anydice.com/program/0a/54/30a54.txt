function:  N:n of S:s from P:n {
  R:  {}
  loop I over {P..[lowest of P+N-1 and #S]} {
    R:  {R, I@S}
  }
  result:  R
}

function:  exprep FS:s N:n {
  SORTED:  [sort FS]
  EXPLODING:  {d [N of SORTED from 1] * 1000+1}
  SAFE:  {d [#FS of SORTED from N+1] * 1000}
  result:  {EXPLODING, SAFE}
}

output [exprep {1,2,3,4,5,5} 1]

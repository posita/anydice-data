INFSIDES: 10 \ sides on influence dice \
INFMAX: 1 \ max range for influence \
function: infmean N:n {result: d{1..SIDES/2} + [count {1..INFMAX} in NdINFSIDES]}
function: inf N:n {
  if N < 0 {result: [neginf [count {1..INFMAX} in -NdINFSIDES]]}
  result: d{0:SIDES/2,1..SIDES/2} + [count {1..INFMAX} in NdINFSIDES]
}
function: neginf N:n {
  if N > 0 {result: d{-SIDES/2..-1,0:SIDES/2} - N}
  result: d{-SIDES/2..-1,1..SIDES/2}
}

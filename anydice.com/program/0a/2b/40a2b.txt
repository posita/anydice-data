COEF: 4
BASE: 100-(15*COEF)

ROLL_A: BASE+(3d6-3)*COEF
ROLL_B: BASE+(3d6-3)*COEF

MAX: [highest of ROLL_A and ROLL_B]
MIN: [lowest of ROLL_A and ROLL_B]
AVG: (ROLL_A+ROLL_B)/2
SIMPLE: ROLL_A

TEST: AVG

COUNT: 1
AVG_SUM: 0
loop COUNT over {1..14}{
AVG_SUM: AVG_SUM + TEST
}
AVG_TOTAL: AVG_SUM/14

output AVG_TOTAL

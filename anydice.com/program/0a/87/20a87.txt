function: roll A:n {
     
    if A >= 25 { result: 25 }   
    if A >= 20 { result: 20 }
    if A >= 15 { result: 15 }
    if A >= 10 { result: 10 }
    if A >= 5  { result: 5 }
    result: 0
}

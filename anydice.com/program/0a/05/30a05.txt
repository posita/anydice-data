PB: 3 
MOD: 4
HITMOD: PB + MOD
AC: HITMOD + 8
X:1
Y:10
N:2
B:1
CRITRANGE: 20
CRITRANGE_TWO: 20
CRIT_MULTIPLIER: 1

function: damage ROLL:n CRITRANGE:n CRIT_MULTIPLIER:n{
 DAMAGE: XdY + MOD
 CRITDAMAGE: CRIT_MULTIPLIER*Y + (X*CRIT_MULTIPLIER)dY + MOD
 if ROLL >= CRITRANGE {result: CRITDAMAGE}
 if ROLL + HITMOD >= AC {result: DAMAGE}
 result:0
}
SEQUENCE: {1..N}
ACTIONDAMAGE:0
loop I over SEQUENCE{
 HITROLL:d20
 ACTIONDAMAGE: ACTIONDAMAGE + [damage HITROLL CRITRANGE CRIT_MULTIPLIER]
 CRITRANGE_TWO: CRITRANGE
 CRITRANGE_TWO: CRITRANGE_TWO-PB
 CRIT_MULTIPLIER_TWO: CRIT_MULTIPLIER + 1
}
SEQUENCE: {1..B}
BONUSDAMAGE:0
loop I over SEQUENCE{
 HITROLL:d20
 BONUSDAMAGE: BONUSDAMAGE + [damage HITROLL CRITRANGE CRIT_MULTIPLIER]
 CRITRANGE_TWO: CRITRANGE_TWO - PB
 CRIT_MULTIPLIER_TWO: CRIT_MULTIPLIER + 1
}

TOTALDAMAGE: ACTIONDAMAGE + BONUSDAMAGE


output TOTALDAMAGE named "TURN 1"

SEQUENCE_TURN: {1..1}
ACTIONDAMAGE:0
loop I over SEQUENCE_TURN{
 HITROLL: [highest 1 of 2d20]
 ACTIONDAMAGE: ACTIONDAMAGE + [damage HITROLL CRITRANGE_TWO CRIT_MULTIPLIER_TWO]
}

TOTALDAMAGE: ACTIONDAMAGE
output TOTALDAMAGE named "TURN 2"

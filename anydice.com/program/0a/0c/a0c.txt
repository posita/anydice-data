function: works N of A:s and B:s {
 R: 0
 loop N over {1..12} {
  R: [highest of R and [count N in {A,B}]]
 }
 COUNT: R - 1
 result: COUNT + 2@[sort {A,B}]
}

function: broke N of A:s and B:s {
 R: 0
 loop N over {1..12} {
  R: [highest of R and [count N in {A,B}]]
 }
 COUNT: R - 1
 result: COUNT + N@[sort {A,B}]
}

output [works 2 of 1d8  and 2d10]
output [broke 2 of 1d8 and 2d10]

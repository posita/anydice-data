function: save A:n with dc B:n {
    result: A >= B
}

function: damage with save SAVE:n is FAIL_ROLL:d on fail otherwise SUCCESS_ROLL:d {
    if SAVE {
        result: SUCCESS_ROLL
    }
    else {
        result: FAIL_ROLL
    }
}

DEX_SAVE: [save d20 +2 with dc 16]
output [damage with save DEX_SAVE is 15d4 on fail otherwise 10d4/2] named "Vitriolic Sphere"
output [damage with save DEX_SAVE is 9d6 on fail otherwise 9d6/2] named "Fireball 4th level"

NUMBEROFDICE: 3    
DIE: 1d6           

function: reroll NUMBER:n {
  TOTALSUM: 0
  loop LOOPING over {1..NUMBER} {
    X: DIE
    X: [checkmax X]
    TOTALSUM: TOTALSUM + X
  }
  result: TOTALSUM
}

function: checkmax N:n {
  if N <= 3 {
    result: 1d6
  }
  result: N
}

output [reroll NUMBEROFDICE] named "Reroll [NUMBEROFDICE]T6 "

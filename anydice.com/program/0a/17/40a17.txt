\Change this for the die count\
COUNT: 3
\Change this for the Number of Rolls (Will add on to previous rolls)\
ROLL:10
\Change this for different dice\
DIE: 20
\Change this to 1 to add the totals together, instead of comparing them\
TOTAL:0


FLOAT:1@COUNTdDIE
output FLOAT named "Roll 1"
loop X over {2..ROLL}
{ 
if TOTAL=1{
SINK:FLOAT+1@COUNTdDIE
output SINK named "Roll [X]"
FLOAT:SINK
} else {
SINK:[lowest of FLOAT and 1@COUNTdDIE]
output SINK named "Roll [X]"
FLOAT:SINK
}
}

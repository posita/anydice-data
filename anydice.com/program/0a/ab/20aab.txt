function: mydice {
  result: {0, 1}
}

function: dangerdice {
  result: {0, -1}
}

function: adjustroll R:s {
  if R < -1 {
    result: -1
  }
  if R > 2 {
    result: 2
  }
  result: R
}

function: roll N M {
  result: Nd[mydice] + Md[dangerdice]
}

function: realroll N M {
  result: [adjustroll [roll N M]]
}

function: roll ROLL:s with LIGHTS:n lights {
  LVAL: [lookup LIGHTS helper]
  LCOUNT: [count LVAL in ROLL]
  MROLL: ROLL + LCOUNTd6
 
  result: MROLL
}

function: lookup LIGHTS:n helper {
  if LIGHTS = 1 { result: {1} }
  if LIGHTS = 2 { result: {1, 3} }
  if LIGHTS = 3 { result: {1, 3, 5} }
  else { result: 0 }
}

\ Poziom trudnosci \
DN : 3


function: ROLL:s count succes{
 result: ((ROLL >= 4) - (ROLL = 6)) + ((ROLL = 6)*2)
}

function: ROLL:s count exalted {
 result: (ROLL = 6) * 2
}

function: ROLL:s count succes for DN{
result: ((ROLL >= 4) - (ROLL = 6)) + ((ROLL = 6)*2) >= DN
}

function: ROLL:s count succes with six{
result: (((ROLL >= 4) - (ROLL = 6)) & ROLL = 6) + ((ROLL = 6)*2)
}

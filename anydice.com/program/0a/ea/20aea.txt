
function : dice map PLUS:n vs AC:n
{
 D : 2@[sort {1,AC-PLUS-1,20}]
 result : 1d{0:D,1:19-D,2}
}

function : save map PLUS:n vs DC:n
{
 D : 2@[sort {1,DC-PLUS-1,20}]
 result : 1d{2:D,1:(20-D)}
}

function : attack MAP:n with DAMAGE:d PLUS:n
{
 D : 0
 if MAP > 0
 {
  D : PLUS
  D : D + MAPdDAMAGE
 }
 result : D
}



J : [attack [dice map 6 vs 13] with 2d{1,2,{3..6}:4} 4]

output [save map 0 vs 14]

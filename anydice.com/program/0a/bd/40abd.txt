set "maximum function depth" to 5
function: explodetopool N:n {
 R: 0
 if N >= 4 {result: 1 + [explodetopool d6]}
 result: R
}

function: legend N:n {
 R: 0
 if N >= 4 {result: 1}
 result: R 
}

LEGEND: [legend d6]
EXPLODE: [explodetopool d6]

output 6dLEGEND + 1dEXPLODE

loop N over {1..6}{
  output Nd[highest of dLEGEND and dLEGEND] + (6-N)dLEGEND + 1dEXPLODE
}

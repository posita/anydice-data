function: oracle O:s plus L:n and M:s vs MOM:n {
  R: -1
  Y: O+L

  if M > MOM {
   DIF: M-MOM
   Y: Y+DIF
  }


  if Y > 7 {
    R: 1
  }

  if O = 12 {
    R: 2
  }
  if O = 1 {
    R: -2
  }
  
  result: R
}
loop L over {-3..3} {
 output [oracle 1d12 plus L and 1d6 vs 1] named "[L]"

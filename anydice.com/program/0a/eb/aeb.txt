
function: highest in SEQUENCE:s plus doubles {
  FACE: 0
  VALUE: 0
  HIGH: 0
  loop N over [sort SEQUENCE] {
     if FACE < N {
        FACE: N
        VALUE: N
     }
     else {
        if FACE = N { VALUE: VALUE + 1 }
     }
     if HIGH < VALUE { HIGH: VALUE }
  }
  result: HIGH
}

loop N over {1..5} { output [highest in Nd6 plus doubles] named "[N] dice" }

loop N over {1..4} { output N + 3 + 4d3 - 8 named "[N] + 4dF" }

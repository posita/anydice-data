MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack 1@1d20 vs DEFENSE]]
}

DEFENSE: d{14}

ATTACK: 6
HITDAMAGE: d10 + 1d6
CRITICALDAMAGE: 2d10 + 2d6
MISSDAMAGE: 0
output [attack] named "Eldritch Blast"

ATTACK: 6
HITDAMAGE: 2d6 + 4
CRITICALDAMAGE: 4d6 + 4
output [attack] named "Scimitar1"

ATTACK: 3
HITDAMAGE: 2d6
CRITICALDAMAGE: 4d6
output [attack] named "Scimitar2"

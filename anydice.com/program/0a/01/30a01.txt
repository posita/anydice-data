PB: 3 
MOD: 4
HITMOD: PB + MOD
AC: HITMOD + 8
X:1
Y:10
N:2
B:1

function: damage ROLL:n{
 DAMAGE: XdY + MOD
 CRITDAMAGE: (2*X)dY + MOD
 if ROLL = 20 {result: CRITDAMAGE}
 if ROLL + HITMOD >= AC {result: DAMAGE}
 result:0
}
SEQUENCE: {1..N}
ACTIONDAMAGE:0
loop I over SEQUENCE{
 HITROLL:d20
 ACTIONDAMAGE: ACTIONDAMAGE + [damage HITROLL]
}
SEQUENCE: {1..B}
BONUSDAMAGE:0
loop I over SEQUENCE{
 HITROLL:d20
 BONUSDAMAGE: BONUSDAMAGE + [damage HITROLL]
}

TOTALDAMAGE: ACTIONDAMAGE + BONUSDAMAGE


output TOTALDAMAGE named "seki"

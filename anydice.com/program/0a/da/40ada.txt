function: roll X:n M:n D:n {
  S: 0
  if X+M >= D-10 { S: 1 }
  if X+M >= D { S: 2 }
  if X+M >= D+10 { S: 3 }
  
  if X=1 & S>0 { S: S-1 }
  if X=20 & S<3 { S: S+1 }

  result: S
}

function: heal O:n B:n {
 if O=3 { result: 4d8 + B }
 if O=2 { result: 2d8 + B }
 if O=1 { result: 0 }
 result: - 1d8
}

M: 18

output [heal [roll 10 M-3 15] 0]
output [heal [roll 1d20 M 15] 0]
output [heal [roll 1d20 M 20] 15]
output [heal [roll 1d20 M 30] 40]

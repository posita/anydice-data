function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

output 3d[attack 1d20 plus 10 vs 0 for 1d10 crit 1d10 on 20] named "EB"
output [attack 1d20 plus 10 vs 0 for 3d10 crit 3d10 on 20] named "Fire bolt"
output 3d[attack 1d20 plus 10 vs 0 for 1d10+5 crit 1d10 on 20] named "Agonizing blast EB"
output [attack 1d20 plus 10 vs 0 for 3d10+5 crit 3d10 on 20] named "Evocation wizard EB"

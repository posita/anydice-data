function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 4
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS
output 3d[attackroll d20 vs ENEMYAC] named "AdV M7"
output 4d[attackroll d20 vs ENEMYAC] named "AdV M7 flurry/mastery"
output 5d[attackroll d20 vs ENEMYAC] named "AdV M7 flurry+mastery"

DAMAGE: 1d8 + 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + 2d6 + ATTACKMOD + DMGBONUS
output 3d[attackroll d20 vs ENEMYAC] named "AdV M7 mark"
output 4d[attackroll d20 vs ENEMYAC] named "AdV M7 marked"
output 5d[attackroll d20 vs ENEMYAC] named "AdV M7 marked+flurry"

ATTACKMOD: 5
DMGBONUS: 0
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS
output 3d[attackroll d20 vs ENEMYAC] named "AdV M8"
output 4d[attackroll d20 vs ENEMYAC] named "AdV M8 flurry"

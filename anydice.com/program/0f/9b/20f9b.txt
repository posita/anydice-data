\** GLOBAL VARIABLES **\
CRIT_: 20
AC_: 16

function: attack ROLL:n HIT_MOD:n hit DMG_DICE DMG_MOD vs AC:n {
 if (ROLL = CRIT_){ result: DMG_DICE + DMG_DICE + DMG_MOD}
 else if (ROLL = 1 | (ROLL + HIT_MOD) < AC) { result: 0}
 else { result: DMG_DICE + DMG_MOD }
}

function: akta dagger ROLL:n{
 result: [attack ROLL 6 hit 1d4 3 vs AC_]
}

function: akta shortsword ROLL:n{
 result: [attack 1d20 6 hit 4d6 3 vs AC_]
}

function: akta two weapon{
 result: [akta shortsword 1d20]+[akta dagger 1d20]
}

function: akta sneak attack{
 result: [attack 1@2d20 6 hit 4d6 3 vs AC_]
}

function: akta assassination combo{
 result: 2d6 + 6d6 + 2d4
}

}

 result: [ethiel quarterstaff]+[ethiel quarterstaff]+[ethiel punch]+[ethiel punch]
}

}

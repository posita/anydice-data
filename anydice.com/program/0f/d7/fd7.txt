BATK: 3d6
BEVD: 2d6

function: evade A:n plus ATK with E:n plus EVD {
  if (E = [maximum of BEVD]) {
    result: 0
  } else if (E + EVD >= A + ATK) {
    result: 0
  } else if (2 * (E + EVD) >= A + ATK) {
    result: (A + ATK) / 2
  } else {
    result: A + ATK
  }
}

output [evade BATK plus 0 with BEVD plus 0] named "Atk +0 vs Evd +0"
output [evade BATK plus 0 with BEVD plus 1] named "Atk +0 vs Evd +1"
output [evade BATK plus 1 with BEVD plus 0] named "Atk +1 vs Evd +0"
output [evade BATK plus 1 with BEVD plus 1] named "Atk +1 vs Evd +1"

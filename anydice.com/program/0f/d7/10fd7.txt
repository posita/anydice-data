\**** Damage calculator duelling Vs archery ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 14-20\
ENEMYAC: d{14..20}


ATTACKMOD: 5
DMGBONUS: 1
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 1d6 + 3d6 + ATTACKMOD/2 + DMGBONUS
CRITDAMAGE: 2d6 + 6d6 + ATTACKMOD/2 + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Rauf lv5 Dagg+Sneak"

ASV : 3 \ Armor Save \
INV: 5 \ Invulnerable Save \
W : 11 \ Wounds \

function: wound ST TO {
    if ST   =  TO {result: 4}
    if ST   >= TO*2 {result: 2}
    if ST   >= TO {result: 3}
    if ST*2 <= TO {result: 6}
    if ST   <  TO {result: 5}
}
function: hitrr BS S AP{
   W: [wound S T]
   if AP=100 {
     SV: 100
   } else {
     SV: [lowest of ASV+AP and INV]
   }
   result: d6>=BS & [highest 1 of d6]>=W & d6<SV
}

function: hit BS S AP {
   W: [wound S T]
   if AP=100 {
     SV: 100
   } else {
     SV: [lowest of ASV+AP and INV]
   }
   result: d6>=BS & d6>=W & d6<SV
}

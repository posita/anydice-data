function: fade FADE:s {
  ROLL: 1@FADE
  OPP: 2@FADE
  DIV: 4
  \ handle the special cases first \
  if ROLL = OPP { result: 0 }
  if ROLL = 20 { DIV: 2 }
  if OPP = 1 { DIV: 2 }
  if OPP = 1 & ROLL = 20 { DIV: 1 }
  if OPP = 20 { DIV: 2 }
  if ROLL = 1 { DIV: 2 }
  if OPP = 20 & ROLL = 1 { DIV: 1 }
  result: (ROLL - OPP) / DIV
\  if ROLL < OPP { result: (ROLL - OPP) / DIV }\
\  else { result: ( OPP - ROLL / DIV * (-1)) }\
}

output [fade 2d20] named "Roll"

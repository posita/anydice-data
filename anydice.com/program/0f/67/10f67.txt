function: evaluate ROLL:s {
  \ we assume that ROLL always has two dice \
  LO: 1@ROLL
  HI: 2@ROLL
  \ handle the special cases first \
  if LO = 10 & HI = 10 { result: -1 } \ double 10 = "crit success/fail" \
  if LO = HI           { result:  0 } \ any other double = crit failure \ 
  if LO + HI = 20      { result: 21 } \ symmetric pair = crit success \
  \ at this point, we know there's no double or symmetric pair \
  if [absolute LO - 10] < [absolute HI - 10] { result: LO } else { result: HI }
}

output [evaluate 2d20]

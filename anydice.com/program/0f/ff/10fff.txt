TARGET: 5
MAX: 1

function: evaluate ROLL:s {
 LIMIT: MAX
 SUCCESSES: ROLL >= TARGET
 if SUCCESSES > 0 { result: SUCCESSES}
 if [count 1 in ROLL] & LIMIT = 1{
  LIMIT: LIMIT-1
  SUCCESSES: SUCCESSES + 1
  result: SUCCESSES
 }
 result: 0
}

loop DICE over {1..5} {
 output [evaluate DICE d6] named "[DICE]d6, TN[TARGET], Flip [MAX]"
}

function: explode N:n helper { 
  if L = 0 | N < B { result: N }
  result: N + [explode D ifge B recurse (L-1)]
}

function: explode D:d ifge B:n recurse L:n  {
  if L < 0 { result: 0 }
  result: [explode D helper]
}
function: count explode D:d ifge B:n recurse L:n { result: 1 + [explode d(D >= B) ifge 1 recurse (L-1)] }

function: explode V:n with D:d if C:n isge M:n recurse L:n { 
  if L < 0 { result: 0 }
  if C < M { result: V }
  if V < 0 { result: V - [explode D ifge M recurse L-1] }
  result: V + [explode D ifge M recurse L-1]
}
function: min amplitude of A:n and B:n explode with D:d ifge M:n recurse L:n { 
  if      [absolute A] < [absolute B] { result: [explode A with D if [absolute B] isge M recurse L] } 
  else if [absolute B] < [absolute A] { result: [explode B with D if [absolute A] isge M recurse L] } 
  result: [explode 0 with D if [absolute B] isge M recurse L]-[explode 0 with D if [absolute A] isge M recurse L]
}
function: min amplitude of A:n and B:n { result: [min amplitude of A and B explode with d2 ifge 1 recurse 0] }

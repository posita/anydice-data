function: attackroll ROLL:n vs AC:n {
if ROLL = 20{ result: 1 }
if ROLL = 1 { result: 0 }
if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: 1 }
result: 0
}

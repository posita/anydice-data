\** GLOBAL VARIABLES **\
CRIT: 20

function: attack ROLL:n HIT_MOD:n hit DMG_DICE DMG_MOD vs AC:n {
 if (ROLL = CRIT){ result: DMG_DICE + DMG_DICE + DMG_MOD}
 else if (ROLL = 1 | (ROLL + HIT_MOD) < AC) { result: 0}
 else { result: DMG_DICE + DMG_MOD }
}

function: akta sneak attack{
 result: [attack 1@2d20 6 hit 4d6 3 vs 15]
}
function: eol guiding bolt{
 result: [attack 1d20 6 hit 4d6 0 vs 15]
}

function: ethiel quarterstaff{
 result: [attack 1d20 7 hit 1d8 4 vs 15]
}

function: ethiel punch{
 result: [attack 1d20 7 hit 1d6 4 vs 15]
}

function: ethiel regular combo{
 result: [ethiel quarterstaff]+[ethiel quarterstaff]+[ethiel punch]
}

function: ethiel flurry of blows combo{
 result: [ethiel quarterstaff]+[ethiel quarterstaff]+[ethiel punch]+[ethiel punch]
}
output [ethiel flurry of blows combo]

\**** Damage calculator ****\

function: savingthrow ROLL:n vs DCMINUSPROF:n { 
 if ROLL + PROFICIENCY + ABILITYMOD >= DCMINUSPROF { result: HALFDAMAGE }
 result: DAMAGE
}

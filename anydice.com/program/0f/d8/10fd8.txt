\**** Damage calculator normal Vs advantage ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 14-20\
ENEMYAC: d{15..17}

ATTACKMOD: 8
DMGBONUS: 1
PROFICIENCY: 4
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 5
DAMAGE: 2d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Tanni"


DIESIZE: 10      \ Size of the dice \
POOLSIZE: 7      \ Number of dice in the pool \
TN: 8            \ Normal target number (must meet or exceed it for a success) \
TNBOON: 7        \ Target number with a boon \
TNBANE: 9        \ Target number with a bane \

\ The results show the probability of rolling a certain number of successes \
output POOLSIZE d (d DIESIZE >= TN) named "[POOLSIZE]d[DIESIZE]"
output POOLSIZE d (d DIESIZE >= TNBOON) named "[POOLSIZE]d[DIESIZE] (boon)"
output POOLSIZE d (d DIESIZE >= TNBANE) named "[POOLSIZE]d[DIESIZE] (bane)"

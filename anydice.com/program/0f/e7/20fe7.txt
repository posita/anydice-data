\ Coin Throw \
loop N over {2..5}{
  output Nd{0,1} >= 2 named "throw [N] coins looking for 2 heads"
}

\ Dice Roll Under \
loop N over {2..5}{
  output 1d6 <= N named "roll d6 under or equal to [N]"
}

\ Dice Roll + Mod \
loop N over {0..3} {
  output N + 1d6 >= 5 named "roll d6 + [N] looking for 5+"
}

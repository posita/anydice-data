\ Old World of Darkness die calculation                             \
\ Explanation: NUM is the number of dice in your dice pool.         \
\ DIFF is the difficulty of the task.                               \
\ REROLL is 1 if you get to reroll 10s, and 0 if you don't.         \
\ The warning "The maximum function depth was exceeded, results are \
\ truncated." is normal when REROLL is enabled.                     \
NUM: 5
DIFF: 7
REROLL: 1

\ Do not edit anything below this line. \
DD: 1d{-1, {0}:(DIFF - 2), {1}:(10 - DIFF), 2}

function: wwd R:s {
  S: R = 1
  T: (R = 2) - (R = -1)
  if REROLL & (T > 0) {
    result: S + T + [wwd TdDD]
  } else {
    result: S + T
  }
}

set "maximum function depth" to 7 / NUM + 3
if NUM = 1 {
  if REROLL {
    output [wwd NUMdDD] named "1 die, difficulty [DIFF], with rerolls"
  } else {
    output [wwd NUMdDD] named "1 die, difficulty [DIFF], no rerolls"
  }
} else {
  if REROLL {
    output [wwd NUMdDD] named "[NUM] dice, difficulty [DIFF], with rerolls"
  } else {
    output [wwd NUMdDD] named "[NUM] dice, difficulty [DIFF], no rerolls"
  }
}

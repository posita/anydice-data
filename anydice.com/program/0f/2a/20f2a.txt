BASEDIE: d6

function: target TARGET:n reroll REROLL:n VALUE:n {
  if VALUE = REROLL {
    result: (VALUE > TARGET) + [target TARGET reroll REROLL BASEDIE]
  }
  if VALUE  > TARGET {
    result: 1
  }
  result: 0
}
set "maximum function depth" to 4


loop N over {1..10}{
DIE: [target 3 reroll 6 BASEDIE]
 output (N-1)d(d6>3) + DIE named "[N]"
}

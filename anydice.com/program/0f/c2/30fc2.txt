SUCCESS: 2
MIXED: 1
BAD: 0
ADVANTAGE: [highest 2 of 3d6]
DISADVANTAGE: [lowest 2 of 3d6]
STANDARD: 2d6

function: roll ROLL:n plus MOD:n {
  if (ROLL + MOD) >= 10 { result: SUCCESS }
  if (ROLL + MOD) >= 7 { result: MIXED }
  result: BAD
}

loop B over {-1..2} {
output [roll STANDARD plus B] named "[B]"
}

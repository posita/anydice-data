function: wodnoreroll D:n again N:n {
 if N = 1 {result: -1}
 result: N >= D
}

function: wodreroll D:n again N:n {
 if N >= 10 { result: 1 + [wodrerollimpl D again d10] }
 if N = 1 {result: -1}
 result: N >= D
}

function: wodrerollimpl D:n again N:n {
 if N >= 10 { result: 1 + [wodrerollimpl D again d10] }
 result: N >= D
}

function: save H:n mod M dmg D bonus B dc DC {
   if H+M >= DC {result: 0} 
   else {result: D+B}
}

function: savehalf H:n mod M dmg D bonus B dc DC {
   if H+M >= DC{result: (D+B)/2} 
   else {result: (D+B)}
}

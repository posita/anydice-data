\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{14..18}

ATTACKMOD: 0
DMGBONUS: 0
PROFICIENCY: 0
ACCBONUS: 3
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 3d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 6d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "SENBONZAKURA attack"

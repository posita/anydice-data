function: winner A:n B:n C:n D:n E:n F:n G:n
{
if ( E+G <= A+C )&(E != 6) {result: 1}
if ((E+F < B+D)|(B=6)) {result: 0}
else {result: -1} 
}

function: attack DA:n DD:n with AA:n AD:n
{
  result: [winner 1d6 1d6 AA AD 1d6 DA DD] 
}

\Attacker Total Attack Bonus\  A:5
\Attacker Weapon Die+Strength\ T:1d8+1
\Defender Total Defense\       D:4
\Defender Total Fort v Threat\ F:3

X:1d20+A>=10+D
\Y: X*(T+10)>1d20+F\
Y: ((T+10)-(1d20+F))

function: damage ROLL:n HIT:n{
 if HIT = 0 {result: -2 }
 if ROLL <-4 {result: -1 }
 if ROLL <= 0 { result: 0 }
 if ROLL <= 4 { result: 1 }
 if ROLL <= 9 { result: 2 }
 if ROLL <= 14 { result: 3 }
 if ROLL <= 19 { result: 4 }
 result: -8
}
output [damage Y X]

D: 1d6
N: 3
ND: NdD
MIN: 1@{D}

function: reroll amount R:s {
  if ((R = MIN) > 0) {
    result: D - 1
  } else {
    result: 0
  }
}

function: reroll R:s {
  if ((R = 1) > 0) {
    result: R - 1 + D
  } else {
    result: R + 0
  }
}

output NdD named "Before reroll"
output [reroll ND] named "After reroll"
output [reroll amount ND] named "Amount"

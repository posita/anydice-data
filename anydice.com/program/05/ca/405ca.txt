\Base Roll Func\
function: roll A:n B:n{
 C: 0
 if (A = 6){
  C:1
 } 
 if (B = 6){
  C: C+1
 }
 result: 1000 + (A + B) * 10 + C
}

function: second and third digit N:n {
 result: 2@N * 10 + 3@N
}

function: fourth digit N:n {
 result: 4@N
}

output [roll 1d6 1d6]
output [second and third digit [roll 1d6 1d6]]
output [fourth digit [roll 1d6 1d6]]

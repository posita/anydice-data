\Base Roll Func\
function: roll A:n B:n{
 C: 0
 if (A = 6){
  C:1
 } 
 if (B = 6){
  C: C+1
 }
 result: 1000 + C * 100 + A + B
}

function: second digit N:n {
 result: 2@N
}

function: third and fourth digit N:n {
 result: 3@N * 10 + 4@N
}

output [roll 1d6 1d6]
output [second digit [roll 1d6 1d6]]
output [third and fourth digit [roll 1d6 1d6]]

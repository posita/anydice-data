D:4 \* Normal Difficulty Number *\
W:6 \* Bonus Die Difficulty Number *\
MINNUMDICE:1 \* Minimum number of dice to be rolled *\
MAXNUMDICE:5 \* Maximum number of dice to be rolled *\
WILDDIE:5 \* Enter 1 if using wild die or 0 if not using wild die *\

function: evaluate ROLL:s WILD:s{
 SUCCESSES: ROLL >= D
 WILDRESULT: WILD = W
 TOTALSUCCESSES: SUCCESSES + WILDRESULT
 if TOTALSUCCESSES > 0 { result: TOTALSUCCESSES}
 result: 0
}

loop DICE over {MINNUMDICE..MAXNUMDICE} {
 if WILDDIE > DICE {CURBONUS: DICE} else {CURBONUS: WILDDIE}
 output [evaluate DICE d6 CURBONUS d6] named "[DICE]d"
}

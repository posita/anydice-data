\ taking highest from a pool of 3 different die types \

function: highest N:n of A:s and B:s and C:s and D:s {
 result: {1..N} @ [sort {A, B, C, D}]
}

loop X over {8,10,12}
{
 loop Y over {0,4,6,8,10,12}
 {
  loop Z over {4,6,8,10,12}
  {
   loop K over {4,6,8,10,12}
   {
    if K >= Z & Z >= Y & Y >= X
    {
     output [highest 2 of dX and dY and dZ and dK] > 8 named "[X]-[Y]-[Z]-[K]"
    }
   }
  }
 }
}

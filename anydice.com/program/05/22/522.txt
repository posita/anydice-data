MISS: 0
HIT: 1
CRIT: 2

ATTACK: 10
COMBADV: 2
DEFENSE: {15, 17, 20}
HIT: 1d6 + 5 + 3
CRIT: 6 + 5 + 3 + 1d6

function: roll ROLL:n vs DEF:n {
 if ROLL=1 { result: MISS }
 if ROLL=20 { 
  if ROLL+ATTACK >= DEF { result: CRIT }
  else { result: HIT }
 }
 if ROLL+ATTACK >= DEF { result: HIT }
}

loop X over DEFENSE {
 output [roll 1d20 vs X] named "Attacking defense [X]"
}

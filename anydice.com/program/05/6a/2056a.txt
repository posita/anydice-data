ROLL_A: 1d100
SKILL: 65
DAMAGE: 3d10+6
output DAMAGE named "Average Damage Roll"
AP: 6

ROLL_B: 1d100
ARMOUR: 18
DEFENSE: 25

OP_A: (ROLL_A <= SKILL) * (ROLL_B <= DEFENSE) * (ROLL_A > ROLL_B)
OP_B: (ROLL_A <= SKILL) * (ROLL_B > DEFENSE)
HIT: (OP_A + OP_B) > 0

ADJ_ARMOUR: [highest of (ARMOUR  - AP) and 0]
ADJ_DAMAGE: [highest of (DAMAGE - ADJ_ARMOUR) and 0]
E_DAMAGE: HIT * ADJ_DAMAGE

output ADJ_ARMOUR named "Adjusted Armour"
output ADJ_DAMAGE named "Damage on Hit"
output HIT named "Hit Chance"
output E_DAMAGE named "E(Damage)"

output E_DAMAGE + ADJ_DAMAGE named "Concentrated Fire"

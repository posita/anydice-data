\Opponents Stats\
ECRD: 0

\Min Function :(\
function: min N:n{
 if N < 1{
  result: 0 
 }else{
  result: N
 }
}

\Base Roll Func\
function: roll A:n B:n{
 C: 0
 if (A = 6){
  C:1
 } 
 if (B = 6){
  C: C+1
 }
 S: 10*(A + B)+ C
 result: S
}

function: parser N:n{
 R: N/10
 result: R
}

function: parsec N:n{
 R: N/10
 C: N-R*10
 result: C
}


output [parser [roll 1d6 1d6]]
output [parsec [roll 1d6 1d6]]

AC: 13
ATTACK_MOD: 3+2
ATTACK_BONUS: 4
DAMAGE_MOD: 3
DAMAGE_DIE: 1d6

\NON EDITABLE\
MISS: 0
HIT: 1
CRIT: 2

function: attackroll ROLL:n against AC_IN:n
{
  if ROLL = 1
  {
    result: MISS
  }
  if ROLL = 20
  {
    if ROLL + ATTACK_MOD >= AC_IN
    {
     result: CRIT
    }
  }
  if ROLL + ATTACK_MOD >= AC_IN
  {
    result: HIT
  }
  else
  {
   result: MISS
  }
}

output [attackroll 1d20 against AC] named "Attack Result"
output [attackroll [highest of 1d20 and 1d20] against AC] named "Attack with advantage"
output [attackroll 1d20+ATTACK_BONUS against AC] named "Result"

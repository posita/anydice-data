AC: 13
ABI_MOD: 3
PROF_BONUS: 2
ATTACK_BONUS: 4
DAMAGE_DIE: 1d6
DAMAGE_MOD: 3
DAMAGE_BONUS: 0


\NON EDITABLE\
ATTACK_MOD: ABI_MOD + PROF_BONUS
DMG_BONUS_TOT: DAMAGE_MOD + DAMAGE_BONUS

MISS: 0
HIT: 1
CRIT: 2


function: attackroll ROLL:n against AC_IN:n
{
  if ROLL = 1
  {
    result: MISS
  }
  if ROLL = 20
  {
    if ROLL + ATTACK_MOD >= AC_IN
    {
     result: CRIT
    }
  }
  if ROLL + ATTACK_MOD >= AC_IN
  {
    result: HIT
  }
  else
  {
   result: MISS
  }
}

function: damageroll DAMAGE:n for ATTACKROLLS:n
{
  if ATTACKROLLS = CRIT
  {
    result: (DAMAGE * 2) + DMG_BONUS_TOT
  }
  if ATTACKROLLS = HIT
  {
    result: DAMAGE + DMG_BONUS_TOT
  }
  else
  {
   result:0
  }
}

output [attackroll 1d20 against AC] named "Attack Result"
output [damageroll DAMAGE_DIE for[attackroll 1d20 against AC]] named "Damage Result"
output [damageroll [highest of DAMAGE_DIE and DAMAGE_DIE] for[attackroll 1d20 against AC]] named "Damage Result with Advantage roll on damage"
output [attackroll [highest of 1d20 and 1d20] against AC] named "Attack with advantage"
output [damageroll DAMAGE_DIE for[attackroll [highest of 1d20 and 1d20] against AC]] named "Damage Result for Attack with advantage"
output [attackroll 1d20+ATTACK_BONUS against AC] named "Attack with flat bonus"
output [damageroll DAMAGE_DIE for[attackroll 1d20+ATTACK_BONUS against AC]] named "Damage Result for Attack with Flat Bonus"

function: resolve BASE:n MOD:n N:n {
  BASE_N: BASE + N
  MOD_N: MOD + [absolute N]

  if BASE_N <= 6 {
    if MOD_N <= 6 {
      result: 3
    }
    else {
      if MOD_N <= 10 {
        result: 2
      }
      else {
        result: 1
      }
    }
  } else {
    if MOD_N <= 6 {
      result: 4
    }
    else {
      if MOD_N <= 10 {
        result: 5
      }
      else {
        result: 6
      }
    }
  }
}

loop N over {-4..4} {
  output [resolve 1d12 1d12 N] named "adjusted by [N]"
}

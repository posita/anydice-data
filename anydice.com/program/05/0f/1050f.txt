DICE: 10

F:DICE-1
loop A over {1..F}{
loop S over {0..F}{
output d{1..A,0:(DICE-A)}+d{1..S,0:(DICE-S)} named "D[DICE] (A:[A] & S:[S])"
}
}

loop T over {10..20}{
loop S over {0..F}{
output d{T-8..T,0}+d{1..S,0:(DICE-S)} named "D[DICE] (A:[T] & S:[S])"
}
loop K over {10..20}{
output d{T-8..T,0}+d{K-8..K,0} named "D[DICE] (A:[T] & S:[K])"
}
}

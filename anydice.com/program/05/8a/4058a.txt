THRESHOLD: 99

function: A:d times B:n at disadvantage {
  TOTAL: 0
  CHANCES: {}
  loop N over {1..B} {
    TOTAL: [lowest of (TOTAL + [lowest of A and A]) and THRESHOLD]
    CHANCES: {CHANCES}
  }
  result: (TOTAL>=THRESHOLD)* B
}


output [1d100 times 1d10 at disadvantage]
output [1d100 times 1 at disadvantage]
output [1d100 times 2 at disadvantage]
output [1d100 times 3 at disadvantage]
output [1d100 times 4 at disadvantage]
output [1d100 times 5 at disadvantage]
output [1d100 times 6 at disadvantage]

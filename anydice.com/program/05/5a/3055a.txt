function: biplode DIE:d {
 MAX: [maximum of DIE]
 result: [biplode  DIE helper]
}

function: biplode N:n helper {
 if N = MAX { result: MAX + [biplode  DIE helper] }
 if N = 1 { result: -MAX + [biplode  DIE helper] }
 result: N
}

function: clamp N:n between MIN:n and MAX:n
{
 if N > MAX {result: MAX}
 if N < MIN {result: MIN}
 result: N
}

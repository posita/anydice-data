function: cofdseries N:n times rote ROTE:n AGAIN:n again{
 TEMP: 0
 loop I over {1..N} {
 TEMP: TEMP+[cofd 1d10 ROTE AGAIN 0]
 }
 result: TEMP
}

function: cofd R:n ROTE:n AGAIN:n I:n{
   if I>3 {result: 0}
   if R >= AGAIN { result: [cofd 1d10 ROTE AGAIN I+1]+1 }
   else if R >= 8 {result: 1}
   else {
    if ROTE = 1 {result: [cofd 1d10 0 AGAIN I+1]}
    else { result: 0 }
   }
}

output [cofdseries 12 times rote 1 8 again]

function: combines ONE:s TWO:s {
 COMBINED: {}
 loop O over {1..#ONE} {
  loop T over {1..#TWO} {
   N: O@ONE + T@TWO
   COMBINED: [adds N to COMBINED]
  }
 }
 result: COMBINED
}
function: adds N:n to S:s {
 if #S = 0 {result: {N}}
 if #S >= 1 {result: {S, N}}
}

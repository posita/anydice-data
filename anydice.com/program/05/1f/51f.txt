\half the ones subtract from the result rounded down, more subtacted than successes is a dramatic failure\

function: roll DICE:n with AGAIN:n again {
 TOTAL: 0
 loop N over {1..DICE}{ TOTAL: TOTAL + [explode d10 AGAIN] }
 result: [highest of (TOTAL+1)/2 and -1]
}

function: explode D:n A:n {
 if D >= A { result: 2 + [explode d10 A]}
 if D = 1 { result: -1}
 if D > 7 { result: 2}
 else { result: 0 }
}

set "maximum function depth" to 10
loop DICE over {1..10} {
 output [roll DICE with 10 again] named "[DICE]d"
}

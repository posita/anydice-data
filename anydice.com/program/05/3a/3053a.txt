function: resolve ROLL:n THRESH:n MOD:n {
  if ROLL < THRESH {
    if MOD < 4 {
      result: 3
    }
    else {
      if MOD < 6 {
        result: 2
      }
      else {
        result: 1
      }
    }
  } else {
    if MOD < 4 {
      result: 4
    }
    else {
      if MOD < 6 {
        result: 5
      }
      else {
        result: 6
      }
    }
  }
  result: 0
}

loop N over {2..6} {
  output [resolve 1d6 N 1d6] named "success threshold = [N]"
}

function: calcpartial {
 if PARTIAL = FALSE {
  result: 1
 }
 if FACING = FR {
  result: DICE != 5 & DICE != 9
 }
 if FACING = RL {
  result: DICE != 3 & DICE != 6 & DICE != 11
 }
}

function: N:n dattack TOHIT:n with DAMAGE:n damage {
 H: 0
 loop X over {1..N} {
  H: H + [hitcheck TOHIT]
 }
 result: H * DAMAGE
}

function: N:n mattack TOHIT:n with SIZE:n size and DAMAGE:n damage{
 H: 0
 loop X over {1..N} {
  H: H + [hitcheck TOHIT]
 }
 result: [mnumcheck DICE SIZE] * H * DAMAGE
}

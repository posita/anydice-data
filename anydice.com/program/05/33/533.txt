function: check ROLL:n {
 if ROLL <= 3 {result: 1} \ penalty \
 if ROLL > 5 {result: 100} \ advantage \
 result: 10 \ partial success \
}

function: otherkind A and B and C {
  
  ROLLONE: [check A] 
  ROLLTWO: [check B] 
  ROLLTHREE: [check C] 
  result: ROLLONE + ROLLTWO + ROLLTHREE
}

function: otherkindplusone A and B and C and D {
  
  ROLLONE: [check A] 
  ROLLTWO: [check B] 
  ROLLTHREE: [check C] 
  ROLLFOUR: [check D] 
  result: ROLLONE + ROLLTWO + ROLLTHREE + ROLLFOUR
}

function: otherkindplustwo A and B and C and D and E {
  
  ROLLONE: [check A] 
  ROLLTWO: [check B] 
  ROLLTHREE: [check C] 
  ROLLFOUR: [check D] 
  ROLLFIVE: [check E] 
  result: ROLLONE + ROLLTWO + ROLLTHREE + ROLLFOUR + ROLLFIVE
}

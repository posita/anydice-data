function: REROLLLOWEST-ONE ROLL:s {
  result: [sort ROLL]
  if result@0 <= 3 {       \ lowest die is result@0
    result@0: d6           \ reroll it once
  }
  result: [sort result]
  if result@0 <= 3 {       \ check again for 1, 2, 3
    result@0: d6           \ reroll again (second iteration)
  }
  result: result@0 + result@1 + result@2 + result@3
}

output [REROLL-LOWEST-ONE 4d6] named "final total"
output [count {11..24} in [REROLL-LOWEST-ONE 4d6]] named "total > 10?"

  if ROLL + DODGE < DC - 10 {result: FAIL}
  if ROLL + DODGE < DC {result: SUCCESS}
  result: CRITICALSUCCESS
 }

 if ROLL + DODGE < DC {result: FAIL}
 if ROLL + DODGE < DC + 10 {result: SUCCESS}
 result: CRITICALSUCCESS
}

  result: DAMAGE
 }
 if DODGERESULT = SUCCESS {
  result: DAMAGE/2
 }
 result: 0
}

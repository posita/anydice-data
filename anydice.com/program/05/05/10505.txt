\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
if ROLL >= 20 { result: CRITDAMAGE }
if ROLL = 1 { result: 0 }
if ROLL + ATTACKMOD >= AC { result: DAMAGE }
result: 0
}

ATTACKMOD: 8
ATTACKTIMES: 10
DAMAGE: 1d4 + 4
CRITDAMAGE: 2d4 + 4

loop ENEMYAC over {5,10,15,20} {
  output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "10 tiny animated objects vs. AC [ENEMYAC]"
}

loop ENEMYAC over {5,10,15,20} {
  output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "10 tiny animated objects vs. AC [ENEMYAC] with advantage"
}

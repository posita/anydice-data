function: reroll ROLL:s as REROLL:d if MIN:n or less {
  if ROLL > MIN {
    result: ROLL + 0
  } else {
    result: REROLL
  }
}

output [reroll 1d10 as d10 if 5 or less] named "1d10, reroll 5 or less"
output [reroll 1d10 as d10 if 4 or less] named "1d10, reroll 4 or less"
output [reroll 1d10 as d10 if 3 or less] named "1d10, reroll 3 or less"
output [reroll 1d10 as d10 if 2 or less] named "1d10, reroll 2 or less"

ABILITY_MOD: 3
PROF: 2
DMG_MOD: ABILITY_MOD
CRIT: 20

function: fifth edition attack ROLL:n DMG_DICE:s USE_PROF:n {
  TO_HIT: ABILITY_MOD + USE_PROF * PROF

  if ROLL = 1 { result: 0 }
  if ROLL >= CRIT { result: 2*DMG_DICE + DMG_MOD }
  if ROLL + TO_HIT >= AC { result: DMG_DICE + DMG_MOD }
  result: 0
}

loop AC over {8..15} {
  output [fifth edition attack d20 1d4 1] named "[AC] AC — Dagger (prof)"
  output [fifth edition attack d20 1d4 0] named "[AC] AC — Dagger (no prof)"
}

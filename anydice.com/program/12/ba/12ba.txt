\ count successes of opposed pools, discarding initial ties \

function: A:s vs B:s {
 \ find die position that determines winner \
 loop P over {1..[highest of #A and #B]} {
  if P@A > P@B { result: [A beats B] }
  if P@A < P@B { result: -[B beats A] }
 }
 \ a draw \
 result: 0
}

function: A beats B {
 S: 1
 \ count dice of winner beating loser \
 loop P over {P + 1..#A} {
  if P@A <= P@B { result: S }
  S: S + 1
 }
 result: S
}

output [4d6 vs 3d6]

\Calculate a KGS player's Max Damage\
function: kgs A:d and B:d {
 result: [explode [highest of A and B]]
}

A:d4
B:d8
\With Advantage\
output [highest of [kgs A and B] and [kgs A and B]] named "With Advantage"
\Regular Roll\
output [kgs A and B] named "Regular"
\With Disadvantage\
output [lowest of [kgs A and B] and [kgs A and B]] named "With Disadvantage"

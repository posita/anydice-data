
\- random2 -\
function: rand N:n {
  if N < 1 {
    result: 0
  }
  else {
    result: 1d{1..N} - 1
  }
}


\- random2avg -\
function: rand N:n avg TIMES:n {
  SUM: 0
  loop I over {1..TIMES}{
    SUM: SUM + [rand N] + 1
  }
  result: SUM / TIMES
}

\- div_rand_round -\
function: NUMER:n divided by DENOM:n random round {
  QUOT: NUMER/DENOM
  REMAIN: NUMER - QUOT * DENOM
  ROUNDUP: REMAIN >= 1dDENOM
  result: ROUNDUP + QUOT
}


NATT: {1..4}
NCMP: {0..4}
AVA: {-1..1}
AVC: {-1..1}

loop NA over AVA{
  loop NC over AVC{
    loop TA over NATT{
      loop TC over NCMP{
        if NA = -1 {RA: [highest 1 of 2d6]}
        if NA = 1  {RA: [lowest 1 of 2d6]}
        if NA = 0  {RA: d6}
        if NC = -1 {RC: [highest 1 of 2d6]}
        if NC = 1  {RC: [lowest 1 of 2d6]}
        if NC = 0  {RC: d6}
        RES: [count {1..TA} in RA] + [count {0..TC} in RC]
        
        output RES named "[NA] [NC] [TA] [TC]"
      }
    }
  }
}

set "explode depth" to 4

function: force die A:d {
  X: #(A)
  Y: [maximum of (1@A)]
  result: (X-1)dY+[explode dY]
}

output [force die 1d6]
output [force die 1d6]+1
output [force die 1d6]+2

output [force die 2d6]
output [force die 2d6]+1
output [force die 2d6]+2

output [force die 3d6]
output [force die 3d6]+1
output [force die 3d6]+2

output [force die 4d6]
output [force die 4d6]+1
output [force die 4d6]+2

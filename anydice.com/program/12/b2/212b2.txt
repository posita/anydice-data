function: explode VALUE:n {
  if VALUE = 10 {result: 10 + 1d10}
  if VALUE = 1 {result: 1 - 1d10}
  result: VALUE
}

function: average of DIE:d {
  MIN: 1@{DIE}
  R: {}
  loop P over {1..#{DIE}} {
    R: {P@{DIE}, R}
  }
  MAX: 1@{R}

  result: ((MAX + MIN) / 2)
}

function: IN:n floor VALUE:n {
 if IN < VALUE {result: VALUE}
 result: IN
}

function: IN:n ceiling VALUE:n {
 if IN > VALUE {result: VALUE}
 result: IN
}

function: IN:n lift VALUE:n {
 if IN > 0 {if IN < VALUE {result: VALUE} else {result: IN}}
 result: 0
}

function: IN:n roof VALUE:n to CUT:n {
 if IN <= CUT {result: [IN ceiling VALUE]}
 result: 0
}


function: attack DICE:n die DIE:n at BASE:n against ARMOR:n at RANGE:n auto AUTO:n headshot WEAK:n crit CRIT:n on THRESH:n {
  DAMAGE: DICE d DIE
  if (WEAK = YES) {BONUS: (BASE - 8)} else {BONUS: BASE}
  ATTACK: [explode 1d10] + BONUS - RANGE 
  if (CRIT = NO) {HITMOD: [[ATTACK floor 0] ceiling (AUTO + 1)]} else {HITMOD: [[ATTACK floor 0] roof (AUTO + 1) to THRESH]}
  THROUGH: [((HITMOD * DAMAGE) - ARMOR) floor 0]
  CRITTHROUGH_A: 0
  if (CRIT = YES) {
    CRITMOD_A: [[([ATTACK floor THRESH] - THRESH) lift (AUTO + 1)] roof (AUTO + 1) to THRESH]
    CRITDAMAGE_A: ([highest (DICE - 1) of DAMAGE] + 6)
    CRITTHROUGH_A: [((CRITMOD_A * CRITDAMAGE_A) - ARMOR) floor 0]
  }

  CRITTHROUGH_B: 0
  if (CRIT = YES) {
    CRITMOD_B: [[([ATTACK floor (THRESH * 2)] - (THRESH * 2)) lift (AUTO + 1)] roof (AUTO + 1) to THRESH]
    CRITDAMAGE_B: ([highest (DICE - 1) of DAMAGE] + 6)
    CRITTHROUGH_B: [((CRITMOD_B * CRITDAMAGE_B) - ARMOR) floor 0]
  }

  TOTAL: THROUGH + CRITTHROUGH_A + CRITTHROUGH_B
  if (WEAK = YES) {AMP: (TOTAL * 2)} else {AMP: TOTAL}
  result: AMP
}


NO: 0
YES: 1
AUTOSMG: 2 \these are intentionally 1 less than the modifiers in the book, it works out in the sim\
AUTOAR: 3
AUTOEXOTIC: 4

A: 11 \target armor sp\
R: 13 \range dc 13 is optimal range, 15>17 etc\
AR: 17 \range dc for autofire\
B: 14 \attacker skill base + any bonuses - any penalties\
T: 10 \threshold for getting a crit, what you must beat the dc by, setting this below the autofire rating for simulations of automatic weapons will introduce a bug\

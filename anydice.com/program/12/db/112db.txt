function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: xattack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= CRIT_RANGE {
  if ROLL + ATTACK >= 0 { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

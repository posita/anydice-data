function: attack ROLL:n {
 if ROLL = 20 | 19 {
  result: CRITICALHIT
 }else {
  result: HIT
 }
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
}

HIT: 1
CRITICALHIT: 2

HITDAMAGE: 2d4+4d8+3
CRITICALDAMAGE: 4d4+8d8+3

output [damage for [attack d20]] named "Average Damage / Hit"

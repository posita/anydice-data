ERRO              : 0

function: ataqueum DADO:n vs CA :n {
 if DADO = 1                               { result: ERRO}
 if DADO = 20 | DADO = MARGEMDECRITICO     { result: 2*ELDRITCHBLAST + 2*HEX + HEXBLADECURSE + CARISMA + SHARDNECROTIC + IVORYCLAW}      
 if DADO + BONUSDEATAQUEUM>= CA            { result: ELDRITCHBLAST + HEX + HEXBLADECURSE + CARISMA + SHARDNECROTIC}
 result: ERRO
}

 if DADO + BONUSDEATAQUEDOIS>= CA          { result: ELDRITCHBLAST + HEX + HEXBLADECURSE + CARISMA + SHARDNECROTIC}
 result: ERRO
}

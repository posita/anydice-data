\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE }
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: conattackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CONCRITDAMAGE }
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: CONDAMAGE }
 result: 0
}

\**** Once Attack Roll - Hand of Harming is messy in anydice, since its hard to track something like this****\
function: onceattackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: ONCECRITDAMAGE }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: ONCEDAMAGE }
 result: 0
}

function: reroll D:d under N:n{
   result: [rerollcore D die D under N]
}

function: rerollcore R:n die D:d under N:n{
   if R < N { result: D } else {result: R}
}

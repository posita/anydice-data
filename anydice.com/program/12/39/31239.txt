set "maximum function depth" to 4

CRIT: {6, 8, 10, 12}

function: wrapper A:n { result: [critsum 1dA A] }

function: VALUE:n is SEQUENCE:s {
 loop P over {1..#SEQUENCE} {
  if P@SEQUENCE = VALUE {result: 1}
 }
 result: 0
}

function: critsum DIE:n MAX:n {
  SUM: DIE
  if [DIE is CRIT] {
    SUM: SUM + [wrapper MAX]
  }
  result: SUM
}


loop M over {6, 8, 10, 12}{
loop N over {10}{
 output 2d[critsum 1dM M] >= N named "[M]"
}
}

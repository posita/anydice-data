function: split A:n using B:n {
  if [highest of A-B and B] <= 3 {
    result: A
  } else {
    result: [lowest of A-B and B]
  }
}

function: keymaster DICE:s {
  TOTAL: 0
  HIGHER: 1@DICE
  LOWER: 2@DICE

  if HIGHER = 1 {
    TOTAL: 2
  } else if HIGHER <= 4 {
    TOTAL: HIGHER
  } else {
    TOTAL: [split HIGHER using d(HIGHER-1)]
  }

  if LOWER <= 3 {
    TOTAL: TOTAL + LOWER
  }

  result: TOTAL
}

output [keymaster 2d6]

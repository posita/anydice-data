function: ROLL:s vardmgonthree {
  if ROLL >= TN {
    result: [count {3..6} in (DMG -1) d6] + 1
  } else {
    result: 0
  }
}

function: ROLL:s vardmgontwo {
  if ROLL >= TN {
    result: [count {2..6} in (DMG -1) d6] + 1
  } else {
    result: 0
  }
}


loop DMG over {1..7}{
  loop TN over {2..12}{
    output (1d12 >= TN) * DMG named "Single D12 TN = [TN] DMG = [DMG]"
    output [count {TN..12} in DMG d12] named "Multiple Attack D12 TN = [TN] DMG = [DMG]"
  }
}

function: collapse A:n between B:n and C:n {
    if A < B {result: B}
    if A > C {result: C}
    result: A
}

function: disruption A:n {
    if A <= 2 {result: 0}
    else if A <= 4 {result: 1}
    else if A >= 5 {result: 2}
}

DICE: 1d6-1
output 2d[disruption DICE]

DICE: 1d6+1
output 1d[disruption DICE]

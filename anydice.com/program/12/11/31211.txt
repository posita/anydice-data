function: attack AC:n ac roll ROLL:n plus BONUS:n for DAMAGE:d crit CRIT_DAMAGE:d {
  TO_HIT: ROLL + BONUS

  if (ROLL = 1 | TO_HIT < AC) {
    result: 0
  } else if (ROLL = 20) {
    result: DAMAGE + CRIT_DAMAGE
  } else {
    result: DAMAGE
  }
}

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{8}

ATTACKMOD: 0
DMGBONUS: 8
PROFICIENCY: 0
ACCBONUS: 0
CRITROLL: 19
ATTACKTIMES: 2
DAMAGE: 1d12+ 1d4 + DMGBONUS
CRITDAMAGE: 2d12+ 2d4 + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Shotgun"

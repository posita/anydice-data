MISS: 0
HIT: 1
CRIT: 2

function: hit roll ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: 0
}

function: attack {
 result: [hit roll d20 vs DEFENSE]
}

ATTACK: 5
DEFENSE: 20

SWORD: {{3..6}:4, {1..2}}
AXE: {{3..12}:7, {1..2}}

HIT: 2dSWORD+3
CRIT: 8dSWORD+3
output [attack] named "Greatsword"

HIT: dAXE+3
CRIT: 6dAXE+3
output [attack] named "Greataxe"

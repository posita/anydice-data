\ Noma's Nova \
CHA_MOD:  5
SPELL_CURSE_BONUS: 0
SHARD_CURSE_BONUS: 0
SHARD_PROCS: 6

function: if CONDITION:n then full A {
  if CONDITION { result: A } else { result: A/2 }
}

function: reroll R:n of S:n {
  if R > S/2 { result: R }
  result: 1dS
}

function: empower N:n of S:n {
  DAMAGE: NdS
  TOTAL_DAMAGE: 0
  loop X over {1..N} {
    if X <= N - CHA_MOD {
      TOTAL_DAMAGE: TOTAL_DAMAGE + X @ DAMAGE
    }
    else {
      TOTAL_DAMAGE: TOTAL_DAMAGE + [reroll X @ DAMAGE of S]
    }
  }
  result: TOTAL_DAMAGE
}

BLIGHT_DAMAGE: ([empower 8 of 8] + SPELL_CURSE_BONUS)
CON_SAVE: 1d20 + 8
BLIGHT_DAMAGE: [if CON_SAVE <= 16 then full BLIGHT_DAMAGE]
TOUCH_DAMAGE: [empower 2 of 8] + SPELL_CURSE_BONUS
WHIP_DAMAGE: ([empower 3 of 6] + SPELL_CURSE_BONUS)
INT_SAVE: 1d20 + 5
WHIP_DAMAGE: [if INT_SAVE <= 16 then full WHIP_DAMAGE]
SHARD_DAMAGE: (3d6 + SHARD_CURSE_BONUS) * SHARD_PROCS

output BLIGHT_DAMAGE + TOUCH_DAMAGE + WHIP_DAMAGE + SHARD_DAMAGE named "Total Damage"
output BLIGHT_DAMAGE named "Blight Damage"
output TOUCH_DAMAGE named "Touch Damage"
output WHIP_DAMAGE named "Whip Damage"
output SHARD_DAMAGE named "Shard Damage"

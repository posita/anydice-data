function: bound V:n {

    result: -1

  result: V
}

  result: [comb ADdD def DDdD]

function: comb AT:s def DE:s {
  L: #AT
  CRIT: 1@AT = 2
  SUCC: [success AT defense DE]

  result: [success SUCC crit CRIT]
} 

function: success S:n crit C:n {
  if S = 0 { result: 0 }
  if S > 1 { result: 1 + C }
  if S < -1 { result: -1 }
  result: S
}

function: success A:n defense D:n {
  result: A - D
}

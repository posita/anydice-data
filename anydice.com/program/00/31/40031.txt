function: internalreroll X:n Y:n
{
	if X = 1
	{
		result: Y
	}
	if X = 2
	{
		result: Y
	}
	else
	{
		result: X
	}
}
function: rd D:d
{
	result: [internalreroll D D]
}
function: rolltwo D:d
{
     result: [highest of D and D]
}
output [rolltwo [rd 2d6]+[rd d8]+[rd d4]]
output [rolltwo [rd 2d6]+[rd d8]]
output [rolltwo [rd d12]+[rd d8]]
output [rolltwo [rd 2d6]]
output [rolltwo [rd d12]]

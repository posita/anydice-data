function: internalreroll X:n Y:n
{
	if X = 1
	{
		result: Y
	}
	if X = 2
	{
		result: Y
	}
	else
	{
		result: X
	}
}
function: nolow D:d
{
	result: [internalreroll D D]
}
function: bestoftwo D:d
{
     result: [highest 1 of D and [nolow D]]
}
function: twobestoftwo D:d
{
     result: [highest 2 of 2dD and 2d[nolow D]]
}

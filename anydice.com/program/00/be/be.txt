function: simulate NUMATTACKS
{
DAMAGE: 0
loop X over {1..NUMATTACKS}
{
DAMAGE:DAMAGE+[attackroll]
}

result: DAMAGE
}

function: attackroll
{
result: ((WEAPON+ATTSTAT)-[floor (DR/DRMOD-(VSDR+DEX))])
}

function: floor DICE:n
{

if DICE<0
{
result:0
}
else
{
result:DICE
}
}

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}


GWF_D_SIX: [reroll {1,2} on d6 to d6]
GWF_D_EIGHT: [reroll {1,2} on d8 to d8]
GWF_D_TEN: [reroll {1,2} on d10 to d10]
GWF_D_TWELVE: [reroll {1,2} on d12 to d12]



output 8dGWF_D_SIX + 6dGWF_D_EIGHT + 3 named "generous DM"
output 8dGWF_D_SIX + 6d8 + 3 named "meanie DM"

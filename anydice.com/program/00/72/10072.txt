AC: d{13..20}
DEX: 3
PROF: 2
SA: 2d6
RAPIER: 1d8
SHORTBOW: 1d6
DAGGER: 1d4
ADV: [highest 1 of 2d20]

\looped against enemy AC\
ENEMYAC: d{10..12}

function: first A:n second B:n vs AC:n {
  if A = 20 {result: 2*(RAPIER + SA) + DEX}
  else {
    if A = 1 {result: 0}
    else {
      if A + DEX + PROF >= AC {result: RAPIER + DEX + SA}
      }
    }
  if B = 20 {result: 2*(DAGGER + SA) + DEX}
  else {
    if B = 1 {result: 0}
    else {
      if B + DEX + PROF >= AC {result: DAGGER + SA}
      }
    }
  result: 0
}
function: bow A:n vs AC:n {
  if A + DEX + PROF >= AC {result: SHORTBOW + SA + DEX}
  if A = 20 {result: 2*(SHORTBOW + SA) + DEX}
  if A = 1 {result: 0}
  result: 0
}

SA: 0
output [first d20 second d20 vs ENEMYAC] named "dual"
output [first ADV second ADV vs ENEMYAC] named "dual Adv"
SA: 2d6
output [first d20 second d20 vs ENEMYAC] named "sneak dual" 
output [first ADV second ADV vs ENEMYAC] named "sneak dual Adv"
SA: 0
output [bow d20 vs ENEMYAC] named "bow"
output [bow ADV vs ENEMYAC] named "bow Adv"
SA: 2d6
output [bow d20 vs ENEMYAC] named "bow sneak" 
output [bow ADV vs ENEMYAC] named "bow sneak Adv"

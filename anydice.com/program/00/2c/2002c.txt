function: armor DAMAGE:s less ARMOR:s {
  if (DAMAGE - ARMOR) < 0 {result: 0}
  result: (DAMAGE - ARMOR)
}

function: double DICE:d less ARMOR:s {
  D: [armor DICE less ARMOR]
  P: D = 0
  M: P*ARMOR + (1-P)*(ARMOR-1)
  result: D + [armor DICE less M]
}

function: crit DICE:s {
  result: (([count 6 in DICE] > 1) * 5)
}

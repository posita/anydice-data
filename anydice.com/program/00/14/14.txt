A: 10 B: 0
C: 5

if A < B {
M: A
 }
else {
M: B
 }

if A=0 {
BB: B+B
R: 1d{0..BB}
output {} named "Rolling [B] (0..[BB])"
output R>=C named "Success vs Difficulty [C]"
}
else if B=0{
AA: A+A
R: 1d{0..AA}
output {} named "Rolling [A] (0..[AA])"
output R>=C named "Success vs Difficulty [C]"
}
else {
AA: A+M
BB: B+M

R: 1d{0..AA} - 1d{0..BB}
output {} named "[A] (0..[AA]) versus [B] (0..[BB])"
output R>0 named "A Wins"
output R<0 named "B Wins"
output R=0 named "Tie"
}

output R named "Expected Outcome"

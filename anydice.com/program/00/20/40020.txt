DICE: {d6 d8}

function: is critical success A:n and B:n {
  result: A >= 6 & A = B
}

function: check A:n and B:n with target TN {
  if [is critical success A and B] | (A + B >= TN) {
    result: 1
  } else {
    result: 0
  }
}

loop T over {7, 10, 13, 16} {
  output [check d6 and d6 with target T] named "d6 + d6 vs [T]"
  output [check d6 and d8 with target T] named "d6 + d8 vs [T]"
  output [check d6 and d10 with target T] named "d6 + d10 vs [T]"
  output [check d6 and d12 with target T] named "d6 + d12 vs [T]"
  output [check d8 and d8 with target T] named "d8 + d8 vs [T]"
  output [check d8 and d10 with target T] named "d8 + d10 vs [T]"
  output [check d8 and d12 with target T] named "d8 + d12 vs [T]"
  output [check d10 and d10 with target T] named "d10 + d10 vs [T]"
  output [check d10 and d12 with target T] named "d10 + d12 vs [T]"
  output [check d12 and d12 with target T] named "d12 + d12 vs [T]"
}

output [is critical success d6 and d6] named "d6 + d6 critical success"
output [is critical success d6 and d8] named "d6 + d8 critical success"
output [is critical success d6 and d10] named "d6 + d10 critical success"
output [is critical success d6 and d12] named "d6 + d12 critical success"
output [is critical success d8 and d8] named "d8 + d8 critical success"
output [is critical success d8 and d10] named "d8 + d10 critical success"
output [is critical success d8 and d12] named "d8 + d12 critical success"
output [is critical success d10 and d10] named "d10 + d10 critical success"
output [is critical success d10 and d12] named "d10 + d12 critical success"
output [is critical success d12 and d12] named "d12 + d12 critical success"

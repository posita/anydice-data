function: attack ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + ARCH >= AC { result: DAMAGE }
 result: 0
}

ENEMYAC: 18

ATTACKMOD: 3

PROFICIENCY: 3
CRITROLL: 20
EXTRAATTACK: 2
DMGBONUS: 2
ACCBONUS: 1
EXTRADMGDICE: 0

ARCH: 0

DAMAGE: 2d{{1..2}, {3..6}:4} + ATTACKMOD + DMGBONUS + EXTRADMGDICE
CRITDAMAGE: 3d{{1..2}, {3..6}:4} + ATTACKMOD + DMGBONUS + 2dEXTRADMGDICE

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "GWF Greatsword + ASI"

output EXTRAATTACKd[attack [highest 1 of 2d20] vs ENEMYAC] named "GWF Greatsword, Reckless + ASI"

DAMAGE: 2d{{1..2}, {3..6}:4} + ATTACKMOD + DMGBONUS + EXTRADMGDICE
CRITDAMAGE: 3d{{1..2}, {3..6}:4} + ATTACKMOD + DMGBONUS + 2dEXTRADMGDICE

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "GWF/GWM Greatsword"

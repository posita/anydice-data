\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 else if ROLL = 1 { result: 0 }
 else if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 else {result: 0}
}



function: attackrol ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAG } 
 else if ROLL = 1 { result: 0 }
 else if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAG }
 else {result: 0}
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}
ENEMYBC: d{15..20}
ATTACKMOD: 4
DMGBONUS: 1d6
PROFICIENCY: 4
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 2



DAMAGE: 4d4 + ATTACKMOD + DMGBONUS
CRITDAMAGE: DAMAGE * 2
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Rabi Feather Sword"

DAMAG: 2d6 + ATTACKMOD + DMGBONUS + 1d8
CRITDAMAG: DAMAG * 2
output 1d[attackrol d20 vs ENEMYBC] named "Rabi T Crossbow"

function: complexluck ROLL:n dc DC:n with DICE:d {
  if ROLL < DC {
    result: DICE
  }
  result: ROLL
}

function: luck on DICE:d with dc DC:n {
  result: [complexluck DICE dc DC with DICE]
}

DC:15

output 1d20 named "Flat roll"
output [lowest 1 of 2d20] named "Disadvantage"
output [highest 1 of 2d20] named "Advantage"
output [luck on 1d20 with dc DC] named "Flat roll, using a luck token if failed"
output [luck on [lowest 1 of 2d20] with dc DC] named "Disadvantage, using a luck token if failed"
output [luck on [highest 1 of 2d20] with dc DC] named "Advantage, using a luck token if failed"

function: complexluck ROLL:n dc DC:n with DICE:d {
  if ROLL < DC {
    result: DICE
  }
  result: ROLL
}

function: luck on DICE:d with dc DC:n {
  result: [complexluck DICE dc DC with DICE]
}

\output [lowest 1 of 2d20] >= DC named "Disadvantage at DC [DC]"
output [highest 1 of 2d20] >= DC named "Advantage at DC [DC]"
output [luck on 1d20 with dc DC] >= DC named "Flat roll, using a luck token if failed at DC [DC]"
output [luck on [lowest 1 of 2d20] with dc DC] >= DC named "Disadvantage, using a luck token if failed at DC [DC]"
output [luck on [highest 1 of 2d20] with dc DC] >= DC named "Advantage, using a luck token if failed at DC [DC]"\

YOU: 6
THEM: 4

CRITICAL: 10

\
 0 is a draw
 1 is a win
 2 is a critical win
 -1 is a loss
 -2 is a critical loss
\

function: evaluate ROLL:n {
 if ROLL >= CRITICAL { result: 2 }
 if ROLL > 0 { result: 1 }
 if ROLL <= -CRITICAL { result: -2 }
 if ROLL < 0 { result: -1 }
 result: 0
}

output [evaluate 1@YOUd20 - 1@THEMd20] named "[YOU] vs [THEM]"

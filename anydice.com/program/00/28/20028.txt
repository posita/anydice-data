function: clock ROLL:s {
  if 1@ROLL = 6 {
    if 2@ROLL = 6 { result: 5 }
    else { result: 3 }
  }
  if 1@ROLL >= 4 & 1@ROLL <= 5 { result: 2 }
  if 1@ROLL >= 1 & 1@ROLL <= 3 { result: 1 }
} 

loop J over {3..6} {
  L: 0
  S: 0
  loop I over {1..J} {
    L: L + [clock 5d6]
    S: S + [clock 4d6]
  }
  output S >= 12 & L < 12 named "After [J] sessions"
}

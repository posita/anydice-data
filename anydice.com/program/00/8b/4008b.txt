MISS: 0
HIT: 1
CRIT: 2

function: check ROLL:n vs DC:n {
  if ROLL >= DC+10 {result: CRIT}
  if ROLL >= DC {result: HIT}
  result: MISS}

function: damage for RESULT:n {
  if RESULT = CRIT {result: (2*(2d10+4))+(1d6)}
  if RESULT = HIT {result: (2d10+4)+(1d6)}
  result: 0}

output [damage for [check 1d20+15 vs 25]] named "Total Damage vs DC 25"

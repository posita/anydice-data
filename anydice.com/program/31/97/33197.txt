function: roll ROLL:s {
 if 1@ROLL < 1 & (#ROLL)@ROLL = -1 {
  result: -1
 }
 result: [highest of 0 and ROLL]
}

DIFFICULTY: 6

NORMAL: d{-1:1, 0:(DIFFICULTY - 2), 1:(11 - DIFFICULTY)}
PLUS_ONE_DICE: d{-1:1, 0:(DIFFICULTY - 2), 1:(11 - DIFFICULTY)}
PLUS_TWO_DICE: d{-1:1, 0:(DIFFICULTY - 2), 1:(11 - DIFFICULTY)}
MINUS_ONE_DIFF: d{-1:1, 0:(DIFFICULTY - 2 - 1), 1:(11 - DIFFICULTY - 1)}

X: 4

output [roll X d NORMAL] named "normal"
output [roll (X+1) d PLUS_ONE_DICE] named "plus one dice"
output [roll (X+2) d PLUS_TWO_DICE] named "plus two dice"
output [roll X d MINUS_ONE_DIFF] named "minus one diff"

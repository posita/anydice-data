function: roll A plus B  {
 S: 0
 loop N over {1..A}{
  S: S + [ring d6]
 }  
 loop M over {1..B}{
  S: S + [skill d12]
 }
 result: S
}

function: ring C:n {
 D: 0
 if (C > 3) { D: D + 1 }
 if (C = 6) { D: D + [ring d6]}
 result : D
}

set "maximum function depth" to 5

function: skill Y:n {
 Z: 0
 if (Y > 5) { Z: Z + 1 }
 if (Y > 10) { Z: Z + [skill d12]}
 result : Z
}
  
set "maximum function depth" to 5

output [roll 2 plus 1]

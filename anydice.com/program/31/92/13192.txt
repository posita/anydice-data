function: attack ROLL:n {
if ROLL = 20 {
result: 2
}
if ROLL = 1 | ROLL + AB < AC {
result: 0 
} 
result: 1
}

function: hit HIT:n for DICE:d plus DB:n {
if HIT > 0 {
result: (HIT d DICE) + DB
} 
result: 0
}

function: dual HITSA:n HITSB:n for DICE:d plus SA:d plus DB:n {
if HITSA = 0 {
result: [hit HITSB for DICE+SA plus 0] 
}
result: [hit HITSA for DICE+SA plus DB] + [hit HITSB for DICE plus 0]
}

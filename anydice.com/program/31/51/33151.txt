function: winner A:n vs B:n {
 if A > B { result: 1 }
 if A < B { result: -1 }
 result: 0
}

loop X over {0..10} {
  output [winner 2d6+X vs 2d6] named "2d6+[X] vs 2d6" 
}

function: target T:n on N:n {
  result: [highest 1 of Nd6] >= T
}

loop Y over {2..6}{
  loop X over {1..6}{
    output [target Y on X] named "[X]d6 target [Y]+"
  }
}

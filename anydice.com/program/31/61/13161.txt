function: gwf DIE:d {
  result: [reroll DIE as DIE if under 3]
}
function: reroll ROLL:n as DIE:d if under LIMIT:n {
  if ROLL < LIMIT { result: DIE }
  else { result: ROLL }
}

loop SIDES over {4} {
  GWF: [gwf dSIDES]
  output 10dGWF+25 named "Twinblade with GWF"
}

output 5d8+25 named "Dual Rapiers (With a spare fighting style)"

output 10d4+25 named "Twinblade (With any other fighting style)"

  GWF: [gwf dSIDES]

function: highest of A:n and B:n {
 if A > B { result: 2 }
 if A = B { result: 1 }
  result: 0
}

\output [highest of 5d6 and 5d6] named "Pilha de d6: Soma"\
\output [highest of [count {3..6} in Nd6] and [count {3..6} in 3d6]] named "Pilha de [N]d6: Sucesso 3+ vs 3d6"\

}

MISS: 0
HIT: 1
CRIT: 2
CRIT_ROLL: 20
function: attack ROLL:n vs AC:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= CRIT_ROLL { result: CRIT }
 if ROLL + TO_HIT_BONUS >= AC { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRIT {
  result: CRIT_DMG
 }
 if ATTACKRESULT = HIT {
  result: HIT_DMG
 }
 result: 0
}
ATK_MOD:5
PROF: 6
TO_HIT_BONUS: ATK_MOD + PROF
HIT_DMG: d10 + 4
CRIT_DMG: 14
ENEMY_AC: d{13..20}
ATK_ROLL: 1d20
RECKLESS: [highest 1 of 2d20]
output [attack ATK_ROLL vs ENEMY_AC] named "Regular Attack"
output [attack RECKLESS vs ENEMY_AC] named "Reckless Attack"
CRIT_ROLL: 19
output [attack ATK_ROLL vs ENEMY_AC] named "Regular Attack 19 crits"
output [attack RECKLESS vs ENEMY_AC] named "Reckless Attack 19 crits"
CRIT_ROLL:18
output [attack ATK_ROLL vs ENEMY_AC] named "Regular Attack 20 crits"

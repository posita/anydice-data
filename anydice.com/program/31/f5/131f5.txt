function: A:n mod B:n {
  result: A - (A/B)*B
}
function: ac X:n {
  result: X >= 10 & [X mod 10] >= 1
}

loop N over {1..7}{
output Nd{1,2,3,0,0,0} named "numero spade con [N] dadi"
output Nd{1,0,0,0,0,0} >= 1 named "1 cavallo/arco/testa con [N] dadi"
if (N > 1) {
  output Nd{1,0,0,0,0,0} >= 2 named "2 cavallo/arco/testa con [N] dadi"
  output [ac Nd{1,10,0,0,0,0}] named "cavallo + arco con [N] dadi"
}
}

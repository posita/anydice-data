function: DAMAGE:d damage with BONUS:n bonus damage STATUS:d status damage RECOIL:n recoil REPEAT:n repeat EXTRA:n extra hits and CRIT:d crit CRITDAMAGE:n crit damage bonus {
 TOTAL: [attack DAMAGE damage BONUS bonus STATUS status RECOIL recoil CRIT crit CRITDAMAGE crit damage helper]
 if REPEAT > 0 {
  loop N over {1..REPEAT}{
   TOTAL: TOTAL + [attack DAMAGE damage BONUS bonus STATUS status RECOIL recoil CRIT crit CRITDAMAGE crit damage helper]
  }
 }
 if EXTRA > 0 {
  loop N over {1..EXTRA}{
   TOTAL: TOTAL + [attack DAMAGE damage 0 bonus 0 status 0 recoil 0 crit 0 crit damage helper]
  }
 }
 result: TOTAL
}

function: attack D:d damage B:n bonus S:d status R:n recoil C:n crit CD:n crit damage helper {
 if C >= 10 {
  result: D + D + B + S + CD - R
 }
 result: D + B + S - R
}

function: matchboom N:d plus M:d {
 LUCKDIESIZE: [highest of [maximum of N] and [maximum of M]]
 result: [matchboomhelper N plus M with LUCKDIESIZE luck]
}

function: matchboomhelper N:n plus M:n with DIESIZE:n luck {
 if (N = M) {
  result: N + M + [burst d(DIESIZE) on a N with DIESIZE die size]
 }
 result: N + M
}

function: burst ROLL:n on a TARGET:n with DIESIZE:n die size{
 if ROLL = TARGET { result: TARGET + [burst d(DIESIZE) on a TARGET with DIESIZE die size] }
 result: ROLL
}

function: damage DIE:n {
  if DIE >= 12 { result: 4 }
  if DIE >= 10 { result: 3 }
  if DIE >= 7 { result: 2 }
  if DIE >= 2 { result: 1 }
  result: 0
}

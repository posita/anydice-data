function: remove N:n from SEQ:s {
  NEW_SEQ: {}
  FLAG: 0
  loop I over { 1..#SEQ } {
    if FLAG = 0 {
      if N = I@SEQ { FLAG: 1 }
      else { NEW_SEQ: { NEW_SEQ, I@SEQ }}
    }
    else { NEW_SEQ: { NEW_SEQ, I@SEQ }}
  }
  result: NEW_SEQ
}

function: fupoolrec AD:s DD:s {
  loop X over AD {
    loop Y over DD {
      if X = Y {
        NEW_AD: [ remove X from AD ]
        NEW_DD: [ remove Y from DD ]
        result: [ fupoolrec NEW_AD NEW_DD ]
      }
    }
  }
  
}

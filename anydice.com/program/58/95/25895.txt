function: luckyroll N:d plus M:d {
 LUCKDIESIZE: [highest of [maximum of N] and [maximum of M]]
 result: [luckyrollhelper N plus M with LUCKDIESIZE luck]
}

function: luckyrollhelper N:n plus M:n with DIESIZE:n luck {
 if (N = M) & (N != 1) {
  result: N + M + [explode d(DIESIZE) on a N with DIESIZE die size]
 }
 result: N + M
}

function: explode ROLL:n on a TARGET:n with DIESIZE:n die size{
 if ROLL = TARGET { result: TARGET + [explode d(DIESIZE) on a TARGET with DIESIZE die size] }
 result: ROLL
}

output [luckyroll d4 plus d4] named "2d4"
output [luckyroll d4 plus d6] named "1d4 + 1d6"

OUTPUT_STATS: 0
OUTPUT_AVERAGES: 1

function: cap A:n {
 if A > 6 {
  result: 6
 }
 result: A
}

function: floor A:n {
 if A < 0 {
  result: 0
 }
 result: A
}

N: 2d{0..3}
L: [cap N+d{0..1}]
R: [floor N-1]

if OUTPUT_STATS {
output N named "normal"
output L named "lucky"
output R named "regency"
}

if OUTPUT_AVERAGES {
output 3d(N) named "normal average"
output 3d(L) named "lucky average"
output 3d(R) named "regency average"
}

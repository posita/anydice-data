function: burstonfour ROLL:s {
  EXTRA: 0
  loop DIE over ROLL {
    if DIE = X {
      EXTRA: 1
    }
    if DIE = 1 {
      EXTRA: 0
    }
  }
  result: ROLL + [burstonfour EXTRA dX]
}
X:4
set "maximum function depth" to 3

function: burstonsix ROLL:s {
  EXTRA: 0
  loop DIE over ROLL {
    if DIE = Y {
      EXTRA: 1
    }
    if DIE = 1 {
      EXTRA: 0
    }
  }
  result: ROLL + [burstonsix EXTRA dY]
}
Y:6
set "maximum function depth" to 3

function: damage DIE:n {
  if DIE >= 25 { result: 6 }
  if DIE >= 17 { result: 5 }
  if DIE >= 12 { result: 4 }
  if DIE >= 7 { result: 3 }
  if DIE >= 5 { result: 2 }
  if DIE >= 2 { result: 1 }
  result: 0
}

function: add B:n to any die in POOL:s {
  SUCCESSES: 0
  BOONS: B
  loop X over {1..#POOL} {
    if X@POOL >= 4 {SUCCESSES: SUCCESSES+1}
    else if X@POOL+BOONS=4 {
      BOONS: BOONS-(4-X@POOL)
      SUCCESSES: SUCCESSES+1
    }
  }
  result: SUCCESSES
}

function: count top N:n in POOL:s {
  SUCCESSES: 0
  loop X over {1..N} {
    if X@POOL >= 4 {SUCCESSES: SUCCESSES+1}
  }
  result: SUCCESSES
}

loop X over {2..10} {
  output [add 1 to any die in Xd6] named "add 1 to any die in [X]d6"
  output [count {4..6} in (X+1)d6] named "roll 1 additional die than [X]d6"
  output [count top X in (X+1)d6] named "reroll 1 die in [X]d6"
}

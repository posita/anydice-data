SAVEBONUS: 5
DAMAGE: 
FAIL:1
SAVE:2

function: save ROLL:n vs dc DC:n {
	if ROLL + SAVEBONUS < DC {result: FAIL}
	if ROLL + SAVEBONUS >= DC {result: SAVE}
}

function: damage for SAVERESULT:n {
	if SAVERESULT = FAIL {
		result: DAMAGE
	 }
	if SAVERESULT = SAVE {
		result: DAMAGE/2
	}
}

output [damage for [save [highest 1 of 1d20] vs dc 15]]

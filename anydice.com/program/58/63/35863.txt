function: burstonfour ROLL:s {
  EXTRA: 0
  loop DIE over ROLL {
    if DIE = X {
      EXTRA: 1
    }
    if DIE = 1 {
      EXTRA: 0
    }
  }
  result: ROLL + [burstonfour EXTRA dX]
}

  if DIE >= 10 { result: 3 }
  if DIE >= 7 { result: 2 }
  if DIE >= 2 { result: 1 }
  result: 0
}

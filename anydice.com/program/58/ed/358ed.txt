output 0 named "Functions Reroll once, All or One dice, More or Less than target, or Specific Targets."

\ Choose your Custom Dice \
CD: d6

function: reroll N:n and less as ROLL:n {
 if ROLL <= N { result: dCD }
 result: ROLL
}
function: reroll N:n and more as ROLL:n {
 if ROLL >= N { result: dCD }
 result: ROLL
}
function: reroll TARGETS:s from ROLL:n as another NEW_DICE:d {
    if [TARGETS contains ROLL] { result: NEW_DICE }
    result: ROLL
}

\ Number of Dices rolled \
POOL: 2
output 0 named "Roll [POOL] dices :"
\ Target for Successes \
TS: 6

output POOLd(dCD>=TS) named "no reroll & successes on 6+"
output POOLd([reroll 5 and more as dCD]>=TS) named "reroll 5+ & successes on 6+"
output POOLd([reroll 1 and less as dCD]>=TS) named "reroll 1s & successes on 6+"
output POOLd([reroll {1,2} from dCD as another dCD]>=TS) named "reroll 1&2 (=2-) & successes on 6+"
output 1d([reroll 1 and less as dCD]>=TS) + (POOL-1)d(dCD>=TS) named "reroll ONE 1 & successes on 6+"

\ Equivalent but with reroll INSIDE the dice. \
\Q: {{d6},{2..6}:6}
output 1d(dQ>=6) + (POOL-1)d(dCD>=TS) named "bis: reroll ONE 1 & successes on 6+"\

function: cut R:n at T:n {
  if R <= T { result: R }
  result: 0
}

function: decide R:s {
  if 1@R = 0 { result: 0 }
  result: 1@R + (R > 0) - 1
}

function: success R:s at T:n {
  if R <= T { result: 1 }
  result: 0
}

D: [ cut 1d6 at 2 ]
E: [ cut 1d10 at 3 ]
S: [ success 1d6 at 2 ]
T: [ success 1d10 at 3 ]

loop N over { 1..10 } {
  output [ decide NdD ] named "[N]d6x3"
}

loop N over { 1..10 } {
  output [ decide NdE ] named "[N]d0x4"
}

function: A:n vs B:n {
  if A > B { result: 1 }
  if A < B { result: -1 }
  result: 0
}

output [ decide 5dD ] named "5d6"
output [ decide 5dD ] + [ decide 3dD ] named "5d6+3d6"

set "maximum function depth" to 3

function: repeat ROLL for TIMES:n hits {
TOTAL: 0
loop X over {1..TIMES}{
TOTAL: TOTAL+ROLL
}
result: TOTAL
}

function: dw COUNT:n dice plus MODIFIERS:n explode on any RF:n {
DOESNT: (RF-1)^COUNT
EXPLODES: 10^COUNT - DOESNT
EXPLODECHANCE:1d{0:DOESNT,1:EXPLODES}
result: COUNTd10+MODIFIERS + EXPLODECHANCE*[dw COUNT dice plus MODIFIERS explode on any RF]
}

output [repeat [dw 2 dice plus 11 explode on any 10] for 3 hits] named "Garian Mastercraft Helstorm Bolt Rifle Hellfire FULL AUTO"

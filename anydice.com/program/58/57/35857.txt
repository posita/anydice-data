function: hr ROLL:s {  
    if ROLL = FAIL_ROLL { 
        result: 0 
    } \ miss \  
    if ROLL = CRIT_ROLL { 
        result: 0 
    } \ crit \  
    else { 
        result: HIT_ROLL_STD 
    } \ normal hit \  
}  
  
FAIL_ROLL: 1  
CRIT_ROLL: 6  
HIT_ROLL_STD: 1d{2..5}  
ABILITY_MOD: 5  
PB: 6  
  
output [hr 1d6]+1d6+(2*ABILITY_MOD)+PB named "Heavy"  
output [hr 1d6]+2d6+ABILITY_MOD+PB named "Medium"  
output [hr 1d6] +3d6+PB named "Light"  
  
output 1d20+ABILITY_MOD+PB named "5e"

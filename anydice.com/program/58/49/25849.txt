function: explode R:d at T:n {
  if R = 1 { result: 1 + [ explode 1d10 at T ] }
  if R <= T { result: 1 }
  result: 0
}

function: decide R:d at T:n {
  if R = 1 { result: 1 + [ explode 1d10 at T ] }
  if R <= T { result: R }
  result: 0
}

loop T over { 1..8 } {
  loop N over { 1..8 } {
    output [ decide Nd10 at T ] named "biggest [N]d vs [T]"
  }
}

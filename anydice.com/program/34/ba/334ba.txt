function: newsys FEAT:s PROF:s {


  TFEAT: [count {3, 6, 9} in FEAT]
  TPROF: [highest of ([count {3,6} in PROF] - [count 1 in PROF]) and 0]
  TOTAL: TFEAT + TPROF

  if TOTAL > 2 {LEVEL: 3}
  if TOTAL = 2 {LEVEL: 2}
  if TOTAL = 1 {LEVEL: 1}
  if TOTAL <= 0 {LEVEL: 0}
  if 1@FEAT = 12 {LEVEL: 2 + TPROF }
 \ if 1@PROF = 6 {LEVEL: 2 + (2@PROF = 6)} \

  result: [lowest of LEVEL and 3]

}

loop N over {0..4} {
  output [newsys d12 Nd6] named "d12 and [N]d6"
}

\ 0 failure, 1 partial success, 2 success, 3 critical. \

SIDED: 4
SHADE: 0
MAXDICE: 4
MAXAID: 3
SUCCESS: 6
PARTIALBOTTOM: 4

function: blades ROLL:s {
 if 1@ROLL >= SUCCESS{
  result: 2 + (2@ROLL >= SUCCESS)
 }
 result: 1@ROLL >= PARTIALBOTTOM
}

loop P over {1..MAXDICE+MAXAID} {
 output [blades PdSIDED] named "[P]d"
}

loop P over {1..3+MAXAID} {
 output [blades Pd6] named "[P]d"
}

loop P over {1..2+MAXAID} {
 output [blades Pd8] named "[P]d"
}

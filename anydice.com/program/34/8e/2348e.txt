function: generalcrazydice DICE:s
{
TOTAL: 0
CHECKED: {0}
loop I over {1..#DICE}
{
if([count I@DICE in CHECKED] = 0)
{
POWER: [count I@DICE in DICE]
TOTAL: (TOTAL + ((I@DICE)^POWER))
CHECKED: {CHECKED,I@DICE}
}
}
if(#DICE = 1)
{result: 1@DICE}
result: TOTAL
}
output [generalcrazydice 1d6] named "One"
output [generalcrazydice 2d6] named "Two"
output [generalcrazydice 3d6] named "Three"
output [generalcrazydice 4d6] named "Four"

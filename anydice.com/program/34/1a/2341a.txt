function: damage against ac AC:n rolling ATKROLL:n plus ATKMOD:n with damage DMGDICE:d plus DMGMOD:n {
  if ATKROLL = 1 { result: 0 }
  if ATKROLL = 20 { result: 2dDMGDICE + DMGMOD }
  if ATKROLL - AC + ATKMOD >= 0 { result: 1dDMGDICE + DMGMOD }
  result: 0
}

output [damage against ac 18 rolling 1d20 plus 9 with damage 2d10 plus 0] named "Funderbolt vs. plate"
output [damage against ac 8 rolling 1d20 plus 10 with damage 1d6 plus 5] named "Varrus HCbow vs. goblin"
output [damage against ac 8 rolling 1d20 plus 5 with damage 1d6 plus 15] named "Varrus sharpshooter HCbow vs. goblin"
output [damage against ac 8 rolling [highest 1 of 2d20] plus 5 with damage 1d6 plus 15] named "Varrus sharpshooter HCbow w/adv. vs. goblin"

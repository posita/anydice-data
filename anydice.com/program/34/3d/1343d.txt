function: hit A:n for AD:d with AB:n and DB:n against AC:n{
  if A >= 20 {result: 2dAD + DB}
  if A + AB >= AC & A != 1 {result: AD + DB}
  result: 0
}
function: first ROLLA:n with ATKBONUSA:n for DICEA:d and DMGBONUSA:n second ROLLB:n with ATKBONUSB:n for DICEB:d and DMGBONUSB:n against AC:n with sneak attack SA:d{
  if ROLLA = 20 & ROLLB = 20 {result: 2dDICEA + 2dDICEB + 2dSA + DMGBONUSA + DMGBONUSB}
  if ROLLA = 20 & ROLLB + ATKBONUSB >= AC & ROLLB != 1 {result: 2dDICEA + 2dSA+ 1dDICEB + DMGBONUSA + DMGBONUSB}
  if ROLLA = 20 {result: 2dDICEA + 2dSA + DMGBONUSA}
  if ROLLA + ATKBONUSA >= AC & ROLLA != 1 & ROLLB = 20 {result: 1dDICEA + 1dSA + 2dDICEB + DMGBONUSA + DMGBONUSB}
  if ROLLA + ATKBONUSA >= AC & ROLLA != 1 & ROLLB + ATKBONUSB >= AC & ROLLB != 1 {result: 1dDICEA + 1dSA + 1dDICEB + DMGBONUSA + DMGBONUSB}
  if ROLLA + ATKBONUSA >= AC & ROLLA != 1 {result: 1dDICEA + 1dSA + DMGBONUSA}
  if ROLLB = 20 {result: 2dDICEB + 2dSA + DMGBONUSB}
  if ROLLB + ATKBONUSB >= AC & ROLLB != 1 {result: 1dDICEB + 1dSA + DMGBONUSB}
  result: 0
}

AC: d{14..20}
STRENGTH: 4
PROF: 3
ENH: 2

function: ROLL:n vs AC:n {
if ROLL = 20 {
result: CRITICAL + DAMAGE
}
if ROLL + ATTACK >= AC {
result: DAMAGE
}
result: 0
}

ATTACK: PROF + STRENGTH + ENH +1
DAMAGE: 2d6 + STRENGTH + ENH +1
CRITICAL: 2d6 output [d20 vs AC] named "ABSI"

ATTACK: PROF + STRENGTH + ENH - 5
DAMAGE: 2d6 + STRENGTH + ENH + 10
CRITICAL: 2d6 output [d20 vs AC] named "GWM"

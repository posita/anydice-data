function: newsys FEAT:s PROF:s{

  TOTAL: [count {3,6,9} in FEAT] + [count {3,6} in PROF]
  
 \ if FEAT = 12 {TOTAL: 1000} \
  if TOTAL > 1 {LEVEL: 2}
  if TOTAL = 1 {LEVEL: 1}
  if TOTAL <= 0 {LEVEL: 0}
 \ if [count {12} in FEAT] {LEVEL: LEVEL +1} \
 \ if [count 6 in PROF] {LEVEL: LEVEL +1} \
  if 1@FEAT = 12 {LEVEL: 2 + (1@PROF = 6)}

  result: [lowest of LEVEL and 3]

}

loop N over {0..4} {
  output [newsys d12 Nd6] named "d12 and [N]d6"
}

BIGAUTO: 2d8+2
SMOLAUTO: 1d6

function: DAMAGE:d damage with CONFIRM:n confirms for EXTRA:n extra hits each {
 TOTAL: DAMAGE
 if CONFIRM > 0 {
  loop N over {1..CONFIRM}{
   TOTAL: TOTAL + DAMAGE
   if EXTRA > 1 {
    loop X over {2..EXTRA} {
     TOTAL: TOTAL + DAMAGE
    }
   }
  }
 }
 result: TOTAL
}

output [BIGAUTO damage with 1 confirms for 1 extra hits each] named "Greatbow w/o skills"
output [BIGAUTO damage with 3 confirms for 1 extra hits each] named "Greatbow w/ Barrage + 4 Risk"
output [BIGAUTO damage with 2 confirms for 1 extra hits each] named "Greatbow w/ Full Auto + 4 Risk"
output [BIGAUTO damage with 3 confirms for 1 extra hits each] named "Greatbow w/ Full Auto + 6 Risk"

output [SMOLAUTO damage with 3 confirms for 3 extra hits each] + [SMOLAUTO damage with 1 confirms for 3 extra hits each] named "Crossbow w/o skills"
output [SMOLAUTO damage with 3 confirms for 3 extra hits each] + [SMOLAUTO damage with 3 confirms for 3 extra hits each] named "Crossbow w/ Barrage + 4 Risk"
output [SMOLAUTO damage with 2 confirms for 3 extra hits each] + [SMOLAUTO damage with 4 confirms for 3 extra hits each] named "Crossbow w/ Full Auto + 4 Risk"
output [SMOLAUTO damage with 3 confirms for 3 extra hits each] + [SMOLAUTO damage with 5 confirms for 3 extra hits each] named "Crossbow w/ Full Auto + 6 Risk"

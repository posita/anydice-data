function: TOTAL:n noofraises {
  result: (TOTAL - 4) / 4
}

function: ROLL:n damageroll {
  result: ROLLd6 + 2d6
}

function: TOTAL:n gotraise {
  result: TOTAL >= 8
}

function: GOTRAISE:n damagerollexp {
  if GOTRAISE = 1 {
    result: 3d [explode d6]
  } else {
    result: 2d6
  }
}

X:[[explode d10] noofraises] 
Y:[d10 gotraise]

output [X damageroll] named "Damage without explosion with multiple raises on skill"
output [Y damagerollexp] named "Damage with explosion with one raise on skill"

function: hit on N:n reroll P:n wound on M:n reroll Q:n save on O:n{
TOHIT: N
TOWOUND: M
REROLLWOUND: Q
TOSAVE: O
result: [hit d6 reroll P]
}

function: hit N:n reroll P:n {
 if N >= TOHIT {result: [wound d6 reroll REROLLWOUND] }
 if N <= P {
 result: [hit d6 reroll 0]
}
 result: 0
}
function: wound N:n reroll P:n {
 if N >= TOWOUND {result: [save d6] }
 if N <= P {
 result: [wound d6 reroll 0]
}
 result: 0
}
function: save N:n {
 if N >= TOSAVE {result: 0 }
 result: 1
}

output 20d[hit on 3 reroll 2 wound on 2 reroll 1 save on 7]

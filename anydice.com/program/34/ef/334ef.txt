MISS: 0
HIT: 1
CRITICALHIT: 2

function: attacka ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + FIRSTATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + FIRSTATTACK - 10 >= DEFENSE { result: CRITICALHIT }
 if ROLL + FIRSTATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: attackb ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + SECONDATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + SECONDATTACK - 10 >= DEFENSE { result: CRITICALHIT }
 if ROLL + SECONDATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: attackc ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + THIRDATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + THIRDATTACK - 10 >= DEFENSE { result: CRITICALHIT }
 if ROLL + THIRDATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKARESULT:n {
 if ATTACKARESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKARESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}
 

function: attacka {
 result: [damage for [attacka d20 vs DEFENSE]] \ first attack \
}
function: attackb {
 result: [damage for [attackb d20 vs DEFENSE]] \ second attack \
}
function: attackc {
 result: [damage for [attackc d20 vs DEFENSE]] \ third attack \
}

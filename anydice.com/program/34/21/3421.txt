
function: adapta A:n {
  if A < 4 { result: 0 }
  if A < 6 { result: 1 }
  if A = 6 { result: 2 + [adapta d6] }
}

function: adapts S:n {
  if S < 2 { result: 0 }
  if S < 5 { result: 1 }
  if S < 6 { result: 2 }
  if S = 6 { result: 3 + [adapts d6] }
}

function: roll attribute A:n skill S:n {
  result: Ad[adapta d6] + Sd[adapts d6]
}


set "maximum function depth" to 5
loop S over {0..6}{
  loop A over {1..8}{
    output [roll attribute A skill S] named "Attribute [A] Skill [S]"
  }
}

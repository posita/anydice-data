set "maximum function depth" to 2

function: oppose A:n TA:n vs B:n TB:n {
 if ( B <= TB & B < A ) { result: 0 }
 if ( A <= TA & A < B ) { result: 1 }
 result: [oppose d20 TA vs d20 TB]
}

CON: 10

loop POT over {10,12,16,18} {
 output [oppose d20 CON vs d20 POT] named "Potency [POT]"
}
loop BANE over {0..3}{
 output 1@(1+BANE)d20 <= CON named "[BANE] Banes"
}

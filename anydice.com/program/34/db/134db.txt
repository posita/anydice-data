function: reroll A:n B:n C:n{ 
 if A <= B { 
  A: 1dC 
 }

}

function: attackroll D:n DD:n DMG:n CRITDICE:n {
 
if  DD >= D {
 D:DD
}

if  D = 20 
{
  X:0
  loop N over {1..CRITDICE} 
  {
  X: X + DMG
  }
 result: X
 }
else 
 { 
 if (D + TOHIT) >= TARGETAC 
  {
  result: DMG
  }
 else 
  { 
  result:0 
  } 
 }
}


function: savageattack A:n B:n { 
 if A >= B { 
  result:A 
 }
 else { 
  result:B 
 } 
}

function: greatweapon DTYPE:n DNUM:n {
 X:0
 loop N over {1..DNUM} {
  X: X + [reroll 1dDTYPE 2 DTYPE]
 }
 result: X
}

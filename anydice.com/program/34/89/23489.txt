\
  Test system 3 :
  Roll 3d6, successes for each if dice < Stat Level
  Stats range from 1 (poor) to 5 (grand master) (6 beeing divinity = 100% 3 success)
  
  Easy task require 1 success
  Moderate task is 2 success
  Difficult task is 3 success

\



function: count VALUES:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 result: COUNT
}

loop LEVEL over {1..6} {
  output [count {1..LEVEL } in 3d6 ] named "Stat lvl [LEVEL]"
}

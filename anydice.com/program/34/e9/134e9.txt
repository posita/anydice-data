\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 12-19\
ENEMYAC: d{12..19}

ATTACKMOD: 3
DMGBONUS: 0

ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d10 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d10 + ATTACKMOD + DMGBONUS

PV: {0, 0, 0, 0, 0, 0, 0, 0, 1, 2, 3, 4, 5, 6, 8, 10, 13, 16}

function: hopeless D:s {
  if 1@D >= 14 {
    S: 1@D / 2 + 2@D / 2 + 3@D / 2 + 4@D / 2 + 5@D / 2 + 6@D / 2 - 30
    if S > 0 {
      result: 0
    } else {
      result: 1
    }
  } else {
    result: 1
  }
}
    
function: pick N:n of D:s {
  if 1@D >= 14 {
    S: 1@D / 2 + 2@D / 2 + 3@D / 2 + 4@D / 2 + 5@D / 2 + 6@D / 2 - 30
    if S > 0 {
      result: N@D
    }
  }
}

function: points D:s {
  if 1@D >= 14 {
    S: 1@D / 2 + 2@D / 2 + 3@D / 2 + 4@D / 2 + 5@D / 2 + 6@D / 2 - 30
    if S > 0 {
      result: (1@D)@PV + (2@D)@PV + (3@D)@PV + (4@D)@PV + (5@D)@PV + (6@D)@PV
    }
  }
}

function: fthree N:n of D:s {
  if 1@D >= 14 {
    S: 1@D / 2 + 2@D / 2 + 3@D / 2 + 4@D / 2 + 5@D / 2 + 6@D / 2 - 30
    if S > 0 {
      result: 361 * N@D + 19 * (N+1)@D + (N+2)@D
    }
  }
}

function: gf N:n {
  result: N / 361
}

function: gs N:n {
  result: (N - 361 * (N / 361)) / 19
}

function: gt N:n {
  result: N - 19 * (N / 19)
}

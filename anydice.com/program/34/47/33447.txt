function: interpret ROLL:s {
 if [count 1 in ROLL]&[count 6 in ROLL] { result: 1 } \ success \
 if [count {1,6} in ROLL] { result: 0 } \ success with stake/time raised \
 result: -1 \ failure\
}

function: interpretsuccess ROLL:s {
 if [lowest of [count 1 in ROLL] and [count 6 in ROLL]]>0 {result: [highest of [count 1 in ROLL] and [count 6 in ROLL]]-[lowest of [count 1 in ROLL] and [count 6 in ROLL]]}\ failure\
}

function: interpretsuccesss ROLL:s {
 if [lowest of [count 1 in ROLL] and [count 6 in ROLL]]=1 & [highest of [count 1 in ROLL] and [count 6 in ROLL]]-[lowest of [count 1 in ROLL] and [count 6 in ROLL]]> 1 {result: 2}
if [lowest of [count 1 in ROLL] and [count 6 in ROLL]]=1 {result: [highest of [count 1 in ROLL] and [count 6 in ROLL]]-[lowest of [count 1 in ROLL] and [count 6 in ROLL]]}
}

loop DICE over {1..10} {
 output [interpretsuccesss DICE d6] named "[DICE]d"
}

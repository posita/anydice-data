\Global Variables\
\looped against enemy AC from 12-20\
ENEMYAC: d{12..20}

SNEAKATTACKCOUNT: 1
ATTACKTIMES: 1
ABILITYMOD: 4
PROFICIENCYMOD: 2
CRITROLL: 20

\Boolean Variables\
DUALWIELD: 1
SNEAKATTACK: 1

\Weapon Dice\
LIGHTCROSSBOW: 1d8
SHORTSWORD: 1d6
DAGGER: 1d4
SHORTBOW: 1d6
LONGBOW: 1d8
NOTHING: 0

TOTALDAMAGE: 0

function: ishit ROLL:n vs AC:n {
	if ROLL = 20 { result: 3 } 
	if ROLL = 1 { result: 0 }
	if ROLL + PROFICIENCYMOD + ABILITYMOD < AC { result: 0 }
	result: 1
}

function: damagetotal A:n and B:n {
	TOTALDAMAGE: 0
	if A + B != 0 {
		if A = 1 {
			TOTALDAMAGE: TOTALDAMAGE + WEAPONA + ABILITYMOD
		}
		else if A = 3{
			TOTALDAMAGE: TOTALDAMAGE + WEAPONA + WEAPONA + ABILITYMOD
		}
		if DUALWIELD = 1{
			if B = 1 {
				TOTALDAMAGE: TOTALDAMAGE + WEAPONB
			}
			else if B = 3{
				TOTALDAMAGE: TOTALDAMAGE + WEAPONB + WEAPONB
			}
		}
                else{
                    B: 0
                }
		if SNEAKATTACK = 1{
			if A + B > 2 {
				TOTALDAMAGE: TOTALDAMAGE + SNEAKATTACKCOUNTd6
			}
			else if A + B > 0{
				TOTALDAMAGE: TOTALDAMAGE + (2 * SNEAKATTACKCOUNT)d6
			}
		}
		result: TOTALDAMAGE
	}
	
	result: 0
}

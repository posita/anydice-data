function: score N:n {
    if N >= 8 { result: 2 }
    if N >= 5 { result: 1 }
    if N = 1 { result: -1 }
    result: 0
}

function: failure N:n {
    if N = 1 { result: 1 }
    result: 0
}

function: partfailure N:n {
    if N >= 2 & N <= 4 { result: 1 }
    result: 0
}

function: success N:n {
    if N >= 5 & N <= 7 { result: 1 }
    result: 0
}

function: crit N:n {
    if N >= 8 { result: 1 }
    result: 0
}

FOUR: [score [explode d4]]
FOURFAIL: [failure [explode d4]]
FOURPART: [partfailure [explode d4]]
FOURCRIT: [crit [explode d4]]
FOURSUCC: [success [explode d4]]

SIX: [score [explode d6]]

loop N over {1..10} {
    output NdFOURFAIL named "[N]d4 failures"
    output NdFOURPART named "[N]d4 partial failures"
    output NdFOURSUCC named "[N]d4 successes"
    output NdFOURCRIT named "[N]d4 crits"
    output NdFOUR named "[N]d4 total"

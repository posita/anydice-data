DISADVANTAGE: -1
NOADVANTAGE: 0
ADVANTAGE: 1
SUPERADVANTAGE: 2
function: attack roll with NUM:n {
  if NUM = -1 {result: [lowest 1 of 2d20]}
  if NUM = 0 {result: d20}
  if NUM = 1 {result: [highest 1 of 2d20]}
  if NUM = 2 {result: [highest 1 of 3d20]}
  result: {}
}
function: damage with ab AB:n and ac AC:n using ROLL:n {
  if ROLL >= 20 {result: 2d8+2d6+2+2}
  if ROLL + AB >= AC {result: 1d8+1d6+2+2}
  result: 0
}

function: damage from hellish rebuke with dex save DEX:n and spell dc DC:n using roll ROLL:n {
  if ROLL >= 20 {result: 2d10/2}
  if ROLL + DEX >= DC {result: 2d10/2}
  result: 2d10
}
function: damage from hellish rebuke with ab AB:n and ac AC:n and dex save DEX:n and spell dc DC:n using ROLL:n {
  if ROLL >= 20 {result: [damage from hellish rebuke with dex save DEX and spell dc DC using roll d20]}
  if ROLL + AB >= AC {result: [damage from hellish rebuke with dex save DEX and spell dc DC using roll d20]}
  result:0
}

function: hr ROLL: s {
  if ROLL = FAIL_ROLL {
    result: 0 \miss\
  }
  if ROLL = CRIT_ROLL {
    result: 0 \crit\
  } else {
    result: HIT_ROLL_STD \normal hit\
  }
}

  FAIL_ROLL: 1
  CRIT_ROLL: 6
  HIT_ROLL_STD: 1 d { 2. .5 }
  WPN_ENCHANT: 0
  ABILITY_MOD: 5
  PB: 6
  HEAVY: 1d6 + (2 * ABILITY_MOD)
  MEDIUM: 2d6 + ABILITY_MOD
  LIGHT: 3d6

output[hr 1 d6] + HEAVY + PB + WPN_ENCHANT named "Heavy"
output[hr 1 d6] + MEDIUM + PB + WPN_ENCHANT named "Medium"
output[hr 1 d6] + LIGHT + PB + WPN_ENCHANT named "Light"

CS:  1d4
PER: 1d6
COR: 1d8

ROLLS: [sort {CS, PER, COR}]
NUMCRIT: 0 + 1@ROLLS<3 + 2@ROLLS<3 + 3@ROLLS<3
TOTAL:0

loop ITR over {1..#ROLLS} {
  ROLL: ITR@ROLLS
  if NUMCRIT> 0 & ITR@ROLLS > 2 {
    CRITPOS: #ROLLS-NUMCRIT+1
    ROLL: CRITPOS@ROLLS}
    TOTAL: TOTAL + ROLL}

CSO: 1d12
PERO:1d6
CORO:1d4

output TOTAL <= 1d10*3

MISS: 0
HIT: 1
CRITICALHIT: 2
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: gwmattack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITICALHIT }
 if ROLL + ATTACK - 5 >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: gwmdamage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE + 10
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE + 10
 }
 result: MISSDAMAGE
}

ATTACK: 2 + 3 + 3
HITDAMAGE: 2d6 + 3 + 1
CRITICALDAMAGE: 4d6 + 3 + 2
MISSDAMAGE: 0
loop AC over {15} {
 output [damage for [attack d20 vs AC]] named "AC:[AC]"
 output [damage for [attack d20+d4 vs AC]] named "AC:[AC] w/ bless"
 output [gwmdamage for [gwmattack d20 vs AC]] named "AC:[AC] w/ GWM"
 output [gwmdamage for [gwmattack d20+d4 vs AC]] named "AC:[AC] w/ GWM w/ bless"
}

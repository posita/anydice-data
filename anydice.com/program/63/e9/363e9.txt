function: nozero A:n and B:n {
        if A - B > 0 & A - B < 5 { result: (A-B)/5+1 }
        if A - B > 0 & A - B >= 5 { result: (A-B)/5 }
        if A - B < 0 & A - B > -5 { result: (A-B)/5-1 }
        if A - B < 0 & A - B <= -5 { result: (A-B)/5 }
        if A = B { result: 0 }
}
output [nozero 1d20 and 1d20]

\ Change these values \
AV: 12    \  This is the AV of the character. You can either directly write down the AV here or fill out the ATT and SKILL field \
DIF: 6    \  This is the DIF of the action, add something in SPEC for an appropriate Specialization that lowers the DIF \
SKILL: 0  \  The Skill of the character.\
ATT: 0    \  The Attribute of the character.\
SPEC: 0   \  The appropriate Specialization that lowers the DIF, can be left at 0 or NO \


\ Function \
if SKILL != 0 & ATT != 0 {AV: SKILL + ATT} 
if SPEC != 0 {DIFF: DIF - SPEC}
else {DIFF: DIF}
DIE: 1d10
VAL: DIE <= AV & DIE > DIFF

output VAL named "Chances"

function: DIE:d mastery RANK:n reroll REROLL:n {
 if RANK < 0 { result: [DIE mastery ([maximum of DIE]+RANK) reroll REROLL reverse] }
 else { result: [DIE mastery RANK reroll REROLL positive] }
}

function: A:n mastery B:n reroll C:n positive {
 if A <= B & C != 0 {
  result: [DIE mastery B reroll (C-1) positive]
 }
 result: A
}

function: A:n mastery B:n reroll C:n reverse {
 if A > B & C != 0 {
  result: [DIE mastery B reroll (C-1) reverse]
 }
 result: A
}

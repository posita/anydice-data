\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 20-25\
ENEMYAC: d{20..25}

ATTACKMOD: 15
DMGBONUS: 10
PROFICIENCY: 0
ACCBONUS: 0
CRITROLL: 19
ATTACKTIMES: 2
DAMAGE: 2d6 + 1d6 + DMGBONUS
CRITDAMAGE: 4d6 + 1d6 + DMGBONUS + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "ExB GREATSWORD Attack modifier"

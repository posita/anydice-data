function: statcheck with MOD {
  ROLL: 2d6 + MOD
  result: [getresult ROLL]
}

function: getresult ROLL:n {
  if ROLL >= 10 { result: 1 }
  else if ROLL >= 7 { result: 0 }
  else { result: -1 }
}

output [statcheck with 0] named "-2"
output [statcheck with 0] named "-1"
output [statcheck with 0] named "+0"
output [statcheck with 0] named "+1"
output [statcheck with 0] named "+2"
output [statcheck with 0] named "+3"

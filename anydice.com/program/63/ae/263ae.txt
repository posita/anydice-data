function: ROLL:n target TARGET:n {
  result: ROLL >= TARGET
}

function: get NUMFOUR:n fourdroplow stats total if at least ATLEASTCOUNT:n dice over TARGET:n and total more than POINTSTHRESHOLD:n {
  TOTAL: 0
  COUNTOVERTARGET: 0
  loop I over {1..NUMFOUR} {
    FOURDROPLOW: [highest 3 of 4d6]
    TOTAL: TOTAL + FOURDROPLOW
    COUNTOVERTARGET: COUNTOVERTARGET + [FOURDROPLOW target TARGET]
  } 
  TOTALPOINTS: TOTAL - 10*NUMFOUR
  MYRESULT: [COUNTOVERTARGET target ATLEASTCOUNT] & [TOTALPOINTS target POINTSTHRESHOLD]
  
  result: MYRESULT
}


function: get chance of set better than pointbuy in NUMSETS:n sets with ATLEASTGOOD:n dice over GOODTHRESHOLD:n {
  BETTERSETCOUNT: 0
  loop I over {1..NUMSETS} {
    BETTERSETCOUNT: BETTERSETCOUNT + [get 6 fourdroplow stats total if at least ATLEASTGOOD dice over GOODTHRESHOLD and total more than 11]
  } 
  result: BETTERSETCOUNT
}

function: get chance of at least ATLEASTCOUNT:n set better than pointbuy with ATLEASTGOOD:n dice over GOODTHRESHOLD:n in NUMSETS:n sets {
  result: [[get chance of set better than pointbuy in NUMSETS sets with ATLEASTGOOD dice over GOODTHRESHOLD] target ATLEASTCOUNT]
}

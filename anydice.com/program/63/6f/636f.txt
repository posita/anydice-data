function: DICE:n dsix ignore S:n negatives{
   TOTAL:0
   V:0
   F:0
   COUNT:S
   DIE: 1d6

   loop N over {1..DICE} {
      V:[determine DIE value]
      F:1@1d6
      if F < 3 {COUNT:COUNT-1}
      TOTAL:TOTAL+F
      VALUE:0 
   }
   result: TOTAL
}

function: determine A:n value{
   if A<3 {result:-1}
   else {
      if A<5 & A>2 {result:0}
      else{
         if A>4 {result:1}
      }
   }
}


output [1 dsix ignore 4 negatives] named "Skill 4"

Q: d{0..3}      \Custom D4 Dice\



function: size of ROLL:s without threes {          \ Remove 3s from the Hits roll \
 COUNT: 0
 loop N over {1..#ROLL} {
   if(N@ROLL != 3){ result: #ROLL -N+1 }
 }
 result: 0
}

\*****************************************************************************\
function: find saves type SV:n from HITSROLL:s {             \find nbr of dices used to roll the saves\
 if(SV=0){result: #HITSROLL}                                      \ 0: reuse same nbr of dices than Hits roll \
 if(SV=1){result: [size of HITSROLL without threes]}              \ 1: remove 3s from the Hits Roll \
 if(SV=2){result: #HITSROLL-[size of HITSROLL without threes]}    \ 2: keep only 3s from the Hits Roll \
 result: 0
}
function: find saves modifier type SV:n from HITSROLL:s {             \find the save modifier\
 if(SV=0){result: 0}                                      \ 0:  \
 if(SV=1){result: 0}                                      \ 1:  \
 if(SV=2){result: 0}                                      \ 2:  \
 result: 0
}

function: window crop SEQUENCE:n {                  \ limit the display range between [-1;100] \
 MIN: 0
 MAX: 100
 if (SEQUENCE < MIN){result: MIN}
 if (SEQUENCE > MAX){result: MAX}
 result: SEQUENCE
}
function: roll hits HITS:d {       \ roll Hits Roll \
 result: HITS
}
function: roll saves SAVES:d {
 result: SAVESdQ
}
function: get unsaved hits from HITSROLL:s and SAVESROLL:s {                 \ get Unsaved Hits \
 result: HITSROLL-SAVESROLL
}
function: apply modifiers HITSMOD:n and SAVESMOD:n to SEQ:s {          \ apply All modifiers \
 result: SEQ+HITSMOD+SAVESMOD
}

function: roll HITSROLL:n {
 result: 4@HITSROLL
}

function: do combat up to step NBR:n with DATAS:s {
 S_UN: [roll hits (1@DATAS)dQ]
 S_DEUX: [find saves type (4@DATAS) from S_UN]
 S_TROIS: [roll saves S_DEUX]
 S_QUATRE: [get unsaved hits from S_UN and S_TROIS]
 S_CINQ: [apply modifiers (2@DATAS) and (3@DATAS) to S_QUATRE]
 if(NBR=1){result: S_UN}                          \roll hits-roll\
 if(NBR=2){result: S_DEUX}                        \find save\
 if(NBR=3){result: S_TROIS}                       \roll save\
 if(NBR=4){result: [window crop S_QUATRE]}        \get the Unsaved Hits (limited from 0 to 100)\
 if(NBR=5){result: [window crop S_CINQ]}          \apply modifiers\
 result: 0
}



POOL: 2
HITS_MOD: 0
SAVES_MOD: -3
SV_TYPE: 2
DATAS: {POOL, HITS_MOD, SAVES_MOD, SV_TYPE}

output [do combat up to step 1 with DATAS] named "hits roll [POOL]A"
output [do combat up to step 2 with DATAS] named "find save"             \output POOLd(dQ>=3)\
output [do combat up to step 3 with DATAS] named "roll save"
output [do combat up to step 4 with DATAS] named "get unsaved hits"
output [do combat up to step 5 with DATAS] named "get final hits modified"

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
\ENEMYAC: d{10..20}\
ENEMYAC: 15

ATTACKMOD: 4
DMGBONUS: 2
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20
ROLL: d20
DAMAGE: 2d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d6 + ATTACKMOD + DMGBONUS
output [attackroll ROLL vs ENEMYAC] named "Normal"

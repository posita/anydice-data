function: ses TP:n TRESH:n {
  result: (TP/TRESH)d3 + TP
}

function: seschance TP:n TRESH:n CHANCE:n{
  if CHANCE = 1 {
    result: (TP/TRESH)d3 + TP
  } else {
    result: TP
  }
}

loop N over {1..5}{
  output Nd{0:5, 1} + 1 named "Kaskaden-Multiplikator für T[N]"

  output (N+1)d6 * (Nd{0:5, 1} + 1) named "Erster Kettenblitz von T[N]-Caster mit Kaskade"
  output (N+1)d6 named "Erster Kettenblitz von T[N]-Caster ohne Kaskade"
  output [ses (N+1)d6 (10-N)] named "Erster Kettenblitz von T[N]-Caster ohne Kaskade, mit SES"
  output [ses ((N+1)d6 * (Nd{0:5, 1} + 1)) (10-N)] named "Erster Kettenblitz von T[N]-Caster mit Kaskade & SES"
  output [seschance ((N+1)d6 * (Nd{0:5, 1} + 1)) (10-N) 1d6] named "Erster Kettenblitz von T[N]-Caster mit Kaskade & 1/6 Chance auf SES"

}

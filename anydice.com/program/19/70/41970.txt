function: roll ROLL:n diff DIFF:n spec S:n {
 if ROLL = 1 { result: -1 }
 if ROLL < DIFF { result: 0 }
 if ROLL >= DIFF & ROLL != 10 { result: 1 }
 if ROLL = 10 & S = 1 { result: 2 }
 if ROLL = 10 & S != 1 { result: 1 }
 result: 1 + [roll d10 diff DIFF spec S]
}
\IGNORE EVERYTHING ABOVE THIS LINE\

\===HOW TO USE===\
\replace the number after 'diff' with the difficulty\
\if rolling with a specialty, set spec to 1\

TOTAL: [roll d10 diff 6 spec 0]

\replace 6 with the number of dice being rolled (this can be a range EX: 2..5 for 2d10 to 5d10)\
\replace 10 with the highest number of successes checked for\
\replace -2 with the lowest number of successes checked for\
\EXAMPLE: if you only need 1 success to pass a roll, change 10 to 1\

loop N over {2} {
 output [highest of [lowest of NdTOTAL and 10] and -2] named "[N]d"
}


function: biplode DIE:d {
 MAX: [maximum of DIE]
 result: [biplode DIE helper ]
}

function: biplode N:n helper {
 if N = MAX { result: MAX + DIE}
 if N = 1 {result: -MAX + DIE}
 result: N
}

set "maximum function depth" to 3

D:[biplode 1d10]
output 2dD named "Biplode 2d10"
output 2d10 named "2d10"
output 1d20 named "1d20"

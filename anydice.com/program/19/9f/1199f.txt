\--- Variables ---\
DIE: [highest 4 of 6d6]+2+5+1
TN: 21
DOSVALUE: 3

\--- DoS Function ---\
function: dos ROLL:n TN:n {
 if ROLL <= TN-5 {
    result: -1
 }
 if ROLL > TN-5 & ROLL < TN {
    result: 0
 }
 if ROLL >= TN & ROLL < TN+5 {
    result: 1
 }
 if ROLL >= TN+5 & ROLL < TN+10 {
    result: 2
 }
 if ROLL >= TN+10 & ROLL < TN+15 {
    result: 3
 }
 if ROLL >= TN+15 {
    result: 4
 }
}

output [dos DIE TN]*DOSVALUE

KEEP: 3

loop SKILL over {0..8}
{
     if(SKILL<4){
          DICEPOOL: 7-SKILL
          output [lowest 3 of (7-SKILL)d12] named "Rank [SKILL] (Lowest 3 of [DICEPOOL]d12)"
     }else{
          DICEPOOL: SKILL-1
          output [highest 3 of (SKILL-1)d12] named "Rank [SKILL] (Highest 3 of [DICEPOOL]d12)"
     }
}

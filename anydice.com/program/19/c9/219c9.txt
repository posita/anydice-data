\ 0 is a miss, 1 is a weak hit, 2 is a strong hit \

function: action AROLL:n AMOD:n CROLL:s MOM:n {
  if MOM < 0 & AROLL = [absolute MOM] {AROLL: 0}
  result: [highest of (AROLL + AMOD) and MOM] > CROLL
}

output [action 1d6 1 2d10 0] named "Action 1d6+1 vs 2d10"
output [action 1d6 1 2d8 0] named "Action 1d6+1 vs 2d8"
output [action 1d6 3 2d10 0] named "Action 1d6+3 vs 2d10"
output [action 1d6 3 2d8 0] named "Action 1d6+3 vs 2d8"

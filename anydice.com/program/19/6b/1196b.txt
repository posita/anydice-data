function: reroll lowest of ROLL:s as REROLL:d if less than MIN:n {
  LOWEST: (#ROLL)@ROLL       \ the lowest die is sorted last \
  if LOWEST >= MIN {
    result: ROLL + 0
  } else {
    REST: {1..#ROLL-1}@ROLL  \ all but the lowest die \
    result: REST + REROLL
  }
}
output 4*([reroll lowest of 1d6 as d6 if less than 2]+[reroll lowest of 1d6 as d6 if less than 2]+2d6+1d10+14)

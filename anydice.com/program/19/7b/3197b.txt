function: fifth edition attack ROLL:n {
if ROLL = 1  {result: 0}
if ROLL >= CRIT {result: 2d{DMG_DICE} + DMG_MOD}
if ROLL+TO_HIT >= AC {result: DMG_DICE + DMG_MOD}
result: 0}

function: fifth edition sneak attack ROLL:n {
if ROLL = 1  {result: 0}
if ROLL >= CRIT {result: 2d{DMG_DICE} + 2d{SNEAK_DICE} + DMG_MOD}
if ROLL+TO_HIT >= AC {result: DMG_DICE + SNEAK_DICE + DMG_MOD}
result: 0}

DMG_DICE: 1d10
SNEAK_DICE: 3d6
DMG_MOD: 4
TO_HIT: 9
CRIT: 19
AC: 16


\output [fifth edition attack 1d20] named "Normal attack"
output [fifth edition sneak attack 1d20] named "Sneak attack"
output [fifth edition attack [highest 1 of 2d20]] named "Advantage Normal attack"
output  [fifth edition sneak attack [highest 1 of 2d20]] named "Advantage Sneak attack"
output [fifth edition attack [highest 1 of 3d20]] named "Elven Accuracy Normal attack"
output [fifth edition sneak attack [highest 1 of 3d20]] named "Elven Accuracy Sneak attack" \


output [fifth edition sneak attack [highest 1 of 1d20]] named "ROGUE-Priority Damage No Advantage"

output [fifth edition sneak attack [highest 1 of 3d20]] named "ROGUE-Priority Damage"

output [fifth edition sneak attack [highest 1 of 3d20]] + [fifth edition attack [highest 1 of 1d20]] named "Fighter-Priority Damage"

output [fifth edition sneak attack [highest 1 of 3d20]] + [fifth edition attack [highest 1 of 3d20]] named "Fighter-Priority Damage Two Advantage"

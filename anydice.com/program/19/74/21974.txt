MISS: 0
HIT: 1
CRT: 2
MSTR: 3 \Модификатор характеристики\
MBON: 2 \Бонус мастерства\
DBON: 0
AC: d{12..22}

function: atc DICE:n ATK vs DEF:n {
 if DICE = 1 { result: MISS }
 if DICE = 20 {result: CRT }
 if DICE + ATK >= DEF { result: HIT }
 result: MISS
}

function: dmg DICE MODS for ATR:n {
 if ATR = CRT {
  result: 2dDICE + MODS
 }
 if ATR = HIT {
  result: DICE+MODS
 }
 result: MISS
}

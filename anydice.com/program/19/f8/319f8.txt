function: dmg H vs P {
  result: [highest of Hd{0,0,0,1,1,1} - Pd{0,0,0,0,1,1} and 0]
}


\ output [highest of 3d{0,0,0,1,1,1} - 1d{0,0,0,0,1,1} and 0] \
loop N over {1..5} {
  output [dmg 3 vs N] named "3 vs [N]"
}

loop N over {1..7} {
  output [dmg 2 vs N] named "2 vs [N]"
}

loop N over {1..3} {

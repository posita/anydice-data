function: successes X:d {
 result: [count {4..6} in X]
}

function: process X:s difficulty D:n {
 loop N over X {
  if N >= D {result: 1}
 }
 result: 0
}

function: roll X:d difficulty D:n {
 result: [process [successes X] difficulty D]
}

D: 2

loop N over {1..10} {
 output [roll Nd6 difficulty D] named "[N]d6 difficulty [D]"
}

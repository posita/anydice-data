function: is N:n odd {
 result: N - (N / 2) * 2
}

function: cap N:n at MAX:n {
 result: [lowest of N and MAX]
}
function: odd floor N:n {
 \ Check if N is even \
 if (N / 2) * 2 = N {
  result: N - 1
 }
 result: N
}

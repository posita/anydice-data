B: 0
SF: 1

if SF = 1 {
   M: 25
   S: 20
} else {
   M: 0
   S: 0
}

T: ((2d7)*5)+B+M
W: ((2d8)*5)+B+S
A: ((2d8)*5)+B+S

function: normalize W:s and A:s and T:s for equipment {
   if T > W+A {
      result: W+((1d3)*5)
   }
   if W+A > T*4 {
      result: W-((1d3)*5)
   }
   result: W
}

function: normalize W:s and A:s and T:s for training {
   if T > W+A {
      result: T-((1d3)*5)
   }
   if W+A > T*4 {
      result: T+((1d3)*5)
   }
   result: T
}

output T named "Training"
output W named "Equipment"

output T > W+A named "Probability of Normalization Against Training"
output W+A > T*4 named "Probability of Normalization Against Equipment"

output [normalize W and A and T for equipment] named "Normalized Equipment Values"
output [normalize W and A and T for training] named "Normalized Training Values"

output (1d3)*5 named "Training Normalization"
output (1d3)*5 named "Weapons Normalization"
output (1d3)*5 named "Armor Normalization"

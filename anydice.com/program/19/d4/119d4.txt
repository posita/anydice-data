function: match DO:n DT:n {

 if DO=DT {result:1}
 result:0
}

function: cmatch DO:n DT:n DC:n {

 if DO=DT=DC {result:2}
 if DO=DT {result:1}
 if DO=DC {result:1}
 if DT=DC {result:1}
 result:0
}

 result: 0
}

 if DM=DO {result:1}
 if DM=DC {result:1}
 result: 0
}

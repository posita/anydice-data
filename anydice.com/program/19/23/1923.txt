N: 8	\ Number of dice \
D: d6	\ Type of die \
E: 5	\ When to explode \
S: 10	\ What target number the dice are compared to \

function: explode P:n { if S>E { if P > (E-1) { result: P + [explode D] } } result: P }
set "maximum function depth" to 5

output Nd([explode D] < S+1) named "[N][D] where each die explodes at [E]+ and are equal or lower than [S]"

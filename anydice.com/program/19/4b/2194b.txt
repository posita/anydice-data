\
 0: Fail.
 1: Success, but with undesired consequences.
 2: Clean success.
 3: Critical success!
\

function: successes ROLL:s {
 result: 2*[count{10} in ROLL] + [count{7..9} in ROLL]
}

function: complication ROLL:s {
 result: 2*[count{7,8} in ROLL] + [count{7..9} in ROLL]
}

loop D over {5..7} {
 output [successes Dd10] named "Successes with [D] in Pool"
 output [complication Dd10] named "Complications with [D] in Pool"
}

function: hit N:n
{
if (N-1)>0
{
result: 1
}
else {result: 0}
}
function: hit N:n and T:n
{
if (N - T-1)>0
{
result: 1
}
else {result: 0}
}
function: hit N:n and T:n and R:n
{
if (N - T- R-1)>0
{
result: 1
}
else {result: 0}
}
function: multi hits H:n and T:n and R:n and D:n
{
COUNT: {0}
TEMP: {0}
loop P over {1..H}
{
COUNT: COUNT+[hit 1d100 and T and R]
}
result: COUNT*D
}

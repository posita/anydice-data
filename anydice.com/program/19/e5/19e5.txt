function: rollstat
{
  S: 1d4
  if 1@S = 4 {
    N: 4
    FLAG: 0
    loop I over {1..6} {
      if FLAG = 0 {
        if 1d2 = 1 {
          N: N+1
        }
        else {
          FLAG: 1
        }
      }
    }
    result: N+6
  }
  S: 1d3
  if 1@S = 1 {
    N: 1
    FLAG: 0
    loop I over {1..6} {
      if FLAG = 0 {
        if 1d2 = 1 {
          N: N-1
        }
        else {
          FLAG: 1
        }
      }
    }
    result: N+6
  }
  S: 1d2
  result: 1@S + 7
}

output [rollstat]

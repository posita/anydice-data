function: rollstat {
  DIE: 1d4
  result: [stathelp DIE 1d2 1d2 1d2 1d2 1d2 1d2]
}

function: stathelp N:n CA:n CB:n CC:n CD:n CE:n CF:n{
  if N = 4 {
    if CA=1 {
      N: 5
      if CB=1 {
        N:6
        if CC=1 {
          N:7
          if CD=1 {
            N:8
            if CE=1 {
              N:9
              if CF=1 {
                N:10
              }
            }
          }
        }
      }
    }
  }
  else if N = 1 {
    if CA=1 {
      N:0
      if CB=1 {
        N:-1
        if CC=1 {
          N:-2
          if CD=1 {
            N:-3
            if CE=1 {
              N:-4
              if CF=1 {
                N:-5
              }
            }
          }
        }
      }
    }
  }
  result: N+6
}

output [rollstat]

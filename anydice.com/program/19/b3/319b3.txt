function: exploding ROLL:n if B:n {
 if ROLL >= B { result: ROLL + 1d6 }
 result: ROLL
}

function: reroll ROLL:n if B:n {
 if ROLL <= B { result: 1d6 }
 result: ROLL
}

function: untrained {
 result: 2d6
}

function: trained {
 result: 2d6 + [exploding 1d6 if 6]
}

function: talented {
 result: [reroll 1d6 if 1] + [reroll 1d6 if 1] + [exploding 1d6 if 6]
}

function: expert {
 result: [reroll 1d6 if 2] + [reroll 1d6 if 2] + [exploding 1d6 if 5]
}

function: master {
 result: [reroll 1d6 if 2] + [reroll 1d6 if 2] + [exploding 1d6 if 4] + 1
}

function: difficulty D:n {
 result: 3d6 + D
}

output [untrained] >= [difficulty 0] named "Untrained"
output [trained] >= [difficulty 0] named "Trained"
output [talented] >= [difficulty 0] named "Talented"
output [expert] >= [difficulty 0] named "Expert"
output [master] >= [difficulty 0] named "Master"

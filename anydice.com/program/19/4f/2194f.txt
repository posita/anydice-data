\
 0: Fail.
 1: Success, but with undesired consequences.
 2: Clean success.
 3: Critical success!
\

function: successes ROLL:s STRESS:s{
 result: 2*[count{10} in ROLL] + [count{7..9} in ROLL] - [count{1} in STRESS] + 2*[count{10} in STRESS] + [count{7..9} in STRESS] 
}

function: complication ROLL:s STRESS:s{
 result: [count{7,8} in ROLL] + [count{7,8} in STRESS]
}

D: 5
S: 1

output [successes Dd10 Sd10] named "Successes with [D] in Pool + [S] Stress"
output [complication Dd10 Sd10] named "Complications with [D] in Pool + [S] Stress"

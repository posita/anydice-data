function: attack D:n plus BONUS:n plus S:n vs DC:n  {
   if D = 1 { result: 0}
   else if D + BONUS + S>= DC { result: 1 } 
   else { result:0}
}

function: legendary attack M:n plus BONUS:n plus S:n vs DC:n  {
   if M = 2 { result: 0}
   else if M + BONUS + S>= DC { result: 1 } 
   else { result:0}
}

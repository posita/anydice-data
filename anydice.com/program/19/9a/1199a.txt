function: evaluate ROLL:s {
 SUCCESSES: ROLL <= X
 if SUCCESSES > 0 { result: SUCCESSES}
 result: 0
}


loop N over {1..7}{
loop X over {1..9}{
output [evaluate Nd10] named "number of passes [N]d10 makes at [X]"
}
}

function: evaluateone ROLL:s {
 SUCCESSES: ROLL <= 9
 if SUCCESSES > 0 { result: SUCCESSES}
 result: 0
}

function: evaluatetwo ROLL:s {
 SUCCESSES: ROLL <= 1
 if SUCCESSES > 0 { result: SUCCESSES}
 result: 0
}

output [evaluateone 2d610] >= [evaluatetwo 5d10] named "Does 2d at 9 beat 5d at 1? O means no. 1 means yes."

   if R >= 20 { result: d8 + 13}

}
function: powershot R:n over N:n {
  if R >= 19 { result: d8 + 23} 
else  if (R + 10) >= (N + 5) { result: d8 + 10} 
else  if (R + 10) >= N {result: d8 + 5} else {result: 0}
}
function: fullpowershot R:n over N:n {
  if R >= 19 { result: d8 + 23} 
else  if (R + 10) >= N { result: d8 + 10} else  {result: 0}
}

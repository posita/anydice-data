WILD: [explode d6]
NORMAL: [explode d6]
PROV: [explode d4-2]

function: max A:n B:n C:n {
    result: 1@[sort {A, B, C}]
}

function: bonus P:n W:n {
    B: [highest of 1@P and 1@W]
    R: [highest of 1@NORMAL and 1@WILD]
    if (B >= 8) { result: R+2 }
    if (B >= 4) { result: R+1 }
    result: R
}

output [highest of NORMAL and WILD] named "Standard"

output [bonus 1@PROV 1@WILD] named "Bonus Modifier"

set "maximum function depth" to 10

function: skill S:n {
  if (S >= 0) {
    result: 3@(S+5)d10
  }
  else {
    result: (-S+3)@(-S+5)d10
  }
}

function: r D:d {
  R: 0
  R: D
  result: R
}

output [middle 1 of 3d6] - [middle 1 of 3d6]
output [middle 1 of 5d20]
output 1d4+1d6+1d8+1d10+1d12+1d20
output [r 1d6]

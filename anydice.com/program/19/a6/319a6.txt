function: exploding ROLL:n if B:n {
 if ROLL >= B { result: ROLL + 1d6 }
 result: ROLL
}

function: reroll ROLL:n if B:n {
 if ROLL <= B { result: 1d6 }
 result: ROLL
}

function: trained {
 result: 2d6 + [exploding 1d6 if 6]
}

function: talented {
 result: [reroll 1d6 if 1] + [reroll 1d6 if 1] + [exploding 1d6 if 6]
}

function: expert {
 result: [reroll 1d6 if 1] + [reroll 1d6 if 1] + [exploding 1d6 if 5]
}

function: master {
 result: [reroll 1d6 if 1] + [reroll 1d6 if 1] + [exploding 1d6 if 5] + 1
}

output [trained] named "Trained"
output [talented] named "Talented"
output [expert] named "Expert"
output [master] named "Master"

function: shoot A:s BS:n {
  OUTPUT : 0
  loop N over {1..#A}{
    OUTPUT : OUTPUT + (N@A >= BS)
  }
  result : OUTPUT
}

function: wound S:n T:n HITS:n {
  if S = T { TARGET : 4 }
  if S > T { TARGET : 3 }
  if S < T { TARGET : 5 }
  if S/T >= 2 { TARGET : 2 }
  if T/S >= 2 { TARGET : 6 }
  result : [shoot HITSd6 TARGET]
}

output [wound 5 5 [shoot 8d6 4]]

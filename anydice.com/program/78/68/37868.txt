set "maximum function depth" to 3

function: bro DICE:d {
  result: [count {6..10} in DICE]
}

function: brah SEQ:s {
  RESULT: 0

  loop RES over SEQ {
    if RES >= 9 {
       RESULT: RESULT + 1 + [brah 1d10]
    }
    if RES >= 6 & RES < 9 {
       RESULT: RESULT + 1
    }
  }

  result: RESULT
}

output [bro 3d10]
output [brah 3d10]

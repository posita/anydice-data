SIZE: 6 \Die size\
ITERATIONS: 15 \Only used if LOOP > 1\
LOOP: 50 \Iterates through 1..LOOP iterations if > 1, max 50\

function: pig {
  ACC: 0
  loop I over {1..ITERATIONS} {
    CURR: 1dSIZE
    ACC: ACC + 1d{-1000, {1..SIZE-1}}
  }
  result: [highest of 0 and ACC]
}

if LOOP > 1 {
  loop I over {1..LOOP} {
    ITERATIONS: I
    output [pig] named "[ITERATIONS] iterations"
  }
} else {
  output [pig] named "Pig"
}

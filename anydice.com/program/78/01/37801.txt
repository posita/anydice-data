function: if CONDITION:n then A else B {
  if CONDITION { result: A } else { result: B }
}

function: if CONDITION:n {
  if CONDITION { result: 1 } else { result: 0 }
}

AC: 20
ATK: -1
DMG: 10 
ROLL: 1d20
ATK_ROLL: ROLL + ATK
NAT: ROLL = 20
HIT: ATK_ROLL >= AC
CRIT: (ATK_ROLL >= AC + 10) | (NAT)
CALC: (HIT * DMG) + (CRIT * (DMG * 2))

output HIT * HIT
output [if HIT | NAT]

output NAT named "nat 20"
output CRIT named "crit"
output HIT named "hit"
output HIT & NAT named "hit and nat 20"
output HIT & CRIT named "hit and crit at the same time"
output CALC named "calc"

function: hit ac AC with bonus ATK for DMG dmg {
 ROLL: 1d20
 ATK_ROLL: ROLL + ATK
 CRIT: (ROLL = 20) | (ROLL != 20 & (ATK_ROLL >= AC + 10))
 HIT: (ATK_ROLL >= AC) & !CRIT
 result: (HIT * DMG) + (CRIT * (DMG * 2))
}

output [hit ac 20 with bonus -1 for 10 dmg] named "Test"

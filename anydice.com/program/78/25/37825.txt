function: explode N:n on SIZE:n {
 if N = SIZE & SIZE < 20 {
  result: N + [explode d(SIZE + 2) on SIZE + 2]
 }
 if N = SIZE & SIZE = 20 {
  result: N + [explode d(SIZE) on SIZE]
 }
 result: N
}

output [explode d4 on 4] named "d4"
output [explode d10 on 10] named "d10"
output [explode d20 on 20] named "d20"

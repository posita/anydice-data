SIZE: 6 \Die size\
ITERATIONS: 15 \Only used if LOOP > 1\
LOOP: 12 \Iterates through 1..LOOP iterations if > 1\

function: iszero DIE:n {
  if DIE = SIZE {
    result: -10000
  }
  result: DIE
}

function: pig {
  ACC: 0
  loop I over {1..ITERATIONS} {
    CURR: 1dSIZE
    ACC: ACC + [iszero CURR]
  }
  result: [highest of 0 and ACC]
}

if LOOP > 1 {
  loop I over {1..LOOP} {
    ITERATIONS: I
    output [pig] named "[ITERATIONS] iterations"
  }
} else {
  output [pig] named "Pig"
}

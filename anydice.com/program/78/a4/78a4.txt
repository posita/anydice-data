MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: save ROLL:n vs DC:n {
 if ROLL = 1 { result: MISS }
 if ROLL + SAVE >= DC { result: HIT }
 result: MISS
}

ATTACK: 6
SAVE: 7
DAMAGE: 1d10+5
loop AC over {5..30} {
 output [attack d20 vs AC]*DAMAGE named "[AC]"
}

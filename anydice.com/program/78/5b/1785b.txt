
\calculates the probability for all damage across multiple attacks\
function: doturn ATTACKS:n ATKBONUS:n AC:n DICESEQ:d{
 HITS: [rollhits ATTACKS ATKBONUS AC]
 DMGTOTAL: 0
 loop N over {#(HITS)}{
  DMGTOTAL: DMGTOTAL + (N@HITS * [rolldamage DICESEQ])
 }
 result: DMGTOTAL
}

\gives the odds of hitting x amount of attacks across ATTACKS against AC with ATKBONUS\
function: rollhits ATTACKS:n ATKBONUS:n AC:n {
 TOTAL: 0
 loop N over {1..ATTACKS}{
  TOTAL: TOTAL + ((d20+ATKBONUS) >= AC)
 }
 result: TOTAL
}

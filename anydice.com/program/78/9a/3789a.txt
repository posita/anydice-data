set "maximum function depth" to 4
function: explode N:n {
 if N = MAX {
  result: N-1+[explode D]
 }
 result: N
}
function: drn D:d {
 MAX: [maximum of D]
 result: [explode D]
}
function: nested D:d N:n times {
 X: 0
 loop N over {1..N}{
  X: X + [ nested sub D ]
 }
 result: X
}
function: nested sub D:n {
 result: d(D*3+4)
}

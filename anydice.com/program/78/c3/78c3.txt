N: 4 \number of dice\
S: 0
A: 0
function: check if S:n succeeded{
  if S > 4 {
    result: 1
  }
  result: 0
}
function: check A:n advantage{
  if A = 4 | A = 6 {
    result: 1
  }
  if A = 1 {
    result: -1
  }
  result: 0
}
loop I over {1..N} {
  X: 1d6
  S: S+[check if X succeeded]
  A: A+[check X advantage]
}
output S named "Successes"
output A named "Advantage"

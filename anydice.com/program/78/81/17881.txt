ATTACK_NUMBER: 1
ATTACK_MIN: 1
ATTACK_MAX: 12
ATTACK_MOD: 0

CRIT_RANGE: 8

DODGE_NUMBER: 1
DODGE_MIN: 1
DODGE_MAX: 12
DODGE_MOD: -1
ARMOR: 1

DAMAGE_NUMBER: 2
DAMAGE_MIN: 1
DAMAGE_MAX: 6
DAMAGE_MOD: 2


function: attack ATTACK_ROLL:n min ATTACK_ROLL_MIN max ATTACK_ROLL_MAX:n vs dodge DODGE_ROLL:n armor ARMOR:n range CRIT_RANGE miss MISS:n graze GRAZE:n hit HIT:n crit CRIT:n {
if ATTACK_ROLL = ATTACK_ROLL_MIN {
 result: MISS
}
else if ATTACK_ROLL < DODGE_ROLL {
 result: [highest of GRAZE - ARMOR and 1]
}
else if ATTACK_ROLL - DODGE_ROLL >= CRIT_RANGE {
 result: CRIT
}
else if ATTACK_ROLL >= DODGE_ROLL {
 result: [highest of HIT - ARMOR and 1]
}
}


ATTACK_ROLL: ATTACK_NUMBERd{ATTACK_MIN..ATTACK_MAX} + ATTACK_MOD
ATTACK_ROLL_MIN: ATTACK_NUMBER * ATTACK_MIN + ATTACK_MOD
ATTACK_ROLL_MAX: ATTACK_NUMBER * ATTACK_MAX + ATTACK_MOD

DODGE_ROLL: DODGE_NUMBERd{DODGE_MIN..DODGE_MAX} + DODGE_MOD

DAMAGE_MISS: 0
DAMAGE_GRAZE: DAMAGE_NUMBER * DAMAGE_MIN + DAMAGE_MOD
DAMAGE_HIT: DAMAGE_NUMBERd{DAMAGE_MIN..DAMAGE_MAX} + DAMAGE_MOD
DAMAGE_CRIT: DAMAGE_NUMBER * DAMAGE_MAX + DAMAGE_MOD

output [attack ATTACK_ROLL min ATTACK_ROLL_MIN max ATTACK_ROLL_MAX vs dodge DODGE_ROLL armor ARMOR range CRIT_RANGE miss DAMAGE_MISS graze DAMAGE_GRAZE hit DAMAGE_HIT crit DAMAGE_CRIT]

function: adv N {
  result: [highest 2 of (2+N)d10]
}

function: norm {
  result: 2d10
}

function: dis N {
 result: [lowest  2 of (2+N)d10]
}

function: rank R {
  if R = 1 {result: [explode d4]}
  if R = 2 {result: [explode d6]}
  if R = 3 {result: [explode d8]}
  if R = 4 {result: [explode d10]}
  if R = 5 {result: [explode d12]}
  if R = 6 {result: 2d[explode d6]}
  if R = 7 {result: [explode d8] + [explode d6]}
  result: {0}
}

loop N over {1..3} {
  loop R over {0..7} {
    output [adv N] + [rank R] named "advantage [N] + rank [R]"
    output [norm] + [rank R] named "rank [R] (normal)"
    output [dis N] + [rank R] named "disadvantage [N] + rank [R]"
}}

\
 0: Disaster
 1: Conflict
 2: Triumph
 3: Twist
\

function: action SCORE:n with TRIUMPH:n {
 if SCORE = 6 {
  result: 2 + (TRIUMPH = 6)
 }
 result: SCORE >= 4
}

function: reaction SCORE:n with TRIUMPH:n {
 result: 6 - SCORE
}


loop D over {0..2} {
 loop P over {0..10} {
  S: 0
  T: 0

  if P <= D {
   F: 2 + (D*2) - P
   S: F@Fd6
  } else {
   S: (D+1)@Pd6
   
   if P > D+1 {
    T: (D+2)@Pd6
   }
  }
  output [reaction S with T] named "[P]d at [D]"
 }
}

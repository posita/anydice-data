MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
  if ROLL = 1 {
    if (ROLL + ATTACK >= DEFENSE + 10) {result: HIT}
  result: MISS
  }
  if ROLL = 20 {
    if (ROLL + ATTACK >= DEFENSE) {result: CRITICALHIT}
    if (ROLL + ATTACK >= DEFENSE - 10) & (ROLL + ATTACK <= DEFENSE + 10) {result: HIT}
  result: MISS
  }
  if (ROLL + ATTACK >= DEFENSE) {
    if (ROLL + ATTACK >= DEFENSE + 10) {result: CRITICALHIT}
  result: HIT
  }
  result: MISS
}


function: damage for ATTACKRESULT:n {
  if ATTACKRESULT = CRITICALHIT {
    result: CRITICALDAMAGE
  }
  if ATTACKRESULT = HIT {
    result: HITDAMAGE
  }
  result: MISSDAMAGE
}

ATTACK: 19
HITDAMAGE: 2d10+10
CRITICALDAMAGE: 2*(2d10+10)
MISSDAMAGE: 0
loop AC over {31..37} {
 output [damage for [attack [highest 1 of d20] vs AC]] named "[AC]"
}

SIZE: 10 \Die size\
COUNT: 1 \Num dice\
ITERATIONS: 5 \Only used if LOOP > 1; d6 max 120 & d10 max 90\
LOOP: 1 \Iterates through 1..LOOP iterations if > 1; d6 max 50 & d10 max 40\

function: pig {
  ACC: 0
  loop I over {1..ITERATIONS} {
    ACC: ACC + COUNTd{-1000, {1..SIZE-1}}
  }
  result: [highest of 0 and ACC]
}

if LOOP > 1 {
  loop I over {1..LOOP} {
    ITERATIONS: I
    output [pig] named "[ITERATIONS] iterations"
  }
} else {
  output [pig] named "[ITERATIONS] Pig"
}

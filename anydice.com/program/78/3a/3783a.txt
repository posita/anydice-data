function: ROLL:n vs AC:n {
 if ROLL + ATTACK >= AC + 10 | ROLL = 20 {
   result: 20
 }  
 if ROLL + ATTACK >= AC {
   result: 10
 }
 if ROLL + ATTACK > AC - 10 {
   result: 1
 }
 if ROLL + ATTACK <= AC - 10 {
   result: -1
 }
}



AC: d{10..20}
STRENGTH: 3
ATTACK: STRENGTH -1
output [d20 vs AC] named "-1"

AC: d{10..20}
STRENGTH: 3
ATTACK: STRENGTH
output [d20 vs AC] named "Flat"

AC: d{10..20}
STRENGTH: 3
ATTACK: STRENGTH +1
output [d20 vs AC] named "+1"


AC: d{10..20}
STRENGTH: 3
ATTACK: STRENGTH +2
output [d20 vs AC] named "+2"

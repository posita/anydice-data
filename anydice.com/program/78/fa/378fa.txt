
function: count T:s in A:s reroll L:n for LT:s {
   result: [count T in A] + [count LT in ([lowest of L and #A-[count T in A]])d6]
}

function: roll N:n for S:n reroll L:n for LS:n {
   result: [count {S..6} in Nd6 reroll L for {LS..6}]
}

output [roll 5 for 5 reroll 3 for 6] named "normal"
output [roll 5 for 6 reroll 3 for 6] named "+1"

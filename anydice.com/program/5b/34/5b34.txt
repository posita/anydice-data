STRENGTH: 0
BAB: 3
ATTACK_ONE: BAB + STRENGTH + -2
WEP_ONE: 1d8
CRITMOD_ONE: 2
THREAT_ONE: 19
ATTACK_TWO: BAB + STRENGTH + -2
WEP_TWO: 1d6
CRITMOD_TWO: 2
THREAT_TWO: 19
AC: 15

NORMHIT: WEP + STRENGTH
CRITHIT: CRITMOD d (WEP + STRENGTH)

function: DTWENTY:n vs AC {
 if DTWENTY = 1 { result: 0 }
 if DTWENTY >= THREAT & DTWENTY + ATTACK >= AC {
  result: [threat d20]
 }
 if DTWENTY + ATTACK >= AC | DTWENTY = 20 { result: NORMHIT }
 result: 0
}

function: threat DTWENTY:n {
 if DTWENTY = 1 { result: NORMHIT }
 if DTWENTY + ATTACK >= AC | DTWENTY = 20 {
  result: CRITHIT
 }
 result: NORMHIT
}

output [d20 vs AC]

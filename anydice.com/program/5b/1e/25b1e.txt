CRIT: 0

\Attack\
function: attack { result: 
   ((CRIT+1)*1 + CRIT)d10 + 5 \Pike\ 
   + 10 \Great Weapon Master\
   + 2 \Rage\
}


\Action\
function: action { result: 
   3d[attack] \Fighter Extra Attack(s)\
   + 1d[attack] + ((CRIT+1)*1)d8 \Dread Ambusher\      
}

\Bonus Action\
function: bonusaction { result: 0 }

\Turn\
function: turn { result: 
   0 +
   + ((CRIT+1) * 2)d6 \Surprise Attack\
   + ((CRIT+1) * 2)d6 \Sneak Attack\
}


function: total { result: 2d[action] + [bonusaction] + [turn] }

output [total]

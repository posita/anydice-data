function: DTWENTY:n vs AC ATTACK:n THREAT:n {
 if DTWENTY = 1 { result: 0 }
 if DTWENTY >= THREAT & DTWENTY + ATTACK >= AC {
  result: [threat d20]
 }
 if DTWENTY + ATTACK >= AC | DTWENTY = 20 { result: NORMHIT & [1d8 and 0] }
 result: 0
}

function: WEP:n and STRENGTH:n

function: threat DTWENTY:n {
 if DTWENTY = 1 { result: NORMHIT }
 if DTWENTY + ATTACK >= AC | DTWENTY = 20 {
  result: CRITHIT
 }
 result: NORMHIT
}

function: attack is BAB:n and STRENGTH:n and MISC:n {
 result: BAB+STRENGTH+MISC
}

 ATTACKS: 7
 HITON: 5
 HIT: 3
 CRIT: 6
 
function: damage ROLL:n {
 if ROLL = 6 {
  result: CRIT
 }
 if ROLL >= HITON {
  result: HIT
 }
 result: 0
}

function: kokilled ROLL:s {
  DAMAGE: 0
  loop N over {1..#ROLL}{
   DAMAGE: DAMAGE + [damage N@ROLL]
  }
  result: DAMAGE
}

output [kokilled ATTACKSd6]

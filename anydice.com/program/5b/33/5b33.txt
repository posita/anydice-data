DMG: 0
STRENGTH: 0
BAB: 4
ATTACK_ONE: BAB + STRENGTH + 0
WEP_ONE: 1d8
CRITMOD_ONE: 2
THREAT_ONE: 19
AC: 15

NORMHIT: WEP_ONE + STRENGTH
CRITHIT: CRITMOD_ONE d (WEP_ONE + STRENGTH)

function: DTWENTY:n vs AC {
 if DTWENTY = 1 { result: 0 }
 if DTWENTY >= THREAT_ONE & DTWENTY + ATTACK_ONE >= AC {
  result: [threat d20]
 }
 if DTWENTY + ATTACK_ONE >= AC | DTWENTY = 20 { result: NORMHIT }
 result: 0
}

function: threat DTWENTY:n {
 if DTWENTY = 1 { result: NORMHIT }
 if DTWENTY + ATTACK_ONE >= AC | DTWENTY = 20 {
  result: CRITHIT
 }
 result: NORMHIT
}

output [d20 vs AC]

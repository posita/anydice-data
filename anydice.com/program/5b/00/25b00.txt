function: ROLL:n replace FILTER:s with REROLL:d {
  if ROLL = FILTER { result: REROLL }
  result: ROLL
}

function: DIE:d reroll FILTER:s {
  result: [DIE replace FILTER with DIE]
}

function: kind DIE:d {
  result: [DIE reroll 1]
}

function: DIE:d keep K:n {
  result: [highest K of DIE]
}

function: DIE:d keep lowest N:n {
  result: [lowest N of DIE]
}

function: DIE:d keep highest N:n {
  result: [highest N of DIE]
}

function: DIE:d with advantage {
  result: [2dDIE keep 1]
}

function: DIE:d with disadvantage {
  result: [2dDIE keep lowest 1]
}

MISS: 0
GLANCE: 1
HIT: 2
CRIT: 3

function: ROLL:n with BONUS:n hits AC:n {
  if ROLL = 1 {
    result: MISS
  }
  if ROLL = 20 {
    if ROLL + BONUS > AC {
      result: CRIT
    }

    result: HIT
  }
  if ROLL + BONUS > AC {
    result: HIT
  }
  if ROLL + BONUS = AC {
    result: GLANCE
  }
  result: MISS
}

function: BONUS:n hits AC:n {
  result: [d20 with BONUS hits AC]
}

function: ROLL:n with BONUS:n saves against DC:n {
  if ROLL + BONUS >= DC {
    result: GLANCE
  }

  result: HIT
}

function: BONUS:n saves against DC:n {
  result: [d20 with BONUS saves against DC]
}

function: do DAMAGE:d with BONUS:n for RESULT:n {
  if RESULT = CRIT {
    result: [maximum of DAMAGE] + DAMAGE + BONUS
  }
  if RESULT = HIT {
    result: DAMAGE + BONUS
  }
  if RESULT = GLANCE {
    result: (DAMAGE + BONUS) / 2
  }

  result: 0
}

function: misha RESULT:n AC:n IS_FIRST_ATTACK:n {
  if RESULT > MISS {
    SPELL_RESULT: [+0 saves against 13]
    DAMAGE: 1d8
    if IS_FIRST_ATTACK = 1 {
      DAMAGE: 2d8
    }
    result: 1d4 + [do 1d10 with +0 for SPELL_RESULT] + [do 1d8 with +5 for RESULT] + [do DAMAGE with +5 for [+9 hits AC]]
  }

  if IS_FIRST_ATTACK = 1 {
    result: [misha [+9 hits AC] AC 0]
  }

  result: 0
}

loop AC over {15..20} {
  output [misha [+9 hits AC] AC 1] named "AC: [AC]"
}

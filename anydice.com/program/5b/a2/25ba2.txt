\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ABILITYSCORE + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

ABILITYSCORE: 5
ACCBONUS: 0
DMGBONUS: 5

CRITROLL: 20
ENEMYAC: 18
DAMAGE: 1d8 + ABILITYSCORE + DMGBONUS
CRITDAMAGE: 2d8 + ABILITYSCORE + DMGBONUS

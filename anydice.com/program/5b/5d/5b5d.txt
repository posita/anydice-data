STRENGTH: 0
BAB: 6
AC: 15
ATTACKS: 1
 HITMOD_ONE: BAB + STRENGTH + -2
    WEP_ONE: 1d8
CRITMOD_ONE: 2
 THREAT_ONE: 19
 HITMOD_TWO: BAB + STRENGTH + -2
    WEP_TWO: 1d6
CRITMOD_TWO: 2
 THREAT_TWO: 19
 HITMOD_TRE: BAB + STRENGTH + -7
    WEP_TRE: 1d8
CRITMOD_TRE: 2
 THREAT_TRE: 20
 HITMOD_FOR: BAB + STRENGTH + -8
    WEP_FOR: 1d6
CRITMOD_FOR: 2
 THREAT_FOR: 20
 HITMOD_FIV: BAB + STRENGTH + -6
    WEP_FIV: 1d6
CRITMOD_FIV: 2
 THREAT_FIV: 20
 HITMOD_SIX: BAB + STRENGTH + -6
    WEP_SIX: 1d6
CRITMOD_SIX: 2
 THREAT_SIX: 20

function: DTWENTY:s HITMOD THREAT vs WEP CRITMOD ac {
 CRITHIT: CRITMOD d (WEP + STRENGTH)
 NORMHIT: WEP + STRENGTH

 if DTWENTY = 1 { result: 0 }
 if DTWENTY >= THREAT & DTWENTY + HITMOD >= AC {
  result: [threat d20]
 }
 if DTWENTY + HITMOD >= AC | DTWENTY = 20 { result: NORMHIT }
 result: 0
}

function: threat DTWENTY:n {
 if DTWENTY = 1 { result: NORMHIT }
 if DTWENTY + HITMOD >= AC | DTWENTY = 20 {
  result: CRITHIT
 }
 result: NORMHIT
}


ONE: [d20 HITMOD_ONE THREAT_ONE vs WEP_ONE CRITMOD_ONE ac]
TWO: [d20 HITMOD_TWO THREAT_TWO vs WEP_TWO CRITMOD_TWO ac]
TRE: [d20 HITMOD_TRE THREAT_TRE vs WEP_TRE CRITMOD_TRE ac]
FOR: [d20 HITMOD_FOR THREAT_FOR vs WEP_FOR CRITMOD_FOR ac]
FIV: [d20 HITMOD_FIV THREAT_FIV vs WEP_FIV CRITMOD_FIV ac]
SIX: [d20 HITMOD_SIX THREAT_SIX vs WEP_SIX CRITMOD_SIX ac]

output ONE+TWO+TRE+FOR+FIV+SIX

function: diffa ROLL:s {
if 2@ROLL = 10 {
  result: 1
}
if 2@ROLL >= 8 {
  result: 2
}

if 2@ROLL >= 6{
  result: 3
}

if 2@ROLL >=2{
  result: 4
}

 result: 5
}

function: diffb ROLL:s {
if 3@ROLL = 10 {
  result: 1
}
if 3@ROLL >= 8 {
  result: 2
}

if 3@ROLL >= 6{
  result: 3
}

if 3@ROLL >=2{
  result: 4
}

 result: 5
}

loop P over {1..4} {
 output [diffa Pd10] named "[P]d Diff 1"
 output [diffb Pd10] named "[P]d Diff 2"
 }

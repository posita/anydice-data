function: compare DIE:n TARGET:n {
  if DIE >= TARGET {
    result: 1
  }
  result: 0
}

function: roll MAG:n HIT:n ARM:n {
  HITS: 0
  loop _ over {1..MAG} {
    HITROLL: 1d6
    HITS: HITS + [compare HITROLL HIT]
  }
  DAMS: 0
  loop I over {HITS} {
    if I != 0 {
      DAMROLL: 1d6
      DAMS: DAMS + [compare DAMROLL ARM]
    }
  }
  result: DAMS
}

output [roll 2 3 3]
output [roll 2 4 4]

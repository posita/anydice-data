function: corner SPEED:n MOD:n DIFF:n PENALTY:d {
 ROLL: 1d20+MOD
 DC: SPEED+DIFF
 FAIL: ROLL < DC
 SUPERFAIL: ROLL < DC-7
 result: ((SPEED)-(FAIL*PENALTY))*!SUPERFAIL
}

SPEED: 15
CORNER: 2
DIFF: 3
COST: 5
PENALTY: 1d8

output [corner SPEED-4 CORNER+2 DIFF PENALTY]
output [corner SPEED CORNER DIFF PENALTY]
output [corner SPEED CORNER DIFF-COST PENALTY]-COST

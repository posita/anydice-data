function: flip to succeed A:n B:n {
    if A = 0 & B = 0 { result: 100 }
    if A < B { result: A*10 + B }
    result: B*10 + A
}

function: flip to fail A:n B:n {
    if A = 0 & B = 0 { result: 100 }
    if A > B { result: A*10 + B }
    result: B*10 + A
}

output d100 named "normal"
output [lowest 1 of 2d100] named "advantage"
output [highest 1 of 2d100] named "disadvantage"
output [flip to succeed 1d10-1 1d10-1] named "flip to succeed"
output [flip to fail 1d10-1 1d10-1] named "flip to fail"

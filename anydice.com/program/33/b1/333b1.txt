function: onering FEAT:s SUCCESS:s{
  if FEAT = 12 {TOTAL: 1000}
  
  TENG: [count {4,6,8,12} in FEAT] + [count {4,6} in SUCCESS]
  
  if TENG > 1 {LEVEL: 2}
  if TENG = 1 {LEVEL: 1}
  if TENG <= 0 {LEVEL: 0}
  if [count 12 in FEAT] {LEVEL: LEVEL +1}
  
  result: LEVEL

}

loop N over {0..4} {
  output [onering d12 Nd6] named "d12 and [N]d6"
}

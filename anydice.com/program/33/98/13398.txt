function: ROLL:n vs AC:n {
if ROLL = 20 {
result: CRITICAL + DAMAGE
}
if ROLL + ATTACK >= AC {
result: DAMAGE
}
result: 0
}

AC: d{14..20}
STRENGTH: 3
PROF: 3

ATTACK: PROF + STRENGTH
DAMAGE: 2d6 + STRENGTH 
CRITICAL: 2d6 output [d20 vs AC] named "Not GWM"

ATTACK: PROF + STRENGTH - 6
DAMAGE: 2d6 + STRENGTH + 9
CRITICAL: 2d6 output [d20 vs AC] named "GWM"

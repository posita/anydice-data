function: reroll elements of ARRAY:s if less than N:n {
 LEN: #ARRAY
 RET:0
 loop A over {1..LEN} {
  if A @ ARRAY < N {
   RET: RET + 1d6
  }
  else {
   RET: RET + A @ ARRAY
  }
 }
result: RET
}

ABILITIES: 6d[reroll elements of [highest 3 of 4d6] if less than 2]
loop P over {1..6} {
 output P @ ABILITIES named "Ability [P]"
}

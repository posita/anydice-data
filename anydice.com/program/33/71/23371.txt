MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{12..22}

ATTACK: 6
HITDAMAGE: d4 + 4
CRITICALDAMAGE: 2d4 + 8
MISSDAMAGE: 0
output [attack] named "Fist"

ATTACK: 5
HITDAMAGE: d6 + 3
CRITICALDAMAGE: 2d6 + 6
output [attack] named "Shortsword"

ATTACK: 5
HITDAMAGE: d8 + 3
CRITICALDAMAGE: 2d8 + 6
output [attack] named "Spear"

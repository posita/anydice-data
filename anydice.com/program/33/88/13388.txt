\**** Damage calculator duelling Vs archery ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: attackrolloff ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGEOFF } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMODOFF + ACCBONUSOFF >= AC { result: DAMAGEOFF }
 result: 0
}

ENEMYAC: d{12..20}

POSITIVE: 1
NEGATIVE: 0

function: attack ATK defenders DEF modifier DRM sign SGN roll DCE:n{
 LN: ATK
 if SGN = POSITIVE { DCE: DCE + DRM }
 else {DCE: DCE - DRM}
 if DCE = 0 { LN: LN / 4 }
 if DCE = {1, 2} { LN: LN / 2 }
 if DCE = {7, 8} { LN: LN *3 / 2 }
 if DCE = 9 { 
  LN: LN * 2
  if LN < 6 { LN: 6 }
 } 
 result: LN / 6
}

loop M over {3..12} {
 output [attack M defenders {6, 6, 5} modifier 2 sign NEGATIVE roll d{0..9}] named "# Step losses | [M] total attack strength"
}

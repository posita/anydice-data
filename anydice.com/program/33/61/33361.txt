function: eroll highest NUMBER:n of DICE:d 
{ 
  SUM: 0 
  loop I over {1..NUMBER} 
  { 
    SUM: SUM + [explode and reroll I@DICE on [maximum of I@DICE]] 
  } 
  result: SUM 
} 
function: explode and reroll NUMBER:n on VALUE:n 
{ 
  if NUMBER = VALUE 
  { 
    BANK: VALUE 
    CHANCE: [explode and reroll d(VALUE) on VALUE] + [explode and reroll d(VALUE) on VALUE] 
    result: (BANK + CHANCE) / 2 
  } 
  result: NUMBER 
} 
set "maximum function depth" to 5

output [eroll highest 3 of 3d6] named "Base"

output [eroll highest 4 of 4d6] + 1 named "Extra Die +1"
output [eroll highest 5 of 5d6] + 2 named "Extra Die +2"
output [eroll highest 6 of 6d6] + 3 named "Extra Die +3"

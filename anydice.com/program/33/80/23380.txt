
function: defence dice helper ROLL_RESULT:s {
  MULT: [count -1 in ROLL_RESULT]
  BASE: 0
  loop P over ROLL_RESULT {
    if P > 0 {
      BASE: BASE + P
    }
  }
  result: (2^MULT)*BASE
}

function: roll N:n defence dice {
  result: [defence dice helper Nd{0,0,1,1,2,-1}]
}

output [roll 4 defence dice] named "Defence Dice"

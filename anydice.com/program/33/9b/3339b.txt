\Ironsworn Dice Roll Results\

function: compare R:n to X:n and Y:n 
{
  if R > X
  {
    if R > Y
    {
      result : 2
    }
    else
    {
      result : 1
    }
  }
  else
  {
    if R > Y
    {
      result : 1
    }
    else
    {
      result : 0
    }
  }
}

function: min A:n or B:n
{
  if A < B
  {
     result: A
  }
  else
  {
    result: B
  }
}

loop MODIFIER over {0..5}
{
  ROLL: [min d6 + MODIFIER or 10]
  output [compare ROLL to d10 and d10] named "[MODIFIER]"
}

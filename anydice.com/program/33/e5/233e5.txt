function: risky A:n B:n C:n {
 VALUES: [sort {A, B, C}]
 HIGH: 1@VALUES
 MID: 2@VALUES
 LOW: 3@VALUES
 if MID > 10 { result: HIGH }
 result: LOW
}

function: riskytwo A:n B:n C:n {
 VALUES: [sort {A, B, C}]
 HIGH: 1@VALUES
 MID: 2@VALUES
 LOW: 3@VALUES
 if (HIGH-MID) <= (MID-LOW) { result: HIGH }
 result: LOW
}

output 3d6
output [middle 1 of 3d20]
output 1d20
output [riskytwo 1d20 1d20 1d20]
output [risky 1d20 1d20 1d20]

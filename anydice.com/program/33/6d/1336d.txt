\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 2
DMGBONUS: 4
PROFICIENCY: 4
ACCBONUS: 4
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d4+4d6+2d8+ ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d4+8d6+ 4d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Booming Blade Sneak attack with Dagger +2 proficiency +4 Attack modifier"

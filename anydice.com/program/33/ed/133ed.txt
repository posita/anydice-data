function: take highest NUM:n of ARRAY:s and reroll if leq TEST:n {
 RET: 0
 loop A over {1..NUM} {
  if A @ ARRAY <= TEST {
   RET: RET + 1@1d6
  }
  else {
   RET: RET + A @ ARRAY
  }
 }
result: RET
}

ABILITIES: 6d[take highest 3 of 4d6 and reroll if leq 1]
loop P over {1..6} {
 output P @ ABILITIES named "Ability [P]"
}

function: A:n prog B:n{
if A <= B{ result: B }
else{ result: A }
}

function: noright A:d{
MAX: [maximum of A]
result: [noright A helper]
}

function: noright N:n helper{
if N = MAX {result: N+2d10}
else {result: N}
}

function: rebote NUMBER:n{
 if NUMBER > 9 {
  if (NUMBER-(NUMBER-9)*2) > 4 { result: NUMBER-(NUMBER-9)*2 }
  if (NUMBER-(NUMBER-9)*2) < 5 { result: NUMBER-8 }
  }
 result: [absolute NUMBER]
}

function: topar NUMBER:n to TOP:n{
 if NUMBER > TOP { result: TOP }
 result: NUMBER
}

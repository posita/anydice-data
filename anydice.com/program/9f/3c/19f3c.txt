function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

MISS: 0
HIT: 1
CRITICALHIT:2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL >= DEFENSE { result: HIT }
 result: MISS
}

function: DAMAGE:n or CRIT:n for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRIT
 }
 if ATTACKRESULT = HIT {
  result: DAMAGE
 }
 result: 0
}

loop AC over {10..30} {
output [3d6+7 or 6d6+7 for [attack d20+7 vs AC]] named "AC=[AC]"

}

loop AC over {10..30} {
output [3d6+17 or 6d6+17 for [attack d20+2 vs AC]] named "GWM - AC=[AC]"
}

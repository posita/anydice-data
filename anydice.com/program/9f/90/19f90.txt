function: reroll N:n once {
 if N <= 2 { result: d6 }
 result: N
}

function: attack N:n with crit on C:n and sneak attack S:n {
 if N >= C { result: 8dREROLLED + (2*S)dREROLLED + 5 }
 result: 2dREROLLED + SdREROLLED + 5
}

REROLLED: [reroll d6 once]

A: [attack d20 with crit on 19 and sneak attack 0]
B: [attack d20 with crit on 19 and sneak attack 1]
C: [attack d20 with crit on 20 and sneak attack 3]

output 2dA
output 2dB
output 1dC

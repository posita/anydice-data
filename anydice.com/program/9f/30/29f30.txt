ATTR: 2d3

loop DIE over {12,10,8,6,4} {
  CHECK: 2dDIE
  CHECK_AID: [lowest 1 of 2dDIE]+dDIE
  CHECK_HIN: [highest 1 of 2dDIE]+dDIE

  \output CHECK named "check [CHECK]"\
  \output CHECK_AID named "aided check [CHECK_AID]"\
  \output CHECK_HIN named "hindered check [CHECK_HIN]"\

  output CHECK <= ATTR named "target [DIE] (<= [CHECK])"
  \output CHECK_AID <= ATTR named "aided target [DIE] (<= [CHECK_AID])"\
  \output CHECK_HIN <= ATTR named "hindered target [DIE] (<= [CHECK_HIN])"\
}

\
output 2d3 named "attribute"
output 2d6 named "check"
output [lowest 2 of 3d6] named "aided check"
output [highest 2 of 3d6] named "hindered check"
\

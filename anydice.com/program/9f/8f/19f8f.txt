if(A=0) {result: 0}
else if(A=1) {result: 1}
else if(A=2) {result: 1}
else if(A=3) {result: 2}
else if(A=4) {result: 2}
else if(A=5) {result: 3}
else if(A=6) {result: 3}
else if(A=7) {result: 4}
else if(A=8) {result: 4}
else if(A=9) {result: 5}
else if(A=10) {result: 5}
}

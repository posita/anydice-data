function: matches A:s and B:n {
  M: 0
  loop N over A{
    if N = B {
      M: M+1
    }
  }
  result: M
}

function: satarichi ROLL:s and DIFICULTAD:n {
  E: [matches ROLL and 6]
  P: [matches ROLL and 1]
  if E >= DIFICULTAD {
    result: 1
  }
  else { 
    if P >= 2 { 
      result: -1 
    }
  }
  result: 0 
}

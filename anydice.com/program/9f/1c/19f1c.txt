function: sigil NUM:n add ADD:n{
CALC:0
FAIL:0
loop N over {1..NUM}{CALC:CALC+[sigil helper d6]}
result:[to zero CALC ADD]
}

function: sigil helper DIE:n {
if DIE = 6 {R: 1+(d6>3)}
else {R: (DIE>3)}
result:R
}

function: to zero NUM:n VAL:n{
if NUM=0{result:0}
result: NUM+VAL
}

loop I over {2..10}{
 loop J over {0..5}{
  output[sigil I add J] named "Sigil [I] add [J]"
}}


\**** Damage calculator duelling Vs archery ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 7
DMGBONUS: 4
PROFICIENCY: 3
ACCBONUS: 0
CRITROLL: 19
ATTACKTIMES: 2
DAMAGE: 2d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 3d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] named "Fighter Greatsword"

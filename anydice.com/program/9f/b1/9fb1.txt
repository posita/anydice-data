\Attributes range from 1 to 5 and add 1D6 per, Skills range from 0 to 7 and add 1D6+1 \
\A Free Dice is awarded automatically, this preserves a bell curve of sorts at all levels\

\Skill\
loop S over {0..7} {
        \Skill Bonus - Skills add 1D6+1 per level\

	\Attribute - Attributes add 1D6 per level\
	loop A over {1..5} {
                N: A + 1 /Free Dice/
		B: S
		output [highest 2 of Nd6] + B named "Skill:[S] Att:+[A], Skill Bonus:+[B] rolling [N] Dice"

		
		
	}
}

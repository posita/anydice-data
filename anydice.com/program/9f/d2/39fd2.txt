\Arbitrary Cased Ammo tester, a tool for Ops & Tactics\


\Change POWDER to determine powder load. -1 = Low Pressure, 0 = Standard Pressure, 1 = High Pressure, 2 = Over Pressure\

POWDER: 0

DICE: 10
SIZE: 8
BONUS: 0
BONUS: BONUS+(DICE*POWDER)

CALIBER: DICEdSIZE+BONUS

if DICE <= 3 { JHP: 2 } else {JHP: DICE-1}
if DICE <= 4 { LJHP: 1 } else {LJHP: DICE-3}

output CALIBER named "FMJ, Hardball, Surplus, Match, Tracer, Non-Lethal, Tungsten, Silver, Acid(Immediate), Cold, Electric, UV, Silver Nitrate, Gel vs Head"
output CALIBER+JHP named "Jacketed Hollow Point"
output CALIBER+LJHP named "JHP, Light"
output CALIBER+DICE named "JHP, Heavy"
output CALIBER+DICE named "Semi-Jacketed Hollow Point"
output CALIBER+(DICE*2) named "Semi-Jacketed Soft Points"
output CALIBER+(DICE*2) named "Soft Points"
output CALIBER+(DICE*3) named "Heavy Soft Points"
output (DICE-(DICE/2))dSIZE + (BONUS-(BONUS/2)) + (DICE/2)dSIZE + (BONUS/2) named "Steel Core"
output (DICE-1)dSIZE + BONUS + (2+((DICE-2)/2))dSIZE + ((2+((DICE-2)/2))*POWDER) named "Armor Piercing"
output (DICE+1)dSIZE + BONUS + POWDER named "Incendiary, Exploding, Gel vs Non-Head"
output (DICE+2)dSIZE + BONUS named "Acid(Total)"
output CALIBER+3 named "Frangible vs unarmored target"
output CALIBER-4 named "Frangible vs armored target"
output CALIBER+4+(DICE*2) named "Wadcutter, 1st range increment, vs unarmored target"
output CALIBER+(DICE*2) named "Semi-Wadcutter, within 2 range increments, vs unarmored target"
output CALIBER+6+(((DICE-3)/2)*3) named "Solid"
output CALIBER-8 named "Ratshot"

FLANKING: 1
ATTACKBONUS: 8 + (FLANKING * 2)
CRITTHREAT: 19
AC: 15

HITCHANCE: 1 - (AC - ATTACKBONUS - 1) / 20
output HITCHANCE named "Hit chance"

CRITCONFIRM: 1 - (AC - ATTACKBONUS - 1) / 20
CRITCHANCE: (20 - CRITTHREAT + 1) / 20 * CRITCONFIRM
output CRITCHANCE named "Crit chance"

NORMALDAMAGE: d8
CRITDAMAGE: 2d8
AVERAGEDAMAGE: HITCHANCE * NORMALDAMAGE + CRITCHANCE * CRITDAMAGE
output AVERAGEDAMAGE named "Average damage"

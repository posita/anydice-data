function: altmiddle NUMBER:n of DICE:d {
 if NUMBER = #DICE { result: DICE }

 if NUMBER = 1 { result: (1 + (#DICE) / 2) @ DICE }

 FROM: 1 + (#DICE - NUMBER + 1) / 2
 TO: FROM + NUMBER - 1
 result: {FROM..TO}@DICE
}

output [middle 2 of 3d6] named "Middle, positions rounding down"
output [altmiddle 2 of 3d6] named "Middle, positions rounding up"

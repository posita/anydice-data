\SEUILS : 2 à 5, DES: 20, 12, 8, 6, NBS: 1 à 4\
\IX 0: rien, 1 : inné, IX 2: élite, IX 3: inné & élite\
SEUILS: {5}
DES: {6}
NBS: {3}
IXS:{3}
ADS: {-1}

loop SEUIL over SEUILS {
  loop DE over DES {
   loop NB over NBS {
     loop IX over IXS {
       loop AD over ADS {
         if (IX < 2) | (NB > 2) {
           if AD != 0 {NBB: NB + 1} else {NBB: NB}
           LANCER: NBB dDE
           Y: [count {1} in LANCER]
           if AD = -1 {Y: Y > 1} else {Y: Y > 0}
           if IX = {1, 3} {I: [LANCER contains DE]} else {I: 0}
           if IX = {2, 3} {X: Y} else {X: 0}
           if DE = 6 {CRITS: {6}}
           if DE = 8 {CRITS: {7, 8}}
           if DE = 12 {CRITS: {9, 10, 11, 12}}
           if DE = 20 {CRITS: {13, 14, 15, 16, 17, 18, 19, 20}}
           CRIT: -1 * ([count {CRITS} in LANCER] >= NB)
           if AD = -1 {SUCCES: -1} else {SUCCES: 0}
           SUCCES: SUCCES + [count {1..SEUIL} in LANCER] + I + X + CRIT
           output SUCCES named "[NB]d[DE] vs [SEUIL] - [IX] - [AD]"
          }
        }
      }
    }
  }
}

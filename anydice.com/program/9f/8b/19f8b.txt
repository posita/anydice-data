\Standard explode\
function: explode DIE:d {
 MAX: [maximum of DIE]
 result: [explode DIE helper]
}
function: explode N:n helper {
 if N = MAX { result: N + [explode DIE helper] }
 result: N
}
set "explode depth" to 1

\AW+ with 1 explode\
function: N:n awplus of DIE:d {
 MAX: [maximum of DIE]
 BEST: [highest 1 of (N+1)dDIE]
 result: [BEST aw DIE helper]
}
\AW- with 1 explode\
function: N:n awminus of DIE:d {
 MAX: [maximum of DIE]
 WORST: [lowest 1 of (N+1)dDIE]
 result: [WORST aw DIE helper]
}
\AW+- helper\
function: N:n aw DIE:d helper {
 if N = MAX { result: N + DIE }
 result: N
}
set "maximum function depth" to 3

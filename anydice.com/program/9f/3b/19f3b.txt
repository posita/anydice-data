set "explode depth" to 1

function: drama D:n A:n {
  if D = 6 {
    result: 6 + [highest 1 of Ad6]
  }
  result: [highest of D and [highest 1 of Ad6]]
}

function: icon A:n {
  if A = 1 {
    result: [explode d6]
  }
  else {
    result: [drama 1d6 (A-1)]
  }
}

loop N over {1..6}{
  output [icon N] named "[N] dice"
}

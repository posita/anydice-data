set "maximum function depth" to 10

function: count NUM:n in explode DIE:d{
 MAX: [maximum of DIE]
 result: [explode count DIE helper]
}
function: explode count N:n helper {
 SUCCESSES: 0
 if N >= NUM { SUCCESSES: SUCCESSES + 1 }
 if N = MAX {result: SUCCESSES + [explode count DIE helper] }
 result: SUCCESSES
}

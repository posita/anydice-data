function: test ROLL:s and LUCK:n vs RANK:n {
 LOWEST: 2@ROLL + LUCK

 if LOWEST > RANK {
  if 1@ROLL = 2@ROLL {
   if LUCK < 0 & 1@ROLL = 1 { result: 0 } 
   result: 2 
  } 
  result: 1 
 }
 if LOWEST = RANK { result: 0 }
 if 1@ROLL = 2@ROLL {
  if LUCK > 0 & 1@ROLL = 8 { result: 0 } 
  result: -2 
 }
 result: -1
}

loop N over {0..9} {
 output [test 2d8 and d{-1..1} vs N] named "Rank[N]"
}

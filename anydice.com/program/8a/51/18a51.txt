TRESHOLD: 4
REROLL: 6
DICE: d6

function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again DICE] }
 result: N >= TRESHOLD
}

NWOD: [nwod REROLL again DICE]

loop N over {1..10} {
 output [lowest of 10 and NdNWOD] named "[N][DICE]"
}

\Ydren Damage\

SCYTHE: 2d4 + 3
VAMPBLADE: 1d6
DART: 1d4+3
GRASP: 2d8
GFB: 2d4+3
CAT: 3d8
STRIKE: 1d8
THUNDER: 2d8
VAMPTOUCH: 3d6
EARTH: 3d12


function: ROLL:s earthdamage AC MOD:s{
 if (ROLL+MOD) < (AC) {
 result: EARTH
 }
 result: EARTH/2
}


output SCYTHE named "Raw Scythe"
output THUNDER named "Thunderwave"
output GRASP named "Shocking Grasp"
output VAMPTOUCH named "Vampiric Touch"
output SCYTHE+VAMPBLADE named "Scythe + Heal"
output SCYTHE+STRIKE named "Scythe + Zephyr"
output CAT named "Catapult"
output SCYTHE+GFB named "Scythe + GFB"
output [1d20 earthdamage 14 1d{-5..5}] named "Erupting Earth"
output SCYTHE + GRASP named "Scythe + Shocking Grasp"
output SCYTHE+VAMPBLADE+GFB named "Scythe + Heal + GFB"
output SCYTHE+GRASP+VAMPBLADE named "Scythe + Heal + Shocking Grasp"
output SCYTHE+VAMPBLADE+GFB+STRIKE named "Scythe + Heal + GFB + Zephyr"

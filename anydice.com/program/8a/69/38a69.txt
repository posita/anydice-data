function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

output [reroll {1,2} on d10 to d10]
output [reroll {1,2,3} on d10 to d10]
output [reroll {1,2} on d6 to d6] + [reroll {1,2} on d6 to d6]
output [reroll {1,2,3} on d6 to d6] + [reroll {1,2,3} on d6 to d6]

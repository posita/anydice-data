\**** Damage calculator elementalist ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-25\
ENEMYAC: d{15..20}

ATTACKMOD: 5
DMGBONUS: 0
PROFICIENCY: 5
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 3d8
CRITDAMAGE: 6d8 
output ATTACKTIMESd[attackroll [highest 1 of 1d20] vs ENEMYAC] named "produce flame/ray or frost"

if ROLL = 1 {result: FUMBLE}
if ROLL = {THREAT..20} {result: CRITICAL}
if ROLL + ATTACK + ESCALATION >= DEFENSE {result: HIT}
result: MISS}

MISS: 1
HIT: 1d10+4
CRITICAL: 2*(HIT)
ATTACK: 5

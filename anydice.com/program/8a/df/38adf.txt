\ Implementation details \
MISS: 0
NORMAL_HIT: 1
CRITICAL_HIT: 2

function: hit with BONUS:n vs AC:n rolling ROLL:n {
   if ROLL >= CRITICAL_THRESHOLD { result: CRITICAL_HIT }

   if ROLL = 1 { result: MISS }

   if ROLL + BONUS - AC >= 0 { result: NORMAL_HIT }

   result: MISS
}


function: attack with BONUS:n vs AC:n rolling ROLL:n with DICE:d and MOD:n {
   HIT: [ hit with BONUS vs AC rolling ROLL ]

   if HIT = MISS { result: 0 }
   
   if HIT = CRITICAL_HIT { result: DICE + DICE + MOD }
   
   result: DICE + MOD
}
\ End implementation details \

PROFICIENCY: 2
CRITICAL_THRESHOLD: 20

NORMAL_ROLL: 1d20
ADVANTAGE: [ highest 1 of 2d20 ]

XBOW: 1d6
TITAN: 1d8

STR_MOD: 7
DEX_MOD: 5

loop TARGET_AC over { 12..20 } {

   DAMAGE_BONUS: DEX_MOD + STR_MOD + 10 + 1
   HIT_BONUS: STR_MOD + PROFICIENCY - 5 + 1d4 + 1
   output [ attack with HIT_BONUS vs TARGET_AC rolling NORMAL_ROLL with TITAN and DAMAGE_BONUS ] named "Titantstring vs AC[TARGET_AC]"

   DAMAGE_BONUS: DEX_MOD + 10 + 1
   HIT_BONUS: DEX_MOD + PROFICIENCY - 5 + 1d4 + 1
   output [ attack with HIT_BONUS vs TARGET_AC rolling NORMAL_ROLL with XBOW and DAMAGE_BONUS ] named "Crossbow vs AC[TARGET_AC]"

   \ Group the results prettier. \
   output {0} named ""
}

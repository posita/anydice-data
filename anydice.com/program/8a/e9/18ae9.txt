DMG_MULT_ON_HIT: 1
DMG_MULT_ON_CRITICAL: 2
ATTACK_ROLL_DIE: d20
ATTACK_ROLL_DIE_ADV: [highest 1 of 2d20]

function: attack ROLL:n vs ENEMY_AC:n {
 if ROLL = 1 { result: DMG_MULT_ON_MISS }
 if ROLL >= CRIT_RANGE {
  if ROLL + ATTACK >= 0 { result: DMG_MULT_ON_CRITICAL }
  result: DMG_MULT_ON_HIT
 }
 if ROLL + ATTACK >= ENEMY_AC { result: DMG_MULT_ON_HIT }
 result: DMG_MULT_ON_MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = DMG_MULT_ON_CRITICAL {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = DMG_MULT_ON_HIT {
  result: HITDAMAGE
 }
 result: DMG_MULT_ON_MISSDAMAGE
}

function: reroll R:n under N:n {
   if R <= N {result: WEAPON_DICE} else {result: R}
}

function: attack {
 result: [damage for [attack ATTACK_ROLL_DIE vs ENEMY_AC]]
}

function: attackadv {
 result: [damage for [attackadv ATTACK_ROLL_DIE_ADV vs ENEMY_AC]]
}

}

}

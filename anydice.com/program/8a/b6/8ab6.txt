function: ovac N:s V {
  result: (N = V) * V
}
function: ovai N:s {
  result: [highest of [highest of [highest of [ovac N 1] and [ovac N 2]] and [highest of [ovac N 3] and [ovac N 4]]] and [highest of [ovac N 5] and [ovac N 6]]]
}
function: ova N {
  if N < 1 {
    result: [lowest 1 of (-N+2)d6]
  } else {
    result: [ovai Nd6]
  }
}

loop N over {-4..12} {
  output [ova N] named "OVA dice (stat: [N])"
}

function: action WROLL:n HROLL:n DROLL:n {
  if WROLL = HROLL & HROLL = DROLL {VAL: 0}
  else if WROLL > HROLL & WROLL > DROLL {VAL: 1}
  else if HROLL > WROLL & HROLL > DROLL {VAL: 1}
  else if DROLL > HROLL & DROLL > WROLL {VAL: 3}
  else if WROLL = HROLL & WROLL > DROLL {VAL: 1}
  else if WROLL = DROLL & WROLL > HROLL {VAL: 2}
  else if HROLL = DROLL & HROLL > WROLL {VAL: 2}

  result: VAL
}

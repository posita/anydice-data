function: impl atk ATK:n max MAX:n dmg DMG:n ac AC:n {
  if ATK = MAX { result: DMG }
  if ATK >= AC { result: DMG }
  result: 0
}
function: atk ATK:d dmg DMG:d ac AC:n {
  result: [impl atk ATK max [maximum of ATK] dmg DMG ac AC]
}
\ single wield long sword \
output [atk 1d10+2 dmg [explode d10] ac 5]
\ dual wield daggers \
output 2d[atk 1d10+2 dmg [explode d4] ac 5]
\ dual wield dagger, shortsword \
output [atk 1d10+2 dmg [explode d4] ac 5] + [atk 1d10+2 dmg [explode d6] ac 5]

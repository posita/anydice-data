function: evaluate ROLL:s {			
SUCCESSES: ROLL <= 14			
if [count 1 in ROLL] { result: 2}			
			
if SUCCESSES > 0 { result: SUCCESSES}			
result: 0			
}			
			
loop DICE over {1..5} {			
output [evaluate DICE d20] named "[DICE]d20"			
}			
			
output 1d{2,1:18,0:1} named "1d20"			
output 2d{2,1:18,0:1} named "2d20"			
output 3d{2,1:18,0:1} named "3d20"			
output 4d{2,1:18,0:1} named "4d20"			
output 5d{2,1:18,0:1} named "5d20"

function: roll A:n{
   B : A
   if A = 10{B: A + [roll 1d10]} 
   result : [eval B]
}

function: eval A:n
{
  TMP: A + BONUS - DIFF
  if TMP < 0 { result: 0}
  if TMP < 4 { result: 1}
  if TMP < 8 { result: 2}
  if TMP < 12 { result: 3}
  if TMP < 16 { result: 4}
  result : 5 + [eval TMP-16]
}

BONUS: 10

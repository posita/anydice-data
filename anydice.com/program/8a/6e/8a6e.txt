function: rage A:n B:n vs AC:n{
	TOHIT: 0
	if A > B { TOHIT: A}
	if B >= A { TOHIT: B}
	if TOHIT = 1 {result: FUMBLE}
	TOHIT: TOHIT + ATTACK + ESCALATION

	if A > 10 { if B > 10 { if TOHIT >= AC { result: CRIT} } }
	if A >= CRITRANGE {result: CRIT}
	if B >= CRITRANGE {result: CRIT}
	if TOHIT >= AC {result: DAMAGE}
	result: MISS
}

DAMAGE: 1d10+4
CRITRANGE: 20
CRIT: 2*(DAMAGE)
FUMBLE: 0
MISS: 1
ATTACK: 5

loop ESCALATION over {0..6} {output [rage 1d20 1d20 vs 17]}

MISS: 0
HIT: 1
CRIT: 2
function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRIT}
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}
ATTACK: 8
function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRIT {
  result: CRITDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}
HITDAMAGE: 1d6+4
MISSDAMAGE: 0
CRITDAMAGE: 2d6+4
output 4d[damage for [attack d20 vs 15]]

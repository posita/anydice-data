\**** Damage calculator full basic damage dealing list ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + DEXTERITY + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 11-20\
ENEMYAC: d{11..20}

\[5] Rogue (AC 17) Dagger Sneak Attack"\
STRENGTH: 1
DEXTERITY: 4
PROFICIENCY: 3
ACCBONUS: 0
DMGBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d4 + 3d6 + STRENGTH + DMGBONUS
CRITDAMAGE: 1d4 + 4 + 3d6 + 18 + STRENGTH + DMGBONUS
output (ATTACKTIMES)d[attackroll d20 vs ENEMYAC] named "[5] Rogue (AC 17) Dagger Sneak Attack"

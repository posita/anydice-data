\
\\ 6 Piece Polyhedral Dice Pool 
\


\ Variables \
DICE: {4,6,8,10,10,12}

\ Function for rolling all the dice \
function: check NUM:n reroll RE:n {

  TOTAL: 0
  loop N over {1..NUM} {
    if (d(N@DICE)>RE) {
      TOTAL: TOTAL + (d(N@DICE)>3)
    }
    else {
      TOTAL: TOTAL + (1@2d(N@DICE) >3)
    }
  }

  result: TOTAL
}

\ function for rolling a single die \
function: roll ROLL:n reroll NUM:n {
  if (ROLL > RE) {
    result: ROLL > 3
  }
  else {
    result: d(N@DICE)>3)
  }
}

\ Loops to test a variety of combinations \
loop N over {1..#DICE} {
  output [check N reroll 1] named "[N]"
}

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: attacktwo ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGETWO } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGETWO }
 result: 0
}

\looped against enemy AC from 13-20\
ENEMYAC: d{13..20}

ATTACKMOD: 6
DMGBONUS: 4
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d6 + DMGBONUS
CRITDAMAGE: 2d8 + DMGBONUS
DAMAGETWO: 1d4 + DMGBONUS
CRITDAMAGETWO: 2d4 + DMGBONUS
output ATTACKTIMESd[attackroll d20 vs ENEMYAC] + ATTACKTIMESd[attacktwo d20 vs ENEMYAC]

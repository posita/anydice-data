\**** Talisman Damage calculator normal attack ****\

function: attackroll ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: ((ROLL + ATTACKBONUS - DEF) + WPNDAMAGEA)*2 + BITEA + BITEB + BITEC} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS >= DEF { result: (ROLL + ATTACKBONUS - DEF)+ WPNDAMAGEA + BITEA + BITEB + BITEC}
 result: 0
}

\looped against enemy DEF\
ENEMYDEF: d{8}

ATTACKBONUS: 2
BITEA: 1d6
BITEB: 0
BITEC: 0
CRITROLL: 12
WPNDAMAGEA: 4
output [attackroll 2d6 vs ENEMYDEF] named "Talismancer attack"
output 2d6

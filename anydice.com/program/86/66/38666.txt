EVASION: 15
AB: 3
NORMATTACKHMG: 1d20-1d6+AB>=EVASION
ACCATTACKHMG: 1d20+AB>=EVASION
ATTACKKNIFE: 1d20+AB>=EVASION
HMGDAMAGE: 2d6+4
KNIFEDAMAGE: 1d3+1

function: knifefirst A:n {
  if A = 1 { result: KNIFEDAMAGE + ACCATTACKHMG*HMGDAMAGE }
  result: NORMATTACKHMG*HMGDAMAGE
}

output [knifefirst ATTACKKNIFE] named "Attack with knife first"
output NORMATTACKHMG*(HMGDAMAGE+3)+ATTACKKNIFE*KNIFEDAMAGE named "Attack with HMG first"

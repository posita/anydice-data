set "explode depth" to 1

function: advantage ROLL:s {
 A: 1@ROLL
 B: 2@ROLL
 C: (1@ROLL=10)d10
 D: (2@ROLL=10)d10
 result: A+B+C+D
}
output [advantage 2d10] named "straight roll"
output [advantage 3d10] named "advantage 1"
output [advantage 4d10] named "advantage 2"
output [advantage 5d10] named "advantage 3"

function: disadvantage ROLL:s {
 A: (#ROLL)@ROLL
 B: (#ROLL-1)@ROLL
 C: ((#ROLL)@ROLL=10)d10
 D: ((#ROLL-1)@ROLL=10)d10
 result: A+B+C+D
}
output [disadvantage 3d10] named "disadvantage 1"
output [disadvantage 4d10] named "disadvantage 2"
output [disadvantage 5d10] named "disadvantage 3"

X: 15
output [advantage 2d10]>X named "straight roll above [X]"
output [advantage 3d10]>X named "advantage 1 above [X]"
output [advantage 4d10]>X named "advantage 2 above [X]"
output [advantage 5d10]>X named "advantage 3 above [X]"

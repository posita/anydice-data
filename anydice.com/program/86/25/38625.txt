function: base N:n T:n {
 if N >= T { result: 1 }
 result: 0
}

function: truncate A:n {
 N: 3
 if A > N { result: N }
 result: A
}

set "maximum function depth" to 4
loop N over {1..2} {
 loop BONUS over {0..3} {
  output [truncate [base [highest 1 of Nd20]+BONUS 11]] named "[N]d20+[BONUS] >= 11"
 }
}
output 1

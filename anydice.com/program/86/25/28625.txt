set "maximum function depth" to 3

function: blackred of A:n and B:n {
 if A = B { result: [blackred of 10 + 1d{0..9} and 10+1d{0..9}]}
 if A < B { result: A }
 result: B
}

SHIFTD: d10

function: shift A:n and B:n {
  if (A = B) { result: 0 }
  if A = B { result: [shift SHIFTD and SHIFTD]}
  if A < B { result: 1 }
  result: -1
}


output ([blackred of 1d10 and 1d10])*[shift SHIFTD and SHIFTD]
output [explode 1d10]-[explode 1d10]

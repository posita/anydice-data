\**** Damage calculator Ninja Dual Wield Rogue ****\

function: attackmain ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: ((ROLL + ATTACKBONUS - DEF) *DEADLYPBNS + BONUSDMG + WPNDAMAGEA)*2 + BURNCROWN} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS - 0 >= DEF { result: (ROLL + ATTACKBONUS - DEF) *DEADLYPBNS + (BONUSDMG) + WPNDAMAGEA + BURNCROWN}
 result: 0
}

function: dualwield ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: WPNDAMAGEB*2 + BURNCROWN} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS >= DEF { result: WPNDAMAGEB + BURNCROWN}
 result: 0
}

\looped against enemy DEF\
ENEMYDEF: d{11}

ATTACKBONUS: 6
BONUSDMG: 0
CRITROLL: 12
CINDERB: 0
WPNDAMAGEA: 12
WPNDAMAGEB: 12
BURNCROWN: 4
DEADLYPBNS: 2
COMBOTIMES: 1
output [attackmain 2d6 vs ENEMYDEF] + ([dualwield 2d6 vs ENEMYDEF] * COMBOTIMES) named "Ninja Rogue Dual Wield"

output 2d6

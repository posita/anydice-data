function: roll {
  result: 1d6+1d4
}
function: interlock N:n STAT:n SKILL:n MOD:n {
  if N = 2 { result: 0 }
  if N = 10 { result: STAT + SKILL + MOD + 10 + [explode [roll]] }
  result: STAT + SKILL + MOD + N
}

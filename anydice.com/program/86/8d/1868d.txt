function: highest DICE:d {
 result: {1}@DICE
}

function: compare C:n D:n E:n K:n {
 if C > D { result: 1 }
 else if D > C { result: 2}
 else if C = D {
  if E > K { if K > 1 {result: 2 } result: 3}
   result: 3 
  }
}

loop N over {1..8}{
 I: [highest Nd6]
 J: [highest 1d6]
 F: 1d6
 output [compare I J F N]
}

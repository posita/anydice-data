function: ROLL:n reroll BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}
function: ROLL:n increment BAD:s {
  if ROLL = BAD { result: ROLL + 1 }
  result: ROLL
}
X: [d6 reroll {1} as d6]
Y: [d6 increment {1}]
XY: [dX increment {1}]

output 8d6 named "Normal"
output 8dY named "Increment 1s"
output 8d6 + 1d4 named "Normal + 1d4"
output 8dX named "Reroll 1s"
output 9d6 named "Normal + 1d6"
output 8dXY named "Reroll 1s, then Increment 1s"
output 8d6 + 1d8 named "Normal + 1d8"

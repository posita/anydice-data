YA: 0    \Yellow d6(10) - only as many as there are players\
YB: 0    \Yellow d8(7)\ 
PA: 0    \Purple d8(7)\
PB: 0    \Purple d12(7)\
BA: 0    \Blue d6(10)\
BB: 0    \Blue d8(9)\
BC: 0    \Blue d12(9)\
GB: 0    \Glittery Blue d6(7)\
RA: 0    \Red d6(10)\
RB: 0    \Red d8(9)\
GA: 0   \Green d20(10)\
CA: 0    \Clear d6(7)\
PIA: 0   \Pink d12(4) - no more than 1 permitted per roll\


\-----------------------------\

\-----------------------------\


  Y: (YAd6 + YBd8) \Sum of Yellows\
  PU: ((PAd8 + PBd12) * 2) \Double Sum of Purples\
  if GB > 0 {
   B: ((BA+GB)d6 + BBd8 + BCd12)*2} \Double Sum of Blues\
  else {
   B: (BAd6 + BBd8 + BCd12)} \Sum of Blues\
  R: (RAd{-1,-2,-3,4,5,6} + RBd{-1,-2,-3,-4,5,6,7,8}) * (RA+RB) \Sum of Red * Num of Red Dice\
  G: GAd20 \Sum of Green\
  C: CAd6 \Sum of Clear\
  PI: PIAd12 \Value of Pink\
  T: (Y+PU+B+R+G+C+PI) \Total Score\

output T named "Total Score"
output Y named "Yellow Score"

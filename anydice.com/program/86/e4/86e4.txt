function: abs N:n
{
  if (N >= 0)
  {
    result: N
  }
  else
  {
    result: N * -1
  }
}

loop D over {-3..3}
{
  Q: ([abs D-1] + 1)

  ROLL: Qd10

  X: 1@ROLL

  L: #ROLL

  Y: L@ROLL

  \output Y named "[Q]d10 bottom value"

  output X named "[Q]d10 top value"

  output [count X in ROLL] named "[Q]d10 count top value"\

  if (D >= 0)
  {
    output X + [count X in ROLL] named "Skill [D]: [Q]d10 top value + count"
  }
  else
  {
    output Y named "Skill [D]: [Q]d10 bottom value"
  }
}

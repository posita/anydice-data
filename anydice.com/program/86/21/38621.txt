function: base N:n T:n B:n {
 if N >= 6 { result: 1 }
 if N + B >= T { result: 1 }
 result: 0
}

function: truncate A:n {
 N: 3
 if A > N { result: N }
 result: A
}

set "maximum function depth" to 4
TARGET: 4
loop N over {1..2} {
 loop BONUS over {0..1} {
  output [truncate [base [highest 1 of Nd6]+BONUS TARGET 0]] named "[N]d6+[BONUS]"
 }
}

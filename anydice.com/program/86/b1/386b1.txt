\ Panda Royale Dice Roller - Statistical Analysis                                            \
\ Rules for the game found at https://lastnightgames.com/games/panda-royale.html             \  

DIE: {6, 8, 12, 20}
DDIE: {6, 8, 12}

DRS: {-1,-2,-3,4,5,6}
DRE: {-1,-2,-3,-4,5,6,7,8}

LDIE: #DIE
LDDIE: #DDIE

N: 1

\ Evaluate Red Combinations \

loop N over {1..10}{

 loop I over {0..N}{
  RSIX: N-I
  REIGHT: I
  R: (RSIXdDRS + REIGHTdDRE) * N
  
  output R named "Red Score for [RSIX] Red d6 and [REIGHT] Red d8"
 }

 loop J over {1..LDIE}{
  IND: J@DIE
  output Nd(IND) named "[N]d[IND] - Standard" 
 }

 loop K over {1..LDDIE}{
  IND: K@DDIE
  output (NdIND)*2 named "[N]d[IND] - Doubled" 
 }

}

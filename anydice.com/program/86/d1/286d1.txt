function: HONE:n HTWO:n vs G:n {
  H: HONE + HTWO
  if G = 1 {result: -1}
  else if H = 2 {result: 0}
  else if H >= 11 {result: -1}
  else if H < G {result: -1}
  else if H > G {
    if G < 4 {result:1}
    else if G < 7 {result: 2}
    else if G < 10 {result: 3}
    else {result: 5}
  }
  else {result: [HONE 1d6 vs G] + [HTWO 1d6 vs G]}
}

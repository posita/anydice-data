set "maximum function depth" to 2

function: blackred of A:n and B:n {
 if A = B { result: [blackred of 10 + 1d10 and 10+1d10]}
 if A < B { result: A }
 result: B
}

function: shift A:n and B:n {
  if (A = B) { result: 0 }
  if A < B { result: 1 }
  result: -1
}

function: shifter A:n B:n{
  if A > 0 {result: 1}
  if A < 0 {result: -1}
  result: 0

}

function: wailandwoe A:n and B:n wailed by WAIL:n woed by WOE:n {
 if WAIL=WOE {result: 0}
 if (WAIL = 10 & (A=10 | B=10)) {result: 10}
 if (WOE = 10 & (A=10 | B=10)) {result: -10}
 if A = B { result: ([blackred of 10+1d10 and 10+1d10])*[shift WAIL and WOE]}
 if A < B { result: A*[shift WAIL and WOE] }
 result: B*[shift WAIL and WOE]
}

output [wailandwoe 1d10 and 1d10 wailed by 1d10 woed by 1d10]
output [explode 1d10]-[explode 1d10]

function: eldblast H:n with W:d and L:n{
 if H < 9 {
  result: 0
 }
 else if H < 20 {
  result: W + L
 }
 else{
  result: W+W+L
 }
}
function: round R:n{
 D: 0
 T: [eldblast 2d20 with 1d10 and 7]

 D: D+T
 T: [eldblast 2d20 with 1d10 and 7]
 D: D+T

 T: [eldblast 2d20 with 4d12 and R]

}
function: rsblast D:n with L:n{

  result: D+L
 }
 else{
  result: D
 }
}

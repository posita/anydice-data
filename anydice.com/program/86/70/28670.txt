\Maor’s Weapon Calculations\

\variable definitions\
AC:  15  \Target AC\
STR:  0  \Strength Modifier\
DEX:  4  \Dexterity Modifier\
PROF: 3  \Proficiency Modifier\
ADV:  1  \0=Disadvantage 1=straight roll 2=advantage\
CR:  20  \Range of Critical Hit\

\setting advantage or disadvantage\
if ADV >= 2 {
 ROLL: [highest 1 of 2d20]
 }
else {
 if ADV <= 0 {
  ROLL: [lowest 1 of 2d20]
  }
else {
 ROLL: 1d20
}}

\Great Weapon Fighting damage calculations\
function: X:n gwfs {
 if X <= 2 {
  result: 1d6
  }
  result: X
}
function: Y:n gwfa {
 if Y <= 2 {
  result: 1d12
  }
  result: Y
}

\Main Attack Function\
function: attack ROLL:n plus BONUS:n vs AC:n miss MISS:d hit HIT:d crit CRIT:d on CRIT_RANGE:n {
    if ROLL >= CRIT_RANGE {
        result: CRIT
    }
    if ROLL + BONUS > AC {
        result: HIT
    }
    result: MISS
}

\Outputs\
output [
attack ROLL plus DEX + PROF + 2 vs AC
    miss 0
    hit  1d8 + DEX
    crit 2d8 + DEX on CR
] named "Longbow"

output [
attack ROLL plus DEX + PROF + 2 vs AC
    miss 0
    hit  1d8 + 1d4 + DEX 
    crit 2d8 + 2d4 + DEX on CR
] named "Longbow Favored Foe"

output [
attack ROLL plus DEX + PROF + 2 - 5 vs AC
    miss 0
    hit  1d8 + DEX +10
    crit 2d8 + DEX + 10 on CR
] named "Longbow Sharpshooter"

output [
attack ROLL plus DEX + PROF vs AC
    miss 0
    hit  1d8 + DEX
    crit 2d8 + DEX on CR
] named "Rapier"

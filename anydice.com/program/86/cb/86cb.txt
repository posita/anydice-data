\NOT WORKING 2d10 + Skill, take highest, bonus +1 for each matching-highest die; take lowest for -ve skills.\

function: abs N:n
{
  if (N >= 0)
  {
    result: N
  }
  else
  {
    result: N * -1
  }
}

function: test D:n ROLL:s
{
  X: 1@ROLL

  if (D > 0)
  {
    result: X + ([count X in ROLL]-1)
  }
  else
  {
    L: #ROLL
    Y: L@ROLL

    result: Y
  }
}


loop D over {-2..5}
{
  DICE: ([abs D-1] + 1)

  ROLL: DICEd10

  output [test D ROLL] named "Skill [D]"
}

TARGET_INT:1
TARGET_DEX:1
SPELL_SAVE:15

SLIVERDMG: 2d6
METEORDMG: 2d6

function: sliver meteor FIRSTROLL:n SECONDROLL:n THIRDROLL:n SLIVER:n {
  DMG: 0
  if FIRSTROLL + TARGET_INT < SPELL_SAVE {
    DMG: DMG + SLIVERDMG
  }
  if SECONDROLL + TARGET_DEX < SPELL_SAVE + (FIRSTROLL + TARGET_INT < SPELL_SAVE)*SLIVER {
    DMG: DMG + METEORDMG
  }
  else {DMG: DMG + METEORDMG/2}
  if THIRDROLL + TARGET_DEX < SPELL_SAVE {DMG: DMG + METEORDMG}
  else {DMG: DMG + METEORDMG/2}
  result: DMG
}

output [sliver meteor 1d20 1d20 1d20 1d4]

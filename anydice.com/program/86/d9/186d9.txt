function: highestof A:n B:n {
  result: 1 @ [sort {A, B}]
}

function: lowestof A:n B:n {
  result: 2 @ [sort {A, B}]
}

function: rerolllow A:n B:n {
  if A <= 0 {result: A}
  result: B
}

function: rerollhigh A:n B:n {
  if A > 0 {result: A}
  result: B
}

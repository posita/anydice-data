function: compare R:n to X:n and Y:n {
 H: 0
 if R < X {
  H: 1
 }
 if R < Y {
  H: H+1
 }
 result : H
}
loop Z over {6,8,10,12}{
loop N over {4,6,8,10,12,20}{
output 1dN > 1dZ named "1d[N] vs 1d[Z]"
loop M over {4,6,8,10,12,20}{
if N <=M{
output [compare 1dZ to 1dN and 1dM] named "1d[N] and 1d[M] vs 1d[Z]"
}
}
}

}

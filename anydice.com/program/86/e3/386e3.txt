loop N over {1..2} {
  output Nd6 named "[N]d6"
  D: N*2
  output [highest N of Dd6] named "highest [N] of [D]d6"
}

\Powerful Sneak\
loop N over {3..6} {
  \output Nd{2,2,3,4,5,6} named "[N]d6 min 2"\
  D: N*2
  H: N/2
  output [highest N of Dd6] + H named "highest [N] of [D]d6 + [H]"
}

\Deadly Sneak\
loop N over {7..10} {
  output Nd{3,3,3,4,5,6} named "[N]d6 min 3"
  D: N*2
  output [highest N of Dd6] + N named "highest [N] of [D]d6 + [N]"
}

function: alta A:n B:n C:n{
if A <= B & B <= C{
    result: B + C -2
}
if B <= A & B <= C{
    result: A + C -2
}
if C <= A & C <= B{
    result: A + B -2
}

}
\ :::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::: \
function: neutra A:n B:n C:n{
if B >= A & A >= C{
    result: B + C - 2
}
if A >= B & B >= C{
    result: A + C - 2
}
if A >= C & C >= B{
    result: B + A - 2
}
if C >= A & A >= B{
    result: B + C - 2
}
if C >= B & B >= A{
    result: A + C - 2
}
if B >= C & C >= A{
    result: B + A - 2
}

}
\ :::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::: \
function: baja A:n B:n C:n{
if A >= B & B >= C{
    result: B + C -2
}
if B >= A & B >= C{
    result: A + C -2
}
if C >= A & C >= B{
    result: A + B -2
}

}
\ :::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::: \
function: pobre A:n B:n C:n{
if A >= B{
   if C >= A{
      result: A - 1
   }else{
      if B >= C{
         result: B - 1
      }else{
         result: C - 1
      }
   }
}else{
   if A >= C{
      result: A - 1
   }else{
      if C >= B{
         result: B - 1
      }else{
         result: C - 1
      }
   }
}

}
\ :::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::: \
function: nula A:n B:n C:n{
if A <= B & B <= C{
    result: A -1
}
if B <= A & B <= C{
    result: B -1
}
if C <= A & C <= B{
    result: C -1
}
}

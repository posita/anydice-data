\**** Damage calculator Cinderblast attack ****\

function: attackroll ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: ((ROLL + ATTACKBONUS - DEF + BONUSDMG + WPNDAMAGEA)*2) + CINDERBLASTDMG} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS >= DEF { result: ROLL + ATTACKBONUS - DEF + BONUSDMG + WPNDAMAGEA + CINDERBLASTDMG}
 result: 0
}

\looped against enemy DEF\
ENEMYDEF: d{11}

ATTACKBONUS: 5
BONUSDMG: 0
CINDERBLASTDMG: 7
CRITROLL: 12
WPNDAMAGEA: 14
output [attackroll 2d6 vs ENEMYDEF] named "Cinderblast attack"
output 2d6

function: hightwo of A:n B:n {
  result: 1@[sort {A,B}] 
}

function: highthree of A:n B:n C:n {
  result: 1@[sort {A,B,C}] 
}

function: highfive of A:n B:n C:n D:n E:n {
  result: 1@[sort {A,B,C,D,E}] 
}

output [hightwo of 1d10 1d10]
output [highthree of 1d10 1d10 1d10]
output [highfive of 1d10 1d10 1d10 1d10 1d10]

function: pro M:n D:n L:n
{
  X:0
  loop I over {1..M} {X:X+(1dD >= L)}
  result:X
}

function: attack A_NUM_OF_DICE:n A_DICE:n A_DC:n B_NUM_OF_DICE:n B_DICE:n B_DC:n
{
  X: [pro A_NUM_OF_DICE A_DICE A_DC] - [pro B_NUM_OF_DICE B_DICE B_DC]
  result: X
}

function: attack X:n
{
  if X > 0 {result: X}
  else {result: 0}
}
function: defend X:n
{
  if X <= 0 {result: -X}
  else {result: 0}
}

X:[attack 7 10 6 9 8 6]
output X
output [attack X]
output [defend X]

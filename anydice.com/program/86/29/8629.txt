function: bcount X:s {
    COUNT: 0
    loop I over {2..#X} {
        COUNT: COUNT + [count {2..6,6} in I@X]
    }
    result: COUNT
}

function: bushido ROLL:s {
    if #ROLL = 1 { result: d{0, 2..6}}
    if ([count 1 in ROLL] < #ROLL){
        SROLL: [sort ROLL]
        HIGHEST: 1@SROLL
        POOL: [bcount SROLL]
        result: HIGHEST + POOL
    } else {
        result:0
    }
}

loop NUMBER over {1..10} {
    output [bushido NUMBERd6] named "[NUMBER]d6"
}

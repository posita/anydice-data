function: fifth edition attack ROLL:n {
if ROLL = 1  {result: 0}
if ROLL >= CRIT {result: 2*DMG_DICE + DMG_MOD}
if ROLL+TO_HIT >= AC {result: DMG_DICE + DMG_MOD}
result: 0}

DMG_DICE: 1d8
DMG_MOD: 2
TO_HIT: 5
CRIT: 20

function: reroll R:n under N:n {
   if R < N { result: 1d20 } else {result: R}
}

loop AC over {8..23} {
 output [fifth edition attack d20] named "[AC] AC"
}

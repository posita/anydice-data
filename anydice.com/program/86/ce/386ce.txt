\**** Damage calculator Machinarm attack ****\

function: attackroll ROLL:n vs DEF:n {
 if ROLL >= CRITROLL { result: (ROLL + ATTACKBONUS - DEF + BONUSDMG + WPNDAMAGEA)*2 + CINDERB} 
 if ROLL <= 2 { result: 0 }
 if ROLL + ATTACKBONUS >= DEF { result: ROLL + ATTACKBONUS - DEF + BONUSDMG + WPNDAMAGEA + CINDERB}
 result: 0
}

\looped against enemy DEF\
ENEMYDEF: d{9}

ATTACKBONUS: 3
BONUSDMG: 0
CINDERB: 7
CRITROLL: 12
WPNDAMAGEA: 12
output [attackroll 2d6 vs ENEMYDEF] named "normal attack"
output 2d6

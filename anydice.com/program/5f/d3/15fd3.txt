

function: special X:n{
	if X!=6{
		result: X
	}
	else{
		result: 0
	}
}

function: delta A:n and C:n with X:n{
	D: A-C+3
	if D>X&X!=0{
		result: -1
	}
	else if D<X&X!=0{
		result: 1
	}
	else if D=X&X!=0{
		result: 0
	}
	else if X=0{
		result: 0
	}
}

loop A over {1..9}{
	loop C over {1..9}{
		output [delta A and C with [special d6]] named "[A] challenges [C]"
	}
}

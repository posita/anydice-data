function: A:s vs B:s {
 loop P over {1..[lowest of #A and #B]} {
  if P@A > P@B { result: [A beats B at P] }
  if P@B > P@A { result: -[B beats A at P] }
 }
 if #A != #B { result: #A - #B }
 result: 0
}

function: A:s beats B:s at P:n {
 S: 1
 loop N over {(P + 1)..#A} {
  if N@A < P@B { result: S }
  S: S + 1
 }
 result: S
}

output [3d6 vs 3d6]

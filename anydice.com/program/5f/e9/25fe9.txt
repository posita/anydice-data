MISS: 0
HIT: 1
CRITICAL_HIT: 2

function: attack ROLL:n mod BONUS:n crit CRIT:n vs DEFENSE:n {
  if ROLL = 1 { result: MISS }
  if ROLL >= CRIT {
    if ROLL + BONUS >= DEFENSE { result: CRITICAL_HIT }
    result: HIT
  }
  if ROLL + BONUS >= DEFENSE { result: HIT }
  result: MISS
}

function: dmgseq ROLL:d mod BONUS:n on attack ATTACK:n {
  if ATTACK = MISS { result: {0} }
  X:ROLL + BONUS
  if ATTACK = CRITICAL_HIT {
    X:X + ROLL
  }
  result: X
}

function: N:n is hit {
  result: N = HIT | N = CRITICAL_HIT
}

function: seq N:n times D:d {
  X:D
  loop ATTEMPT over {2..N} {
    X: X + D
  }
  result: X
}

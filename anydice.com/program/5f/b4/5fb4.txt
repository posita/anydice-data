set "maximum function depth" to 100
function: factorial A:n {
  ACC: 1
  loop N over {1..A}{
    ACC: ACC*N
  }
  result: ACC
}

function: N:n choose K:n {
  result: [factorial N]/[factorial K]/[factorial N-K]
}

function: current S:n n N:n p P:n k K:n x X:n {
  R: [N-S choose P]
  if R + K >= X { result: {S,K}}
  else { result: [current S+1 n N p P k R+K x X]}
}

function: starting S:n n N:n p P:n k K:n x X:n {
  result: [current S+1 n N p P k K x X]
}

function: N:n buckles P:n index X:n {
  ACC: {}
  S: 0
  K: 0
  loop I over {0..P-2}{
  R: [starting S n N p P-I-1 k K x X]
  S: 1@R
  K: 2@R
  ACC:{ACC,S}
 }
 result: {ACC,S+X-K} 
}

function: mydie ROLL:n {
  VAL: {1, 1, 1, 1, 1, 3, 3, 3, 5, 5, 7}
  SEL: [11 buckles 4 index ROLL] 
  ACC: 0
  loop N over SEL{
    ACC: ACC+N@VAL
  }  
 result: ACC
}

output [mydie d[11 choose 4]] named "mydie"

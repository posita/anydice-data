set "maximum function depth" to 100
function: factorial A:n {
  ACC: 1
  loop N over {1..A}{
    ACC: ACC*N
  }
  result: ACC
}

function: N:n choose K:n {
  result: [factorial N]/[factorial K]/[factorial N-K]
}

function: inner CURRENT:n N:n K:n LIMIT:n INDEX:n {
  NEWLIMIT: [N-CURRENT choose K] + LIMIT
  if NEWLIMIT >= INDEX { result: {CURRENT,LIMIT}}
  else { result: [inner CURRENT+1 N K NEWLIMIT INDEX]}
}

function: N:n buckles K:n index INDEX:n {
  ACC: {}
  CURRENT: 0
  LIMIT: 0
  loop I over {0..K-2}{
    R: [inner CURRENT+1 N K-I-1 LIMIT INDEX]
    CURRENT: 1@R
    LIMIT: 2@R
    ACC:{ACC,CURRENT}
  }
  result: {ACC,CURRENT+INDEX-LIMIT} 
}

function: mydie ROLL:n {
  VAL: {1, 1, 1, 1, 1, 3, 3, 3, 5, 5, 7}
  SEL: [11 buckles 4 index ROLL] 
  ACC: 0
  loop N over SEL{
    ACC: ACC+N@VAL
  }  
  result: ACC
}

output [mydie d[11 choose 4]] named "mydie"

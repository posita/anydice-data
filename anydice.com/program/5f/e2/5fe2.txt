N:5
ROLL: Nd6

function: sum all I in ROLL:s{
 result: I*( ROLL=I )
}

function: max of A:n and B:n{
 if A > B {
  result: A
 }
 else {
  result: B
 }
}

function: myroll ROLL:s{

 A: [sum all 1 in ROLL]

 loop I over {2..6}{
  B: [sum all I in ROLL]
  A: [max of A and B] 
 }

 result: A
}

output [myroll ROLL]

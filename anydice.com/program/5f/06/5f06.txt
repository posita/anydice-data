function: ROLL:d reroll below A:n {
 N:#ROLL
 SIDE:[maximum of ROLL]/N
 result: [reroll NdSIDE]
}

function: reroll DIE:n {
 if DIE < A {
  result: NdSIDE}
 result: DIE
}

loop N over {4,6,8,10,12}{
HN: N/2
 output 1dN named "d[N]"
 output [highest of 1dN and N/2] named "d[N] OR [HN]"
 loop M over {2..N}{
  output [1dN reroll below M] named "d[N] reroll < [M]"
 }
}

function: deefive S:n{
	if S<=2{
		result: -2
	}
	else if S<=4{
		result: -1
	}
	else if S<=6{
		result: 0
	}
	else if S<=8{
		result: 1
	}
	else{
		result: 2
	}
}

function: delta A:n and C:n with X:n{
	D: A-C
	if D>X{
		result: -1
	}
	else if D<X{
		result: 1
	}
	else{
		result: 0
	}
}

output [deefive d10] named "check"
loop A over {1..9}{
	loop C over {1..9}{
		output [delta A and C with [deefive d10]] named "[A] challenges [C]"
	}
}

function: hit D:s over N:n with mod M:n and crit over C:n {
  if D>C {result: 2} else {if (D+M)>N {result: 1} else {result: 0}}
}

function: damage with dice DD:s and mod MM:n hitting with dice D:s over N:n with mod M:n and crit over C:n {
  X: [hit D over N with mod M and crit over C]
  if X = 0 {
    result: 0
  }
  else {
    if X = 1 { 
      result: DD+MM
    }
    else {
      result: (2*DD) + MM
    }
  }

}

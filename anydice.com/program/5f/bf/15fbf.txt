function: soak DIE:n using LIMIT:n {
  if DIE <= LIMIT { result: 0 }
  else { result: DIE-LIMIT }
}

function: fight ROLL:d num COUNT:n armor ARMOR:n {
  result: [soak COUNTdROLL using ARMOR]
}


GUARD: 1d8+1
GOBLIN: 1d6+2
BLACKBEAR: 1d6+2+2d4+2
BROWNBEAR: 1d8+4+2d6+4


MAX: 4
ARMOR: 12
OPPONENT: GOBLIN

loop P over {1..MAX}{
  output [fight OPPONENT num P armor ARMOR] named "[P] opponents"
}

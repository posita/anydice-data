H_SKILL: 1
H_PROF: 2
HERBALISM: d20 + H_SKILL + H_PROF

loop RARITY_MOD over {-10,0,10,20} {
 HERBALISM_CHECK: d100 + HERBALISM - RARITY_MOD
 if RARITY_MOD = -10 {output HERBALISM_CHECK named "Common"}
 else if RARITY_MOD = 0 {output HERBALISM_CHECK named "Uncommon"}
 else if RARITY_MOD = 10 {output HERBALISM_CHECK named "Rare"}
 else if RARITY_MOD = 20 {output HERBALISM_CHECK named "Very Rare"}
}

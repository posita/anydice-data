\ A result of 1 means the one with a higher skill won, 2 means they lost and 0 is a draw \

MORESKILLED: 95
LESSSKILLED: 90

function: roll TENSA:n and ONESA:n under SKILLA:n vs TENSB:n and ONESB:n under SKILLB:n
{
 if ONESA+TENSA = 0
    { 
       A: 4
    }
 else if TENSA*10+ONESA > SKILLA
 {
    if ONESA = 0 | ONESA = 5
    { 
       A: 4
    }
    else 
    { 
       A: 3
    }
 }
 else if TENSA*10+ONESA <= SKILLA
 {
     if ONESA = 0 | ONESA = 5
     {
         A: 1
     }
     else 
     {
         A: 2
     }
 }
 if ONESB+TENSB = 0
 { 
       B: 4
 }
 else if TENSB*10+ONESB > SKILLB
 {
    if ONESB = 0 | ONESB = 5
    { 
       B: 4
    }
    else 
    { 
       B: 3
    }
 }
 else if TENSB*10+ONESB <= SKILLB
 {
     if ONESB = 0 | ONESB = 5
     {
         B: 1
     }
     else 
     {
         B: 2
     }
 }
 if A < B
 {
         result: 1
 }
 else if A > B
 {
         result: 2
 }
 else
 {
         if TENSA*10+ONESA = TENSB*10+ONESB
         {
             result: 0
         }
         if TENSA*10+ONESA < TENSB*10+ONESB
         {
             result: 1
         }
         else
         {
             result: 2
         }
 }

}

function: roll DIEA:n under SKILLA:n vs DIEB:n under SKILLB:n 
{
   if DIEA <= SKILLA & DIEB > SKILLB
   {
       result: 1
   }
   if DIEA > SKILLA & DIEB <= SKILLB
   {
       result: 2
   }
   if DIEA <= SKILLA & DIEB <= SKILLB
   { 
       if DIEA >= DIEB
       {
           result: 1
       }
       else
       {
           result: 2
       }
    }
    if DIEA > SKILLA & DIEB > SKILLB
    { 
           result: 1
    }
}

function: alt DIEA:n under SKILLA:n vs DIEB:n under SKILLB:n 
{
   if DIEA > SKILLA & DIEB > SKILLB
   { 
           result: 0
   }
   else if DIEA <= SKILLA & DIEB > SKILLB
   {
       result: 1
   }
   else if DIEA > SKILLA & DIEB <= SKILLB
   {
       result: 2
   }
   else
   { 
       if DIEA > DIEB
       {
           result: 1
       }
       if DIEA < DIEB 
       {
           result: 2
       }
       else
       {
           result: 0
       }
    }
}

       {
           result: 2
       }
       else
       {
           result: [reroll DIEA under SKILLA vs DIEB under SKILLB]
       }
    }
}

ENEMY_AC: 10
DEX: 2

function: roll to hit HIT_ROLL:n with attack ATTACK:n {
    if HIT_ROLL>ENEMY_AC {
        result: ATTACK
    } else {
        result: 0
    }
}

function: dagger {
    result: [roll to hit 1d20+DEX with attack 1d4]
}

function: rapier {
    result: [roll to hit 1d20+DEX with attack 1d8]
}

function: dual dagger {
    result: [dagger] + [dagger]
}


output [dagger] named "Single dagger"
output [dual dagger] named "Dual wielded dagger"
output [rapier] named "Rapier"

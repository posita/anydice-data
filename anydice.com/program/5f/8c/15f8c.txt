TO_HIT: 1d20+6
ENEMY_AC: 18

ONE_H_DMG: 1d6+1d6+2d12+1
TWO_H_DMG: 1d8+1d6+2d12+1

ONE_H_CRIT: 1d6
TWO_H_CRIT: 1d8

function: TOHIT:n vs AC:n for DMG:n plus CRIT:n {
 RES: 0
 if TOHIT >= AC {
  result: RES + DMG
 }
 if TOHIT = 20 {
  result: RES + CRIT
 }
 result: RES
}

function: TOHIT:n vs AC:n for DMG:n plus CRIT:n twice {
 result: [TOHIT vs AC for DMG plus CRIT] + [TOHIT vs AC for DMG plus CRIT]
}

output [TO_HIT vs ENEMY_AC for ONE_H_DMG plus ONE_H_CRIT twice] named "One Handed"
output [TO_HIT vs ENEMY_AC for TWO_H_DMG plus TWO_H_CRIT twice] named "Two Handed"

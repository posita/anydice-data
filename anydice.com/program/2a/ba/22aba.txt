\ FUMBLE: 1 take a Complication for an extra Success \
\ FUMBLE: 0 take only regular successes \
FUMBLE:0
\ Base dice are d6 \
D:d{FUMBLE,0:3,1,2}
\ Half dice are d4 \
H:d{FUMBLE,0:2,1}

loop N over {1..25}
{
  DICE: N / 2
  HALF: N - DICE * 2
  FRACTION:HALF * 5
  output DICEdD + HALFdH named "[DICE].[FRACTION]d Fumble:[FUMBLE]"
}

function: is even VALUE {
  HALF: VALUE / 2
  result: (HALF * 2) = VALUE
}

function: median DICES {
  NUMDICES: #(DICES)
  if [is even NUMDICES] = 1 {
    result: (NUMDICES / 2)@DICES
  } else {
    result: (NUMDICES / 2 + 1)@DICES
  }
}

function: test LEVEL {
  result: [median ((LEVEL-1)*2+1)d6] + 1*LEVEL - 1
}

\ EDGE: 1 take a Complication for an extra Success on max roll \
\ EDGE: 0 take only regular successes \
EDGE:0
\ Base dice are d6 \
D:d{1,2,0:3,EDGE}
\ Half dice are d12 \
H:d{1,2,0:9,EDGE}

loop N over {1..25}
{
  DICE: N / 2
  HALF: N - DICE*2
  FRACTION:HALF * 5
  output DICEdD + HALFdH named "[DICE].[FRACTION]d Edge:[EDGE]"
}

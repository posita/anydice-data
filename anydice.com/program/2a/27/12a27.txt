\ Simple Melee Weapons (Medium) \

function: ROLL:n vs AC:n {
 if ROLL >= THREAT & ROLL + ATTACK >= AC {
  result: [threat d20]
 }
 if ROLL + ATTACK >= AC | ROLL = 20 { result: DAMAGE }
 result: 0
}

function: threat ROLL:n {
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: CRITICAL d DAMAGE
 }
 result: DAMAGE
}

AC: d{14..30}
BAB: 1
STRENGTH: 3
ATTACK: BAB + STRENGTH

DAMAGE: 1d6 + STRENGTH
THREAT: 20
CRITICAL: 2
output [d20 vs AC] named "Battle Aspergillum"

DAMAGE: 1d4 + STRENGTH
THREAT: 19
CRITICAL: 2
output [d20 vs AC] named "Brass Knife"

DAMAGE: 1d3 + STRENGTH
THREAT: 20
CRITICAL: 2
output [d20 vs AC] named "Brass Knuckles"

DAMAGE: 1d4 + STRENGTH
THREAT: 19
CRITICAL: 2
output [d20 vs AC] named "Cestus"

function: ROLL:n add BONUS:d to TARGET:s {
  if ROLL = TARGET { result: ROLL + BONUS }
  result: ROLL
}

function: explode DIE:d as BONUS:d {
  result: [DIE add BONUS to [maximum of DIE]]
}

function: curve H:n M:n AC:n{

if M = 1 {
A: 5
D: 10
}
else {
A: 10
D: 0
}

if H = 20{
result: 2d8 + 2d6 + 4d6 + 7 + D
}

if H + A >= AC & H>1{
result: 1d8 + 1d6 + 2d6 + 7 + D
}
result: 0
}

}

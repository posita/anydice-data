TARGET_AC: {21}

STR: 2 
PROF: 3
CRIT: 2

GREATSWORD_ATTACK: {STR + PROF}

GREATSWORD_ROLL: 1d{{1..2}:2, {4..6}:((6 * 6) - 4) / (6 - 2)}
GREATSWORD_ROLL_CRIT: {CRIT}d{{1..2}:2, {4..6}:((6 * 6) - 4) / (6 - 2)}

GREAT_WEAPON_MASTER_A: -5
GREAT_WEAPON_MASTER_D: 10

GREATSWORD_DAMAGE: GREATSWORD_ROLL + STR + PROF
GREATSWORD_DAMAGE_CRIT: GREATSWORD_ROLL_CRIT + STR + PROF

MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{TARGET_AC}

ATTACK: GREATSWORD_ATTACK
HITDAMAGE: GREATSWORD_DAMAGE
CRITICALDAMAGE: GREATSWORD_DAMAGE_CRIT
MISSDAMAGE: 0
output [attack] named "GS"

ATTACK: GREATSWORD_ATTACK + 2
HITDAMAGE: GREATSWORD_DAMAGE
CRITICALDAMAGE: GREATSWORD_DAMAGE_CRIT
MISSDAMAGE: 0
output [attack] named "GS +2"

ATTACK: GREATSWORD_ATTACK + 3
HITDAMAGE: GREATSWORD_DAMAGE
CRITICALDAMAGE: GREATSWORD_DAMAGE_CRIT
MISSDAMAGE: 0
output [attack] named "GS +3"

ATTACK: GREATSWORD_ATTACK + GREAT_WEAPON_MASTER_A
HITDAMAGE: GREATSWORD_DAMAGE + GREAT_WEAPON_MASTER_D
CRITICALDAMAGE: GREATSWORD_DAMAGE_CRIT + GREAT_WEAPON_MASTER_D
output [attack] named "GS GWM"

ATTACK: GREATSWORD_ATTACK + GREAT_WEAPON_MASTER_A + 2
HITDAMAGE: GREATSWORD_DAMAGE + GREAT_WEAPON_MASTER_D
CRITICALDAMAGE: GREATSWORD_DAMAGE_CRIT + GREAT_WEAPON_MASTER_D
output [attack] named "GS GWM +2"

ATTACK: GREATSWORD_ATTACK + GREAT_WEAPON_MASTER_A + 3
HITDAMAGE: GREATSWORD_DAMAGE + GREAT_WEAPON_MASTER_D
CRITICALDAMAGE: GREATSWORD_DAMAGE_CRIT + GREAT_WEAPON_MASTER_D
output [attack] named "GS GWM +3"

function: attack {
 result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]
}

ATTACK: GREATSWORD_ATTACK
HITDAMAGE: GREATSWORD_DAMAGE
CRITICALDAMAGE: GREATSWORD_DAMAGE_CRIT
output [attack] named "Adv GS"

ATTACK: GREATSWORD_ATTACK + GREAT_WEAPON_MASTER_A
HITDAMAGE: GREATSWORD_DAMAGE + GREAT_WEAPON_MASTER_D
CRITICALDAMAGE: GREATSWORD_DAMAGE_CRIT + GREAT_WEAPON_MASTER_D
output [attack] named "Adv GS GWM"

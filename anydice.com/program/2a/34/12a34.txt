function: ra ROLL:n {
if ROLL = 1 { result: 1d6}
if ROLL = 2 { result: 1d6}
result: ROLL
}

function: rb ROLL:n{
if ROLL = 6 { result: 1 + ([ra d6] > 2)}
result: ROLL > 2}

function: w ROLL:n {
if ROLL = 1 {result: 1d6 > 5}
result: ROLL > 5
}

X: [w [rb [ra d6]]d6]

loop N over {1..(16*6-1)}{
 X: X + ([w [rb [ra d6]]d6])}

output X named "Avg Mortal Wounds"

function: A:n with tension T:n {
  if A = 0 { result: A }
  if A < 0 { result: A - T }
  if A > 0 { result: A + T }
}

function: A:n succeeds on THR:n and crits on CRIT:n {
   result: (A >= CRIT) + (A >= THR) - (A <= -CRIT) - (A <= -THR)
}

function: W:d minus B:d tension T:n success THR:n crit CRIT:n {
   result: [[(W - B) with tension T] succeeds on THR and crits on CRIT]
}

loop T over {0..8} {
output [d8 minus d10 tension T success 2 crit 8] named "d8-d8 + [T]; 2:4"
}

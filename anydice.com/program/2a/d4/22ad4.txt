\ FUMBLE: 1 take a Complication for an extra Success \
\ FUMBLE: 0 take only regular successes \
FUMBLE:0
\ Base dice are d6 \
D:d{FUMBLE,0:3,1,2}
\ Quarter dice are d8 \
Q:d{0:7,1}

loop N over {1..51}
{
  DICE: N / 4
  QUARTER: N - DICE * 4
  FRAC:QUARTER * 25
  if (FRAC > 0) {output DICEdD + QUARTERdQ named "[DICE].[FRAC]D Fumble:[FUMBLE]"}
  else {output DICEdD named "[DICE].00D Fumble:[FUMBLE]"}
}

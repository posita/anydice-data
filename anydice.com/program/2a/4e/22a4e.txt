function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output 3d4 named "3d4"
output 3d[reroll {1,2} on d4] named "3d4 GWF"
output 2d6 named "2d6"
output 2d[reroll {1,2} on d6] named "2d6 GWF"
output 1d12 named "1d12"
output 6d2 named "6d2"

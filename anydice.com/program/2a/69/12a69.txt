\Normal attack vs Adv\
AC: 15 \Enter Target AC\
H: 11 \Your To Hit\

if AC > H { X: AC } else { X: H+1 }
NORM: (21 + H - X)*100/20  \named "% to Hit vs AC[X]"\
ADV: (100- ((X-H-1)^2)/2^2) \named "% to Hit vs AC[X] w Adv"\
SADV: (100- ((X-H-1)^3)/2^3) \named "% to Hit vs AC[X] w Adv"\

output NORM
output ADV
output SADV

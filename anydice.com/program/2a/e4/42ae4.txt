function: hr of A:n and B:n and C:n {
  if A >= B & A >= C {result: A}
  if B >= A & B >= C {result: B}
  result: C
}

function: compare A:n or B:n or C:n against ROLL:s {
 R: [hr of A and B and C]
 if (R > 1@ROLL) { result: 1 }
 if (R = 1@ROLL) { result: 0 }
 result: -1
}

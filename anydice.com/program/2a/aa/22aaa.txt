\ EDGE: 1 take a Complication for an extra Success on max roll \
\ EDGE: 0 take only regular successes \
EDGE:0
\ Base dice are d6 \
D:d{1,2,0:3,EDGE}
\ Third dice are smaller d6s \
T:d{1,0:5}

loop N over {1..38}
{
  DICE: N / 3
  THIRD: N - DICE*3
  output DICEdD + THIRDdT named "[DICE]d+[THIRD] Edge:[EDGE]"
}

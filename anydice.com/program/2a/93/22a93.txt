\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}


\looped against enemy AC from 10-25\
ENEMYAC: d{10..25}

ATTACKMOD: 5
PROFICIENCY: 3
ATTACK: 1d10
DMGBONUS: ATTACKMOD
ACCBONUS: 1
CRITROLL: 20
DAMAGE: ATTACK + DMGBONUS
CRITDAMAGE: 2*ATTACK + 2*DMGBONUS

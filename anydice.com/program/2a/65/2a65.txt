function: second of A:s and B:s{
 ONE: 1@A
 TWO: 2@A
 THREE: 1@B
 if THREE < TWO{
  result: TWO
 } if THREE < ONE{
  result: THREE
 } else {
  result: ONE
 }
}

N: 6
loop I over {4..(4+N)}{
 H: I-(I/2*2)
 if H = 0{
  output 2@(I/2)d20 named "skill level [I]"
 } else {
  output [second of (I/2)d20 and d12] named "skill level [I]"
 }
}

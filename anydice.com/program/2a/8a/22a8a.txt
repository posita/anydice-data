function: group check S:n out of N:n {
  result: S >= N/2
}

loop T over {2..20} {
  X: d20 >= T
  output X named "single roll, target [T]"

  loop N over {2..20} {
    if (N/2)*2 = N {
      output [group check NdX out of N] named "even group of [N], target [T]"
    } else {
      output [group check (2*N)dX out of 2*N] named "odd group of [N], target [T]"
    }
  }
}

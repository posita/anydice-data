function: ATTACKS:s vs AC:n with FAILURE_EFFECT:n for DAMAGE:d FORCEFUL:n TWIN:n DEADLY:d {
  SUM: {0}
  N: 0
  F: 0
  T: 0
  FE: 0
  LAST: 0
  loop ATTACK over ATTACKS {
    S: 0
    N: N + 1
    if LAST = 0 { LAST: ATTACK }
    else if ATTACK < LAST { 
      LAST: ATTACK
      FE: FAILURE_EFFECT
    }
    if (N = 2) { 
      F: FORCEFUL 
      T: TWIN
    } else if (N > 2) { 
      F: 2*FORCEFUL 
      T: TWIN
    }
    SUM: SUM + [1d20 plus ATTACK vs AC with FE for DAMAGE FORCEFUL TWIN DEADLY] 
  }
  result: SUM
}

function: ROLL:n plus ATTACK:n vs AC:n with FAILURE_EFFECT:n for DAMAGE:d FORCEFUL:n TWIN:n DEADLY:d {
  if (ROLL + ATTACK >= (AC+10)) { result: 2*DAMAGE+DEADLY+FORCEFUL+TWIN }
  if (ROLL + ATTACK >= AC) {
    if (ROLL = 20){ result: 2*DAMAGE+DEADLY+FORCEFUL+TWIN }
    result: DAMAGE+FORCEFUL+TWIN
  }
  if (ROLL = 20){ result: DAMAGE+FORCEFUL+TWIN }
  result: FAILURE_EFFECT
}

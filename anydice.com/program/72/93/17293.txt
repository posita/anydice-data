MISS: 0
HIT: 1
CRITICALHIT: 2
function: attack ROLL:n vs AC:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 { result: CRITICALHIT }
 if ROLL + TOHIT >= AC { result: HIT }
 result: MISS
}
function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: 0
}
function: attack {
result: [damage for [attack d20 vs AC]]
}
AC: d{10..25}

STRENGTH: 5
RAGEDAMAGE: 2
PROF: 2
TOHIT: STRENGTH + PROF
HITDAMAGE: 2d6 + [explode 1d12] + STRENGTH + RAGEDAMAGE
CRITICALDAMAGE: 4d6 + STRENGTH + RAGEDAMAGE

output [attack] named "1 Hit Damage"
output 2d[attack] named "2 Hits Damage"
output 3d[attack] named "3 Hits Damage"
output 4d[attack] named "4 Hits Damage"

function: max X Y {
  if Y > X {
    result: Y
  } else {
    result: X
  }
}

function: min X Y {
  if Y < X {
    result: Y
  } else {
    result: X
  }
}

function: roll ROLL:n with SKILL {
  if ROLL = 100 {
    result: -20
  } if ROLL = 1 {
    result: 20
  } if ROLL <= SKILL {
    result: [max (SKILL/10 - ROLL/10) 1]
  } if ROLL > SKILL {
    result: [min (SKILL/10 - ROLL/10) (-1)]
  }
}

function: suc ROLL:s in D {
  result: (ROLL >= D) - (ROLL = 1)
}

function: res ROLL:n SK D {
  if ROLL <= 4 {
    result: -50
  } else if (ROLL + SK) < D {
    result: (ROLL + SK)/5 - D/5 - 10
  } else if ROLL >= 17 & (ROLL + SK) >= D {
    result: 50
  } else {
    result: (ROLL + SK)/5 - D/5
  }
}
    
X: [res 3d6 20 34]
output X named "Probabilities"
output X >= 0 named "Probabilities"
output 3d6 named "3d6 Bell-curve"

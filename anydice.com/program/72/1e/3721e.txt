function: explode N:n {
 if N = 10 { result: N + [explode d10] }
 result: N
}

function: ROLL:n wild WILD:n diff DIFF:n {
  if ROLL = 1 {
    if WILD < DIFF { result: -1000 }
    result: 0
  }
  if ROLL  = 10 { result: [explode d10] }
  if ROLL  >= DIFF { result: 1 }
  result: 0
}

function: ROLL:n diff DIFF:n {
if ROLL = 10 { result: 0 }  
if ROLL >= DIFF { result: 1 }
  result: 0
}

	ROLL: POOL-1
	result: [highest of -1 and ROLLd[d10 diff DIFF] + [d10 wild d10 diff DIFF]]
}

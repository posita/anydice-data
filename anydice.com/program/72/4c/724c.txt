DIETYPE: d6
LOWESTAMOUNTOFDICE: 2
HIGHESTAMOUNTOFDICE: 5

function: find longest set in DICE:s {
  LONGESTSET: 0

  loop P over {2..[maximum of DIETYPE]} {
    SET: [count P in DICE]
    if SET > LONGESTSET { LONGESTSET: SET}
  }

  if LONGESTSET > 1 { result: LONGESTSET}
  else       { result: 0 }
}

loop X over {LOWESTAMOUNTOFDICE..HIGHESTAMOUNTOFDICE} {
  output 0 named "------------------- [X]D6 -------------------"

  output [find longest set in Xd6] named "Longest set in [X]d6 that is not ones"

  output [count 1 in Xd6] named "Counting ones in [X]d6"
}

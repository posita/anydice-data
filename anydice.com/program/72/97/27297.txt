P: 16
DAMAGE: 1d20+80
DICE: 6d4
MODIFIER: -4
UNARMED: 0

FORCE: 10
ALACRITY: 10
AWARENESS: 10
PSYCHE: 10
MAGIC: 10

ENEMYALACRITY: 10

ACTIONPOINTS: 20+ALACRITY*8-[highest of ALACRITY-10 and 0]*4
ACTIONCOST: ACTIONPOINTS

FORCEMODIFIER: FORCE*10-[highest of FORCE-10 and 0]*5
PALADINMODIFIER: (FORCE+PSYCHE)*10-[highest of FORCE+PSYCHE/2-10 and 0]*5
PALADINMODIFIERCC: (FORCE+PSYCHE)*10-[highest of FORCE+PSYCHE/2-10 and 0]*5
CAP: [lowest of P and [maximum of DAMAGE]/5]
CAPCC: [lowest of (P+1)/2 and [maximum of DAMAGE]/5]
ROGUEBONUS: 35+15*P
ROGUEBONUSCC: 10+15*(P/2)
MONKBONUSCC: (P+1)/2
ACCURACY: 0
CRITCHANCE: 0
MONKACCURACY: 0
MONKCRITCHANCE: 0
MONKACCURACYCC: 0
MONKCRITCHANCECC: 0
ENEMYDODGE: d10+ENEMYALACRITY
if ENEMYALACRITY <= 0 {
 ENEMYDODGE: 0
}

loop N over {1..(ACTIONPOINTS/ACTIONCOST)}{
 ACCURACY: ACCURACY + (DICE+MODIFIER+(AWARENESS-10) >= ENEMYDODGE)
 CRITCHANCE: d200+5*(ALACRITY-10)+5*[highest of (DICE+MODIFIER+(AWARENESS-10)-ENEMYDODGE) and 0] >= 160+MODIFIER*5
 MONKACCURACY: MONKACCURACY + (DICE+MODIFIER+(AWARENESS-10)+P*UNARMED >= ENEMYDODGE)
 MONKCRITCHANCE: d200+5*(ALACRITY-10)+5*[highest of (DICE+MODIFIER+(AWARENESS-10)+P*UNARMED-ENEMYDODGE) and 0] >= 160+MODIFIER*5
 MONKACCURACYCC: MONKACCURACYCC + (DICE+MODIFIER+(AWARENESS-10)+MONKBONUSCC*UNARMED >= ENEMYDODGE)
 MONKCRITCHANCECC: d200+5*(ALACRITY-10)+5*[highest of (DICE+MODIFIER+(AWARENESS-10)+MONKBONUSCC*UNARMED-ENEMYDODGE) and 0] >= 160+MODIFIER*5
}
output (((DAMAGE)*(FORCEMODIFIER+CRITCHANCE*MAGIC*10))/100)*ACCURACY named "Base Breech Rifle Damage Per Round"
output (((DAMAGE+P*UNARMED)*(FORCEMODIFIER+MONKCRITCHANCE*MAGIC*10))/100)*MONKACCURACY named "Monk Breech Rifle Damage Per Round"
output (((DAMAGE)*(FORCEMODIFIER+ROGUEBONUS+CRITCHANCE*MAGIC*10))/100)*ACCURACY named "Rogue Breech Rifle Sneak Damage Per Round"
output (((DAMAGE)*(PALADINMODIFIER+CRITCHANCE*MAGIC*10))/100)*ACCURACY named "Paladin Breech Rifle Damage Per Round"
output (((DAMAGE+CAPd6-CAP)*(FORCEMODIFIER+CRITCHANCE*MAGIC*10))/100)*ACCURACY named "Fighter Breech Rifle Damage Per Round"
output (((DAMAGE+MONKBONUSCC*UNARMED)*(FORCEMODIFIER+ROGUEBONUSCC+MONKCRITCHANCECC*MAGIC*10))/100)*MONKACCURACYCC named "Agent (Rogue/Monk) Breech Rifle Sneak Damage Per Round"
output (((DAMAGE+CAPCCd6-CAPCC+MONKBONUSCC*UNARMED)*(FORCEMODIFIER+MONKCRITCHANCECC*MAGIC*10))/100)*MONKACCURACYCC named "Brawler (Monk/Fighter) Breech Rifle Damage Per Round"
output (((DAMAGE+CAPCCd6-CAPCC)*(FORCEMODIFIER+ROGUEBONUSCC+CRITCHANCE*MAGIC*10))/100)*ACCURACY named "Thug (Rogue/Fighter) Breech Rifle Sneak Damage Per Round"
output (((DAMAGE+CAPCC)*(PALADINMODIFIERCC+CRITCHANCE*MAGIC*10))/100)*ACCURACY named "Harrier (Fighter/Paladin) Breech Rifle Damage Per Round"
output (((DAMAGE+MONKBONUSCC*UNARMED)*(PALADINMODIFIERCC+MONKCRITCHANCECC*MAGIC*10))/100)*MONKACCURACYCC named "Disciple (Monk/Paladin) Breech Rifle Damage Per Round"
output (((DAMAGE)*(PALADINMODIFIERCC+ROGUEBONUSCC+CRITCHANCE*MAGIC*10))/100)*ACCURACY named "Slayer (Rogue/Paladin) Breech Rifle Sneak Damage Per Round"

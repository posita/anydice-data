function: explode N:n {
 if N = 1 {
  result: N - d10
 }
 if N = 10 {
  result: N + d10
 }
 result: N
}

function: damage N:n {
 result: N
}

function: autodamageiii N:n {
 result: N*3
}

function: autodamagevi N:n {
 result: N*4
}
function: autodamageii N:n {
 result: N*2
}
function: crit N:n {
 if N = 6 {
 result: 1
 }
 result: 0
}

DAMAGE: 4d([damage d6])
AUTODAMAGEII: 2d([autodamageii d6])
AUTODAMAGEIII: 2d([autodamageiii d6])
AUTODAMAGEVI: 2d([autodamagevi d6])
CRIT: 2d([crit d6])
ATTACK: [explode 1d10]+16
DODGE: [explode 1d10]+12

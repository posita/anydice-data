loop N over {1..10}{
 INITIAL_LEVEL: N
 MAXIMUM_LEVEL: 10
 DAMAGEMODIFIER: ((100+(3+2*INITIAL_LEVEL)*INITIAL_LEVEL)*(4*INITIAL_LEVEL+10+MAXIMUM_LEVEL))/100
 CHARGE: N^2+10*N
 DAMAGE: INITIAL_LEVELd20
 AWARENESS: 20
 ENEMYALACRITY: 10
 ACCURACY: INITIAL_LEVELd2+(10-INITIAL_LEVEL)*2
 HITRATE: ACCURACY+(AWARENESS-10) >= d10+ENEMYALACRITY
 output (HITRATE)*(DAMAGE+DAMAGEMODIFIER) named "Level [N] Spell"
}

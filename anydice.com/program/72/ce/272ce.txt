function: N:s advantage A:s {
 result: [highest of 2@[sort {N, A}] and 1@A]
}

function: N:s disadvantage D:s {
 result: [lowest of 2@[sort {N, D}] and (#D)@D]
}

ONE: d20
TWO: 2d20
THREE: 3d20

output 3@THREE named "3D"
output [ONE disadvantage TWO] named "2D"
output [TWO disadvantage ONE] named "1D"
output 2@THREE named "normal"
output [TWO advantage ONE] named "1A"
output [ONE advantage TWO] named "2A"
output 1@THREE named "3A"

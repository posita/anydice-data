function: dungeon ROLL:n {
  MIN: ROLL >= 7
  MAJ: ROLL >= 10
  result: MIN + MAJ
}

function: dragon ROLL:n {
  MIN: ROLL >= 10
  MAJ: ROLL >= 18
  result: MIN + MAJ
}

function: interpret blades ROLL:n {
  if 1@ROLL = 6 {
    result: 2 + (2@ROLL = 6)
  }

  result: 1@ROLL >= 4
}

function: blades COUNT:n {
  if COUNT = 0 {
    ROLL: 2@2d6
  } else {
    ROLL: COUNTd6
  }

  result: [interpret blades ROLL]
}


loop I over {0..3} {
  output [dungeon 2d6+I] named "Dungeon World +[I]"
  output [dragon 1d20+I] named "Dragon World +[I]"
  output [blades I] named "Blades [I]"
  \output 0 named ""\
}

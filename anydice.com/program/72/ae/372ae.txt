function: reroll lowest A:s B:s {
  if B < 5 & B <= A {
    result: A + d8
  }
  if A < 4 & A <= B {
    result: B + d6
  }
  result: A + B
}

function: reroll lowest versus average A:s B:s {
  A_DIFF: (A * 2) - 7
  B_DIFF: (B * 2) - 9

  if B < 5 & B_DIFF <= A_DIFF {
    result: A + d8
  }
  if A < 4 & A_DIFF <= B_DIFF {
    result: B + d6
  }
  result: A + B
}

  B_DIFF: B - 8

  if B < 5 & B_DIFF <= A_DIFF {
    result: A + d8
  }
  if A < 4 & A_DIFF <= B_DIFF {
    result: B + d6
  }
  result: A + B
}

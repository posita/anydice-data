function: reroll ROLL:n {
 if ROLL = 1 { result: 1d5 }
 if ROLL = 2 {result: 1d5 }
 else { result: ROLL }
}

function: higher RA:n and RB:n {
 if RA>RB { result: RA}
 if RB>RA { result: RB}
 else {result: RA}

}

SECOND: 1d5
THIRD: 1d5

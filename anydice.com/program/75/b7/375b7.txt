BRUTAL_DICE: 3

function: ROLL:n plus ATTACK_BONUS attack ac AC:n with DAMAGE_DICE:d plus BONUS:n {
  if ROLL = 20 {
    DD: 0
    loop I over {1..BRUTAL_DICE+1} {
      DD: DD + DAMAGE_DICE
    }
    result: 2*DD + BONUS
  }
  if ROLL + ATTACK_BONUS >= AC {
    result: DAMAGE_DICE + BONUS
  }
  
  result: 0
}

function: swing SWINGS:n attacks of ROLL:n plus ATTACK_BONUS:n against ac AC:n with DAMAGE_DICE:d plus BONUS:n {
  if SWINGS <= 0 {
    result: 0
  }
  DAMAGE: 0
  loop I over {1..SWINGS} {
    DAMAGE: DAMAGE + [ROLL plus ATTACK_BONUS attack ac AC with DAMAGE_DICE plus BONUS]
  }
  result: DAMAGE
}

AC: 20
SWINGS: 2

output [swing SWINGS attacks of [highest 1 of 2d20] plus 16 against ac AC with 1d10 plus 10] named "[SWINGS] Swings against AC [AC] (Advantage)"
output [swing SWINGS attacks of [highest 1 of 1d20] plus 16 against ac AC with 1d10 plus 10] named "[SWINGS] Swings against AC [AC] (Straight)"
output [swing SWINGS attacks of [lowest 1 of 2d20] plus 16 against ac AC with 1d10 plus 10] named "[SWINGS] Swings against AC [AC] (Disadvantage)"

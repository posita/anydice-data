function: sj X:n c C:n {
 if C { result: 3 }
 if X=6 { result: 2 }
 if X>3 { result: 1 }
 result: 0
}

function: judge X:d crit C:n {
 CRIT: 0
 if C {
  CRIT: [count 6 in X]>=2
 }
 result: [sj 1@X c CRIT]
}

set "position order" to "lowest first"
output [judge 2d6 crit 0] named "0d"
set "position order" to "highest first"
output [judge 1d6 crit 0] named "1d"
output [judge 2d6 crit 1] named "2d"
output [judge 3d6 crit 1] named "3d"
output [judge 4d6 crit 1] named "4d"

\ This stuff sets up the Skew rules; you should leave it alone.  Scroll almost to the bottom to decide what dice will be rolled. \

function: U:d {
  if (#U = 1 & 1@{U} = 0) { result: {} } \ Because AnyDice says 0d12 means one die. \
  result: {[maximum of 1@U]:#U}
}
function: U:d and V:d {
  result: {[U], [V]}
}
function: U:d and V:d and W:d {
  result: {[U], [V], [W]}
}
function: U:d and V:d and W:d and X:d {
  result: {[U], [V], [W], [X]}
}
function: U:d and V:d and W:d and X:d and Y:d {
  result: {[U], [V], [W], [X], [Y]}
}
function: U:d and V:d and W:d and X:d and Y:d and Z:d {
  result: {[U], [V], [W], [X], [Y], [Z]}
}

function: tail of S:s {
  RESULT: {}
  loop I over {2..#S} {
    RESULT: {RESULT, I@S}
  }
  result: RESULT
}

function: score S:s {
  RESULT: 0
  loop I over S {
    CANDIDATE: I * [count I in S]
    if CANDIDATE > RESULT { RESULT: CANDIDATE }
  }
  result: RESULT
}
function: roll R:s and combine with S:s {
  if #R = 0 { result: [score S] }
  TAIL: [tail of R]
  RESULT: {}
  loop I over {1..1@R} {
    RESULT: {RESULT, [roll TAIL and combine with {S, I}]}
  }
  result: RESULT
}

function: skew R:s {
 result: d[roll R and combine with {}]
}
function: degrees N:n {
  if N < 0 { result: -[degrees -N] }
  result: ([absolute N] + 4) / 5
}
function: A:s versus B:s {
  result: [degrees [skew A] - [skew B]]
}

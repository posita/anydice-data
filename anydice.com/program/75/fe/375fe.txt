function: armordiff INCOMINGDAMAGE:n vs ARMOR:n {
 if INCOMINGDAMAGE >= ARMOR { result: INCOMINGDAMAGE-ARMOR }
 result: 0
}

function: antimaterielrifle ROLL:n vs DEFENSE:n and ARMOR:n {
 OUTGOINGDAMAGE: 0
 if ROLL = 1 { OUTGOINGDAMAGE: 0}
 if ROLL >= 20 {
  OUTGOINGDAMAGE: [highest 2 of 4d6]
 }
 if ROLL >= DEFENSE { OUTGOINGDAMAGE: 2d6}
 result: [armordiff OUTGOINGDAMAGE vs ARMOR]
}

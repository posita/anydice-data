function: reroll R:n under N:n this D {
   if R < N { result: D } else {result: R}
}

function: hit H:n mod M crit C dmg D bonus B ac A {
   if H >= C {result: D+D+B}
   else if H+M > A {result: D+B} 
   else {result: 0}
}

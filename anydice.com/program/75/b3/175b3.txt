\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + ATTACKMOD >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 10-20\
ENEMYAC: d{10..20}

\assuming >= level 17\
PROFICIENCY: 6
ABILITYMOD: 5
ENHANCEMENT: 3

\these modifiers are constant\
GWMDAMAGE: 10
GWMHIT: -5
LIFEDRINKER: ABILITYMOD
HEXDMG: 1d6

ATTACKMOD: ABILITYMOD + LIFEDRINKER + PROFICIENCY + ENHANCEMENT
DMGBONUS: ABILITYMOD + LIFEDRINKER + ENHANCEMENT
CRITROLL: 20
ATTACKTIMES: 2
DAMAGE: 1d8 + DMGBONUS
CRITDAMAGE: 2d8 + DMGBONUS

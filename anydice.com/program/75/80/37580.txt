DICEMIM: 1 \Starting number of dice rolled\
DICEMAX: 5 \Ending number of dice rolled\
DIEBONUS: 32 \Bonus per die\
CURRENTPROGRESS: 70 \Your current project progress\
PROGRESSTARGET: 250\The total progress the project needs\
OMAKEBONUS: 0 \Omake bonus points\
NOBONUSDICE: 0 \Number of extra dice rolled without a bonus\

function: evaluate ROLL:s {
 SUCCESSES: ROLL >= 0
 if SUCCESSES > 0 { result: 1 }
 result: 0
}

loop DICE over {DICEMIM..DICEMAX} {
 output [evaluate (1*DICE)d50+(1*NOBONUSDICE)d50+DIEBONUS*DICE+CURRENTPROGRESS-PROGRESSTARGET+OMAKEBONUS] named "[DICE]"
}

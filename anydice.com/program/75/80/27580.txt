function: dmg RT AC D:d DB:n {
 if MHC=2 {result: D+[maximum of D]+DB}
 if MHC=1 {result: D+DB}
 result: 0
}

function: atkdmg ADN:n B:n D:d DB:n AC:n {
 if ADN=1 {R: 1@2d20}
 else if NAD=-1 {R: 2@2d20}
 else {R: 1d20} 

 if R=20 {result: D+[maximum of D]+DB}
 if R>=AC {result: D+DB}
 result: 0
}

DEX:5
PB:3

output [atkdmg 1d20 DEX+PB 1d4 DEX+6 8+DEX]

MISS: 0
HIT: 1

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 {
  if ROLL + ATTACK >= (DEFENSE + 10) { result: HIT}
  result: MISS
 }
 if ROLL = 20 {
  if ROLL + ATTACK >= (DEFENSE - 10) { result: HIT}
  result: MISS
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: ROLL:n with advantage {
 result: [highest 1 of 2d20]
}

function: ROLL:n with disadvantage {
 result: [lowest 1 of 2d20]
}

ATTACK: 1
loop DC over {1..30} {
 output [attack d20 vs DC] named "[DC]" 
 \ output [attack [d20 with advantage] vs DC] named "[DC with advantage]" \
 \ output [attack [d20 with disadvantage ] vs DC ] named "[DC with disadvantage]" \
}

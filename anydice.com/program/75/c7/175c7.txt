function: pool ROLL:s { 
  result: 1@ROLL
} 
\rolling normal dice\

function: drag ROLL:s {
  result: 1@ROLL
} 
\rolling negation dice\

function: D:s negate P:s {
  if 1@P = 1@D {
    result: 0
  } else {
    result: 1@P
  }
} \if negation = normal, -1. otherwise, result is normal\

loop A over {1..6} {
  loop B over {0..3} {
    output [[drag Bd6] negate [pool Ad6]] named "[A]d -- [B] Drag"
  }
}

function: pool ROLL:s { 
  loop W over {1..6} {
    if 1@ROLL = [W] {
      result: [W]
    }
  }
} \rolling normal dice\

function: drag ROLL:s {
  loop M over {1..6} {
    if 1@ROLL = [M] {
      result: [M]
    }
  }
} \rolling negation dice\

function: negate P:s equal D:s {
  if 1@P = 1@D {
    result: -1
  } else {
    result: 1@P
  }
} \if negation = normal, -1. otherwise, result is normal\

loop B over {0..3} {
  loop A over {1..6} {
    output [negate [pool Ad6] equal [drag Bd6]] named "[A]d Pool --[B]d Drag"
  }
}

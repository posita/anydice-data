function: comparison A:s B:s {
  R: 0
  loop N over {1..[highest of #A and #B]} {
   if (N@A = 0) & (N@B >= 4) { R: R - 1}
   if (N@B = 0) & (N@A >= 4) { R: R + 1}
   if (N@A > N@B) & (N@A > 0) & (N@B > 0) { R: R + 1 }
   if (N@B > N@A) & (N@A > 0) & (N@B > 0) { R: R - 1 }
  }
  result: R
}

output [comparison 2d6 2d6]
output [comparison 3d6 2d6]
output [comparison 4d6 2d6]

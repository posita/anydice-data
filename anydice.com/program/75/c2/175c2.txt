function: pool ROLL:s { 
  loop W over {1..6} {
    if 1@ROLL = [W] {
      result: [W]
    }
  }
} \normal dice\

function: drag ROLL:s {
  loop M over {1..6} {
    if 1@ROLL = [M] {
      result: [M]
    }
  }
} \dice that negate normal dice\

function: negate P:s equal D:s {
  if 1@P = 1@D {
    result: -1
  } else {
    result: 1@P
  }
} \negation process\

loop A over {1..6} {
  loop B over {0..3} {
    output [negate [pool Ad6] equal [drag Bd6]] named "[A]d Pool --[B]d Drag"
  }
}

\alchemy is a function\
\ROLL:s is a Sequence variable\
\1@ROLL checks what the dice rolled\
\if else statements for success tier\
\remember = is checking value, == is assigning value\
\yes = 1, no = 0\

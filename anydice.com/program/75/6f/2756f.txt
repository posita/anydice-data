function: sj X:n c C:n {
 if C { result: 3 }
 if X=6 { result: 2 }
 if X>3 { result: 1 }
 result: 0
}

function: judge X:d focus F:n {
 CRIT: 0
 FOCUS: 0
 if F {
  CRIT: [count 6 in X]>=2
  FOCUS: [highest 1 of X]
 }
 else {
  FOCUS: [lowest 1 of X]
 }
 result: [sj FOCUS c CRIT]
}

function: strength S:n {
 N: S
 C: 1
 if S<=0 {
  N: 2
  C: 0
 }
 result: [judge Nd6 focus C]
}

loop S over {0..6}{
 output [strength S] named "d[S]"
}

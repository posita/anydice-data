function: hit HIT:d crit CRIT:d atk DICE:n MOD:n ac AC:n res RES:n {
ATK: DICE+MOD
if ATK>=AC+10 | (DICE=20 & ATK>=AC) { result: CRIT-RES }
if ATK>=AC | (DICE=20 & ATK>AC-10) { result:HIT-RES }
result: 0
}

output [hit 3d8+4 crit (3d8+4)*2+d8 atk d20 12 ac 21 res 0]
output [hit 3d10+4 crit (3d10+4)*2 atk d20 12 ac 21 res 0]

output [hit 4d8+4 crit (4d8+4)*2+d8 atk d20 22 ac 29 res 0]
output [hit 4d10+4 crit (4d10+4)*2 atk d20 22 ac 29 res 0]

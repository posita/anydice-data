function: edgesnag EDGE:s SNAG:s {
  loop X over {1..#EDGE} {
    if X@EDGE >= X@SNAG { result: X@EDGE }
  }
  if #EDGE < #SNAG { result: #SNAG@SNAG }
  result: 1
}

output [edgesnag {5,5,3} {6,5}] named "5,5,3 vs 6,5"
output [edgesnag {4,3} {5,5,3,2}] named "4,3 vs 5,5,3,2"
output [edgesnag {3} {5}] named "3 vs 5"
output [edgesnag 3d6 2d6] named "3d6 vs 2d6"
output [edgesnag 2d6 4d6] named "2d6 vs 4d6"
output [edgesnag 1d6 1d6] named "1d6 vs 1d6"

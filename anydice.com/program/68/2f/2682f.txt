\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{11..17}



ATTACKMOD: 3
DMGBONUS: 1
PROFICIENCY: 2
ACCBONUS: 1
CRITROLL: 20
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS
output [attackroll d20 vs ENEMYAC] named "+1 flail"

ACCBONUS: 0
DAMAGE: 1d12 + ATTACKMOD
CRITDAMAGE: 2d12 + ATTACKMOD
output [attackroll d20 vs ENEMYAC] named "Greataxe"

ACCBONUS: 0
DAMAGE: 1d12 + ATTACKMOD
CRITDAMAGE: 2d12 + ATTACKMOD
output [attackroll d20 vs ENEMYAC] named "TrownHammer"

ACCBONUS: 1
DAMAGE: 1d8 + 1d{0..1}*1d4+ ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + 1d{0..1}*2d4 + ATTACKMOD + DMGBONUS
output [attackroll d20 vs ENEMYAC] named "Thrown+1Flail"

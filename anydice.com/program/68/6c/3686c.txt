function: hit HIT:d crit CRIT:d atk DICE:n MOD:n ac AC:n res RES:n {
ATK: DICE+MOD
if ATK>=AC+10 | (DICE=20 & ATK>=AC) { result: CRIT-RES }
if ATK>=AC | (DICE=20 & ATK>AC-10) { result:HIT-RES }
result: 0
}

output [hit 2d8+4 crit (2d8+4)*2+d8 atk d20 9 ac 18 res 0]
output [hit 2d10+4 crit (2d10+4)*2 atk d20 9 ac 18 res 0]

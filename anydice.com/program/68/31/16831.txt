function: roll TIMES:n die SIDES:n count SUCC:n explode on EXPL:n {
  SUCCESSES: 0
  DIE: 1dSIDES
  loop N over {1..TIMES} {
    if N = 1 {
        SUCCESSES: SUCCESSES + [DIE success on SUCC explode on EXPL]
    }
    else {
        SUCCESSES: SUCCESSES + [DIE success on SUCC]
    }
  }
  result: SUCCESSES
}

function: N:n success on SUCC:n explode on EXPL:n {
  SUCCESSES: 0
  if N <= 1 { SUCCESSES: -1 }
  if N >= SUCC { SUCCESSES: 1 }
  if N >= EXPL { SUCCESSES: SUCCESSES + [DIE success on SUCC explode on EXPL] }
  result: SUCCESSES
}

function: N:n success on SUCC:n {
  SUCCESSES: 0
  if N >= SUCC { SUCCESSES: 1 }
  result: SUCCESSES
}

output [roll 4 die 6 count 5 explode on 6] named "roll 4d6 count 5+, explode on 6+"

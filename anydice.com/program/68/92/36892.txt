loop M over {3..6} {
output 1@Md6 named "Highest in [M]d" 
}

loop D over {3..6} {
output [count {4..6} in Dd6] named "Normal Hits in [D]d" 
}

function: thegoldhack R:n again N:n {
 if N >= R { result: 1 + [thegoldhack R again d6] }
 result: N >= 4
}

TGH: [thegoldhack 6 again d6]

loop N over {3..6} {
 output [lowest of 6 and NdTGH] named "Magic Hits in [N]d"
}

function: roll N:n open D:n{
 if N <= 3 {
  if D = 90 {result: -1d100}
 } 
 if N >= D {
  if D < 93 {result: N + [roll 1d100 open (D+1)]}
   result: N
 }
 result: N
}
function: opposed R:n{
 ARMOUR: 0
 if R >= 20 + ARMOUR {result: (R - 20 - ARMOUR)*5}
 if R < 0 {result: R/2}
 result: 0
}
output [opposed [roll 1d100 open 90] - [roll 1d100 open 90]]

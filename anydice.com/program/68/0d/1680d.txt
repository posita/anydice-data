set "explode depth" to 3
function: successes in ROLL:s {
  \ explosions count, and so do their extra dice (whatever is rolled) so just double-count them \
  result: [count {1,1..3} in ROLL]
}
LOW: {1..4}
MID: {5}
HIGH: {6}
DIE: [explode d6]
loop DICE over LOW { \ manually switch this to get around 5s timeout \
  output [successes in DICEdDIE] named "[DICE]d6"
}

\ BROKEN COMPASS \

function: player roll A:s {
 ONES: [count 1 in A]
 TWOS: [count 2 in A] 
 THREES: [count 3 in A] 
 FOURS: [count 4 in A] 
 FIVES: [count 5 in A] 
 SIXES: [count 6 in A] 
 
 LIST: [sort {ONES, TWOS, THREES, FOURS, FIVES, SIXES}]

 if 1@LIST > 1 { 
   if 2@LIST > 1 {result: 1@LIST + (1/2@LIST) }
   result: 1@LIST 
 }
 result: 1
}

loop N over {1..9}{
 output [player roll Nd6] named "[N]d6" 
}

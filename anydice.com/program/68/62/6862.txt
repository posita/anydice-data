function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d10] }
 result: N >= 8
}

function: nwoda N:n {
 result: N >= 8
}

NWOD: [nwod 10 again d10]
NWODA: [nwoda d10]

loop N over {1..5} {
 output [lowest of 5 and NdNWOD] named "[N]d10"
}
loop N over {1..5} {
 output [lowest of 5 and NdNWODA] named "[N]d10 without reroll"
}

function: success ROLL:n {
  if ROLL = 6 { result: 2 }
  else if ROLL = 4 | ROLL = 5 { result: 1 }
  else { result: 0 }
}

function: count successes in DICE:s {
  SUM: 0
  loop DIE over DICE {
    SUM: SUM + [success DIE]
  }
  result: SUM
}

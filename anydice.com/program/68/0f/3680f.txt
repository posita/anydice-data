function: ROLL:n rerolls BAD:s as REROLL:d {
  if ROLL = BAD { result: REROLL }
  result: ROLL
}

function: DICE:d reroll BAD:s {
  TOTAL: 0
  MAX: [maximum of DICE]
  loop X over {1..#DICE} {
    NEW: [1@DICE rerolls BAD as 1@DICE]
    TOTAL: TOTAL + NEW
  }
  result: TOTAL
}

output [2d6 reroll {1,2}]
output [1d10 reroll {1,2}]

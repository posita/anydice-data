function: hc AC:n THB:n N:n{
  BASE: (1d20+THB)>=AC
  RES: 0
  loop I over {1..N}{
    RES: RES + BASE
  }
  result: RES > 0
}

function: crit DIE:d BASE:n{
  result: DIE * (1d20>=BASE)
}

function: optdmg DIST:s DMG:n {
  NEW: {}
  loop I over {1..#DIST} {
    if I@DIST = 0 {NEW: {NEW: 0}}
    if I@DIST > 0 {NEW: {NEW, I@DIST + DMG}}
  }
  result: NEW
}

set "position order" to "lowest first"
DIE: {1,2,3,9,9,9}                \ 9's represent successes, the other numbers represent the focus cost required to turn the result into a success \

function: roll DICE:s with F:n focus{
 SUCCESSES:[count 9 in DICE]      \ count the number of successes \
 SORTED:[sort DICE]               \ sort results lowest to highest (this is why successes needed to be represented by a large number) \
 loop X over {1..#SORTED} {       \ loop through the non-success results until their sum is greater than your focus points \
  if F<{1..X}@SORTED { result: SUCCESSES + X-1 }
 }
 result:#SORTED
}

loop DICE over {1..5} {
 loop FOCUS over {1..4} {
  output [roll DICEdDIE with FOCUS focus]
 }
}

function: attack ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + ARCH >= AC { result: DAMAGE }
 result: 0
}

ENEMYAC: d{12..19}

ATTACKMOD: 5

PROFICIENCY: 3
CRITROLL: 20
EXTRAATTACK: 2
DMGBONUS: 0
ACCBONUS: 0
EXTRADMGDICE: 0

ARCH: 0

DAMAGE: 2d{{1..2}, {3..6}:4} + ATTACKMOD + DMGBONUS + EXTRADMGDICE
CRITDAMAGE: 3d{{1..2}, {3..6}:4} + ATTACKMOD + DMGBONUS + 2dEXTRADMGDICE

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "GWF Greatsword"

DAMAGE: 1d6 + ATTACKMOD + DMGBONUS + EXTRADMGDICE
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS + 2dEXTRADMGDICE

output (EXTRAATTACK + 1)d[attack d20 vs ENEMYAC] named "TWF Shortswords"

ACCBONUS: 0
DAMAGE: 2d6 + ATTACKMOD + DMGBONUS + EXTRADMGDICE
CRITDAMAGE: 3d6 + ATTACKMOD + DMGBONUS + 2dEXTRADMGDICE

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "Defense/None Greatsword"

DAMAGE: 1d8 + ATTACKMOD + DMGBONUS + 2 + EXTRADMGDICE
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS + 2 + 2dEXTRADMGDICE

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "Dueling Longsword"

ARCH: 2
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS + EXTRADMGDICE
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS + 2dEXTRADMGDICE

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "Archery Longbow"

ARCH: 0
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS + EXTRADMGDICE
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS + 2dEXTRADMGDICE

output EXTRAATTACKd[attack d20 vs ENEMYAC] named "Protection/None 1d8"

BASEDIE: d10

function: target TARGET:n explode EXPLODE:n reroll REROLL:s VALUE:n {
  if VALUE = EXPLODE {
    result: 1 + [target TARGET explode EXPLODE reroll REROLL BASEDIE]
  }
  if VALUE  >= TARGET {
    result: 1
  }
  if (REROLL = VALUE)>0 {
    result: [target TARGET explode EXPLODE reroll {} BASEDIE]
  }
  result: 0
}

DIE: [target 8 explode 10 reroll {} BASEDIE]


loop N over {1..10}{
  output [lowest of NdDIE and 12] named "[N]"
}

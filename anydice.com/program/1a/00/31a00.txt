
function: biplode DIE:d {
 MAX: [maximum of DIE]
 result: [biplode DIE helper ]
}

function: biplode N:n helper {
 if N = MAX { result: MAX + DIE}
 if N = 1 {result: -MAX + DIE}
 result: N
}


D:[biplode 1d10]
loop N over {6, 5, 4, 3}
{
DIS:N - 2
output [lowest 2 of NdD] named "Biplode 2d10 -[DIS]"
}
output 2dD named "Biplode 2d10"
loop N over {3, 4, 5, 6}
{
ADV:N-2
output [highest 2 of NdD] named "Biplode 2d10 +[ADV]"
}

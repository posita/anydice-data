function: unopposed ROLL:s TARGET:n {
 SUCCESSES: ROLL <= TARGET
 result: SUCCESSES
}

function: ftf ROLL:s TARGET:n DROLL:s DEF:n{
 if DROLL <= DEF { result: ROLL <= (T-DROLL) }
 else { result: [unopposed ROLL TARGET] }
}

function: burst ROLL:s {
 D: 1d20
 if D > T { SUCCESSES: ROLL <= T }
 else { SUCCESSES: ROLL <= (T-D) }
 result: SUCCESSES
}

T: 18
D: 11
B: 3

output [unopposed Bd20 T] named "B[B] unopposed"
output [ftf Bd20 T 1d20 D] named "B[B] face2face"

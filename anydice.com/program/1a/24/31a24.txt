loop REROLLS over {0..4}{
	SKILLPTS: REROLLS
	SUCCESSES: [count {5, 6} in 5d6]
	loop DISCARD over {1..5}
	{
                if #(SUCCESSES )>= 5
                {
                        SKILLPTS: SKILLPTS - 10
                }
		if SKILLPTS > 0
		{
			SUCCESSES: SUCCESSES + [count {2} in 1d2]
			SKILLPTS: SKILLPTS - 1
		}
	}
        

	REROLLNUMBER: REROLLS
	output SUCCESSES named "Successes at [REROLLNUMBER] Skill Points"
}

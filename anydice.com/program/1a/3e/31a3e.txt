function: reg attack ROLL:n {
if ROLL = 1  {result: 0}
if ROLL >= CRIT {result: 2d{DMG_DICE} + DMG_MOD}
if ROLL+TO_HIT >= AC {result: DMG_DICE + DMG_MOD}
result: 0
}

function: sneak attack ROLL:n {
if ROLL = 1  {result: 0}
if ROLL >= CRIT {result: 2d{DMG_DICE} + 2d{SNEAK_DICE} + 2d{BOOMING_DICE} + DMG_MOD}
if ROLL+TO_HIT >= AC {result: DMG_DICE + SNEAK_DICE + BOOMING_DICE + DMG_MOD}
result: 0
}

function: reroll lowest of ROLL:s as REROLL:d if less than MIN:n {
  LOWEST: (#ROLL)@ROLL       \ the lowest die is sorted last \
  if LOWEST >= MIN {
    result: ROLL + 0
  } else {
    REST: {1..#ROLL-1}@ROLL  \ all but the lowest die \
    result: REST + REROLL
  }
}

function: piercer sneak attack ROLL:n {
if ROLL = 1  {result: 0}
if ROLL >= CRIT {result: [reroll lowest of 3d{DMG_DICE} as DMG_DICE if less than 5] + 2d{SNEAK_DICE} + 2d{BOOMING_DICE} + DMG_MOD}
if ROLL+TO_HIT >= AC {result: [reroll lowest of DMG_DICE as DMG_DICE if less than 5] + SNEAK_DICE + BOOMING_DICE + DMG_MOD}
result: 0
}

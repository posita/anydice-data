
function: explo N:n and S:n{
if(S=-1){if N>=0{N:0}else{N: N + [explo Nd{-1,1,0} and -1] }}
else{if N<=0{N:0}else{N: N + [explo Nd{-1,1,0} and 1] }}
result: N}
function: rdf N:n {
if N > 0 {result: N + [explo Nd{-1,1,0} and 1]}
if N < 0 {result: N + [explo Nd{-1,1,0} and -1]}
result: N}
set "maximum function depth" to 10
output [rdf 4d{-1,1,0}] named "4dF+"
output [rdf 2d{-1,1,0}] named "2dF+"
output [rdf 1d{-1,1,0}] named "1dF+"

AC: 14
INT_MOD: 3
PROFICIENCY: 3

ADVANTAGE:0

\ Stuff defined by level \
FLAME_DICE: 0
MULTIATTACKS: 1

function: hits
{
 result: ([highest 1 of (1 + ADVANTAGE)d20] + INT_MOD + PROFICIENCY) > AC
}

function: multiattack hit count
{
 result: MULTIATTACKSd[hits]
}

function: any multiattack hit
{
 result: [multiattack hit count] > 0
}

output [hits] * (1d8 + INT_MOD + FLAME_DICEd8) named "Green Flame Blade Damage"

output [hits] * (INT_MOD + FLAME_DICEd8) named "Green Flame Blade Second Target Damage"

output [hits] * (1d8 + INT_MOD + FLAME_DICEd8 + INT_MOD + FLAME_DICEd8) named "Green Flame Blade Total Damage (2 Targets)"

output [hits] named "Green Flame Blade Does Hit"

output [multiattack hit count] * (1d8 + INT_MOD) named "Multiattack Damage"

output [multiattack hit count] > 0 named "Multiattack Does Hit"

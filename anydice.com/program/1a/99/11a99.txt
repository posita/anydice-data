DIFFICULTY: 4
MINIMUM: DIFFICULTY

function: roll ROLL:s {
 if 1@ROLL < 1 & (#ROLL)@ROLL = -1 {
  result: -1
 }
 result: [highest of 0 and ROLL]
}

function: roll ROLL:n with willpower {
 result: [highest of 0 and ROLL + 1]
}

function: specialty N:n {
 if N = 10 { result: 2 }
 if N = 1 { result: -1 }
 result: N >= DIFFICULTY
}

NORMAL: d{-1:1, 0:(DIFFICULTY - 2), 1:(11 - DIFFICULTY)}

SPECIALTY: [specialty d10]

function: COUNT:n minimum MIN:n {
  if COUNT < 0 { result: -1 }
  if COUNT < MIN { result: 0 }
  result: 1
}

loop X over {MINIMUM..11} {
  output [[roll X d NORMAL] minimum MINIMUM] named "normal, [X] dice"
}
loop X over {MINIMUM..11} {
  output [[roll X d NORMAL with willpower] minimum MINIMUM] named "normal willpower, [X] dice"
}
loop X over {MINIMUM..11} {
  output [[roll X d SPECIALTY] minimum MINIMUM] named "speciality, [X] dice"
}
loop X over {MINIMUM..11} {
  output [[roll X d SPECIALTY with willpower] minimum MINIMUM] named "speciality willpower, [X] dice"
}

function: NORM attack ROLL:n {
if ROLL = 1  {result: 0}
if ROLL >= CRIT {result: 2d{DMG_DICE} + DMG_MOD}
if ROLL+TO_HIT >= AC {result: DMG_DICE + DMG_MOD}
result: 0
}


function: GWM attack ROLL:n {
if ROLL = 1  {result: 0}
if ROLL >= CRIT {result: 2d{DMG_DICE} + DMG_MOD + 10}
if ROLL+TO_HIT-5 >= AC {result: DMG_DICE + DMG_MOD + 10}
result: 0
}


DMG_DICE: 1d10

DMG_MOD: 3
TO_HIT: 6
CRIT: 20
AC: 16

output 1d20 + 1
output [highest 1 of 2d20] + 1

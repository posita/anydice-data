\**** Lae'zel GWM ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 4
ENCHANT: 2
ADV: 0
COLOSSUS: 0
DMGBONUS: ENCHANT
PROFICIENCY: 3
CRITROLL: 20


ACCBONUS: ENCHANT - 5
DAMAGE: 1d12 + 1d4 + ATTACKMOD + DMGBONUS + 10 + COLOSSUS*1d8
CRITDAMAGE: 2d12 + 2d4 + ATTACKMOD + DMGBONUS + 10 + COLOSSUS*2d8
output [attackroll [highest 1 of (ADV+1)d20] vs ENEMYAC] named "Great Weapon Master"

ACCBONUS: ENCHANT
DAMAGE: 1d12 + 1d4 + ATTACKMOD + DMGBONUS + COLOSSUS*1d8
CRITDAMAGE: 2d12 + 2d4 + ATTACKMOD + DMGBONUS + COLOSSUS*2d8
output [attackroll [highest 1 of (ADV+1)d20] vs ENEMYAC] named "No GWM"

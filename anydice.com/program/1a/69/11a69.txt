function: owod roll COUNT:n dice target TARGET:n specialized SPECIALIZED:n willpower WILLPOWER:n {
  THRESHOLD:0
  if TARGET > 9 {
    THRESHOLD: TARGET-9
  }
  RELABELED: {1, 2:(TARGET-2), 9:(10-TARGET), 10}
  result: [owod roll COUNT d RELABELED target TARGET specialized SPECIALIZED willpower WILLPOWER]
}

function: owod roll ROLL:s target TARGET:n specialized SPECIALIZED:n willpower WILLPOWER:n {

  MAXSUCCESS: 10
  SUPERSUCCESSROLLS: {11}
  if SPECIALIZED > 0 {
    MAXSUCCESS: 9
    SUPERSUCCESSROLLS: {10}
  }
  SUCCESSROLLS: {TARGET..MAXSUCCESS}
  BOTCHROLLS: {1}

  SUPERSUCCESSES: [count SUPERSUCCESSROLLS in ROLL]
  SUCCESSES: [count SUCCESSROLLS in ROLL]
  BOTCHES: [count BOTCHROLLS in ROLL]

  THRESHOLDSUPERSUCCESSES: SUPERSUCCESSES-THRESHOLD
  THRESHOLDSUCCESSES: SUCCESSES
  THRESHOLDWILLPOWER: WILLPOWER

  if THRESHOLDSUPERSUCCESSES < 0 {
    THRESHOLDSUCCESSES: SUCCESSES+THRESHOLDSUPERSUCCESSES

    if THRESHOLDSUCCESSES < 0 {
      THRESHOLDWILLPOWER: WILLPOWER+THRESHOLDSUCCESSES

      if THRESHOLDWILLPOWER < 0 {
        THRESHOLDWILLPOWER: 0
        }

      THRESHOLDSUCCESSES: 0
      }

    THRESHOLDSUPERSUCCESSES: 0
    }

  FINALTHRESHOLD: THRESHOLDSUPERSUCCESSES+THRESHOLDSUCCESSES+THRESHOLDWILLPOWER

  if (FINALTHRESHOLD = 0) & (BOTCHES > 0) {
    result: -1
  }

  else {
    BOTCHESSUPERSUCCESSES: THRESHOLDSUPERSUCCESSES-BOTCHES
    BOTCHESSUCCESSES: THRESHOLDSUCCESSES
    BOTCHESWILLPOWER: THRESHOLDWILLPOWER

    if BOTCHESSUPERSUCCESSES < 0 {
      BOTCHESSUCCESSES: THRESHOLDSUCCESSES+BOTCHESSUPERSUCCESSES

      if BOTCHESSUCCESSES < 0 {
        BOTCHESWILLPOWER: THRESHOLDWILLPOWER+BOTCHESSUCCESSES
      }
    }

  FINALSUCCESSES: BOTCHESSUCCESSES+(BOTCHESSUPERSUCCESSES*(1+SPECIALIZED))+BOTCHESWILLPOWER

  if FINALSUCCESSES < 0 {
    result: 0
  }
  result: FINALSUCCESSES}

}

output [owod roll 4 dice target 6 specialized 1 willpower 0]

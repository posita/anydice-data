function: craftcheck body A mind B mixed C proficiency D bonus E {
if C=1  {result: 1d20 + E + D }
if C>1 & C<=9  {result: 1d20 + E + D + A + B/2 }
if C<20 & C>=10 {result: 1d20 + E + D + A/2 + B }
if C=20 {result: 1d20 + E + D + A + B }
}

loop N over {1,2,11,20}{
output [craftcheck body 4 mind 2 mixed N proficiency 4 bonus 1]
}
output [craftcheck body 4 mind 0 mixed 20 proficiency 4 bonus 1] named "raw body"
output [craftcheck body 0 mind 3 mixed 20 proficiency 4 bonus 1] named "raw mind"

function: explode ROLL:s boost BOOST:n mod MOD:n limit LIMIT:n {
  X: [lowest of [count 8 in ROLL] and LIMIT]
  if X = 0 {
    result: ROLL + BOOST * MOD
  } else {
    result: ROLL + [explode Xd8 boost BOOST mod MOD limit LIMIT - X] + MOD
  }
}

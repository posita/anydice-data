\ Refined Mirror Function \
\ ROLL: the raw d20 result \
\ BASE: the number to subtract the raw roll from \
function: mirror mechanic ROLL:n base BASE:n {
    ROLLED: ROLL - 1
    MIRROR: BASE - ROLL
    
    if ROLLED > MIRROR { result: ROLLED }
    result: MIRROR
}

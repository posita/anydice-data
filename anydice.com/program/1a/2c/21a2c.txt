
SKILLMIN: 0    \ skill. modifies custom d10 die \
SKILLMAX: 3
RATINGMIN: 1   \ rating. number of dice in the pool \
RATINGMAX: 5

loop SKILL over {SKILLMIN..SKILLMAX}{
 \ create a custom d10 where every number above 3+skill is a zero instead \
 DIE: {1..(SKILL+3),0:(7-SKILL)}
 
 loop RATING over {RATINGMIN..RATINGMAX}{
  output RATING d DIE named "[RATING]d, skill [SKILL]"
 }

}

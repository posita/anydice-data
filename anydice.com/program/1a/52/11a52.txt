function: X:n rif Y:n aim Z {

if X = 1 | Y = 1 { result: 0 }

else { result: (X+Y)*Z }

}

function: X:n pist Y:n {

if X = 1 | Y = 1 { result: 0 }

else { result: X*Y }

}

function: rifledamage X {
    result: [1d10 rif 1d10 aim X]
}

function: pistoldamage {
    result: [1d4 pist 1d10]
}

output [rifledamage 6] named "Rifle!"
output [pistoldamage] named "Pistol!"

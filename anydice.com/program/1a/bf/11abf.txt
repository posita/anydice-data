function: plusone ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE  }
 result: 0
}

ENEMYAC: d{12..20}

ATTACKMOD: 7
PROFICIENCY: 6
CRITROLL: 20
ATTACKTIMES: 2
DAMAGEBONUS: 6
ACCBONUS: 2

DAMAGE: 2d6 + ATTACKMOD + DAMAGEBONUS
CRITDAMAGE: 8d6 + ATTACKMOD + DAMAGEBONUS


function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}


output 2d[reroll {1,2} on d6] >= 6
output 2d6 >=6
output 2d[reroll {6} on d6] >= 6
output 2d[reroll {1,2} on d6] >= 9
output 2d6 >=9
output 2d[reroll {6} on d6] >= 9

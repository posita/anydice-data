
F:{-1,0,1}
BONUS:{0,0,1}
X:3

function: roll DICE:s {
 N:[count 1 in DICE]
 result: DICE + NdBONUS
}

function: roll DICE:s and limit bonus rolls {
 N:[count 1 in DICE]
 result: DICE + [lowest of N and X]dBONUS
}

\ 1. roll (4+X)dF, sum the result, count N 1's rolled, add NdBONUS \
output [roll (4+X)dF] named "[4+X]dF no limits"

\ 2. roll (4+X)dF, sum the result, count N 1's rolled, add (min N,X)dBONUS \
output [roll (4+X)dF and limit bonus rolls] named "[4+X]dF with max [X] bonus rolls"

\ 3. roll (4)dF, sum the result PLUS roll XdF, sum the result, count N 1's rolled, add NdBONUS \
output 4dF + [roll XdF] named "[4+X]dF only roll bonus dice from X dice"

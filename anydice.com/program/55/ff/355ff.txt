function: min LOW:n {
  result: (1d6>=LOW)
}

function: min LOW:n on DIE:n {
  if DIE >= LOW { result: 1 }
  result: 0
}

function: ceaseless ROLL:n {
  if ROLL = 1 {
  result: 1d6
  }
  result: ROLL
}

function: relentless ROLL:n min MIN:n {
  if ROLL < MIN {
  result: 1d6
  }
  result: ROLL
}

function: damage ROLL:n NORMAL:n CRIT:n hitting on MIN:n {
  if ROLL = 6 {
    result: CRIT
  }
  if ROLL >= MIN {
    result: NORMAL
  }
  result: 0
}

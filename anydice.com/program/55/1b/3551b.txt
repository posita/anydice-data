function: degree of ROLL:n against TN:n {
  if ROLL <= 4 | (ROLL = 5 & TN >= 15) | (ROLL = 6 & TN >= 16) {
    result: 1
  }
  if ROLL = 18 | (ROLL = 17 & TN <= 15) | (ROLL >= TN + 10) {
    result: 4
  }
  if ROLL <= TN {
    result: 2
  }
  if ROLL > TN {
    result: 3
  }
}

loop TN over {5..16} {
  output [degree of 3d6 against TN] named "When TN = [TN]: Chance for each Degree of Success/Failure"
}
output 3d6 named "General roll result"

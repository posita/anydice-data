function: roll with B boons {
  result: 2d6+[highest 1 of Bd6]
}

function: roll with B banes {
  result: 2d6-[highest 1 of Bd6]
}

loop P over {0..10} {
 X: [roll with P boons]
 output X >= 18 named "Roll with [P] boons: Critical success!"
 output X >= 10 & X < 18 named "Roll with [P] boons: Complete success."
 output X >= 5 & X < 10 named "Roll with [P] boons: Complicated success."
 output X > 0 & X < 5 named "Roll with [P] boons: Failure..."
 output X <= 0 named "Roll with [P] boons: Fumble...!"
}

loop P over {0..10} {
 X: [roll with P banes]
 output X >= 18 named "Roll with [P] banes: Critical success!"
 output X >= 10 & X < 18 named "Roll with [P] banes: Complete success."
 output X >= 5 & X < 10 named "Roll with [P] banes: Complicated success."
 output X > 0 & X < 5 named "Roll with [P] banes: Failure..."
 output X <= 0 named "Roll with [P] banes: Fumble...!"
}

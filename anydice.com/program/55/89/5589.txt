function: ROLL:n vs AC:n x DMG:n {
 if ROLL >= THREAT & ROLL + ATTACK >= AC {
  result: [threat d20]
 }
 if ROLL + ATTACK >= AC | ROLL = 20 { result: DAMAGE * DMG }
 result: 0
}

function: threat ROLL:n {
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: CRITICAL d DAMAGE
 }
 result: DAMAGE
}

AC: 28
STRENGTH: 17

CRITICAL: 3
DMG_DICE: 6d6

DAMAGE: DMG_DICE + 17
ATTACK: 23

output [d20 vs AC x 2] + [d20 vs AC x 1] + [d20 - 5 vs AC x 1] + [d20 - 10 vs AC x 1] + [d20 - 15 vs AC x 1] named "Deadly Aim"

DAMAGE: DMG_DICE + 7
ATTACK: 23 + 5

output [d20 vs AC x 2] + [d20 vs AC x 1] + [d20 - 5 vs AC x 1] + [d20 - 10 vs AC x 1] + [d20 - 15 vs AC x 1] named "Full Attack"

function: virtue A:n vice E:n threshold T:n depth D:n hold H:n {
  S: [absolute A-E]
  RA: 0
  RE: 0
  if A>T & E!=1 {RE: T-E}
  if E<T & A!=6 {RA: A-T}
  if !(RA|RE) | D=0 | S>=H {result: S}
  if RA>RE {result: [virtue d6 vice E threshold T-1 depth D-1 hold H]}
  else {result: [virtue A vice d6 threshold T+1 depth D-1 hold H]}
}



loop T over {1..6}{
  loop H over {1..5}{
    output [virtue d6 vice d6 threshold T depth 1 hold H] named "threshold [T], hold at [H]"
  }
}

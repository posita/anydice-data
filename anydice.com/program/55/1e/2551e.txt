function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: boost LIMIT:s above ALLOWED:n on ROLL:n by BONUS:n {
    X: [highest of ([count LIMIT in ROLL] - ALLOWED) and 0]
    result: ROLL + BONUS * X
}

output 5d4+5 named "Med-Low Dynastic"

output 5d[reroll {1} on d4 to d4]+2 named "Non-Dynastic Reroll 1s once"

output [boost {1} above 1 on 5d[reroll {1} on d4 to d4] by 1]+2 named "First Ruler"

output 5d3+7 named "Non-Dynastic Reroll 1s always"

output [highest 5 of 7d4]+2 named "Non-Dynastic roll 7 drop 2"

output [highest 5 of 8d4]+2 named "Non-Dynastic roll 8 drop 2"

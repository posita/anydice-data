function: event ROLL:n plus STAT:n vs TN:n for BS:n  {
    if ROLL = 1 {
        result: 0
    }
    
    if ROLL + STAT >= TN {
        result: BS
    }
    result: 0
}

DECK: d{1:34,2:18}
PIPCARD: 3d{2..10}
FACECARD:([event d10 plus 5 vs 10 for 1]*30)

function: explore DRAW:n   {
    if DRAW = 1 {
        result: PIPCARD
    }
    
    if DRAW = 2 {
        result: FACECARD
    }
    result: 0
}

output 4d([explore DECK])

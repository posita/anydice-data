function: virtue A:n vice E:n threshold T:n depth D:n {
  S: [absolute A-E]
  RA: 0
  RE: 0
  if A>T & E!=1 {RE: T-E}
  if E<T & A!=6 {RA: A-T}
  if !(RA|RE) | D=0 {result: S}
  if RA>RE {result: [virtue d6 vice E threshold T-1 depth D-1]} 
  else {result: [virtue A vice d6 threshold T+1 depth D-1]}
}

loop D over {2..4}{
  loop T over {2..5}{
    output [virtue d6 vice d6 threshold T depth D-1] named "threshold [T], [D] roll(s) deep"
  }
}

\ Empowered Spell Calculator \

NUM_DICE: 4
DIE_SIZE: 8    \ Damage = 4d8 \
CHA_MOD:  5    \ +5 Charisma  \



DAMAGE: NUM_DICEdDIE_SIZE
TOTAL_DAMAGE: 0

function: reroll R:n {
  if R > DIE_SIZE/2 { result: R }
  result: 1dDIE_SIZE
}

loop X over {1..NUM_DICE} {
  if X <= NUM_DICE - CHA_MOD {
    TOTAL_DAMAGE: TOTAL_DAMAGE + X @ DAMAGE
  }
  else {
    TOTAL_DAMAGE: TOTAL_DAMAGE + [reroll X @ DAMAGE]
  }
}

output DAMAGE named "Normal Damage"
output TOTAL_DAMAGE named "Empowered Spell"

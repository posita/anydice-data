STAT: [highest 3 of 4d6]

function: average STATS:s {
  RES: 0
  loop N over {1..#STATS} {
    RES: RES + N@STATS
  }
  result: RES/#STATS
}

function: reroll DICE:s {
  if 2@DICE<15 {
    result: [reroll #DICEdSTAT]
  }
  result: [average DICE]
}

output [average 6dSTAT]
output [average [highest 6 of 7dSTAT]]

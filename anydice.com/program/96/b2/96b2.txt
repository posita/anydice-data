function: modulo B of A{
X: A/B
result: A-(X*B)
}

function: matchcrit A:s and B:s and C:s{
if A > B{
if A > C{X:A}
else{X:C}
Y:B
}
else{
if B>C{X:B}
else{X:C}
Y:A
}
if X=Y{
if [modulo 2 of X]=0 {result:30}
if [modulo 2 of X]=1 {result:-1} 
}
else{result: X+Y}
}

output [matchcrit 1d10 and 1d10 and 1d10]

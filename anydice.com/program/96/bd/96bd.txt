MISS: 0
MISSDAMAGE: 0 
HIT: 1
CRITICALHIT: 2
ATTACK: 7



function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {result: CRITICALHIT}
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]
}

DEFENSE: 15

HITDAMAGE: d12 + 5
CRITICALDAMAGE: 3d12 + 5
output 3d[damage for [attack [highest 1 of 1d20] vs DEFENSE]] named "Great Axe(Normal)"

output 3d[damage for [attack [lowest 1 of 2d20] vs DEFENSE]] named "Great Axe(Disadvantage)"

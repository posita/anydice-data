SHOWASSASSINATE: 0    \ Output the calculations for Tazyl's assassinate round \
SHOWSUCCESSCHANCES: 0 \ Show the chance of success of rolling any TN using advantage, disadvantage, lucky \

ATK:  1d20  \ attack roll: use 1@2d20 for advantage or 2@2d20 for disadvantage \
HIT:  +12    \ bonus to attack roll \
AC:   17    \ target AC: attack hits if ATK + HIT >= AC \
CRIT:   20  \ minimum attack roll to crit \
DMG_DICE:  2d6+1d6  \ damage dice rolled on successful hit \
DMG_BONUS: 6 \ extra damage added on successful hit - not multiplied by crit \ 
BONUS: 2d6+1d8  \ bonus damage on first hit \

function: attack ROLL:n {
  \ return the number of times damage is applied on an attack roll of ROLL \
  if ROLL <= 1 { result: 0 }     \ natural 1 is always a miss \
  if ROLL >= 20 | ((ROLL + HIT >= AC) & (ROLL >= CRIT)) { result: 2 }  \ critical hit deals damage damage twice \
  result: ROLL + HIT >= AC
}

function: multiattack N:n times {
  result: [roll [attack ATK] to multiattack N times]
}
function: roll MULT:n DICE:d damage dice plus EXTRA:n {
  if MULT <= 0 {
    result: 0
  }
  result: (MULT d DICE) + EXTRA
}
function: roll FIRSTHIT:n to multiattack N:n times {
  if N <= 0 {
    result: 0
  }
  result: [roll FIRSTHIT (DMG_DICE+BONUS) damage dice plus DMG_BONUS] + (N-1) d [roll [attack ATK] DMG_DICE damage dice plus DMG_BONUS]  
}


\ output [multiattack 1 times] named "multiattack"\

BASEDMG: 5+2+1
NUM_ATTACKS: 4

ATK: 1@2d20
BONUS: 2d6+1d8

 output 2@2d20 named "disadvantage"

 SSBETTER: SS>NORMAL

LEVELS: {2, 2, 2, 2, 2}
PARTY_DICE: {8, 10, 8, 6, 12}
PARTY_BONUS: {2, 3, 2, -1, 5}

loop N over {1..#PARTY_DICE} {
  LEVEL: N@LEVELS
  HITDIE: N@PARTY_DICE
  BONUS: N@PARTY_BONUS
  output HITDIE+BONUS+(LEVEL-1)d([highest of 1 and (1dHITDIE + BONUS)]) named "Party #[N]: Lv.[LEVEL] d[HITDIE]+[BONUS] Rolled"
  output HITDIE+BONUS+(LEVEL-1)*([highest of 1 and ((HITDIE/2+1) + BONUS)]) named "Party #[N]: Lv.[LEVEL] d[HITDIE]+[BONUS] Average"
}

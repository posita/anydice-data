function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

output 1d10 named "1d12"

output [reroll {1,2,3,4,5} on 1d12 to 6] named "1d12 hi"

output 2d10 named "2d10" 

output [reroll {1,2,3,4} on 1d10 to 5]+[reroll {1,2,3,4} on 1d10 to 5] named "2d10 hi"

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
if ROLL >= CRITROLL { result: CRITDAMAGE }
if ROLL = 1 { result: 0 }
if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

ATTACKMOD: 7
DMGBONUS: 4+3+10
PROFICIENCY: 6
ACCBONUS: -2
CRITROLL: 20
ATTACKTIMES: 3
DAMAGE: 1d12 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 6d12 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "Ykra lvl w GWM 3+ weapon"

function: reroll R:n under N:n this D {
   if R < N { result: D } else {result: R}
}

function: hit H:n mod M crit C dmg D bonus B ac A {
   if H >= C {result: D+D+B}
   else if H+M > A {result: D+B} 
   else {result: 0}
}

GWS: [reroll 1d6 under 3 this d6]

RA: [hit d20 mod 9 crit 20 dmg d8+d6 bonus 4 ac 16]
RT: [hit d20 mod 7 crit 20 dmg d8+d8 bonus 4 ac 16]
RD: [hit d20 mod 7 crit 20 dmg d8+d6 bonus 6 ac 16]

output RA+RA named "Ranger Archer"
output RT+RT named "Ranger Two-Weap (2 attacks)"
output RT+RT+RT named "Ranger Two-Weap (3 attacks)"
output RD+RD named "Ranger Duelist"

function: reroll R:n under N:n this D {
   if R < N { result: D } else {result: R}
}

function: hit H:n mod M crit C dmg D bonus B ac A {
   if H >= C {result: D+D+B}
   else if H+M > A {result: D+B} 
   else {result: 0}
}

GWS: [reroll 1d6 under 3 this d6]

GSC: [hit d20 mod 7 crit 19 dmg GWS+GWS bonus 4 ac 16]
GWB: [hit d20 mod 4 crit 19 dmg d8 bonus 1 ac 16]
TW: [hit d20 mod 7 crit 19 dmg d6 bonus 4 ac 16]
TWB: [hit d20 mod 7 crit 19 dmg d8 bonus 4 ac 16]


output GSC+GSC named "GW fighter: Greatsword"
output TW+TW+TW named "Two-weapon fighter: Short sword"

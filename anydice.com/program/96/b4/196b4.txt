function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

output 1d8 named "1d8"

output [reroll {1,2,3} on 1d8 to 4] named "1d8 hi"

output [reroll {1,2} on 1d6 to 3]+[reroll {1,2} on 1d6 to 3] named "2d6 hi"

output 2d6 named "2d6"

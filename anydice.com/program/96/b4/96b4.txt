function: modulo B of A{
X: A/B
result: A-(X*B)
}

function: matchcrit A:s and B:s and C:s{
if A=B & B=C &[modulo 2 of A] = 1{result: 1}

if A < B{
if A > C{X:A}
else{X:C}
Y:B
}
else{
if B>C{X:B}
else{X:C}
Y:A
}
if X=Y & [modulo 2 of X]=0 {result:20}
else{result: X+Y}
}



function: matchcritnorm A:s and B:s{
X: A
Y: B
if X=Y{
if [modulo 2 of X]=0 {result:20}
if [modulo 2 of X]=1 {result:1} 
}
else{result: X+Y}
}
output d20 named "normal d20 roll"
output 1@2d20 named "d20 with advantage"

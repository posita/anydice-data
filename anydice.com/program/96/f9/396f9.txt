function: ROLL:n vs AC:n {
 if ROLL = 20 {
  result: [threat d20]
 }
 if ROLL + ATTACK >= AC | ROLL = 20 { result: DAMAGE }
 result: 0
}

function: threat ROLL:n {
 if ROLL + ATTACK >= AC | ROLL = 20 {
  result: 2dDAMAGE + STRENGTH
 }
 result: DAMAGE
}

AC: d{16}
STRENGTH: 2
ATTACK: 2 + STRENGTH

DAMAGE: 1d6 
output 10d[d20 vs AC] named "Ten Goblins"

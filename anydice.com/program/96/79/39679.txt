\DAMAGE MODIFIERS AND COMBINATIONS\
DMGMOD: 4
PROFBONUS: 3
ATKMOD: DMGMOD + PROFBONUS
MISSDMG: 0 \This allows support for the "Graze" Weapon mastery property. Simply change 0 to the appropriate modifier\
WEAPDMG_A: 1d6 
WEAPDMG_B: 1d6  
CRIT_RANGE: {20} \If Crit on 19-20, change to "{19,20}"\

\Damage Modifiers for things that add bonus damage to EACH HIT [i.e. Hunter's Mark, Hex, Magic Weapons, etc.]\
BONUSDMG_A: 0
BONUSDMG_B: 0

MISS:0  NO:0
HIT:1   YES:1 
CRIT:20


ADVANTAGE: NO \CASE SENSITIVE. "YES" OR "NO"\


AC: 18
\MAKE SURE YOU SET YOUR AC TARGETS CORRECTLY!
"d{N1..N2}" will provide output for the AVERAGE Hit chance and Damage over a range of AC targets.
"d{N1}" or "N1" will provide output for a single AC target.
This can also be looped over a sequence to provide individual results for multiple AC targets\
 

\ATTACK ROLL SECTION\
function: attack ROLL:n vs AC:n {
 if ROLL >= CRIT_RANGE {result: CRIT}
 if ROLL + ATKMOD >= AC {result: HIT}
 if ROLL + ATKMOD < AC {result: MISS}
}
if ADVANTAGE = YES {
  output [highest of [attack d20 vs AC] and [attack d20 vs AC]] named "Hit chance with advantage vs target AC"
}
else if ADVANTAGE = NO {
 output [attack d20 vs AC] named "Hit chance without Advantage vs target AC "
}


\DAMAGE SECTION\
function: damage for ATTACKRESULT:s {
D: 0
 loop A over ATTACKRESULT {
  if  A = CRIT {D: 2dWEAPDMG_A + DMGMOD}
  if  A = HIT {D: WEAPDMG_A + DMGMOD}
  else if A = MISS {D: MISSDMG}
  if !A = MISS {D: D + [bonusdmg for ATTACKRESULT]} 
}
result: D
}
function: bonusdmg for ATTACKRESULT:n {
  result: BONUSDMG_A + BONUSDMG_B 
}
if ADVANTAGE = NO {
 function: attack {
  result:  [damage for [attack d20 vs AC]]
 }
 function: N:n attack {
  result: [damage for Nd[attack d20 vs AC]] 
 }
}
if ADVANTAGE = YES {
 function: attack {
  result: [damage for [highest of [attack d20 vs AC] and [attack d20 vs AC]]]
 }
 function: N:n attack {
  result: [damage for Nd[highest of [attack d20 vs AC] and [attack d20 vs AC]]]
 }
}


\OUTPUT SECTION: INPUT OUTPUT FUNCTION BELOW: "output Nd[attack] named "______""\

output  d[attack] named "Weapon attack x1"
output  2d[attack] named "Weapon attack x2"
output  3d[attack] named "Weapon attack x3"
output  4d[attack] named "TWF/Nick + FoB"

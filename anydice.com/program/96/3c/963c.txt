function: A:n mod N:n {
  result: A - (N * (A/N))
}

function: montyhall DOOR:n PICK:n SWITCH:n {
  if SWITCH = 1 { result: PICK = DOOR }
  result: ([(PICK +1) mod 3] + 1 = DOOR) | ([(PICK +2) mod 3] +1 = DOOR)
}

output [montyhall 1d3 1d3 1] named "No switch"
output [montyhall 1d3 1d3 1d2] named "Switch randomly"
output [montyhall 1d3 1d3 2] named "Always switch"

function: oklessee SWITCH:n D:n {
  if SWITCH = 1 { result: D = 1 }
  result: (D = 2) | (D = 3)
}

output [oklessee 1d2 1d3]

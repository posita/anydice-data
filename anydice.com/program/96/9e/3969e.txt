function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

MISS: 0
HIT: 1
CRITICALHIT: 2

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attackdamage {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{10..20}

ATTACK: 0
HITDAMAGE: 2d6 + 13
CRITICALDAMAGE: 4d6+13
MISSDAMAGE: 0
output [attackdamage] named "2014 Level 4"

ATTACK: 6
HITDAMAGE: 2d6+5
CRITICALDAMAGE: 4d6+5
output [attackdamage] named "2024 Level 4"

BOTCH: -1
MISS: 0
HIT: d8
CRITICALHIT: [maximum of d8]
ROLL: d100
STAT: 40
STATBONUS: d20 + STAT

function: attack ROLL:n vs STAT:n {
 if ROLL = 100 { result: BOTCH }
 if ROLL = 1 {
  result: CRITICALHIT
 }
 if ROLL <= STAT { result: HIT }
 result: MISS
}

function: sacrifice ROLL:n vs STATBONUS:n {
 if ROLL = 100 { result: BOTCH }
 if ROLL = 1 {
  result: CRITICALHIT
 }
 if ROLL <= STATBONUS { result: HIT }
 result: MISS
}

output [attack ROLL vs STAT] named "attack damage roll at [STAT]"

output [sacrifice ROLL vs STATBONUS] named "sacrifice damage roll at [STAT]"

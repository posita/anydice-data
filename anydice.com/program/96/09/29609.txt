function: successes in DICE:s {
  SUC: 0
  MAXVAL: 0
  SIXES: 0
  loop I over {1..#DICE} {
    if (I@DICE > MAXVAL) {MAXVAL: I@DICE}
    if (I@DICE = 6) {SIXES: SIXES + 1}
  }
  
  if (SIXES > 1) {MAXVAL: 7}

  result: MAXVAL
}

function: RES:n graded {
  GRAD: 0
  
  if (RES = 7) {GRAD: 3}
  if (RES = 6) {GRAD: 2}
  if (RES < 6) {GRAD: 1}
  if (RES < 4) {GRAD: 0}

  result: GRAD
}

loop X over {1..6} {
  output [[successes in Xd6] graded] named "[X]d6"
}

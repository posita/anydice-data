function: interpret X:n with CALAMITY:n {
  if X = CALAMITY { result: 0 }
  else { result: X }
}

function: getdice X:n with CALAMITY:n {
  if X > 3 { result: [ highest 4 of Xd[interpret 1d6 with CALAMITY] ] }
  if X = 3 { result: 3d[interpret 1d6 with CALAMITY] + 1d[interpret 1d4 with CALAMITY] }
  if X = 2 { result: 2d[interpret 1d6 with CALAMITY] + 2d[interpret 1d4 with CALAMITY] }
  if X = 1 { result: 1d[interpret 1d6 with CALAMITY] + 3d[interpret 1d4 with CALAMITY] }
  result: 4d[interpret 1d4 with CALAMITY]
}

function: roll X:n {
  CALAMITY: 1d12
  result: [getdice X with CALAMITY]
}

loop DICE over {0..10} {
 output [roll DICE] named "[DICE] Dice"
}

\function: compare A:s vs B:s{
X: A > B
if X = 1{result: A}
else {result: B}
}\

function: modulo B of A{
X: A/B
result: A-(X*B)
}

function: matchcrit A:s and B:s{
X: A
Y: B
if X=Y{
if [modulo 2 of X]=0 {result:20}
if [modulo 2 of X]=1 {result:1} 
}
else{result: X+Y}
}

output [matchcrit 1d10 and 1d10]

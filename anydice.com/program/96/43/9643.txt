N: 4
ROLL: Nd10

function: ore ROLL:s {
 SET: 0
 SETNUMBER: 0
 loop NUMBER over {1..10} {
  if [count NUMBER in ROLL] > SET & [count NUMBER in ROLL] >= 1{
   SET: [count NUMBER in ROLL]
   SETNUMBER: NUMBER}
  else if [count NUMBER in ROLL] = SET & NUMBER > SETNUMBER {
   SETNUMBER: NUMBER}
 }
 if SETNUMBER = 10 {
  SET: SET*10
  SETNUMBER: 0}
 else {
  SET: SET*10}
 if SET+SETNUMBER < 20{
  result: 0}
 result: SET + SETNUMBER
}

output [ore 4d10] named "ORE w/ 4 dice"

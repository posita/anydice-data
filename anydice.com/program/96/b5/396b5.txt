function: attack ROLLS:s plus MOD:n against AC:n duelcrit DUELCRIT:d crit CRIT:d duelhit DUELHIT:d hit HIT:d add ADD:n {
    HIGHEST: (1 @ ROLLS)
    ATK: HIGHEST + MOD
    if HIGHEST >= 19{ \ crit \
        if ATK >= (AC + 5){ result: DUELCRIT + ADD }
        else { result: CRIT + ADD }
    }
    else {
        if ATK >= AC {
            if ATK >= (AC+5) { result: DUELHIT + ADD }
            else { result: HIT + ADD }
        }
        else {
            \ if ATK > (AC - 10) { result: 1d8 } \
            \ else { \
                result: 0
            \ } \
        }
    }
}

function: reroll lowest of ROLL:s as REROLL:d if less than MIN:n {
  if ((#ROLL)@ROLL) >= MIN {
     result: ROLL + 0
  } else {
    result: (({1..#ROLL-1}@ROLL) + REROLL)
  }
}

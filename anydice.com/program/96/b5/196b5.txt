function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

output 1d10 named "1d10"

output [reroll {1,2,3,4} on 1d10 to 5] named "1d10 hi"

output 3d6 named "3d6" 

output [reroll {1,2} on 1d6 to 3]+[reroll {1,2} on 1d6 to 3]+[reroll {1,2} on 1d6 to 3] named "3d6 hi"

F: d{-1..1}

function: flip FN:n up for DICE:s {
  SORTED_DICE: [reverse [sort DICE]]
  TOTAL: 0
  loop N over {1..FN} {
    TOTAL: TOTAL + [absolute N@SORTED_DICE]
  }
  loop N over {FN + 1..#SORTED_DICE} {
    TOTAL: TOTAL + N@SORTED_DICE
  }
  result: TOTAL
}

function: flip FN:n down for DICE:s {
  SORTED_DICE: [sort DICE]
  TOTAL: 0
  loop N over {1..FN} {
    TOTAL: TOTAL - [absolute N@SORTED_DICE]
  }
  loop N over {FN + 1..#SORTED_DICE} {
    TOTAL: TOTAL + N@SORTED_DICE
  }
  result: TOTAL
}

function: ignore FN:n minuses for DICE:s {
  SORTED_DICE: [reverse [sort DICE]]
  TOTAL: 0
  loop N over {1..FN} {
    if N@SORTED_DICE > 0 {
      TOTAL: TOTAL + [absolute N@SORTED_DICE]
    }
  }
  loop N over {FN + 1..#SORTED_DICE} {
    TOTAL: TOTAL + N@SORTED_DICE
  }
  result: TOTAL
}

function: ignore FN:n pluses for DICE:s {
  SORTED_DICE: [sort DICE]
  TOTAL: 0
  loop N over {1..FN} {
    if N@SORTED_DICE < 0 {
      TOTAL: TOTAL + N@SORTED_DICE
    }
  }
  loop N over {(FN + 1)..#SORTED_DICE} {
    TOTAL: TOTAL + N@SORTED_DICE
  }
  result: TOTAL
}

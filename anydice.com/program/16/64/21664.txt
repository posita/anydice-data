\ TFC Gambling \

DC: 5 + 2d10   \ fixed \
SONE: 1d20+6   \ insight \
STWO: 1d20+6   \ sleight of hand \
STHREE: 1d20   \ intimidation/deception \
CHECK: SONE >= DC & STWO >= DC & STHREE >= DC

function: profit SUCC:n {
  if SUCC = 0 { result: 0 }
  if SUCC = 1 { result: 100 }
  if SUCC = 2 { result: 175 }
  if SUCC = 3 { result: 250 }
  result: -1
}

output [profit CHECK + CHECK + CHECK]

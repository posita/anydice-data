function: farthest from middle ROLL:s {
  HI: 1@ROLL
  MID: (#ROLL/2+1)@ROLL
  LO: #ROLL@ROLL
  if HI - MID > MID - LO { result: HI }
  if HI - MID < MID - LO { result: LO }
  result: d{HI, LO} \ break ties by coin flip \
}

function: coin flip low high ROLL:s {
  HI: 1@ROLL
  LO: #ROLL@ROLL
  result: d{HI, LO}
}

loop N over {3,5} {
  output [farthest from middle Nd20] named "roll [N]d20, take farthest from middle"
  output [coin flip low high Nd20] named "roll [N]d20, flip coin to choose lowest or highest"
}

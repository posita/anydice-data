set "explode depth" to 10

RANGE: 40

DTIE: d{0..1}
ZFIVE: d{0..5}

function: clamp N:n between MIN:n and MAX:n {
  if N > MAX { result: MAX }
  if N < MIN { result: MIN }
  result: N
}

function: extend N:n {
  if N = 1 { result: 1 - [explode ZFIVE] }
  if N = 6 { result: 6 + [explode ZFIVE] }
  result: N
}

\ Opposed roll \
output [clamp [extend d6]-[extend d6]-DTIE between -RANGE and RANGE] named "d6X vs"

\ Static action \
output [clamp [extend d6]-4 between -RANGE and RANGE] named "d6X-4"

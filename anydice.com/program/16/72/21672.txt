function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output 2d8 named "Level 1 Divine Smite"
output 3d8 named "Level 2 Divine Smite"
output 4d8 named "Level 3 Divine Smite"
output 5d8 named "Level 4 Divine Smite"

\ how many points you get \
POINTS: {-1, 0, 0, 1, 1, 2}

\ how many points you get on explosions \
EXPLODEPOINTS: {0, 0, 0, 1, 1, 2}

\ which d6 values explode \
EXPLODES: {0, 0, 0, 1, 1, 0}

\ maximum rolls per die, to eliminate ~0% tail \
set "maximum function depth" to 2

\ graph tail cutoff, to eliminate ~0% tail \
CUTOFF: 20

function: score N:n {
 if N@EXPLODES { result: N@POINTS + [explode d6] }
 result: N@POINTS
}

function: explode N:n {
 if N@EXPLODES { result: N@EXPLODEPOINTS + [explode d6] }
 result: N@EXPLODEPOINTS
}

DIE: [score d6]

loop N over {1..8} {
 output [lowest of CUTOFF and NdDIE] named "[N]d"
}

function: die A:s armor B:n pen C:n {
 S:0
 TA: B - C
 if (TA < 0){
  TA: 0
 }
 loop N over A{
   if (N > TA) {
      S : S + (N - TA)
   }
 }
  result: S
}

output [die 1d6  armor 3 pen 0] named "Example"

output [die 1d6  armor 10 pen 0] named "Test Armor > Damage"

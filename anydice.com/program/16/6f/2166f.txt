function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output 1d6 named "Level 1 Sword Burst"
output 2d6 named "Level 5 Sword Burst"
output 3d6 named "Level 11 Sword Burst"
output 4d6 named "Level 17 Sword Burst"

function: diminish explode DIE:d {
  MAX: [maximum of DIE]
  result: [diminish explode DIE helper]
}

function: diminish explode N:n helper {
  if N = MAX
  {
    if N = 12
    {
      MAX: 10
      result: N + [diminish explode 1d10 helper]
    }
    if N = 10
    {
      MAX: 8
      result: N + [diminish explode 1d8 helper]
    }
    if N = 8
    {
      MAX: 6
      result: N + [diminish explode 1d6 helper]
    }
    if N = 6
    {
      MAX: 4
      result: N + [diminish explode 1d4 helper]
    }
    if N = 4
    {
      result: N
    }
  }
  else
  {
    result: N
  }
}

output [diminish explode 1d8]

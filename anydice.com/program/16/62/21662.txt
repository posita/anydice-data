function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output 8d6 named "Level 1 Lightning Bolt and Fireball"
output 8d[reroll {1,2,3} on d6] named "EMPOWERED Level 1 Lightning Bolt and Fireball"

function: dnd D:n {
  if D=1  {result:-50}
  if D=20 {result:50}
  result: D
}
function: gtb A:n a B:n a C:n{
  if A=1 {result:1}
  if B>C {result:1}
  result: 0
}
function: res A:n a B:n a C:n a D:n{
  if B=0 {result: A+C*(D+1)}
  result: A
}
M:0
S:0
U:2
loop I over {1..U}
{ 
  S:S-I
  X:[dnd d20]
  X:X+M
  B:0
  N:U-I+1
  loop J over {1..I}
  {
      Y:[dnd d20]
      B:[gtb B a Y a X]
  }
  S:[res S a B a I a N]
}
output S named "res"

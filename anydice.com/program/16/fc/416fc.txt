function: reroll REROLLS:s on ROLL:n to REPLACEMENT:d {
  if [REROLL contains ROLL] {
    result: REPLACEMENT
  }
  result: ROLL
}

function: reroll REROLL:s on DICE:d {
  result: [reroll REROLL on DICE to DICE]
}


loop N over { 2, 3, 4, 5, 6 } {
  X: d6 >= 5
  output NdX named "[N]d6"
  output [reroll 0 on NdX] named "[N]d6 rerolled"
}


\ ones function returns the digit in the ones place of a number \
function: ones NUM:n {
 R: NUM/10
 Z: R*10
 result: NUM-Z
}

\ the sumdigits function adds the individual digits of a number together \
function: sumdigits NUM:n {
 if NUM=0 {result:0}
 else {result:[ones NUM] + [sumdigits NUM/10]}
}

\ repeat the sumdigits function until the result is a single digit \
function: sumdigitsrepeat NUM:n {
 if NUM < 10 {result:NUM}
 else {result:[sumdigitsrepeat [sumdigits NUM]]}
}

output [sumdigitsrepeat 2d20]

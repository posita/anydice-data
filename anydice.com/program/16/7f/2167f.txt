function: farthest from middle ROLL:s {
  HI: 1@ROLL
  MID: 2@ROLL
  LO: 3@ROLL
  if HI - MID > MID - LO { result: HI }
  if HI - MID < MID - LO { result: LO }
  result: MID
}

function: coin flip low high ROLL:s {
  HI: 1@ROLL
  LO: 3@ROLL
  result: d{HI, LO}
}

output [farthest from middle 3d20] named "roll 3d20, take farthest from middle"
output [coin flip low high 3d20] named "roll 3d20, flip coin to choose lowest or highest"

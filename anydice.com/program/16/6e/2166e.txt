function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output 1d{4..13} named "Level 1 Eldritch (Agonizing) Blast"
output 2d{4..13} named "Level 5 Eldritch (Agonizing) Blast"
output 3d{5..14} named "Level 11 Eldritch (Agonizing) Blast"
output 4d{6..15} named "Level 17 Eldritch (Agonizing) Blast"

DIFFICULTY: 8
POOL: 8

function: one R:n {
  if R = 1 { result: -1 }
  if R >= DIFFICULTY { result: 1 }
  result: 0
}

SUM: 0
loop N over {1..POOL} {
  SUM: SUM + [one d10]
}

function: final S:n {
  NEG: S @ -1          
  POS: S @ 1          
  if NEG > 0 & POS = 0 { result: -1 }
  else {
    if S < -1 { result: -1 } else { result: S }
  }
}
output [final SUM] named "[POOL]d10"

ATTACK_BONUS:5
ATTACK_BONUS_DIE:0 \1d4 for bless\
TARGET_AC:11
ADVANTAGE:0  \-1 is disadvantage, 0 is normal, 1 is advantage, 2 is trivantage\
DAMAGE: 1
CRIT_DAMAGE:0
CRIT_RANGE:20 \20 for most people, 19/18 for champions\
MULTIATTACK:24


TOTAL:0

if ADVANTAGE=0 {ROLL:1d20}
if ADVANTAGE=1 {ROLL:1@2d20}
if ADVANTAGE=2 {ROLL:1@3d20}
if ADVANTAGE=-1 {ROLL:2@2d20}

ROLL:ROLL+ATTACK_BONUS_DIE


function: attack ROLL:n {
 if ROLL = 1 {result:0 }
 if ROLL >=CRIT_RANGE {result:DAMAGE+CRIT_DAMAGE }
 if ROLL + ATTACK_BONUS >= TARGET_AC { result:DAMAGE }
 result:0
}

function: multiattack MULTIATTACK:n {
TOTAL:0
loop N over {1..MULTIATTACK} {
 TOTAL:TOTAL+[attack ROLL]
 }
result:TOTAL
}
output [multiattack MULTIATTACK]

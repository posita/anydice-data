function: caltrop explode DIE:d {
  MAX: [maximum of DIE]
  result: [caltrop explode DIE helper]
}

function: caltrop explode N:n helper {
  if N = MAX
  {
    MAX: 4
    result: N + [caltrop explode 1d4 helper]
  }
  else
  {
    result: N
  }
}

set "maximum function depth" to 6

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output 6d6 named "Level 1 Scorching Ray (All Rays)"
output 6d[reroll {1..3} on d6] named "EMPOWERED Level 1 Scorching Ray (All Rays)"

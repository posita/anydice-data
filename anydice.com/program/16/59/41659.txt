function: diminish explode DIE:d {
  MAX: [maximum of DIE]
  result: [diminish explode DIE helper]
}

function: diminish explode N:n helper {
  if N = MAX
  {
    if N = 12
    {
      result: N + [diminish explode 1d10]
    }
    if N = 10
    {
      result: N + [diminish explode 1d8]
    }
    if N = 8
    {
      result: N + [diminish explode 1d6]
    }
    if N = 6
    {
      result: N + [diminish explode 1d4]
    }
    if N = 4
    {
      result: N
    }
  }
  else
  {
    result: N
  }
}

output [diminish explode 1d8]

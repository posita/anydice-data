TARGET_AC: 15
ADVANTAGE: 1
NUMBER_RAPTORS: 8

MISS: 0
HIT: 1
CRITICALHIT: 2
ATTACK: 4
BITEDAMAGE: d6 + 2
BITECRITICALDAMAGE: 2d6 + 2
CLAWDAMAGE: d4 + 2
CLAWCRITICALDAMAGE: 2d4 + 2
MISSDAMAGE: 0

TOTALDAMAGE: 0

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n with HITDAMAGE:n and CRITDAMAGE:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: damage for ATTACKRESULT:n with HITDAMAGE:n and CRITDAMAGE:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

loop N over {1..NUMBER_RAPTORS}{
if ADVANTAGE = 1 {
  TOTALDAMAGE: TOTALDAMAGE 
   + [damage for [attack [highest 1 of 2d20] vs TARGET_AC] with BITEDAMAGE and BITECRITICALDAMAGE] 
   + [damage for [attack [highest 1 of 2d20] vs TARGET_AC] with CLAWDAMAGE and CLAWCRITICALDAMAGE]
 } else {
  TOTALDAMAGE: TOTALDAMAGE 
  + [damage for [attack [highest 1 of 2d20] vs TARGET_AC] with BITEDAMAGE and BITECRITICALDAMAGE] 
  + [damage for [attack [highest 1 of 2d20] vs TARGET_AC] with CLAWDAMAGE and CLAWCRITICALDAMAGE]
 }
}

output TOTALDAMAGE named "All Raptors"

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output 3d{2..5} named "Level 1 Magic Missile (All Darts)"
output 3d[reroll {2,3} on d{2..5}] named "EMPOWERED Level 1 Magic Missile (All Darts)"

\
 Blades in the Dark
 0: Things go bad.
 1: Success, but with undesired consequences.
 2: Clean success.
 3: Critical success!
\

function: blades ROLL:s {
 if 1@ROLL = 8 {
  result: 2 + (2@ROLL = 8)
 }
 result: 1@ROLL >= 6
}

output [blades 2@2d8] named "0d"
loop P over {1..7} {
 output [blades Pd8] named "[P]d"
}

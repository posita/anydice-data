function: hitme X Y {
  result: X >= Y
}

function: dmg P R A {
  result: R + P - A
}

function: spray {
  result: [hitme 2d6 4] * [highest of [dmg 12 2d6 15] and 0]
} 

function: tail {
  result: [hitme 2d6 4] * [dmg 14 2d6 15]
} 

function: claw {
  result: [hitme 2d6 4] * [dmg 13 2d6 15]
} 


function: tailboost {
  result: [hitme 2d6 4] * [dmg 14 3d6 15]
} 

function: clock {
  result: [spray] + [tailboost] + [claw] + [claw] + [tail] + [tail] 
}

function: enrageclock {
  result: [spray] + [tailboost] + 2 + [claw] + 2 + [claw] + 2 + [tail] + 2 + [tail] + 2 
}


output [clock] + [enrageclock] >= 58
output [enrageclock] >= 58

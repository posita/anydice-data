\**** Damage calculator normal Vs advantage ****\
LUCKYCOUNTER: 3

function: attackroll ROLL:n vs AC:n LUCKYCOUNTER:n LUCKYD:n{
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 if LUCKYCOUNTER > 0 & LUCKYD > 0 {
  LUCKYCOUNTER: LUCKYCOUNTER - 1
  result: [attackroll LUCKYD vs AC 0 0]
 }
 result: 0
}

ACCBONUS: -5

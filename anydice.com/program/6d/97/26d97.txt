function: highest sum of tied in ROLL:s {
    HIGH: 1@ROLL
    LAST: HIGH
    COUNT: 1
    loop N over {2..(#ROLL)} {
       CUR: N@ROLL
       if CUR = LAST {
           COUNT: COUNT + 1
       }
       else {
           COUNT: 0
       }
       if (COUNT * CUR) > HIGH {
           HIGH: COUNT * CUR
       }
    }
    result: HIGH
}

loop N over {1..5} {
    output [highest sum of tied in Nd10] named "[N]d10"
}

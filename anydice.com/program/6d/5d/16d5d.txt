function: frame MOD {
F: 12 - MOD
if F>20 {F: 20}
if F<0 {F: 0}

result: {0:F*F, 1:2*F, 2:2*F+1, 3:2*F+2, 4:2*F+3, 5:2*F+6, 6:2*F+7, 7:2*F+8, 8:2*F+9, 10:124+4*F-F*F, 15:240-20*F}
}

function: bless MOD {
result: {[frame MOD+1],[frame MOD+2],[frame MOD+3],[frame MOD+4]}
}

loop N over {0..6}{
NPC: [highest 1 of 2d[frame 10-2*N]]

FIGHTER: [highest 1 of 2d[bless 10-2*N]]

output FIGHTER - NPC named "[N]"
}

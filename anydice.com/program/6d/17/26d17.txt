\

Fighting styles
* Dueling
When you are wielding a melee weapon in one hand and no other weapons, you gain a +2 bonus to damage rolls with that weapon.
* Great Weapon Fighting
When you roll a 1 or 2 on a damage die for an Attack you make with a melee weapon that you are wielding with two hands, you can reroll the die and must use the new roll, even if the new roll is a 1 or a 2.


\


CRIT_THRESHOLD: 20

TARGET_AC: 17
ADVANTAGE_ROLL: [highest 1 of 2d20]
EA_ROLL: [highest 1 of 3d20] \ELVEN_ADVANTAGE\
ROLL: 1d20


function: R:n basicattack{
    if R >= CRIT_THRESHOLD {result: 2 d DMG + DMG_MOD}
    if (R + HIT_MOD) >= TARGET_AC {result: 1 d DMG + DMG_MOD}
    if (R + HIT_MOD) < TARGET_AC {result: 0}
}

function: R:n gwfattack{
    if R >= CRIT_THRESHOLD {result: 4d{{1..2}:2, {3..6}:8} + DMG_MOD}
    if (R + HIT_MOD) >= TARGET_AC {result: 2d{{1..2}:2, {3..6}:8} + DMG_MOD}
    if (R + HIT_MOD) < TARGET_AC {result: 0}
}

    if R >= CRIT_THRESHOLD {result: 4d{{1..2}:2, {3..6}:8} + DMG_MOD}
    if (R + HIT_MOD) >= TARGET_AC {result: 2d{{1..2}:2, {3..6}:8} + DMG_MOD}
    if (R + HIT_MOD) < TARGET_AC {result: 0}
}

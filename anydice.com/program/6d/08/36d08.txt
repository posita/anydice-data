function: power A:n {
 if A <= 11{result: 1}
 if A >= 17{result: 3}
 result: 2
}
function: resistance A:n {
 if A <= 10{result: 3}
 if A >= 17{result: 1}
 result: 2
}

M: 3

output [resistance 2d10+M] named "Resistance"
output [power 2d10+2+C] named "Edge"
output [lowest of [power 2d10+C] + 1 and 3] named "Double Edge"

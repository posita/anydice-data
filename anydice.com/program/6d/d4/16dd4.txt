function: wild WILDROLL:n N dice {
   if N > 1 {
   OTHERDICE: (N-1)d6 
   if WILDROLL = 6 {
      WILDEFFECT: [explode 1d6] 
      result: OTHERDICE + WILDEFFECT + WILDROLL} 
   if WILDROLL < 6 & WILDROLL > 1{
      result: OTHERDICE + WILDROLL}
   if WILDROLL = 1 {
      if N > 2 {
         WILDEFFECT: [lowest (N-2) of OTHERDICE]
         result: WILDEFFECT + WILDROLL}
      if N = 2 {result: WILDROLL}
      }
   }
   else {WILDEFFECT: [explode 1d6] result: WILDEFFECT}
}

output [wild 1d6 3 dice]

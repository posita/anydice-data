function: edged N:n {
  if N = 6 {
    result: 1 + [edged d6] 
  }
  if N = 5 {
    result: 1
  } 
  result: 0
}

function: edie {
  result: [edged d6]
}

function: die {
  result: d{0,0,1}
}

function: reroll N:n {
  if N < 5 {
    result: [count {5, 6} in d6] 
  }
  result: 1
}

function: rdie {
  result: [reroll d6] 
}

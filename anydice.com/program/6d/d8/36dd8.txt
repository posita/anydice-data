DICE: 2d10
function: roll DICE with bonus for doubles {
  result: [sum DICE]
}

function: roll with bonus for doubles in ROLL {
  DOUBLES: 0
  loop X over ROLL {
    if (ROLL = X) > 1 {
      DOUBLES: 1
    }
  }
  if DOUBLES {
    result: [roll DICE with bonus for doubles 3d10]
  } else {
    result: [roll DICE with bonus for doubles ROLL]
  }
}

output [roll with bonus for doubles in 2d10]

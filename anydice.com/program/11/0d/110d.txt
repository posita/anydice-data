function: pentastar S:s {
 SIXES: S=6
 if SIXES>0{
  result: 5+SIXES
 }
 else {
  result: 1@[sort S]
 }
}

function: pentastar DICE:s penalty PDICE:s {
 PIPS:{}
 loop PIP over {1..6}{
  COUNT:[count PIP in DICE]-[count PIP in PDICE]
  PIPS: {PIPS, PIP:COUNT}
 }
 if #PIPS=0{PIPS:{0}}
 result: [pentastar PIPS]
}

loop D over {3}{
 output [pentastar Dd6] named "PentaStar [D]d6"
 loop P over {4}{
  output [pentastar Dd6 penalty Pd6] named "PentaStar [D]d6 w/ penalty [P]d6"
 }
}

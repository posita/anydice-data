\ Swarmkeeper/Gloomstalker Ranger dual shortsword lvl 3 \

\ Constants \
LVL: 3
ATTACKS: 1
SHORT_SWORD: 1d6
DEX_MOD: 3
SMALL_FURY: LVL
HUNTER_MARK: 1d6
SWARM_DAMAGE: 1d6
GLOOM_ATTACK: 1d8
SS_CRIT_RANGE: 20
HIT: 1d20

\ SWARM SHORT SWORD FIRST ATTACK \
SWARM_FIRST: ATTACKS*(SHORT_SWORD+DEX_MOD+SMALL_FURY+HUNTER_MARK) + (SWARM_DAMAGE)
output SWARM_FIRST named "SWARMKEEPER_FIRST_ATTACK"

\ SHORT SWORD SUBSEQUENT ATTACKS \
SWARM_DUAL: ATTACKS *(SHORT_SWORD+DEX_MOD+HUNTER_MARK)+ (SHORT_SWORD+DEX_MOD+HUNTER_MARK)+SWARM_DAMAGE
output SWARM_DUAL named "SUBSEQUENT_SWARMKEEPER_ATTACKS"

\ GLOOMSTALKER FIRST ATTACK \
GLOOM_FIRST: ATTACKS*(SHORT_SWORD+DEX_MOD+SMALL_FURY+HUNTER_MARK) + (SHORT_SWORD+DEX_MOD+HUNTER_MARK+GLOOM_ATTACK)
output GLOOM_FIRST named "GLOOMSTALKER_FIRST_ATTACK"

\ GLOOMSTALKER SUBSEQUENT ATTACKS \
GLOOM_DUAL: ATTACKS*(SHORT_SWORD+DEX_MOD+HUNTER_MARK) + (SHORT_SWORD+DEX_MOD+HUNTER_MARK)
output GLOOM_DUAL named "SUBSEQUENT_GLOOMSTALKER_ATTACKS"

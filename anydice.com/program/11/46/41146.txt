\Lancer damage simulator\

ARMOR: 1 \assume average armor is 1\
TN:  9 \E-Def or Evasion target number - reduce by 1 if Field Analyst and also change 20 to 19 below\
ACC: 0 \Accuracy applied\ 
DIFF: 0 \Difficulty applied - either accuracy or difficulty should be 0\
RELIABLE: 2

ROLL: d20 + [highest 1 of ACCd6] - [highest 1 of DIFFd6]
function: dmgcalc R:n TN:n {
   if R >= TN {
\change 20 to 19 below if Field Analyst\
    if R >= 20 {
      result: [highest 1 of 2d6 and RELIABLE]-ARMOR
    } else {
      result: [highest of 1d6-ARMOR and RELIABLE]
    }
  } else {
    result: RELIABLE-ARMOR
  }
}

output [dmgcalc ROLL TN]

\ 4: Sucesso Total, 3: Sucesso Parcial, 2: Efeito Excessivo, 1: Fracasso \
function: desfecho JOGADA:s {
  EFEITOS: [count {8..10} in JOGADA]
  CONTROLES: [count {6..7} in JOGADA]
  FALHAS: [count {1..5} in JOGADA]
  
  if CONTROLES >= EFEITOS & EFEITOS > 0 { result: 4 } \ Sucesso Total \
  if CONTROLES > 0 & EFEITOS = 0 { result: 3 }        \ Sucesso Parcial \
  if EFEITOS > CONTROLES { result: 2 }                \ Sucesso Excessivo \
  if FALHAS = #JOGADA { result: 1 }                   \ Fracasso \
  
  result: 0
}

loop N over {1..4} {
  output [desfecho Nd10] named "[N]d10"
}

\script para calcular o dano médio levando em conta seus críticos e também as chances de nem passar na defesa ou mesmo rolar aquele 1 natural\

\altere apenas os números ao fim de cada linha\

\margem de ameaça\ M: 20
\multiplicador de crítico\ X: 2
\dano da arma\ A: 1d10
\dano extra numérico (flat)\ F: 14
\dados de dano extra\ D: 1d6
\bônus no teste de ataque\ B: 20
\defesa do inimigo\ C: 30

\não altere nada daqui pra baixo\

\chance de erro\ E: [highest of 1 and (C-B-1)]
\chance de acerto crítico\ I: [lowest of (20-E) and (20-M+1)]
\chance de acerto comum\ H: (20-E-I)

\dano acerto comum\ J: (A+F+D)
\dano acerto crítico\ K: (A*X+F+D)
\dano acerto lancinante\ L: ((A+F)*X+D)

V: ((J*H)+(K*I))/20
output V named "dano médio"

W: ((J*H)+(K*I)+(V*E))/20
output W named "dano médio com concentração"

Y: ((J*H)+(L*I))/20
output Y named "dano médio com lancinante"

Z: ((J*H)+(L*I)+(Y*E))/20
output Z named "dano médio com lancinante e concentração"

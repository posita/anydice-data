\Loner 3e Oracle Dice mechanic\
function: CHANCE:n over RISK:n {
  if CHANCE = RISK {
    result: 1
  }
  if CHANCE > RISK & CHANCE > 3 & RISK > 3 {
    result: 3
  }
  if CHANCE > RISK & CHANCE > 3 & RISK < 4 {
    result: 2
  }
  if CHANCE > RISK & CHANCE < 4 & RISK < 4 {
    result: 1
  }
  if CHANCE < RISK & CHANCE > 3 & RISK > 3 {
    result: -3
  }
  if CHANCE < RISK & CHANCE < 4 & RISK > 3 {
    result: -2
  }
  if CHANCE < RISK & CHANCE < 4 & RISK < 4 {
    result: -1
  }
}

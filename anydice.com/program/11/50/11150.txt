function: ROLL:n plus ATTACK vs AC:n {  
if ROLL >= THREAT & ROLL + ATTACK >= AC {result: [threat d20]}  
if ROLL + ATTACK >= AC | ROLL = 20 { result: DAMAGE }  
result: 0}  
   
function: threat ROLL:n {  
if ROLL + ATTACK + CONFIRM >= AC | ROLL = 20 {result: CRITICAL d DAMAGE}  
result: DAMAGE}  
   
AC: d{14..20}  
STRENGTH: 5
BAB: 1  
CONFIRM: 0  

WMOD: 1
THREAT: 20  
CRITICAL: 4
HIT: STRENGTH + BAB + WMOD 
DAMAGE: 2d4 + WMOD + STRENGTH * 3/2 
output [ d20 plus HIT vs AC] named "Olaf"

STRENGTH: 4
BAB: 3  
CONFIRM: 0  

WMOD: 0
THREAT: 19  
CRITICAL: 2
HIT: STRENGTH + BAB + WMOD

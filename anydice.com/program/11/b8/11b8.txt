MISS: 0
NORMAL: 1
CRIT: 2

function: critdie A:n {
   if A>=4 { result: CRIT}
   result: NORMAL
}

function: test AT:n plus ROLL:n again DEF:n plus DEFROLL:n {
  if ROLL=1 {result: MISS}

  if ROLL>1 {

    if ROLL=6 { result: [critdie d6] }

    if AT+ROLL >= DEF+DEFROLL { result: NORMAL }
    result: MISS
  }
}

output [test 1 plus d6 again 6 plus d6]

function: test STAT:n with BONUS:n SKILL:n vs TARGET:n via BAG:d {
  TOTAL_BONUS: STAT + BONUS + SKILL

  result: [helper BAG TOTAL_BONUS TARGET]
}

function: helper A:n AA:n B:n {
  if A = 999 {
    result: 1
  } else if A = -999 {
    result: 0
  } else if A + AA - B >= 0{
    result: 1
  } else {
    result: 0
  }
}

function: succeed by test STAT:n with BONUS:n SKILL:n vs TARGET:n via BAG:d {
  TOTAL_BONUS: STAT + BONUS + SKILL

  result: [amount BAG TOTAL_BONUS TARGET]
}

function: amount A:n AA:n B:n {
  if A = 999 {
    result: AA - B
  } else if A = -999 {
    result: 0 - B
  } else {
    result: A + AA - B
  }
}

function: evenodds NR:n {
  HALF: NR / 2
  if HALF * 2 != NR {
    result: NR-1
  } else {
    result: NR
  }
}

function: wild ROLL:s {
  SUCCESSES: ROLL >= 3
  SIXES: ROLL = 6

  VALUE: 0

  if SUCCESSES { VALUE: 1}
  if SIXES { VALUE: VALUE + [wild SIXESd6] }
  result: VALUE
}

function: successes ROLL:s {
  result: ROLL >= 3
}

loop N over {0..3} {
    O: N + 1
    output [successes Nd6] + [wild 1d6] named "[O]D"
    output [successes Nd6] + [wild 2d6] named "[O]D+1"
    output [successes Nd6] + [wild 3d6] named "[O]D+2"
}

output [successes 4d6] + [wild 1d6]

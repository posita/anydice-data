\margem de ameaça\ M: 20
\multiplicador de crítico\ X: 2
\dano da arma\ A: 1d10
\dano extra numérico (flat)\ F: 14
\dados de dano extra\ D:1d10
\defesa\ C: 30
\bônus de ataque\ B: 20

\chance de erro\ E: [highest of 5 and 5*(C-B-1)] 
\chance de acerto crítico\ I: [lowest of 100-E and 5*(20-M+1)] 
\chance de acerto comum\ H:(100-E-I)

\dano comum\ J:(A+F+D)
\dano crítico\ K:(A*X+F+D)
\dano lancinante\ L:((A+F)*X+D)

V: ((J*H)+(K*I))/100
output V named "dano médio"

W:((J*H)+(K*I)+(V*E))/100
output W named "dano médio com concentração"

Y: ((J*H)+(L*I))/100
output Y named "dano médio lancinante"

Z: ((J*H)+(L*I)+(Y*E))/100
output Z named "dano médio lancinante com concentração"

output H named "comum" 
output I named "critico" 
output E named "erro"

function: wild die COUNT:n {
    result: [wild die COUNT helper]
}

function: wild die COUNT:n helper {
    VALUE: 0
    NEWCOUNT: 0

    loop B over {1..COUNT} {
        X: [to number 1d6]
        if X = 6 {
            VALUE: 1
            NEWCOUNT: NEWCOUNT + 1
        }
    }

    result: VALUE + [wild die NEWCOUNT helper]
}

function: to number N:n {
    result: N
}

output [wild die 6]

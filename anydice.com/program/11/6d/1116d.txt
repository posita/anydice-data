function: ave AVE:s {
  LO: 1@AVE
  HI: 2@AVE
  \ Special Cases: \
  if LO = 20 & HI = 20 { result: 22 } \ double 20 = "triple critical": Triples damage/success, as opposed to a normal critical. \
  if LO = 10 & HI = 10 { result: -1 } \ double 10 = "critical draw": Auto triple critical, depends on success or failure  \
  if LO = 1 & HI = 1   { result: -2 } \ double 20 = "triple fumble": The worst ever. \
  if LO + HI = 20      { result: 21 } \ mirror pair = critical success \
  if LO = HI           { result:  0 } \ equal pair = critical failure \ 
  \ Any other roll: \
  if [absolute LO - 10] < [absolute HI - 10] { result: LO } else { result: HI }
}

output [ave 2d20] named "Averaged Roll"

MISS: 0
NORMAL: 1
CRIT: 2

function: critdie A:n {
   if A>=4 { result: CRIT}
   result: NORMAL
}

function: test AT:n plus ROLL:n again DEF:n plus DEFROLL:n {
  if ROLL=1 {result: MISS}

  if ROLL>=2 {

    if ROLL=10 { result: [critdie d6] }

    if AT + ROLL > DEF + DEFROLL { result: NORMAL }
    result: MISS
  }
}

loop N over {1..6} {
  output [test N plus d10 again 1 plus d10] named "atak [N]"
}

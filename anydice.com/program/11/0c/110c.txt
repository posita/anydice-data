function: pentastar S:s {
 SIXES: S=6
 if SIXES>0{
  result: 5+SIXES
 }
 else {
  result: 1@[sort S]
 }
}

function: pentastar DICE:s penalty PDICE:s {
 PIPS:{}
 loop PIP over {1..6}{
  COUNT:[count PIP in DICE]-[count PIP in PDICE]
  PIPS: {PIPS, PIP:COUNT}
 }
 if #PIPS=0{PIPS:{0}}
 result: [pentastar PIPS]
}
NDICE:3
output [pentastar NDICEd6] named "PentaStar [NDICE]d6"
output [pentastar NDICEd6 penalty 1d6] named "PentaStar [NDICE]d6 pen 1d6"

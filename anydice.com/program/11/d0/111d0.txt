function: plusone ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE  }
 result: 0
}

function: onedfour ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: ACRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD >= AC { result: ADAMAGE  }
 result: 0
}

ENEMYAC: d{12..20}

ATTACKMOD: 4
PROFICIENCY: 3
CRITROLL: 20
ATTACKTIMES: 1
DAMAGEBONUS: 1
ACCBONUS: 1
SNEAKATTACK: 3d6

DAMAGE: 1d8 + ATTACKMOD + DAMAGEBONUS + SNEAKATTACK
CRITDAMAGE: 2d8 + ATTACKMOD + DAMAGEBONUS + SNEAKATTACK*2

ADAMAGE: 1d8 + ATTACKMOD + 1d4 + SNEAKATTACK
ACRITDAMAGE: 2d8 + ATTACKMOD + 2d4 + SNEAKATTACK*2

output ATTACKTIMESd[plusone d20 vs ENEMYAC] named "+1 v 12-20 AC"
output ATTACKTIMESd[onedfour d20 vs ENEMYAC] named "+1d4 v 12-20 AC"

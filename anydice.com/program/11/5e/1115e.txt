function : cd J:s to R:s {
X : 1@J
S : #R
loop N over {1..S} {
  if N@R>S {  X : X +1 }
}
if 1@J=6 { X:X+[count{1@J} in J]-1 }
result : X
}

output [cd 1d6 to 4d6 ] named "14 4"
output [cd 2d6 to 3d6 ] named "23 4"
output [cd 3d6 to 2d6 ] named "32 4"
output [cd 4d6 to 1d6 ] named "41 4"
output [cd 4d6 to 2d6 ] named "42 4"

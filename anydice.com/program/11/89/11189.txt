function: rune D:s {
  HI: 1@D
  LO: 2@D
  if HI = 20 { result: 1 }
  if LO = 20 { result: 1 }
  if HI = LO & LO > 1 { result: 1 }
  result: 0
}

function: crit D:s {
  HI: D
  if HI=2 { result: 1 }
  result: 0
}

function: adv D:s {
  HI: 1@D
  LO: 2@D
  if HI = 20 { result: 1 }
  if LO = 20 { result: 1 }
  result: 0
}

output [crit 1d20] named "Normal"
output [adv 2d20] named "Advantage"
output [rune 2d20] named "Rune Crit"

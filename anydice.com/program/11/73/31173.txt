\
\\ 6 Piece Polyhedral Dice Pool 
\


\ Variables \
DICE: {4,6,8,10,12,20}

\ Function for the dice roll \
function: roll NUM:n {

  TOTAL: 0

  loop N over {1..NUM} {
    TOTAL: TOTAL + (d(N@DICE)>3)
  }

  result: TOTAL
}

\ Loops to test a variety of combinations \
loop N over {1..#DICE} {
  output [roll N] named "[N]"
}

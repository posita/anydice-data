set "explode depth" to 3

function: rolll A:n B:n C:n D {
 X: ({1, 2}@[sort {A, B, C}] - D)
 if X >= D + 10 { result : 3}
 if X >= D + 5 { result : 2}
 if X >= D { result : 1}
 result: 0
}

function: roll A:n B:n D {
 X: ({1, 2}@[sort {A, B}])

 \if X >= D + 10 { result : 3}
 if X >= D + 5 { result : 2}\
 if X >= D { result : 1}
 result: 0
}


loop M over {6, 8, 10}{
loop N over {5, 7, 9, 11, 13}{
 output [roll [explode dM] [explode dM] N] named "exp 2d[M] over [N]"
}
loop N over {5, 7, 9, 11, 13}{
 output [roll 1dM 1dM N] named "2d[M] over [N]"
}
}
\ 
output [roll 1d6 1d6 9]
output [rolll 1d6 1d6 1d6 10]
output [rolll [explode d6] [explode d6] [explode d6] 10]
output [roll 1d6 1d6 7]
output [roll 1d8 1d8 1d8 10]
output [roll [explode d8] [explode d8] [explode d8] 10]
output [roll 1d10 1d10 1d10 10]
output [roll [explode d10] [explode d10] [explode d10] 10]

function : cd J:s to R:s {
X : 1@J
if 1@J=6 {
  loop N over {2..#J} { \si 6 est tiré, rajouter 1 pour chaque dé contigu au plus grand tirage (6 ou 5,4,...)\
    if (N+1)@J=6 | (N+1)@J=N@J-1 { X : X+1 } else { result : X }
  }
}
result : X
}

output [cd 1d6 to 4d6 ] named "14 4"

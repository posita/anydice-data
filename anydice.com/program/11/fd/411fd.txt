function: ROLL:s vardmgonthree {
  if ROLL >= TN {
    result: [count {3..6} in (DMG -1) d6] + 1
  } else {
    result: 0
  }
}

function: ROLL:s vardmgontwo {
  if ROLL >= TN {
    result: [count {2..6} in (DMG -1) d6] + 1
  } else {
    result: 0
  }
}


loop DMG over {1..7}{
  loop TN over {2..12}{
   loop N over {1..6}{
      output [count {(TN - N)..6} in DMG d6]  named "Pilot Die D6 = [N] TN = [TN] DMG = [DMG]"
   }

   loop N over {1..4}{
      output [count {(TN - N)..8} in DMG d8]  named "Pilot Die D4 = [N] TN = [TN] DMG = [DMG]"
   }
   loop N over {1..8}{
      output [count {(TN - N)..4} in DMG d4]  named "Pilot Die D8 = [N] TN = [TN] DMG = [DMG]"
   }  
  
  }
}

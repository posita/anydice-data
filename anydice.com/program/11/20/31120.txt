function: wildcard TRAIT:d {
	result: [highest of [explode TRAIT] and [explode d6]]
}

function: damage {
	X: [explode d6] + [explode d6]
	result: X

}

function: damage with raise {
	X: [damage] + [explode d6]
	result: X
}

function: hit parry PARRY:n with FGHT:n {
	if FGHT >= PARRY + 4 {
		result: [damage with raise]}
	else if FGHT >= PARRY {
		result: [damage]
	}
	result: 0
}

function: fighting A:d against parry PARRY:n{
 FIGHTING_ROLL: [wildcard A]
 result: [hit parry PARRY with FIGHTING_ROLL]
}

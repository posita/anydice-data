function: challenge ACTION:n vs CHALLENGE:n {
  if ACTION > CHALLENGE {result: 1}
  result: 0
}

function: ironsworn ACTION:n {
  result: (([challenge (ACTION) vs (1d10)])+([challenge (ACTION) vs (1d10)]))
}

STAT: 2
ADDS: 0

ACTION_SCORE: 1d6 + STAT + ADDS

\ Loop STAT from 0 to 5 \
loop STAT over {0..5} {
  ACTION_SCORE: 1d6 + STAT + ADDS
  
  output [ironsworn ACTION_SCORE] named "Stat [STAT] (2=Strong, 1=Weak, 0=Miss)"
  
  \ Uncomment the line below if you also want to see 'Any Hit' vs 'Miss' for each stat \
  \ output [ironsworn ACTION_SCORE] >= 1 named "Stat [STAT] Likelihood of Any Hit" \
}

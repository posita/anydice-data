function: nwodeval ROLL:s {
 SUCCESSES: ROLL >= 8
 if SUCCESSES > 0 { result: SUCCESSES + (ROLL = 10) }
 if [count 1 in ROLL] { result: -1 }
 result: 0
}

function: larpeval ROLL:s ADD:n {
 SUCCESSES: (ROLL+ADD >= 10) + (ROLL+ADD-10)/5
 if SUCCESSES > 0 { result: SUCCESSES }
 if [count 1 in ROLL] { result: -1 }
 result: 0
}

loop DICE over {1..15} {
 output [nwodeval DICE d10] named "nWoD Tabletop [DICE]d"
}

function: DIE success {
 if DIE = 1 {N: 0}
 if DIE > 1 & DIE < 4 {N: 0}
 if DIE > 3 & DIE < 8 {N: 1}
 if DIE > 7 & DIE < 12 {N: 2}
 if DIE = 12 {N: 3}
 result: N
}

function: DIE extra{
 if DIE > 0 & DIE < 4 {N: 0}
 if DIE > 3 & DIE < 8 {N: 1}
 if DIE > 7 & DIE < 12 {N: 2}
 if DIE = 12 {N: 3}
 result: N
}

function: DIE:n modifier{
 result: [DIE extra]
}

function: roll BASE:n{
 result: [BASE success]
}

function: explode BASE:n on MAX:n with CORDIE:n{
 if BASE = MAX {
 result: [BASE success] + [CORDIE extra]
 }
 result: [BASE success]
}

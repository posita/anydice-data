function: normal ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE + CRITSNEAKATK } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE + SNEAKATK }
 result: 0
}

function: surprise ROLL:n vs AC:n {
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: CRITDAMAGE + CRITSNEAKATK + BUGBEAR }
 result: 0
}

ENEMYAC: d{12..17}

ATTACKMOD: 3
PROFICIENCY: 3
CRITROLL: 20
ATTACKTIMES: 1
DMGBONUS: 1d8
ACCBONUS: 0
BUGBEAR: 4d6

DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS*2

SNEAKATK: 2d6
CRITSNEAKATK: 4d6

output ATTACKTIMESd[normal d20 vs ENEMYAC] named "Rogue Booming Blade"
output ATTACKTIMESd[surprise [highest 1 of 2d20] vs ENEMYAC] named "Surprise Round"

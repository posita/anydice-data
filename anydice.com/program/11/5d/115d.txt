function: nwod R:n again N:n {
 if N >= R { result: 1 + [nwod R again d10] }
 result: N >= 8
}

NWOD: [nwod 10 again d10]

loop N over {1..15} {
 output [lowest of 10 and NdNWOD] named "nWoD [N]d"
}

function: larp N:n {
 result: [highest of 0 and (N+[explode d10])-5]/5
}

loop N over {1..15} {
 output [lowest of 10 and [larp N]] named "LARP [N]"
}

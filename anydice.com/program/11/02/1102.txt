function: remove index I:n from SEQ:s{
 NEWSET: {}
 loop N over {1..#SEQ}{
  if N!=I{
   NEWSET: {NEWSET,N@SEQ}
  }
 }
 result: NEWSET
}
function: SET:s pentastar dice with PSET:s penalty dice{
RSET: SET
loop P over {1..#PSET}{
 loop S over {1..#RSET}{
  if P@PSET=S@RSET{
   RSET: [remove index S from RSET]
   PSET: [remove index P from PSET]
  }
 }
}
 SIXES: RSET=6
 if SIXES>0{
  result: 5+SIXES
 }
 else {
  result: 1@[sort RSET]
 }
}
loop LEVEL over {1}{
 loop PENALTIES over {0..2}{
  output [LEVELd6 pentastar dice with PENALTIESd6 penalty dice] named "L[LEVEL] w/ [PENALTIES]pen."
 }
}

DICEDAMAGE: 1d12
FLATDAMAGE: 18
HITBONUS: 0
ATTACKS: 3
OPROHB: 3d8
OPROHT: 3
CRIT: 20
BCRIT: 0

function: attack B:n HIT:n DD:n FD:n {
  if HIT >= CRIT {
    \on crit\
    result: DD + FD + (1-BCRIT) * DD + BCRIT * [maximum of DICEDAMAGE]
  }
  if HIT + B >= 8 {
    \on hit\
    result: DD + FD
  } else {
    \on miss\
    result: 0
  }
}

function: oproh OHB:n ATK:n {
  if ATK = 0 {
    result: 0
  }
  result: ATK + OHB
}

function: atk {
   result: [attack HITBONUS 1d20 DICEDAMAGE FLATDAMAGE]
}

function: atkh H:n {
   result: [attack HITBONUS+H 1d20 DICEDAMAGE FLATDAMAGE]
}

function: atkd D:d {
   result: [attack HITBONUS 1d20 DICEDAMAGE+D FLATDAMAGE]
}

function: adv {
   result: [highest of 1d20 and 1d20]
}

function: eadv {
   result: [highest of 1d20 and [highest of 1d20 and 1d20]]
}

function: aatk {
   result: [attack HITBONUS [adv] DICEDAMAGE FLATDAMAGE]
}

output [oproh OPROHB OPROHTd[attack HITBONUS 1d20 DICEDAMAGE FLATDAMAGE]] + (ATTACKS-OPROHT)d[attack HITBONUS 1d20 DICEDAMAGE FLATDAMAGE] named "DPR"
output [oproh OPROHB OPROHTd[attack HITBONUS [highest of 1d20 and 1d20] DICEDAMAGE FLATDAMAGE]] + (ATTACKS-OPROHT)d[attack HITBONUS [highest of 1d20 and 1d20] DICEDAMAGE FLATDAMAGE] named "DPR (advantage)"
output [oproh OPROHB OPROHTd[attack HITBONUS [highest of 1d20 and [highest of 1d20 and 1d20]] DICEDAMAGE FLATDAMAGE]] + (ATTACKS-OPROHT)d[attack HITBONUS [highest of 1d20 and [highest of 1d20 and 1d20]] DICEDAMAGE FLATDAMAGE] named "DPR (elven)"

output [oproh 1d8 4d[atk]] named "specific"
output [oproh 2d8 2d[attack HITBONUS [adv] DICEDAMAGE+2d12 FLATDAMAGE]] named "specific2"
output [oproh 2d8 2d[attack HITBONUS [eadv] DICEDAMAGE+2d12 FLATDAMAGE]] named "specific3"

\function: rr one D:n DD:d {
  if D = 1  {
    result: DD
  } 
  result: D
}
function: two one D:n {
  if D = 1  {
    result: 2
  } 
  result: D
}\

function: attackgs ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: GSCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: GSDAMAGE }
 result: 0
}

function: attackkgs ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: KGSCRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: KGSDAMAGE }
 result: 0
}

function: attackga ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: GACRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: GADAMAGE }
 result: 0
}

ENEMYAC: d{13..19}

ATTACKMOD: 5
DMGBONUS: 0
PROFICIENCY: 4
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 2
GSDAMAGE: 2d6 + ATTACKMOD + DMGBONUS
GSCRITDAMAGE: 5d6 + ATTACKMOD + DMGBONUS
KGSDAMAGE: 3d4 + ATTACKMOD + DMGBONUS
KGSCRITDAMAGE: 7d4 + ATTACKMOD + DMGBONUS
GADAMAGE: 2d6 + ATTACKMOD + DMGBONUS
GACRITDAMAGE: 3d12 + ATTACKMOD + DMGBONUS

output ATTACKTIMESd[attackgs d20 vs ENEMYAC] named "Greatsword"
output ATTACKTIMESd[attackkgs d20 vs ENEMYAC] named "Knight Greatsword"
output ATTACKTIMESd[attackga d20 vs ENEMYAC] named "Greataxe"

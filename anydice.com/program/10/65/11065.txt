function: attack ROLL:n plus ATTACK_BONUS:n for DAMAGE:d plus DAMAGE_BONUS:n and EXTRA_DAMAGE:d damage {
  if ROLL = 1 {
    result: 0
  } else if ROLL = 20 {
    result: DAMAGE + DAMAGE + DAMAGE_BONUS + EXTRA_DAMAGE + EXTRA_DAMAGE
  } else if ROLL + ATTACK_BONUS >= 16 {
    result: DAMAGE + DAMAGE_BONUS + EXTRA_DAMAGE
  } else {
    result: 0
  }
}

function: save DC:n against ROLL:n for DAMAGE:d and BONUS_DAMAGE:d damage {
  if ROLL >= DC {
    result: DAMAGE + BONUS_DAMAGE
  } else {
    result: 0
  }
}

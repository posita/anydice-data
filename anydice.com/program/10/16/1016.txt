function: haveho COUNT:n {
 MYONE: 1
 MYTWO: 2*10^2
 MYTHREE: 3*10^4
 MYFOUR: 4*10^6
 MYFIVE: 5*10^8
 MYSIX: 6*10^10
 result: [hinspect COUNTd{MYONE,MYTWO,MYTHREE,MYFOUR,MYFIVE,MYSIX}]
}

function: hinspect NUMBY:n {
 if NUMBY>=MYONE
    {HO: 0+(NUMBY/MYONE)}
 if NUMBY>=MYTWO
    {HO: 1+(NUMBY/MYTWO)}
 if NUMBY>=MYTHREE
    {HO: 2+(NUMBY/MYTHREE)}
 if NUMBY>=MYFOUR
    {HO: 3+(NUMBY/MYFOUR)}
 if NUMBY>=MYFIVE
    {HO: 4+(NUMBY/MYFIVE)}
 if NUMBY>=MYSIX
    {HO: 5+(NUMBY/MYSIX)}
 result: HO
}

output [haveho 6]
output [highest 1 of 6d6]

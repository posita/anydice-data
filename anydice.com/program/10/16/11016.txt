function: gwf DIE:d {
  result: [reroll DIE as DIE if under 3]
}
function: reroll ROLL:n as DIE:d if under LIMIT:n {
  if ROLL < LIMIT { result: DIE }
  else { result: ROLL }
}

loop SIDES over {6} {
  output 2d[gwf dSIDES] named "Maul with GWF"
}

function: X:n gwf {
if X = 1 {result: 1d12}
if X = 2 {result: 1d12}
result: X
}

output 2d6 named "Maul without GWF"

output [d12 gwf] named "Great Axe GWF"

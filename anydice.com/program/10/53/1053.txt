
function: explode DIE:n at N:n {
    if DIE = N {result: N + [explode d6 at N]}
    result: DIE
}

function: explode M:n doubles below K:n {
    if (M <= K) {B: [explode d6 at M]}
    else {B: d6}
    result: M + B
}

set "maximum function depth" to 5

loop A over {0..6} {
    output [explode d6 doubles below A] named "[A]"
}

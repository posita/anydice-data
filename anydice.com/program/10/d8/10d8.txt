SUCCESSDIE: d6 > 4
N: 4

loop MAXMULT over {0..2*N}{

function: ROLL:s with mult {
 if MAXMULT > N {
  MULTB: MAXMULT - N
  MULTA: N
 } else { 
  MULTA: MAXMULT
  MULTB: 0 
 }
  
 R: 0
 loop P over {1..#ROLL} {
  if P@ROLL {
   if MULTB > 0 {
    R: R + 3
    MULTB: MULTB - 1
   } else if MULTA > 0 {
    R: R + 2
    MULTA: MULTA - 1
   } else { R: R + 1 }
  }
 }
 result: R
}
output [N d SUCCESSDIE with mult] named "[MAXMULT]" 
}

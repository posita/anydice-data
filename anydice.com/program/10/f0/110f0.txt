function: stdroll X:n Y:n BONUS:n D:n {
  if ( (X + Y + BONUS) >= 7 ) {
    result: 1dD
  }
}

function: advdmg X:n Y:n BONUS:n D:n {
  if ( (X + Y) >= 10 ) {
    result: [highest of 1dD and 1dD]
  }

  if ( (X + Y + BONUS) >= 7 ) {
    result: 1dD
  }
}

function: lowroll X:n Y:n BONUS:n {
  if ( (X + Y + BONUS) >= 7 ) {
    if ( X < Y ) { result: X  }
    result: Y
  }
}

function: rolldist X:n {
  result: X
}

function: under A:n of B:n and C:n {
if B != C & A <= [lowest of B and C] & A < [highest of B and C] {result:A}
if B = C & A < B {result:A}
result: 0
}

function: above A:n of B:n and C:n {
if B != C & A > [lowest of B and C] & A >= [highest of B and C] {result:A}
if B = C & A > B {result: A}
result: 0
}

function: between A:n of B:n and C:n {
if A > [lowest of B and C] & A < [highest of B and C] {result:A}
result: 0
}


function: outcome A:n of B:n and C:n of D:n and E:n {
if A+D+E = B & B = C {result: 0}
if A = [lowest of B and C] {result:-2}
if [under A+D+E of B and C] {result: -1} 
if A = [highest of B and C] {result: 0}
if [between A+D+E of B and C] {result: +2}
if [above A+D+E of B and C] {result: +1}
}

A: 1d13
B: 1d13
C: 1d13
D: d{2,1,1,0}
E: d{1,0,-2,-2}

output D+E

output [outcome A of B and C of 0 and 0]
output [outcome A of B and C of D and E]

output [under A of B and C]

output A = [lowest of B and C]

output [between A of B and C]

output A = [highest of B and C]

output [above A of B and C]

output A = B & B = C

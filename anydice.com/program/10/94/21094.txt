set "maximum function depth" to 2
function : copy mephit N:d
{
 J : Nd{0:3,1}d(1d4)
 result : N + [copy mephit J]
}

function : calc N:n M:n
{
 if N = 1 {result : M * N}
 else if N = 2 {result : (3 * M * N)/2}
 else if N < 7 {result : 2 * M * N}
 else if N < 11 {result : (5 * M * N)/2}
 else if N < 15 {result : 3 * M * N}
 else {result : 4 * M * N}
}

loop N over {1..5}
{
 J : [copy mephit N]
 output [lowest of [calc J 100] and 1000]
}

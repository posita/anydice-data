TOTAL_DICE: 5
ABILITY: 10
FOCUS: 3

function: ROLL:s under ABILITY:n and FOCUS:n {
 NON_CRITICAL: (ROLL <= ABILITY) + (ROLL <= FOCUS)
 CRITICAL: (ROLL <= ABILITY) + (ROLL = 1)
 if CRITICAL > NON_CRITICAL { result: CRITICAL}
 else {result: NON_CRITICAL}
}

loop N over {1..TOTAL_DICE} {
  output [Nd20 under ABILITY and FOCUS]
}

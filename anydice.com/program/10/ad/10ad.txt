function: roll POOL:n at TARGET:n {
 A: 0
 loop X over {1..POOL} {
  B: A + [proll 1d10 at TARGET]
  A: B
 }
 result: A
}

function: proll D:n at TARGET:n {
 if D >= (10 - TARGET) {
  result: 1
 } else {
  result: 0
 }
}

loop ATT over {1..5} {
 loop SKILL over {0..5} {
  output [roll ATT at SKILL] named "Rolling [ATT] dice at skill [SKILL]"
 }
}

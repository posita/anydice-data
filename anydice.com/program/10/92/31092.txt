MISS: 0
HIT: 1
CRIT: 2
function: attack ROLL:n vs AC:s{
 if ROLL = 1{result: MISS}
 if ROLL = 20{
  if ROLL + ACC >= AC{result:CRIT}
  result: HIT
 }
 if ROLL+ACC >= AC{result:HIT}
 result: MISS
}
MISSDAMAGE: 0
function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}
ACC: 12
HITDAMAGE: 12+16
CRITICALDAMAGE: HITDAMAGE * 2
loop DEX over {5..20}{
 output [attack d20 vs d20+DEX] named "[DEX]"
}

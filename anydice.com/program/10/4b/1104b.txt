\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

ENEMYAC: d{15..20}

ATTACKMOD: 3
PROFICIENCY: 2
CRITROLL: 20


DMGBONUS: 2
ACCBONUS: 0
ATTACKTIMES: 2
ADVANTAGE: 1
DAMAGE: 1d6 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d6 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of ADVANTAGEd20] vs ENEMYAC] named "Dualwielding"



DMGBONUS: 4
ACCBONUS: 0
ATTACKTIMES: 1
ADVANTAGE: 1
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of ADVANTAGEd20] vs ENEMYAC] named "Singlewielding"

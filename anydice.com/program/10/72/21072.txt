function: stunt points for DICE:s and ACTION_DIE:n vs TARGET:n {
  if DICE + ACTION_DIE < TARGET { result: 0 }  \ miss \
  if 1@DICE = 2@DICE | ACTION_DIE = DICE { result: 1 }  \ pair \
  result: 0  \ hit but no pair \
}

  output [stunt points for 2d6 and d20 vs TARGET] named "hit on [TARGET]"
}

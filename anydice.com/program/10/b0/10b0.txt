function: roll POOL:n at SKILL:n {
 A: 0
 loop X over {1..POOL} {
  B: A + [proll 1d6 at SKILL]
  A: B
 }
 result: A
}

function: proll D:n at SKILL:n {
 if D <= SKILL {
  result: 1
 } else {
  result: 0
 }
}

loop ATT over {1..5} {
 loop SKILL over {1..5} {
  output [roll ATT at SKILL] named "Rolling [ATT] dice at skill [SKILL]"
 }
}

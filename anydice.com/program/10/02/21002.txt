PASSIVE: 0
ACTIVE: 0

function: modifier for A:n vs B {
  OFFSET: B*5
  THRESHOLD: 60-OFFSET
  RAW: A + 1d100
  DIFFERENCE: RAW - THRESHOLD

  MODIFIER: DIFFERENCE / 10
  
  result: MODIFIER
}


output [modifier for -5d10 vs {PASSIVE}] named "Active -5 vs Passive [PASSIVE]"

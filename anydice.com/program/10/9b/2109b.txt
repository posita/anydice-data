CHALLENGE:2d10

output (d6-1 > 1@CHALLENGE) + (d6-1 > 2@CHALLENGE) named "-1"

loop N over {0..6}{

ACTION:[lowest of 10 and (d6 + N)]
CONFIDENCE:[lowest of 10 and (d{2, 3, 4, 5, 6, 6} + N)]
DESPERATION:[lowest of 10 and (d{1, 1, 2, 3, 4, 5} + N)]

output (ACTION > 1@CHALLENGE) + (ACTION > 2@CHALLENGE) named "+[N]"
output (CONFIDENCE > 1@CHALLENGE) + (CONFIDENCE > 2@CHALLENGE) named "+[N] with confidence"
output (DESPERATION > 1@CHALLENGE) + (DESPERATION > 2@CHALLENGE) named "+[N] in desperation"
}

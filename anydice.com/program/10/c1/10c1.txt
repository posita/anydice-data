L: 3             \greater than this counts as a success\
REROLLS: 1       \how many rerolls you get\
\-----------------------------------\
N:1
COUNTER: 1
function: die C:n {
 if (C<=L) & (COUNTER <= REROLLS) {
  COUNTER: COUNTER + 1
  result: [reroll d6]
 }
 else {
  result: C>L
 }
}

function: reroll C:n {
 result: C>L
}

 output Nd[die d6]
 output Nd[reroll d6]


function: attack ROLL:n plus BONUS:n vs AC:n hit HIT:d miss MISS:d crit CRIT:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: MISS
    }
    if ROLL >= CRIT_RANGE {
        result: CRIT
    }
    if ROLL + BONUS >= AC {
        result: HIT
    }
    result: MISS
}


function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

ADV: [highest 1 of 2d20]

DISADV: [lowest 1 of 2d20]

ELV_ACC: [highest 1 of 3d20]

function: ba BONUS_ACTION:d attacks ATTACKS:n roll ROLL:d plus BONUS:d vs AC:d for DMG:d crit CRIT_DMG:d on CRIT_RANGE:d {

    NO_CRIT: BONUS_ACTION
    ALREADY_CRIT: BONUS_ACTION

    ATTACK: [attack ROLL plus BONUS vs AC for DMG crit CRIT_DMG on CRIT_RANGE]

    loop N over {1 .. ATTACKS}{
        NO_CRIT: [attack ROLL plus BONUS vs AC 
            hit  DMG + NO_CRIT
            miss NO_CRIT
            crit DMG + CRIT_DMG + ALREADY_CRIT on CRIT_RANGE
        ]
        ALREADY_CRIT: ATTACK + ALREADY_CRIT
    }

    result: NO_CRIT
}

AC: d{12..18}
ROLL: d20
COMP_ROLL: d20

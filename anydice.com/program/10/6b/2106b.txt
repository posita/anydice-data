function: save  RESULT:n {
 if ARMOR >= RESULT {
   if VULNERABLE = 1 { 
     VULNERABLE: 0
     result: [save d6] 
   }
   result: 0
 }
 result: 1
}

function: hit RESULT:n {
 if NOSAVE = 1 { result: 1 }
 if RESULT >= TOHIT {
   if WEAKEN = 1{ 
    WEAKEN: 0
    result: [hit d6]
   }
   if RESULT = 6 {
     if PRECISION = 1 {
        if CRITBLOW = 1 {result: 1 + [save d6] }
        result: 1
     }
     if CRITBLOW  = 1 { result: [save d6] + [save d6] }
   }
   result: [save d6]
 }
 if REROLL = 1 {
   REROLL : 0
   result: [hit d6]
 }
 result: 0
}

function: moral DICEONE:n and DICETWO:n damage DAMAGE:d{
  if DICEONE+DICETWO-MORALMALUS >= MORAL {
    if PANIKED = 1 {
      PANIKED: 0
      if DICEONE >= 4 {
        if DICETWO >= 4 {
          if DAMAGE = 1 {
            result: [moral d6 and d6 damage d3]
          }
          result: [moral d6 and d6 damage DAMAGE]
        }

        if DAMAGE = 1 {
          result: [moral d6 and DICETWO damage d3]
        }
          result: [moral d6 and DICETWO damage DAMAGE]
      }
      if DICETWO >= 4 {
        if DAMAGE = 1 {
          result: [moral d6 and DICEONE damage d3]
        }
        result: [moral d6 and DICEONE damage DAMAGE]
      }
      if DAMAGE = 1 {
        result: [moral DICEONE and DICETWO damage d3]
      }
    }
    result: 0
  }
  if PANIKED = 1 {
    PANIKED: 0
    if DAMAGE = 1 {
      result: [moral d6 and DICETWO damage d3]
    }
  }
  result : DAMAGE + MORALWOUND
}

function: takemoral WOUND:n {
   if WOUND >=1 {
      result: WOUND + [moral d6 and d6 damage d3]
   }
   result: 0 
}

function: attack {
   WOUND: DICEd[hit d6] + BONUSHITd[save d6]
   result : [takemoral WOUND]
}

function: doattack {
  TOHIT : TOHIT - TOHITMALUS
  if TOHIT > 6 {TOHIT: 6}
  if TOHIT < 2 {TOHIT: 2}
  DICE: DICE+ BONUSDICE
  if REAR= 1 {
     ARMORMALUS: ARMORMALUS + 2
     MORALMALUS: MORALMALUS + 2
  }
  else {
    if FLANK = 1 {
       ARMORMALUS: ARMORMALUS + 1
       MORALMALUS: MORALMALUS + 1
    }
  }
  ARMOR: ARMOR - ARMORMALUS
  if ARMOR > 6 {ARMOR: 6}
  if ARMOR < 2 {ARMOR: 2}
  if CHARGE = 1 {
     if DISORDERLY = 0 {
       REROLL : 1
     }
  }
  result: [attack]
}

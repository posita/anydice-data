\ Paladin/Hexblade 15/5 (17/3?, 19,1?) \
\ EB Build can go straight warlock, or sorlock \
\ Longsword/Spear + Shield or Glaive \


\ Constants \
ATTACKS: 1
EB_CASTS: 1
CHA_MOD: 3
WEAPON_MOD: 0
EB: 1d10
HEX: 1d6
GLAIVE: 1d10
LONGSWORD: 1d8
SPEAR: 1d6
MADDENING_HEX: CHA_MOD
PAM_BA: 1d4
DUELING_BONUS: 2

\ Great Weapon Fighting \
\ 1d[gwfdice 10] == 1d10 w/ GWF \
function: gwfdice N {
 result: {{1..2}, {3..N}:N/2+1}
}

\ GWF Attacks \
GWF_GLAIVE: 1d[gwfdice 10]
GWF_PAM_BA: 1d[gwfdice 4]


\ Sustained, with hex up \
\ LS + Maddening Hex \
LONGSWORD_SUSTAINED: (ATTACKS*(LONGSWORD+CHA_MOD+WEAPON_MOD+DUELING_BONUS+HEX))+MADDENING_HEX
output LONGSWORD_SUSTAINED named "LONGSWORD_SUSTAINED"

\ Spear + PAM BA \
SPEAR_SUSTAINED: (ATTACKS*(SPEAR+CHA_MOD+WEAPON_MOD+DUELING_BONUS+HEX))+(PAM_BA+DUELING_BONUS+CHA_MOD+WEAPON_MOD+HEX)
output SPEAR_SUSTAINED named "SPEAR_SUSTAINED"

\ EB + Maddening Hex \
\ Assuming Agonizing Blast \
EB_SUSTAINED: (EB_CASTS*(EB+CHA_MOD+HEX))+MADDENING_HEX
output EB_SUSTAINED named "EB_SUSTAINED"

\ Glaive + PAM BA \
\ With GWF \
PAM_SUSTAINED: (ATTACKS*(GWF_GLAIVE+CHA_MOD+WEAPON_MOD+HEX))+(GWF_PAM_BA+CHA_MOD+WEAPON_MOD+HEX)
output PAM_SUSTAINED named "PAM_SUSTAINED"

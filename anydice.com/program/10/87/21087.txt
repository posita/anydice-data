function: attack ROLL:n plus BONUS:n vs AC:n for DMG:d crit CRIT_DMG:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: DMG + CRIT_DMG
    }
    if ROLL + BONUS >= AC {
        result: DMG
    }
    result: 0
}

AC: d{15..20}

output [attack 1d20 plus 9 vs AC for 1d8+5+6d6 crit 1d8+6d6 on 20] named "12th-level rogue"
output [attack 1d20 plus 9 vs AC for 1d4+5+6d6 crit 1d4+6d6 on 19]
output [attack 1d20 plus 10 vs AC for 1d8+5+7d6 crit 1d8+7d6 on 20] named "13th-level rogue"
output [attack 1d20 plus 10 vs AC for 1d4+5+7d6 crit 1d4+7d6 on 19]

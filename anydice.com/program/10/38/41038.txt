ATKMOD: 14 + 4
AC: 22
HIT: 1d20 + ATKMOD
NORMALDAMAGE: 2d8 + 2d6
CRITDAMAGE: 2 * (2d12 + 2d6) + 1d12

function: hitresult NUMBER:n {
    if NUMBER >= AC + 10 {
        result: 2
    }
    if NUMBER >= AC {
        result: 1
    }
    else {
        result: 0
    }
}

function: damage NUMBER:n {
    if NUMBER >= AC + 10 {
        result: CRITDAMAGE
    }
    if NUMBER >= AC {
        result: NORMALDAMAGE
    }
    else {
        result: 0
    }
}

output [hitresult HIT]
output [damage HIT]

function: expfour N:n {
 if N = 4 { result: N + [expfour d4] }
 result: N
}
function: expsix N:n {
 if N = 6 { result: N + [expfour d4] }
 result: N
}
function: expeight N:n {
 if N = 8 { result: N + [expfour d4] }
 result: N
}

function: flail A:n p B:n m C:n {
  if A = B {
    result: A - C
  }
  result: A + B - C
}

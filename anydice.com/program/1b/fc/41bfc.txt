function: check NUM actions delve equipment SIZE resistance RES {
 result: NUMdSIZE >= RES
}

function: fallout on STRESS with DIE:s {
 result: DIE <= STRESS
}

function: stress STRESS:s fallout FALLOUT:s {
 result: (FALLOUT <= STRESS)
}

loop EQPT over {4,6,8,10,12} {
 loop RES over {8,10,12,14,16} {
  loop ACT over {1..6}{
   output [check 1 actions delve equipment EQPT resistance RES] named "check [ACT] actions delve equipment [EQPT] resist [RES]"
  }
 }
}

\
loop F over {4,6,8,10,12} {
 loop N over {4,6,8,10,12}{
  output [stress 1dN fallout 1dF] named "chance of taking fallout using a 1d[N] delve with a D[F] fallout die"
 }
}

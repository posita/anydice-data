function: ATR:n attribute FEAT:n feat with SKILL:n skill {
  if (FEAT = 1) {
    result: 3
  } else if (ATR = FEAT) {
    if (ATR <= SKILL) {
      result: 3
    } else {
      result: -1
    }
  } else if (ATR <= SKILL) {
    if (FEAT <= SKILL) {
      result: 2
    } else {
      result: 1
    }
  } else {
    result: 0
  }
}

TARGET_AC: {19}

STR: 5 
PROF: 5
RAGE: 3
CRIT: 2
FELL_HANDED_A: 1
SAVAGE_ATTACKS: 1
BRUTAL_CRIT: 2
BRUTAL_FIST_DIE: 4

GREATAXE_ATTACK: {FELL_HANDED_A + STR + PROF}

BRUTAL_FIST_ROLL: 1d{{1..3}, {BRUTAL_FIST_DIE}:3}
BRUTAL_FIST_ROLL_CRIT: CRITd{{1..3}, {BRUTAL_FIST_DIE}:3}
GREATAXE_ROLL: 1d{{1..3}, {4..12}:7}
GREATAXE_ROLL_CRIT: {CRIT + SAVAGE_ATTACKS + BRUTAL_CRIT}d{{1..3}, {4..12}:7}

GREAT_WEAPON_MASTER_A: -5
GREAT_WEAPON_MASTER_D: 10

GREATAXE_DAMAGE: GREATAXE_ROLL + STR
GREATAXE_DAMAGE_CRIT: GREATAXE_ROLL_CRIT + STR

MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{TARGET_AC}

ATTACK: GREATAXE_ATTACK
HITDAMAGE: GREATAXE_DAMAGE
CRITICALDAMAGE: GREATAXE_DAMAGE_CRIT
MISSDAMAGE: 0
output [attack] named "GA"

ATTACK: GREATAXE_ATTACK + GREAT_WEAPON_MASTER_A
HITDAMAGE: GREATAXE_DAMAGE + GREAT_WEAPON_MASTER_D
CRITICALDAMAGE: GREATAXE_DAMAGE_CRIT + GREAT_WEAPON_MASTER_D
output [attack] named "GA GWM"

ATTACK: GREATAXE_ATTACK
HITDAMAGE: GREATAXE_DAMAGE + RAGE
CRITICALDAMAGE: GREATAXE_DAMAGE_CRIT + RAGE
output [attack] named "GA Rg"

ATTACK: GREATAXE_ATTACK + GREAT_WEAPON_MASTER_A
HITDAMAGE: GREATAXE_DAMAGE + GREAT_WEAPON_MASTER_D + RAGE
CRITICALDAMAGE: GREATAXE_DAMAGE_CRIT + GREAT_WEAPON_MASTER_D + RAGE
output [attack] named "GA Rg GWM"

function: attack {
 result: [damage for [attack [highest 1 of 2d20] vs DEFENSE]]
}

output [attack] named "Adv GA GWM"

output [attack] named "Adv GA Rg GWM"

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output [highest 1 of 1d6] named "normal"
output [highest 1 of 2d6] named "savage"
output [reroll {1,2} on d6] named "great weapon master"
output [reroll {1,2} on [highest 1 of 2d6]] named "savage + great weapon master"

function: attack ROLL:n max DIE_MAX:n cap ARMOR_CAP:n {
  if ROLL = 1 { result: 0 }
  if ROLL = DIE_MAX {
    if ROLL > ARMOR_CAP { result: ARMOR_CAP * 2 }
    result: ROLL * 2
  }
  if ROLL > ARMOR_CAP { result: ARMOR_CAP }
  result: ROLL
}

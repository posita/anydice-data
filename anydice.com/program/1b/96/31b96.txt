function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

function: roll DICEA:d and DICEB:d {
    result: [return biggest of DICEA and DICEB]
}

function: return biggest of A:n and B:n {
    if A > B { result: A }
    else { result: B }
}

function: savage gwm DICE:d {
    result: [roll [reroll {1,2} on DICE] and [reroll {1,2} on DICE]]
}

output [highest 1 of 1d10] named "normal"
output [highest 1 of 2d10] named "savage"
output [reroll {1,2} on d10] named "great weapon master"
output [savage gwm d10] named "savage + great weapon master (reroll any 1 or 2 on any die)"

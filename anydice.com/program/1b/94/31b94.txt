function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

function: roll DICEA:d and DICEB:d {
    result: [DICEA bigger than DICEB]
}

function: A:n bigger than B:n {
    if A > B { result: A }
    else { result: B }
}

\Rending USR\
\outputs: rending dice\
function: rend DICE:n {
 if DICE = 6 { result: 6 + d3 }
 result: DICE
}

function: hit DICE:n bs BS:n twin-linked TWINLINK:n {
 if DICE >= 7-BS {
  result: 1
 }
 else {
  if TWINLINK = 1 {
   result: [hit d6 BS 0]
  }
  if BS > 5 {
   result: [hit d6 BS-5 0]
  }
 result: 0
 }
}

NOTHIT: 0

function: penetrate HITROLL:n s STRENGTH:n pen dice N:n rending RENDING:n { 
 if HITROLL = 1 {
  if RENDING = 1 {
   result: STRENGTH+[rend d6]
  }
  result: STRENGTH+Nd6
 }
 else { result: NOTHIT }
}

NOTPEN: 0

function: result PENROLL:n GLANCE:n AV:n {
 PEN: 1
 if PENROLL = NOTHIT { result: NOTHIT }
 if PENROLL+GLANCE < AV { result: NOTPEN }
 if PENROLL+GLANCE > AV { result: PEN }
}

\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: lucky ROLL:n {
 if ROLL = 1 {result: 1d20}
 result: ROLL
}

\output [lucky 1d20] named "Lucky Probabilities"\

\looped against enemy AC from 15-20\
ENEMYAC: d{15..20}

DMGBONUS: 17

ATTACKTIMES: 3

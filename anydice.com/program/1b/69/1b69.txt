\Wounds\
function: melee FIGHTERONE:n with WSONE and S vs FIGHTERTWO:n with WSTWO and T
{
  HITS: (FIGHTERONE+WSONE) - (FIGHTERTWO+WSTWO)
  if HITS > 0
  {
     TOWOUND: HITS
     if TOWOUND >= 4-(S-T)
     {
       result: 1
     }
     else
     {
       result: 0
     }
  }
  else
  {
    result: 0
  }
}

loop T over {1..7}
{
  loop S over {1..7}
  {
    loop WSTWO over {1..7}
    {
      loop WSONE over {1..7}
      { 
      output [melee d6 with WSONE and S vs d6 with WSTWO and T] named "Hits with WS1 [WSONE] S [S] vs. WS2 [WSTWO] T [T]"
      }
    }  
  }
}

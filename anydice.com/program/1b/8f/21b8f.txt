\SETTINGS\
M:2 \ how many attacks\
DICES:2 \ how many dices on weapon \
AC: 15
MAP: 5
MAP_AGILE: 4
FF_INIT: 2
ATTACK : 11
STRENGTH : 4

\attack : dice vs AC map how much multiple attack penalty is there\

function: ROLL:n vs AC:n map MAP:n deadly DEADLY:n {
 if ROLL + ATTACK - MAP >= AC - FF + 10 | ROLL = 20 
 	{
 		result: 2dDAMAGE + DEADLY
	}
 if ROLL + ATTACK - MAP >= AC - FF{ result: DAMAGE }
 result: 0
}

\HATCHET\
DAMAGE: DICESd6 + STRENGTH
DEADLY: 0

SUM: 0
FF: FF_INIT
loop N over {0..M-1}{
 SUM: SUM + [d20 vs AC map MAP_AGILE*N deadly DEADLY]
 FF: 2
}
output SUM named "Hatchet [M] attacks"

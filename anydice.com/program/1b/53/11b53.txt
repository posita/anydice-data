BRUTAL_FIST_ROLL: 1d{{1..3}, {BRUTAL_FIST_DIE}:3}
BRUTAL_FIST_ROLL_CRIT: CRITd{{1..3}, {BRUTAL_FIST_DIE}:3}
GREATAXE_ROLL: 1d{{1..3}, {4..12}:7}
GREATAXE_ROLL_CRIT: {CRIT + SAVAGE_ATTACKS + BRUTAL_CRIT}d{{1..3}, {4..12}:7}

GREAT_WEAPON_MASTER_A: -5

GREATAXE_DAMAGE: GREATAXE_ROLL + BRUTAL_FIST_ROLL + STR + PROF
GREATAXE_DAMAGE_CRIT: GREATAXE_ROLL_CRIT + BRUTAL_FIST_ROLL_CRIT + STR + PROF

MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  if ROLL + ATTACK >= DEFENSE { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

DEFENSE: d{12..25}

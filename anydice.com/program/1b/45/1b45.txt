function: statistic ATTRIBUTE:n vs ROLL:n {
 if ROLL > ATTRIBUTE {result: -1}
 result: ATTRIBUTE - ROLL
}

function: contested MAIN:n vs OPPOSED:n rolling ROLL:n {
 RESULT: 10 + MAIN - OPPOSED - ROLL
 if RESULT < 0 {result: -1}
 result: RESULT
}

function: opposed MAIN:n vs OPPOSED:n {
if MAIN < 0 {result: -1}
if OPPOSED < 0 {result: MAIN}
if OPPOSED > MAIN {result: -1}
result: MAIN - OPPOSED
}
loop N over {-4..4} {
 Y: 10
 X: Y + N
 output [contested X vs Y rolling 3d6] named "Contested Roll [X] vs [Y]"
}

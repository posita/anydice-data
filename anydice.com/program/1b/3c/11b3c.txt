\**** Great Weapon Fighting modifiable Calculator ****\

function: attackroll ROLL:n vs AC:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 result: 0
}

function: sixgwfight ROLLX:n {
 if ROLLX <= 2 { result:d6 }
 result: ROLLX
}

function: twelvegwfight ROLLX:n {
 if ROLLX <= 2 { result:d12 }
 result: ROLLX
}

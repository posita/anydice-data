function: roll N:s PROF {
if N = 1 { result: PROF * d10 + d4 + d6 }
if N = 2 { result: PROF * d10 + d4 + d8 }
if N = 3 { result: PROF * d10 + d6 + d8 }
if N = 4 { result: PROF * d10 + d6 + d10 }
if N = 5 { result: PROF * d10 + d4 + d6 + d8 }
if N = 6 { result: PROF * d10 + d4 + d6 + d10 }
if N = 7 { result: PROF * d10 + d4 + d8 + d10 }
if N = 8 { result: PROF * d10 + d6 + d8 + d10 }
if N = 9 { result: PROF * d10 + d6 + d8 + d12 }
if N = 10 { result: PROF * d10 + d6 + d10 + d12 }
if N = 11 { result: PROF * d10 + d8 + d10 + d12 }
}


loop I over {1..11} {
\output [roll I 0] named "[I] level"\
\output [roll I 1] named "[I] level prof"\
\output [roll I 1]+[roll d9 0] named "[I] level fusion"\
output ([roll I 0] + [roll I 1] + [roll I 1] + [roll d9 0])/3 named "[I] level average"
}

\output [roll d11 0] named "summary"
output [roll d11 1] named "summary prof"

output [roll d11 0]+[roll d11 1] named "summary fusion"

output ([roll d11 0] + [roll d11 1] + [roll d11 0] + [roll d11 1])/3 named "summary average"\

function: roll H:n and M:n and L:n {
  if H > L {
    if H > M { result: H + (M>4) + (L>4) }
    else { result: M + (H>4) + (L>4) }
  }
  else if M > L { result: M + (H>4) + (L>4) }
  else { result: L + (H>4) + (M>4) }
}

AC: 20
ATKMOD: 5
PROF: 6
SA: 5d6

function: first A:n second B:n {
  if A + ATKMOD + PROF >= AC & B + ATKMOD + PROF >= AC {result: 1d8 + 1d8 + ATKMOD + ATKMOD + SA}
  if A + ATKMOD + PROF >= AC {result: 1d8 + ATKMOD + SA}
  if B + ATKMOD + PROF >= AC {result: 1d8 + ATKMOD + SA}
  result: 0
}

output [first d20 second d20] named "Two Attacks"

function: attack ROLL:n vs AC:n {
 if ROLL + PROF + ATKMOD - 5 >= AC { result: DAMAGE }
 result: 0
}

DAMAGE: 1d8 + ATKMOD + SA + 10

output [attack [highest 1 of 2d20] vs AC] named "One Attack with Adv"

\ An example of rolling a d20 and adding a modifier as well as examples for rolling it twice and taking the highest or lowest \

loop I over {-3,-2, -1, 0, 1, 2, 3}{
 output 2d6 + I named "2d6 [I] modifier normally"
 output [highest 1 of 2d6] + I named "2d6 [I] modifier with Advantage"
 output [lowest 1 of 2d6] + I named "2d6 [I] modifier with Disadvantage"
}

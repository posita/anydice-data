\ 1. Nbr min. de réussites parfaites (6s) \
\ 2. Nbr min. de réussites aprtielles (4-5) \
\ 3. Nbr min. de succès (mixtes : 4, 5 ou 6) \

function: check SMIN:n PMIN:n TOTALMIN:n in JET:s {
  S: [count 6 in JET]      \ Réussites parfaites \
  P: [count {4,5} in JET]  \ Réussites partielles \
  
  if (S >= SMIN) & (P >= PMIN) & ((S+P) >= TOTALMIN) { result: 1 }
  result: 0
}

\ CONFIGURATION DE LA LOOP : NOMBRE DE DÉS \
LOOP: {3..8} 

loop N over LOOP {
  
  \ --- SEUIL 1 --- \
  output [check 0 0 1 in Nd6] named "[N]d Seuil 1 : Global (Total 1+)"
  output [check 1 0 1 in Nd6] named "[N]d Seuil 1 : 1 Parfaites"

  \ --- SEUIL 2 --- \
  output [check 0 0 2 in Nd6] named "[N]d Seuil 2 : Global (Total 2+)"
  output [check 2 0 2 in Nd6] named "[N]d Seuil 2 : 2 Parfaites"
  output [check 1 1 2 in Nd6] named "[N]d Seuil 2 : 1 Parf + 1 Std"

  \ --- SEUIL 3 --- \
  output [check 0 0 3 in Nd6] named "[N]d Seuil 3 : Global (Total 3+)"
  output [check 3 0 3 in Nd6] named "[N]d Seuil 3 : 3 Parfaites"
  output [check 2 1 3 in Nd6] named "[N]d Seuil 3 : 2 Parf + 1 Std"
  output [check 1 2 3 in Nd6] named "[N]d Seuil 3 : 1 Parf + 2 Std"

  \ --- SEUIL 4 --- \
  output [check 0 0 4 in Nd6] named "[N]d Seuil 4 : Global (Total 4+)"
  output [check 4 0 4 in Nd6] named "[N]d Seuil 4 : 4 Parfaites"
  output [check 3 1 4 in Nd6] named "[N]d Seuil 4 : 3 Parf + 1 Std"
  output [check 2 2 4 in Nd6] named "[N]d Seuil 4 : 2 Parf + 2 Std"
  output [check 1 3 4 in Nd6] named "[N]d Seuil 4 : 1 Parf + 3 Std"
}

function: ROLL:s count dice { 
   X: ROLL 
   if X > 4 { result: X-4 } 
   else if X = 1 { result: -10000 }
   else { result: 0 }
} 

function: NUM:n group fails { 
   X: NUM
   if X < 0 { result: -1 } 
   else { result: X }
}

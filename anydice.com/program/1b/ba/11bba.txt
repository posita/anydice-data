\ Battles \ 
function: NUMBER:n ROLL:d vs OPPOSED:n with DIFFERENCE:n {
 if ROLL = 20 & OPPOSED < 20 { result: 2 }
 if ROLL > OPPOSED + DIFFERENCE { result: 1 }
 if ROLL < OPPOSED + DIFFERENCE & OPPOSED < 20 { result: -1 }
 if ROLL < 20 & OPPOSED = 20 { result: -2 }
 result: 0
}

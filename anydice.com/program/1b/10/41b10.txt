\ Roll Calculations. Do Not Change \

function: roll ROLL:s {
 if 1@ROLL < 1 & (#ROLL)@ROLL = -1 {
  result: -1
 }
 result: [highest of 0 and ROLL]
}

function: roll ROLL:n with willpower {
 result: [highest of 0 and ROLL + 1]
}

function: compare N:n {
 if N = 10 { result: 1 }
 if N = 1 { result: -1 }
 result: N >= COMPARE
}

function: exploding N:n {
 if N = 10 { result: 2 }
 if N = 1 { result: -1 }
 result: N >= DIFFICULTY
}

function: compsplode N:n {
 if N = 10 { result: 2 }
 if N = 1 { result: -1 }
 result: N >= COMPARE
}

\ Roll Difficulties. To change difficulties for normal rolls change the top number, the bottom number only changes the difficulty on compared rolls. \
DIFFICULTY: 6
COMPARE: 8

\ Function Names. Do Not Change \
NORMAL: d{-1:1, 0:(DIFFICULTY - 2), 1:(11 - DIFFICULTY)}
COMPARE: [compare d10]
EXPLODING: [exploding d10]
COMPSPLODE: [compsplode d10]

\ Result Examples. Adding additional loops and outputs can grab different results. Example outputs are shown below. \
X: 4
output [roll X d NORMAL] named "Normal"
output [roll X d COMPARE] named "Compare"
output [roll X d EXPLODING] named "Specialty"
output [roll X d COMPSPLODE] named "Compare Specialty"
output [roll X d NORMAL with willpower] named "Normal Willpower"
output [roll X d COMPARE with willpower] named "Compare Willpower"
output [roll X d EXPLODING with willpower] named "Specialty Willpower"
output [roll X d COMPSPLODE with willpower] named "Compare Specialty Willpower"

loop X over {1..2} {
 output [roll X d NORMAL] named "[X]d Normal"
}

function: reroll REROLL:s on ROLL:n to REPLACEMENT:d {
    if [REROLL contains ROLL] {
        result: REPLACEMENT
    }
    result: ROLL
}

function: reroll REROLL:s on DICE:d {
    result: [reroll REROLL on DICE to DICE]
}

output 3d6 named "Level 1 Burning Hands"
output 3d[reroll {1,2,3} on d6] named "EMPOWERED Level 1 Burning Hands"
output 4d6 named "Level 2 Burning Hands"
output 4d[reroll {1,2,3} on d6] named "EMPOWERED Level 2 Burning Hands"
output 5d6 named "Level 3 Burning Hands"

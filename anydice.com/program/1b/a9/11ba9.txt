\**** Damage calculator ****\

function: attackroll ROLL:n vs AC:n {
if ROLL >= CRITROLL { result: CRITDAMAGE }
if ROLL = 1 { result: 0 }
if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
result: 0
}

\looped against enemy AC from 15-20\
ENEMYAC: d{18}

ATTACKMOD: 2
DMGBONUS: 0
PROFICIENCY: 2
ACCBONUS: 0
CRITROLL: 20
ATTACKTIMES: 10
DAMAGE: 2d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll 1d20 vs ENEMYAC] named "vs Fighter"

ENEMYAC: d{17}

ATTACKMOD: 2
DMGBONUS: 0
PROFICIENCY: 2
ACCBONUS: 0

DAMAGE: 2d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 4d8 + ATTACKMOD + DMGBONUS
output ATTACKTIMESd[attackroll [highest 1 of 2d20] vs ENEMYAC] named "vs Barbarian"

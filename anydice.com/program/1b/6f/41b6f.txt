function: attack ROLL:n max DIE_MAX:n cap ARMOR_CAP:n {
  if ROLL = 1 { result: 0 }
  if ROLL = DIE_MAX {
    if ROLL > ARMOR_CAP { result: ARMOR_CAP * 2 }
    result: ROLL * 2
  }
  if ROLL > ARMOR_CAP { result: ARMOR_CAP }
  result: ROLL
}

DAGGER: [attack d4 max 4 cap 6]
SHORTSWORD: [attack d6 max 6 cap 6]
LONGSWORD: [attack d8 max 8 cap 6]
GREATSWORD: [attack d12 max 12 cap 6]

output 10dDAGGER named "Dagger x10"
output 10dSHORTSWORD named "Shortsword x10"
output 10dLONGSWORD named "Longsword x10"
output 10dGREATSWORD named "Greatsword x10"

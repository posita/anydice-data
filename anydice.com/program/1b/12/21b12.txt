H: 8    \Hero die\
S: 8    \Skill die\
C: 8    \Career die\
E: 0    \Expertise die\
A: 0    \Advantage die\
D: 0    \Number of difficulty dice\
TN: 5   \Target number for action dice\
AN: 2   \Activation number for difficulty dice\
OH: 8   \opposition hero die\
OS: 8   \opposition skill die\
OC: 8   \opposition career die\
OE: 0    \opposition expertise die\
OA: 0    \opposition advantage die\
OD: 0    \Number of difficulty dice\

function: evaluate HERO:d and STAT:d and CAREER:d and EXPERTISE:d and ADVANTAGE:d and DIFFICULTY:n vs OPHERO:d and OPSTAT:d and OPCAREER:d and OPEXPERTISE:d and OPADVANTAGE:d and OPDIFFICULTY:n {
  result: (((HERO >= TN) + (STAT >= TN) + (CAREER >= TN) + (EXPERTISE >= TN) + (ADVANTAGE >= TN)) - DIFFICULTYd(d4<=AN)) - (((OPHERO >= TN) + (OPSTAT >= TN) + (OPCAREER >= TN) + (OPEXPERTISE >= TN) + (OPADVANTAGE >= TN)) - OPDIFFICULTYd(d4<=AN))
}

output [evaluate 1dH and 1dS and 1dC and 1dE and 1dA and D vs 1dOH and 1dOS and 1dOC and 1dOE and 1dOA and OD]

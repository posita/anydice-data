function: normal ROLL:n max DIE_MAX:n cap ARMOR_CAP:n {
  if ROLL = 1 { result: 0 }
  if ROLL = DIE_MAX {
    if ROLL > ARMOR_CAP { result: ARMOR_CAP * 2 }
    result: ROLL * 2
  }
  if ROLL > ARMOR_CAP { result: ARMOR_CAP }
  result: ROLL
}

function: advantage A:n and B:n max DIE_MAX:n cap ARMOR_CAP:n {
  if A = 1 & B = 1 { result: 0 }
  if A = DIE_MAX | B = DIE_MAX {
    if DIE_MAX > ARMOR_CAP { result: ARMOR_CAP * 2 }
    result: DIE_MAX * 2
  }
  BEST: [highest of A and B]
  if BEST > ARMOR_CAP { result: ARMOR_CAP }
  result: BEST
}

function: disadvantage A:n and B:n max DIE_MAX:n cap ARMOR_CAP:n {
  if A = 1 | B = 1 { result: 0 }
  if A = DIE_MAX & B = DIE_MAX {
    if DIE_MAX > ARMOR_CAP { result: ARMOR_CAP * 2 }
    result: DIE_MAX * 2
  }
  WORST: [lowest of A and B]
  if WORST > ARMOR_CAP { result: ARMOR_CAP }
  result: WORST
}

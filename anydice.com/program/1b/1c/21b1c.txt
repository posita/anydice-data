function: stunt DICE:s STUNT:n DC:n {
  if 1@DICE + 2@DICE + STUNT < DC {
    result: 0
  }
  loop X over {1..#DICE-1}{
    if (X@DICE = (X+1)@DICE | X@DICE = STUNT | (X+1)@DICE = STUNT) & X@DICE + (X+1)@DICE + STUNT >= DC { result: STUNT }
  }
  result: 0
}

DC: 0
output [stunt 2d6 1d6 DC] named "2d6"
output [stunt 3d6 1d6 DC] named "3d6"
output [stunt 4d6 1d6 DC] named "4d6"
output [stunt 5d6 1d6 DC] named "5d6"
output [stunt 6d6 1d6 DC] named "6d6"

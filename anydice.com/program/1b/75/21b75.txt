function: save S:n armor piercing AP:n damage D:n armor SV:n{
  if S >= SV + AP {
    result:0
  }
  result:D
}

function: wound W:n strength S:n armor piercing AP:n damage D:n against t T:n armor SV:n
{
  if S >= 2*T{
    if W >= 2{
      result:[save 1d6 armor piercing AP damage D armor SV]
    }
  } else if 2*S <= T{
    if W >= 6{
      result:[save 1d6 armor piercing AP damage D armor SV]
    }
  } else if S > T{
    if W >= 3{
      result:[save 1d6 armor piercing AP damage D armor SV]
    }
  } else if S < T{
    if W >= 5{
      result:[save 1d6 armor piercing AP damage D armor SV]
    }
  } else {
    if W >= 4{
      result:[save 1d6 armor piercing AP damage D armor SV]
    }
  }
  result: 0
}

function: attack A:n ballistics BS:n strength S:n armor piercing AP:n damage D:n against t T:n armor SV:n {
  if A >= BS {
    result:[wound 1d6 strength S armor piercing AP damage D against t T armor SV]
  }
  result:0
}

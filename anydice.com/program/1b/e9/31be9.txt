\
 Blades in the Dark
 0: Things go bad.                                   (Rolling nothing above 3)
 1: Success, but with undesired consequences.        (Rolling at least one die above 3)
 2: Clean success.                                   (Rolling at least one 6)
 3: Critical success!                                (Rolling at least two 6's)
\

function: blades ROLL:s {
 if 1@ROLL = 6 {
  result: 2 + (2@ROLL = 6)
 }
 result: 1@ROLL >= 4
}

output [blades 2@2d6] named "0d"
loop P over {1..20} {
 output [blades Pd6] named "[P]d"
}

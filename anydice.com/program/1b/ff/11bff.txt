function: add A to B {
 result: A + B
}

function: booleanize NUMBER:n for AC:n do DMG{
 if NUMBER > AC { result: DMG }
 else { result: 0 }
}

function: beatac AC with DICE add VALUE do DMG {
 Y: DICE + VALUE
 result: [booleanize Y for AC do DMG]
}

function: dmgavg DMG against AC with DICE add VALUE {
 X: [beatac AC with DICE add VALUE do DMG]
 result : X
}

MISS: 0
HIT: 1
CRITICALHIT: 2


function: fateroll ATK:n with ROLL:n vs DEFENSE:n {

ATKRLL: ATK + ROLL

 if ROLL = 1 { result: MISS }
 if ROLL = 20 {
  result: CRITICALHIT
 }
 if ATKRLL >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for WEAPONDAMAGE:n with ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: 2 * WEAPONDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: WEAPONDAMAGE
 }
 result: 0
}

function: attack with WEAPON:n {
 result: [damage for WEAPON with [fateroll WEAPON with d20 vs DEFENSE]]
}

DEFENSE: 20



output [attack with [explode d6]] named "Derringer"
output [attack with 2d[explode d6]] named "Sawn-Off Shotgun"
output [attack with 1d[explode d8]] named "Magnum Pistol"
output [attack with 1d[explode d8] - 1] named "LeMat Pistol"
output [attack with 2d[explode d6] - 1] named "LeMat Shotgun"

output [attack with 3d[explode d12]] named "Elephant Gun"

function: penetration N:n {
    result: [lowest of [absolute d10 - d10] and N]
}

function: D:n armor A:n {
    result: [highest of 0 and D - A]
}

function: hit on ACC:n dam D:n pen P:n with ROLL:n {
    if ROLL < ACC {
        result: 0
    } else {
        result: D + [penetration P]
    } 

}

function: high impact on ACC:n dam D:n pen P:n with ROLL:n {
    if ROLL < ACC {
        result: 0
    } else {
        result: D + [penetration P] + [lowest of ROLL - ACC and [highest of D and 2]]
    }     
}

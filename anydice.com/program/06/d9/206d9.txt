function: pbta ROLL:n plus N:n {
  if ROLL + N <= 6 {
    result: 0
  }
  if ROLL + N > 6 & ROLL + N < 10 {
    result: 1
  }
  result: 2
}

function: ROLL:s target T:n D:n {
  A: 1@ROLL
  B: 2@ROLL
  N: T - D

  if A = B & A >= N { result: -1 }
  if A = B & A < N { result: 3 }
  \if A = B { result: 3 }\
  if A < N & B < N { result: 2 }
  if A < N & B >= N { result: 1 }
  if B < N & A >= N { result: 1 }

  result: 0
}

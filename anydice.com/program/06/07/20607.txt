function : damage DAMAGE:d DPLUS:n attack DICE:s APLUS:n vs AC:n
{
	MULTIPLIER : 0
	OUT : 0
	if (DICE = 20) {MULTIPLIER : 2}
	else if (DICE + APLUS >= AC) {MULTIPLIER : 1}
	
	if (MULTIPLIER > 0)
	{
		OUT : DPLUS
		loop N over {1..MULTIPLIER}
		{
			OUT : OUT + DAMAGE
		}
	}
	
	result : OUT
}


function : great weapon fighter DICE:s
{
 H : 0
 loop N over {1..#DICE}
 {
  if N@DICE > 2 {H : H + N@DICE}
  else {H : H + 1d6}
 }
 result : H
}

function : great weapon master DAMAGE:s DPLUS:n attack DICE:s APLUS vs AC:n
{
 result : [damage DAMAGE DPLUS+10 attack DICE APLUS-5 vs AC]
}

function : get modifiers DICE:d
{
 result : (DICE - 10)/2
}

HM : 1@6d[get modifiers [highest 3 of 4d6]] + 2
HD : 1d20
D : [great weapon fighter 2d6]
loop N over {6..25}
{
 output [damage D 0 attack HD HM vs N] named "Normal Attack AC [N]"
 output [great weapon master D 0 attack HD HM vs N] named "Great Weapon Master AC [N]"
}

\**** Damage calculator ****\

function: useprecision ROLL:n and PRECISION:n {
 if ROLL >= CRITROLL { result: 0 } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + MAXPRECISION < AC { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + PRECISION >= AC { result: 1 }
 result: 0
}

function: attackroll ROLL:n and PRECISION:n {
 if ROLL >= CRITROLL { result: CRITDAMAGE } 
 if ROLL = 1 { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS >= AC { result: DAMAGE }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + MAXPRECISION < AC { result: 0 }
 if ROLL + PROFICIENCY + ATTACKMOD + ACCBONUS + PRECISION >= AC { result: DAMAGE }
 result: 0
}

\looped against enemy AC from 15-20\
AC: 20

ATTACKMOD: 5
PROFICIENCY: 4
ACCBONUS: -2
DMGBONUS: 13
CRITROLL: 20
ATTACKTIMES: 1
DAMAGE: 1d8 + ATTACKMOD + DMGBONUS
CRITDAMAGE: 2d8 + ATTACKMOD + DMGBONUS
MAXPRECISION: 5

output ATTACKTIMESd[useprecision d20 and d8] named "Longbow +1"
output ATTACKTIMESd[attackroll d20 and d8] named "Longbow +1"

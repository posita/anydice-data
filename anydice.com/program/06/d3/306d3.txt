S: {0,0,0,1,1,2}
P: {0,1,1,1,1,2}

function: minzero N:n with pow POW:n arm ARM:n {
   if N <= 0 {result: 0}
   result: (NdP + POWdS) / ARM
}

function: warc with cha CHA:n mat MAT:n def DEF:n cover COV:n pow POW:n arm ARM:n {
   result: [minzero (CHAdS + MATdP) - (DEFdS + COVdP) with pow POW arm ARM] 
}

output [warc with cha 1 mat 3 def 3 cover 2 pow 4 arm 3]

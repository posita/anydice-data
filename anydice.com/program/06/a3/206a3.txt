function: explode N:n {
 if N = 1 { result: 1 - d10 }
 if N = 10 { result: 10 + d10 }
 result: N
}

function: attack ATT:n defense DEF:n armor AC:n {
    \ Check if attack roll is higher than defense roll \
    if(ATT > DEF) {
        \ Damage is attack minus opponents armor \        
        DAMAGE: (5d6 - AC) 

        \ Damage cannot be less than zero \
        result: [highest of DAMAGE and 0]
    } 
    \ Attack roll failed, 0 damage \
    result: 0
}

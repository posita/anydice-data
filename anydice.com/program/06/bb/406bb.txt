function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= CRITTRESH { result: CRITICALHIT }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}
function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT { result: CRITICALDAMAGE }
 if ATTACKRESULT = HIT { result: HITDAMAGE }
 if ATTACK = 20 { result: HITDAMAGE }
 result: MISSDAMAGE
}
function: attack {
 result: [damage for [attack d20 vs DEFENSE]]
}

function: damagepally for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT { result: CRITICALDAMAGE + 10d8 }
 if ATTACKRESULT = HIT { result: HITDAMAGE + 5d8 }
 if ATTACK = 20 { result: HITDAMAGE + 5d8 }
 result: MISSDAMAGE
}
function: attackpally {
 TEMP: [damagepally for [attack d20 vs DEFENSE]]
 result: TEMP
}

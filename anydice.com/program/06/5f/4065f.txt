function: EXTRA-SUCCESS for S:n {
    R: d6
    if R > 4 { result: 1 }  /* extra success on 5-6 */
    result: 0                /* everything else */
}

function: TOTAL-SUCCESSES {
    INITIAL: 0
    loop 10 times {
        R: d6
        if R > 3 { INITIAL += 1 }   /* success on 4-6 */
    }

    EXTRA: 0
    loop INITIAL times {
        EXTRA += [EXTRA-SUCCESS for INITIAL]
    }

    result: INITIAL + EXTRA
}

output [TOTAL-SUCCESSES] named "total successes"

\
output [highest of [explode d10] and [explode d6]] named "wild"
output [explode d10] named "extra"

function: myexplode N:n on SIZE:n {
 if N = SIZE {
  result: N + [myexplode d(SIZE) on SIZE]
 }
 result: N
}

function: roll A:n B:n { 		\  Tar in två tärningsslag. \
  if A + B = 0 {
    if COUNTWITHZERO { result: 0 }	\ Tar med dubbelnoll om så är önskat. \
  }
  else {
    if A*10+B <= SKILLVALUE {		\ Jämför slaget med färdighetsvärdet. \
       result: A + B + GAMBLE}		\ Räknar ut Resultatet. GAMBLE adderas om man har satsat. \
    else if COUNTWITHZERO { result: 0 }	\ Skriver ut nollresultat om så är önskat. \
  }
}

function: flipped A:n B:n {
  if A < B { result: [roll A B] }	\ Jämför vilken tärning som är högst \
  result: [roll B A]
}

DICE: {0..9}			\ Tärningstyp. T10, där noll utläses som 0 \
ROLL: 1d(DICE)			\ Tärningsslag. \
SKILLVALUE: 90			\ Färdighetsvärdet. \
COUNTWITHZERO: 1		\ Om nollresultat ska skrivas ut. (0 = nej) \
GAMBLE: 0			\ Om satsat (0 = nej) \

output {} named "FÄRDIGHETSSLAG"
output [roll ROLL ROLL] named "1T100 mot [SKILLVALUE]"
output [flipped ROLL ROLL] named "Flippad 1T100 mot [SKILLVALUE]"

SKILLVALUEORG: SKILLVALUE	\ Används i output nedan. \
SKILLVALUE: SKILLVALUE-20	\ Förändrar värden för satsning. -20 på färdigheten. \
GAMBLE: 4			\ Förändrar värden för satsning. +4 på Resultatet. \

output [roll ROLL ROLL] named "Satsad 1T100 mot [SKILLVALUEORG]"
output [flipped ROLL ROLL] named "Satsad och flippad 1T100 mot [SKILLVALUEORG]"

\ --- \

SKILLVALUE: 160 		 \ Siffran krävs då grundegenskaper använder en annan skala. \
COUNTWITHZERO: 1		 \ Ska alltid räkna med noll i Resultatet. \

output {} named "GRUNDEGENSKAPERSSLAG"
output 2d(DICE) named "2T10"
output [roll ROLL+ROLL 0] named "Satsad 2T10"

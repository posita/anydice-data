function: reroll lowest of ROLL:s as REROLL:d if less than MIN:n {
  LOWEST: (#ROLL)@ROLL       \ the lowest die is sorted last \
  if LOWEST >= MIN {
    result: ROLL + 0
  } else {
    REST: {1..#ROLL-1}@ROLL  \ all but the lowest die \
    result: REST + REROLL
  }
}

output 1d6 named "1d6"
output [highest 1 of 2d6] named "extra roll before"
loop N over {1..6}{
  output [reroll lowest of 1d6 as d6 if less than N] named "extra roll after <[N]"
}

\   1,2,3,4,5,6,7,8,9,0 \
X: {0,0,1,1,2,2}
XD: 6

function: xtail N:n {
  if N < XD {result: 0}
  result: 1 + [xtail dXD]
}

function: xdice N:n {
  if N < 3 {result: 0}
  if N < 5 {result: 1}
  if N < 6 {result: 2}
  if N = 6 {result: 2 + [xtail dXD]}
}

loop N over {1..4} {
  output [lowest of Nd[xdice dXD] and 10] named "[N]dX"
}

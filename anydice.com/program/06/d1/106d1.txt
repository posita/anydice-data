function: ironclaw DIFFERENCE:n {
 if DIFFERENCE < 0 { result: -1 }
 if DIFFERENCE = 0 { result: 0  }
 if DIFFERENCE >= 5 { result: 2 }
 result: 1
}

function: high P:n minus O:n {
 result: [highest 1 of P] - [highest 1 of O]
}

loop N over {1,2} {
 loop M over {4,6,8,10,12} {
  output [highest of [high 2d8 minus NdM] and -1] named "2d8-[N]d[M]"
 }
}

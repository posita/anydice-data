function: attack ROLL:n plus BONUS:n vs AC:n hit HIT:d crit CRIT:d on CRIT_RANGE:n {
    if ROLL = 1 {
        result: 0
    }
    if ROLL >= CRIT_RANGE {
        result: CRIT
    }
    if ROLL + BONUS >= AC {
        result: HIT
    }
    result: 0
}

output [attack 1d20 plus 4+3 vs 15
  hit 2d6 + 4
  crit 4d6 + 4 on 20
] named "Greatsword"

output [attack 1d20 plus 4+3 vs 15
  hit [highest of 2d6 and 2d6] + 4
  crit [highest of 4d6 and 4d6] + 4 on 20
] named "SA Greatsword"

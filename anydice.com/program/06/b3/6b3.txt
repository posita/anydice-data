A:6 \amount of dice\
B:3 \range\
C:9 \basic to-hit\

loop N over {1..B}{
  X: N
  H: C
  if (N = 1) {H: H-1}
  output Ad(d10 >= H) named "range [X]"
}
loop N over {C+1..10}{
  X: B+N-C
  output Ad(d10 >= N) named "range [X]"
}
loop N over {1..A-1}{
  R: A-N
  X: B+10-C+N
  output Rd(d10 >= 10) named "range [X]"
}

function: banner {
  result: 1
}

function: offbanner {
  result: 0
}

function: threestar {
  R: 1 \d100\
  if R = 1 { result: [banner] }
  if R <= 3 { result: [offbanner] }
  result: 0
}

function: fourstar {
  if d20 = 1 { 
    if d4 = 1 { result: [offbanner] }
    result: [banner]
  }
  result: 0
}

T: 0
loop N over { 1..47 }{
  T: T + [threestar]
}
loop N over { 1..5 }{
  T: T + [fourstar]
}
T: T + [banner]
T: T + [offbanner]

output T

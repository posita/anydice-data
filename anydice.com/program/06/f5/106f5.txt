CRIT_RANGE: 20
NORMAL_HIT: d20
HIT_ADV: [highest 1 of 2d20]

MISS: 0
HIT: 1
CRITICALHIT: 2

function: attack ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= CRIT_RANGE {
  if ROLL + ATTACK >= 0 { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: attackadv ROLL:n vs DEFENSE:n {
 if ROLL = 1 { result: MISS }
 if ROLL >= CRIT_RANGE {
  if ROLL + ATTACK >= 0 { result: CRITICALHIT }
  result: HIT
 }
 if ROLL + ATTACK >= DEFENSE { result: HIT }
 result: MISS
}

function: damage for ATTACKRESULT:n {
 if ATTACKRESULT = CRITICALHIT {
  result: CRITICALDAMAGE
 }
 if ATTACKRESULT = HIT {
  result: HITDAMAGE
 }
 result: MISSDAMAGE
}

function: attack {
 result: [damage for [attack NORMAL_HIT vs DEFENSE]]
}

function: attackadv {
 result: [damage for [attackadv HIT_ADV vs DEFENSE]]
}

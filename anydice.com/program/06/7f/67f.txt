
function: minz NUM:n {
  if NUM >= 0 { result: NUM} 
  result: 0
}

function: getdamage DAMAGE CHECK:n {
  if CHECK >= 0 { result: [minz DAMAGE + (CHECK/5)]} 
  result: 0
}


loop DC over {-10, -5, 0, 5, 10, 15, 20} {
 loop DAMAGE over {-10, -5, 0, 5 ,10, 15, 20} {
   CHECK: 2d10 + DC - 2d10
   output [getdamage DAMAGE CHECK] named "[DAMAGE] damage at [DC]"
 }
}

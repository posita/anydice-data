SKILL: 30
DICESIZE: 8

function: SUM:n open unfairdice NUMBER:n {
  if SUM > SKILL {
    result: {SKILL}
  }
  else {result: 1d([count 1 in (NUMBERdDICESIZE)])}
}

loop N over {1..10} {
  output [NdDICESIZE open unfairdice N] named "sannolikheten för antalet 1:or på [N]st [DICESIZE]sidiga tärningar med en summa under [SKILL]"
}

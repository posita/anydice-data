function: difficulty D:n {
 if D>6 {result: [highest of 1d12 and 1d(2*(D-4))]}
 result: 1d(2*(D+2))
}

function: high O:n {
 result: [highest 1 of 2d8] - [difficulty O]
}

loop D over {1..5} {
  X:2*(D+2)
  output [highest of [high D] and -1] named "d[X]"
}
loop D over {6..8} {
  Y:2*(D-4)
  output [highest of [high D] and -1] named "d12+d[Y]"
}

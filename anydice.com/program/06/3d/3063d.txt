function: count VALUES:s against MINUS:s in SEQUENCE:s {
 COUNT: 0
 loop P over {1..#VALUES} {
  COUNT: COUNT + (P@VALUES = SEQUENCE)
 }
 loop P over {1..#MINUS} {
  COUNT: COUNT - (P@MINUS = SEQUENCE)
 }
 result: COUNT
}

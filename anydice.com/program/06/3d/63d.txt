FROM: 3
TO: 5

function: get NUMBER in DICE:s {
SIX: [count 6 in DICE]*6
FIVE: [count 5 in DICE]*5
FOUR: [count 4 in DICE]*4
THREE: [count 3 in DICE]*3
TWO: [count 2 in DICE]*2
ONE: [count 1 in DICE]

result: NUMBER@[sort {ONE, TWO, THREE, FOUR, FIVE, SIX}]
}

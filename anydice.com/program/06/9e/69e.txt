  \ En uträkning för sannolikheten om man slår om maxutfallet vid genesistärningar. \
GENDICE: {0,1,2}

function: SUM:n open genesisdice MAX:n ADDMAX:n {
  if SUM = MAX {
    result: [NdGENDICE open genesisdice MAX MAX+ADDMAX]
  }
  else { result: NdGENDICE + ADDMAX }
}

loop N over {2..6} {
  output [NdGENDICE open genesisdice N*2 0] named "[N] genesistärningar"
}

\              (varibler)
  N antal tärningar
  NdGENDICE är N stycken genesistärningar.
  SUM är tärningsutfallet
  MAX är maxresultatet
  ADDMAX adderar maxresultatet vid omslag.

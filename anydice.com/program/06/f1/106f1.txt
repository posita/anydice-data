function: banner {
  result: d2 - 1
}

function: offbanner {
  result: d64 <= 10
}


function: X:n threestar {
  if X = 1 { result: [banner] }
  if X <= 3 { result: [offbanner] }
  result: 0
}

function: X:n fourrainbow {
  if X=1 { result: [offbanner] }
  result: [banner]
}

function: X:n fourstar {
  if X = 1 { 
    result: [d4 fourrainbow]
  }
  result: 0
}

T: 0
loop N over { 1..47 }{
  T: T + [d100 threestar]
}
loop N over { 1..5 }{
  T: T + [d20 fourstar]
}
T: T + [banner]
T: T + [offbanner]

output T named "Stepup"
